196 /**
197 Conservative culling test for up to 32 planes.
198 Returns true if there exists a <CODE>plane[p]</CODE> for
199 which the entire object is in the negative half space
200 (opposite the plane normal).
201
202 <CODE>testMask</CODE> and <CODE>childMask</CODE>
203 are used for optimizing bounding volume hierarchies.
204 The version of this method that produces childMask
205 is slower than the version without; it should only
206 be used for parent nodes.
207
208 @param cullingPlaneIndex The index of the first plane for which
209 the entire object is in the negative half-space. The function
210 exits early when one plane is found. -1 when the function
211 returns false (i.e. when no plane culls the whole object).
212
213 @param testMask If bit <I>p</I> is 0, the
214 bounding volume automatically passes the culling test for
215 <CODE>plane[p]</CODE> (i.e. it is known that the volume
216 is entirely within the positive half space). The function
217 must return false if testMask is 0 and test all planes
218 when testMask is -1 (0xFFFFFFFF).
219
220 @param childMask Test mask for the children of this volume.
221
222 */
223 bool culledBy
224 (const Array<Plane>& plane,

237 /** less than or equal to containment */
238 inline bool contains(const AABox& other) const {
239 return
240 (other.hi.x <= hi.x) &&
241 (other.hi.y <= hi.y) &&
242 (other.hi.z <= hi.z) &&
243 (other.lo.x >= lo.x) &&
244 (other.lo.y >= lo.y) &&
245 (other.lo.z >= lo.z);
246 }
247
248 inline bool contains(const Point3& point) const {
249 return
250 (point.x >= lo.x) &&
251 (point.y >= lo.y) &&
252 (point.z >= lo.z) &&
253 (point.x <= hi.x) &&
254 (point.y <= hi.y) &&
255 (point.z <= hi.z);
256 }

258 inline float area() const {
259 if (isEmpty()) { return 0; }
260 Vector3 diag = hi - lo;
261 return 2.0f * (diag.x * diag.y + diag.y * diag.z + diag.x * diag.z);
262 }
263
264 inline float volume() const {
265 if (isEmpty()) { return 0; }
266 Vector3 diag = hi - lo;
267 return diag.x * diag.y * diag.z;
268 }

281 /** Return the intersection of the two boxes */
282 AABox intersect(const AABox& other) const {
283 if (isEmpty() || other.isEmpty()) {
284 return empty();
285 }
286
287 const Point3& H = hi.min(other.hi);
288 const Point3& L = lo.max(other.lo).min(H);
289
290 if (H.x < L.x && H.y < L.y && H.z < L.z) {
291 return empty();
292 } else {
293 return AABox(L, H);
294 }
295 }
