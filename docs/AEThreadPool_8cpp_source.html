61 void AEThreadPool::pulse(AEThread& pulseThread)
62 {
63 if (m_shutdownRequested)
64 {
65 lock_guard<mutex> guard(m_mtx);
66
67 m_tasks.clear();
68 for (auto& thread : m_threads) { thread->killAndJoin(); }
69 m_threads.clear();
70 pulseThread.killAndJoin();
71 return;
72 }
73
74 {
75 // Prune threads if we're above the soft limit
76 if (m_threadCount > m_softMaximumThreadCount)
77 {
78 lock_guard<mutex> guard(m_mtx);
79
80 m_threads.remove_if([this](unique_ptr<AEThread>& thread)
81 {
82 if (thread->isWorking())
83 return false;
84
85 --this->m_threadCount;
86 return true;
87 });
88 }
89
90 // Adjust thread count if we need to
91 if (m_threadCount < m_minimumThreadCount)
92 {
93 lock_guard<mutex> guard(m_mtx);
94 while (m_threadCount < m_minimumThreadCount)
95 spawnThread();
96 }
97 }
98
99 {
100 lock_guard<mutex> guard(m_mtx);
101
102 // Remove dead tasks
103 const auto remove_dead_tasks = [](unique_ptr<Task>& task) { return task->m_done; };
104 m_tasks.remove_if(remove_dead_tasks);
105 m_highPriorityTasks.remove_if(remove_dead_tasks);
106
107 const auto internal_task_runner = [](unique_ptr<Task>& task, AEThread& poolThread)
108 {
109 task->m_running = true;
110 task->m_task(poolThread);
111 task->m_running = false;
112 poolThread.requestKill();
113 if (task->isFireOnceTask())
114 task->m_done = true;
115 else
116 task->m_lastPulse = steady_clock::now();
117 };
118
119 // Check every thread to see if it's doing work
120 for (auto& thread : m_threads)
121 {
122 if (thread->isDone())
123 {
124 const auto now = steady_clock::now();
125
126 for (auto& task : m_highPriorityTasks)
127 {
128 if (task->m_running)
129 continue;
130
131 thread->setWork([internal_task_runner, &task](AEThread& poolThread)
132 {
133 internal_task_runner(task, poolThread);
134 });
135
136 thread->reboot();
137 break;
138 }
139
140 // We might have work at this point, so check again
141 if (thread->isWorking())
142 continue;
143
144 for (auto& task : m_tasks)
145 {
146 if (task->canFire(now))
147 {
148 thread->setWork([internal_task_runner, &task](AEThread& poolThread)
149 {
150 internal_task_runner(task, poolThread);
151 });
152
153 thread->reboot();
154 break;
155 }
156 }
157 }
158 }
159
160 // Spawn new threads for any high priority tasks that weren't dispatched already
161 for (auto& task : m_highPriorityTasks)
162 {
163 if (task->m_running)
164 continue;
165
166 // Try to spawn a thread, give up if we can't
167 auto new_thread = spawnThread();
168 if (new_thread == nullptr)
169 break;
170
171 new_thread->setWork([internal_task_runner, &task](AEThread& poolThread)
172 {
173 internal_task_runner(task, poolThread);
174 });
175 new_thread->reboot();
176 }
177 }
178 }
