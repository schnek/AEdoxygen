18inline bool inRangeYZX(const float* v1, const float* v2, const float r, const float h)
19{
20 const float dx = v2[0] - v1[0];
21 const float dy = v2[1] - v1[1]; // elevation
22 const float dz = v2[2] - v1[2];
23 return (dx * dx + dz * dz) < r * r && fabsf(dy) < h;
24}

247enum TargetFilter : uint32_t
248{
249 // Standard filters
250 TargetFilter_None = 0, // 0
251 TargetFilter_Closest = 1 << 0, // 1
252 TargetFilter_Friendly = 1 << 1, // 2
253 TargetFilter_NotCurrent = 1 << 2, // 4
254 TargetFilter_Wounded = 1 << 3, // 8
255 TargetFilter_SecondMostHated = 1 << 4, // 16
256 TargetFilter_Aggroed = 1 << 5, // 32
257 TargetFilter_Corpse = 1 << 6, // 64
258 TargetFilter_InMeleeRange = 1 << 7, // 128
259 TargetFilter_InRangeOnly = 1 << 8, // 256
260 TargetFilter_IgnoreSpecialStates = 1 << 9, // 512 - not really a TargetFilter, more like requirement for spell
261 TargetFilter_IgnoreLineOfSight = 1 << 10, // 1024
262 TargetFilter_Current = 1 << 11, // 2048
263 TargetFilter_LowestHealth = 1 << 12, // 4096
264 TargetFilter_Health = 1 << 13, // 8192
265 TargetFilter_AOE = 1 << 14, // 16348 - not really a Filter just no Target for AOE spells
266 TargetFilter_Self = 1 << 15, // 32768 - mostlikely return ourself unless we set aura filtering and we dont have it
267 TargetFilter_Caster = 1 << 16, // 65536 - Mana Based Class
268 TargetFilter_Casting = 1 << 17, // 131072 - Target is Casting currently
269 TargetFilter_Player = 1 << 18, // 262144 - Players Only
270
271 // Predefined filters
272 TargetFilter_ClosestFriendly = TargetFilter_Closest | TargetFilter_Friendly, // 3
273 TargetFilter_ClosestNotCurrent = TargetFilter_Closest | TargetFilter_NotCurrent, // 5
274 TargetFilter_WoundedFriendly = TargetFilter_Wounded | TargetFilter_Friendly, // 10
275 TargetFilter_FriendlyCorpse = TargetFilter_Corpse | TargetFilter_Friendly, // 66
276 TargetFilter_ClosestFriendlyCorpse = TargetFilter_Closest | TargetFilter_FriendlyCorpse, // 67
277 TargetFilter_CurrentInRangeOnly = TargetFilter_Current | TargetFilter_InRangeOnly, // 2304
278 TargetFilter_WoundedFriendlyLowestHealth = TargetFilter_Wounded | TargetFilter_Friendly | TargetFilter_LowestHealth, // 4106
279 TargetFilter_WoundedFriendlyLowestHealthInRange = TargetFilter_Wounded | TargetFilter_Friendly | TargetFilter_LowestHealth | TargetFilter_InRangeOnly, // 4362
280 TargetFilter_CasterWhileCasting = TargetFilter_Casting | TargetFilter_Caster, // 196608
281 TargetFilter_SelfBelowHealth = TargetFilter_Self | TargetFilter_Health // 40960
282};
