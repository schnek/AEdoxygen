40void AccountMgr::addAccount(Field* field)
41{
42 auto account = std::make_unique<Account>();
43
44 account->AccountId = field[0].asUint32();
45
46 std::string accountName = field[1].asCString();
47 std::string encryptedPassword = field[2].asCString();
48
49 account->AccountFlags = field[3].asUint8();
50 account->Banned = field[4].asUint32();
51 account->forcedLanguage = field[5].asCString();
52 account->Muted = field[6].asUint32();
53
54 if (static_cast<uint32_t>(UNIXTIME) > account->Banned && account->Banned != 0 && account->Banned != 1)
55 {
56 account->Banned = 0;
57 sLogger.debug("Account {}'s ban has expired.", accountName);
58 sLogonSQL->Execute("UPDATE accounts SET banned = 0 WHERE id = %u", account->AccountId);
59 }
60
61 if (account->forcedLanguage != "enUS")
62 account->forcedLocale = true;
63 else
64 account->forcedLocale = false;
65
66 if (static_cast<uint32_t>(UNIXTIME) > account->Muted && account->Muted != 0 && account->Muted != 1)
67 {
68 account->Muted = 0;
69 sLogger.debug("Account {}'s mute has expired.", accountName);
70 sLogonSQL->Execute("UPDATE accounts SET muted = 0 WHERE id = %u", account->AccountId);
71 }
72
73 if (encryptedPassword.size() == 40)
74 {
75 BigNumber bn;
76 bn.SetHexStr(encryptedPassword.c_str());
77 if (bn.GetNumBytes() < 20)
78 {
79 memcpy(account->SrpHash, bn.AsByteArray(), bn.GetNumBytes());
80 for (auto n = bn.GetNumBytes(); n <= 19; n++)
81 account->SrpHash[n] = static_cast<uint8_t>(0);
82
83 std::reverse(std::begin(account->SrpHash), std::end(account->SrpHash));
84 }
85 else
86 {
87 memcpy(account->SrpHash, bn.AsByteArray(), 20);
88 std::reverse(std::begin(account->SrpHash), std::end(account->SrpHash));
89 }
90 }
91 else
92 {
93 sLogger.failure("Account `{}` has incorrect number of bytes in encrypted password! Disabling.", accountName);
94 memset(account->SrpHash, 0, 20);
95 }
96
97 AscEmu::Util::Strings::toUpperCase(accountName);
98
99 _accountMap.insert_or_assign(accountName, std::move(account));
100}

173void AccountMgr::reloadAccounts(bool silent)
174{
175 std::lock_guard lock(accountMgrMutex);
176
177 if (!silent)
178 sLogger.info("[AccountMgr] Reloading Accounts...");
179
180 std::set<std::string> account_list;
181
182 auto result = sLogonSQL->Query("SELECT id, acc_name, encrypted_password, flags, banned, forceLanguage, muted FROM accounts");
183 if (result)
184 {
185 do
186 {
187 Field* field = result->Fetch();
188 std::string accountName = field[1].asCString();
189
190 AscEmu::Util::Strings::toUpperCase(accountName);
191
192 const auto account = _getAccountByNameLockFree(accountName);
193 if (account == nullptr)
194 addAccount(field);
195 else
196 updateAccount(account, field);
197
198 account_list.insert(accountName);
199
200 } while (result->NextRow());
201 }
202
203 for (auto accounts = _accountMap.begin(); accounts != _accountMap.end();)
204 {
205 auto accountsCopy = accounts;
206 ++accounts;
207
208 if (account_list.find(accountsCopy->first) == account_list.end())
209 _accountMap.erase(accountsCopy);
210 else
211 accountsCopy->second->UsernamePtr = const_cast<std::string*>(&accountsCopy->first);
212 }
213
214 if (!silent)
215 sLogger.info("[AccountMgr] Found {} accounts.", static_cast<uint32_t>(_accountMap.size()));
216}
