78bool ChatCommandHandler::handleAchievementResetCommand(const char* args, WorldSession* m_session)
79{
80 Player* selected_player = GetSelectedPlayer(m_session, true, true);
81 if (selected_player == nullptr)
82 return true;
83
84 if (!*args)
85 return false;
86
87 bool reset_achievement = true;
88 bool reset_criteria = false;
89 bool resetAll = false;
90 uint32_t achievement_id = 0;
91
92 std::string criteria(args, 9);
93
94 if (AscEmu::Util::Strings::isEqual(criteria, "criteria "))
95 {
96 achievement_id = std::stoul(args + 9);
97 if (achievement_id == 0)
98 {
99 if (!AscEmu::Util::Strings::isEqual(args + 9, "all"))
100 return false;
101
102 resetAll = true;
103 }
104 reset_criteria = true;
105 reset_achievement = false;
106 }
107 else if (AscEmu::Util::Strings::isEqual(args, "all"))
108 {
109 resetAll = true;
110 reset_criteria = true;
111 }
112 else
113 {
114 achievement_id = std::stoul(args);
115 if (achievement_id == 0)
116 {
117 if (args == nullptr)
118 return false;
119
120 const char* ptr = strstr(args, "|Hachievement:");
121 if (ptr == nullptr)
122 return false;
123
124 // achievement id is just past "|Hachievement:" (14 bytes)
125 achievement_id = std::stoul(ptr + 14);
126
127 if (achievement_id == 0)
128 return false;
129 }
130 }
131
132 if (reset_achievement)
133 selected_player->getAchievementMgr()->gmResetAchievement(achievement_id, resetAll);
134
135 if (reset_criteria)
136 selected_player->getAchievementMgr()->gmResetCriteria(achievement_id, resetAll);
137
138 return true;
139}
