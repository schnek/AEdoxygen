49void AchievementMgr::loadFromDb(QueryResult* _achievementResult, QueryResult* _criteriaResult)
50{
51 if (_achievementResult)
52 {
53 do
54 {
55 Field* field = _achievementResult->Fetch();
56 uint32_t id = field[0].asUint32();
57 if (m_completedAchievements[id] == 0)
58 m_completedAchievements[id] = field[1].asUint32();
59 else
60 sLogger.failure("Duplicate completed achievement {} for player {}, skipping", id, m_player->getGuidLow());
61 } while (_achievementResult->NextRow());
62 }
63
64 if (_criteriaResult)
65 {
66 do
67 {
68 Field* field = _criteriaResult->Fetch();
69 uint32_t progress_id = field[0].asUint32();
70 if (m_criteriaProgress[progress_id] == nullptr)
71 {
72 m_criteriaProgress[progress_id] = std::make_unique<CriteriaProgress>(progress_id, field[1].asUint32(), static_cast<time_t>(field[2].asUint64()));
73 }
74 else
75 sLogger.failure("Duplicate criteria progress {} for player {}, skipping", progress_id, m_player->getGuidLow());
76
77 } while (_criteriaResult->NextRow());
78 }
79}

81void AchievementMgr::saveToDb(QueryBuffer* _buffer)
82{
83 if (!m_completedAchievements.empty())
84 {
85 std::ostringstream ss;
86
87 ss << "DELETE FROM character_achievement WHERE guid = ";
88 ss << m_player->getGuidLow();
89 ss << ";";
90
91 if (_buffer == nullptr)
92 CharacterDatabase.ExecuteNA(ss.str().c_str());
93 else
94 _buffer->AddQueryNA(ss.str().c_str());
95
96 ss.rdbuf()->str("");
97
98 ss << "INSERT INTO character_achievement VALUES ";
99 bool first = true;
100 for (auto iterCompletedAchievements : m_completedAchievements)
101 {
102 if (ss.str().length() >= 16000)
103 {
104 // SQL query length is limited to 16384 characters
105 if (_buffer == nullptr)
106 CharacterDatabase.ExecuteNA(ss.str().c_str());
107 else
108 _buffer->AddQueryNA(ss.str().c_str());
109
110 ss.str("");
111 ss << "INSERT INTO character_achievement VALUES ";
112 first = true;
113 }
114
115 if (!first)
116 ss << ", ";
117 else
118 first = false;
119
120 ss << "(" << m_player->getGuidLow() << ", " << iterCompletedAchievements.first << ", " << iterCompletedAchievements.second << ")";
121 }
122
123 if (_buffer == nullptr)
124 CharacterDatabase.ExecuteNA(ss.str().c_str());
125 else
126 _buffer->AddQueryNA(ss.str().c_str());
127 }
128
129 if (!m_criteriaProgress.empty())
130 {
131 std::ostringstream ss;
132
133 ss << "DELETE FROM character_achievement_progress WHERE guid = ";
134 ss << m_player->getGuidLow();
135 ss << ";";
136
137 if (_buffer == nullptr)
138 CharacterDatabase.ExecuteNA(ss.str().c_str());
139 else
140 _buffer->AddQueryNA(ss.str().c_str());
141
142 ss.rdbuf()->str("");
143
144 ss << "INSERT INTO character_achievement_progress VALUES ";
145
146 bool first = true;
147 for (const auto& iterCriteriaProgress : m_criteriaProgress)
148 {
149 if (canSaveAchievementProgressToDB(iterCriteriaProgress.second.get()))
150 {
151 // only save some progresses, others will be updated when character logs in
152 if (ss.str().length() >= 16000)
153 {
154 // SQL query length is limited to 16384 characters
155 if (_buffer == nullptr)
156 CharacterDatabase.ExecuteNA(ss.str().c_str());
157 else
158 _buffer->AddQueryNA(ss.str().c_str());
159 ss.str("");
160 ss << "INSERT INTO character_achievement_progress VALUES ";
161 first = true;
162 }
163
164 if (!first)
165 ss << ", ";
166 else
167 first = false;
168
169 ss << "(" << m_player->getGuidLow() << ", " << iterCriteriaProgress.first << ", " << iterCriteriaProgress.second->counter << ", " << iterCriteriaProgress.second->date << ")";
170 }
171 }
172
173 if (!first)
174 {
175 // don't execute query if there's no entries to save
176 if (_buffer == nullptr)
177 CharacterDatabase.ExecuteNA(ss.str().c_str());
178 else
179 _buffer->AddQueryNA(ss.str().c_str());
180 }

184bool AchievementMgr::canCompleteCriteria(WDB::Structures::AchievementCriteriaEntry const* _achievementCriteria, AchievementCriteriaTypes _type, Player* _player) const
185{
186 switch (_type)
187 {
188 case ACHIEVEMENT_CRITERIA_TYPE_REACH_LEVEL:
189 case ACHIEVEMENT_CRITERIA_TYPE_COMPLETE_QUEST_COUNT:
190 case ACHIEVEMENT_CRITERIA_TYPE_COMPLETE_QUESTS_IN_ZONE:
191 case ACHIEVEMENT_CRITERIA_TYPE_COMPLETE_QUEST:
192 case ACHIEVEMENT_CRITERIA_TYPE_GAIN_REPUTATION:
193 case ACHIEVEMENT_CRITERIA_TYPE_GAIN_EXALTED_REPUTATION:
194 case ACHIEVEMENT_CRITERIA_TYPE_NUMBER_OF_MOUNTS:
195 case ACHIEVEMENT_CRITERIA_TYPE_REACH_SKILL_LEVEL:
196 case ACHIEVEMENT_CRITERIA_TYPE_LEARN_SKILL_LEVEL:
197 case ACHIEVEMENT_CRITERIA_TYPE_BUY_BANK_SLOT:
198 case ACHIEVEMENT_CRITERIA_TYPE_WIN_RATED_ARENA:
199 case ACHIEVEMENT_CRITERIA_TYPE_ROLL_NEED_ON_LOOT:
200 case ACHIEVEMENT_CRITERIA_TYPE_ROLL_GREED_ON_LOOT:
201 return true;
202 case ACHIEVEMENT_CRITERIA_TYPE_EXPLORE_AREA:
203 return _player->hasOverlayUncovered(_achievementCriteria->explore_area.areaReference);
204 case ACHIEVEMENT_CRITERIA_TYPE_COMPLETE_ACHIEVEMENT:
205 return m_completedAchievements.find(_achievementCriteria->complete_achievement.linkedAchievement) != m_completedAchievements.end();
206 case ACHIEVEMENT_CRITERIA_TYPE_LEARN_SPELL:
207 return _player->hasSpell(_achievementCriteria->learn_spell.spellID);
208 default:
209 break;
210 }
211
212 return false;
213}

301/// depend on the achievement type. Generally, miscvalue1 is an ID of some type (quest ID,
302/// item ID, faction ID, etc.), and miscvalue2 is the amount to increase the progress.

304{
305 if (m_player->getSession()->HasGMPermissions() && worldConfig.gm.disableAchievements)

311 selectedGUID = getPlayer()->getTargetGuid();
312 }
313 AchievementCriteriaEntryList const & achievementCriteriaList = sObjectMgr.getAchievementCriteriaByType(_type);
314 for (const auto achievementCriteria : achievementCriteriaList)
315 {
316 if (isCompletedCriteria(achievementCriteria))
317 continue;
318
319 if (achievementCriteria->groupFlag & ACHIEVEMENT_CRITERIA_GROUP_NOT_IN_GROUP && getPlayer()->getGroup())
320 continue;
321
322 const auto achievement = sAchievementStore.lookupEntry(achievementCriteria->referredAchievement);
323 if (!achievement)

326 // achievement requires a faction of which the player is not a member

1198 doneCompleted = true;
1199 }

1205 {

1250 }
