66inline uint32_t secsToTimeBitFields(time_t secs)
67{
68 tm* lt = localtime(&secs);
69 return static_cast<uint32_t>((lt->tm_year - 100) << 24 | lt->tm_mon << 20 | (lt->tm_mday - 1) << 14 | lt->tm_wday << 11 | lt->tm_hour << 6 | lt->tm_min);
70}
