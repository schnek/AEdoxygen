520void AlteracValley::AVNode::Assault(Player* plr)
521{
522 // player assaulted the control point.
523 // safety check
524 DLLLogDetail("AlteracValley : AVNode::Assault(%s) : entering", m_template->m_name);
525 if (plr->isTeamAlliance() && (m_state == AV_NODE_STATE_ALLIANCE_ASSAULTING || m_state == AV_NODE_STATE_ALLIANCE_CONTROLLED))
526 return;
527
528 if (plr->isTeamHorde() && (m_state == AV_NODE_STATE_HORDE_ASSAULTING || m_state == AV_NODE_STATE_HORDE_CONTROLLED))
529 return;
530
531 if (m_destroyed)
532 return;
533
534 // are we returning the node to us?
535 if ((plr->isTeamAlliance() && m_lastState == AV_NODE_STATE_ALLIANCE_CONTROLLED) ||
536 (plr->isTeamHorde() && m_lastState == AV_NODE_STATE_HORDE_CONTROLLED))
537 {
538 // set the state for capture
539 m_state = plr->isTeamAlliance() ? AV_NODE_STATE_ALLIANCE_ASSAULTING : AV_NODE_STATE_HORDE_ASSAULTING;
540
541 // no timer delay
542 Capture();
543
544 // pvp data
545 if (m_template->m_isGraveyard)
546 plr->m_bgScore.MiscData[BattlegroundDef::AV_GRAVEYARDS_DEFENDED]++;
547 else
548 plr->m_bgScore.MiscData[BattlegroundDef::AV_TOWERS_DEFENDED]++;
549
550 return;
551 }
552
553 // start timer
554 sEventMgr.RemoveEvents(m_bg, EVENT_AV_CAPTURE_CP_0 + m_nodeId);
555
556 sEventMgr.AddEvent(m_bg, &AlteracValley::EventAssaultControlPoint, m_nodeId, EVENT_AV_CAPTURE_CP_0 + m_nodeId, TimeVars::Minute * 4 * 1000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
557
558 // update state
559 ChangeState(plr->isTeamHorde() ? AV_NODE_STATE_HORDE_ASSAULTING : AV_NODE_STATE_ALLIANCE_ASSAULTING);
560
561 // pvp data
562 if (m_template->m_isGraveyard)
563 {
564 // send message
565 m_bg->sendChatMessage(CHAT_MSG_BG_EVENT_ALLIANCE + plr->getTeam(), 0, "%s claims the %s! If left unchallenged, the %s will control it!", plr->getName().c_str(), m_template->m_name,
566 plr->isTeamHorde() ? "Horde" : "Alliance");
567
568 plr->m_bgScore.MiscData[BattlegroundDef::AV_GRAVEYARDS_ASSAULTED]++;
569 }
570 else
571 {
572 m_bg->Herald("%s is under attack! If left unchecked the %s will destroy it!", m_template->m_name, plr->isTeamHorde() ? "Horde" : "Alliance");
573 plr->m_bgScore.MiscData[BattlegroundDef::AV_TOWERS_ASSAULTED]++;
574 }
