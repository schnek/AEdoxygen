209void AnubarakAI::summonFrostSphere(CreatureAIFunc pThis)
210{
211 // Get a random starting index within the valid range
212 uint8_t startAt = Util::getRandomUInt(0, static_cast<uint32_t>(sphereGuids.size() - 1));
213 uint8_t attempts = 0;
214
215 do {
216 // Get the current sphere index to check
217 uint8_t currentIndex = (startAt + attempts) % 6;
218
219 Creature* pSphere = getCreature()->getWorldMapCreature(sphereGuids[currentIndex]);
220
221 if (pSphere && !pSphere->hasAurasWithId(anubarak::SPELL_FROST_SPHERE))
222 {
223 // Create and summon a new frost sphere
224 Creature* summon = summonCreature(anubarak::NPC_FROST_SPHERE, anubarak::SphereSpawn[currentIndex]);
225
226 if (summon)
227 {
228 // If summoning successful, update the sphereGuids with the new creature's GUID
229 sphereGuids[currentIndex] = summon->getGuid();
230 }
231
232 // Exit the loop, we found a suitable sphere or summoned a new one
233 break;
234 }
235
236 // Increment attempts counter to move to the next index in the next iteration
237 attempts++;
238
239 } while (attempts < sphereGuids.size());
240
241 repeatFunctionFromScheduler(pThis, 20s);
242}

245{
246 // Check if the burrowGuids vector is not empty
247 if (!burrowGuids.empty())
248 {
249 // Get a random index within the valid range
250 uint32_t at = Util::getRandomUInt(0, static_cast<uint32_t>(burrowGuids.size() - 1));
251
252 // Get the iterator pointing to the selected element in the vector
253 auto i = burrowGuids.begin() + at;
254
255 // Check if the element at the selected index is a valid creature
256 if (Creature* pBurrow = getCreature()->getWorldMapCreature(*i))
257 {
258 // Cast the spell on the valid creature
259 pBurrow->castSpell(pBurrow, 66340, false);
260 }
261 }
262
263 repeatFunctionFromScheduler(pThis, 4s);

420{

498void SpikeAI::handlePermafrostHit(Creature* pCreature)
499{
500 if (pCreature->getEntry() != anubarak::NPC_FROST_SPHERE)
501 return;
502
503 if (getScriptPhase() == anubarak::PHASE_NO_MOVEMENT)
504 return;
505
506 if (getCreature()->IsWithinDistInMap(pCreature, 7.0f))
507 {
508 switch (getScriptPhase())
509 {
510 case anubarak::PHASE_IMPALE_NORMAL:
511 _removeAura(anubarak::SPELL_SPIKE_SPEED1);
512 break;
513 case anubarak::PHASE_IMPALE_MIDDLE:
514 _removeAura(anubarak::SPELL_SPIKE_SPEED2);
515 break;
516 case anubarak::PHASE_IMPALE_FAST:
517 _removeAura(anubarak::SPELL_SPIKE_SPEED3);
518 break;
519 default:
520 break;
521 }
522
523 castSpellOnSelf(anubarak::SPELL_SPIKE_FAIL, true);
524
525 // After the spikes hit the icy surface they can't move for about ~5 seconds
526 setScriptPhase(anubarak::PHASE_NO_MOVEMENT);
527 phaseTimer = 5 * TimeVarsMs::Second;
528 setRooted(false);
529 getMovementManager()->moveIdle();
530 getMovementManager()->clear();
531 }
