169void ArathiBasin::SpawnBuff(uint32_t x)
170{
171 uint32_t chosen_buffid = buffentries[Util::getRandomUInt(2)];
172 auto gameobject_info = sMySQLStore.getGameObjectProperties(chosen_buffid);
173 if (gameobject_info == nullptr)
174 return;
175
176 if (m_buffs[x] == nullptr)
177 {
178 m_buffs[x] = spawnGameObject(chosen_buffid, LocationVector(BuffCoordinates[x][0], BuffCoordinates[x][1], BuffCoordinates[x][2],
179 BuffCoordinates[x][3]), 0, 114, 1);
180
181 m_buffs[x]->setLocalRotation(0.f, 0.f, BuffRotations[x][0], BuffRotations[x][1]);
182 m_buffs[x]->setState(GO_STATE_CLOSED);
183 m_buffs[x]->setGoType(GAMEOBJECT_TYPE_TRAP);
184 m_buffs[x]->setAnimationProgress(100);
185 m_buffs[x]->PushToWorld(m_mapMgr);
186 }
187 else
188 {
189 // only need to reassign guid if the entry changes.
190 if (m_buffs[x]->IsInWorld())
191 m_buffs[x]->RemoveFromWorld(false);
192
193 if (chosen_buffid != m_buffs[x]->getEntry())
194 {
195 m_buffs[x]->SetNewGuid(m_mapMgr->generateGameobjectGuid());
196 m_buffs[x]->setEntry(chosen_buffid);
197 m_buffs[x]->SetGameObjectProperties(gameobject_info);
198 }
199
200 m_buffs[x]->PushToWorld(m_mapMgr);
201 }

204void ArathiBasin::SpawnControlPoint(uint32_t Id, uint32_t Type)
205{
206 auto gameobject_info = sMySQLStore.getGameObjectProperties(ControlPointGoIds[Id][Type]);
207 if (gameobject_info == nullptr)
208 return;
209
210 auto gi_aura = gameobject_info->raw.parameter_3 ? sMySQLStore.getGameObjectProperties(gameobject_info->raw.parameter_3) : nullptr;
211
212 if (m_controlPoints[Id] == nullptr)
213 {
214 m_controlPoints[Id] = spawnGameObject(gameobject_info->entry, LocationVector(ControlPointCoordinates[Id][0], ControlPointCoordinates[Id][1],
215 ControlPointCoordinates[Id][2], ControlPointCoordinates[Id][3]), 0, 35, 1.0f);

217 m_controlPoints[Id]->setLocalRotation(0.f, 0.f, ControlPointRotations[Id][0], ControlPointRotations[Id][1]);

221 m_controlPoints[Id]->setDynamicFlags(GO_DYN_FLAG_INTERACTABLE);
222 m_controlPoints[Id]->setDisplayId(gameobject_info->display_id);
223
224 switch (Type)
225 {
226 case AB_SPAWN_TYPE_ALLIANCE_ASSAULT:
227 case AB_SPAWN_TYPE_ALLIANCE_CONTROLLED:
228 m_controlPoints[Id]->SetFaction(2);

240
241 m_controlPoints[Id]->PushToWorld(m_mapMgr);

243 else

246 m_controlPoints[Id]->RemoveFromWorld(false);

269 }
270
271 m_controlPoints[Id]->SetGameObjectProperties(gameobject_info);
272 m_controlPoints[Id]->PushToWorld(m_mapMgr);
273 }
274
275 if (gi_aura == nullptr)
276 {
277 // remove it if it exists
278 if (m_controlPointAuras[Id] != nullptr && m_controlPointAuras[Id]->IsInWorld())
279 m_controlPointAuras[Id]->RemoveFromWorld(false);

286 m_controlPointAuras[Id] = spawnGameObject(gi_aura->entry, LocationVector(ControlPointCoordinates[Id][0], ControlPointCoordinates[Id][1],
287 ControlPointCoordinates[Id][2], ControlPointCoordinates[Id][3]), 0, 35, 1.0f);

297 if (m_controlPointAuras[Id]->IsInWorld())

300 // re-spawn the aura
301 m_controlPointAuras[Id]->SetNewGuid(m_mapMgr->generateGameobjectGuid());

471void ArathiBasin::EventUpdateResources(uint32_t Team)
472{
473 uint32_t resource_fields[2] = { WORLDSTATE_AB_ALLIANCE_RESOURCES, WORLDSTATE_AB_HORDE_RESOURCES };
474
475 uint32_t current_resources = m_resources[Team];
476 uint32_t current_bases = m_capturedBases[Team];
477
478 if (current_bases > 5)
479 current_bases = 5;
480
481 // figure out how much resources we have to add to that team based on the number of captured bases.
482 current_resources += (PointBonusPerUpdate[current_bases]);
483
484 // did it change?
485 if (current_resources == m_resources[Team])
486 return;
487
488 // check for overflow
489 if (current_resources > RESOURCES_WINVAL)
490 current_resources = RESOURCES_WINVAL;
491
492 m_resources[Team] = current_resources;
493 if ((current_resources - m_lastRepGainResources[Team]) >= resourcesToGainBR)
494 {
495 std::lock_guard<std::recursive_mutex> lock(m_mutex);
496 for (std::set<Player*>::iterator itr = m_players[Team].begin(); itr != m_players[Team].end(); ++itr)
497 {
498 uint32_t fact = (*itr)->isTeamHorde() ? 510 : 509; //The Defilers : The League of Arathor
499 (*itr)->modFactionStanding(fact, 10);
500 }
501 m_lastRepGainResources[Team] += resourcesToGainBR;
502 }
503
504 if ((current_resources - m_lastHonorGainResources[Team]) >= resourcesToGainBH)
505 {
506 uint32_t honorToAdd = m_honorPerKill;
507 std::lock_guard<std::recursive_mutex> lock(m_mutex);
508 for (std::set<Player*>::iterator itr = m_players[Team].begin(); itr != m_players[Team].end(); ++itr)
509 {
510 (*itr)->m_bgScore.BonusHonor += honorToAdd;
511 HonorHandler::AddHonorPointsToPlayer((*itr), honorToAdd);
512 }
513
514 updatePvPData();
515 m_lastHonorGainResources[Team] += resourcesToGainBH;
516 }
517
518 // update the world states
519 setWorldState(resource_fields[Team], current_resources);
520
521 if (current_resources >= RESOURCES_WARNING_THRESHOLD && !m_nearingVictory[Team])
522 {
523 m_nearingVictory[Team] = true;
524 sendChatMessage(Team ? CHAT_MSG_BG_EVENT_HORDE : CHAT_MSG_BG_EVENT_ALLIANCE, static_cast<uint64_t>(0), "The %s has gathered %u resources and is nearing victory!", Team ? "Horde" : "Alliance", current_resources);
525 uint32_t sound = BattlegroundDef::ALLIANCE_BGALMOSTEND - Team;
526 playSoundToAll(sound);
527 }
528
529 // check for winning condition
530 if (current_resources == RESOURCES_WINVAL)
531 {
532 sEventMgr.RemoveEvents(this);
533 sEventMgr.AddEvent(static_cast<Battleground*>(this), &Battleground::close, EVENT_BATTLEGROUND_CLOSE, 120000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
534
535 this->endBattleground(Team == TEAM_ALLIANCE ? TEAM_ALLIANCE : TEAM_HORDE);
536 }
537}
