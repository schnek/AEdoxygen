49TriangleBoundary::TriangleBoundary(LocationVector const& pointA, LocationVector const& pointB, LocationVector const& pointC, bool isInverted) :
50 AreaBoundary(isInverted), _a(pointA), _b(pointB), _c(pointC), _abx(_b.getDoublePositionX()-_a.getDoublePositionX()), _bcx(_c.getDoublePositionX()-_b.getDoublePositionX()), _cax(_a.getDoublePositionX() - _c.getDoublePositionX()), _aby(_b.getDoublePositionY()-_a.getDoublePositionY()), _bcy(_c.getDoublePositionY()-_b.getDoublePositionY()), _cay(_a.getDoublePositionY() - _c.getDoublePositionY()) { }
51bool TriangleBoundary::isWithinBoundaryArea(LocationVector const* pos) const
52{
53 // half-plane signs
54 bool sign1 = ((-_b.getDoublePositionX() + pos->getPositionX()) * _aby - (-_b.getDoublePositionY() + pos->getPositionY()) * _abx) < 0;
55 bool sign2 = ((-_c.getDoublePositionX() + pos->getPositionX()) * _bcy - (-_c.getDoublePositionY() + pos->getPositionY()) * _bcx) < 0;
56 bool sign3 = ((-_a.getDoublePositionX() + pos->getPositionX()) * _cay - (-_a.getDoublePositionY() + pos->getPositionY()) * _cax) < 0;
57
58 // if all signs are the same, the point is inside the triangle
59 return ((sign1 == sign2) && (sign2 == sign3));
60}

64ParallelogramBoundary::ParallelogramBoundary(LocationVector const& cornerA, LocationVector const& cornerB, LocationVector const& cornerD, bool isInverted) :
65 AreaBoundary(isInverted), _a(cornerA), _b(cornerB), _d(cornerD), _c(DoublePosition(_d.getDoublePositionX() + (_b.getDoublePositionX() - _a.getDoublePositionX()), _d.getDoublePositionY() + (_b.getDoublePositionY() - _a.getDoublePositionY()))), _abx(_b.getDoublePositionX() - _a.getDoublePositionX()), _dax(_a.getDoublePositionX() - _d.getDoublePositionX()), _aby(_b.getDoublePositionY() - _a.getDoublePositionY()), _day(_a.getDoublePositionY() - _d.getDoublePositionY()) { }
66bool ParallelogramBoundary::isWithinBoundaryArea(LocationVector const* pos) const
67{
68 // half-plane signs
69 bool sign1 = ((-_b.getDoublePositionX() + pos->getPositionX()) * _aby - (-_b.getDoublePositionY() + pos->getPositionY()) * _abx) < 0;
70 bool sign2 = ((-_a.getDoublePositionX() + pos->getPositionX()) * _day - (-_a.getDoublePositionY() + pos->getPositionY()) * _dax) < 0;
71 bool sign3 = ((-_d.getDoublePositionY() + pos->getPositionY()) * _abx - (-_d.getDoublePositionX() + pos->getPositionX()) * _aby) < 0; // AB = -CD
72 bool sign4 = ((-_c.getDoublePositionY() + pos->getPositionY()) * _dax - (-_c.getDoublePositionX() + pos->getPositionX()) * _day) < 0; // DA = -BC
73
74 // if all signs are equal, the point is inside
75 return ((sign1 == sign2) && (sign2 == sign3) && (sign3 == sign4));
76}
