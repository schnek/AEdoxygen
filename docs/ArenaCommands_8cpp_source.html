35bool ChatCommandHandler::HandleArenaCreateTeam(const char* args, WorldSession* m_session)
36{
37 uint32_t teamType;
38 char teamName[1000];
39
40 auto player = GetSelectedPlayer(m_session, true, true);
41 if (sscanf(args, "%u %[^\n]", &teamType, teamName) != 2)
42 {
43 systemMessage(m_session, "Invalid syntax. Usage: .arena createteam <type> <name>");
44 return true;
45 }
46
47 uint8_t internalType = GetArenaTeamInternalType(teamType, m_session);
48 if (internalType == 10)
49 return true;
50
51 if (player == nullptr)
52 {
53 systemMessage(m_session, "Selected player not found!");
54 return true;
55 }
56
57 if (player->isInArenaTeam(internalType))
58 {
59 redSystemMessage(m_session, "Player: {} is already in an arena team of that type!", player->getName());
60 return true;
61 }
62
63 ArenaTeamEmblem emblem{ .emblemStyle = 22, .emblemColour = 4292133532UL, .borderStyle = 1,
64 .borderColour = 4294931722UL, .backgroundColour = 4284906803UL };
65
66 if (auto* const arenaTeam = sObjectMgr.createArenaTeam(internalType, player, teamName, 1500, emblem))
67 {
68 player->setArenaTeam(arenaTeam->m_type, arenaTeam);
69 greenSystemMessage(m_session, "Arena team created for Player: {} Type: {}", player->getName(), teamType);
70 }
71
72 return true;
73}
