74 return;

142 ss << ",'0 0 0 0 0 0'";
143 }
144 }
145
146 for (; i < 10; ++i)
147 {
148 ss << ",'0 0 0 0 0 0'";

151 ss << ")";

155void ArenaTeam::destroy()
156{
157 std::vector<CachedCharacterInfo const*> toDestroyMembers;
158 toDestroyMembers.reserve(m_memberCount);
159
160 char buffer[1024];
161 snprintf(buffer, 1024, "The arena team, '%s', disbanded.", m_name.c_str());
162
163 sendPacket(SmsgMessageChat(SystemMessagePacket(buffer)).serialise().get());
164
165 for (uint32_t i = 0; i < m_memberCount; ++i)
166 {
167 if (m_members[i].Info)
168 toDestroyMembers.push_back(m_members[i].Info);
169 }
170
171 for (auto& itr : toDestroyMembers)
172 removeMember(itr);
173
174 // TODO: arena team is not removed from db? -Appled
175 sObjectMgr.removeArenaTeam(this);
176}

210bool ArenaTeam::addMember(CachedCharacterInfo const* cachedCharInfo)
211{
212 if (!cachedCharInfo)
213 return false;
214
215 if (Player* loggedInPlayer = sObjectMgr.getPlayer(cachedCharInfo->guid))
216 {
217 if (m_memberCount >= m_slots)
218 return false;
219
220 memset(&m_members[m_memberCount], 0, sizeof(ArenaTeamMember));
221 m_members[m_memberCount].PersonalRating = 1500;
222 m_members[m_memberCount++].Info = cachedCharInfo;
223 saveToDB();
224
225#if VERSION_STRING != Classic
226 loggedInPlayer->setArenaTeamId(m_type, m_id);
227 loggedInPlayer->setArenaTeamMemberRank(m_type, 1);
228#endif
229 }
230
231 return true;
232}

234bool ArenaTeam::removeMember(CachedCharacterInfo const* cachedCharInfo)
235{
236 if (!cachedCharInfo)
237 return false;
238
239 for (uint32_t i = 0; i < m_memberCount; ++i)
240 {
241 if (m_members[i].Info == cachedCharInfo)
242 {
243 /* memcpy all the blocks in front of him back (so we only loop O(members) instead of O(slots) */
244 for (uint32_t j = (i + 1); j < m_memberCount; ++j)
245 memcpy(&m_members[j - 1], &m_members[j], sizeof(ArenaTeamMember));
246
247 --m_memberCount;
248 saveToDB();
249
250#if VERSION_STRING != Classic
251 if (Player* loggedInPlayer = sObjectMgr.getPlayer(cachedCharInfo->guid))
252 {
253 loggedInPlayer->setArenaTeamId(m_type, 0);
254 loggedInPlayer->setArenaTeam(m_type, nullptr);
255 }
256#endif
257
258 return true;
259 }
260 }
261
262 return false;

352void ArenaTeam::_allocateSlots(uint16_t type)
353{
354 uint32_t Slots = 0;
355 if (type == ARENA_TEAM_TYPE_2V2)
356 Slots = 4;
357 else if (type == ARENA_TEAM_TYPE_3V3)
358 Slots = 6;
359 else if (type == ARENA_TEAM_TYPE_5V5)
360 Slots = 10;
361
362 if (Slots == 0)
363 {
364 sLogger.failure("Tried to allocate Slot 0 in ArenaTeam::AllocateSlots");
365 return;
366 }
367
368 m_members = std::make_unique<ArenaTeamMember[]>(Slots);
369 std::fill(m_members.get(), m_members.get() + Slots, ArenaTeamMember());
370 m_slots = Slots;
371 m_memberCount = 0;
372}
