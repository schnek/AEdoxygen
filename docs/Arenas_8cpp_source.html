104bool Arena::HandleFinishBattlegroundRewardCalculation(PlayerTeam _winningTeam)
105{
106 // update arena team stats
107 if (rated_match)
108 {
109 m_deltaRating[0] = m_deltaRating[1] = 0;
110 for (uint8_t i = 0; i < 2; ++i)
111 {
112 uint8_t j = i ? 0 : 1; // opposing side
113 bool outcome;
114
115 if (m_teams[i] == NULL || m_teams[j] == NULL)
116 continue;
117
118 outcome = (i == _winningTeam);
119 if (outcome)
120 {
121 m_teams[i]->m_stats.won_season++;
122 m_teams[i]->m_stats.won_week++;
123 }
124
125 m_deltaRating[i] = CalcDeltaRating(m_teams[i]->m_stats.rating, m_teams[j]->m_stats.rating, outcome);
126 m_teams[i]->m_stats.rating += m_deltaRating[i];
127 if (static_cast<int32_t>(m_teams[i]->m_stats.rating) < 0) m_teams[i]->m_stats.rating = 0;
128
129 for (std::set<uint32_t>::iterator itr = m_players2[i].begin(); itr != m_players2[i].end(); ++itr)
130 {
131 const auto info = sObjectMgr.getCachedCharacterInfo(*itr);
132 if (info)
133 {
134 ArenaTeamMember* tp = m_teams[i]->getMember(info);
135
136 if (tp != NULL)
137 {
138 tp->PersonalRating += CalcDeltaRating(tp->PersonalRating, m_teams[j]->m_stats.rating, outcome);
139 if (static_cast<int32_t>(tp->PersonalRating) < 0)
140 tp->PersonalRating = 0;
141
142 if (outcome)
143 {
144 ++(tp->Won_ThisWeek);
145 ++(tp->Won_ThisSeason);
146 }
147 }
148 }
149 }
150
151 m_teams[i]->saveToDB();
152 }
153 }
154
155 sObjectMgr.updateArenaTeamRankings();
156
157 m_nextPvPUpdateTime = 0;
158 updatePvPData();
159 playSoundToAll(_winningTeam ? BattlegroundDef::ALLIANCEWINS : BattlegroundDef::HORDEWINS);
160
161 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_CLOSE);
162 sEventMgr.AddEvent(static_cast<Battleground*>(this), &Battleground::close, EVENT_BATTLEGROUND_CLOSE, 120000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
163
164 for (uint8_t i = 0; i < 2; i++)
165 {
166 bool victorious = (i == _winningTeam);
167 std::set<Player*>::iterator itr = m_players[i].begin();
168 for (; itr != m_players[i].end(); ++itr)
169 {
170 Player* plr = *itr;
171 if (plr != NULL)
172 {
173 sHookInterface.OnArenaFinish(plr, plr->getArenaTeam(m_arenateamtype), victorious, rated_match);
174 plr->resetAllCooldowns();
175 }
176 }
177 }
178
179 /* Prevent honor being given to people in arena */
180 return false;

183void Arena::OnAddPlayer(Player* _player)
184{
185 if (_player == NULL)
186 return;
187
188 _player->m_deathVision = true;
189
190 // remove all buffs (exclude talents, include flasks)
191 for (uint16_t x = AuraSlots::REMOVABLE_SLOT_START; x < AuraSlots::REMOVABLE_SLOT_END; x++)
192 {
193 if (auto* const aur = _player->getAuraWithAuraSlot(x))
194 {
195 if (!aur->getSpellInfo()->getDurationIndex() && aur->getSpellInfo()->getAttributesExC() & ATTRIBUTESEXC_CAN_PERSIST_AND_CASTED_WHILE_DEAD)
196 continue;
197
198 aur->removeAura();
199 }
200 }
201 // On arena start all conjured items are removed
202 _player->getItemInterface()->RemoveAllConjured();
203 // On arena start remove all temp enchants
204 _player->removeTempItemEnchantsOnArena();
205
206 // Before the arena starts all your cooldowns are reset
207 if (!m_hasStarted && _player->IsInWorld())
208 _player->resetAllCooldowns();
209
210 // if (_player->m_isGmInvisible == false)
211 // Make sure the player isn't a GM an isn't invisible (monitoring?)
212 if (!_player->m_isGmInvisible)
213 {
214 if (!m_hasStarted && _player->IsInWorld())
215 _player->castSpell(_player, ARENA_PREPARATION, true);
216
217 m_playersCount[_player->getTeam()]++;
218 UpdatePlayerCounts();
219 }
220 // If they're still queued for the arena, remove them from the queue
221 if (_player->isQueuedForBg())
222 _player->setIsQueuedForBg(false);
223
224 // Add the green/gold team flag
225 auto aura = sSpellMgr.newAura(sSpellMgr.getSpellInfo((_player->getInitialTeam()) ? 35775 - _player->getBgTeam() : 32725 - _player->getBgTeam()), -1, _player, _player, true);
226 _player->addAura(std::move(aura));
227
228 _player->setFfaPvpFlag();
229
230 m_playersAlive.insert(_player->getGuidLow());
231}

383uint32_t Arena::CalcDeltaRating(uint32_t _oldRating, uint32_t _opponentRating, bool _outcome)
384{
385 double power = (int)(_opponentRating - _oldRating) / 400.0f;
386 double divisor = pow(10.0, power);
387 divisor += 1.0;
388
389 double winChance = 1.0 / divisor;
390
391 // New Rating Calculation via Elo
392 // New Rating = Old Rating + K * (_outcome - Expected Win Chance)
393 // _outcome = 1 for a win and 0 for a loss (0.5 for a draw ... but we cant have that)
394 // K is the maximum possible change
395 // Through investigation, K was estimated to be 32 (same as chess)
396 double multiplier = (_outcome ? 1.0 : 0.0) - winChance;
397 return Util::long2int32(32.0 * multiplier);
398}
