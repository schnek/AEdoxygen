52/**
53 \brief Dynamic 1D array tuned for performance.
54
55 Objects must have a default constructor (constructor that
56 takes no arguments) in order to be used with this template.
57 You will get the error "no appropriate default constructor found"
58 if they do not.
59
60 Do not use with objects that overload placement <code>operator new</code>,
61 since the speed of Array is partly due to pooled allocation.
62
63 Array is highly optimized compared to std::vector.
64 Array operations are less expensive than on std::vector and for large
65 amounts of data, Array consumes only 1.5x the total size of the
66 data, while std::vector consumes 2.0x. The default
67 array takes up zero heap space. The first resize (or append)
68 operation grows it to a reasonable internal size so it is efficient
69 to append to small arrays.
70
71 Then Array needs to copy
72 data internally on a resize operation it correctly invokes copy
73 constructors of the elements (the MSVC6 implementation of
74 std::vector uses realloc, which can create memory leaks for classes
75 containing references and pointers). Array provides a guaranteed
76 safe way to access the underlying data as a flat C array --
77 Array::getCArray. Although (T*)std::vector::begin() can be used for
78 this purpose, it is not guaranteed to succeed on all platforms.
79
80 To serialize an array, see G3D::serialize.
81
82 The template parameter MIN_ELEMENTS indicates the smallest number of
83 elements that will be allocated. The default of 10 is designed to avoid
84 the overhead of repeatedly allocating the array as it grows from 1, to 2, and so on.
85 If you are creating a lot of small Arrays, however, you may want to set this smaller
86 to reduce the memory cost. Once the array has been allocated, it will never
87 deallocate the underlying array unless MIN_ELEMENTS is set to 0, MIN_BYTES is 0, and the array
88 is empty.
89
90 Do not subclass an Array.
91
92 \sa G3D::SmallArray
93 */
94template <class T, size_t MIN_ELEMENTS = 10>

146 /**
147 Allocates a new array of size numAllocated (not a parameter to the method)
148 and then copies at most oldNum elements from the old array to it. Destructors are
149 called for oldNum elements of the old array.
150 */
151 void realloc(size_t oldNum) {
152 T* oldData = data;
153
154 // The allocation is separate from the constructor invocation because we don't want
155 // to pay for the cost of constructors until the newly allocated
156 // elements are actually revealed to the application. They
157 // will be constructed in the resize() method.
158
159 data = (T*)m_memoryManager->alloc(sizeof(T) * numAllocated);
160 alwaysAssertM(data, "Memory manager returned NULL: out of memory?");
161
162 // Call the copy constructors
163 {const size_t N = G3D::min(oldNum, numAllocated);
164 const T* end = data + N;
165 T* oldPtr = oldData;
166 for (T* ptr = data; ptr < end; ++ptr, ++oldPtr) {
167
168 // Use placement new to invoke the constructor at the location
169 // that we determined. Use the copy constructor to make the assignment.
170 const T* constructed = new (ptr) T(*oldPtr);
171
172 (void)constructed;
173 debugAssertM(constructed == ptr,
174 "new returned a different address than the one provided by Array.");
175 }}
176
177 // Call destructors on the old array (if there is no destructor, this will compile away)
178 {const T* end = oldData + oldNum;
179 for (T* ptr = oldData; ptr < end; ++ptr) {
180 ptr->~T();
181 }}
182
183 m_memoryManager->free(oldData);
184 }

347 /** Resizes this to match the size of \a other and then copies the data from other using memcpy. This is only safe for POD types */
348 void copyPOD(const Array<T>& other) {
349 static_assert(std::is_standard_layout_v<T> && std::is_trivial_v<T>, "copyPOD called on non-POD type");
350 if (numAllocated < other.num) {
351 m_memoryManager->free(data);
352 data = NULL;
353 if (other.data) {
354 data = (T*)m_memoryManager->alloc(sizeof(T) * other.num);
355 }
356 numAllocated = other.num;
357 }
358
359 num = other.num;
360 if (other.data && (num > 0)) {
361 System::memcpy(data, other.data, sizeof(T) * num);
362 }
363 }

365 /** Resizes this to just barely match the size of \a other + itself and then copies the data to the end of the array from other using memcpy.
366 This is only safe for POD types */
367 void appendPOD(const Array<T>& other) {
368 static_assert(std::is_standard_layout_v<T> && std::is_trivial_v<T>, "appendPOD called on non-POD type");
369 const size_t oldSize = num;
370 num += other.num;
371 if (numAllocated < num) {
372 alwaysAssertM(other.data, "non-zero array with no allocated space");
373 T* old = data;
374 data = (T*)m_memoryManager->alloc(sizeof(T) * num);
375 System::memcpy(data, old, sizeof(T) * oldSize);
376 m_memoryManager->free(old);
377 numAllocated = num;
378 }
379 if (other.data) {
380 System::memcpy((data + oldSize), other.data, sizeof(T) * other.num);
381 }
382 }

475 /** Resize this array to consume exactly the capacity required by its size.
476 \sa clear, resize, capacity, size
477 */
478 void trimToSize() {
479 if (size() != capacity()) {
480 size_t oldNum = numAllocated;
481 numAllocated = size();
482 realloc(oldNum);
483 }
484 }

486 /** @param shrinkIfNecessary if false, memory will never be
487 reallocated when the array shrinks. This makes resizing much
488 faster but can waste memory. Default = true.
489
490 \sa clear, trimToSize
491 */
492 void resize(size_t n, bool shrinkIfNecessary = true) {
493 alwaysAssertM(n < 0xFFFFFFFF, "This implementation does not support arrays with more than 2^32 elements, although the size in memory may be larger.");
494 if (num == n) {
495 return;
496 }
497
498 size_t oldNum = num;
499 num = n;
500
501 // Call the destructors on newly hidden elements if there are any
502 for (size_t i = num; i < oldNum; ++i) {
503 (data + i)->~T();
504 }
505
506 // Once allocated, always maintain MIN_ELEMENTS elements or 32 bytes, whichever is higher.
507 const size_t minSize = G3D::max(MIN_ELEMENTS, (size_t)(MIN_BYTES / sizeof(T)));
508
509 if ((MIN_ELEMENTS == 0) && (MIN_BYTES == 0) && (n == 0) && shrinkIfNecessary) {
510 // Deallocate the array completely
511 numAllocated = 0;
512 m_memoryManager->free(data);
513 data = NULL;
514 return;
515 }
516
517 if (num > numAllocated) {
518 // Grow the underlying array
519
520 if (numAllocated == 0) {
521 // First allocation; grow to exactly the size requested to avoid wasting space.
522 numAllocated = n;
523 debugAssert(oldNum == 0);
524 realloc(oldNum);
525 } else {
526
527 if (num < minSize) {
528 // Grow to at least the minimum size
529 numAllocated = minSize;
530
531 } else {
532
533 // Increase the underlying size of the array. Grow aggressively
534 // up to 64k, less aggressively up to 400k, and then grow relatively
535 // slowly (1.5x per resize) to avoid excessive space consumption.
536 //
537 // These numbers are tweaked according to performance tests.
538
539 double growFactor = 3.0f;
540
541 size_t oldSizeBytes = numAllocated * sizeof(T);
542 if (oldSizeBytes > 10000000) {
543 // Conserve memory more tightly above 10 MB
544 growFactor = 1.2f;
545 } else if (oldSizeBytes > 400000) {
546 // Avoid bloat above 400k
547 growFactor = 1.5f;
548 } else if (oldSizeBytes > 64000) {
549 // This is what std:: uses at all times
550 growFactor = 2.0f;
551 }
552
553 numAllocated = (num - numAllocated) + (size_t)(numAllocated * growFactor);
554
555 if (numAllocated < minSize) {
556 numAllocated = minSize;
557 }
558 }
559
560 realloc(oldNum);
561 }
562
563 } else if ((num <= numAllocated / 3) && shrinkIfNecessary && (num > minSize)) {
564 // Shrink the underlying array
565
566 // Only copy over old elements that still remain after resizing
567 // (destructors were called for others if we're shrinking)
568 realloc(min(num, oldNum));
569
570 }
571
572 // Call the constructors on newly revealed elements.
573 // Do not use parens because we don't want the intializer
574 // invoked for POD types.
575 for (size_t i = oldNum; i < num; ++i) {
576 new (data + i) T;
577 }
578 }

580 /**
581 Add an element to the end of the array. Will not shrink the underlying array
582 under any circumstances. It is safe to append an element that is already
583 in the array.
584 */
585 inline void append(const T& value) {
586
587 if (num < numAllocated) {
588 // This is a simple situation; just stick it in the next free slot using
589 // the copy constructor.
590 new (data + num) T(value);
591 ++num;
592 } else if (inArray(&value)) {
593 // The value was in the original array; resizing
594 // is dangerous because it may move the value
595 // we have a reference to.
596 T tmp = value;
597 append(tmp);
598 } else {
599 // Here we run the empty initializer where we don't have to, but
600 // this simplifies the computation.
601 resize(num + 1, DONT_SHRINK_UNDERLYING_ARRAY);
602 data[num - 1] = value;
603 }
604 }

607 inline void append(const T& v1, const T& v2) {
608 if (inArray(&v1) || inArray(&v2)) {
609 // Copy into temporaries so that the references won't break when
610 // the array resizes.
611 T t1 = v1;
612 T t2 = v2;
613 append(t1, t2);
614 } else if (num + 1 < numAllocated) {
615 // This is a simple situation; just stick it in the next free slot using
616 // the copy constructor.
617 new (data + num) T(v1);
618 new (data + num + 1) T(v2);
619 num += 2;
620 } else {
621 // Resize the array. Note that neither value is already in the array.
622 resize(num + 2, DONT_SHRINK_UNDERLYING_ARRAY);
623 data[num - 2] = v1;
624 data[num - 1] = v2;
625 }
626 }

744 /**
745 Append the elements of array. Cannot be called with this array
746 as an argument.
747 */
748 void append(const Array<T>& array) {

1176 /** The output arrays are resized with fastClear() so that if they are already of the same size
1177 as this array no memory is allocated during partitioning.
1178
1179 @param comparator A function, or class instance with an overloaded operator() that compares
1180 two elements of type <code>T</code> and returns 0 if they are equal, -1 if the second is smaller,
1181 and 1 if the first is smaller (i.e., following the conventions of std::string::compare). For example:
1182
1183 <pre>
1184 int compare(int A, int B) {
1185 if (A < B) {
1186 return 1;
1187 } else if (A == B) {
1188 return 0;
1189 } else {
1190 return -1;
1191 }
1192 }
1193 </pre>
1194 */
1195 template<typename Comparator>
1196 void partition(
1197 const T& partitionElement,
1198 Array<T>& ltArray,
1199 Array<T>& eqArray,
1200 Array<T>& gtArray,
1201 const Comparator& comparator) const {
1202
1203 // Make sure all arrays are independent
1204 debugAssert(&ltArray != this);
1205 debugAssert(&eqArray != this);
1206 debugAssert(&gtArray != this);
1207 debugAssert(&ltArray != &eqArray);
1208 debugAssert(&ltArray != &gtArray);
1209 debugAssert(&eqArray != &gtArray);
1210
1211 // Clear the arrays
1212 ltArray.fastClear();
1213 eqArray.fastClear();
1214 gtArray.fastClear();
1215
1216 // Form a table of buckets for lt, eq, and gt
1217 Array<T>* bucket[3] = {&ltArray, &eqArray, &gtArray};
1218
1219 for (size_t i = 0; i < num; ++i) {
1220 int c = comparator(partitionElement, data[i]);
1221 debugAssertM(c >= -1 && c <= 1, "Comparator returned an illegal value.");
1222
1223 // Insert into the correct bucket, 0, 1, or 2
1224 bucket[c + 1]->append(data[i]);
1225 }
1226 }

1240 /**
1241 Paritions the array into those below the median, those above the median, and those elements
1242 equal to the median in expected O(n) time using quickselect. If the array has an even
1243 number of different elements, the median for partition purposes is the largest value
1244 less than the median.
1245
1246 @param tempArray used for working scratch space
1247 @param comparator see parition() for a discussion.*/
1248 template<typename Comparator>
1249 void medianPartition(
1250 Array<T>& ltMedian,
1251 Array<T>& eqMedian,
1252 Array<T>& gtMedian,
1253 Array<T>& tempArray,
1254 const Comparator& comparator) const {
1255
1256 ltMedian.fastClear();
1257 eqMedian.fastClear();
1258 gtMedian.fastClear();
1259
1260 // Handle trivial cases first
1261 switch (size()) {
1262 case 0:
1263 // Array is empty; no parition is possible
1264 return;
1265
1266 case 1:
1267 // One element
1268 eqMedian.append(first());
1269 return;
1270
1271 case 2:
1272 {
1273 // Two element array; median is the smaller
1274 int c = comparator(first(), last());
1275
1276 switch (c) {
1277 case -1:
1278 // first was bigger
1279 eqMedian.append(last());
1280 gtMedian.append(first());
1281 break;
1282
1283 case 0:
1284 // Both equal to the median
1285 eqMedian.append(first(), last());
1286 break;
1287
1288 case 1:
1289 // Last was bigger
1290 eqMedian.append(first());
1291 gtMedian.append(last());
1292 break;
1293 }
1294 }
1295 return;
1296 }
1297
1298 // All other cases use a recursive randomized median
1299
1300 // Number of values less than all in the current arrays
1301 int ltBoost = 0;
1302
1303 // Number of values greater than all in the current arrays
1304 int gtBoost = 0;
1305
1306 // For even length arrays, force the gt array to be one larger than the
1307 // lt array:
1308 // [1 2 3] size = 3, choose half = (s + 1) /2
1309 //
1310 int lowerHalfSize, upperHalfSize;
1311 if (isEven(size())) {
1312 lowerHalfSize = size() / 2;
1313 upperHalfSize = lowerHalfSize + 1;
1314 } else {
1315 lowerHalfSize = upperHalfSize = (size() + 1) / 2;
1316 }
1317 const T* xPtr = NULL;
1318
1319 // Maintain pointers to the arrays; we'll switch these around during sorting
1320 // to avoid copies.
1321 const Array<T>* source = this;
1322 Array<T>* lt = &ltMedian;
1323 Array<T>* eq = &eqMedian;
1324 Array<T>* gt = &gtMedian;
1325 Array<T>* extra = &tempArray;
1326
1327 while (true) {
1328 // Choose a random element -- choose the middle element; this is theoretically
1329 // suboptimal, but for loosly sorted array is actually the best strategy
1330
1331 xPtr = &(source->middle());
1332 if (source->size() == 1) {
1333 // Done; there's only one element left
1334 break;
1335 }
1336 const T& x = *xPtr;
1337
1338 // Note: partition (fast) clears the arrays for us
1339 source->partition(x, *lt, *eq, *gt, comparator);
1340
1341 int L = lt->size() + ltBoost + eq->size();
1342 int U = gt->size() + gtBoost + eq->size();
1343 if ((L >= lowerHalfSize) &&
1344 (U >= upperHalfSize)) {
1345
1346 // x must be the partition median
1347 break;
1348
1349 } else if (L < lowerHalfSize) {
1350
1351 // x must be smaller than the median. Recurse into the 'gt' array.
1352 ltBoost += lt->size() + eq->size();
1353
1354 // The new gt array will be the old source array, unless
1355 // that was the this pointer (i.e., unless we are on the
1356 // first iteration)
1357 Array<T>* newGt = (source == this) ? extra : const_cast<Array<T>*>(source);
1358
1359 // Now set up the gt array as the new source
1360 source = gt;
1361 gt = newGt;
1362
1363 } else {
1364
1365 // x must be bigger than the median. Recurse into the 'lt' array.
1366 gtBoost += gt->size() + eq->size();
1367
1368 // The new lt array will be the old source array, unless
1369 // that was the this pointer (i.e., unless we are on the
1370 // first iteration)
1371 Array<T>* newLt = (source == this) ? extra : const_cast<Array<T>*>(source);
1372
1373 // Now set up the lt array as the new source
1374 source = lt;
1375 lt = newLt;
1376 }
1377 }
1378
1379 // Now that we know the median, make a copy of it (since we're about to destroy the array that it
1380 // points into).
1381 T median = *xPtr;
1382 xPtr = NULL;
1383
1384 // Partition the original array (note that this fast clears for us)
1385 partition(median, ltMedian, eqMedian, gtMedian, comparator);
1386 }
