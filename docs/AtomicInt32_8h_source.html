73# if defined(G3D_WINDOWS)

104# if defined(G3D_WINDOWS)

140 /** Atomic test-and-set: if <code>*this == comperand</code> then <code>*this := exchange</code> else do nothing.
141 In both cases, returns the old value of <code>*this</code>.
142
143 Performs an atomic comparison of this with the Comperand value.
144 If this is equal to the Comperand value, the Exchange value is stored in this.
145 Otherwise, no operation is performed.
146
147 Under VC6 the sign bit may be lost.
148 */
149 int32 compareAndSet(const int32 comperand, const int32 exchange) {
150# if defined(G3D_WINDOWS)
151 return InterlockedCompareExchange(&m_value, exchange, comperand);
152# elif defined(G3D_LINUX) || defined(G3D_FREEBSD) || defined(G3D_OSX)
153# if defined(__aarch64__)
154 return __sync_val_compare_and_swap(&m_value, comperand, exchange);
155# else
156 // Based on Apache Portable Runtime
157 // http://koders.com/c/fid3B6631EE94542CDBAA03E822CA780CBA1B024822.aspx
158 int32 ret;
159 asm volatile ("lock; cmpxchgl %1, %2"
160 : "=a" (ret)
161 : "r" (exchange), "m" (m_value), "0"(comperand)
162 : "memory", "cc");
163 return ret;
164
165 // Note that OSAtomicCompareAndSwap32 does not return a useful value for us
166 // so it can't satisfy the cmpxchgl contract.
167# endif
168# endif
169 }
