131void AuctionHouse::loadAuctionsFromDB()
132{
133 auto result = CharacterDatabase.Query("SELECT * FROM auctions WHERE auctionhouse =%u", getId());
134 if (!result)
135 return;
136
137 do
138 {
139 Field* fields = result->Fetch();
140 const auto auctionId = fields[0].asUint32();
141
142 auto pItem = sObjectMgr.loadItem(fields[2].asUint32());
143 if (!pItem)
144 {
145 CharacterDatabase.Execute("DELETE FROM auctions WHERE auctionId=%u", auctionId);
146 continue;
147 }
148
149 auctions.try_emplace(auctionId, std::make_unique<Auction>(fields, std::move(pItem)));
150 }
151 while (result->NextRow());
152}

154void AuctionHouse::updateAuctions()
155{
156 std::lock_guard guard(auctionLock);
157 std::lock_guard lock(removalLock);
158
159 auto const time = static_cast<uint32_t>(UNIXTIME);
160 for (auto itr = auctions.begin(); itr != auctions.end();)
161 {
162 const auto& auction = itr->second;
163 ++itr;
164
165 if (auction->isRemoved)
166 continue;
167
168 if (time >= auction->expireTime)
169 {
170 if (auction->highestBidderGuid.getGuidLow() == 0)
171 {
172 auction->removedType = AUCTION_REMOVE_EXPIRED;
173 this->sendAuctionExpiredNotificationPacket(auction.get());
174 }
175 else
176 {
177 auction->removedType = AUCTION_REMOVE_WON;
178 }
179
180 auction->isRemoved = true;
181 removalList.push_back(auction.get());
182 }
183 }

198void AuctionHouse::removeAuction(Auction* auction)
199{
200 sLogger.debug("{}: Removing auction {}, reason {}.", auctionHouseEntryDbc->id, auction->Id, auction->removedType);
201
202 char subject[100];
203 char body[200];
204 switch (auction->removedType)
205 {
206 case AUCTION_REMOVE_EXPIRED:
207 {
208 // ItemEntry:0:3
209 snprintf(subject, 100, "%u:0:3", auction->auctionItem->getEntry());
210
211 // Auction expired, resend item, no money to owner.
212 sMailSystem.SendAutomatedMessage(MAIL_TYPE_AUCTION, auctionHouseEntryDbc->id, auction->ownerGuid, subject, "", 0, 0, auction->auctionItem->getGuid(), MAIL_STATIONERY_AUCTION, MAIL_CHECK_MASK_COPIED);
213 }
214 break;
215 case AUCTION_REMOVE_WON:
216 {
217 // ItemEntry:0:1
218 snprintf(subject, 100, "%u:0:1", auction->auctionItem->getEntry());
219
220 // <owner player guid>:bid:buyout
221 snprintf(body, 200, "%X:%s:%s", auction->ownerGuid.getGuidLow(), std::to_string(auction->highestBid).c_str(), std::to_string(auction->buyoutPrice).c_str());
222
223 // Auction won by highest bidder. He gets the item.
224 sMailSystem.SendAutomatedMessage(MAIL_TYPE_AUCTION, auctionHouseEntryDbc->id, auction->highestBidderGuid, subject, body, 0, 0, auction->auctionItem->getGuid(), MAIL_STATIONERY_AUCTION, MAIL_CHECK_MASK_COPIED);
225
226 // Send a mail to the owner with his cut of the price.
227 const auto auction_cut = Util::float2int32(cutPercent * static_cast<float_t>(auction->highestBid));
228 auto amount = auction->highestBid - auction_cut + auction->depositAmount;
229
230 // ItemEntry:0:2
231 snprintf(subject, 100, "%u:0:2", auction->auctionItem->getEntry());
232
233 // <hex player guid>:bid:0:deposit:cut
234 if (auction->highestBid == auction->buyoutPrice) // Buyout
235 snprintf(body, 200, "%X:%s:%s:%u:%u", auction->highestBidderGuid.getGuidLow(), std::to_string(auction->highestBid).c_str(), std::to_string(auction->buyoutPrice).c_str(), auction->depositAmount, (unsigned int)auction_cut);
236 else
237 snprintf(body, 200, "%X:%s:0:%u:%u", auction->highestBidderGuid.getGuidLow(), std::to_string(auction->highestBid).c_str(), auction->depositAmount, (unsigned int)auction_cut);
238
239 // send message away.
240 sMailSystem.SendAutomatedMessage(MAIL_TYPE_AUCTION, auctionHouseEntryDbc->id, auction->ownerGuid, subject, body, static_cast<uint32_t>(amount), 0, 0, MAIL_STATIONERY_AUCTION, MAIL_CHECK_MASK_COPIED);
241
242 // If it's not a buyout (otherwise the players has been already notified)
243 if (auction->highestBid < auction->buyoutPrice || auction->buyoutPrice == 0)
244 {
245 this->sendAuctionBuyOutNotificationPacket(auction);
246 }
247 }
248 break;
249 case AUCTION_REMOVE_CANCELLED:
250 {
251 if (auction->auctionItem)
252 {
253 snprintf(subject, 100, "%u:0:5", auction->auctionItem->getEntry());
254 const auto cut = Util::float2int32(cutPercent * static_cast<float_t>(auction->highestBid));
255 Player* plr = sObjectMgr.getPlayer(auction->ownerGuid.getGuidLow());
256 if (cut && plr && plr->hasEnoughCoinage(static_cast<uint32_t>(cut)))
257 plr->modCoinage(-cut);
258
259 sMailSystem.SendAutomatedMessage(MAIL_TYPE_AUCTION, getId(), auction->ownerGuid, subject, "", 0, 0, auction->auctionItem->getGuid(), MAIL_STATIONERY_AUCTION, MAIL_CHECK_MASK_COPIED);
260 }
261 // return bidders money
262 if (auction->highestBidderGuid)
263 {
264 sMailSystem.SendAutomatedMessage(MAIL_TYPE_AUCTION, getId(), auction->highestBidderGuid, subject, "", static_cast<uint32_t>(auction->highestBid), 0, 0, MAIL_STATIONERY_AUCTION, MAIL_CHECK_MASK_COPIED);
265 }
266
267 }
268 break;
269 }
270
271 // Destroy the item from memory (it still remains in the db)
272 auction->auctionItem = nullptr;
273
274 // Finally destroy the auction instance.
275 auction->deleteFromDB();
276
277 // Remove the auction from the hashmap.
278 auctionLock.lock();
279 auctions.erase(auction->Id);
280 auctionLock.unlock();
281}

427void AuctionHouse::sendAuctionList(Player* player, AscEmu::Packets::CmsgAuctionListItems srlPacket)
428{
429 std::vector<AuctionPacketList> auctionPacketList{};
430
431 uint32_t count = 0;
432 uint32_t totalcount = 0;
433
434 // convert auction string to lowercase for faster parsing.
435 if (srlPacket.searchedName.length() > 0)
436 {
437 for (uint32_t j = 0; j < srlPacket.searchedName.length(); ++j)
438 srlPacket.searchedName[j] = static_cast<char>(tolower(srlPacket.searchedName[j]));
439 }
440
441 std::lock_guard<std::mutex> guard(auctionLock);
442
443 for (auto& auction : auctions)
444 {
445 if (auction.second->isRemoved)
446 continue;
447
448 ItemProperties const* proto = auction.second->auctionItem->getItemProperties();
449
450 // Check the auction for parameters
451
452 // inventory type
453 if (srlPacket.auctionSlotId != 0xffffffff && srlPacket.auctionSlotId != proto->InventoryType)
454 continue;
455
456 // class
457 if (srlPacket.auctionMainCategory != 0xffffffff && srlPacket.auctionMainCategory != proto->Class)
458 continue;
459
460 // subclass
461 if (srlPacket.auctionSubCategory != 0xffffffff && srlPacket.auctionSubCategory != proto->SubClass)
462 continue;
463
464 // this is going to hurt. - name
465 std::string proto_lower = proto->lowercase_name;
466 if (srlPacket.searchedName.length() > 0 && !AscEmu::Util::Strings::contains(srlPacket.searchedName, proto_lower))
467 continue;
468
469 // rarity
470 if (srlPacket.quality != 0xffffffff && srlPacket.quality > proto->Quality)
471 continue;
472
473 // level range check - lower boundary
474 if (srlPacket.levelMin && proto->RequiredLevel < srlPacket.levelMin)
475 continue;
476
477 // level range check - high boundary
478 if (srlPacket.levelMax && proto->RequiredLevel > srlPacket.levelMax)
479 continue;
480
481 // usable check - this will hurt too :(
482 if (srlPacket.usable)
483 {
484 // allowed class
485 if (proto->AllowableClass && !(player->getClassMask() & proto->AllowableClass))
486 continue;
487
488 if (proto->RequiredLevel && proto->RequiredLevel > player->getLevel())
489 continue;
490
491 if (proto->AllowableRace && !(player->getRaceMask() & proto->AllowableRace))
492 continue;
493
494 if (proto->Class == 4 && proto->SubClass && !(player->getArmorProficiency() & (((uint32_t)(1)) << proto->SubClass)))
495 continue;
496
497 if (proto->Class == 2 && proto->SubClass && !(player->getWeaponProficiency() & (((uint32_t)(1)) << proto->SubClass)))
498 continue;
499
500 if (proto->RequiredSkill && (!player->hasSkillLine(proto->RequiredSkill) || proto->RequiredSkillRank > player->getSkillLineCurrent(proto->RequiredSkill, true)))
501 continue;
502 }
503
504 if (count < 50 && totalcount >= srlPacket.listFrom)
505 {
506 ++count;
507
508 auctionPacketList.push_back(auction.second->getListMember());
509 }
510
511 ++totalcount;
512 }
513
514 player->sendPacket(SmsgAuctionListResult(static_cast<uint32_t>(auctionPacketList.size()), auctionPacketList, totalcount, 300).serialise().get());
515}
