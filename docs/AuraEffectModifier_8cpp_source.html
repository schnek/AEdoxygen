58void AuraEffectModifier::applyEffect(bool apply, bool skipScriptCheck/* = false*/)
59{
60 // Do not apply or remove effect multiple times
61 if (mActive == apply)
62 return;
63
64 mActive = apply;
65
66 if (getAura())
67 {
68 if (skipScriptCheck)
69 {
70 (*getAura().*SpellAuraHandler[getAuraEffectType()])(this, apply);
71 }
72 else
73 {
74 const auto scriptResult = sScriptMgr.callScriptedAuraBeforeAuraEffect(getAura(), this, apply);
75 if (scriptResult != SpellScriptExecuteState::EXECUTE_PREVENT)
76 (*getAura().*SpellAuraHandler[getAuraEffectType()])(this, apply);
77 }
78 }
79 else
80 sLogger.failure("AuraEffectModifier::applyEffect fatal Error invalid Aura!");
81}
