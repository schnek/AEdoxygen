253 // SRP6 //////////////////////////////////////////////////////////////////////////////////////////////////////
254 // Challenge
255 //
256 // First we will generate the Verifier value using the following formulas
257 //
258 // x = SHA1(s | SHA1(I | ":" | P))
259 // v = g^x % N
260 //
261 // The SHA1(I | ":" | P) part for x we have in the account database, this is the encrypted password, reversed
262 // N is a safe prime
263 // g is the generator
264 // | means concatenation in this contect
265 //
266 //

268 Sha1Hash sha;
269 sha.updateData(s.AsByteArray(), 32);
270 sha.updateData(m_account->SrpHash, 20);
271 sha.finalize();
272
273 BigNumber x;
274 x.SetBinary(sha.getDigest(), SHA_DIGEST_LENGTH);

276
277 // Next we generate b, and B which are the public and private values of the server
278 //
279 // b = random()
280 // B = k*v + g^b % N
281 //
282 // in our case the multiplier parameters, k = 3
283
284 b.SetRand(152);
285 uint8_t k = 3;
286
287 BigNumber gmod = g.ModExp(b, N);
288 B = ((v * k) + gmod) % N;
289 ASSERT(gmod.GetNumBytes() <= 32);
290
291 BigNumber unk;
292 unk.SetRand(128);
293
294 // Now we send B, g, N and s to the client as a challenge, asking the client for the proof
295 sAuthLogonChallenge_S challenge;
296 challenge.cmd = 0;
297 challenge.error = 0;
298 challenge.unk2 = CE_SUCCESS;
299 memcpy(challenge.B, B.AsByteArray(), 32);
300 challenge.g_len = 1;
301 challenge.g = (g.AsByteArray())[0];
302 challenge.N_len = 32;
303 memcpy(challenge.N, N.AsByteArray(), 32);

305 memcpy(challenge.unk3, unk.AsByteArray(), 16);
306 challenge.unk4 = 0;

311void AuthSocket::HandleProof()
312{
313 if (readBuffer.GetSize() < sizeof(sAuthLogonProof_C))
314 {
315 sLogger.failure("[AuthLogonProof] The packet received is larger than expected, refusing to handle it!");
316 return;
317 }
318
319 // patch
320 if (m_patch && !m_account)
321 {
322 //RemoveReadBufferBytes(75,false);
323 readBuffer.Remove(75);
324 sLogger.debug("[AuthLogonProof] Intitiating PatchJob");
325 uint8_t bytes[2] = { 0x01, 0x0a };
326 Send(bytes, 2);
327 PatchMgr::getInstance().InitiatePatch(m_patch, this);
328 return;
329 }
330
331 if (!m_account)
332 return;
333
334 sLogger.debug("[AuthLogonProof] Interleaving and checking proof...");
335
336 sAuthLogonProof_C lp;
337 //Read(sizeof(sAuthLogonProof_C), (uint8_t*)&lp);
338 readBuffer.Read(&lp, sizeof(sAuthLogonProof_C));
339
340 // SRP6 //////////////////////////////////////////////////////////////////////////////////////////////////////
341 //
342 // Now comes the famous secret Xi Chi fraternity handshake ( http://www.youtube.com/watch?v=jJSYBoI2si0 ),
343 // generating a session key
344 //
345 // A = g^a % N
346 // u = SHA1( A | B )
347 //
348 //
349
350 BigNumber A;
351 A.SetBinary(lp.A, 32);
352
353 Sha1Hash sha;
354 sha.updateBigNumbers(&A, &B, 0);
355 sha.finalize();
356
357 BigNumber u;
358 u.SetBinary(sha.getDigest(), 20);
359
360 // S session key key, S = ( A * v^u ) ^ b
361 BigNumber S = (A * (v.ModExp(u, N))).ModExp(b, N);
362
363 // Generate M
364 // M = H(H(N) xor H(g), H(I), s, A, B, K) according to http://srp.stanford.edu/design.html
365 uint8_t t[32];
366 uint8_t t1[16];
367 uint8_t vK[40];
368 memcpy(t, S.AsByteArray(), 32);
369 for (int i = 0; i < 16; i++)
370 {
371 t1[i] = t[i * 2];
372 }
373 sha.initialize();
374 sha.updateData(t1, 16);
375 sha.finalize();
376 for (int i = 0; i < 20; i++)
377 {
378 vK[i * 2] = sha.getDigest()[i];
379 }
380 for (int i = 0; i < 16; i++)
381 {
382 t1[i] = t[i * 2 + 1];
383 }
384 sha.initialize();
385 sha.updateData(t1, 16);
386 sha.finalize();
387 for (int i = 0; i < 20; i++)
388 {
389 vK[i * 2 + 1] = sha.getDigest()[i];
390 }
391 m_sessionkey.SetBinary(vK, 40);
392
393 uint8_t hash[20];
394
395 sha.initialize();
396 sha.updateBigNumbers(&N, NULL);
397 sha.finalize();
398 memcpy(hash, sha.getDigest(), 20);
399 sha.initialize();
400 sha.updateBigNumbers(&g, NULL);
401 sha.finalize();
402 for (int i = 0; i < 20; i++)
403 {
404 hash[i] ^= sha.getDigest()[i];
405 }
406 BigNumber t3;
407 t3.SetBinary(hash, 20);
408
409 sha.initialize();
410 sha.updateData((const uint8_t*)m_account->UsernamePtr->c_str(), (int)m_account->UsernamePtr->size());
411 sha.finalize();
412
413 BigNumber t4;
414 t4.SetBinary(sha.getDigest(), 20);
415
416 sha.initialize();
417 sha.updateBigNumbers(&t3, &t4, &s, &A, &B, &m_sessionkey, NULL);
418 sha.finalize();
419
420 BigNumber M;
421 M.SetBinary(sha.getDigest(), 20);
422
423 // Compare the M value the client sent us to the one we generated, this proves we both have the same values
424 // which proves we have the same username-password pairs
425 if (memcmp(lp.M1, M.AsByteArray(), 20) != 0)
426 {
427 // Authentication failed.
428 //SendProofError(4, 0);
429 SendChallengeError(CE_NO_ACCOUNT);
430 sLogger.debug("[AuthLogonProof] M values don't match. ( Either invalid password or the logon server is bugged. )");
431 return;
432 }
433
434 // Store sessionkey
435 m_account->SetSessionKey(m_sessionkey.AsByteArray());
436
437 // let the client know
438 sha.initialize();
439 sha.updateBigNumbers(&A, &M, &m_sessionkey, 0);
440 sha.finalize();
441
442 //SendProofError(0, sha.GetDigest());
443 sendAuthProof(sha);
444 sLogger.debug("[AuthLogonProof] Authentication Success.");
445
446 // we're authenticated now :)
447 m_authenticated = true;
448
449 // Don't update when IP banned, but update anyway if it's an account ban
450 sLogonSQL->Execute("UPDATE accounts SET lastlogin=NOW(), lastip='%s' WHERE id = %u;", GetRemoteIP().c_str(), m_account->AccountId);
