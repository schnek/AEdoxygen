33void PatchMgr::initialize()
34{
35 // Load patches
36 sLogger.info("PatchMgr : Loading Patches...");
37
38 const std::string patchDirectory = "./ClientPatches";
39
40 if (!fs::exists(patchDirectory) || !fs::is_directory(patchDirectory))
41 {
42 sLogger.warning("No patches found.");
43 return;
44 }
45
46 for (const auto& entry : fs::directory_iterator(patchDirectory))
47 {
48 if (!entry.is_regular_file())
49 continue;
50
51 std::string filePath = entry.path().string();
52 std::string fileName = entry.path().filename().string();
53 uint32_t srcversion;
54 char locality[5] = { 0 };
55
56 if (sscanf(fileName.c_str(), "%4s%u.", locality, &srcversion) != 2)
57 continue;
58
59 std::ifstream file(filePath, std::ios::binary | std::ios::ate);
60 if (!file.is_open())
61 {
62 sLogger.failure("Cannot open {}", filePath);
63 continue;
64 }
65
66 std::streamsize size = file.tellg();
67 file.seekg(0, std::ios::beg);
68
69 auto pPatch = std::make_unique<Patch>();
70 pPatch->FileSize = static_cast<uint32_t>(size);
71 pPatch->Data = std::make_unique<uint8_t[]>(pPatch->FileSize);
72 pPatch->Version = srcversion;
73 for (uint32_t i = 0; i < 4; ++i)
74 {
75 pPatch->Locality[i] = static_cast<char>(std::tolower(locality[i]));
76 }
77
78 pPatch->Locality[4] = '\0';
79 pPatch->uLocality = *reinterpret_cast<uint32_t*>(pPatch->Locality);
80
81 if (!file.read(reinterpret_cast<char*>(pPatch->Data.get()), pPatch->FileSize))
82 {
83 sLogger.failure("Cannot read {}", filePath);
84 continue;
85 }
86
87 file.close();
88
89 sLogger.info("PatchMgr : Found patch for {} locale `{}` ({} bytes).", srcversion, locality, pPatch->FileSize);
90
91 // md5hash the file
92 MD5Hash md5;
93 md5.initialize();
94 md5.updateData(pPatch->Data.get(), pPatch->FileSize);
95 md5.finalize();
96 std::memcpy(pPatch->MD5, md5.getDigest(), MD5_DIGEST_LENGTH);
97
98 // add the patch to the patchlist
99 m_patches.push_back(std::move(pPatch));
100 }
101}

103Patch* PatchMgr::FindPatchForClient(uint32_t Version, const char* Locality)
104{
105 char tmplocality[5];
106 Patch* fallbackPatch = nullptr;
107
108 for (uint32_t i = 0; i < 4; ++i)
109 tmplocality[i] = static_cast<char>(tolower(Locality[i]));
110
111 tmplocality[4] = 0;
112 uint32_t ulocality = *(uint32_t*)tmplocality;
113
114 for (auto itr = m_patches.begin(); itr != m_patches.end(); ++itr)
115 {
116 // since localities are always 4 bytes we can do a simple int compare,
117 // saving a string compare ;)
118 if ((*itr)->uLocality == ulocality)
119 {
120 if (fallbackPatch == nullptr && (*itr)->Version == 0)
121 fallbackPatch = (*itr).get();
122
123 if ((*itr)->Version == Version)
124 return (*itr).get();
125 }
126 }
127
128 return fallbackPatch;
129}

140void PatchMgr::UpdateJobs()
141{
142 std::lock_guard lock(m_patchJobLock);
143
144 for (auto itr = m_patchJobs.begin(); itr != m_patchJobs.end();)
145 {
146 std::list<std::unique_ptr<PatchJob>>::iterator itr2 = itr;
147 ++itr;
148
149 if (!(*itr2)->Update())
150 {
151 (*itr2)->GetClient()->m_patchJob = nullptr;
152 m_patchJobs.erase(itr2);
153 }
154 }
155}

197bool PatchJob::Update()
198{
199 // don't update unless the write buffer is empty
200 m_client->BurstBegin();
201 if (m_client->writeBuffer.GetSize() != 0)
202 {
203 m_client->BurstEnd();
204 return true;
205 }
206
207 // send 1500 byte chunks
208 TransferDataPacket header;
209 header.cmd = 0x31;
210 header.chunk_size = static_cast<uint16_t>((m_bytesLeft > 1500) ? 1500 : m_bytesLeft);
211 //LogDebug("PatchJob : Sending %u byte chunk", header.chunk_size);
212
213 bool result = m_client->BurstSend((const uint8_t*)&header, sizeof(TransferDataPacket));
214 if (result)
215 {
216 result = m_client->BurstSend(m_dataPointer, header.chunk_size);
217 if (result)
218 {
219 m_dataPointer += header.chunk_size;
220 m_bytesSent += header.chunk_size;
221 m_bytesLeft -= header.chunk_size;
222 }
223 }
224
225 if (result)
226 m_client->BurstPush();
227
228 m_client->BurstEnd();
229
230 // no need to check the result here, could just be a full buffer and not necessarily a fatal error.
231 return (m_bytesLeft > 0) ? true : false;
232}

234bool PatchMgr::InitiatePatch(Patch* pPatch, AuthSocket* pClient)
235{
236 // send initiate packet
237 TransferInitiatePacket init;
238 init.cmd = 0x30;
239 init.strsize = 5;
240 init.name[0] = 'P';
241 init.name[1] = 'a';
242 init.name[2] = 't';
243 init.name[3] = 'c';
244 init.name[4] = 'h';
245 init.name[5] = '\0';
246 init.filesize = pPatch->FileSize;
247 memcpy(init.md5hash, pPatch->MD5, MD5_DIGEST_LENGTH);
248
249 // send it to the client
250 pClient->BurstBegin();
251 bool result = pClient->BurstSend((const uint8_t*)&init, sizeof(TransferInitiatePacket));
252 if (result)
253 pClient->BurstPush();
254
255 pClient->BurstEnd();
256 return result;
257}
