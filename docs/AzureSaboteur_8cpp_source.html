38void AzureSaboteurAI::AIUpdate(unsigned long time_passed)
39{
40 if (!mInstance || mInstance->getLocalData(DATA_MAIN_EVENT_STATE) == EncounterStates::NotStarted)
41 {
42 scriptEvents.resetEvents();
43 return;
44 }
45
46 scriptEvents.updateEvents(time_passed, getScriptPhase());
47
48 while (uint32_t eventId = scriptEvents.getFinishedEvent())
49 {
50 switch (eventId)
51 {
52 case 1:
53 {
54 startMovement();
55 } break;
56 case 2:
57 {
58 if (castCounter < 2)
59 {
60 ++castCounter;
61 getCreature()->castSpell(getCreature(), SPELL_SHIELD_DISRUPTION, false);
62 scriptEvents.addEvent(2, 1 * TimeVarsMs::Second);
63 }
64 else
65 {
66 scriptEvents.addEvent(3, 2 * TimeVarsMs::Second);
67 }
68 } break;
69 case 3:
70 {
71 mInstance->setLocalData(DATA_START_BOSS_ENCOUNTER, 1);
72 getCreature()->castSpell(getCreature(), SPELL_TELEPORT_VISUAL, false);
73 scriptEvents.addEvent(4, 1 * TimeVarsMs::Second);
74 } break;
75 case 4:
76 {
77 getCreature()->ToSummon()->unSummon();
78 } break;
79 default:
80 break;
81 }
82 }
83}
