57//.battleground menu
58bool ChatCommandHandler::HandleBGMenuCommand(const char* args, WorldSession* m_session)
59{
60 if (!*args)
61 return false;
62
63 uint32_t type = atoi(args);
64 if (type >= BATTLEGROUND_NUM_TYPES)
65 return false;
66
67 Player* selected_player = GetSelectedPlayer(m_session, true, true);
68 if (selected_player == nullptr)
69 return true;
70
71#if VERSION_STRING <= WotLK
72 sBattlegroundManager.handleBattlegroundListPacket(selected_player->getSession(), type);
73#else
74 WoWGuid guid;
75 guid.Init(uint64_t(0));
76 sBattlegroundManager.handleBattlegroundListPacket(guid, selected_player->getSession(), type);
77#endif
78
79 return true;
80}

165 redSystemMessage(m_session, "You are not in a battleground.");

171 m_session->LocalizedWorldSrv(m_session->GetPlayer()->getBattleground()->GetNameID()));
172
173 sEventMgr.RemoveEvents(m_session->GetPlayer()->getBattleground(), EVENT_BATTLEGROUND_COUNTDOWN);

175 m_session->GetPlayer()->getBattleground()->startBattleground();
176
177 return true;
