142void BaseMap::loadSpawns(bool reload)
143{
144 if (reload) // perform cleanup
145 {
146 for (uint32_t x = 0; x < Map::Cell::_sizeX; x++)
147 {
148 for (uint32_t y = 0; y < Map::Cell::_sizeY; y++)
149 {
150 if ((*spawns[x])[y])
151 {
152 CellSpawns* sp = (*spawns[x])[y].get();
153 for (CreatureSpawnList::iterator i = sp->CreatureSpawns.begin(); i != sp->CreatureSpawns.end(); ++i)
154 delete(*i);
155 for (GameobjectSpawnList::iterator it = sp->GameobjectSpawns.begin(); it != sp->GameobjectSpawns.end(); ++it)
156 delete(*it);
157
158 (*spawns[x])[y] = nullptr;
159 }
160 }
161 }
162 }
163
164 CreatureSpawnCount = 0;
165 for (const auto& cspawn : sMySQLStore._creatureSpawnsStore[this->_mapId])
166 {
167 if (!sMySQLStore.isTransportMap(this->_mapId))
168 {
169 uint32_t cellx = CellHandler<MapMgr>::getPosX(cspawn->x);
170 uint32_t celly = CellHandler<MapMgr>::getPosY(cspawn->y);
171 if (spawns[cellx] == nullptr)
172 {
173 spawns[cellx] = std::make_unique<std::array<std::unique_ptr<CellSpawns>, Map::Cell::_sizeY>>();
174 std::fill(spawns[cellx]->begin(), spawns[cellx]->end(), nullptr);
175 }
176
177 if ((*spawns[cellx])[celly] == nullptr)
178 (*spawns[cellx])[celly] = std::make_unique<CellSpawns>();
179
180 (*spawns[cellx])[celly]->CreatureSpawns.push_back(cspawn);
181 ++CreatureSpawnCount;
182 }
183 }
184
185 GameObjectSpawnCount = 0;
186 for (const auto& go_spawn : sMySQLStore._gameobjectSpawnsStore[this->_mapId])
187 {
188 GameObjectOverrides m_overrides = GAMEOBJECT_NORMAL_DISTANCE;
189 if (GameObjectProperties const* gameobject_properties = sMySQLStore.getGameObjectProperties(go_spawn->entry))
190 {
191
192 // Check if GameObject is Large
193 if (gameobject_properties->isLargeGameObject())
194 m_overrides = GAMEOBJECT_AREAWIDE; // not implemented yet
195
196 // Check if GameObject is Infinite
197 if (gameobject_properties->isInfiniteGameObject())
198 m_overrides = GAMEOBJECT_MAPWIDE;
199 }
200
201 switch (m_overrides)
202 {
203 case GAMEOBJECT_NORMAL_DISTANCE:
204 {
205 uint32_t cellx = CellHandler<MapMgr>::getPosX(go_spawn->spawnPoint.x);
206 uint32_t celly = CellHandler<MapMgr>::getPosY(go_spawn->spawnPoint.y);
207 if (spawns[cellx] == nullptr)
208 {
209 spawns[cellx] = std::make_unique<std::array<std::unique_ptr<CellSpawns>, Map::Cell::_sizeY>>();
210 std::fill(spawns[cellx]->begin(), spawns[cellx]->end(), nullptr);
211 }
212
213 if ((*spawns[cellx])[celly] == nullptr)
214 (*spawns[cellx])[celly] = std::make_unique<CellSpawns>();
215
216 (*spawns[cellx])[celly]->GameobjectSpawns.push_back(go_spawn);
217
218 ++GameObjectSpawnCount;
219 } break;
220 case GAMEOBJECT_MAPWIDE:
221 {
222 mapWideSpawns.GameobjectSpawns.push_back(go_spawn);
223 ++GameObjectSpawnCount;
224 } break;
225 case GAMEOBJECT_AREAWIDE:
226 {
227 areaWideSpawns.GameobjectSpawns.push_back(go_spawn);
228 ++GameObjectSpawnCount;
229 } break;
230 }
231 }
232 sLogger.info("MapMgr : {} creatures / {} gobjects on map {} cached.", CreatureSpawnCount, GameObjectSpawnCount, _mapId);
233}
