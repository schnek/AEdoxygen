117void Battleground::buildPvPUpdateDataPacket(WorldPacket* data)
118{
119 data->Initialize(MSG_PVP_LOG_DATA);
120 data->reserve(10 * (m_players[0].size() + m_players[1].size()) + 50);
121
122 BGScore* bs;
123 if (isTypeArena(m_type))
124 {
125 if (!m_hasEnded)
126 return;
127
128 *data << uint8_t(1);
129
130 if (!Rated())
131 {
132 *data << uint32_t(0); //uint32_t(negative rating)
133 *data << uint32_t(0); //uint32_t(positive rating)
134 *data << uint32_t(0); //uint32_t(0)[<-this is the new field in 3.1]
135 *data << uint8_t(0); //name if available / which is a null-terminated string, and we send an uint8_t(0), so we provide a zero length name string
136 *data << uint32_t(0);
137 *data << uint32_t(0);
138 *data << uint32_t(0);
139 *data << uint8_t(0);
140 }
141 else
142 {
143 /* Grab some arena teams */
144 auto** teams = dynamic_cast< Arena* >(this)->GetTeams();
145
146 if (teams[0])
147 {
148 *data << uint32_t(0);
149 *data << uint32_t(3000 + m_deltaRating[0]);
150 *data << uint32_t(0);
151 *data << uint8_t(0);
152 }
153 else
154 {
155 *data << uint32_t(0);
156 *data << uint32_t(0);
157 *data << uint32_t(0);
158 *data << uint8_t(0);
159 }
160
161 if (teams[1])
162 {
163 *data << uint32_t(0);
164 *data << uint32_t(3000 + m_deltaRating[1]);
165 *data << uint32_t(0);
166 *data << uint8_t(0);
167 }
168 else
169 {
170 *data << uint32_t(0);
171 *data << uint32_t(0);
172 *data << uint32_t(0);
173 *data << uint8_t(0);
174 }
175 }
176
177 *data << uint8_t(1);
178 *data << uint8_t(m_winningTeam);
179
180 *data << uint32_t((m_players[0].size() + m_players[1].size()) - m_invisGMs);
181
182 for (auto& m_player : m_players)
183 {
184 for (const auto itr : m_player)
185 {
186 if (itr->m_isGmInvisible)
187 continue;
188
189 *data << itr->getGuid();
190 bs = &itr->m_bgScore;
191 *data << bs->KillingBlows;
192
193 *data << uint8_t(itr->getBgTeam());
194
195 *data << bs->DamageDone;
196 *data << bs->HealingDone;
197 *data << uint32_t(0);
198 }
199 }
200 }
201 else
202 {
203 *data << uint8_t(0);
204 if (m_hasEnded)
205 {
206 *data << uint8_t(1);
207 *data << uint8_t(m_winningTeam ? 0 : 1);
208 }
209 else
210 {
211 *data << uint8_t(0); // If the game has ended - this will be 1
212 }
213
214 *data << uint32_t((m_players[0].size() + m_players[1].size()) - m_invisGMs);
215
216 const uint32_t FieldCount = getFieldCount(getType());
217 for (auto& m_player : m_players)
218 {
219 for (const auto itr : m_player)
220 {
221 if (itr != nullptr)
222 {
223 if (itr->m_isGmInvisible)
224 continue;
225
226 *data << itr->getGuid();
227 bs = &itr->m_bgScore;
228
229 *data << bs->KillingBlows;
230 *data << bs->HonorableKills;
231 *data << bs->Deaths;
232 *data << bs->BonusHonor;
233 *data << bs->DamageDone;
234 *data << bs->HealingDone;
235
236 *data << FieldCount;
237 for (uint32_t x = 0; x < FieldCount; ++x)
238 *data << bs->MiscData[x];
239 }
240 }
241 }
242 }

250void Battleground::addPlayer(Player* plr, uint32_t team)
251{
252 std::lock_guard lock(m_mutex);
253
254 // This is called when the player is added, not when they port. So, they're essentially still queued, but not inside the bg yet
255 m_pendPlayers[team].insert(plr->getGuidLow());
256
257 // Send a packet telling them that they can enter
258 plr->setPendingBattleground(this);
259 sBattlegroundManager.sendBattlefieldStatus(plr, BattlegroundDef::STATUS_READY, m_type, m_id, 80000, m_mapMgr->getBaseMap()->getMapId(), Rated()); // You will be removed from the queue in 2 minutes.
260
261 // Add an event to remove them in 1 minute 20 seconds time

293void Battleground::portPlayer(Player* plr, bool skip_teleport)
294{
295 std::lock_guard lock(m_mutex);
296
297 if (m_hasEnded)
298 {
299 plr->getSession()->systemMessage(plr->getSession()->LocalizedWorldSrv(ServerString::SS_YOU_CANNOT_JOIN_BG_AS_IT_HAS_ALREADY_ENDED));
300 sBattlegroundManager.sendBattlefieldStatus(plr, BattlegroundDef::STATUS_NOFLAGS, 0, 0, 0, 0, 0);
301 plr->setPendingBattleground(nullptr);
302 return;
303 }
304
305 m_pendPlayers[plr->getBgTeam()].erase(plr->getGuidLow());
306 if (m_players[plr->getBgTeam()].find(plr) != m_players[plr->getBgTeam()].end())
307 return;
308
309 plr->setFullHealthMana();
310 plr->setTeam(plr->getBgTeam());
311
312 // Do not let everyone know an invisible gm has joined.
313 if (plr->m_isGmInvisible == false)
314 distributePacketToTeam(AscEmu::Packets::SmsgBattlegroundPlayerJoined(plr->getGuid()).serialise().get(), plr->getBgTeam());
315 else
316 ++m_invisGMs;
317
318 m_players[plr->getBgTeam()].insert(plr);
319
320 // remove from any auto queue remove events
321 sEventMgr.RemoveEvents(plr, EVENT_BATTLEGROUND_QUEUE_UPDATE);
322
323 if (!skip_teleport)
324 if (plr->IsInWorld())
325 plr->removeFromWorld();
326
327 plr->setPendingBattleground(nullptr);
328 plr->setBattleground(this);
329 plr->setLastBattlegroundPetId(0);
330 plr->setLastBattlegroundPetSpell(0);
331
332 if (!plr->isPvpFlagSet())
333 plr->setPvpFlag();
334
335 plr->removeAllAurasByAuraInterruptFlag(AURA_INTERRUPT_ON_PVP_ENTER);
336
337 // Reset the score
338 memset(&plr->m_bgScore, 0, sizeof(BGScore));
339
340 // update pvp data
341 updatePvPData();
342
343 // add the player to the group
344 if (plr->getGroup() && !Rated())
345 {
346 // remove them from their group
347 plr->getGroup()->RemovePlayer(plr->getPlayerInfo());
348 }
349
350 if (!m_countdownStage)
351 {
352 m_countdownStage = 1;
353 sEventMgr.AddEvent(this, &Battleground::eventCountdown, EVENT_BATTLEGROUND_COUNTDOWN, 30000, 0, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
354 sEventMgr.ModifyEventTimeLeft(this, EVENT_BATTLEGROUND_COUNTDOWN, 10000);
355 }
356
357 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_CLOSE);
358
359 if (!skip_teleport)
360 {
361 // This is where we actually teleport the player to the battleground
362 plr->safeTeleport(m_mapMgr, GetStartingCoords(plr->getBgTeam()));
363 sBattlegroundManager.sendBattlefieldStatus(plr, BattlegroundDef::STATUS_TIME, m_type, m_id, static_cast<uint32_t>(UNIXTIME) - m_startTime, m_mapMgr->getBaseMap()->getMapId(), Rated()); // Elapsed time is the last argument
364 }
365 else
366 {
367 // If we are not ported, call this immediatelly, otherwise its called after teleportation in Player::OnPushToWorld
368 OnAddPlayer(plr);
369 }
370}

439void Battleground::endBattleground(PlayerTeam winningTeam)
440{
441 std::lock_guard lock(m_mutex);
442
443 this->m_hasEnded = true;
444 this->m_winningTeam = winningTeam;
445 this->m_nextPvPUpdateTime = 0;
446
447 const auto losingTeam = winningTeam == TEAM_ALLIANCE ? TEAM_HORDE : TEAM_ALLIANCE;
448
449 /* If we still need to handle reward calculations */
450 if (this->HandleFinishBattlegroundRewardCalculation(winningTeam))
451 {
452 for (const auto winningPlr : m_players[winningTeam])
453 {
454 if (winningPlr->isQueuedForRbg())
455 {
456 winningPlr->applyRandomBattlegroundReward(true);
457 winningPlr->setHasWonRbgToday(true);
458 }
459
460 winningPlr->saveToDB(false);
461 }
462 for (const auto losingPlr : m_players[losingTeam])
463 {
464 if (losingPlr->isQueuedForRbg())
465 {
466 losingPlr->applyRandomBattlegroundReward(false);
467 }
468
469 losingPlr->saveToDB(false);
470 }
471
472 this->addHonorToTeam(winningTeam, 3 * 185);
473 this->addHonorToTeam(losingTeam, 1 * 185);
474 }
475
476 this->playSoundToAll(winningTeam == TEAM_ALLIANCE ? BattlegroundDef::ALLIANCEWINS : BattlegroundDef::HORDEWINS);
477
478 this->updatePvPData();
479}

572void Battleground::removePlayer(Player* plr, bool logout)
573{
574 std::lock_guard lock(m_mutex);
575
576 // Don't show invisible gm's leaving the game.
577 if (plr->m_isGmInvisible == false)
578 distributePacketToAll(AscEmu::Packets::SmsgBattlegroundPlayerLeft(plr->getGuid()).serialise().get());
579 else
580 --m_invisGMs;
581
582 // Call subclassed virtual method
583 OnRemovePlayer(plr);
584
585 // Clean-up
586 plr->setBattleground(nullptr);
587 plr->setFullHealthMana();
588 plr->setLastBattlegroundPetId(0);
589 plr->setLastBattlegroundPetSpell(0);
590 m_players[plr->getBgTeam()].erase(plr);
591 memset(&plr->m_bgScore, 0, sizeof(BGScore));
592
593 // are we in the group?
594 if (plr->getGroup() == m_groups[plr->getBgTeam()])
595 plr->getGroup()->RemovePlayer(plr->getPlayerInfo());
596
597 // reset team
598 plr->resetTeam();
599
600 // revive the player if he is dead
601 if (!plr->isAlive())
602 {
603 plr->setHealth(plr->getMaxHealth());
604 plr->resurrect();
605 }
606
607 plr->removeAllAurasById(32727); // Arena preparation
608 plr->removeAllAurasById(44521); // BG preparation
609 plr->removeAllAurasById(44535);
610 plr->removeAllAurasById(21074);
611
612 plr->setMoveRoot(false);
613
614 // teleport out
615 if (!logout)
616 {
617 if (!m_hasEnded)
618 if(!plr->getSession()->HasGMPermissions())
619 plr->castSpell(plr, BattlegroundDef::DESERTER, true);
620
621 if (!IS_INSTANCE(plr->getBGEntryMapId()))
622 plr->safeTeleport(plr->getBGEntryMapId(), plr->getBGEntryInstanceId(), plr->getBGEntryPosition());
623 else
624 plr->safeTeleport(plr->getBindMapId(), 0, plr->getBindPosition());
625
626 sBattlegroundManager.sendBattlefieldStatus(plr, BattlegroundDef::STATUS_NOFLAGS, 0, 0, 0, 0, 0);
627 }
628
629 if (m_players[0].size() == 0 && m_players[1].size() == 0)
630 {
631 // create an inactive event
632 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_CLOSE);
633 this->close();
634 }
635
636 plr->setBgTeam(plr->getTeam());
637}

663void Battleground::eventCountdown()
664{
665 if (m_countdownStage == 1)
666 {
667 std::lock_guard lock(m_mutex);
668 m_countdownStage = 2;
669
670 for (auto& m_player : m_players)
671 {
672 for (const auto itr : m_player)
673 if (itr && itr->getSession())
674 itr->getSession()->SystemMessage(itr->getSession()->LocalizedWorldSrv(ServerString::SS_BATTLE_BEGIN_ONE_MINUTE),
675 itr->getSession()->LocalizedWorldSrv(GetNameID()));
676 }
677 }
678 else if (m_countdownStage == 2)
679 {
680 std::lock_guard lock(m_mutex);
681
682 m_countdownStage = 3;
683
684 for (auto& m_player : m_players)
685 {
686 for (const auto itr : m_player)
687 if (itr && itr->getSession())
688 itr->getSession()->SystemMessage(itr->getSession()->LocalizedWorldSrv(ServerString::SS_THIRTY_SECONDS_UNTIL_THE_BATTLE),
689 itr->getSession()->LocalizedWorldSrv(GetNameID()));
690 }
691 }
692 else if (m_countdownStage == 3)
693 {
694 std::lock_guard lock(m_mutex);
695
696 m_countdownStage = 4;
697
698 for (auto& m_player : m_players)
699 {
700 for (const auto itr : m_player)
701 if (itr && itr->getSession())
702 itr->getSession()->SystemMessage(itr->getSession()->LocalizedWorldSrv(ServerString::SS_FIFTEEN_SECONDS_UNTIL_THE_BATTLE),
703 itr->getSession()->LocalizedWorldSrv(GetNameID()));
704 }
705
706 sEventMgr.ModifyEventTime(this, EVENT_BATTLEGROUND_COUNTDOWN, 150);
707 sEventMgr.ModifyEventTimeLeft(this, EVENT_BATTLEGROUND_COUNTDOWN, 15000);
708 }
709 else
710 {
711 std::lock_guard lock(m_mutex);
712 for (auto& m_player : m_players)
713 {
714 for (const auto itr : m_player)
715 if (itr && itr->getSession())
716 itr->getSession()->SystemMessage(itr->getSession()->LocalizedWorldSrv(ServerString::SS_THE_BATTLE_FOR_HAS_BEGUN),
717 itr->getSession()->LocalizedWorldSrv(GetNameID()));
718 }
719
720 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_COUNTDOWN);
721 this->startBattleground();
722 }
