29template<class T, class BoundsFunc = BoundsTrait<T> >
30class BIHWrap
31{
32 template<class RayCallback>
33 struct MDLCallback
34 {
35 const T* const* objects;
36 RayCallback& _callback;
37 uint32_t objects_size;
38
39 MDLCallback(RayCallback& callback, const T* const* objects_array, uint32_t objects_size ) : objects(objects_array), _callback(callback), objects_size(objects_size) { }
40
41 /// Intersect ray
42 bool operator() (const G3D::Ray& ray, uint32_t idx, float& maxDist, bool /*stopAtFirst*/)
43 {
44 if (idx >= objects_size)
45 return false;
46 if (const T* obj = objects[idx])
47 return _callback(ray, *obj, maxDist/*, stopAtFirst*/);
48 return false;
49 }
50
51 /// Intersect point
52 void operator() (const G3D::Vector3& p, uint32_t idx)
53 {
54 if (idx >= objects_size)
55 return;
56 if (const T* obj = objects[idx])
57 _callback(p, *obj);
58 }
59 };

63 BIH m_tree;

69public:

72 void insert(const T& obj)

74 ++unbalanced_times;
75 m_objects_to_push.insert(&obj);

78 void remove(const T& obj)
79 {
80 ++unbalanced_times;
81 uint32_t Idx = 0;
82 const T * temp;
83 if (m_obj2Idx.getRemove(&obj, temp, Idx))
84 m_objects[Idx] = NULL;
85 else
86 m_objects_to_push.remove(&obj);
87 }
88
89 void balance()
90 {
91 if (unbalanced_times == 0)
92 return;
93
94 unbalanced_times = 0;
95 m_objects.fastClear();
96 m_obj2Idx.getKeys(m_objects);
97 m_objects_to_push.getMembers(m_objects);
98 //assert that m_obj2Idx has all the keys
99
100 m_tree.build(m_objects, BoundsFunc::getBounds2);
101 }
