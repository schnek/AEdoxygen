23{

36void BIH::subdivide(int left, int right, std::vector<uint32_t> &tempTree, buildData &dat, AABound &gridBox, AABound &nodeBox, int nodeIndex, int depth, BuildStats &stats)
37{
38 if ((right - left + 1) <= dat.maxPrims || depth >= MAX_STACK_SIZE)
39 {
40 // write leaf node
41 stats.updateLeaf(depth, right - left + 1);
42 createNode(tempTree, nodeIndex, left, right);
43 return;
44 }
45 // calculate extents
46 int axis = -1, prevAxis, rightOrig;
47 float clipL = G3D::fnan(), clipR = G3D::fnan(), prevClip = G3D::fnan();
48 float split = G3D::fnan(), prevSplit;
49 bool wasLeft = true;
50 while (true)
51 {
52 prevAxis = axis;
53 prevSplit = split;
54 // perform quick consistency checks
55 G3D::Vector3 d( gridBox.hi - gridBox.lo );
56 if (d.x < 0 || d.y < 0 || d.z < 0)
57 throw std::logic_error("negative node extents");
58 for (int i = 0; i < 3; i++)
59 {
60 if (nodeBox.hi[i] < gridBox.lo[i] || nodeBox.lo[i] > gridBox.hi[i])
61 {
62 //UI.printError(Module.ACCEL, "Reached tree area in error - discarding node with: %d objects", right - left + 1);
63 throw std::logic_error("invalid node overlap");
64 }
65 }
66 // find longest axis
67 axis = d.primaryAxis();
68 split = 0.5f * (gridBox.lo[axis] + gridBox.hi[axis]);
69 // partition L/R subsets
70 clipL = -G3D::finf();
71 clipR = G3D::finf();
72 rightOrig = right; // save this for later
73 float nodeL = G3D::finf();
74 float nodeR = -G3D::finf();
75 for (int i = left; i <= right;)
76 {
77 int obj = dat.indices[i];
78 float minb = dat.primBound[obj].low()[axis];
79 float maxb = dat.primBound[obj].high()[axis];
80 float center = (minb + maxb) * 0.5f;
81 if (center <= split)
82 {
83 // stay left
84 i++;
85 if (clipL < maxb)
86 clipL = maxb;
87 }
88 else
89 {
90 // move to the right most
91 int t = dat.indices[i];
92 dat.indices[i] = dat.indices[right];
93 dat.indices[right] = t;
94 right--;
95 if (clipR > minb)
96 clipR = minb;
97 }
98 nodeL = std::min(nodeL, minb);
99 nodeR = std::max(nodeR, maxb);
100 }
101 // check for empty space
102 if (nodeL > nodeBox.lo[axis] && nodeR < nodeBox.hi[axis])
103 {
104 float nodeBoxW = nodeBox.hi[axis] - nodeBox.lo[axis];
105 float nodeNewW = nodeR - nodeL;
106 // node box is too big compare to space occupied by primitives?
107 if (1.3f * nodeNewW < nodeBoxW)
108 {
109 stats.updateBVH2();
110 int nextIndex = static_cast<int>(tempTree.size());
111 // allocate child
112 tempTree.push_back(0);
113 tempTree.push_back(0);
114 tempTree.push_back(0);
115 // write bvh2 clip node
116 stats.updateInner();
117 tempTree[nodeIndex + 0] = (axis << 30) | (1 << 29) | nextIndex;
118 tempTree[nodeIndex + 1] = floatToRawIntBits(nodeL);
119 tempTree[nodeIndex + 2] = floatToRawIntBits(nodeR);
120 // update nodebox and recurse
121 nodeBox.lo[axis] = nodeL;
122 nodeBox.hi[axis] = nodeR;
123 subdivide(left, rightOrig, tempTree, dat, gridBox, nodeBox, nextIndex, depth + 1, stats);
124 return;
125 }
126 }
127 // ensure we are making progress in the subdivision
128 if (right == rightOrig)
129 {
130 // all left
131 if (prevAxis == axis && G3D::fuzzyEq(prevSplit, split)) {
132 // we are stuck here - create a leaf
133 stats.updateLeaf(depth, right - left + 1);
134 createNode(tempTree, nodeIndex, left, right);
135 return;
136 }
137 if (clipL <= split) {
138 // keep looping on left half
139 gridBox.hi[axis] = split;
140 prevClip = clipL;
141 wasLeft = true;
142 continue;
143 }
144 gridBox.hi[axis] = split;
145 prevClip = G3D::fnan();
146 }
147 else if (left > right)
148 {
149 // all right
150 right = rightOrig;
151 if (prevAxis == axis && G3D::fuzzyEq(prevSplit, split)) {
152 // we are stuck here - create a leaf
153 stats.updateLeaf(depth, right - left + 1);
154 createNode(tempTree, nodeIndex, left, right);
155 return;
156 }
157 if (clipR >= split) {
158 // keep looping on right half
159 gridBox.lo[axis] = split;
160 prevClip = clipR;
161 wasLeft = false;
162 continue;
163 }
164 gridBox.lo[axis] = split;
165 prevClip = G3D::fnan();
166 }
167 else
168 {
169 // we are actually splitting stuff
170 if (prevAxis != -1 && !std::isnan(prevClip))
171 {
172 // second time through - lets create the previous split
173 // since it produced empty space
174 int nextIndex = static_cast<int>(tempTree.size());
175 // allocate child node
176 tempTree.push_back(0);
177 tempTree.push_back(0);
178 tempTree.push_back(0);
179 if (wasLeft) {
180 // create a node with a left child
181 // write leaf node
182 stats.updateInner();
183 tempTree[nodeIndex + 0] = (prevAxis << 30) | nextIndex;
184 tempTree[nodeIndex + 1] = floatToRawIntBits(prevClip);
185 tempTree[nodeIndex + 2] = floatToRawIntBits(G3D::finf());
186 } else {
187 // create a node with a right child
188 // write leaf node
189 stats.updateInner();
190 tempTree[nodeIndex + 0] = (prevAxis << 30) | (nextIndex - 3);
191 tempTree[nodeIndex + 1] = floatToRawIntBits(-G3D::finf());
192 tempTree[nodeIndex + 2] = floatToRawIntBits(prevClip);
193 }
194 // count stats for the unused leaf
195 depth++;
196 stats.updateLeaf(depth, 0);
197 // now we keep going as we are, with a new nodeIndex:
198 nodeIndex = nextIndex;
199 }
200 break;
201 }
202 }
203 // compute index of child nodes
204 int nextIndex = static_cast<int>(tempTree.size());
205 // allocate left node
206 int nl = right - left + 1;
207 int nr = rightOrig - (right + 1) + 1;
208 if (nl > 0) {
209 tempTree.push_back(0);
210 tempTree.push_back(0);
211 tempTree.push_back(0);
212 } else
213 nextIndex -= 3;
214 // allocate right node
215 if (nr > 0) {
216 tempTree.push_back(0);
217 tempTree.push_back(0);
218 tempTree.push_back(0);
219 }
220 // write leaf node
221 stats.updateInner();
222 tempTree[nodeIndex + 0] = (axis << 30) | nextIndex;
223 tempTree[nodeIndex + 1] = floatToRawIntBits(clipL);
224 tempTree[nodeIndex + 2] = floatToRawIntBits(clipR);
225 // prepare L/R child boxes
226 AABound gridBoxL(gridBox), gridBoxR(gridBox);
227 AABound nodeBoxL(nodeBox), nodeBoxR(nodeBox);
228 gridBoxL.hi[axis] = gridBoxR.lo[axis] = split;
229 nodeBoxL.hi[axis] = clipL;
230 nodeBoxR.lo[axis] = clipR;
231 // recurse
232 if (nl > 0)
233 subdivide(left, right, tempTree, dat, gridBoxL, nodeBoxL, nextIndex, depth + 1, stats);
234 else
235 stats.updateLeaf(depth + 1, 0);
236 if (nr > 0)
237 subdivide(right + 1, rightOrig, tempTree, dat, gridBoxR, nodeBoxR, nextIndex + 3, depth + 1, stats);
238 else
239 stats.updateLeaf(depth + 1, 0);
240}

242bool BIH::writeToFile(FILE* wf) const
243{
244 uint32_t treeSize = static_cast<uint32_t>(tree.size());
245 size_t check = 0;
246 check += fwrite(&bounds.low(), sizeof(float), 3, wf);
247 check += fwrite(&bounds.high(), sizeof(float), 3, wf);
248 check += fwrite(&treeSize, sizeof(uint32_t), 1, wf);
249 check += fwrite(&tree[0], sizeof(uint32_t), treeSize, wf);
250 uint32_t count = static_cast<uint32_t>(objects.size());
251 check += fwrite(&count, sizeof(uint32_t), 1, wf);
252 check += fwrite(&objects[0], sizeof(uint32_t), count, wf);
253 return (check == (3 + 3 + 2 + treeSize + count));
254}

256bool BIH::readFromFile(FILE* rf)
257{
258 uint32_t treeSize;
259 G3D::Vector3 lo, hi;
260 uint32_t check = 0, count = 0;
261 check += static_cast<uint32_t>(fread(&lo, sizeof(float), 3, rf));
262 check += static_cast<uint32_t>(fread(&hi, sizeof(float), 3, rf));
263 bounds = G3D::AABox(lo, hi);
264 check += static_cast<uint32_t>(fread(&treeSize, sizeof(uint32_t), 1, rf));
265 tree.resize(treeSize);
266 check += static_cast<uint32_t>(fread(&tree[0], sizeof(uint32_t), treeSize, rf));
267 check += static_cast<uint32_t>(fread(&count, sizeof(uint32_t), 1, rf));
268 objects.resize(count); // = new uint32_t[nObjects];
269 check += static_cast<uint32_t>(fread(&objects[0], sizeof(uint32_t), count, rf));
270 return (uint64_t(check) == uint64_t(3 + 3 + 1 + 1 + uint64_t(treeSize) + uint64_t(count)));
271}

273void BIH::BuildStats::updateLeaf(int depth, int n)
274{
275 numLeaves++;
276 minDepth = std::min(depth, minDepth);
277 maxDepth = std::max(depth, maxDepth);
278 sumDepth += depth;
279 minObjects = std::min(n, minObjects);
280 maxObjects = std::max(n, maxObjects);
281 sumObjects += n;
282 int nl = std::min(n, 5);
283 ++numLeavesN[nl];
284}
