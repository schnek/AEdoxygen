49void BroadcastMgr::sendBroadcast()
50{
51 MySQLDataStore::WorldBroadcastContainer* its = sMySQLStore.getWorldBroadcastStore();
52 for (MySQLDataStore::WorldBroadcastContainer::iterator itr = its->begin(); itr != its->end(); ++itr)
53 {
54 if (itr->second.nextUpdate < (uint32_t)UNIXTIME)
55 {
56 sWorld.sendBroadcastMessageById(itr->second.id);
57
58 uint32_t randomMins = (itr->second.randomInterval ? Util::getRandomUInt(itr->second.randomInterval) : 0);
59 uint32_t intervalMins = itr->second.interval;
60
61 itr->second.nextUpdate = randomMins + intervalMins + (uint32_t)UNIXTIME;
62 }
63 }
