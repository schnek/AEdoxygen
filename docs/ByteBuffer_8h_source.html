474 uint32_t ReadPackedTime()
475 {
476 uint32_t packedDate = read<uint32_t>();
477 tm lt = tm();
478
479 lt.tm_min = packedDate & 0x3F;
480 lt.tm_hour = (packedDate >> 6) & 0x1F;
481 //lt.tm_wday = (packedDate >> 11) & 7;
482 lt.tm_mday = ((packedDate >> 14) & 0x3F) + 1;
483 lt.tm_mon = (packedDate >> 20) & 0xF;
484 lt.tm_year = ((packedDate >> 24) & 0x1F) + 100;
485
486 return uint32_t(mktime(&lt));

563 {
564 size_t mask_position = wpos();

579 {
580 uint64_t guid = 0;

583 *this >> mask;

601 template <typename T> void PutBits(size_t pos, T value, uint32_t bitCount)
602 {
603 for (uint32_t i = 0; i < bitCount; ++i)
604 {
605 size_t wp = (pos + i) / 8;
606 size_t bit = (pos + i) % 8;
607 if ((value >> (bitCount - i - 1)) & 1)
608 _storage[wp] |= 1 << (7 - bit);
609 else
610 _storage[wp] &= ~(1 << (7 - bit));
611 }
612 }
