59Channel* ChannelMgr::getOrCreateChannel(std::string name, Player const* player, uint32_t typeId)
60{
61 auto channelList = &m_channelList[0];
62 if (m_seperateChannels && player && name != worldConfig.getGmClientChannelName())
63 channelList = &m_channelList[player->getTeam()];
64
65 {
66 std::lock_guard<std::mutex> configGuard(m_mutexConfig);
67 for (const auto& m_bannedChannel : m_bannedChannels)
68 {
69 if (name == m_bannedChannel)
70 return nullptr;
71 }
72 }
73
74 std::lock_guard<std::mutex> channelGuard(m_mutexChannels);
75 const auto teamId = (m_seperateChannels && player) ? player->getTeam() : TEAM_ALLIANCE;
76
77 const auto [channelItr, _] = channelList->try_emplace(name, Util::LazyInstanceCreator([&name, teamId, typeId] {
78 return std::make_unique<Channel>(name, teamId, typeId);

83void ChannelMgr::removeChannel(Channel const* channel)

90 channelList = &m_channelList[channel->getChannelTeam()];

94 for (auto channelListMember = channelList->begin(); channelListMember != channelList->end(); ++channelListMember)
95 {
96 if (channelListMember->second.get() == channel)

138bool ChannelMgr::canPlayerJoinDefaultChannel(Player const* player, WDB::Structures::AreaTableEntry const* areaEntry, WDB::Structures::ChatChannelsEntry const* channelDbc) const
139{
140 if (player == nullptr || channelDbc == nullptr)
141 return false;
142
143 // City specific channels
144 if (channelDbc->flags & (CHANNEL_DBC_CITY_ONLY_1 | CHANNEL_DBC_CITY_ONLY_2))
145 {
146 if (areaEntry == nullptr || !(areaEntry->flags & MapManagement::AreaManagement::AREA_CITY_AREA))
147 return false;
148 }
149
150 // Channels with zone name
151 if (channelDbc->flags & CHANNEL_DBC_HAS_ZONENAME)
152 {
153 if (areaEntry == nullptr)
154 return false;
155 }
156
157 // Guild recruitment channel
158 if (channelDbc->flags & CHANNEL_DBC_GUILD_RECRUIT)
159 {
160 if (player->getGuild() != nullptr)
161 return false;
162 }
163
164 return true;
165}

167std::string ChannelMgr::generateChannelName(WDB::Structures::ChatChannelsEntry const* channelDbc, WDB::Structures::AreaTableEntry const* areaEntry) const
168{
169#if VERSION_STRING < Cata
170 char* channelNameDbc = channelDbc->name_pattern[sWorld.getDbcLocaleLanguageId()];
171#else
172 char* channelNameDbc = channelDbc->name_pattern[0];
173#endif
174
175 if (channelDbc->flags & CHANNEL_DBC_GLOBAL || !(channelDbc->flags & CHANNEL_DBC_HAS_ZONENAME))
176 return std::string(channelNameDbc);
177
178 char channelName[95];
179 char const* defaultAreaName = "City";
180
181 if (const auto defaultArea = MapManagement::AreaManagement::AreaStorage::GetAreaById(3459))
182 {
183#if VERSION_STRING < Cata
184 defaultAreaName = defaultArea->area_name[sWorld.getDbcLocaleLanguageId()];
185#else
186 defaultAreaName = defaultArea->area_name[0];
187#endif
188 }
189
190 // City specific channels
191 if (channelDbc->flags & (CHANNEL_DBC_CITY_ONLY_1 | CHANNEL_DBC_CITY_ONLY_2))
192 {
193 std::snprintf(channelName, 95, channelNameDbc, defaultAreaName);
194 }
195 else
196 {
197 if (areaEntry != nullptr)
198 {
199#if VERSION_STRING < Cata
200 std::snprintf(channelName, 95, channelNameDbc, areaEntry->area_name[sWorld.getDbcLocaleLanguageId()]);
201#else
202 std::snprintf(channelName, 95, channelNameDbc, areaEntry->area_name);
203#endif
204 }
205 else
206 {
207 std::snprintf(channelName, 95, channelNameDbc, defaultAreaName);
208 }
209 }
210
211 return std::string(channelName);
212}
