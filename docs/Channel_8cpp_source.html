20Channel::Channel(std::string name, uint8_t team, uint32_t channelId/* = 0*/) :
21 m_channelName(name), m_channelTeam(team), m_channelId(channelId)
22{
23 const auto channelDbc = sChatChannelsStore.lookupEntry(channelId);
24 if (channelDbc != nullptr)
25 {
26 // Default channels
27 m_channelFlags |= CHANNEL_FLAGS_GENERAL;
28 m_announcePlayers = false;
29
30 // Trade channel
31 if (channelDbc->flags & CHANNEL_DBC_TRADE)
32 m_channelFlags |= CHANNEL_FLAGS_TRADE;
33
34 // Channels that are active in cities
35 if (channelDbc->flags & (CHANNEL_DBC_CITY_ONLY_1 | CHANNEL_DBC_CITY_ONLY_2))
36 m_channelFlags |= CHANNEL_FLAGS_CITY;
37
38 // LookingForGroup channel
39 if (channelDbc->flags & CHANNEL_DBC_LFG)
40 m_channelFlags |= CHANNEL_FLAGS_LFG;
41 else
42 m_channelFlags |= CHANNEL_FLAGS_NOT_LFG;
43 }
44 else
45 {
46 // Player custom created channels
47 m_channelFlags = CHANNEL_FLAGS_CUSTOM;
48 m_channelId = 0;
49 }
50
51 for (const auto& channelName : sChannelMgr.m_minimumChannel)
52 {
53 if (name != channelName)
54 {
55 m_minimumLevel = 10;
56 m_announcePlayers = false;
57 break;
58 }
59 }
60}

85void Channel::attemptJoin(Player* plr, std::string password, bool skipCheck/* = false*/)
86{
87 if (!skipCheck && m_channelFlags & CHANNEL_FLAGS_GENERAL)
88 {
89 const auto areaEntry = plr->GetArea();
90
91 const auto channelDbc = sChatChannelsStore.lookupEntry(getChannelId());
92 if (!sChannelMgr.canPlayerJoinDefaultChannel(plr, areaEntry, channelDbc))
93 return;
94 }
95
96 uint8_t memberFlags = CHANNEL_MEMBER_FLAG_NONE;
97
98 if (!(m_channelFlags & CHANNEL_FLAGS_GENERAL) && plr->getSession()->CanUseCommand('c'))
99 memberFlags |= CHANNEL_MEMBER_FLAG_MODERATOR;
100
101 if (!m_channelPassword.empty() && strcmp(m_channelPassword.c_str(), password.c_str()) != 0)
102 {
103 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_WRONGPASS, m_channelName).serialise().get());
104 return;
105 }
106
107 m_mutexChannel.lock();
108
109 if (m_bannedMembers.find(plr->getGuidLow()) != m_bannedMembers.end())
110 {
111 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOURBANNED, m_channelName).serialise().get());
112 m_mutexChannel.unlock();
113 return;
114 }
115
116 if (m_members.find(plr) != m_members.end())
117 {
118 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_ALREADY_ON, m_channelName).serialise().get());
119 m_mutexChannel.unlock();
120 return;
121 }
122
123 if (m_members.empty() && !(m_channelFlags & CHANNEL_FLAGS_GENERAL))
124 memberFlags |= CHANNEL_MEMBER_FLAG_OWNER;
125
126 m_members.insert(std::make_pair(plr, memberFlags));
127 m_mutexChannel.unlock();
128
129 plr->joinedChannel(this);
130
131 // Announce player join to other members in channel
132 if (m_announcePlayers)
133 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_JOINED, m_channelName, plr->getGuid()).serialise().get(), nullptr);
134
135 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOUJOINED, m_channelName, 0, m_channelFlags, m_channelId).serialise().get());
136}

138void Channel::leaveChannel(Player* plr, bool sendPacket/* = true*/)
139{
140 m_mutexChannel.lock();
141
142 const MemberMap::const_iterator itr = m_members.find(plr);
143 if (itr == m_members.end())
144 {
145 // Player is not on this channel
146 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTON, m_channelName).serialise().get());
147 m_mutexChannel.unlock();
148 return;
149 }
150
151 const auto memberFlags = itr->second;
152 m_members.erase(itr);
153 m_mutexChannel.unlock();
154
155 plr->leftChannel(this);
156
157 // If player is channel owner, find new owner for channel
158 if (memberFlags & CHANNEL_MEMBER_FLAG_OWNER)
159 setOwner(nullptr, nullptr);
160
161 // Do not send packet in teleport or logout
162 if (sendPacket && !(plr->getSession() && (plr->getSession()->IsLoggingOut() || plr->getTeleportState() == 1)))
163 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOULEFT, m_channelName, 0, 0, m_channelId).serialise().get());
164
165 // Announce player leave to other members in channel
166 if (m_announcePlayers)
167 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_LEFT, m_channelName, plr->getGuid()).serialise().get());
168
169 // If channel is now empty, delete it
170 if (m_members.empty())
171 sChannelMgr.removeChannel(this);
172}

185void Channel::say(Player* plr, std::string message, Player* for_gm_client, bool forced)
186{
187 if (!forced)
188 {
189 std::lock_guard<std::mutex> guard(m_mutexChannel);
190
191 const MemberMap::const_iterator itr = m_members.find(plr);
192 if (itr == m_members.end())
193 {
194 // Player is not on channel
195 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTON, m_channelName).serialise().get());
196 return;
197 }
198
199 if (itr->second & CHANNEL_MEMBER_FLAG_MUTED)
200 {
201 // Player is muted
202 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOUCANTSPEAK, m_channelName).serialise().get());
203 return;
204 }
205
206 if (m_muted && !(itr->second & CHANNEL_MEMBER_FLAG_VOICED) && !(itr->second & CHANNEL_MEMBER_FLAG_MODERATOR) && !(itr->second & CHANNEL_MEMBER_FLAG_OWNER))
207 {
208 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOUCANTSPEAK, m_channelName).serialise().get());
209 return;
210 }
211 }
212
213 if (plr->getLevel() < m_minimumLevel)
214 {
215 plr->broadcastMessage("You must be level %u to speak in the channel, '%s'.", m_minimumLevel, m_channelName.c_str());
216 return;
217 }
218
219 // Send message
220 const uint8_t gmFlag = plr->isGMFlagSet() ? 4U : 0U;
221 if (for_gm_client != nullptr)
222 for_gm_client->sendPacket(SmsgMessageChat(CHAT_MSG_CHANNEL, LANG_UNIVERSAL, gmFlag, message, plr->getGuid(), "", 0, m_channelName).serialise().get());
223 else
224 sendToAll(SmsgMessageChat(CHAT_MSG_CHANNEL, LANG_UNIVERSAL, gmFlag, message, plr->getGuid(), "", 0, m_channelName).serialise().get());
225}

227void Channel::invitePlayer(Player* plr, Player* new_player)
228{
229 std::lock_guard<std::mutex> guard(m_mutexChannel);
230
231 if (m_members.find(plr) == m_members.end())
232 {
233 // Player is not on channel
234 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTON, m_channelName).serialise().get());
235 return;
236 }
237
238 if (m_members.find(new_player) != m_members.end())
239 {
240 // Invited player is already on the channel
241 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_ALREADY_ON, m_channelName, new_player->getGuid()).serialise().get());
242 return;
243 }
244
245 new_player->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_INVITED, m_channelName, plr->getGuid()).serialise().get());
246 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOU_INVITED, m_channelName, 0, 0, 0, 0, new_player->getName()).serialise().get());
247}

249void Channel::kickOrBanPlayer(Player* plr, Player* die_player, bool ban)
250{
251 m_mutexChannel.lock();
252
253 const MemberMap::const_iterator me_itr = m_members.find(plr);
254 if (me_itr == m_members.end())
255 {
256 // Player is not on channel
257 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTON, m_channelName).serialise().get());
258 m_mutexChannel.unlock();
259 return;
260 }
261
262 const MemberMap::const_iterator itr = m_members.find(die_player);
263 if (itr == m_members.end())
264 {
265 // Kicked player is not on channel
266 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOT_ON_2, m_channelName, 0, 0, 0, 0, die_player->getName()).serialise().get());
267 m_mutexChannel.unlock();
268 return;
269 }
270
271 if (!(me_itr->second & (CHANNEL_MEMBER_FLAG_OWNER | CHANNEL_MEMBER_FLAG_MODERATOR)) && !plr->getSession()->CanUseCommand('a'))
272 {
273 // Player is not a moderator
274 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTMOD, m_channelName).serialise().get());
275 m_mutexChannel.unlock();
276 return;
277 }
278
279 m_mutexChannel.unlock();
280
281 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_KICKED, m_channelName, die_player->getGuid(), 0, 0, 0, std::string(), plr->getGuid()).serialise().get());
282
283 if (ban)
284 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_BANNED, m_channelName, die_player->getGuid(), 0, 0, 0, std::string(), plr->getGuid()).serialise().get());
285
286 m_mutexChannel.lock();
287
288 const auto memberFlags = itr->second;
289 m_members.erase(itr);
290 if (ban)
291 m_bannedMembers.insert(die_player->getGuidLow());
292
293 m_mutexChannel.unlock();
294
295 // Find new owner for channel
296 if (memberFlags & CHANNEL_MEMBER_FLAG_OWNER)
297 setOwner(nullptr, nullptr);
298
299 die_player->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_YOULEFT, m_channelName, 0, 0, m_channelId).serialise().get());
300}

669void Channel::setOwner(Player* plr, Player const* newOwner)
670{
671 // Here both player and new owner can be nullptr
672 if (plr != nullptr)
673 {
674 std::lock_guard<std::mutex> guard(m_mutexChannel);
675
676 const MemberMap::const_iterator itr = m_members.find(plr);
677 if (itr == m_members.end())
678 {
679 // Player is not on channel
680 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTON, m_channelName).serialise().get());
681 return;
682 }
683
684 if (!(itr->second & (CHANNEL_MEMBER_FLAG_OWNER | CHANNEL_MEMBER_FLAG_MODERATOR)) && !plr->getSession()->CanUseCommand('a'))
685 {
686 // Player is not a moderator
687 plr->sendPacket(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_NOTMOD, m_channelName).serialise().get());
688 return;
689 }
690 }
691
692 Player const* owner = nullptr;
693 uint8_t oldMemberFlags = CHANNEL_MEMBER_FLAG_NONE;
694
695 Player const* oldOwner = nullptr;
696 uint8_t oldOwnerFlags = CHANNEL_MEMBER_FLAG_NONE;
697
698 if (newOwner == nullptr)
699 {
700 std::lock_guard<std::mutex> guard(m_mutexChannel);
701
702 // Find new random owner
703 for (auto& member : m_members)
704 {
705 if (member.second & CHANNEL_MEMBER_FLAG_OWNER)
706 {
707 // Remove the old owner
708 oldOwner = member.first;
709 oldOwnerFlags = member.second;
710 member.second &= ~CHANNEL_MEMBER_FLAG_OWNER;
711 }
712 else if (owner == nullptr)
713 {
714 owner = member.first;
715 oldMemberFlags = member.second;
716 member.second |= CHANNEL_MEMBER_FLAG_OWNER;
717 }
718 }
719 }
720 else
721 {
722 std::lock_guard<std::mutex> guard(m_mutexChannel);
723
724 // Set newOwner to owner
725 for (auto& member : m_members)
726 {
727 if (member.second & CHANNEL_MEMBER_FLAG_OWNER)
728 {
729 // Remove the old owner
730 oldOwner = member.first;
731 oldOwnerFlags = member.second;
732 member.second &= ~CHANNEL_MEMBER_FLAG_OWNER;
733 }
734 else
735 {
736 if (newOwner == member.first)
737 {
738 owner = member.first;
739 oldMemberFlags = member.second;
740 member.second |= CHANNEL_MEMBER_FLAG_OWNER;
741 }
742 }
743 }
744 }
745
746 if (oldOwner != nullptr)
747 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_MODE_CHG, m_channelName, oldOwner->getGuid(), oldOwnerFlags, 0, (oldOwnerFlags &= ~CHANNEL_MEMBER_FLAG_OWNER)).serialise().get());
748
749 // Channel possibly empty
750 if (owner == nullptr)
751 return;
752
753 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_CHGOWNER, m_channelName, owner->getGuid()).serialise().get());
754
755 // Send the mode change
756 sendToAll(SmsgChannelNotify(CHANNEL_NOTIFY_FLAG_MODE_CHG, m_channelName, owner->getGuid(), oldMemberFlags, 0, (oldMemberFlags |= CHANNEL_MEMBER_FLAG_OWNER)).serialise().get());
757}

760{

762

766 // Player is not on channel

772 std::vector<SmsgChannelListMembers> members;

786 if (!(member.second & CHANNEL_MEMBER_FLAG_MUTED))
787 memberFlags |= CHANNEL_MEMBER_FLAG_VOICED;

794

796 memberFlags |= CHANNEL_MEMBER_FLAG_CUSTOM;
797
798 members.push_back({ member.first->getGuid(), memberFlags });
799 }
800
801 plr->sendPacket(SmsgChannelList(chatQuery, m_channelName, m_channelFlags, members).serialise().get());
802}
