62bool ChatCommandHandler::HandleCharLevelUpCommand(const char* args, WorldSession* m_session)
63{
64 uint32_t levels = atoi(args);
65
66 if (levels == 0)
67 {
68 redSystemMessage(m_session, "Command must be in format: .character levelup <level>.");
69 redSystemMessage(m_session, "A 0 level is not allowed.");
70 return true;
71 }
72
73 auto player_target = GetSelectedPlayer(m_session, true, true);
74 if (player_target == nullptr)
75 return true;
76
77 sGMLog.writefromsession(m_session, "used level up command on %s, with %u levels", player_target->getName().c_str(), levels);

79 levels += player_target->getLevel();

81 if (levels > worldConfig.player.playerLevelCap)
82 levels = worldConfig.player.playerLevelCap;
83
84 auto level_info = sObjectMgr.getLevelInfo(player_target->getRace(), player_target->getClass(), levels);
85 if (level_info == nullptr)
86 {
87 redSystemMessage(m_session, "No LevelInfo for Leve: {}, Race: {}, Class: {}", levels, player_target->getRace(), player_target->getClass());
88 return true;
89 }
90
91 player_target->applyLevelInfo(levels);
92
93 if (player_target != m_session->GetPlayer())
94 {
95 blueSystemMessage(m_session, "{} leveled up to level: {}", player_target->getName().c_str(), levels);
96 blueSystemMessage(player_target->getSession(), "{} leveled you up to {}.", m_session->GetPlayer()->getName().c_str(), levels);
97 sGMLog.writefromsession(m_session, "leveled player %s to level %u", player_target->getName().c_str(), levels);
98 }
99 else
100 {
101 blueSystemMessage(m_session, "You leveled yourself to {}", levels);
102 }
103
104 player_target->sendFriendStatus(true);
105
106 return true;
107}

109int32_t ChatCommandHandler::getSpellIDFromLink(const char* spelllink)
110{
111 if (spelllink == NULL)
112 return 0;
113

123uint16_t ChatCommandHandler::getItemIDFromLink(const char* itemlink, uint32_t* itemid)
124{
125 if (itemlink == NULL)
126 {
127 *itemid = 0;
128 return 0;
129 }
130 uint16_t slen = (uint16_t)strlen(itemlink);
131 const char* ptr = strstr(itemlink, "|Hitem:");
132 if (ptr == NULL)
133 {
134 *itemid = 0;
135 return slen;
136 }
137 ptr += 7; // item id is just past "|Hitem:" (7 bytes)
138 *itemid = atoi(ptr);
139 ptr = strstr(itemlink, "|r"); // the end of the item link
140 if (ptr == NULL) // item link was invalid
141 {
142 *itemid = 0;
143 return slen;
144 }
145 ptr += 2;
146 return (ptr - itemlink) & 0x0000ffff;
147}

183{
184 uint16_t skill;
185 uint16_t min;
186 uint16_t max;
187
188 if (sscanf(args, "%hu %hu %hu", &skill, &min, &max) < 1)
189 {
190 redSystemMessage(m_session, "Command must be at least in format: .character learnskill <skillid>.");

194
195 if (min == 0)

197
198 if (max == 0)

201 auto player_target = GetSelectedPlayer(m_session, true, true);

203 return true;

206 {

209 else

211
212 return true;
213 }
214
215 player_target->addSkillLine(skill, min, max);
216
217 if (player_target == m_session->GetPlayer())
218 {
219 blueSystemMessage(m_session, "Adding skill line {}", skill);
220 }
221 else
222 {
223 systemMessage(player_target->getSession(), "{} taught you skill line {}.", m_session->GetPlayer()->getName(), skill);
224 blueSystemMessage(m_session, "Skill line {} added to player: {}", skill, player_target->getName());
225 sGMLog.writefromsession(m_session, "used add skill of %u %u %u on %s", skill, min, max, player_target->getName().c_str());
226 }
227
228 return true;
229}
