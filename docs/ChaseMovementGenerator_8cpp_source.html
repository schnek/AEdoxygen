25static bool isMutualChase(Unit* owner, Unit* target)
26{
27 if (target->getMovementManager()->getCurrentMovementGeneratorType() != CHASE_MOTION_TYPE)
28 return false;
29
30 if (ChaseMovementGenerator* movement = dynamic_cast<ChaseMovementGenerator*>(target->getMovementManager()->getCurrentMovementGenerator()))
31 return movement->getTarget() == owner;
32
33 return false;
34}

36static bool positionOkay(Unit* owner, Unit* target, Optional<float> minDistance, Optional<float> maxDistance, Optional<ChaseAngle> angle)
37{
38 float const distSq = owner->GetDistance2dSq(target);
39 if (minDistance && distSq < Util::square(*minDistance))
40 return false;
41 if (maxDistance && distSq > Util::square(*maxDistance))
42 return false;
43 if (angle && !angle->isAngleOkay(target->getRelativeAngle(owner)))
44 return false;
45 // Objects outside of Line of Sight cannot be detected
46 if (worldConfig.terrainCollision.isCollisionEnabled)
47 {
48 if (!owner->IsWithinLOSInMap(target))
49 return false;
50 }
51 return true;
52}

90bool ChaseMovementGenerator::update(Unit* owner, uint32_t diff)
91{
92 // owner might be dead or gone (can we even get nullptr here?)
93 if (!owner || !owner->isAlive())
94 return false;
95
96 // our target might have gone away
97 Unit* const target = getTarget();
98 if (!target || !target->IsInWorld())
99 return false;
100
101 // the owner might be unable to move (rooted or casting), or we have lost the target, pause movement
102 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell() || hasLostTarget(owner, target))
103 {
104 owner->stopMoving();
105 _lastTargetPosition.reset();
106 if (Creature* cOwner = owner->ToCreature())
107 cOwner->getAIInterface()->setCannotReachTarget(false);
108 return true;
109 }
110
111 bool const mutualChase = isMutualChase(owner, target);
112 float const hitboxSum = owner->getCombatReach() + target->getCombatReach();
113 float const minRange = _range ? _range->MinRange + hitboxSum : CONTACT_DISTANCE;
114 float const minTarget = (_range ? _range->MinTolerance : 0.0f) + hitboxSum;
115 float const maxRange = _range ? _range->MaxRange + hitboxSum : owner->getMeleeRange(target); // melee range already includes hitboxes
116 float const maxTarget = _range ? _range->MaxTolerance + hitboxSum : CONTACT_DISTANCE + hitboxSum;
117 Optional<ChaseAngle> angle = mutualChase ? Optional<ChaseAngle>() : _angle;
118
119 // periodically check if we're already in the expected range...
120 _rangeCheckTimer->updateTimer(diff);
121 if (_rangeCheckTimer->isTimePassed())
122 {
123 _rangeCheckTimer->resetInterval(RANGE_CHECK_INTERVAL);
124 if (hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED) && positionOkay(owner, target, _movingTowards ? Optional<float>() : minTarget, _movingTowards ? maxTarget : Optional<float>(), angle))
125 {
126 removeFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
127 _path = nullptr;
128 if (Creature* cOwner = owner->ToCreature())
129 cOwner->getAIInterface()->setCannotReachTarget(false);
130 owner->stopMoving();
131 owner->setInFront(target);
132 doMovementInform(owner, target);
133 return true;
134 }
135 }
136
137 // if we're done moving, we want to clean up
138 if (owner->hasUnitStateFlag(UNIT_STATE_CHASE_MOVE) && owner->movespline->Finalized())
139 {
140 removeFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
141 _path = nullptr;
142 if (Creature* cOwner = owner->ToCreature())
143 cOwner->getAIInterface()->setCannotReachTarget(false);
144 owner->removeUnitStateFlag(UNIT_STATE_CHASE_MOVE);
145 owner->setInFront(target);
146 doMovementInform(owner, target);
147 }
148
149 // if the target moved, we have to consider whether to adjust
150 if (!_lastTargetPosition || target->GetPosition() != _lastTargetPosition.value() || mutualChase != _mutualChase)
151 {
152 _lastTargetPosition = target->GetPosition();
153 _mutualChase = mutualChase;
154 if (owner->hasUnitStateFlag(UNIT_STATE_CHASE_MOVE) || !positionOkay(owner, target, minRange, maxRange, angle))
155 {
156 Creature* const cOwner = owner->ToCreature();
157 // can we get to the target?
158 if (cOwner && !target->isInAccessiblePlaceFor(cOwner))
159 {
160 cOwner->getAIInterface()->setCannotReachTarget(true);
161 cOwner->stopMoving();
162 _path = nullptr;
163 return true;
164 }
165
166 // figure out which way we want to move
167 bool const moveToward = !owner->isInDist(target, maxRange);
168
169 // make a new path if we have to...
170 if (!_path || moveToward != _movingTowards)
171 _path = std::make_unique<PathGenerator>(owner);
172
173 float x, y, z;
174 bool shortenPath;
175 // if we want to move toward the target and there's no fixed angle...
176 if (moveToward && !angle)
177 {
178 // ...we'll pathfind to the center, then shorten the path
179 target->getPosition(x, y, z);
180 shortenPath = true;
181 }
182 else
183 {
184 // otherwise, we fall back to nearpoint finding
185 target->getNearPoint(owner, x, y, z, (moveToward ? maxTarget : minTarget) - hitboxSum, angle ? target->toAbsoluteAngle(angle->RelativeAngle) : target->getAbsoluteAngle(owner));
186 shortenPath = false;
187 }
188
189 if (owner->isHovering())
190 owner->updateAllowedPositionZ(x, y, z);
191
192 bool forcedest = owner->canFly() || owner->isInWater();
193
194 bool success = _path->calculatePath(x, y, z, forcedest);
195 if (!success || (_path->getPathType() & (PATHFIND_NOPATH /* | PATHFIND_INCOMPLETE*/)))
196 {
197 if (cOwner)
198 cOwner->getAIInterface()->setCannotReachTarget(true);
199 owner->stopMoving();
200 return true;
201 }
202
203 if (shortenPath)
204 _path->shortenPathUntilDist(positionToVector3(target->GetPosition()), maxTarget);
205
206 if (cOwner)
207 cOwner->getAIInterface()->setCannotReachTarget(false);
208
209 bool walk = false;
210 if (cOwner && !cOwner->isPet())
211 {
212 switch (cOwner->getMovementTemplate().getChase())
213 {
214 case CreatureChaseMovementType::CanWalk:
215 walk = owner->isWalking();
216 break;
217 case CreatureChaseMovementType::AlwaysWalk:
218 walk = true;
219 break;
220 default:
221 break;
222 }
223 }
224
225 owner->addUnitStateFlag(UNIT_STATE_CHASE_MOVE);
226 addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
227
228 MovementMgr::MoveSplineInit init(owner);
229 init.MovebyPath(_path->getPath());
230 init.SetWalk(walk);
231 init.SetFacing(target);
232 init.Launch();
233 }
234 }
235
236 // and then, finally, we're done for the tick
237 return true;
238}
