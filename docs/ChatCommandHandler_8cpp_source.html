63std::optional<std::string_view> ChatCommandHandler::normalizeCommandInput(const char* raw)
64{
65 if (!raw)
66 return std::nullopt;
67
68 // skip leading whitespace
69 const char* p = raw;
70 while (*p && std::isspace(static_cast<unsigned char>(*p)))
71 ++p;
72
73 // must start with '.' or '!'
74 if (*p != '.' && *p != '!')
75 return std::nullopt;
76
77 // reject ".."
78 if (p[1] == '.')
79 return std::nullopt;
80
81 // skip the sigil
82 ++p;
83
84 // spaces after sigil
85 while (*p && std::isspace(static_cast<unsigned char>(*p)))
86 ++p;
87
88 if (*p == '\0')
89 return std::nullopt;
90
91 // build string_view to the end
92 return std::string_view{ p, std::strlen(p) };
93}

95// Resolve the 1st token to a unique top-level command word (case-insensitive, abbreviation).
96// Returns true and writes 'outTop' on unique match (with permission), otherwise false.
97// Top-level == registry entries whose command has exactly one word.
98bool ChatCommandHandler::resolveTopLevelAbbrev(std::string_view tok0, WorldSession* s, std::string& outTop) const
99{
100 auto &reg = sCommandTableStorage.getCommandRegistry();
101
102 auto lo = [](char c){
103 return static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
104 };
105 auto istarts_with = [&](std::string_view a, std::string_view b){
106 if (b.size() > a.size())
107 return false;
108
109 for (size_t i=0; i<b.size(); ++i)
110 if (lo(a[i]) != lo(b[i]))
111 return false;
112
113 return true;
114 };
115
116 std::vector<std::string> matches;
117
118 for (const auto& e : reg)
119 {
120 if (e.command.empty())
121 continue;
122
123 // split command into words (top-level == exactly 1 word)
124 std::istringstream is(e.command);
125 std::string w0, more;
126 if (!(is >> w0) || (is >> more))
127 continue; // not exactly one word
128
129 const char perm = e.commandPermission.empty() ? '0' : e.commandPermission[0];
130 if (perm != '0' && !s->CanUseCommand(perm))
131 continue;
132
133 // exact (case-insensitive) wins immediately
134 if (w0.size() == tok0.size())
135 {
136 bool eq = true;
137 for (size_t i=0; i<w0.size(); ++i)
138 {
139 if (lo(w0[i]) != lo(tok0[i]))
140 {
141 eq = false;
142 break;
143 }
144 }
145
146 if (eq)
147 {
148 outTop = w0;
149 return true;
150 }
151 }
152
153 // abbreviation candidate
154 if (istarts_with(w0, tok0))
155 matches.push_back(std::move(w0));
156 }
157
158 if (matches.size() == 1)
159 {
160 outTop = matches[0];
161 return true;
162 }
163
164 return false; // unknown or ambiguous
165}

167bool ChatCommandHandler::executeCommandFlat(std::string_view text, WorldSession* m_session)
168{
169 if (text.empty())
170 return false;
171
172 // split input into tokens (views)
173 std::vector<std::string_view> tokens;
174 {
175 size_t p = 0;
176 while (p < text.size())
177 {
178 while (p < text.size() && std::isspace(static_cast<unsigned char>(text[p])))
179 ++p;
180
181 if (p >= text.size())
182 break;
183
184 size_t q = p;
185 while (q < text.size() && !std::isspace(static_cast<unsigned char>(text[q])))
186 ++q;
187
188 tokens.emplace_back(text.substr(p, q - p));
189 p = q;
190 }
191 }
192 if (tokens.empty())
193 return false;
194
195 // resolve only the first token to a top-level command
196 std::string top;
197 if (!resolveTopLevelAbbrev(tokens[0], m_session, top))
198 return false; // unknown or ambiguous top-level
199
200 auto lo = [](char c){
201 return static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
202 };
203 auto istarts_with = [&](std::string_view a, std::string_view b){
204 if (b.size() > a.size())
205 return false;
206
207 for (size_t i=0; i<b.size(); ++i)
208 if (lo(a[i]) != lo(b[i]))
209 return false;
210
211 return true;
212 };
213
214 // find deepest matching command (multi-token abbrev)
215 const ChatCommand* chosen = nullptr;
216 std::string chosenCmd;
217 size_t matchedDepth = 0;
218
219 const auto& reg = sCommandTableStorage.getCommandRegistry();
220
221 for (const auto& e : reg)
222 {
223 if (e.command.empty())
224 continue;
225
226 // split command into words
227 std::vector<std::string> words;
228 {
229 std::istringstream is(e.command);
230 for (std::string w; is >> w; ) words.push_back(std::move(w));
231 }
232 if (words.empty())
233 continue;
234
235 // cannot be deeper than what the user typed
236 if (words.size() > tokens.size())
237 continue;
238
239 // first word must equal the resolved top-level exactly (case-insensitive)
240 if (words[0].size() != top.size()) {
241 // fast-path size check (kept to mirror your structure)
242 }
243 {
244 bool eq = true;
245 for (size_t i=0; i<top.size(); ++i)
246 {
247 if (lo(words[0][i]) != lo(top[i]))
248 {
249 eq = false;
250 break;
251 }
252 }
253 if (!eq)
254 continue;
255 }
256
257 // remaining words (if any) are matched by abbreviation
258 bool ok = true;
259 for (size_t i = 1; i < words.size(); ++i)
260 {
261 if (!istarts_with(words[i], tokens[i]))
262 {
263 ok = false;
264 break;
265 }
266 }
267
268 if (!ok)
269 continue;
270
271 const char perm = e.commandPermission.empty() ? '0' : e.commandPermission[0];
272 if (perm != '0' && !m_session->CanUseCommand(perm))
273 continue;
274
275 // prefer deepest match
276 if (words.size() > matchedDepth)
277 {
278 chosen = &e;
279 chosenCmd = e.command;
280 matchedDepth = words.size();
281 }
282 }
283
284 // If there is no matching entry, treat it as a node prefix: show its children.
285 if (!chosen)
286 {
287 // Build the prefix string: <top> + remaining typed tokens (as text),
288 // but do NOT auto-append anything else.
289 std::string prefix = top;
290 for (size_t i = 1; i < tokens.size(); ++i)
291 {
292 prefix.push_back(' ');
293 prefix.append(tokens[i].data(), tokens[i].size());
294 }
295 std::string prefixWithSpace = prefix + ' ';
296
297 // List subcommands under this prefix the user can use.
298 bool any = false;
299 for (const auto& sub : reg)
300 {
301 if (sub.command.size() <= prefixWithSpace.size())
302 continue;
303
304 // case-insensitive startswith on the whole prefix string
305 bool starts = true;
306 for (size_t k = 0; k < prefixWithSpace.size(); ++k)
307 {
308 if (lo(sub.command[k]) != lo(prefixWithSpace[k]))
309 {
310 starts = false;
311 break;
312 }
313 }
314
315 if (!starts)
316 continue;
317
318 const char p = sub.commandPermission.empty() ? '0' : sub.commandPermission[0];
319 if (p != '0' && !m_session->CanUseCommand(p))
320 continue;
321
322 auto help = sub.help.empty() ? "No Help Available" : sub.help;
323 blueSystemMessage(m_session, " {} - {}", sub.command.substr(prefixWithSpace.size()), help);
324
325 any = true;
326 }
327
328 if (!any)
329 systemMessage(m_session, "There is no help for that command");
330 else
331 greenSystemMessage(m_session, "Available Subcommands:");
332
333 return true;
334 }
335
336 // Exact/deepest entry found either node (no handler) or leaf (handler)
337
338 if (!chosen->handler)
339 {
340 if (!chosen->help.empty())
341 SendMultilineMessage(m_session, chosen->help.c_str());
342 else
343 greenSystemMessage(m_session, "Available Subcommands:");
344
345 const std::string prefix = chosenCmd + ' ';
346 bool any = false;
347
348 for (const auto& sub : reg)
349 {
350 if (sub.command.size() <= prefix.size())
351 continue;
352
353 bool starts = true;
354 for (size_t k = 0; k < prefix.size(); ++k)
355 {
356 if (lo(sub.command[k]) != lo(prefix[k]))
357 {
358 starts = false;
359 break;
360 }
361 }
362
363 if (!starts)
364 continue;
365
366 const char p = sub.commandPermission.empty() ? '0' : sub.commandPermission[0];
367 if (p != '0' && !m_session->CanUseCommand(p))
368 continue;
369
370 auto help = sub.help.empty() ? "No Help Available" : sub.help;
371 blueSystemMessage(m_session, " {} - {}", sub.command.substr(prefix.size()), help);
372 any = true;
373 }
374
375 if (!any && chosen->help.empty())
376 systemMessage(m_session, "There is no help for that command");
377
378 return true;
379 }
380
381 // pass "args" = everything after the matchedDepth words (not after all tokens!)
382 std::string_view args = text;
383 {
384 size_t consumed = 0, seen = 0;
385 while (consumed < text.size() && seen < matchedDepth)
386 {
387 while (consumed < text.size() && std::isspace(static_cast<unsigned char>(text[consumed])))
388 ++consumed;
389
390 while (consumed < text.size() && !std::isspace(static_cast<unsigned char>(text[consumed])))
391 ++consumed;
392
393 ++seen;
394 }
395 while (consumed < text.size() && std::isspace(static_cast<unsigned char>(text[consumed])))
396 ++consumed;
397 args = text.substr(consumed);
398 }
399
400 // check if min argument count was reached
401 const size_t argc = countWords(args);
402
403 if (argc < chosen->minArgCount)
404 {
405 if (!chosen->help.empty())
406 SendMultilineMessage(m_session, chosen->help.c_str());
407 else
408 redSystemMessage(m_session, "Incorrect syntax specified. Try .help %s for the correct syntax.", chosenCmd.c_str());
409
410 return true;
411 }
412
413 // actually execute the command
414 const bool ok = chosen->handler(this, args, m_session);
415 if (!ok)
416 {
417 if (!chosen->help.empty())
418 SendMultilineMessage(m_session, chosen->help.c_str());
419 else
420 redSystemMessage(m_session, "Incorrect syntax specified. Try .help %s for the correct syntax.", chosenCmd.c_str());
421 }
422
423 return true;
424}

431int ChatCommandHandler::ParseCommands(const char* text, WorldSession* session)
432{
433 if (!session)
434 return 0;
435
436 if (!*text)
437 return 0;
438
439 if (!session->HasGMPermissions() && worldConfig.server.requireGmForCommands)
440 return 0;
441
442 auto normalized = normalizeCommandInput(text);
443 if (!normalized)
444 return 0;
445
446 try
447 {
448 bool success = executeCommand(*normalized, session);
449 if (!success)
450 {
451 systemMessage(session, "There is no such command, or you do not have access to it.");
452 }
453 }
454 catch (AscEmu::Exception::PlayerNotFoundException& e)
455 {
456 // TODO: Handle this properly (what do we do when we're running commands with no player object?)
457 sLogger.failure("PlayerNotFoundException occurred when processing command [{}]. Exception: {}", text, e.AEwhat());
458 }
459
460 return 1;
461}

463Player* ChatCommandHandler::GetSelectedPlayer(WorldSession* m_session, bool showerror, bool auto_self)
464{
465 if (m_session == nullptr)
466 return nullptr;
467
468 bool is_creature = false;
469 Player* player_target = nullptr;
470 uint64_t guid = m_session->GetPlayer()->getTargetGuid();
471
472 WoWGuid wowGuid;
473 wowGuid.Init(guid);
474
475 switch (wowGuid.getHigh())
476 {
477 case HighGuid::Pet:
478 case HighGuid::Unit:
479 case HighGuid::Vehicle:
480 {
481 is_creature = true;
482 break;
483 }
484 default:
485 break;
486 }
487
488 if (guid == 0 || is_creature)
489 {
490 if (auto_self)
491 {
492 m_session->systemMessage("Auto-targeting self.");
493 player_target = m_session->GetPlayer();
494 }
495 else
496 {
497 if (showerror)
498 m_session->systemMessage("This command requires a selected player.");
499
500 return nullptr;
501 }
502 }
503 else
504 {
505 player_target = m_session->GetPlayer()->getWorldMap()->getPlayer((uint32_t)guid);
506 }
507
508 return player_target;
509}

641void ChatCommandHandler::SendHighlightedName(WorldSession* m_session, const char* prefix, const char* full_name, std::string & lowercase_name, std::string & highlight, uint32_t id)
642{
643 char message[1024];
644 char start[50];
645 start[0] = 0;
646 message[0] = 0;
647
648 snprintf(start, 50, "%s %u: %s", prefix, id, MSG_COLOR_WHITE);
649
650 auto highlight_length = highlight.length();
651 std::string fullname = std::string(full_name);
652 size_t offset = lowercase_name.find(highlight);
653 auto remaining = fullname.size() - offset - highlight_length;
654
655 strcat(message, start);
656 strncat(message, fullname.c_str(), offset);
657 strcat(message, MSG_COLOR_LIGHTRED);
658 strncat(message, (fullname.c_str() + offset), highlight_length);
659 strcat(message, MSG_COLOR_WHITE);
660 if (remaining > 0)
661 strncat(message, (fullname.c_str() + offset + highlight_length), remaining);
662
663 systemMessage(m_session, message);
