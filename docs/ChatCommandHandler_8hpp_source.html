45 // Variadic template version of systemMessage
46 template<typename... Args>
47 void systemMessage(WorldSession* _session, const std::string& _format, Args&&... _args)
48 {
49 // Use the custom StringFormat function to format the string
50 std::string formattedMessage = AscEmu::StringFormat(_format, std::forward<Args>(_args)...);
51
52 // Send the formatted message via packet
53 sendSystemMessagePacket(_session, formattedMessage);
54 }
55
56 // Variadic template version of redSystemMessage
57 template<typename... Args>
58 void colorSystemMessage(WorldSession* _session, const std::string _colorCode, const std::string& _format, Args&&... _args)
59 {
60 // Use the custom StringFormat function to format the string
61 const std::string formattedMessage = AscEmu::StringFormat(_format, std::forward<Args>(_args)...);
62 std::string coloredMessage = _colorCode + formattedMessage + "|r";
63
64 // Send the formatted message via packet
65 sendSystemMessagePacket(_session, coloredMessage);
66 }
67
68 // Variadic template version of redSystemMessage
69 template<typename... Args>
70 void redSystemMessage(WorldSession* _session, const std::string& _format, Args&&... _args)
71 {
72 colorSystemMessage(_session, MSG_COLOR_LIGHTRED, _format, std::forward<Args>(_args)...);
73 }
