48inline size_t countWords(const char* argz)
49{
50 if (!argz)
51 return 0;
52
53 size_t count = 0;
54 bool inWord = false;
55
56 for (const char* p = argz; *p; ++p)
57 {
58 if (std::isspace(static_cast<unsigned char>(*p)))
59 {
60 inWord = false;
61 }
62 else if (!inWord)
63 {
64 inWord = true;
65 ++count;
66 }
67 }
