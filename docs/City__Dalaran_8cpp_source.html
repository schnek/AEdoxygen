13class SilverCovenantMageGuard : public CreatureAIScript
14{
15public:
16 static CreatureAIScript* Create(Creature* c) { return new SilverCovenantMageGuard(c); }
17 explicit SilverCovenantMageGuard(Creature* pCreature) : CreatureAIScript(pCreature)
18 {
19 getCreature()->addUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
20 RegisterAIUpdateEvent(1500);
21 }
22
23 void StartDefense()
24 {
25 Player* player = getNearestPlayer();
26 if (player == nullptr)
27 return;
28
29 //Don't do anything with alliance players
30 if (player->isTeamAlliance())
31 return;
32
33 float player_x = player->GetPositionX();
34 float player_y = player->GetPositionY();
35 float player_z = player->GetPositionZ();
36
37 //hardcoded values...
38 // the guards should cast the spell if someone is behind them...
39 if (player_x < 5761.9f && player_x >5738.68f && player_y < 732.12f && player_y >712.09f && player_z > 635.0f)
40 {
41 getCreature()->setTargetGuid(player->getGuid());
42 getCreature()->eventCastSpell(player, sSpellMgr.getSpellInfo(54028));
43 }
44 else
45 {
46 getCreature()->setTargetGuid(0); //Reset target... ugly
47 }
48 }
49
50 void AIUpdate() override
51 {
52 if (getCreature()->getInRangePlayersCount() > 0)
53 StartDefense();
54 }
55};
