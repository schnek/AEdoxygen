32template<class T>
33void ConfusedMovementGenerator<T>::doInitialize(T* owner)
34{
35 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING | MOVEMENTGENERATOR_FLAG_TRANSITORY | MOVEMENTGENERATOR_FLAG_DEACTIVATED);
36 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED);
37
38 if (!owner || !owner->isAlive())
39 return;
40
41 // TODO: UNIT_FIELD_FLAGS should not be handled by generators
42 owner->addUnitFlags(UNIT_FLAG_CONFUSED);
43 owner->stopMoving();
44
45 _timer->resetInterval(0);
46 owner->getPosition(_x, _y, _z);
47 _path = nullptr;
48}

58template<class T>
59bool ConfusedMovementGenerator<T>::doUpdate(T* owner, uint32_t diff)
60{
61 if (!owner || !owner->isAlive())
62 return false;
63
64 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
65 {
66 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
67 owner->stopMoving();
68 _path = nullptr;
69 return true;
70 }
71 else
72 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
73
74 // waiting for next move
75 _timer->updateTimer(diff);
76 if ((MovementGenerator::hasFlag(MOVEMENTGENERATOR_FLAG_SPEED_UPDATE_PENDING) && !owner->movespline->Finalized()) || (_timer->isTimePassed() && owner->movespline->Finalized()))
77 {
78 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_TRANSITORY);
79
80 LocationVector destination(_x, _y, _z);
81 float distance = 4.0f * Util::getRandomFloat(0.0f, 1.0f) - 2.0f;
82 float angle = Util::getRandomFloat(0.0f, 1.0f) * float(M_PI) * 2.0f;
83 owner->movePositionToFirstCollision(destination, distance, angle);
84
85 // Check if the destination is in LOS
86 if (!owner->IsWithinLOS(destination))
87 {
88 // Retry later on
89 _timer->resetInterval(200);
90 return true;
91 }
92
93 if (!_path)
94 {
95 _path = std::make_unique<PathGenerator>(owner);
96 _path->setPathLengthLimit(30.0f);
97 }
98
99 bool result = _path->calculatePath(destination.getPositionX(), destination.getPositionY(), destination.getPositionZ());
100 if (!result || (_path->getPathType() & PATHFIND_NOPATH)
101 || (_path->getPathType() & PATHFIND_SHORTCUT)
102 || (_path->getPathType() & PATHFIND_FARFROMPOLY))
103 {
104 _timer->resetInterval(100);
105 return true;
106 }
107
108 owner->addUnitStateFlag(UNIT_STATE_CONFUSED_MOVE);
109
110 MovementMgr::MoveSplineInit init(owner);
111 init.MovebyPath(_path->getPath());
112 init.SetWalk(true);
113 int32_t traveltime = init.Launch();
114 _timer->resetInterval(std::max(300u, Util::getRandomUInt(traveltime / 2, traveltime * 2)));
115 }
116
117 return true;
118}
