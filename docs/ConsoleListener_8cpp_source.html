38{

63bool StartConsoleListener()
64{
65 if (worldConfig.remoteConsole.isEnabled == false)
66 {
67 return false;
68 }
69
70 std::string consoleListenHost = worldConfig.remoteConsole.host;
71 uint32_t consoleListenPort = worldConfig.remoteConsole.port;
72
73 g_pListenSocket = std::make_unique<ListenSocket<ConsoleSocket>>(consoleListenHost.c_str(), consoleListenPort);
74
75 if (g_pListenSocket->IsOpen() == false)
76 {
77 g_pListenSocket->Close();
78 g_pListenSocket = nullptr;
79
80 return false;
81 }
82
83 sConsoleAuthMgr.initialize();
84 return true;
85}

99void RemoteConsole::Write(const char* Format, ...)
100{
101 char obuf[65536];
102 va_list ap;
103
104 va_start(ap, Format);
105 vsnprintf(obuf, 65536, Format, ap);
106 va_end(ap);
107
108 if (*obuf == '\0')
109 return;
110
111 m_pSocket->Send((const uint8_t*)obuf, (uint32_t)strlen(obuf));
112}

156void processConsoleInput(BaseConsole* baseConsole, std::string consoleInput, bool isWebClient)
157{
158 ConsoleCommand commandList;
159
160 std::stringstream consoleInputStream(consoleInput);
161
162 std::string commandName;
163 std::string commandVars;
164
165 consoleInputStream >> commandName;
166 std::getline(consoleInputStream, commandVars);
167
168 bool commandFound = false;
169 bool isHelpCommand = false;
170 for (int i = 0; Commands[i].consoleCommand.empty() == false; ++i)
171 {
172 if (commandName.empty())
173 break;
174
175 if (commandName.compare("Help") == 0 || commandName.compare("help") == 0 || commandName.compare("?") == 0)
176 {
177 isHelpCommand = true;
178 break;
179 }
180
181 if (Commands[i].consoleCommand.compare(commandName) == 0)
182 {
183 commandFound = true;
184 commandList = Commands[i];
185 break;
186 }
187 }
188
189 if (isHelpCommand)
190 {
191 if (isWebClient == false)
192 {
193 baseConsole->Write("================================================================================\r\n");
194 baseConsole->Write("| Console::Help |\r\n");
195 baseConsole->Write("================================================================================\r\n");
196 baseConsole->Write("| %20s | %53s |\r\n", "Name", "Arguments");
197 baseConsole->Write("================================================================================\r\n");
198
199 for (int j = 0; Commands[j].consoleCommand.empty() == false; ++j)
200 {
201 baseConsole->Write("| %20s | %53s |\r\n", Commands[j].consoleCommand.c_str(), Commands[j].argumentFormat.c_str());
202 }
203
204 baseConsole->Write("================================================================================\r\n");
205 baseConsole->Write("| type 'quit' to terminate a Remote Console Session |\r\n");
206 baseConsole->Write("================================================================================\r\n");
207 }
208 }
209 else
210 {
211 if (commandFound)
212 {
213 if (isWebClient == false)
214 {
215 baseConsole->Write("Received command: %s\r\n", commandName.c_str());
216 }
217
218 if (commandList.argumentCount > 0 && commandVars.empty() == false)
219 {
220 if (isWebClient == false)
221 {
222 baseConsole->Write("Received vars: %s\r\n", commandVars.c_str());
223 }
224 }
225
226 if (!commandList.CommandPointer(baseConsole, commandList.argumentCount, commandVars, isWebClient))
227 {
228 if (isWebClient == false)
229 {
230 baseConsole->Write("[!]Error, '%s' used an incorrect syntax, the correct syntax is: '%s'.\r\n\r\n",
231 commandList.consoleCommand.c_str(), commandList.argumentFormat.c_str());
232 }
233 }
234 }
235 else
236 {
237 if (isWebClient == false)
238 {
239 baseConsole->Write("[!]Error, Command '%s' doesn't exist. Type '?' or 'help' to get a command overview.\r\n\r\n",
240 commandName.c_str());
241 }
242 }
243 }
244}
