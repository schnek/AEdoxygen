63void ConsoleSocket::handleConsoleInput()
64{
65 uint32_t readLength = (uint32_t)readBuffer.GetSize();
66 if ((readLength + mInputBufferPosition) >= mInputBufferLength)
67 {
68 Disconnect();
69 return;
70 }
71
72 readBuffer.Read(&mInputBuffer[mInputBufferPosition], readLength);
73 mInputBufferPosition += readLength;
74
75 char* inputChar = strchr(mInputBuffer.get(), '\n');
76 while (inputChar != NULL)
77 {
78 uint32_t inputLength = (uint32_t)((inputChar + 1) - mInputBuffer.get());
79 if (*(inputChar - 1) == '\r')
80 {
81 *(inputChar - 1) = '\0';
82 }
83
84 *inputChar = '\0';
85
86 if (mInputBuffer[0] != '\0')
87 {
88 switch (mConsoleSocketState)
89 {
90 case ConsoleDefines::RemoteConsoleState::WaitForUsername:
91 {
92 mConsoleAuthName = std::string(mInputBuffer.get());
93 mRemoteConsole->Write("password: ");
94 mConsoleSocketState = ConsoleDefines::RemoteConsoleState::WaitForPassword;
95
96 } break;
97 case ConsoleDefines::RemoteConsoleState::WaitForPassword:
98 {
99 mConsoleAuthPassword = std::string(mInputBuffer.get());
100 mRemoteConsole->Write("\r\nAttempting to authenticate. Please wait.\r\n");
101
102 mRequestId = sConsoleAuthMgr.getGeneratedId();
103 sConsoleAuthMgr.addRequestIdSocket(mRequestId, this);
104
105 testConsoleLogin(mConsoleAuthName, mConsoleAuthPassword, mRequestId);
106
107 } break;
108 case ConsoleDefines::RemoteConsoleState::UserLoggedIn:
109 {
110 if (AscEmu::Util::Strings::isEqual(mInputBuffer.get(), "quit"))
111 {
112 Disconnect();
113 break;
114 }
115
116 if (AscEmu::Util::Strings::isEqual(mInputBuffer.get(), "webclient"))
117 {
118 isWebClient = true;
119 break;
120 }
121 }
122 default:
123 {
124 processConsoleInput(mRemoteConsole.get(), mInputBuffer.get(), isWebClient);
125
126 } break;
127 }
128 }
129
130 if (inputLength == mInputBufferPosition)
131 {
132 mInputBuffer[0] = '\0';
133 mInputBufferPosition = 0;
134 }
135 else
136 {
137 memcpy(mInputBuffer.get(), &mInputBuffer[inputLength], mInputBufferPosition - inputLength);
138 mInputBufferPosition -= inputLength;
139 }
140
141 inputChar = strchr(mInputBuffer.get(), '\n');
142 }
143}
