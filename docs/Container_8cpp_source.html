106std::tuple<bool, std::unique_ptr<Item>> Container::addItem(int16_t slot, std::unique_ptr<Item> itemHolder)
107{
108 if (slot < 0 || (uint32_t)slot >= getItemProperties()->ContainerSlots)
109 return { false, std::move(itemHolder) };
110
111 if (m_Slot[slot])
112 {
113 sLogger.failure("Container::addItem: Bad container item {} slot {}", itemHolder->getGuidLow(), slot);
114 return { false, std::move(itemHolder) };
115 }
116
117 if (!m_owner)
118 return { false, std::move(itemHolder) };
119
120 auto* item = itemHolder.get();
121 m_Slot[slot] = std::move(itemHolder);
122 item->m_isDirty = true;
123
124 item->setContainer(this);
125 item->setOwner(m_owner);
126
127 if (item->getItemProperties()->Bonding == ITEM_BIND_ON_PICKUP)
128 {
129 if (item->getItemProperties()->Flags & ITEM_FLAG_ACCOUNTBOUND)
130 item->addFlags(ITEM_FLAG_ACCOUNTBOUND);
131 else
132 item->addFlags(ITEM_FLAG_SOULBOUND);
133 }
134
135 setSlot(slot, item->getGuid());
136
137 forceCreationUpdate(item);
138
139 return { true, nullptr };
140}

212void Container::swapItems(int8_t SrcSlot, int8_t DstSlot)
213{
214 if (SrcSlot < 0 || SrcSlot >= static_cast<int8_t>(m_itemProperties->ContainerSlots))
215 return;
216
217 if (DstSlot < 0 || DstSlot >= static_cast<int8_t>(m_itemProperties->ContainerSlots))
218 return;
219
220 uint32_t destMaxCount = (m_owner->m_cheats.hasItemStackCheat) ? 0x7fffffff : ((m_Slot[DstSlot]) ? m_Slot[DstSlot]->getItemProperties()->MaxCount : 0);
221 if (m_Slot[DstSlot] && m_Slot[SrcSlot] && m_Slot[DstSlot]->getEntry() == m_Slot[SrcSlot]->getEntry() && m_Slot[SrcSlot]->m_wrappedItemId == 0 && m_Slot[DstSlot]->m_wrappedItemId == 0 && destMaxCount > 1)
222 {
223 uint32_t total = m_Slot[SrcSlot]->getStackCount() + m_Slot[DstSlot]->getStackCount();
224 m_Slot[DstSlot]->m_isDirty = m_Slot[SrcSlot]->m_isDirty = true;
225 if (total <= destMaxCount)
226 {
227 m_Slot[DstSlot]->modStackCount(m_Slot[SrcSlot]->getStackCount());
228 safeFullRemoveItemFromSlot(SrcSlot);
229 return;
230 }
231 else
232 {
233 if (m_Slot[DstSlot]->getStackCount() != destMaxCount)
234 {
235 int32_t delta = destMaxCount - m_Slot[DstSlot]->getStackCount();
236 m_Slot[DstSlot]->setStackCount(destMaxCount);
237 m_Slot[SrcSlot]->modStackCount(-delta);
238 return;
239 }
240 }
241 }
242
243 std::unique_ptr<Item> temp = std::move(m_Slot[SrcSlot]);
244 m_Slot[SrcSlot] = std::move(m_Slot[DstSlot]);
245 m_Slot[DstSlot] = std::move(temp);
246
247 if (m_Slot[DstSlot])
248 {
249 setSlot(DstSlot, m_Slot[DstSlot]->getGuid());
250 m_Slot[DstSlot]->m_isDirty = true;
251 }
252 else
253 {
254 setSlot(DstSlot, 0);
255 }
256
257 if (m_Slot[SrcSlot])
258 {
259 setSlot(SrcSlot, m_Slot[SrcSlot]->getGuid());
260 m_Slot[SrcSlot]->m_isDirty = true;
261 }
262 else
263 {
264 setSlot(SrcSlot, 0);
265 }
266}

268std::unique_ptr<Item> Container::safeRemoveAndRetreiveItemFromSlot(int16_t slot, bool destroy)
269{
270 if (slot < 0 || static_cast<uint32_t>(slot) >= getItemProperties()->ContainerSlots)
271 return nullptr;
272
273 Item* rawItem = m_Slot[slot].get();
274 if (!rawItem || rawItem == this)
275 return nullptr;
276
277 if (rawItem->getOwner() != m_owner)
278 return nullptr;
279
280 std::unique_ptr<Item> item = std::move(m_Slot[slot]);
281 setSlot(slot, 0);
282
283 item->setContainer(nullptr);
284
285 if (destroy)
286 {
287 if (item->IsInWorld())
288 item->removeFromWorld();
289
290 item->deleteFromDB();
291 }
292
293 return item;
294}
