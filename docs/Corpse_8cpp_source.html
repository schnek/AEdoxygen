113void Corpse::saveToDB()
114{
115 std::stringstream ss;
116 ss.rdbuf()->str("");
117 ss << "REPLACE INTO corpses (guid, positionx, positiony, positionz, orientation, zoneId, mapId, data, instanceid) VALUES ("
118 << getGuidLow()
119 << ", '"
120 << GetPositionX()
121 << "', '" << GetPositionY()
122 << "', '" << GetPositionZ()
123 << "', '" << GetOrientation()
124 << "', '" << getZoneId()
125 << "', '" << GetMapId()
126
127 << "', '";
128 ss << getGuid() << " " << getOType() << " " << getEntry() << " " << getScale() << " ";
129 ss << getOwnerGuid() << " " << getDisplayId() << " ";
130
131 for (uint8_t i = 0; i < WOWCORPSE_ITEM_COUNT; ++i)
132 ss << getItem(i) << " ";
133
134 ss << getBytes1() << " " << getBytes2() << " " << getFlags() << " " << getDynamicFlags() << " ";
135
136 ss << "', " << GetInstanceID() << ")";
137
138 CharacterDatabase.Execute(ss.str().c_str());
139}

177void Corpse::spawnBones()
178{
179 setFlags(CORPSE_FLAG_BONE | CORPSE_FLAG_UNK1);
180 setOwnerNotifyMap(0);
181
182 for (uint8_t i = 0; i < EQUIPMENT_SLOT_END; ++i)
183 if (getItem(i))
184 setItem(i, 0);
185
186 deleteFromDB();
187 setCorpseState(CORPSE_STATE_BONES);
188}
