62void StartCrashHandler()
63{
64 // Firstly, check if there is a debugger present. There isn't any point in
65 // handling crashes internally if we have a debugger attached, that would
66 // just piss us off. :P
67
68 // Check for a debugger.
69#if !(defined(__x86_64__) || defined(__x86_64) || defined(__amd64__) || defined(_WIN64))
70 DWORD code;
71
72 __asm
73 {
74 MOV EAX, FS:[0x18]
75 MOV EAX, DWORD PTR [EAX + 0x30]
76 MOV ECX, DWORD PTR [EAX]
77 MOV [DWORD PTR code], ECX
78 }
79
80 if(code & 0x00010000)
81 {
82 // We got a debugger. We'll tell it to not exit on a crash but instead break into debugger.
83 ON_CRASH_BREAK_DEBUGGER = true;
84 }
85 else
86 {
87 // No debugger. On crash, we'll call OnCrash to save etc.
88 ON_CRASH_BREAK_DEBUGGER = false;
89 }
90#else
91 ON_CRASH_BREAK_DEBUGGER = (IsDebuggerPresent() == TRUE) ? true : false;
92#endif
93
94 if(!ON_CRASH_BREAK_DEBUGGER)
95 {
96 _CrtSetReportMode( _CRT_ERROR, _CRTDBG_MODE_FILE );
97 _CrtSetReportFile( _CRT_ERROR, _CRTDBG_FILE_STDERR );
98 _CrtSetReportMode( _CRT_ASSERT, _CRTDBG_MODE_FILE );
99 _CrtSetReportFile( _CRT_ASSERT, _CRTDBG_FILE_STDERR );
100 }
101}

227void CStackWalker::OnCallstackEntry(CallstackEntryType eType, CallstackEntry & entry)
228{
229 CHAR buffer[STACKWALK_MAX_NAMELEN];
230 if((eType != lastEntry) && (entry.offset != 0))
231 {
232 if(entry.name[0] == 0)
233 strcpy(entry.name, "(function-name not available)");
234 if(entry.undName[0] != 0)
235 strcpy(entry.name, entry.undName);
236 if(entry.undFullName[0] != 0)
237 strcpy(entry.name, entry.undFullName);
238
239 char* p = strrchr(entry.loadedImageName, '\\');
240 if(!p)
241 p = entry.loadedImageName;
242 else
243 ++p;
244
245 if(entry.lineFileName[0] == 0)
246 {
247 if(entry.name[0] == 0)
248 sprintf(entry.name, "%lld", entry.offset);
249
250 sprintf(buffer, "%s!%s Line %u\n", p, entry.name, entry.lineNumber);
251 }
252 else
253 sprintf(buffer, "%s!%s Line %u\n", p, entry.name, entry.lineNumber);
254
255 OnOutput(buffer);
256 }

272int __cdecl HandleCrash(PEXCEPTION_POINTERS pExceptPtrs)
273{
274 if(pExceptPtrs == 0)
275 {
276 // Raise an exception :P
277 __try
278 {
279 RaiseException(EXCEPTION_BREAKPOINT, 0, 0, 0);
280 }
281 __except(HandleCrash(GetExceptionInformation()), EXCEPTION_CONTINUE_EXECUTION)
282 {
283 }
284 }
285
286 /* only allow one thread to crash. */
287 if(!m_crashLock.attemptAcquire())
288 {
289 TerminateThread(GetCurrentThread(), static_cast<DWORD>(-1));
290 // not reached
291 }
292
293 if(died)
294 {
295 TerminateProcess(GetCurrentProcess(), static_cast<UINT>(-1));
296 // not reached:P
297 }
298
299 died = true;
300
301 // Create the date/time string
302 time_t curtime = time(NULL);
303 tm* pTime = localtime(&curtime);
304 char filename[maxPathLength];
305 TCHAR modname[maxPathLength * 2];
306 ZeroMemory(modname, sizeof(modname));
307 if(GetModuleFileName(0, modname, maxPathLength * 2 - 2) <= 0)
308 strcpy(modname, "UNKNOWN");
309
310 char* mname = strrchr(modname, '\\');
311 (void*)mname++; // Remove the last
312
313 sprintf(filename, "CrashDumps\\dump-%s-%s-%u-%u-%u-%u-%u-%u-%u.dmp",
314 mname, AE_BUILD_HASH, pTime->tm_year + 1900, pTime->tm_mon + 1, pTime->tm_mday,
315 pTime->tm_hour, pTime->tm_min, pTime->tm_sec, GetCurrentThreadId());
316
317
318 HANDLE hDump = CreateFile(filename, GENERIC_WRITE, 0, NULL, CREATE_ALWAYS,
319 FILE_ATTRIBUTE_NORMAL | FILE_FLAG_WRITE_THROUGH, 0);
320
321 if(hDump == INVALID_HANDLE_VALUE)
322 {
323 // Create the directory first
324 CreateDirectory("CrashDumps", 0);
325 hDump = CreateFile(filename, GENERIC_WRITE, 0, NULL, CREATE_ALWAYS,
326 FILE_ATTRIBUTE_NORMAL | FILE_FLAG_WRITE_THROUGH, 0);
327 }
328
329 sLogger.failure("Server has crashed. Creating crash dump file {}", filename);
330
331 if(hDump == INVALID_HANDLE_VALUE)
332 {
333 sLogger.failure("Could not open crash dump file.");
334 }
335 else
336 {
337 MINIDUMP_EXCEPTION_INFORMATION info;
338 info.ClientPointers = FALSE;
339 info.ExceptionPointers = pExceptPtrs;
340 info.ThreadId = GetCurrentThreadId();
341
342
343 MiniDumpWriteDump(GetCurrentProcess(), GetCurrentProcessId(),
344 hDump, MiniDumpWithIndirectlyReferencedMemory, &info, 0, 0);
345
346 CloseHandle(hDump);
347 }
348
349 SetPriorityClass(GetCurrentProcess(), BELOW_NORMAL_PRIORITY_CLASS);
350 OnCrash(!ON_CRASH_BREAK_DEBUGGER);
351
352 sLogger.finalize();
353 return EXCEPTION_CONTINUE_SEARCH;
354}
