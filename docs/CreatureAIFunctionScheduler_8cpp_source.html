26CreatureAIFunc CreatureAIFunctionScheduler::addAIFunction(Function pFunction, SchedulerArgs const& pScheduler, FunctionArgs const& funcArgs)
27{
28 // Create a new shared_ptr<CreatureAIFunction> object and initialize it
29 std::shared_ptr<CreatureAIFunction> newAIFunction = std::make_shared<CreatureAIFunction>(mOwner, pFunction, pScheduler, funcArgs);
30
31 // When there is no Timer set dont add it to the Scheduler Directly execute it.
32 // This Skips all checks
33 if (!newAIFunction->getInitialCooldown().count())
34 {
35 // Add One usage
36 newAIFunction->increaseCount();
37
38 // Execute our Function
39 newAIFunction->getFunction()(newAIFunction);
40
41 return newAIFunction;
42 }
43
44 // Add the new object to the mCreatureAIFunctions list
45 mCreatureAIFunctionsTemporary.push_back(newAIFunction);
46
47 // Return the pointer to the underlying CreatureAIFunction object
48 return newAIFunction;
49}

51CreatureAIFunc CreatureAIFunctionScheduler::addAISpell(FunctionArgs funcArgs, SchedulerArgs const& pScheduler)
52{
53 // first lets check if this function already exists so we cannot add them multiple times by accident
54 for (const auto& existingFunction : mCreatureAIFunctions)
55 {
56 if (existingFunction->getFunctionArgs().getSpellId() == funcArgs.getSpellId())
57 {
58 // Return the pointer to the underlying CreatureAIFunction object
59 return existingFunction;
60 }
61 }
62
63 const auto spellInfo = sSpellMgr.getSpellInfo(funcArgs.getSpellId());
64 if (spellInfo != nullptr)
65 {
66 // Add a new Function
67 return addAIFunction(&CreatureAIScript::CreatureAIFunc_CastSpell, pScheduler, funcArgs);
68 }
69
70 sLogger.failure("tried to add invalid spell with id {}", funcArgs.getSpellId());
71
72 // Return an empty weak_ptr
73 return CreatureAIFunc();
74}

168void CreatureAIFunctionScheduler::update(unsigned long time_passed)
169{
170 // Copy all temporary functions to the scheduler
171 mCreatureAIFunctions.insert(mCreatureAIFunctions.end(), mCreatureAIFunctionsTemporary.begin(), mCreatureAIFunctionsTemporary.end());
172 mCreatureAIFunctionsTemporary.clear(); // Empty temporary function vector
173
174 if (!mCreatureAIFunctions.size())
175 return; // Nothing to do, so skip
176
177 // Randomize the order of functions
178 // Warning when 2 functions have 2 seconds timer the first always would executes before the second one...
179 // Randomizing the order could lead to unexpected behaviour when u rely on the order of insertion
180 Util::randomShuffleVector(mCreatureAIFunctions);
181
182 // Delete functions marked for removal
183 mCreatureAIFunctions.erase(std::remove_if(mCreatureAIFunctions.begin(), mCreatureAIFunctions.end(), [](const auto& function) {
184 return function->mGarbage;
185 }), mCreatureAIFunctions.end());
186
187 for (auto functions_iterator = mCreatureAIFunctions.begin(); functions_iterator != mCreatureAIFunctions.end(); ++functions_iterator)
188 {
189 auto function = *functions_iterator;
190
191 // Roll the Dice :)
192 float rolled = Util::getRandomFloat(100.0f);
193
194 if (!function->isEnabled())
195 continue; // Function is currently disabled, skip
196
197 if (function->getCombatUsage() && !mOwner->_isInCombat())
198 continue; // Function should only execute in combat, but not in combat now, skip
199
200 if (!function->isAvailableForScriptPhase(mOwner->getScriptPhase()))
201 continue; // Execution should happen in another phase, skip
202
203 if (!function->getInitialCooldown().count() && !function->isChanceMeet(0.0f))
204 continue; // Chance is 0 and no cooldown timer, assume it should execute always, but it didn't meet the chance, skip
205
206 if (!function->usesLeft())
207 {
208 function->mGarbage = true; // No uses left, mark for removal
209 continue;
210 }
211
212 if (!function->isHpInPercentRange(static_cast<float>(mOwner->_getHealthPercent())))
213 continue; // Not in HP range, skip
214
215 // We are a Spell Function so lets do some cheks dependant on that
216 if (function->getFunctionArgs().getSpellId())
217 {
218 if (mOwner->getCreature()->hasUnitStateFlag(UNIT_STATE_STUNNED | UNIT_STATE_FLEEING | UNIT_STATE_IN_FLIGHT | UNIT_STATE_CHARMED | UNIT_STATE_CONFUSED))
219 continue; // do not cast any spell while stunned/feared/silenced/charmed/confused
220
221 if (mOwner->_isCasting())
222 continue; // Creature is casting, skip
223
224 // Check if creature can cast the spell based on required Mana/Power
225 auto mSpellInfo = sSpellMgr.getSpellInfo(function->getFunctionArgs().getSpellId());
226 if (mSpellInfo->getPowerType() == POWER_TYPE_MANA)
227 {
228 if (mSpellInfo->getManaCost() > mOwner->getCreature()->getPower(POWER_TYPE_MANA))
229 continue; // Not enough Mana, skip
230 }
231 else if (mSpellInfo->getPowerType() == POWER_TYPE_FOCUS)
232 {
233 if (mSpellInfo->getManaCost() > mOwner->getCreature()->getPower(POWER_TYPE_FOCUS))
234 continue; // Not enough Focus, skip
235 }
236
237 // Aura stacking: Only cast the spell when it's under the specified amount
238 if (mOwner->getCreature()->getAuraCountForId(mSpellInfo->getId()) >= function->getFunctionArgs().getMaxStackCount())
239 continue; // Aura stack count reached, skip
240 }
241
242 // Casting Should be Possible reduce Timers
243 function->decreaseCooldownTimer(Milliseconds(time_passed));
244
245 // When our Timer is over Execute the Function
246 if (function->getCooldownTimer().count() <= 0)
247 {
248 if (function->getFunction())
249 {
250 // If the chance is not met, try again in 1.5 seconds
251 if (!function->isChanceMeet(rolled))
252 {
253 function->setCooldownTimer(1500ms);
254 continue;
255 }
256
257 // Execute the Function
258 function->increaseCount();
259 function->getFunction()(function);
260
261 // Reset the cooldown timer
262 if (function->getFunctionArgs().getSpellId() && function->getFunctionArgs().getUseSpellCD())
263 function->setCooldownTimer(function->getSpellCooldown());
264 else
265 function->setCooldownTimer(function->getInitialCooldown());
266 }
267 }
268 }

287void CreatureAIFunctionScheduler::delayAllEvents(Milliseconds time)
288{
289 if (!mCreatureAIFunctions.empty())
290 {
291 for (auto itr = mCreatureAIFunctions.begin(); itr != mCreatureAIFunctions.end();)
292 {
293 // Only Delay Timers that are not Finished and in our Current Phase
294 if (itr->get()->getCooldownTimer().count() > 0 && (itr->get()->isAvailableForScriptPhase(mOwner->getScriptPhase())))
295 itr->get()->setCooldownTimer(itr->get()->getCooldownTimer() + time);
296
297 ++itr;
298 }
299 }
