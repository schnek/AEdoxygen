12CreatureAIFunction::CreatureAIFunction(CreatureAIScript* owner, Function pFunction, SchedulerArgs const& schedArgs, FunctionArgs const& funcArgs)
13{
14 mOwner = owner;
15 mFunction = pFunction;
16 schedulerArgs = schedArgs;
17 functionArgs = funcArgs;
18 mCooldownTimer = schedArgs.getInitialCooldown();
19
20 // Set CastTime and SpellDefault CD when its a Spell
21 if (const auto spellInfo = sSpellMgr.getSpellInfo(functionArgs.getSpellId()))
22 {
23 uint32_t castTime = GetCastTime(sSpellCastTimesStore.lookupEntry(spellInfo->getCastingTimeIndex())) ? GetCastTime(sSpellCastTimesStore.lookupEntry(spellInfo->getCastingTimeIndex())) : 500;
24 setCastTimer(Milliseconds(castTime));
25
26 // Set RecoveryTimer
27 // When not Available use Default Time
28 Milliseconds cooldown = Milliseconds(spellInfo->getRecoveryTime()) + getCastTimer();
29 if (!cooldown.count())
30 schedulerArgs.setSpellCooldown(mCooldownTimer);
31 else
32 schedulerArgs.setSpellCooldown(Milliseconds(cooldown));
33 }
34}
