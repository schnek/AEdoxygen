122void CreatureAIScript::_internalAIUpdate(unsigned long time_passed)
123{
124 //sLogger.debug("CreatureAIScript::_internalAIUpdate() called");
125
126 // New AI Function Scheduler
127 mCreatureAIScheduler->update(time_passed);
128
129 updateAITimers(time_passed);
130 AIUpdate(time_passed);
131
132 // idleemotes
133 if (!_isInCombat() && isIdleEmoteEnabled)
134 {
135 if (_isTimerFinished(getIdleEmoteTimerId()))
136 {
137 sendRandomDBChatMessage(mEmotesOnIdle, nullptr);
138 generateNextRandomIdleEmoteTime();
139 }
140 }
141
142 // old AIUpdate stuff is now handled by customAIUpdateTimer. Keep this until all scripts are updated to new logic.
143 if (mCustomAIUpdateDelayTimerId != 0)
144 {
145 if (!_isTimerFinished(mCustomAIUpdateDelayTimerId))
146 return;
147
148 AIUpdate();
149
150 _resetTimer(mCustomAIUpdateDelayTimerId, mCustomAIUpdateDelay);
151 }
152 else
153 {
154 m_oldAIUpdate->updateTimer(time_passed);
155
156 // old Timer AIUpdate
157 if (m_oldAIUpdate->isTimePassed())
158 {
159 AIUpdate();
160 m_oldAIUpdate->resetInterval(1000);
161 }
162 }
163}

261Creature* CreatureAIScript::spawnCreature(uint32_t entry, float posX, float posY, float posZ, float posO, uint32_t factionId /* = 0*/, uint32_t phase /*= 1*/)
262{
263 CreatureProperties const* creatureProperties = sMySQLStore.getCreatureProperties(entry);
264 if (creatureProperties == nullptr)
265 {
266 sLogger.failure("tried to create an invalid creature with entry {}!", entry);
267 return nullptr;
268 }
269
270 if (_creature->getWorldMap()->getInterface() == nullptr)
271 return nullptr;
272
273 Creature* creature = _creature->getWorldMap()->getInterface()->spawnCreature(entry, LocationVector(posX, posY, posZ, posO), true, true, 0, 0, phase);
274 if (creature == nullptr)
275 return nullptr;
276
277 if (factionId != 0)
278 creature->setFaction(factionId);
279 else
280 creature->setFaction(creatureProperties->Faction);
281
282 return creature;
283}

579void CreatureAIScript::setWaypointToMove(uint32_t pathId, uint32_t pWaypointId)
580{
581 auto _path = getCustomPath(pathId);
582 WaypointNode const &waypoint = _path->nodes[pWaypointId];
583
584 MovementMgr::MoveSplineInit init(getCreature());
585 init.MoveTo(waypoint.x, waypoint.y, waypoint.z);
586
587 //! Accepts angles such as 0.00001 and -0.00001, 0 must be ignored, default value in waypoint table
588 if (waypoint.orientation && waypoint.delay)
589 init.SetFacing(waypoint.orientation);
590
591 switch (waypoint.moveType)
592 {
593#if VERSION_STRING >= WotLK
594 case WAYPOINT_MOVE_TYPE_LAND:
595 init.SetAnimation(ANIMATION_FLAG_GROUND);
596 break;
597 case WAYPOINT_MOVE_TYPE_TAKEOFF:
598 init.SetAnimation(ANIMATION_FLAG_HOVER);
599 break;
600#endif
601 case WAYPOINT_MOVE_TYPE_RUN:
602 init.SetWalk(false);
603 break;
604 case WAYPOINT_MOVE_TYPE_WALK:
605 init.SetWalk(true);
606 break;
607 default:
608 break;
609 }
610
611 init.Launch();
612}

1170void CreatureAIScript::CreatureAIFunc_CastSpell(CreatureAIFunc pThis)
1171{
1172 auto sharedFunc = pThis.lock();
1173 if (sharedFunc)
1174 {
1175 // Setup Filters
1176 FilterArgs mTargetFilters = sharedFunc->getFunctionArgs().getTargetFilters();
1177
1178 // Get a Spelltarget
1179 Unit* target = selectUnitTarget(mTargetFilters);
1180
1181 // Only Cast Spell when our Target is Valid
1182 if (target || mTargetFilters.hasFilter(TargetFilter_AOE))
1183 {
1184 uint32_t spellId = sharedFunc->getFunctionArgs().getSpellId();
1185 bool triggered = sharedFunc->getFunctionArgs().isTriggered();
1186
1187 // Cast Spell
1188 if (mTargetFilters.hasFilter(TargetFilter_AOE))
1189 castSpellAOE(spellId, triggered);
1190 else
1191 castSpell(target, spellId, triggered);
1192
1193 // send announcements on casttime beginn
1194 if (!sharedFunc->getFunctionArgs().getAnnouncement().empty())
1195 {
1196 sendChatMessage(CHAT_MSG_RAID_BOSS_EMOTE, LANG_UNIVERSAL, sharedFunc->getFunctionArgs().getAnnouncement());
1197 }
1198
1199 // Add Message On CastTimer Ended Slot MessageSlot::SLOT_DEFAULT
1200 if (sharedFunc->getFunctionArgs().getAIMessages(MessageSlot::SLOT_DEFAULT).size())
1201 {
1202 Message message;
1203 for (const auto text : sharedFunc->getFunctionArgs().getAIMessages(MessageSlot::SLOT_DEFAULT))
1204 message.addDBMessage(text.messageId, MessageSlot::SLOT_DEFAULT, text.useTarget);
1205
1206 addMessage(message, DoOnceScheduler(sharedFunc->getCastTimer()));
1207 }
1208
1209 // Add Message On CastTimer Ended Slot MessageSlot::SLOT_BONUS
1210 if (sharedFunc->getFunctionArgs().getAIMessages(MessageSlot::SLOT_BONUS).size())
1211 {
1212 Message message;
1213 for (const auto text : sharedFunc->getFunctionArgs().getAIMessages(MessageSlot::SLOT_BONUS))
1214 message.addDBMessage(text.messageId, MessageSlot::SLOT_BONUS, text.useTarget);
1215
1216 addMessage(message, DoOnceScheduler(sharedFunc->getCastTimer()));
1217 }
1218 }
