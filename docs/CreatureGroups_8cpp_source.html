23void FormationMgr::addCreatureToGroup(uint32_t leaderSpawnId, Creature* creature)
24{
25 WorldMap* map = creature->getWorldMap();
26
27 const auto [itr, inserted] = map->CreatureGroupHolder.try_emplace(leaderSpawnId, Util::LazyInstanceCreator([leaderSpawnId] {
28 return std::make_unique<CreatureGroup>(leaderSpawnId);
29 }));
30
31 if (!inserted)
32 {
33 //Add member to an existing group
34 sLogger.debug("FormationMgr : Group found: {}, inserting creature {}, Group InstanceID {}", leaderSpawnId, creature->getGuid(), creature->GetInstanceID());
35
36 // With dynamic spawn the creature may have just respawned
37 // we need to find previous instance of creature and delete it from the formation, as it'll be invalidated
38 for (const auto& pair : map->_sqlids_creatures)
39 {
40 if (pair.first == creature->getSpawnId())
41 {
42 Creature* other = pair.second;
43 if (other == creature)
44 continue;
45
46 if (itr->second->hasMember(other))
47 itr->second->removeMember(other);
48 }
49 }
50 }
51 else
52 {
53 //Create new group
54 sLogger.debug("FormationMgr : Group not found: {}. Creating new group.", leaderSpawnId);
55 }
56
57 itr->second->addMember(creature);
58}

60void FormationMgr::removeCreatureFromGroup(CreatureGroup* group, Creature* member)
61{
62 sLogger.debug("FormationMgr : Deleting member pointer to GUID: {} from group {}", group->getLeaderSpawnId(), member->getSpawnId());
63 group->removeMember(member);
64
65 if (group->isEmpty())
66 {
67 WorldMap* map = member->getWorldMap();
68
69 sLogger.debug("FormationMgr : Deleting group with InstanceID {}", member->GetInstanceID());
70 auto itr = map->CreatureGroupHolder.find(group->getLeaderSpawnId());
71 ASSERT(itr != map->CreatureGroupHolder.end() && "Not registered group in map");
72 map->CreatureGroupHolder.erase(itr);
73 }
74}

76void FormationMgr::loadCreatureFormations()
77{
78 auto oldMSTime = Util::TimeNow();
79
80 //Get group data
81 auto result = WorldDatabase.Query("SELECT leaderGUID, memberGUID, dist, angle, groupAI, point_1, point_2 FROM creature_formations ORDER BY leaderGUID");
82 if (!result)
83 {
84 sLogger.debug("FormationMgr : Loaded 0 creatures in formations. DB table `creature_formations` is empty!");
85 return;
86 }
87
88 uint32_t count = 0;
89 std::unordered_set<uint32_t> leaderSpawnIds;
90 do
91 {
92 Field* fields = result->Fetch();
93
94 //Load group member data
95 FormationInfo member;
96 member.LeaderSpawnId = fields[0].asUint32();
97 uint32_t memberSpawnId = fields[1].asUint32();
98 member.FollowDist = 0.f;
99 member.FollowAngle = 0.f;
100
101 //If creature is group leader we may skip loading of dist/angle
102 if (member.LeaderSpawnId != memberSpawnId)
103 {
104 member.FollowDist = fields[2].asFloat();
105 member.FollowAngle = fields[3].asFloat() * float(M_PI) / 180.0f;
106 }
107
108 member.GroupAI = fields[4].asUint32();
109 for (uint8_t i = 0; i < 2; ++i)
110 member.LeaderWaypointIDs[i] = fields[5 + i].asUint16();
111
112 // check data correctness
113 {
114 auto spawnResult = WorldDatabase.Query("SELECT * FROM creature_spawns WHERE id = %u", member.LeaderSpawnId);
115 if (spawnResult == nullptr)
116 {
117 sLogger.failure("FormationMgr : creature_formations table leader guid {} incorrect (not exist)", member.LeaderSpawnId);
118 continue;
119 }
120
121 spawnResult = nullptr;
122 spawnResult = WorldDatabase.Query("SELECT * FROM creature_spawns WHERE id = %u", memberSpawnId);
123 if (spawnResult == nullptr)
124 {
125 sLogger.failure("FormationMgr : creature_formations table member guid {} incorrect (not exist)", memberSpawnId);
126 continue;
127 }
128
129 leaderSpawnIds.insert(member.LeaderSpawnId);
130 }
131
132 _creatureGroupMap.emplace(memberSpawnId, std::move(member));
133 ++count;
134 } while (result->NextRow());
135
136 for (uint32_t leaderSpawnId : leaderSpawnIds)
137 {
138 if (!_creatureGroupMap.count(leaderSpawnId))
139 {
140 sLogger.failure("FormationMgr : creature_formation contains leader spawn {} which is not included on its formation, removing", leaderSpawnId);
141 for (auto itr = _creatureGroupMap.begin(); itr != _creatureGroupMap.end();)
142 {
143 if (itr->second.LeaderSpawnId == leaderSpawnId)
144 {
145 itr = _creatureGroupMap.erase(itr);
146 continue;
147 }
148
149 ++itr;
150 }
151 }
152 }
153
154 sLogger.debug("FormationMgr : Loaded {} creatures in formations in {} ms", count, Util::GetTimeDifferenceToNow(oldMSTime));
155}

208void CreatureGroup::memberEngagingTarget(Creature* member, Unit* target)
209{
210 // used to prevent recursive calls
211 if (_engaging)
212 return;
213
214 const auto groupAI = sFormationMgr->getFormationInfo(member->getSpawnId())->GroupAI;
215 if (!groupAI)
216 return;
217
218 if (member == _leader)
219 {
220 if (!(groupAI & FLAG_MEMBERS_ASSIST_LEADER))
221 return;
222 }
223 else if (!(groupAI & FLAG_LEADER_ASSISTS_MEMBER))
224 return;
225
226 _engaging = true;
227
228 for (auto const& pair : _members)
229 {
230 Creature* other = pair.first;
231 if (other == member)
232 continue;
233
234 if (!other->isAlive())
235 continue;
236
237 if (((other != _leader && (groupAI & FLAG_MEMBERS_ASSIST_LEADER)) || (other == _leader && (groupAI & FLAG_LEADER_ASSISTS_MEMBER))))
238 other->getAIInterface()->onHostileAction(target);
239 }
240
241 _engaging = false;
242}

244void CreatureGroup::formationReset(bool dismiss)
245{
246 for (auto const& pair : _members)
247 {
248 if (pair.first != _leader && pair.first->isAlive())
249 {
250 if (dismiss)
251 pair.first->getMovementManager()->initialize();
252 else
253 pair.first->getMovementManager()->moveIdle();
254 sLogger.debug("FormationMgr : CreatureGroup::FormationReset: Set {} movement for member {}", dismiss ? "default" : "idle", pair.first->getGuid());
255 }
256 }
257
258 _formed = !dismiss;
259}

261void CreatureGroup::leaderStartedMoving()
262{
263 if (!_leader)
264 return;
265
266 for (auto const& pair : _members)
267 {
268 Creature* member = pair.first;
269 if (member == _leader || !member->isAlive() || member->getAIInterface()->isEngaged() || !(pair.second->GroupAI & FLAG_IDLE_IN_FORMATION))
270 continue;
271
272 float angle = pair.second->FollowAngle + float(M_PI); // for some reason, someone thought it was a great idea to invert relativ angles...
273 float dist = pair.second->FollowDist;
274
275 if (!member->hasUnitStateFlag(UNIT_STATE_FOLLOW_FORMATION))
276 member->getMovementManager()->moveFormation(_leader, dist, angle, pair.second->LeaderWaypointIDs[0], pair.second->LeaderWaypointIDs[1]);
277 }
278}

280bool CreatureGroup::canLeaderStartMoving() const
281{

283 {
284 if (pair.first != _leader && pair.first->isAlive())
285 {
286 if (pair.first->getAIInterface()->isEngaged() || pair.first->isReturningHome())
287 return false;
288 }
289 }
290
291 return true;
