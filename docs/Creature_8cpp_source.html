49uint8_t CreatureProperties::generateRandomDisplayIdAndReturnGender(uint32_t* displayId) const
50{
51 if (isTriggerNpc)
52 {
53 // Trigger npcs should load their invisible model
54 *displayId = getInvisibleModelForTriggerNpc();
55 return 0;
56 }
57
58 const uint32_t models[] = { Male_DisplayID, Male_DisplayID2, Female_DisplayID, Female_DisplayID2 };
59 if (!models[0] && !models[1] && !models[2] && !models[3])
60 {
61 // All models are invalid
62 sLogger.failure("CreatureProperties : All display IDs are invalid for creature entry {}", Id);
63 return 0;
64 }
65
66 while (true)
67 {
68 const auto res = Util::getRandomUInt(3);
69 if (models[res])
70 {
71 *displayId = models[res];
72 return res < 2 ? 0U : 1U;
73 }
74 }

199{
200 if (controller == nullptr)
201 return;
202
203 const uint16_t flags = isVehicle() ? 0x0800 : 0;
204
205 // Send the actionbar
206 AscEmu::Packets::SmsgPetActionsArray actions{};
207 if (isVehicle())
208 {
209 auto itr = creature_properties->castable_spells.cbegin();
210 for (uint8_t i = 0; i < 10; ++i)
211 {
212 if (itr != creature_properties->castable_spells.cend())
213 {
214 actions[i] = AscEmu::Packets::packPetActionButtonData(*itr, PET_SPELL_STATE_DEFAULT);
215 ++itr;
216 }
217 else
218 {
219 actions[i] = 0;
220 }
221 }
222 }
223 else
224 {
225 actions[0] = AscEmu::Packets::packPetActionButtonData(PET_ACTION_ATTACK, PET_SPELL_STATE_SET_ACTION);
226 actions[1] = AscEmu::Packets::packPetActionButtonData(PET_ACTION_FOLLOW, PET_SPELL_STATE_SET_ACTION);
227 actions[2] = AscEmu::Packets::packPetActionButtonData(PET_ACTION_STAY, PET_SPELL_STATE_SET_ACTION);
228
229 auto itr = creature_properties->castable_spells.cbegin();
230 for (uint8_t i = 3; i < 7; ++i)
231 {
232 if (itr != creature_properties->castable_spells.cend())
233 {
234 actions[i] = AscEmu::Packets::packPetActionButtonData(*itr, PET_SPELL_STATE_DEFAULT);
235 ++itr;
236 }
237 else
238 {
239 actions[i] = 0;
240 }
241 }
242
243 actions[7] = AscEmu::Packets::packPetActionButtonData(PET_STATE_AGGRESSIVE, PET_SPELL_STATE_SET_REACT);
244 actions[8] = AscEmu::Packets::packPetActionButtonData(PET_STATE_DEFENSIVE, PET_SPELL_STATE_SET_REACT);

249 controller->sendPacket(AscEmu::Packets::SmsgPetSpells(getGuid(), familyId, duration,
250 PET_STATE_PASSIVE, PET_ACTION_FOLLOW, flags, std::move(actions), AscEmu::Packets::SmsgPetSpellsVector()).serialise().get());

468void Creature::setDeathState(DeathState s)
469{
470 Unit::setDeathState(s);
471
472 if (s == ALIVE)
473 this->removeUnitFlags(UNIT_FLAG_DEAD);
474
475 if (s == JUST_DIED)
476 {
477 // Respawn Handling
478 const auto now = Util::getTimeNow();
479
480 m_corpseRemoveTime = now + m_corpseDelay;
481
482 uint32_t respawnDelay = m_respawnDelay;
483
484 if (isDungeonBoss() && !m_respawnDelay)
485 m_respawnTime = std::numeric_limits<time_t>::max(); // never respawn in this instance
486 else
487 m_respawnTime = now + respawnDelay + m_corpseDelay;
488
489 saveRespawnTime();
490
491 setTargetGuid(0);
492
493 setUnitFlags(UNIT_NPC_FLAG_NONE);
494
495 setMountDisplayId(0);
496
497 if (m_enslaveSpell)
498 RemoveEnslave();
499
500 //Dismiss group if is leader
501 if (m_formation && m_formation->getLeader() == this)
502 m_formation->formationReset(true);
503
504 for (uint8_t i = 0; i < CURRENT_SPELL_MAX; ++i)
505 interruptSpellWithSpellType(static_cast<CurrentSpellType>(i));
506
507 // if it's not a Pet, and not a summon and it has skinningloot then we will allow skinning
508 if (getCreatedByGuid() == 0 && getSummonedByGuid() == 0 && sLootMgr.isSkinnable(creature_properties->Id))
509 addUnitFlags(UNIT_FLAG_SKINNABLE);
510
511 bool needsFalling = (IsFlying() || isHovering()) && !isUnderWater();
512 setMoveHover(false);
513 setMoveDisableGravity(false);
514
515 if (needsFalling)
516 getMovementManager()->moveFall();
517
518 Unit::setDeathState(CORPSE);
519 }
520 else if (s == JUST_RESPAWNED)
521 {
522 if (isPet())
523 {
524 setFullHealth();
525 }
526 else
527 {
528 uint32_t curhealth = getMaxHealth();
529 setPower(POWER_TYPE_MANA, getMaxPower(POWER_TYPE_MANA));
530 setHealth((m_deathState == ALIVE || m_deathState == JUST_RESPAWNED) ? curhealth : 0);
531 }
532
533 setTaggerGuid(nullptr);
534
535 getAIInterface()->setCannotReachTarget(false);
536 updateMovementFlags();
537
538 removeUnitStateFlag(UNIT_STATE_ALL_ERASABLE);
539
540 OnRespawn(getWorldMap());
541
542 motion_Initialize();
543 Unit::setDeathState(ALIVE);
544 }
545}

718void Creature::updateMovementFlags()
719{
720 // Do not update movement flags if creature is controlled by a player (charm/vehicle)
721 if (getUnitOwner())
722 return;
723
724 // Set the movement flags if the creature is in that mode. (Only fly if actually in air, only swim if in water, etc)
725 const float ground = getFloorZ();
726
727 bool isInAir;

733#endif

738 if (getMovementTemplate().isFlightAllowed())

740 if (isAlive())

750 setMoveDisableGravity(true);

759 setMoveCanFly(false);

762 setMoveDisableGravity(false);

765 needsFalling = true;
766 }
767 }
768
769 if (isHovering() && (!hasAuraWithAuraEffect(SPELL_AURA_HOVER) || !isAlive()))

771 setMoveHover(false);
772 needsFalling = true;
773 }
774
775 // Make unit fall
776 if (needsFalling && !isUnderWater())
777 getMovementManager()->moveFall();
778 }
779
780 if (!isInAir)
781 {
782 removeUnitMovementFlag(MOVEFLAG_FALLING);

785 setMoveCanFly(false);

797 if (!hasUnitMovementFlag(MOVEFLAG_SWIMMING))
798 setMoveSwim(true);
799 }

1351 else

1360

1364void Creature::CalcStat(uint8_t type)
1365{
1366 int32_t pos = 0;
1367 int32_t neg = 0;
1368
1369 if (StatModPct[type] < 0)
1370 neg = (m_baseStats[type] * abs(StatModPct[type]) / 100);
1371 else
1372 pos = (m_baseStats[type] * StatModPct[type]) / 100;
1373
1374 if (isPet())
1375 {
1376 Player* owner = static_cast<Pet*>(this)->getPlayerOwner();
1377 if (type == STAT_STAMINA && owner)
1378 pos += int32_t(0.45f * owner->getStat(STAT_STAMINA));
1379 else if (type == STAT_INTELLECT && owner && getCreatedBySpellId())
1380 pos += int32_t(0.30f * owner->getStat(STAT_INTELLECT));
1381 }
1382
1383 if (TotalStatModPct[type] < 0)
1384 neg += (m_baseStats[type] + pos - neg) * abs(TotalStatModPct[type]) / 100;
1385 else
1386 pos += (m_baseStats[type] + pos - neg) * TotalStatModPct[type] / 100;
1387
1388 if (FlatStatMod[type] < 0)
1389 neg += abs(FlatStatMod[type]);
1390 else
1391 pos += FlatStatMod[type];
1392
1393#if VERSION_STRING != Classic
1394 setPosStat(type, pos);
1395 setNegStat(type, neg);
1396#endif
1397
1398 int32_t tot = m_baseStats[type] + pos - neg;
1399 setStat(type, tot > 0 ? tot : 0);
1400
1401 switch (type)
1402 {
1403 case STAT_STRENGTH:
1404 {
1405 //Attack Power
1406 if (!isPet()) //We calculate pet's later
1407 {
1408 uint32_t str = getStat(STAT_STRENGTH);
1409 int32_t AP = (str * 2 - 20);
1410 if (AP < 0) AP = 0;
1411 setAttackPower(AP);
1412 }
1413 calculateDamage();
1414 }
1415 break;
1416 case STAT_AGILITY:
1417 {
1418 //Ranged Attack Power (Does any creature use this?)
1419 int32_t RAP = getLevel() + getStat(STAT_AGILITY) - 10;
1420 if (RAP < 0)
1421 RAP = 0;
1422
1423 setRangedAttackPower(RAP);
1424 }
1425 break;
1426 case STAT_STAMINA:
1427 {
1428#if VERSION_STRING != Classic
1429 //Health
1430 uint32_t hp = getBaseHealth();
1431 uint32_t stat_bonus = getPosStat(STAT_STAMINA)- getNegStat(STAT_STAMINA);
1432 if (static_cast<int32_t>(stat_bonus) < 0) stat_bonus = 0;
1433
1434 uint32_t bonus = stat_bonus * 10 + m_healthfromspell;
1435 uint32_t res = hp + bonus;
1436
1437 if (res < hp)
1438 res = hp;
1439 setMaxHealth(res);
1440 if (getHealth() > getMaxHealth())
1441 setHealth(getMaxHealth());
1442#endif
1443 }
1444 break;
1445 case STAT_INTELLECT:
1446 {
1447#if VERSION_STRING != Classic
1448 if (getPowerType() == POWER_TYPE_MANA)
1449 {
1450 uint32_t mana = getBaseMana();
1451 uint32_t stat_bonus = getPosStat(STAT_INTELLECT) - getNegStat(STAT_INTELLECT);
1452 if (static_cast<int32_t>(stat_bonus) < 0) stat_bonus = 0;
1453
1454 uint32_t bonus = stat_bonus * 15;
1455 uint32_t res = mana + bonus;
1456
1457 if (res < mana) res = mana;
1458 setMaxPower(POWER_TYPE_MANA, res);
1459 }
1460#endif
1461 }
1462 break;
1463 }
