402// Nestlewood Owlkin - Quest 9303
403class NestlewoodOwlkin : public CreatureAIScript
404{
405public:
406 static CreatureAIScript* Create(Creature* c) { return new NestlewoodOwlkin(c); }
407 explicit NestlewoodOwlkin(Creature* pCreature) : CreatureAIScript(pCreature)
408 {
409 respawn = false;
410 reset = false;
411 }
412
413 void AIUpdate() override
414 {
415 if (!reset)
416 {
417 if (getCreature()->hasAurasWithId(29528) && !respawn)
418 {
419 reset = true;
420 getCreature()->setMoveRoot(true);
421 RemoveAIUpdateEvent();
422 GiveKillCredit();
423 }
424 }
425 else
426 {
427 respawn = true;
428 getCreature()->setMoveRoot(false);
429 getCreature()->Despawn(0, 10000); // respawn delay 10 seconds
430 }
431 }
432
433 void GiveKillCredit()
434 {
435 if (getCreature()->hasAurasWithId(29528))
436 {
437 if (auto* player = getCreature()->getWorldMap()->getPlayer(static_cast<uint32_t>(getCreature()->getTargetGuid())))
438 {
439 if (!player->hasQuestInQuestLog(9303) || player->hasQuestFinished(9303))
440 return;
441
442 if (auto* questLog = player->getQuestLogByQuestId(9303))
443 {
444 if (questLog->getMobCountByIndex(0) < 6)
445 {
446 questLog->incrementMobCountForIndex(0);
447 questLog->sendUpdateAddKill(0);
448 questLog->updatePlayerFields();
449
450 RegisterAIUpdateEvent(240000); // update after 4 mins
451 }
452 }
453 }
454 }
455 }
456
457 void OnLoad() override
458 {
459 RegisterAIUpdateEvent(4000);
460 reset = false;
461 respawn = false;
462 }
463
464private:
465 bool reset;
466 bool respawn;
