18void DatabaseUpdater::initBaseIfNeeded(const std::string& dbName, const std::string& dbBaseType, Database& dbPointer)
19{
20 auto dbResult = dbPointer.Query("SHOW TABLES FROM %s", dbName.c_str());
21 if (dbResult == nullptr)
22 {
23 sLogger.info("Database: Your Database {} has no tables. AE is setting up the database for you.", dbName);
24 setupDatabase(dbBaseType, dbPointer);
25 }
26
27 // save 100% (current queue)
28 if (auto const queue = dbPointer.GetQueueSize())
29 {
30 // set up bar size
31 const int barSize = 70;
32
33 // set up progress
34 float currentProgress = 0.0f;
35
36 // update the line
37 while (currentProgress < 1.0f)
38 {
39 // calc percentage
40 const size_t sendQueues = queue - dbPointer.GetQueueSize();
41 currentProgress = static_cast<float>(sendQueues) / static_cast<float>(queue);
42
43 std::cout << "Creating '" << dbName << "' : ";
44 const auto position = static_cast<int>(barSize * currentProgress);
45 for (auto i = 0; i < barSize; ++i)
46 {
47 if (i < position)
48 std::cout << "-";
49 else if (i == position)
50 std::cout << ">";
51 else
52 std::cout << " ";
53 }
54
55 std::cout << " | " << int(currentProgress * 100.0f);
56 if (currentProgress < 1.0f)
57 {
58 std::cout << " %\r";
59 std::cout.flush();
60 }
61 else
62 std::cout << " %\n";
63
64 Arcemu::Sleep(250);
65 }
66 }

69void DatabaseUpdater::setupDatabase(const std::string& database, Database& dbPointer)

72 fs::path baseFilePath = fs::current_path();

117void DatabaseUpdater::applyUpdatesForDatabase(const std::string& database, Database& dbPointer)
118{
119 const std::string sqlUpdateDir = "sql/" + database + "/updates";
120
121 //////////////////////////////////////////////////////////////////////////////////////////
122 // 1. get current version
123 auto result = dbPointer.Query("SELECT LastUpdate FROM %s_db_version ORDER BY LastUpdate DESC LIMIT 1", database.c_str());
124
125 if (!result)
126 {
127 sLogger.failure("{}_db_version query failed!", database);
128 return;
129 }
130
131 Field* fields = result->Fetch();
132 const std::string dbLastUpdate = fields[0].asCString();
133
134 sLogger.info("Database {} Version : {}", database, dbLastUpdate);
135
136 const auto lastUpdateMajor = Util::readMajorVersionFromString(dbLastUpdate);
137 const auto lastUpdateMinor = Util::readMinorVersionFromString(dbLastUpdate);
138
139 //////////////////////////////////////////////////////////////////////////////////////////
140 // 2. check if update folder exist in *dir*/sql/
141 std::map<uint32_t, DatabaseUpdateFile> updateSqlStore;
142
143 uint32_t count = 0;
144 std::vector<std::string> updateFiles;
145
146 for (auto& p : fs::recursive_directory_iterator(sqlUpdateDir))
147 {
148 const std::string filePathName = p.path().string();
149 updateFiles.push_back(filePathName);
150 }
151
152 // In Windows, recursive_directory_iterator seems to get files sorted but
153 // in Linux they are in random order -Appled
154 std::ranges::sort(updateFiles);
155
156 for (const auto& filePathName : updateFiles)
157 {
158 std::string fileName = filePathName;
159 fileName.erase(0, sqlUpdateDir.size() + 1);
160
161 const uint32_t majorVersion = Util::readMajorVersionFromString(fileName);
162 const uint32_t minorVersion = Util::readMinorVersionFromString(fileName);
163
164 DatabaseUpdateFile dbUpdateFile;
165 dbUpdateFile.fullName = filePathName;
166 dbUpdateFile.majorVersion = majorVersion;
167 dbUpdateFile.minorVersion = minorVersion;
168
169 //\todo Remove me
170 //sLogger.info("Available file in updates dir: {}", filePathName);
171
172 updateSqlStore.emplace(count, dbUpdateFile);
173 ++count;
174 }
175
176 updateFiles.clear();
177
178 //////////////////////////////////////////////////////////////////////////////////////////
179 // 3. save filenames into vector, when newer than current db version
180 std::map<uint32_t, DatabaseUpdateFile> applyNewUpdateFilesStore;
181
182 if (!updateSqlStore.empty())
183 {
184 //sLogger.debug("=========== New {} update files in {} ===========", database, sqlUpdateDir);
185 //compare it with latest update in mysql
186 for (const auto& update : updateSqlStore)
187 {
188 bool addToUpdateFiles = false;
189 if (update.second.majorVersion == lastUpdateMajor && update.second.minorVersion > lastUpdateMinor)
190 addToUpdateFiles = true;
191
192 if (update.second.majorVersion > lastUpdateMajor)
193 addToUpdateFiles = true;
194
195 if (addToUpdateFiles)
196 {
197 applyNewUpdateFilesStore.emplace(update);
198 sLogger.debug("Updatefile {}, Major({}), Minor({}) - added and ready to be applied!", update.second.fullName, update.second.majorVersion, update.second.minorVersion);
199 }
200 }
201 }
202
203 //////////////////////////////////////////////////////////////////////////////////////////
204 // 4. open/parse files and apply to db
205 if (!applyNewUpdateFilesStore.empty())
206 {
207 sLogger.info("=========== Applying sql updates from {} ===========", sqlUpdateDir);
208
209 for (const auto& [_, updateFilePath] : applyNewUpdateFilesStore)
210 {
211 const fs::path sqlFile = fs::current_path() /= updateFilePath.fullName;
212
213 if (fs::exists(sqlFile))
214 {
215 sLogger.info("{}", updateFilePath.fullName);
216 std::string loadedFile = Util::readFileIntoString(sqlFile);
217 // Make sure newlines are same in all files -Appled
218 loadedFile = std::regex_replace(loadedFile, std::regex("\r\n+"), "\n");
219
220 // split into seperated string
221 std::vector<std::string> seglist;
222 std::string delimiter = ";\n";
223
224 size_t pos = 0;
225 std::string token;
226 while ((pos = loadedFile.find(delimiter)) != std::string::npos)
227 {
228 token = loadedFile.substr(0, pos);
229 seglist.emplace_back(token + ";");
230 loadedFile.erase(0, pos + delimiter.length());
231 }
232
233 for (const auto& statements : seglist)
234 {
235 if (dbPointer.WaitExecuteNA(statements.c_str()))
236 continue;
237 }
238 }
239 }
240 }
