124std::unique_ptr<QueryResult> Database::Query(const char* QueryString, ...)
125{
126 va_list vlist;
127 va_start(vlist, QueryString);
128
129 // Get buffer size
130 va_list vlist_copy;
131 va_copy(vlist_copy, vlist);
132 const auto size = vsnprintf(nullptr, 0, QueryString, vlist_copy);
133 va_end(vlist_copy);
134
135 if (size < 0)
136 {
137 va_end(vlist);
138 return nullptr;
139 }
140
141 std::string sql(size, '\0');
142 vsnprintf(&sql[0], static_cast<size_t>(size) + 1, QueryString, vlist);
143 va_end(vlist);
144
145 // Send the query
146 std::unique_ptr<QueryResult> qResult;
147 DatabaseConnection* con = GetFreeConnection();
148
149 if (_SendQuery(con, sql.data(), false))
150 qResult = _StoreQueryResult(con);
151
152 con->Busy.release();
153 return qResult;
154}

301void Database::PerformQueryBuffer(QueryBuffer* b, DatabaseConnection* ccon)
302{
303 if (!b->queries.size())
304 return;
305
306 DatabaseConnection* con = ccon;
307 if (ccon == NULL)
308 con = GetFreeConnection();
309
310 _BeginTransaction(con);
311
312 for (auto itr = b->queries.begin(); itr != b->queries.end(); ++itr)
313 {
314 _SendQuery(con, (*itr).data(), false);
315 }
316
317 _EndTransaction(con);
318
319 if (ccon == NULL)
320 con->Busy.release();
321}

420void AsyncQuery::Perform()
421{
422 DatabaseConnection* conn = db->GetFreeConnection();
423 for (std::vector<AsyncQueryResult>::iterator itr = queries.begin(); itr != queries.end(); ++itr)
424 itr->result = db->FQuery(itr->query.data(), conn);
425
426 conn->Busy.release();
427 func->run(queries);
