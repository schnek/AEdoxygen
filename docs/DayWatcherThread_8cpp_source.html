162bool DayWatcherThread::has_timeout_expired(tm* now_time, tm* last_time, uint32_t timeoutval)
163{
164 switch (timeoutval)
165 {
166 case WEEKLY:
167 return abs(now_time->tm_yday - last_time->tm_yday) >= 7;
168 case MONTHLY:
169 return now_time->tm_mon != last_time->tm_mon;
170 case HOURLY:
171 return now_time->tm_hour != last_time->tm_hour || now_time->tm_mday != last_time->tm_mday || now_time->tm_mon != last_time->tm_mon;
172 case DAILY:
173 return now_time->tm_mday != last_time->tm_mday && now_time->tm_hour == 4;
174 }
175
176 return false;
177}

192void DayWatcherThread::update_arena()
193{
194 sLogger.info("DayWatcherThread : Running Weekly Arena Point Maintenance...");
195
196 auto result = CharacterDatabase.Query("SELECT guid, arenaPoints FROM characters");
197 uint32_t arenapointsPerTeam[3] = { 0 };
198 if (result)
199 {
200 do
201 {
202 Field* field = result->Fetch();
203 uint32_t guid = field[0].asUint32();
204 uint32_t arenapoints = field[1].asUint32();
205 uint32_t orig_arenapoints = arenapoints;
206
207 for (uint8_t i = 0; i < 3; ++i)
208 arenapointsPerTeam[i] = 0;
209
210 // are we in any arena teams?
211 for (uint8_t i = 0; i < 3; ++i)
212 {
213 if (const auto arenaTeam = sObjectMgr.getArenaTeamByGuid(guid, i))
214 {
215 const auto arenaTeamMember = arenaTeam->getMemberByGuid(guid);
216 if (arenaTeamMember == nullptr || arenaTeam->m_stats.played_week < 10 || arenaTeamMember->Played_ThisWeek * 100 / arenaTeam->m_stats.played_week < 30)
217 continue;
218
219 const double arenaStatsRating = arenaTeam->m_stats.rating;
220 double arenaPoints;
221
222 if (arenaStatsRating <= 510.0)
223 continue;
224
225 if (arenaStatsRating > 510.0 && arenaStatsRating <= 1500.0) // 510 < X <= 1500"
226 {
227 arenaPoints = 0.22 * arenaStatsRating + 14.0;
228 }
229 else
230 {
231 const double power = -0.00412 * arenaStatsRating;
232 double divisor = pow(2.71828, power);
233 divisor *= 1639.28;
234 divisor += 1.0;
235
236 arenaPoints = 1511.26 / divisor;
237 }
238
239 if (arenaTeam->m_type == ARENA_TEAM_TYPE_2V2)
240 {
241 arenaPoints *= 0.76;
242 arenaPoints *= worldConfig.getFloatRate(RATE_ARENAPOINTMULTIPLIER2X);
243 }
244 else if (arenaTeam->m_type == ARENA_TEAM_TYPE_3V3)
245 {
246 arenaPoints *= 0.88;
247 arenaPoints *= worldConfig.getFloatRate(RATE_ARENAPOINTMULTIPLIER3X);
248 }
249 else
250 {
251 arenaPoints *= worldConfig.getFloatRate(RATE_ARENAPOINTMULTIPLIER5X);
252 }
253
254 if (arenaPoints > 1.0)
255 arenapointsPerTeam[i] += Util::long2int32(double(ceil(arenaPoints)));
256 }
257 }
258
259 arenapointsPerTeam[0] = static_cast<uint32_t>(std::max(arenapointsPerTeam[0], arenapointsPerTeam[1]));
260 arenapoints += static_cast<uint32_t>(std::max(arenapointsPerTeam[0], arenapointsPerTeam[2]));
261 if (arenapoints > 5000)
262 arenapoints = 5000;
263
264 if (orig_arenapoints != arenapoints)
265 {
266 auto player = sObjectMgr.getPlayer(guid);
267 if (player != nullptr)
268 {
269 player->addArenaPoints(arenapoints, false);
270
271 // update fields (no uint lock)
272 sEventMgr.AddEvent(player, &Player::updateArenaPoints, EVENT_PLAYER_UPDATE, 100, 1, 0);
273 player->getSession()->systemMessage("Your arena points have been updated! Check your PvP tab!");
274 }
275
276 CharacterDatabase.Execute("UPDATE characters SET arenaPoints = %u WHERE guid = %u", arenapoints, guid);
277 }
278 } while (result->NextRow());
279 }
280
281 sObjectMgr.updateArenaTeamWeekly();
282
283 m_lastArenaTime = UNIXTIME;
284 dupe_tm_pointer(localtime(&m_lastArenaTime), &m_localLastArenaTime);
285 m_updateDBSettings = true;
286}
