81 return true;

85 auto creature_spawn_result = WorldDatabase.Query("SELECT entry FROM creature_spawns WHERE map = %u GROUP BY(entry)", map);
86 if (creature_spawn_result)
87 {
88 {
89 do
90 {
91 Field* fields = creature_spawn_result->Fetch();
92 creatureEntries.push_back(fields[0].asUint32());
93
94 } while (creature_spawn_result->NextRow());
95 }
96 }
97
98 //prepare new table for dump
99 char my_table[1400];
100 sprintf(my_table, "CREATE TABLE `creature_ai_scripts_%s` (`min_build` int NOT nullptr DEFAULT '8606',`max_build` int NOT nullptr DEFAULT '12340',`entry` int unsigned NOT nullptr,\
101 `difficulty` tinyint unsigned NOT nullptr DEFAULT '0',`phase` tinyint unsigned NOT nullptr DEFAULT '0',`event` tinyint unsigned NOT nullptr DEFAULT '0',`action` tinyint unsigned NOT nullptr DEFAULT '0',\
102 `maxCount` tinyint unsigned NOT nullptr DEFAULT '0',`chance` float unsigned NOT nullptr DEFAULT '1',`spell` int unsigned NOT nullptr DEFAULT '0',`spell_type` int NOT nullptr DEFAULT '0',`triggered` tinyint(1) NOT nullptr DEFAULT '0',\
103 `target` tinyint NOT nullptr DEFAULT '0',`cooldownMin` int NOT nullptr DEFAULT '0',`cooldownMax` int unsigned NOT nullptr DEFAULT '0',`minHealth` float NOT nullptr DEFAULT '0',\
104 `maxHealth` float NOT nullptr DEFAULT '100',`textId` int unsigned NOT nullptr DEFAULT '0',`misc1` int NOT nullptr DEFAULT '0',`comments` text CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci,\
105 UNIQUE KEY `entry` (`min_build`,`max_build`,`entry`,`difficulty`,`phase`,`spell`,`event`,`action`,`textId`) USING BTREE) ENGINE = MyISAM DEFAULT CHARSET = utf8mb4 COLLATE = utf8mb4_unicode_ci COMMENT = 'AI System'", args);
106
107 WorldDatabase.Execute(my_table);
108
109 uint32_t count = 0;
110 for (auto entry : creatureEntries)
111 {
112 auto creature_properties = sMySQLStore.getCreatureProperties(entry);
113 if (creature_properties == nullptr)
114 {
115 redSystemMessage(session, "Creature with entry {} is not a valid entry (no properties information in database)", entry);

132 creature_spawn->bytes0 = creature_spawn->setbyte(0, 2, gender);
133 creature_spawn->stand_state = 0;
134 creature_spawn->death_state = 0;
135 creature_spawn->channel_target_creature = creature_spawn->channel_target_go = creature_spawn->channel_spell = 0;
136 creature_spawn->MountedDisplayID = 0;
137 creature_spawn->sheath_state = 0;
138
139 creature_spawn->Item1SlotEntry = creature_properties->itemslot_1;
140 creature_spawn->Item2SlotEntry = creature_properties->itemslot_2;
141 creature_spawn->Item3SlotEntry = creature_properties->itemslot_3;
142
143 creature_spawn->CanFly = 0;
144 creature_spawn->phase = session->GetPlayer()->GetPhase();
145
146 if (auto creature = session->GetPlayer()->getWorldMap()->createCreature(entry))
147 {
148 creature->Load(creature_spawn, 0, nullptr);
149 creature->m_loadedFromDB = true;
150 creature->PushToWorld(session->GetPlayer()->getWorldMap());
151
152 // Add to map
153 uint32_t x = session->GetPlayer()->getWorldMap()->getPosX(session->GetPlayer()->GetPositionX());
154 uint32_t y = session->GetPlayer()->getWorldMap()->getPosY(session->GetPlayer()->GetPositionY());
155 session->GetPlayer()->getWorldMap()->getBaseMap()->getSpawnsListAndCreate(x, y)->CreatureSpawns.push_back(creature_spawn);
156 MapCell* map_cell = session->GetPlayer()->getWorldMap()->getCell(x, y);
157 if (map_cell != nullptr)
158 map_cell->setLoaded();
159

162 if (aiSpells->fromDB)

170 if (cooldown == 0xFFFFFFFF) //4294967295
171 cooldown = 10000;
172
173 std::string remove = "'";
174 std::string name = sMySQLStore.getCreatureProperties(entry)->Name;
175 name.erase(std::remove_if(name.begin(), name.end(),
176 [&remove](const char& c) {
177 return remove.find(c) != std::string::npos;
178 }),
179 name.end());
180
181 std::string spellname = aiSpells->mSpellInfo->getName();
182 spellname.erase(std::remove_if(spellname.begin(), spellname.end(),
183 [&remove](const char& c) {
184 return remove.find(c) != std::string::npos;
185 }),
186 spellname.end());
187
188 std::string comment = name + " - " + spellname;
189
190 char my_insert1[700];
191 sprintf(my_insert1, "INSERT INTO creature_ai_scripts_%s VALUES (8606,12340,%u,4,0,5,1,0,%f,%u,%u,0,%u,%u,%u,0,100,0,0,'%s')", args, entry, chance, spell, spelltype, target, cooldown, cooldown, comment.c_str());
192
193 WorldDatabase.Execute(my_insert1);
194 ++count;
195 }
196
197 creature->RemoveFromWorld(false, true);
198 }
199 }
200
201 systemMessage(session, "Dumped: {} hardcoded scripts to creature_ai_scripts_dump", count);
202
203 return true;

571 uint32_t value;
572 if (sscanf(args, "%u %u %u", &bytes, &offset, &value) != 3)

574 redSystemMessage(m_session, "Command must be in format <bytes> <offset> <value>.");
575 return true;
576 }
577
578 auto unit_target = GetSelectedUnit(m_session, true);
579 if (unit_target == nullptr)
580 return true;

582 if (offset > 3)
583 return true;
584

586 {
587 case 0:
588 {
589 unit_target->setBytes0ForOffset(offset, static_cast<uint8_t>(value));
590 } break;
591 case 1:
592 {
593 unit_target->setBytes1ForOffset(offset, static_cast<uint8_t>(value));

595 case 2:
596 {
597 unit_target->setBytes2ForOffset(offset, static_cast<uint8_t>(value));
598 } break;
599 default:
600 {
601 redSystemMessage(m_session, "Bytes {} are not existent. Choose from 0, 1 or 2", bytes);
602 return true;
603 }
604 }
605
606 greenSystemMessage(m_session, "Unit Bytes {} Offset {} set to Value {}", bytes, offset, value);
607
