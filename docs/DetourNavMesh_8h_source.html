153/// Defines a polygon within a dtMeshTile object.
154/// @ingroup detour
155struct dtPoly
156{
157 /// Index to first link in linked list. (Or #DT_NULL_LINK if there is no link.)
158 unsigned int firstLink;
159
160 /// The indices of the polygon's vertices.
161 /// The actual vertices are located in dtMeshTile::verts.
162 unsigned short verts[DT_VERTS_PER_POLYGON];
163
164 /// Packed data representing neighbor polygons references and flags for each edge.
165 unsigned short neis[DT_VERTS_PER_POLYGON];
166
167 /// The user defined polygon flags.
168 unsigned short flags;
169
170 /// The number of vertices in the polygon.
171 unsigned char vertCount;
172
173 /// The bit packed area id and polygon type.
174 /// @note Use the structure's set and get methods to acess this value.
175 unsigned char areaAndtype;
176
177 /// Sets the user defined area id. [Limit: < #DT_MAX_AREAS]
178 inline void setArea(unsigned char a) { areaAndtype = (areaAndtype & 0xc0) | (a & 0x3f); }
179
180 /// Sets the polygon type. (See: #dtPolyTypes.)
181 inline void setType(unsigned char t) { areaAndtype = (areaAndtype & 0x3f) | (t << 6); }
182
183 /// Gets the user defined area id.
184 inline unsigned char getArea() const { return areaAndtype & 0x3f; }
185
186 /// Gets the polygon type. (See: #dtPolyTypes)
187 inline unsigned char getType() const { return areaAndtype >> 6; }
188};

279/// Defines a navigation mesh tile.
280/// @ingroup detour
281struct dtMeshTile
282{
283 unsigned int salt; ///< Counter describing modifications to the tile.
284
285 unsigned int linksFreeList; ///< Index to the next free link.
286 dtMeshHeader* header; ///< The tile header.
287 dtPoly* polys; ///< The tile polygons. [Size: dtMeshHeader::polyCount]
288 float* verts; ///< The tile vertices. [(x, y, z) * dtMeshHeader::vertCount]
289 dtLink* links; ///< The tile links. [Size: dtMeshHeader::maxLinkCount]
290 dtPolyDetail* detailMeshes; ///< The tile's detail sub-meshes. [Size: dtMeshHeader::detailMeshCount]
291
292 /// The detail mesh's unique vertices. [(x, y, z) * dtMeshHeader::detailVertCount]
293 float* detailVerts;
294
295 /// The detail mesh's triangles. [(vertA, vertB, vertC, triFlags) * dtMeshHeader::detailTriCount].
296 /// See dtDetailTriEdgeFlags and dtGetDetailTriEdgeFlags.
297 unsigned char* detailTris;
298
299 /// The tile bounding volume nodes. [Size: dtMeshHeader::bvNodeCount]
300 /// (Will be null if bounding volumes are disabled.)
301 dtBVNode* bvTree;
302
303 dtOffMeshConnection* offMeshCons; ///< The tile off-mesh connections. [Size: dtMeshHeader::offMeshConCount]
304
305 unsigned char* data; ///< The tile data. (Not directly accessed under normal situations.)
306 int dataSize; ///< Size of the tile data.
307 int flags; ///< Tile flags. (See: #dtTileFlags)
308 dtMeshTile* next; ///< The next free tile, or the next tile in the spatial grid.
309private:
310 dtMeshTile(const dtMeshTile&);
311 dtMeshTile& operator=(const dtMeshTile&);
312};

520 /// @{
521 /// @name Encoding and Decoding
522 /// These functions are generally meant for internal use only.
523
524 /// Derives a standard polygon reference.
525 /// @note This function is generally meant for internal use only.
526 /// @param[in] salt The tile's salt value.
527 /// @param[in] it The index of the tile.
528 /// @param[in] ip The index of the polygon within the tile.
529 inline dtPolyRef encodePolyId(unsigned int salt, unsigned int it, unsigned int ip) const
530 {
531#ifdef DT_POLYREF64
532 return ((dtPolyRef)salt << (DT_POLY_BITS+DT_TILE_BITS)) | ((dtPolyRef)it << DT_POLY_BITS) | (dtPolyRef)ip;
533#else
534 return ((dtPolyRef)salt << (m_polyBits+m_tileBits)) | ((dtPolyRef)it << m_polyBits) | (dtPolyRef)ip;
535#endif
536 }
537
538 /// Decodes a standard polygon reference.
539 /// @note This function is generally meant for internal use only.
540 /// @param[in] ref The polygon reference to decode.
541 /// @param[out] salt The tile's salt value.
542 /// @param[out] it The index of the tile.
543 /// @param[out] ip The index of the polygon within the tile.
544 /// @see #encodePolyId
545 inline void decodePolyId(dtPolyRef ref, unsigned int& salt, unsigned int& it, unsigned int& ip) const
546 {
547#ifdef DT_POLYREF64
548 const dtPolyRef saltMask = ((dtPolyRef)1<<DT_SALT_BITS)-1;
549 const dtPolyRef tileMask = ((dtPolyRef)1<<DT_TILE_BITS)-1;
550 const dtPolyRef polyMask = ((dtPolyRef)1<<DT_POLY_BITS)-1;
551 salt = (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
552 it = (unsigned int)((ref >> DT_POLY_BITS) & tileMask);
553 ip = (unsigned int)(ref & polyMask);
554#else
555 const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
556 const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
557 const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
558 salt = (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
559 it = (unsigned int)((ref >> m_polyBits) & tileMask);
560 ip = (unsigned int)(ref & polyMask);
561#endif
562 }

564 /// Extracts a tile's salt value from the specified polygon reference.
565 /// @note This function is generally meant for internal use only.
566 /// @param[in] ref The polygon reference.
567 /// @see #encodePolyId
568 inline unsigned int decodePolyIdSalt(dtPolyRef ref) const
569 {
570#ifdef DT_POLYREF64
571 const dtPolyRef saltMask = ((dtPolyRef)1<<DT_SALT_BITS)-1;
572 return (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
573#else
574 const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
575 return (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
576#endif
577 }
578
579 /// Extracts the tile's index from the specified polygon reference.
580 /// @note This function is generally meant for internal use only.
581 /// @param[in] ref The polygon reference.
582 /// @see #encodePolyId
583 inline unsigned int decodePolyIdTile(dtPolyRef ref) const
584 {
585#ifdef DT_POLYREF64
586 const dtPolyRef tileMask = ((dtPolyRef)1<<DT_TILE_BITS)-1;
587 return (unsigned int)((ref >> DT_POLY_BITS) & tileMask);

696/**
697
698@typedef dtPolyRef
699@par
700
701Polygon references are subject to the same invalidate/preserve/restore
702rules that apply to #dtTileRef's. If the #dtTileRef for the polygon's
703tile changes, the polygon reference becomes invalid.
704
705Changing a polygon's flags, area id, etc. does not impact its polygon
706reference.
707
708@typedef dtTileRef
709@par
710
711The following changes will invalidate a tile reference:
712
713- The referenced tile has been removed from the navigation mesh.
714- The navigation mesh has been initialized using a different set
715 of #dtNavMeshParams.
716
717A tile reference is preserved/restored if the tile is added to a navigation
718mesh initialized with the original #dtNavMeshParams and is added at the
719original reference location. (E.g. The lastRef parameter is used with
720dtNavMesh::addTile.)
721
722Basically, if the storage structure of a tile changes, its associated
723tile reference changes.
724
725
726@var unsigned short dtPoly::neis[DT_VERTS_PER_POLYGON]
727@par
728
729Each entry represents data for the edge starting at the vertex of the same index.
730E.g. The entry at index n represents the edge data for vertex[n] to vertex[n+1].
731
732A value of zero indicates the edge has no polygon connection. (It makes up the
733border of the navigation mesh.)
734
735The information can be extracted as follows:
736@code
737neighborRef = neis[n] & 0xff; // Get the neighbor polygon reference.
738
739if (neis[n] & #DT_EX_LINK)
740{
741 // The edge is an external (portal) edge.
742}
743@endcode
744
745@var float dtMeshHeader::bvQuantFactor
746@par
747
748This value is used for converting between world and bounding volume coordinates.
749For example:
750@code
751const float cs = 1.0f / tile->header->bvQuantFactor;
752const dtBVNode* n = &tile->bvTree[i];
753if (n->i >= 0)
754{
755 // This is a leaf node.
756 float worldMinX = tile->header->bmin[0] + n->bmin[0]*cs;
757 float worldMinY = tile->header->bmin[0] + n->bmin[1]*cs;
758 // Etc...
759}
760@endcode
761
762@struct dtMeshTile
763@par
764
765Tiles generally only exist within the context of a dtNavMesh object.
766
767Some tile content is optional. For example, a tile may not contain any
768off-mesh connections. In this case the associated pointer will be null.
769
770If a detail mesh exists it will share vertices with the base polygon mesh.
771Only the vertices unique to the detail mesh will be stored in #detailVerts.
772
773@warning Tiles returned by a dtNavMesh object are not guarenteed to be populated.
774For example: The tile at a location might not have been loaded yet, or may have been removed.
775In this case, pointers will be null. So if in doubt, check the polygon count in the
776tile's header to determine if a tile has polygons defined.
777
778@var float dtOffMeshConnection::pos[6]
779@par
780
781For a properly built navigation mesh, vertex A will always be within the bounds of the mesh.
782Vertex B is not required to be within the bounds of the mesh.
783
