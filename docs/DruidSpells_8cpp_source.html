28bool Starfall(uint8_t effectIndex, Spell* pSpell)
29{
30 Unit* m_caster = pSpell->getUnitCaster();
31 if (m_caster == NULL)
32 return true;
33
34 uint8_t am = 0;
35 for (const auto& itr : m_caster->getInRangeObjectsSet())
36 {
37 if (!itr || !itr->isCreatureOrPlayer())
38 continue;
39
40 Unit* Target = static_cast<Unit*>(itr);
41 if (Target->isValidTarget(m_caster) && m_caster->CalcDistance(itr) <= pSpell->getEffectRadius(effectIndex))
42 {
43 m_caster->castSpell(Target, pSpell->getSpellInfo()->calculateEffectValue(effectIndex, m_caster), true);
44 ++am;
45 if (am >= 2)
46 return true;
47 }
48 }
49 return true;
50}

52bool PredatoryStrikes(uint8_t effectIndex, Aura* a, bool apply)
53{
54 Unit* m_target = a->getOwner();
55 int32_t realamount = 0;
56
57 realamount = (a->getEffectDamage(effectIndex) * m_target->getLevel()) / 100;
58
59 if (apply)
60 {
61 m_target->modAttackPowerMods(realamount);
62 }
63 else
64 m_target->modAttackPowerMods(-realamount);
65
66 m_target->calculateDamage();
67
68 return true;
69}
