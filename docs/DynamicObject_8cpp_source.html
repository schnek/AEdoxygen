115void DynamicObject::updateTargets()
116{
117 if (m_aliveDuration == 0)
118 return;
119
120 if (m_aliveDuration >= 100)
121 {
122 float radius = getRadius() * getRadius();
123
124 for (const auto& itr : getInRangeObjectsSet())
125 {
126 Object* object = itr;
127 if (!object || !object->isCreatureOrPlayer() || !static_cast<Unit*>(object)->isAlive())
128 continue;
129
130 Unit* target = static_cast<Unit*>(object);
131
132 if (!m_unitCaster->isValidTarget(target, m_spellInfo))
133 continue;
134
135 // skip units already hit, their range will be tested later
136 if (m_targets.find(target->getGuid()) != m_targets.end())
137 continue;
138
139 if (getDistanceSq(target) <= radius)
140 {
141 auto aura = sSpellMgr.newAura(m_spellInfo, m_aliveDuration, m_unitCaster, target, true);
142 for (uint8_t i = 0; i < 3; ++i)
143 {
144 if (m_spellInfo->getEffect(i) == SPELL_EFFECT_PERSISTENT_AREA_AURA)
145 {
146 aura->addAuraEffect(static_cast<AuraEffect>(m_spellInfo->getEffectApplyAuraName(i)),
147 m_spellInfo->getEffectBasePoints(i) + 1, m_spellInfo->getEffectMiscValue(i), 1.0f, false, i);
148 }
149 }
150
151 target->addAura(std::move(aura));
152
153 m_targets.insert(target->getGuid());
154 }
155 }
156
157 for (auto jtr = m_targets.begin(); jtr != m_targets.end();)
158 {
159 Unit* target = getWorldMap() ? getWorldMap()->getUnit(*jtr) : nullptr;
160
161 auto jtr2 = jtr;
162 ++jtr;
163
164 if (target && getDistanceSq(target) > radius)
165 {
166 target->removeAllAurasById(m_spellInfo->getId());
167 m_targets.erase(jtr2);
168 }
169 }
170
171 m_aliveDuration -= 100;
172 }
173 else
174 {
175 m_aliveDuration = 0;
176 }
177
178 if (m_aliveDuration == 0)
179 remove();
180}

190void DynamicObject::remove()
191{
192 if (!IsInWorld())
193 {
194 delete this;
195 return;
196 }
197
198 for (auto const targetGuid : m_targets)
199 {
200 if (Unit* target = m_WorldMap->getUnit(targetGuid))
201 target->removeAllAurasById(m_spellInfo->getId());
202 }
203
204 sendGameobjectDespawnAnim();
205
206 if (IsInWorld())
207 RemoveFromWorld(true);
208
209 if (m_unitCaster && m_spellInfo->getChannelInterruptFlags() != 0)
210 {
211 m_unitCaster->setChannelObjectGuid(0);
212 m_unitCaster->setChannelSpellId(0);
213 }
214
215 delete this;
216}
