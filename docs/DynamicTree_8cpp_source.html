218bool DynamicMapTree::getIntersectionTime(const uint32_t phasemask, const G3D::Ray& ray,
219 const G3D::Vector3& endPos, float& maxDist) const
220{
221 float distance = maxDist;
222 DynamicTreeIntersectionCallback callback(phasemask);
223 impl->intersectRay(ray, callback, distance, endPos);
224 if (callback.didHit())
225 maxDist = distance;
226 return callback.didHit();
227}

229bool DynamicMapTree::getObjectHitPos(const uint32_t phasemask, const G3D::Vector3& startPos,
230 const G3D::Vector3& endPos, G3D::Vector3& resultHit,
231 float modifyDist) const
232{
233 bool result = false;
234 float maxDist = (endPos - startPos).magnitude();
235 // valid map coords should *never ever* produce float overflow, but this would produce NaNs too
236 ASSERT(maxDist < std::numeric_limits<float>::max());
237 // prevent NaN values which can cause BIH intersection to enter infinite loop
238 if (maxDist < 1e-10f)
239 {
240 resultHit = endPos;
241 return false;
242 }
243 G3D::Vector3 dir = (endPos - startPos)/maxDist; // direction with length of 1
244 G3D::Ray ray(startPos, dir);
245 float dist = maxDist;
246 if (getIntersectionTime(phasemask, ray, endPos, dist))
247 {
248 resultHit = startPos + dir * dist;
249 if (modifyDist < 0)
250 {
251 if ((resultHit - startPos).magnitude() > -modifyDist)
252 resultHit = resultHit + dir*modifyDist;
253 else
254 resultHit = startPos;
255 }
256 else
257 resultHit = resultHit + dir*modifyDist;
258
259 result = true;
260 }
261 else
262 {
263 resultHit = endPos;
264 result = false;
265 }
266 return result;
267}

269bool DynamicMapTree::isInLineOfSight(float x1, float y1, float z1, float x2, float y2, float z2, uint32_t phasemask) const
270{
271 G3D::Vector3 v1(x1, y1, z1), v2(x2, y2, z2);
272
273 float maxDist = (v2 - v1).magnitude();
274
275 if (!G3D::fuzzyGt(maxDist, 0) )
276 return true;
277
278 G3D::Ray r(v1, (v2-v1) / maxDist);
279 DynamicTreeIntersectionCallback callback(phasemask);
280 impl->intersectRay(r, callback, maxDist, v2);
281
282 return !callback.did_hit;
283}

285float DynamicMapTree::getHeight(float x, float y, float z, float maxSearchDist, uint32_t phasemask) const
286{
287 G3D::Vector3 v(x, y, z);
288 G3D::Ray r(v, G3D::Vector3(0, 0, -1));
289 DynamicTreeIntersectionCallback callback(phasemask);
290 impl->intersectZAllignedRay(r, callback, maxSearchDist);
291
292 if (callback.didHit())
293 return v.z - maxSearchDist;
294 else
295 return -G3D::finf();

298bool DynamicMapTree::getAreaInfo(float x, float y, float& z, uint32_t phasemask, uint32_t& flags, int32_t& adtId, int32_t& rootId, int32_t& groupId) const
299{
300 G3D::Vector3 v(x, y, z + 0.5f);
301 DynamicTreeAreaInfoCallback intersectionCallBack(phasemask);
302 impl->intersectPoint(v, intersectionCallBack);
303 if (intersectionCallBack.GetAreaInfo().result)
304 {
305 flags = intersectionCallBack.GetAreaInfo().flags;
306 adtId = intersectionCallBack.GetAreaInfo().adtId;
307 rootId = intersectionCallBack.GetAreaInfo().rootId;
308 groupId = intersectionCallBack.GetAreaInfo().groupId;
309 z = intersectionCallBack.GetAreaInfo().ground_Z;
310 return true;
311 }
312 return false;
313}

315void DynamicMapTree::getAreaAndLiquidData(float x, float y, float z, uint32_t phasemask, uint8_t reqLiquidType, VMAP::AreaAndLiquidData& data) const
316{
317 G3D::Vector3 v(x, y, z + 0.5f);
318 DynamicTreeLocationInfoCallback intersectionCallBack(phasemask);
319 impl->intersectPoint(v, intersectionCallBack);
320 if (intersectionCallBack.GetLocationInfo().hitModel)
321 {
322 data.floorZ = intersectionCallBack.GetLocationInfo().ground_Z;
323 uint32_t liquidType = intersectionCallBack.GetLocationInfo().hitModel->GetLiquidType();
324 float liquidLevel;
325 if (!reqLiquidType || VMAP::VMapFactory::createOrGetVMapManager()->GetLiquidFlagsPtr(liquidType) & reqLiquidType)
326 if (intersectionCallBack.GetHitModel()->getLiquidLevel(v, intersectionCallBack.GetLocationInfo(), liquidLevel))
327 data.liquidInfo.emplace(liquidType, liquidLevel);
328
329 data.areaInfo.emplace(0,
330 intersectionCallBack.GetLocationInfo().rootId,
331 intersectionCallBack.GetLocationInfo().hitModel->GetWmoID(),
332 intersectionCallBack.GetLocationInfo().hitModel->GetMogpFlags());
333 }
