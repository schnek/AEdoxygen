77 bool EquipmentSetMgr::LoadfromDB(QueryResult* result)
78 {
79 if (result == NULL)
80 return false;
81
82 uint32_t setcount = 0;
83
84 do
85 {
86 if (setcount >= 10)
87 {
88 sLogger.failure("There were more than 10 equipment sets for GUID: {}", ownerGUID);
89 return true;
90 }
91
92 Field* fields = result->Fetch();
93
94 EquipmentSets.try_emplace(fields[1].asUint32(), Util::LazyInstanceCreator([fields] {
95 return std::make_unique<EquipmentSet>(fields);
96 }));
97
98 setcount++;
99 }
100 while (result->NextRow());
101
102 return true;
103 }

105 bool EquipmentSetMgr::SavetoDB(QueryBuffer* buf)
106 {
107 if (buf == NULL)
108 return false;
109
110 std::stringstream ds;
111 ds << "DELETE FROM equipmentsets WHERE ownerguid = ";
112 ds << ownerGUID;
113
114 buf->AddQueryNA(ds.str().c_str());
115
116 for (EquipmentSetStorage::iterator itr = EquipmentSets.begin(); itr != EquipmentSets.end(); ++itr)
117 {
118 const auto& set = itr->second;
119
120 std::stringstream ss;
121
122 ss << "INSERT INTO equipmentsets VALUES('";
123 ss << ownerGUID << "','";
124 ss << set->SetGUID << "','";
125 ss << set->SetID << "','";
126 ss << CharacterDatabase.EscapeString(set->SetName) << "','";
127 ss << set->IconName << "'";
128
129 for (uint32_t j = 0; j < set->ItemGUID.size(); ++j)
130 {
131 ss << ",'";
132 ss << set->ItemGUID[j];
133 ss << "'";
134 }
135
136 ss << ")";
137
138 buf->AddQueryNA(ss.str().c_str());
139 }
140
141 return true;
142 }
