15ErekemAI::ErekemAI(Creature* pCreature) : CreatureAIScript(pCreature)
16{
17 // Instance Script
18 mInstance = getInstanceScript();
19
20 // Spells
21 addAISpell(Erekem::SPELL_EARTH_SHIELD, 100.0f, TARGET_SELF);
22 addAISpell(Erekem::SPELL_BLOODLUST, 100.0f, TARGET_SELF);
23 addAISpell(Erekem::SPELL_LIGHTNING_BOLT, 44.0f, TARGET_RANDOM_SINGLE, 0, 2);
24 addAISpell(Erekem::SPELL_EARTH_SHOCK, 44.0f, TARGET_RANDOM_SINGLE, 0, 8);
25
26 // Only Casted When hes Guards are Dead so make Chance 0 and Handle in Update
27 if (windfury = addAISpell(Erekem::SPELL_WINDFURY, 0.0f, TARGET_ATTACKING))
28 {
29 windfury->mIsTriggered = true;
30 windfury->setAvailableForScriptPhase({ 1 });
31 }
32
33 breakBonds = addAISpell(Erekem::SPELL_BREAK_BONDS, 0.0f, TARGET_SELF);
34
35 // Only Cast This on The Lowest Health In Range Friendly Target
36 addAISpell(Erekem::SPELL_CHAIN_HEAL, 33.0f, 3, [this]() { return getBestUnitTarget(TargetFilter_WoundedFriendlyLowestHealthInRange, 0.0f, 40.0f); });
37 addAISpell(Erekem::SPELL_EARTH_SHIELD, 100.0f, 20, [this]() { return getBestUnitTarget(TargetFilter_WoundedFriendlyLowestHealthInRange, 0.0f, 30.0f); });
38
39 if (CreatureAISpells* stormstrike = addAISpell(Erekem::SPELL_STORMSTRIKE, 100.0f, TARGET_ATTACKING))
40 stormstrike->setAvailableForScriptPhase({ 2 });
41
42 // Emotes
43 addEmoteForEvent(Event_OnCombatStart, Erekem::SAY_AGGRO);
44 addEmoteForEvent(Event_OnTargetDied, Erekem::SAY_SLAY1);
45 addEmoteForEvent(Event_OnTargetDied, Erekem::SAY_SLAY2);
46 addEmoteForEvent(Event_OnTargetDied, Erekem::SAY_SLAY3);
47 addEmoteForEvent(Event_OnDied, Erekem::SAY_DEATH);
48}

60void ErekemAI::AIUpdate(unsigned long /*time_passed*/)
61{
62 if (mInstance)
63 {
64 for (uint32_t i = DATA_EREKEM_GUARD_1; i <= DATA_EREKEM_GUARD_2; ++i)
65 {
66 Creature* guard = mInstance->GetCreatureByGuid(mInstance->getLocalData(i));
67
68 if (guard && guard->isAlive() && checkGuardAuras(guard))
69 {
70 _castAISpell(breakBonds);
71 return;
72 }
73 }
74 }
75
76 if (getScriptPhase() == 1 && !checkGuardsAlive())
77 {
78 setScriptPhase(2);
79 _castAISpell(windfury);
80 }
81}

98bool ErekemAI::checkGuardAuras(Creature* guard)
99{
100 uint32_t mechanicMask = MOVEMENT_IMPAIRMENTS_AND_LOSS_OF_CONTROL_MASK;
101
102 static std::list<AuraEffect> const AuraImmunityList =
103 {
104 SPELL_AURA_MOD_STUN,
105 SPELL_AURA_MOD_DECREASE_SPEED,
106 SPELL_AURA_MOD_ROOT,
107 SPELL_AURA_MOD_CONFUSE,
108 SPELL_AURA_MOD_FEAR
109 };
110
111 if (guard->hasAuraWithMechanic(SpellMechanic(mechanicMask)))
112 return true;
113
114 for (AuraEffect type : AuraImmunityList)
115 if (guard->hasAuraWithAuraEffect(type))
116 return true;
117
118 return false;
119}
