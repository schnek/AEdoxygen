230 EventableObjectHolder* GetEventHolder(int32_t InstanceId)
231 {
232 std::lock_guard<std::mutex> guard(holderLock);
233
234 HolderMap::iterator itr = mHolders.find(InstanceId);
235
236 if (itr == mHolders.end())
237 {
238 return nullptr;
239 }
240
241 return itr->second;
242 }
243
244 void AddEventHolder(EventableObjectHolder* holder, int32_t InstanceId)
245 {
246 std::lock_guard<std::mutex> guard(holderLock);
247
248 mHolders.insert(HolderMap::value_type(InstanceId, holder));
249 }
250
251 void RemoveEventHolder(int32_t InstanceId)
252 {
253 std::lock_guard<std::mutex> guard(holderLock);
254
255 mHolders.erase(InstanceId);
256 }
257
258 void RemoveEventHolder(EventableObjectHolder* holder)
259 {
260 std::lock_guard<std::mutex> guard(holderLock);
261
262 HolderMap::iterator itr = mHolders.begin();
263 for (; itr != mHolders.end(); ++itr)
264 {
265 if (itr->second == holder)
266 {
267 mHolders.erase(itr);
268 return;
269 }
270 }
271 }
272
273 protected:
274 HolderMap mHolders;
275 std::mutex holderLock;
