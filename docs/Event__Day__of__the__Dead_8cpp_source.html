24void Catrina(Player* pPlayer, Unit* pUnit)
25{
26 if (pPlayer == nullptr || pUnit == nullptr)
27 {
28 return;
29 }
30
31 if (pPlayer->hasAurasWithId(SPELL_HONOR_THE_DEAD) == false)
32 {
33 if (pPlayer->isDead() == false)
34 {
35 pUnit->castSpell(pPlayer, SPELL_HONOR_THE_DEAD, true);
36 }
37 }
38
39#if VERSION_STRING > TBC

43 }
44#endif
45}
