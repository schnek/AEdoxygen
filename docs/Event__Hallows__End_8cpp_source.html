146 {

152

232 if (!proto)
233 return;
234
235 slotresult = pPlayer->getItemInterface()->FindFreeInventorySlot(proto);
236
237 if (!slotresult.Result)
238 {
239 pPlayer->getItemInterface()->buildInventoryChangeError(NULL, NULL, INV_ERR_INVENTORY_FULL);
240 return;
241 }
242 else
243 {
244 if (pPlayer->getItemInterface()->GetItemCount(32971, false) == 0)
245 {
246 auto itm = sObjectMgr.createItem(32971, pPlayer);
247 if (itm == nullptr)
248 return;
249
250 const auto [result, returnedItem] = pPlayer->getItemInterface()->SafeAddItem(std::move(itm), slotresult.ContainerSlot, slotresult.Slot);
251 if (!result)
252 {
253 DLLLogDetail("Error while adding item %u to player %s", returnedItem->getEntry(), pPlayer->getName().c_str());
254 }
255 }
256 else
257 {
258 pPlayer->getItemInterface()->buildInventoryChangeError(NULL, NULL, INV_ERR_INVENTORY_FULL);
