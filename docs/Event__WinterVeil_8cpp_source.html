50void WinterReveler(Player* pPlayer, Unit* pUnit)
51{
52 if (pUnit->getEntry() == 15760)
53 {
54 uint32_t Winteritem = 0;
55 SlotResult slotresult;
56
57 uint32_t chance = Util::getRandomUInt(2);
58 switch (chance)
59 {
60 case 0:
61 {
62 ItemProperties const* proto = sMySQLStore.getItemProperties(21212);
63 if (!proto)
64 return;
65
66 slotresult = pPlayer->getItemInterface()->FindFreeInventorySlot(proto);
67 Winteritem = 21212;
68 }
69 break;
70 case 1:
71 {
72 ItemProperties const* proto = sMySQLStore.getItemProperties(21519);
73 if (!proto)
74 return;
75
76 slotresult = pPlayer->getItemInterface()->FindFreeInventorySlot(proto);
77 Winteritem = 21519;
78 }
79 break;
80 case 2:
81 {
82 ItemProperties const* proto = sMySQLStore.getItemProperties(34191);
83 if (!proto)
84 return;
85
86 slotresult = pPlayer->getItemInterface()->FindFreeInventorySlot(proto);
87 Winteritem = 34191;
88 }
89 break;
90
91 }
92
93 if (!slotresult.Result)
94 {
95 pPlayer->getItemInterface()->buildInventoryChangeError(NULL, NULL, INV_ERR_INVENTORY_FULL);
96 return;
97 }
98 else
99 {
100 auto item = sObjectMgr.createItem(Winteritem, pPlayer);
101 if (item == nullptr)
102 return;
103
104 item->setStackCount(5);
105 const auto [item_add_result, returnedItem] = pPlayer->getItemInterface()->SafeAddItem(std::move(item), slotresult.ContainerSlot, slotresult.Slot);
106 if (!item_add_result)
107 {
108 DLLLogDetail("Error while adding item %u to player %s", returnedItem->getEntry(), pPlayer->getName().c_str());
109 }
110 else
111 {
112 pUnit->castSpell(pPlayer, 26218, true);
113 }
114 }
115 }
