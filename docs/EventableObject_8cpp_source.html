50void EventableObject::event_AddEvent(std::shared_ptr<TimedEvent> ptr)
51{
52 m_lock.lock();
53
54 if (m_holder == nullptr)
55 {
56 m_event_Instanceid = event_GetInstanceID();
57 m_holder = sEventMgr.GetEventHolder(m_event_Instanceid);
58
59 if (m_holder == nullptr)
60 {
61 // We still couldn't find an eventholder for us so let's run in WorldRunnable
62 m_event_Instanceid = WORLD_INSTANCE;
63 m_holder = sEventMgr.GetEventHolder(m_event_Instanceid);
64 }
65 }
66
67 // We still couldn't find an event holder for ourselves :(
68 if (m_holder == nullptr)
69 {
70 sLogger.failure("EventableObject::event_AddEvent not able to find a event holder, return!");
71 m_lock.unlock();
72 return;
73 }
74
75 // If we are flagged not to run in WorldRunnable then we won't!
76 // This is much better than adding us to the eventholder and removing on an update
77 if (m_event_Instanceid == WORLD_INSTANCE && (ptr->eventFlag & EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT))
78 {
79 m_lock.unlock();
80 return;
81 }
82
83 ptr->instanceId = m_event_Instanceid;
84 m_events.emplace(ptr->eventType, ptr);
85 m_lock.unlock();
86
87 /* Add to event manager */
88 m_holder->AddEvent(std::move(ptr));
89}

290void EventableObjectHolder::Update(time_t time_difference)
291{
292 std::lock_guard lock(m_lock);
293
294 // Insert any pending objects in the insert pool.
295 m_insertPoolLock.lock();
296 InsertableQueue::iterator iqi;
297 InsertableQueue::iterator iq2 = m_insertPool.begin();
298 while (iq2 != m_insertPool.end())
299 {
300 iqi = iq2++;
301 if (!((*iqi)->deleted || (*iqi)->instanceId != mInstanceId))
302 m_events.push_back(std::move((*iqi)));
303
304 m_insertPool.erase(iqi);
305 }
306 m_insertPoolLock.unlock();
307
308 // Now we can proceed normally.
309 EventList::iterator itr = m_events.begin();
310 EventList::iterator it2;
311 TimedEvent* ev;
312
313 while (itr != m_events.end())
314 {
315 it2 = itr++;
316
317 if ((*it2)->instanceId != mInstanceId || (*it2)->deleted)
318 {
319 // remove from this list.
320 m_events.erase(it2);
321
322 continue;
323 }
324
325 // Event Update Procedure
326 ev = it2->get();
327
328 if (ev->currTime <= time_difference)
329 {
330 // execute the callback
331 if (ev->eventFlag & EVENT_FLAG_DELETES_OBJECT)
332 {
333 ev->deleted = true;
334 ev->cb->execute();
335 m_events.erase(it2);
336 continue;
337 }
338 else
339 ev->cb->execute();
340
341 // check if the event is expired now.
342 if (ev->repeats && --ev->repeats == 0)
343 {
344 // Event expired :>
345
346 /* remove the event from the object */
347 /*obj = (EventableObject*)ev->obj;
348 obj->event_RemoveByPointer(ev);*/
349
350 /* remove the event from here */
351 ev->deleted = true;
352 m_events.erase(it2);
353
354 continue;
355 }
356 else if (ev->deleted)
357 {
358 // event is now deleted
359 m_events.erase(it2);
360 continue;
361 }
362
363 // event has to repeat again, reset the timer
364 ev->currTime = ev->msTime;
365 }
366 else
367 {
368 // event is still "waiting", subtract time difference
369 ev->currTime -= time_difference;
370 }
371 }

374void EventableObject::event_Relocate()
375{
376 std::lock_guard lock(m_lock);
377
378 EventableObjectHolder* nh = sEventMgr.GetEventHolder(event_GetInstanceID());
379 if (nh != m_holder)
380 {
381 // whee, we changed event holder :>
382 // doing this will change the instanceid on all the events, as well as add to the new holder.
383
384 //If nh is NULL then we were removed from world. There's no reason to be added to WORLD_INSTANCE EventMgr, let's just wait till something will add us again to world.
385 if (nh == NULL)
386 {
387 //set instaceId to 0 to each event of this EventableObject, so EventableObjectHolder::Update() will remove them from its EventList.
388 for (EventMap::iterator itr = m_events.begin(); itr != m_events.end(); ++itr)
389 {
390 itr->second->instanceId = 0;
391 }
392 // reset our instance id.
393 m_event_Instanceid = 0;
394 }
395 else
396 {
397 nh->AddObject(this);
398 // reset our instance id
399 m_event_Instanceid = nh->GetInstanceID();
400 }
401 // reset our m_holder pointer
402 m_holder = nh;
403 }

434void EventableObjectHolder::AddObject(EventableObject* obj)
435{
436 // transfer all of this objects events into our holder
437 if (!m_lock.try_lock())
438 {
439 // The other thread is obviously occupied. We have to use an insert pool here, otherwise
440 // if 2 threads relocate at once we'll hit a deadlock situation.
441 m_insertPoolLock.lock();
442 EventMap::iterator it2;
443
444 for (EventMap::iterator itr = obj->m_events.begin(); itr != obj->m_events.end(); ++itr)
445 {
446 // ignore deleted events (shouldn't be any in here, actually)
447 if (itr->second->deleted)
448 {
449 /*it2 = itr++;
450 itr->second->DecRef();
451 obj->m_events.erase(it2);*/
452 continue;
453 }
454
455 itr->second->instanceId = mInstanceId;
456 m_insertPool.push_back(itr->second);
457 }
458
459 // Release the insert pool.
460 m_insertPoolLock.unlock();
461 }
462 else
463 {
464 for (EventMap::iterator itr = obj->m_events.begin(); itr != obj->m_events.end(); ++itr)
465 {
466 // ignore deleted events
467 if (itr->second->deleted)
468 continue;
469
470 itr->second->instanceId = mInstanceId;
471 m_events.push_back(itr->second);
472 }
473 m_lock.unlock();
