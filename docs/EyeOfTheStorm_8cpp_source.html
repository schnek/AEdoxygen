260bool EyeOfTheStorm::HookHandleRepop(Player* plr)
261{
262 uint32_t t = plr->getTeam();
263 float dist = 999999.0f;
264 float distcur;
265 LocationVector dest;
266
267 dest.ChangeCoords({ EOTSStartLocations[t][0], EOTSStartLocations[t][1], EOTSStartLocations[t][2] });
268
269 for (uint8_t i = 0; i < EOTS_TOWER_COUNT; ++i)
270 {
271 if (m_CPBanner[i] &&
272 (((m_CPBanner[i]->getEntry() == EOTS_BANNER_ALLIANCE) && (t == TEAM_ALLIANCE)) ||
273 ((m_CPBanner[i]->getEntry() == EOTS_BANNER_HORDE) && (t == TEAM_HORDE))))
274 {
275 distcur = plr->GetPositionNC().Distance2DSq({ EOTSGraveyardLocations[i][0], EOTSGraveyardLocations[i][1] });
276 if (distcur < dist)
277 {
278 dist = distcur;
279 dest.ChangeCoords({ EOTSGraveyardLocations[i][0], EOTSGraveyardLocations[i][1], EOTSGraveyardLocations[i][2] });
280 }
281 }
282 }
283
284 plr->safeTeleport(plr->GetMapId(), plr->GetInstanceID(), dest);
285 return true;

288void EyeOfTheStorm::HookOnAreaTrigger(Player* plr, uint32_t id)
289{
290 int32_t tid = -1;
291 int32_t bonusid = -1;
292 switch (id)
293 {
294 case 4476: // BE Tower
295 tid = EOTS_TOWER_BE;
296 break;
297 case 4568: // BE Tower bonus
298 bonusid = EOTS_TOWER_BE;
299 break;
300 case 4514: // Fel Reaver Tower
301 tid = EOTS_TOWER_FELREAVER;
302 break;
303 case 4569: // Fel Reaver Tower bonus
304 bonusid = EOTS_TOWER_FELREAVER;
305 break;
306 case 4518: // Draenei Tower
307 tid = EOTS_TOWER_DRAENEI;
308 break;
309 case 4571: // Draenei Tower bonus
310 bonusid = EOTS_TOWER_DRAENEI;
311 break;
312 case 4516: // Mage Tower
313 tid = EOTS_TOWER_MAGE;
314 break;
315 case 4570: // Mage Tower bonus
316 bonusid = EOTS_TOWER_MAGE;
317 break;
318 case 4512:
319 case 4515:
320 case 4517:
321 case 4519:
322 case 4530:
323 case 4531:
324 case 5866:
325 break;
326 default:
327 DLLLogDetail("Encountered unhandled areatrigger id %u", id);
328 return;
329 break;
330 }
331
332 if (plr->isDead()) // on ne buff pas les joueurs morts ;)
333 return;
334
335 if (bonusid > -1)
336 {
337 uint32_t spellid = 0;
338 uint32_t x = (uint32_t)bonusid;
339 if (EOTSm_buffs[x] && EOTSm_buffs[x]->IsInWorld())
340 {
341 spellid = EOTSm_buffs[x]->GetGameObjectProperties()->raw.parameter_3;
342 SpellInfo const* sp = sSpellMgr.getSpellInfo(spellid);
343 if (sp)
344 {
345 Spell* pSpell = sSpellMgr.newSpell(plr, sp, true, nullptr);
346 SpellCastTargets targets(plr->getGuid());
347 pSpell->prepare(&targets);
348 }
349 EOTSm_buffs[x]->despawn(0, EOTS_BUFF_RESPAWN_TIME);
350 }
351 }
352
353 if (tid < 0)
354 return;
355
356 if (!m_hasStarted)
357 {
358 Anticheat_Log->writefromsession(plr->getSession(), "%s tried to hook the flag in eye of the storm before battleground (ID %u) started.", plr->getName().c_str(), this->m_id);
359 sendChatMessage(CHAT_MSG_BG_EVENT_NEUTRAL, plr->getGuid(), "%s will be removed from the game for cheating.", plr->getName().c_str());
360 // Remove player from battleground.
361 this->removePlayer(plr, false);
362 // Kick player from server.
363 plr->kickFromServer(TimeVarsMs::Second * 6);
364 return;
365 }
366
367 const auto team = plr->getTeam();
368 if (plr->getGuidLow() != m_flagHolder)
369 return;
370
371 int32_t val;
372
373 uint32_t towers = 0;
374 if (team == TEAM_ALLIANCE)
375 val = EOTS_BANNER_ALLIANCE;
376 else
377 val = EOTS_BANNER_HORDE;
378
379 if (!m_CPBanner[tid] || m_CPBanner[tid]->getEntry() != static_cast<uint32_t>(val))
380 return; // not captured by our team
381
382 for (uint8_t i = 0; i < EOTS_TOWER_COUNT; ++i)
383 {
384 if (m_CPBanner[i] && m_CPBanner[i]->getEntry() == static_cast<uint32_t>(val))
385 towers++;
386 }
387
388 /*
389 Points from flag captures
390 * 1 towers controlled = 75 points
391 * 2 towers controlled = 85 points
392 * 3 towers controlled = 100 points
393 * 4 towers controlled = 500 points
394 */
395
396 // 25 is guessed
397 const static uint32_t points[5] = { 25, 75, 85, 100, 500 };
398 const char* msgs[2] = { "The Alliance have captured the flag.", "The Horde have captured the flag." };
399
400 sendChatMessage(CHAT_MSG_BG_EVENT_ALLIANCE + team, 0, msgs[team]);
401 GivePoints(team, points[towers]);
402
403 DropFlag2(plr, id);
404 setWorldState(EOTS_NETHERWING_FLAG_READY, 1);
405
406 plr->removeAllAurasById(EOTS_NETHERWING_FLAG_SPELL);
407 plr->m_bgScore.MiscData[BattlegroundDef::EOTS_FLAGS_CAPTURED]++;
408 updatePvPData();
409}

809void EyeOfTheStorm::GeneratePoints()
810{
811 uint32_t towers[2] = { 0, 0 };
812
813 /*
814 # Unlike Arathi Basin, points are always generated in 2 seconds intervals no matter how many towers are controlled by both teams.
815 # Each claimed tower generates victory points for the controlling team. The more towers your team owns, the faster your team gains points
816
817 * 1 tower controlled = 1 point/tick (0.5 points per second)
818 * 2 towers controlled = 2 points/tick (1 point per second)
819 * 3 towers controlled = 5 points/tick (2.5 points per second)
820 * 4 towers controlled = 10 points/tick (5 points per second)
821
822 */
823 uint32_t pointspertick[5] = { 0, 1, 2, 5, 10 };
824
825 for (uint8_t i = 0; i < EOTS_TOWER_COUNT; ++i)
826 {
827 if (m_CPBanner[i] && m_CPBanner[i]->getEntry() == EOTS_BANNER_ALLIANCE)
828 towers[0]++;
829 else if (m_CPBanner[i] && m_CPBanner[i]->getEntry() == EOTS_BANNER_HORDE)
830 towers[1]++;
831 }
832
833 for (uint8_t i = 0; i < 2; ++i)
834 {
835 if (towers[i] == 0)
836 {
837 //printf("EOTS: No points on team %u\n", i);
838 continue;
839 }
840
841 if (GivePoints(i, pointspertick[towers[i]]))
842 return;
843 }
844}

846bool EyeOfTheStorm::GivePoints(uint32_t team, uint32_t points)
847{
848 //printf("EOTS: Give team %u %u points.\n", team, points);
849
850 m_points[team] += points;
851 if ((m_points[team] - m_lastHonorGainPoints[team]) >= resourcesToGainBH)
852 {
853 std::lock_guard<std::recursive_mutex> lock(m_mutex);
854
855 uint32_t honorToAdd = m_honorPerKill;
856 for (std::set<Player*>::iterator itr = m_players[team].begin(); itr != m_players[team].end(); ++itr)
857 {
858 (*itr)->m_bgScore.BonusHonor += honorToAdd;
859 HonorHandler::AddHonorPointsToPlayer((*itr), honorToAdd);
860 }
861
862 updatePvPData();
863 m_lastHonorGainPoints[team] += resourcesToGainBH;
864 }
865
866 if (m_points[team] >= 1600)
867 {
868 m_points[team] = 1600;
869
870 sEventMgr.RemoveEvents(this);
871 sEventMgr.AddEvent(static_cast<Battleground*>(this), &Battleground::close, EVENT_BATTLEGROUND_CLOSE, 120000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
872
873 setWorldState(WORLDSTATE_EOTS_ALLIANCE_VICTORYPOINTS + team, m_points[team]);
874
875 this->endBattleground(team == TEAM_ALLIANCE ? TEAM_ALLIANCE : TEAM_HORDE);
876 return true;
877 }
878
879 setWorldState(WORLDSTATE_EOTS_ALLIANCE_VICTORYPOINTS + team, m_points[team]);
880 return false;
881}
