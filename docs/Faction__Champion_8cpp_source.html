28void ChampionControllerAI::DoAction(int32_t action)
29{
30 switch (action)
31 {
32 case champions::ACTION_SUMMON:
33 {
34 summonChampions();
35 } break;
36 case champions::ACTION_START:
37 {
38 for (SummonList::iterator i = summons.begin(); i != summons.end(); ++i)
39 {
40 if (Creature* summon = getCreature()->getWorldMapCreature(*i))
41 {
42 summon->getAIInterface()->setReactState(REACT_AGGRESSIVE);
43 summon->removeUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
44 summon->getAIInterface()->setIgnorePlayerCombat(false);
45 }
46 }
47 } break;
48 case champions::ACTION_INPROGRESS:
49 {
50 if (!mInProgress)
51 {
52 mChampionsNotStarted = 0;
53 mChampionsFailed = 0;
54 mChampionsKilled = 0;
55 mInProgress = true;
56
57 getInstanceScript()->setBossState(DATA_FACTION_CRUSADERS, EncounterStates::InProgress);
58 }
59 } break;
60 case champions::ACTION_FAILED:
61 {
62 mChampionsFailed++;
63 const auto championsKilled = static_cast<uint32_t>(mChampionsFailed + mChampionsKilled);
64 if (championsKilled >= summons.size())
65 {
66 getInstanceScript()->setBossState(DATA_FACTION_CRUSADERS, EncounterStates::Failed);
67 summons.despawnAll();
68 despawn();
69 }
70 } break;
71 case champions::ACTION_PERFORMED:
72 {
73 mChampionsKilled++;
74 if (mChampionsKilled == 1)
75 {
76 getInstanceScript()->setLocalData(DATA_FACTION_CRUSADERS, 0); // Used in Resilience will Fix Achievement
77 }
78 else if (mChampionsKilled >= summons.size())
79 {
80 getInstanceScript()->setBossState(DATA_FACTION_CRUSADERS, EncounterStates::Performed);
81 summons.despawnAll();
82 despawn();
83 }
84 } break;
85 }

88void ChampionControllerAI::summonChampions()
89{
90 PlayerTeam teamInInstance = getInstanceScript()->getInstance()->getTeamIdInInstance();
91
92 std::vector<LocationVector> vChampionJumpOrigin;
93 if (teamInInstance == TEAM_ALLIANCE)
94 for (uint8_t i = 0; i < 5; i++)
95 vChampionJumpOrigin.push_back(FactionChampionLoc[i]);
96 else
97 for (uint8_t i = 5; i < 10; i++)
98 vChampionJumpOrigin.push_back(FactionChampionLoc[i]);
99
100 std::vector<LocationVector> vChampionJumpTarget;
101 for (uint8_t i = 10; i < 20; i++)
102 vChampionJumpTarget.push_back(FactionChampionLoc[i]);
103
104 std::vector<uint32_t> vChampionEntries = selectChampions(teamInInstance);
105
106 for (uint8_t i = 0; i < vChampionEntries.size(); ++i)
107 {
108 uint8_t pos = Util::getRandomUInt(0, static_cast<uint32_t>(vChampionJumpTarget.size() - 1));
109 const auto maxChampionOrigin = static_cast<uint32_t>(vChampionJumpOrigin.size() - 1);
110 if (Creature* champion = summonCreature(vChampionEntries[i], vChampionJumpOrigin[Util::getRandomUInt(0, maxChampionOrigin)], CreatureSummonDespawnType::MANUAL_DESPAWN))
111 {
112 summons.summon(champion);
113 champion->getAIInterface()->setReactState(REACT_PASSIVE);
114 champion->setUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
115 champion->getAIInterface()->setIgnorePlayerCombat(false);
116
117 if (teamInInstance == TEAM_ALLIANCE)
118 {
119 champion->SetSpawnLocation(vChampionJumpTarget[pos].getPositionX(), vChampionJumpTarget[pos].getPositionY(), vChampionJumpTarget[pos].getPositionZ(), 0);
120 champion->getMovementManager()->moveJump(vChampionJumpTarget[pos], 20.0f, 20.0f);
121 champion->SetOrientation(0);
122 }
123 else
124 {
125 champion->SetSpawnLocation((ToCCommonLoc[1].getPositionX() * 2) - vChampionJumpTarget[pos].getPositionX(), vChampionJumpTarget[pos].getPositionY(), vChampionJumpTarget[pos].getPositionZ(), 3);
126 champion->getMovementManager()->moveJump((ToCCommonLoc[1].getPositionX() * 2) - vChampionJumpTarget[pos].getPositionX(), vChampionJumpTarget[pos].getPositionY(), vChampionJumpTarget[pos].getPositionZ(), vChampionJumpTarget[pos].getOrientation(), 20.0f, 20.0f);
127 champion->SetOrientation(3);
128 }
129 }
130 vChampionJumpTarget.erase(vChampionJumpTarget.begin() + pos);
131 }

134std::vector<uint32_t> ChampionControllerAI::selectChampions(PlayerTeam playerTeam)
135{
136 std::vector<uint32_t> vHealersEntries;
137 vHealersEntries.clear();
138 vHealersEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_DRUID_RESTORATION : NPC_ALLIANCE_DRUID_RESTORATION);
139 vHealersEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_PALADIN_HOLY : NPC_ALLIANCE_PALADIN_HOLY);
140 vHealersEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_PRIEST_DISCIPLINE : NPC_ALLIANCE_PRIEST_DISCIPLINE);
141 vHealersEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_SHAMAN_RESTORATION : NPC_ALLIANCE_SHAMAN_RESTORATION);
142
143 std::vector<uint32_t> vOtherEntries;
144 vOtherEntries.clear();
145 vOtherEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_DEATH_KNIGHT : NPC_ALLIANCE_DEATH_KNIGHT);
146 vOtherEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_HUNTER : NPC_ALLIANCE_HUNTER);
147 vOtherEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_MAGE : NPC_ALLIANCE_MAGE);
148 vOtherEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_ROGUE : NPC_ALLIANCE_ROGUE);
149 vOtherEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_WARLOCK : NPC_ALLIANCE_WARLOCK);
150 vOtherEntries.push_back(playerTeam == TEAM_ALLIANCE ? NPC_HORDE_WARRIOR : NPC_ALLIANCE_WARRIOR);
151
152 uint8_t healersSubtracted = 2;
153
154 // Max Allowed Healers
155 if (getInstanceScript()->getInstance()->getDifficulty() == InstanceDifficulty::RAID_25MAN_NORMAL || getInstanceScript()->getInstance()->getDifficulty() == InstanceDifficulty::RAID_25MAN_HEROIC)
156 healersSubtracted = 1;
157
158 // Fill Healers
159 for (uint8_t i = 0; i < healersSubtracted; ++i)
160 {
161 uint8_t pos = Util::getRandomUInt(0, static_cast<uint32_t>(vHealersEntries.size() - 1));
162 switch (vHealersEntries[pos])
163 {
164 case NPC_ALLIANCE_DRUID_RESTORATION:
165 vOtherEntries.push_back(NPC_ALLIANCE_DRUID_BALANCE);
166 break;
167 case NPC_HORDE_DRUID_RESTORATION:
168 vOtherEntries.push_back(NPC_HORDE_DRUID_BALANCE);
169 break;
170 case NPC_ALLIANCE_PALADIN_HOLY:
171 vOtherEntries.push_back(NPC_ALLIANCE_PALADIN_RETRIBUTION);
172 break;
173 case NPC_HORDE_PALADIN_HOLY:
174 vOtherEntries.push_back(NPC_HORDE_PALADIN_RETRIBUTION);
175 break;
176 case NPC_ALLIANCE_PRIEST_DISCIPLINE:
177 vOtherEntries.push_back(NPC_ALLIANCE_PRIEST_SHADOW);
178 break;
179 case NPC_HORDE_PRIEST_DISCIPLINE:
180 vOtherEntries.push_back(NPC_HORDE_PRIEST_SHADOW);
181 break;
182 case NPC_ALLIANCE_SHAMAN_RESTORATION:
183 vOtherEntries.push_back(NPC_ALLIANCE_SHAMAN_ENHANCEMENT);
184 break;
185 case NPC_HORDE_SHAMAN_RESTORATION:
186 vOtherEntries.push_back(NPC_HORDE_SHAMAN_ENHANCEMENT);
187 break;
188 default:
189 break;
190 }
191 vHealersEntries.erase(vHealersEntries.begin() + pos);
192 }
193
194 // Damage Dealers
195 if (getInstanceScript()->getInstance()->getDifficulty() == InstanceDifficulty::RAID_10MAN_NORMAL || getInstanceScript()->getInstance()->getDifficulty() == InstanceDifficulty::RAID_10MAN_HEROIC)
196 for (uint8_t i = 0; i < 4; ++i)
197 vOtherEntries.erase(vOtherEntries.begin() + Util::getRandomUInt(0, static_cast<uint32_t>(vOtherEntries.size() - 1)));
198
199 std::vector<uint32_t> vChampionEntries;
200 vChampionEntries.clear();
201
202 // Fill Our Entries
203 for (uint8_t i = 0; i < vHealersEntries.size(); ++i)
204 vChampionEntries.push_back(vHealersEntries[i]);
205 for (uint8_t i = 0; i < vOtherEntries.size(); ++i)
206 vChampionEntries.push_back(vOtherEntries[i]);
207
208 return vChampionEntries;
209}

292float FactionChampionsAI::calculateThreat(float distance, uint32_t armor, uint32_t health) const
293{
294 float dist_mod = (mAIType == champions::AI_MELEE || mAIType == champions::AI_PET) ? 15.0f / (15.0f + distance) : 1.0f;
295 float armor_mod = (mAIType == champions::AI_MELEE || mAIType == champions::AI_PET) ? armor / 16635.0f : 0.0f;
296 float eh = (health + 1) * (1.0f + armor_mod);
297 return dist_mod * 30000.0f / eh;
298}

309void FactionChampionsAI::update(CreatureAIFunc pThis)
310{
311 // Power Management
312 if (getCreature()->getPowerType() == POWER_TYPE_MANA)
313 getCreature()->modPower(POWER_TYPE_MANA, getCreature()->getMaxPower(POWER_TYPE_MANA) / 3);
314
315 // Threath Modifiers
316 for (ThreatReference* ref : getCreature()->getThreatManager().getModifiableThreatList())
317 if (Player* victim = ref->getVictim()->ToPlayer())
318 {
319 ref->scaleThreat(0.0f);
320 ref->addThreat(1000000.0f * calculateThreat(getCreature()->getDistance2d(victim), victim->getResistance(0), victim->getHealth()));
321 }
322}
