91void FileSystem::Dir::computeZipListing(const std::string& zipfile, const std::string& _pathInsideZipfile) {
92#if _HAVE_ZIP /* G3DFIX: Use ZIP-library only if defined */
93 const std::string& pathInsideZipfile = FilePath::canonicalize(_pathInsideZipfile);
94 struct zip* z = zip_open( FilePath::removeTrailingSlash(zipfile).c_str(), ZIP_CHECKCONS, NULL );
95 debugAssert(z);
96
97 int count = zip_get_num_files( z );
98 Set<std::string> alreadyAdded;
99 for (int i = 0; i < count; ++i) {
100 struct zip_stat info;
101 zip_stat_init( &info ); // TODO: Docs unclear if zip_stat_init is required.
102 zip_stat_index( z, i, ZIP_FL_NOCASE, &info );
103
104 // Fully-qualified name of a file inside zipfile
105 std::string name = FilePath::canonicalize(info.name);
106
107 if (beginsWith(name, pathInsideZipfile)) {
108 // We found something inside the directory we were looking for,
109 // so the directory itself must exist
110 exists = true;
111
112 // For building the cached directory listing, extract only elements that do not contain
113 // additional subdirectories.
114
115 size_t start = pathInsideZipfile.size();
116 if ((name.length() > start) && isSlash(name[start])) {
117 ++start;
118 }
119
120 size_t end = findSlash(name, start);
121 if (end == std::string::npos) {
122 // There are no more slashes; add this name
123 name = name.substr(start);
124 if (alreadyAdded.insert(name)) {
125 Entry& e = nodeArray.next();
126 e.name = name;
127 e.type = FILE_TYPE;
128 }
129 } else {
130 // There are more slashes, indicating that this is a directory
131 name = name.substr(start, end);
132 if (alreadyAdded.insert(name)) {
133 Entry& e = nodeArray.next();
134 e.name = FilePath::removeTrailingSlash(name);
135 e.type = DIR_TYPE;
136 }
137 }
138 }
139 }
140
141 zip_close(z);
142 z = NULL;
143#else
144 (void)zipfile;
145 (void)_pathInsideZipfile;
146#endif
147}

150FileSystem::Dir& FileSystem::getContents(const std::string& path, bool forceUpdate) {
151 const std::string& key =
152# if defined(G3D_WINDOWS)
153 FilePath::canonicalize(FilePath::removeTrailingSlash(toLower(FilePath::canonicalize(resolve(path)))));
154# else
155 FilePath::canonicalize(FilePath::removeTrailingSlash(FilePath::canonicalize(resolve(path))));
156# endif
157
158 RealTime now = System::time();
159 Dir& dir = m_cache.getCreate(key);
160
161 if ((now > dir.lastChecked + cacheLifetime()) || forceUpdate) {
162 dir = Dir();
163
164 // Out of date: update
165 dir.lastChecked = now;
166# ifdef G3D_WINDOWS
167 // On windows, we have to use GetFileAttributes (http://msdn.microsoft.com/en-us/library/aa364944(v=vs.85).aspx) instead of
168 // stat in order to work with network shares
169 const DWORD st = GetFileAttributesA(key.c_str());
170 const bool exists = (st != INVALID_FILE_ATTRIBUTES);
171 const bool isDirectory = (st & FILE_ATTRIBUTE_DIRECTORY) != 0;
172# else
173 struct _stat st;
174 const bool exists = _stat(key.c_str(), &st) != -1;
175 const bool isDirectory = (st.st_mode & S_IFDIR) != 0;
176# endif
177
178 // Does this path exist on the real filesystem?
179 if (exists && isDirectory) {
180
181 // Is this path actually a directory?
182 if (isDirectory) {
183 dir.exists = true;
184 // Update contents
185# ifdef G3D_WINDOWS
186 const std::string& filespec = FilePath::concat(key, "*");
187 struct _finddata_t fileinfo;
188 intptr_t handle = _findfirst(filespec.c_str(), &fileinfo);
189 debugAssert(handle != -1);
190 int result = 0;
191 do {
192 if ((strcmp(fileinfo.name, ".") != 0) && (strcmp(fileinfo.name, "..") != 0)) {
193 Entry& e = dir.nodeArray.next();
194 e.name = fileinfo.name;
195 if ((fileinfo.attrib & _A_SUBDIR) != 0) {
196 e.type = DIR_TYPE;
197 } else {
198 e.type = FILE_TYPE;
199 }
200 }
201
202 result = _findnext(handle, &fileinfo);
203 } while (result == 0);
204 _findclose(handle);
205
206# else
207 DIR* listing = opendir(key.c_str());
208 debugAssertM(listing, "opendir failed on '" + key + "'");
209 struct dirent* entry = readdir(listing);
210 while (entry != NULL) {
211 if ((strcmp(entry->d_name, "..") != 0) && (strcmp(entry->d_name, ".") != 0)) {
212 Entry& e = dir.nodeArray.next();
213 e.name = entry->d_name;
214
215# ifdef _DIRENT_HAVE_D_TYPE
216 // Not all POSIX systems support this field
217 // http://www.delorie.com/gnu/docs/glibc/libc_270.html
218 switch (entry->d_type) {
219 case DT_DIR:
220 e.type = DIR_TYPE;
221 break;
222
223 case DT_REG:
224 e.type = FILE_TYPE;
225 break;
226
227 case DT_UNKNOWN:
228 default:
229 e.type = UNKNOWN;
230 break;
231 }
232# endif
233 }
234 entry = readdir(listing);
235 }
236 closedir(listing);
237 listing = NULL;
238 entry = NULL;
239# endif
240 }
241
242 } else {
243 std::string zip;
244
245 if (exists && isZipfile(path)) {
246 // This is a zipfile; get its root
247 dir.isZipfile = true;
248 dir.computeZipListing(path, "");
249
250 } else if (inZipfile(path, zip)) {
251
252 // There is a zipfile somewhere in the path. Does
253 // the rest of the path exist inside the zipfile?
254 dir.inZipfile = true;
255 dir.computeZipListing(zip, path.substr(zip.length() + 1));
256 }
257 }
258 }
259
260 return dir;
261}

264bool FileSystem::_inZipfile(const std::string& _path, std::string& z) {
265 const std::string& path = FilePath::expandEnvironmentVariables(_path);
266
267 // Reject trivial cases before parsing
268
269 // Look at all sub-paths containing periods.
270 // For each, ask if it is a zipfile.
271 size_t current = 0;
272 current = path.find('.', current);
273
274 while (current != std::string::npos) {
275 // xxxxx/foo.zip/yyyyy
276 current = path.find('.', current);
277
278 // Look forward for the next slash
279 size_t s = findSlash(path, current);
280
281 if (s == std::string::npos) {
282 // No more slashes
283 return false;
284 }
285
286 z = path.substr(0, s);
287 if (_isZipfile(z)) {
288 return true;
289 }
290
291 current = s + 1;
292 }
293
294 z = "";
295 return false;
296}

299bool FileSystem::_isZipfile(const std::string& _filename) {
300 const std::string& filename = FilePath::canonicalize(FilePath::expandEnvironmentVariables(_filename));
301
302
303
304 FILE* f = fopen(FilePath::removeTrailingSlash(filename).c_str(), "r");
305 if (f == NULL) {
306 return false;
307 }
308 uint8 header[4];
309 (void)fread(header, 4, 1, f);
310
311 const uint8 zipHeader[4] = {0x50, 0x4b, 0x03, 0x04};
312 for (int i = 0; i < 4; ++i) {
313 if (header[i] != zipHeader[i]) {
314 fclose(f);
315 return false;
316 }
317 }
318
319 fclose(f);
320 return true;
321}

324FILE* FileSystem::_fopen(const char* _filename, const char* mode) {
325 const std::string& filename = FilePath::canonicalize(FilePath::expandEnvironmentVariables(_filename));
326
327 for (const char* m = mode; *m != '\0'; ++m) {
328 if (*m == 'w') {
329 // Purge the cache entry for the parent of this directory

331 break;

340void FileSystem::_clearCache(const std::string& _path) {
341 const std::string& path = FilePath::expandEnvironmentVariables(_path);
342
343 if ((path == "") || FilePath::isRoot(path)) {
344 m_cache.clear();
345 } else {
346 Array<std::string> keys;
347 m_cache.getKeys(keys);
348
349 const std::string& prefix =
350# ifdef G3D_WINDOWS
351 toLower(FilePath::canonicalize(FilePath::removeTrailingSlash(_resolve(path))));
352# else
353 FilePath::canonicalize(FilePath::removeTrailingSlash(_resolve(path)));
354# endif
355 const std::string& prefixSlash = prefix + "/";
356
357 for (int k = 0; k < keys.size(); ++k) {
358 const std::string& key = keys[k];
359 if ((key == prefix) || beginsWith(key, prefixSlash)) {
360 m_cache.remove(keys[k]);
361 }
362 }
363 }

372void FileSystem::_createDirectory(const std::string& _dir) {
373 const std::string& dir = FilePath::expandEnvironmentVariables(_dir);
374
375 if (dir == "") {
376 return;
377 }
378
379 std::string d = _resolve(dir);
380
381 // Add a trailing / if there isn't one.
382 switch (d[d.size() - 1]) {
383 case '/':
384 case '\\':
385 break;
386
387 default:
388 d += "/";
389 }
390
391 // If it already exists, do nothing
392 if (_exists(FilePath::removeTrailingSlash(d))) {
393 return;
394 }
395
396 // Parse the name apart
397 std::string root, base, ext;
398 Array<std::string> path;
399
400 std::string lead;
401 FilePath::parse(d, root, path, base, ext);
402 debugAssert(base == "");
403 debugAssert(ext == "");
404
405 // Begin with an extra period so "c:\" becomes "c:\.\" after
406 // appending a path and "c:" becomes "c:.\", not root: "c:\"
407 std::string p = root + ".";
408
409 // Create any intermediate that doesn't exist
410 for (int i = 0; i < path.size(); ++i) {
411 p += "/" + path[i];
412 if (! _exists(p)) {
413 // Windows only requires one argument to mkdir,
414 // where as unix also requires the permissions.
415# ifndef G3D_WINDOWS
416 mkdir(p.c_str(), 0777);
417# else
418 _mkdir(p.c_str());
419# endif
420 }
421 }
422
423 _clearCache(FilePath::parent(FilePath::removeTrailingSlash(d)));
424}

444bool FileSystem::_exists(const std::string& _f, bool trustCache, bool caseSensitive) {
445 const std::string& f = FilePath::expandEnvironmentVariables(_f);
446
447 if (FilePath::isRoot(f)) {
448# ifdef G3D_WINDOWS
449 const std::string& winname = toLower(f.substr(0, 1)) + ":\\";
450 return _drives().contains(winname);
451# else
452 return true;
453# endif
454 }
455
456 const std::string& path = FilePath::removeTrailingSlash(f);
457 const std::string& parentPath = FilePath::parent(path);
458
459 const bool forceUpdate = ! trustCache;
460 const Dir& entry = getContents(parentPath, forceUpdate);
461
462 if (FilePath::containsWildcards(f)) {
463 if (! entry.exists) {
464 // The directory didn't exist, so neither do its contents
465 return false;
466 }
467
468 const std::string& pattern = FilePath::baseExt(path);
469
470 // See if any element of entry matches the wild card
471 for (int i = 0; i < entry.nodeArray.size(); ++i) {
472 if (FilePath::matches(entry.nodeArray[i].name, pattern, caseSensitive)) {
473 return true;
474 }
475 }
476
477 // Could not find a match
478 return false;
479
480 } else {
481 return entry.exists && entry.contains(FilePath::baseExt(path), caseSensitive);
482 }
483}

511std::string FileSystem::_resolve(const std::string& _filename, const std::string& _cwd) {
512 const std::string& filename = FilePath::expandEnvironmentVariables(_filename);
513 const std::string& cwd = FilePath::expandEnvironmentVariables(_cwd);
514
515 if (filename.size() >= 1) {
516 if (isSlash(filename[0])) {
517 // Already resolved
518 return filename;
519 } else {
520
521 #ifdef G3D_WINDOWS
522 if ((filename.size() >= 2) && (filename[1] == ':')) {
523 // There is a drive spec on the front.
524 if ((filename.size() >= 3) && isSlash(filename[2])) {
525 // Already fully qualified
526 return filename;
527 } else {
528 // The drive spec is relative to the
529 // working directory on that drive.
530 debugAssertM(false, "Files of the form d:path are"
531 " not supported (use a fully qualified"
532 " name).");
533 return filename;
534 }
535 }
536 #endif
537 }
538 }
539
540 // Prepend the working directory.
541 return FilePath::concat(cwd, filename);
542}

583int64 FileSystem::_size(const std::string& _filename) {
584 const std::string& filename = FilePath::canonicalize(FilePath::expandEnvironmentVariables(_filename));
585
586 struct stat64 st;
587 int result = stat64(filename.c_str(), &st);
588
589 if (result == -1) {
590#if _HAVE_ZIP /* G3DFIX: Use ZIP-library only if defined */
591 std::string zip, contents;
592 if (zipfileExists(filename, zip, contents)) {
593 int64 requiredMem;
594
595 struct zip *z = zip_open( zip.c_str(), ZIP_CHECKCONS, NULL );
596 debugAssertM(z != NULL, zip + ": zip open failed.");
597 {
598 struct zip_stat info;
599 zip_stat_init( &info ); // Docs unclear if zip_stat_init is required.
600 int success = zip_stat( z, contents.c_str(), ZIP_FL_NOCASE, &info );
601 (void) success;
602 debugAssertM(success == 0, zip + ": " + contents + ": zip stat failed.");
603 requiredMem = info.size;
604 }
605 zip_close(z);
606 return requiredMem;
607 } else {
608#endif
609 return -1;
610#if _HAVE_ZIP /* G3DFIX: Use ZIP-library only if defined */
611 }
612#endif
613 }
614
615 return st.st_size;
616}

619void FileSystem::listHelper
620(const std::string& shortSpec,
621 const std::string& parentPath,
622 Array<std::string>& result,
623 const ListSettings& settings) {
624
625 Dir& dir = getContents(parentPath, false);
626
627 if (! dir.exists) {
628 return;
629 }
630
631 for (int i = 0; i < dir.nodeArray.size(); ++i) {
632 Entry& entry = dir.nodeArray[i];
633 // See if it matches the spec
634 if (FilePath::matches(entry.name, shortSpec, settings.caseSensitive)) {
635
636 if ((entry.type == UNKNOWN) &&
637 (! (settings.files && settings.directories) ||
638 settings.recursive)) {
639 // Update the type: it is unknown and we'll need to branch onit below
640 entry.type = isDirectory(FilePath::concat(parentPath, entry.name)) ? DIR_TYPE : FILE_TYPE;
641 }
642
643 if ((settings.files && settings.directories) ||
644 (settings.files && (entry.type == FILE_TYPE)) ||
645 (settings.directories && (entry.type == DIR_TYPE))) {
646
647 if (settings.includeParentPath) {
648 result.append(FilePath::concat(parentPath, entry.name));
649 } else {
650 result.append(entry.name);
651 }
652 }
653 } // match
654
655 if (settings.recursive) {
656 if (entry.type == UNKNOWN) {
657 entry.type = isDirectory(FilePath::concat(parentPath, entry.name)) ? DIR_TYPE : FILE_TYPE;
658 }
659
660 if (entry.type == DIR_TYPE) {
661 listHelper(shortSpec, FilePath::concat(parentPath, entry.name), result, settings);
662 }
663 }
664 } // for
665}

701bool FilePath::isRoot(const std::string& f) {
702# ifdef G3D_WINDOWS
703 if (f.length() < 2) {
704 return false;
705 }
706
707 if (f[1] == ':') {
708 if (f.length() == 2) {
709 // e.g., "x:"
710 return true;
711 } else if ((f.length() == 3) && isSlash(f[2])) {
712 // e.g., "x:\"
713 return true;
714 }
715 }
716
717 // Windows shares are considered roots, but only if this does not include a path inside the share
718 if (isSlash(f[0]) && isSlash(f[1])) {
719 size_t i = f.find("/", 3);
720 size_t j = f.find("\\", 3);
721
722 if (i == std::string::npos) {
723 i = j;
724 }
725
726 // e.g., "\\foo\", "\\foo"
727 return ((i == std::string::npos) || (i == f.length() - 1));
728 }
729# else
730 if (f == "/") {
731 return true;
732 }
733# endif
734
735 return false;
736}

769 size_t i = findLastSlash(filename);
770
771# ifdef G3D_WINDOWS
772 size_t j = filename.rfind(":");
773 if ((i == std::string::npos) && (j != std::string::npos)) {

786std::string FilePath::base(const std::string& path) {
787 std::string filename = baseExt(path);
788 size_t i = filename.rfind(".");
789 if (i == std::string::npos) {
790 // No extension

840void FilePath::parse
841(const std::string& filename,
842 std::string& root,
843 Array<std::string>& path,
844 std::string& base,
845 std::string& ext) {
846
847 std::string f = filename;
848
849 root = "";
850 path.clear();
851 base = "";
852 ext = "";
853
854 if (f == "") {
855 // Empty filename
856 return;
857 }
858
859 // See if there is a root/drive spec.
860 if ((f.size() >= 2) && (f[1] == ':')) {
861
862 if ((f.size() > 2) && isSlash(f[2])) {
863
864 // e.g. c:\foo
865 root = f.substr(0, 3);
866 f = f.substr(3, f.size() - 3);
867
868 } else {
869
870 // e.g. c:foo
871 root = f.substr(2);
872 f = f.substr(2, f.size() - 2);
873
874 }
875
876 } else if ((f.size() >= 2) && isSlash(f[0]) && isSlash(f[1])) {
877
878 // e.g. //foo
879 root = f.substr(0, 2);
880 f = f.substr(2, f.size() - 2);
881
882 } else if (isSlash(f[0])) {
883
884 root = f.substr(0, 1);
885 f = f.substr(1, f.size() - 1);
886
887 }
888
889 // Pull the extension off
890 {
891 // Find the period
892 size_t i = f.rfind('.');
893
894 if (i != std::string::npos) {
895 // Make sure it is after the last slash!
896 size_t j = findLastSlash(f);
897 if ((j == std::string::npos) || (i > j)) {
898 ext = f.substr(i + 1, f.size() - i - 1);
899 f = f.substr(0, i);
900 }
901 }
902 }
903
904 // Pull the basename off
905 {
906 // Find the last slash
907 size_t i = findLastSlash(f);
908
909 if (i == std::string::npos) {
910
911 // There is no slash; the basename is the whole thing
912 base = f;
913 f = "";
914
915 } else if ((i != std::string::npos) && (i < f.size() - 1)) {
916
917 base = f.substr(i + 1, f.size() - i - 1);
918 f = f.substr(0, i);
919
920 }
921 }
922
923 // Parse what remains into path.
924 size_t prev, cur = 0;
925
926 while (cur < f.size()) {
927 prev = cur;
928
929 // Allow either slash
930 size_t i = f.find('/', prev + 1);
931 size_t j = f.find('\\', prev + 1);
932 if (i == std::string::npos) {
933 i = f.size();
934 }
935
936 if (j == std::string::npos) {
937 j = f.size();
938 }
939
940 cur = min(i, j);
941
942 if (cur == std::string::npos) {
943 cur = f.size();
944 }
945
946 path.append(f.substr(prev, cur - prev));
947 ++cur;
948 }
949}

952std::string FilePath::expandEnvironmentVariables(const std::string& path) {
953 // Search for pattern
954 size_t end = path.find_first_of('$', 0);
955 if (end == std::string::npos) {
956 // Pattern does not exist
957 return path;
958 }
959
960 size_t start = 0;
961 std::string result;
962 while (end != std::string::npos) {
963 const std::string& before = path.substr(start, end - start);
964 result += before;
965 start = end + 1;
966 std::string var;
967 if (path[start] == '(') {
968 // Search for close paren
969 end = path.find_first_of(')', start + 1);
970 if (end == std::string::npos) {
971 throw std::string("Missing close paren in environment variable in \"") + path + "\"";
972 }
973 var = path.substr(start + 1, end - start - 1);
974 } else {
975 // Search for slash or end of string
976 end = path.find_first_of('/', start);
977 size_t i = path.find_first_of('\\', start);
978 if ((size_t(i) != std::string::npos) && ((end == std::string::npos) || (size_t(i) < end))) {
979 end = i;
980 }
981 if (end == std::string::npos) {
982 // If the varible goes to the end of the string, it is the rest of the string
983 end = path.size();
984 } else {
985 --end;
986 }
987 var = path.substr(start, end - start + 1);
988 }
989
990 if (! var.empty()) {
991 const char* value = getenv(var.c_str());
992
993 if (value == NULL) {
994 throw (std::string("LocalLightingEnvironment variable \"") + var + "\" not defined for path \"" + path + "\"");
995 } else {
996 result += value;
997 }
998 } else {
999 // We just parsed an "empty" variable, which was probably a default share on Windows, e.g.,
1000 // "\\mycomputer\c$", and not a variable name.
1001 result += "$";
1002 }
1003
1004 start = end + 1;
1005 end = path.find_first_of('$', start);
1006 }
1007
1008 // Paste on the remainder of the source path
1009 if (start < path.size()) {
1010 result += path.substr(start);
1011 }
1012
1013 return result;
1014}
