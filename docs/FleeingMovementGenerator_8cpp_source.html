60template<class T>
61bool FleeingMovementGenerator<T>::doUpdate(T* owner, uint32_t diff)
62{
63 if (!owner || !owner->isAlive())
64 return false;
65
66 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
67 {
68 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
69 owner->stopMoving();
70 _path = nullptr;
71 return true;
72 }
73 else
74 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
75
76 _timer->updateTimer(diff);
77 if ((MovementGenerator::hasFlag(MOVEMENTGENERATOR_FLAG_SPEED_UPDATE_PENDING) && !owner->movespline->Finalized()) || (_timer->isTimePassed() && owner->movespline->Finalized()))
78 {
79 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_TRANSITORY);
80 setTargetLocation(owner);
81 }
82
83 return true;
84}

125template<class T>
126void FleeingMovementGenerator<T>::setTargetLocation(T* owner)
127{
128 if (!owner || !owner->isAlive())
129 return;
130
131 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
132 {
133 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
134 owner->stopMoving();
135 _path = nullptr;
136 return;
137 }
138
139 LocationVector destination = owner->GetPosition();
140 getPoint(owner, destination);
141
142 // Add LOS check for target point
143 if (!owner->IsWithinLOS(destination))
144 {
145 _timer->resetInterval(200);
146 return;
147 }
148
149 if (!_path)
150 {
151 _path = std::make_unique<PathGenerator>(owner);
152 _path->setPathLengthLimit(30.0f);
153 }
154
155 bool result = _path->calculatePath(destination.getPositionX(), destination.getPositionY(), destination.getPositionZ());
156 if (!result || (_path->getPathType() & PATHFIND_NOPATH)
157 || (_path->getPathType() & PATHFIND_SHORTCUT)
158 || (_path->getPathType() & PATHFIND_FARFROMPOLY))
159 {
160 _timer->resetInterval(100);
161 return;
162 }
163
164 owner->addUnitStateFlag(UNIT_STATE_FLEEING_MOVE);
165
166 MovementMgr::MoveSplineInit init(owner);
167 init.MovebyPath(_path->getPath());
168 init.SetWalk(false);
169 int32_t traveltime = init.Launch();
170 _timer->resetInterval(traveltime + Util::getRandomUInt(800, 1500));
171}

173template<class T>
174void FleeingMovementGenerator<T>::getPoint(T* owner, LocationVector &position)
175{
176 float casterDistance, casterAngle;
177 if (Unit* fleeTarget = (*owner).getWorldMapUnit(_fleeTargetGUID))
178 {
179 casterDistance = fleeTarget->getDistance(owner);
180 if (casterDistance > 0.2f)
181 casterAngle = fleeTarget->getAbsoluteAngle(owner);
182 else
183 casterAngle = Util::getRandomFloat(0.0f, 2.0f * float(M_PI));
184 }
185 else
186 {
187 casterDistance = 0.0f;
188 casterAngle = Util::getRandomFloat(0.0f, 2.0f * float(M_PI));
189 }
190
191 float distance, angle;
192 if (casterDistance < MIN_QUIET_DISTANCE)
193 {
194 distance = Util::getRandomFloat(0.4f, 1.3f) * (MIN_QUIET_DISTANCE - casterDistance);
195 angle = casterAngle + Util::getRandomFloat(-float(M_PI) / 8.0f, float(M_PI) / 8.0f);
196 }
197 else if (casterDistance > MAX_QUIET_DISTANCE)
198 {
199 distance = Util::getRandomFloat(0.4f, 1.0f) * (MAX_QUIET_DISTANCE - MIN_QUIET_DISTANCE);
200 angle = -casterAngle + Util::getRandomFloat(-float(M_PI) / 4.0f, float(M_PI) / 4.0f);
201 }
202 else // we are inside quiet range
203 {
204 distance = Util::getRandomFloat(0.6f, 1.2f) * (MAX_QUIET_DISTANCE - MIN_QUIET_DISTANCE);
205 angle = Util::getRandomFloat(0.0f, 2.0f * float(M_PI));
206 }
207
208 owner->movePositionToFirstCollision(position, distance, angle);
209}
