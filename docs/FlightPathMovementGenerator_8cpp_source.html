18uint32_t FlightPathMovementGenerator::getPathAtMapEnd() const
19{
20 if (_currentNode >= _path.size())
21 return static_cast<uint32_t>(_path.size());
22
23 uint32_t curMapId = _path[_currentNode]->mapid;
24 for (uint32_t i = _currentNode; i < _path.size(); ++i)
25 if (_path[i]->mapid != curMapId)
26 return i;
27
28 return static_cast<uint32_t>(_path.size());
29}

31static bool isNodeIncludedInShortenedPath(WDB::Structures::TaxiPathNodeEntry const* path1, WDB::Structures::TaxiPathNodeEntry const* path2)
32{
33 return path1->mapid != path2->mapid || std::pow(path1->x - path2->x, 2) + std::pow(path1->y - path2->y, 2) > (40.0f * 40.0f);

36void FlightPathMovementGenerator::loadPath(Player* player)
37{
38 _pointsForPathSwitch.clear();
39 std::deque<uint32_t> const& taxi = player->getTaxiData()->getPath();
40 for (uint32_t src = 0, dst = 1; dst < taxi.size(); src = dst++)
41 {
42 uint32_t path, cost;
43 sTaxiMgr.getTaxiPath(taxi[src], taxi[dst], path, cost);
44 if (path > sTaxiPathNodesByPath.size())
45 return;
46
47 TaxiPathNodeList const& nodes = sTaxiPathNodesByPath[path];
48 if (!nodes.empty())
49 {
50 WDB::Structures::TaxiPathNodeEntry const* start = nodes[0];
51 WDB::Structures::TaxiPathNodeEntry const* end = nodes[nodes.size() - 1];
52 bool passedPreviousSegmentProximityCheck = false;
53 for (uint32_t i = 0; i < nodes.size(); ++i)
54 {
55 if (passedPreviousSegmentProximityCheck || !src || _path.empty() || isNodeIncludedInShortenedPath(_path[_path.size() - 1], nodes[i]))
56 {
57 if ((!src || (isNodeIncludedInShortenedPath(start, nodes[i]) && i >= 2)) &&
58 (dst == taxi.size() - 1 || (isNodeIncludedInShortenedPath(end, nodes[i]) && i < nodes.size() - 1)))
59 {
60 passedPreviousSegmentProximityCheck = true;
61 _path.push_back(nodes[i]);
62 }
63 }
64 else
65 {
66 _path.pop_back();
67 --_pointsForPathSwitch.back().PathIndex;
68 }
69 }
70 }
71
72 _pointsForPathSwitch.push_back({ uint32_t(_path.size() - 1), int32_t(cost) });
73 }
74}

90void FlightPathMovementGenerator::doFinalize(Player* player, bool active, bool /*movementInform*/)
91{
92 addFlag(MOVEMENTGENERATOR_FLAG_FINALIZED);
93 if (!active)
94 return;
95
96 // remove flag to prevent send object build movement packets for flight state
97 player->removeUnitStateFlag(UNIT_STATE_IN_FLIGHT);
98
99 if (!player->getTaxiData()->nodeAfterTeleport)
100 {
101 player->cleanupAfterTaxiFlight();
102 }
103 else
104 {
105 player->dismount();
106 player->removeUnitFlags(UNIT_FLAG_LOCK_PLAYER | UNIT_FLAG_MOUNTED_TAXI);
107 }
108
109 if (!player->isOnTaxi())
110 {
111 // update position and orientation for landing point
112 // this prevent cheating with landing point at lags
113 player->stopMoving();
114 float mapHeight = player->getWorldMap()->getHeight(player->GetPhase(), LocationVector(_path[getCurrentNode()]->x, _path[getCurrentNode()]->y, _path[getCurrentNode()]->z));
115 // When the player reaches the last flight point, teleport to destination
116 player->safeTeleport(_path[getCurrentNode()]->mapid, 0, LocationVector(_path[getCurrentNode()]->x, _path[getCurrentNode()]->y, mapHeight, player->GetOrientation()));
117 }
118
119#if VERSION_STRING >= TBC
120 player->removePlayerFlags(PLAYER_FLAG_TAXI_TIME_TEST);
121#endif
122}

124void FlightPathMovementGenerator::doReset(Player* player)
125{
126 removeFlag(MOVEMENTGENERATOR_FLAG_DEACTIVATED);
127
128 player->addUnitStateFlag(UNIT_STATE_IN_FLIGHT);
129 player->addUnitFlags(UNIT_FLAG_LOCK_PLAYER | UNIT_FLAG_MOUNTED_TAXI);
130
131 MovementMgr::MoveSplineInit init(player);
132 for (uint32_t i = getCurrentNode(); i != getPathAtMapEnd(); ++i)
133 {
134 G3D::Vector3 vertice(_path[i]->x, _path[i]->y, _path[i]->z);
135 init.Path().push_back(vertice);
136 }
137 init.SetFirstPointId(getCurrentNode());
138 init.SetFly();
139#if VERSION_STRING > WotLK
140 init.SetSmooth();
141 init.SetUncompressed();
142 init.SetWalk(true);
143#endif
144 init.SetVelocity(32.0f);
145 init.Launch();
146}

148bool FlightPathMovementGenerator::doUpdate(Player* player, uint32_t /*diff*/)
149{
150 if (!player)
151 return false;
152
153 uint32_t pointId = (uint32_t)player->movespline->currentPathIdx();
154 if (pointId > _currentNode)
155 {
156 bool departureEvent = true;
157 do
158 {
159 doEventIfAny(player, _path[_currentNode], departureEvent);
160 while (!_pointsForPathSwitch.empty() && _pointsForPathSwitch.front().PathIndex <= _currentNode)
161 {
162 _pointsForPathSwitch.pop_front();
163 player->getTaxiData()->nextTaxiDestination();
164 if (!_pointsForPathSwitch.empty())
165 {
166#if VERSION_STRING > TBC
167 player->updateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GOLD_SPENT_FOR_TRAVELLING, _pointsForPathSwitch.front().Cost);
168#endif
169 player->modCoinage(-_pointsForPathSwitch.front().Cost);
170 }
171 }
172
173 if (pointId == _currentNode)
174 break;
175
176 if (_currentNode == _preloadTargetNode)
177 preloadEndGrid();
178
179 _currentNode += (uint32_t)departureEvent;
180 departureEvent = !departureEvent;
181 } while (true);
182 }
183
184 return _currentNode < (_path.size() - 1);
185}

222void FlightPathMovementGenerator::initEndGridInfo()
223{
224 // Storage to preload flightmaster cell at end of flight. For multi-stop flights, this will
225 // be reinitialized for each flightmaster at the end of each spline (or stop) in the flight.
226
227 // Number of nodes in path.
228 const auto nodeCount = static_cast<uint32_t>(_path.size());
229 // MapId of last node
230 const uint32_t endNode = nodeCount - 1;
231 _endMapId = _path[endNode]->mapid;
232 // Node where Destination Cell gets Loaded
233 _preloadTargetNode = nodeCount - 3;
234 // EndCell Position
235 _endGridX = _path[endNode]->x;
236 _endGridY = _path[endNode]->y;
237}

239void FlightPathMovementGenerator::preloadEndGrid()
240{
241 // used to preload the final grid where the flightmaster is
242 WorldMap* endMap = sMapMgr.findWorldMap(_endMapId);
243
244 // Load the MapCell
245 if (endMap)

247 sLogger.debug("FlightPathMovementGenerator:: Preloading of Cell({}, {}) for map {} at node index {}/{}", _endGridX, _endGridY, _endMapId, _preloadTargetNode, (uint32_t)(_path.size() - 1));
248 if (endMap->getCellByCoords(_endGridX, _endGridY))
249 endMap->getCellByCoords(_endGridX, _endGridY)->setActivity(true);
250 }
