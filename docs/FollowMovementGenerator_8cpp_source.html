35static bool positionOkay(Unit* owner, Unit* target, float range, Optional<ChaseAngle> angle = {})
36{
37 if (owner->GetDistance2dSq(target) > Util::square(owner->getCombatReach() + target->getCombatReach() + range))
38 return false;
39
40 return !angle || angle->isAngleOkay(target->getRelativeAngle(owner));
41}

61bool FollowMovementGenerator::update(Unit* owner, uint32_t diff)
62{
63 // owner might be dead or gone
64 if (!owner || !owner->isAlive())
65 return false;
66
67 // our target might have gone away
68 Unit* const target = getTarget();
69 if (!target || !target->IsInWorld())
70 return false;
71
72 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
73 {
74 _path = nullptr;
75 owner->stopMoving();
76 _lastTargetPosition.reset();
77 return true;
78 }
79
80 _checkTimer->updateTimer(diff);
81 if (_checkTimer->isTimePassed())
82 {
83 _checkTimer->resetInterval(CHECK_INTERVAL);
84 if (hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED) && positionOkay(owner, target, _range, _angle))
85 {
86 removeFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
87 _path = nullptr;
88 owner->stopMoving();
89 _lastTargetPosition.reset();
90 doMovementInform(owner, target);
91 return true;
92 }
93 }
94
95 if (owner->hasUnitStateFlag(UNIT_STATE_FOLLOW_MOVE) && owner->movespline->Finalized())
96 {
97 removeFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
98 _path = nullptr;
99 owner->removeUnitStateFlag(UNIT_STATE_FOLLOW_MOVE);
100 doMovementInform(owner, target);
101 }
102
103 if (!_lastTargetPosition || _lastTargetPosition->distanceSquare(target->GetPosition()) > 0.0f)
104 {
105 _lastTargetPosition = target->GetPosition();
106 if (owner->hasUnitStateFlag(UNIT_STATE_FOLLOW_MOVE) || !positionOkay(owner, target, _range + FOLLOW_RANGE_TOLERANCE))
107 {
108 if (!_path)
109 _path = std::make_unique<PathGenerator>(owner);
110
111 float x, y, z;
112
113 // select angle
114 float tAngle;
115 float const curAngle = target->getRelativeAngle(owner);
116 if (_angle.isAngleOkay(curAngle))
117 tAngle = curAngle;
118 else
119 {
120 float const diffUpper = normalizeOrientation(curAngle - _angle.upperBound());
121 float const diffLower = normalizeOrientation(_angle.lowerBound() - curAngle);
122 if (diffUpper < diffLower)
123 tAngle = _angle.upperBound();
124 else
125 tAngle = _angle.lowerBound();
126 }
127
128 target->getNearPoint(owner, x, y, z, _range, target->toAbsoluteAngle(tAngle));
129
130 if (owner->isHovering())
131 owner->updateAllowedPositionZ(x, y, z);
132
133 // pets are allowed to "cheat" on pathfinding when following their master
134 bool allowShortcut = false;
135 if (Unit* oPet = owner)
136 {
137 auto* const petOwner = oPet->getUnitOwner();
138 if (petOwner && target->getGuid() == petOwner->getGuid())
139 allowShortcut = true;
140 }
141
142 bool success = _path->calculatePath(x, y, z, allowShortcut);
143 if (!success || (_path->getPathType() & PATHFIND_NOPATH))
144 {
145 owner->stopMoving();
146 return true;
147 }
148
149 owner->addUnitStateFlag(UNIT_STATE_FOLLOW_MOVE);
150 addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
151
152 MovementMgr::MoveSplineInit init(owner);
153 init.MovebyPath(_path->getPath());
154 init.SetWalk(target->isWalking());
155 init.SetFacing(target->GetOrientation());
156 init.Launch();
157 }
158 }
159 return true;
160}
