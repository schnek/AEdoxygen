53bool FormationMovementGenerator::doUpdate(Creature* owner, uint32_t diff)
54{
55 Unit* target = getTarget();
56
57 if (!owner || !target)
58 return false;
59
60 // Owner cannot move. Reset all fields and wait for next action
61 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
62 {
63 addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
64 owner->stopMoving();
65 _nextMoveTimer->resetInterval(0);
66 _hasPredictedDestination = false;
67 return true;
68 }
69
70 // If target is not moving and destination has been predicted and if we are on the same spline, we stop as well
71 if (target->movespline->Finalized() && target->movespline->GetId() == _lastLeaderSplineID && _hasPredictedDestination)
72 {
73 addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
74 owner->stopMoving();
75 _nextMoveTimer->resetInterval(0);
76 _hasPredictedDestination = false;
77 return true;
78 }
79
80 if (!owner->movespline->Finalized())
81 owner->SetSpawnLocation(owner->GetPosition());
82
83 // Formation leader has launched a new spline, launch a new one for our member as well
84 // This action does not reset the regular movement launch cycle interval
85 if (!target->movespline->Finalized() && target->movespline->GetId() != _lastLeaderSplineID)
86 {
87 // Update formation angle
88 if (_point1 && target->isCreature())
89 {
90 if (CreatureGroup* formation = target->ToCreature()->getFormation())
91 {
92 if (Creature* leader = formation->getLeader())
93 {
94 const auto currentWaypoint = leader->getCurrentWaypointInfo().first;
95 if (currentWaypoint == _point1 || currentWaypoint == _point2)
96 _angle = float(M_PI) * 2 - _angle;
97 }
98 }
99 }
100
101 launchMovement(owner, target);
102 _lastLeaderSplineID = target->movespline->GetId();
103 return true;
104 }
105
106 _nextMoveTimer->updateTimer(diff);
107 if (_nextMoveTimer->isTimePassed())
108 {
109 _nextMoveTimer->resetInterval(FORMATION_MOVEMENT_INTERVAL);
110
111 // Our leader has a different position than on our last check, launch movement.
112 if (_lastLeaderPosition != target->GetPosition())
113 {
114 launchMovement(owner, target);
115 return true;
116 }
117 }
118
119 // We have reached our destination before launching a new movement. Alling facing with leader
120 if (owner->hasUnitStateFlag(UNIT_STATE_FOLLOW_FORMATION_MOVE) && owner->movespline->Finalized())
121 {
122 owner->removeUnitStateFlag(UNIT_STATE_FOLLOW_FORMATION_MOVE);
123 owner->setFacingTo(target->GetOrientation());
124 movementInform(owner);
125 }
126
127 return true;
128}

130void FormationMovementGenerator::launchMovement(Creature* owner, Unit* target)
131{
132 float relativeAngle = 0.f;
133
134 // Determine our relative angle to our current spline destination point
135 if (!target->movespline->Finalized())
136 relativeAngle = target->getRelativeAngle(vector3ToPosition(target->movespline->CurrentDestination()));
137
138 //////////////////////////////////////////////////////////////////////////////////////////
139 // Destination calculation
140
141 /// According to sniff data, formation members have a periodic move interal of 1,2s.
142 /// Each of these splines has a exact duration of 1650ms +- 1ms when no pathfinding is involved.
143 /// To get a representative result like that we have to predict our formation leader's path
144 /// and apply our formation shape based on that destination.
145
146 LocationVector dest = target->GetPosition();
147 float velocity = 0.f;
148
149 // Formation leader is moving. Predict our destination
150 if (!target->movespline->Finalized())
151 {
152 // Pick up leader's spline velocity
153 velocity = target->movespline->Velocity();
154
155 // Calculate travel distance to get a 1650ms result
156 float travelDist = velocity * 1.65f;
157
158 // Move destination ahead...
159 target->movePositionToFirstCollision(dest, travelDist, relativeAngle);
160 // ... and apply formation shape
161 target->movePositionToFirstCollision(dest, _range, _angle + relativeAngle);
162
163 float distance = owner->GetPosition().getExactDist(dest);
164
165 // Calculate catchup speed mod (Limit to a maximum of 50% of our original velocity
166 float velocityMod = std::min<float>(distance / travelDist, 1.5f);
167
168 // Now we will always stay synch with our leader
169 velocity *= velocityMod;
170 _hasPredictedDestination = true;
171 }
172 else
173 {
174 // Formation leader is not moving. Just apply the base formation shape on his position.
175 target->movePositionToFirstCollision(dest, _range, _angle + relativeAngle);
176 _hasPredictedDestination = false;
177 }
178
179 // Leader is not moving, so just pick up his default walk speed
180 if (velocity == 0.f)
181 velocity = target->getSpeedRate(TYPE_WALK, true);
182
183 MovementMgr::MoveSplineInit init(owner);
184 init.MoveTo(positionToVector3(dest));
185 init.SetVelocity(velocity);
186 init.Launch();
187
188 _lastLeaderPosition = target->GetPosition();
189 owner->addUnitStateFlag(UNIT_STATE_FOLLOW_FORMATION_MOVE);
190 removeFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
191}
