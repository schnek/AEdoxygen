14 systemMessage(m_session, "--- Current Active Events ---");
15 for (const auto& gameEventPair : sGameEventMgr.mGameEvents)

17 auto gameEvent = gameEventPair.second.get();
18 if (gameEvent->GetState() == GAMEEVENT_ACTIVE)

20 systemMessage(m_session, "{} - {}", gameEvent->event_id, gameEvent->description);

23 {

110bool ChatCommandHandler::HandleEventReloadAllEvents(const char* /*args*/, WorldSession* m_session)
111{
112 systemMessage(m_session, "Beginning reload of all game events");
113 auto startTime = Util::TimeNow();
114
115 // Stop all events, then clear storage
116 for (const auto& eventPair : sGameEventMgr.mGameEvents)
117 {
118 auto gameEvent = eventPair.second.get();
119 gameEvent->StopEvent();
120 }
121
122 // Clear game events
123 sGameEventMgr.mGameEvents.clear();
124
125 // Reload events from DB
126 sGameEventMgr.LoadFromDB();
127
128 systemMessage(m_session, "Reloaded all game events in {} ms", Util::GetTimeDifferenceToNow(startTime));
129 return true;
