36void GameEventMgr::StartArenaEvents()
37{
38 for (auto i = 57; i <= 60; ++i)
39 {
40 auto gameEvent = GetEventById(i);
41 if (gameEvent == nullptr)
42 {
43 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "Missing arena event (id: {})", i);
44 continue;
45 }
46
47 if (i - 52 == worldConfig.arena.arenaSeason && worldConfig.arena.arenaProgress == 1)
48 gameEvent->SetState(GAMEEVENT_ACTIVE_FORCED);
49 else
50 gameEvent->SetState(GAMEEVENT_INACTIVE_FORCED);
51 }
52}

306void GameEventMgr::GameEventMgrThread::Update()
307{
308 //sLogger.info("GameEventMgr : Tick!");
309 auto now = time(0);
310
311 for (const auto& gameEventPair : sGameEventMgr.mGameEvents)
312 {
313 GameEvent* gameEvent = gameEventPair.second.get();
314
315 // Don't alter manual events
316 if (!gameEvent->isValid())
317 continue;
318
319 auto startTime = time_t(gameEvent->start);
320 if (startTime < now && now < gameEvent->end)
321 {
322 if ((now - startTime) % (gameEvent->occurence * 60) < gameEvent->length * 60)
323 {
324 // Event should start
325 if (gameEvent->state != GAMEEVENT_INACTIVE_FORCED)
326 {
327 gameEvent->StartEvent();
328 continue;
329 }
330 }
331 continue;
332 }
333
334 // Event should stop
335 if (gameEvent->state != GAMEEVENT_ACTIVE_FORCED)
336 {
337 gameEvent->StopEvent();
338 }
339 }
