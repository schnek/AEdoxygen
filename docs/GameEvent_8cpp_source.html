50GameEventState GameEvent::OnStateChange(GameEventState pOldState, GameEventState pNewState)
51{
52 if (mEventScript != nullptr)
53 pNewState = mEventScript->OnEventStateChange(this, pOldState, pNewState);
54
55 if (pOldState == pNewState)
56 return pNewState;
57
58 // Save new state to DB before calling handler
59 const char* updateQuery = "REPLACE INTO gameevent_save (event_entry, state, next_start) VALUES (%u, %u, %u)";
60 CharacterDatabase.Execute(updateQuery, event_id, pNewState, nextstart);
61
62 bool shouldStop = true;
63 bool shouldStart = true;
64
65 switch (pNewState)
66 {
67 case GAMEEVENT_INACTIVE: // Despawn all entities
68 case GAMEEVENT_INACTIVE_FORCED:
69 if (mEventScript != nullptr)
70 shouldStop = mEventScript->OnBeforeEventStop(this, pOldState);
71
72 if (shouldStop)
73 DestroyAllEntities();
74
75 if (mEventScript != nullptr)
76 mEventScript->OnAfterEventStop(this, pOldState);
77
78 break;
79 case GAMEEVENT_ACTIVE: // Spawn all entities
80 case GAMEEVENT_ACTIVE_FORCED:
81 case GAMEEVENT_PREPARING: // Not yet supported, but should be used for things like darkmoon preparation
82 if (mEventScript != nullptr)
83 shouldStart = mEventScript->OnBeforeEventStart(this, pOldState);
84
85 if (shouldStart)
86 {
87 DestroyAllEntities();
88 SpawnAllEntities();
89 }
90
91 if (mEventScript != nullptr)
92 mEventScript->OnAfterEventStart(this, pOldState);
93 break;
94 }
95
96 return pNewState;
97}

172 if (npc.waypoint_group != 0)

174 // todo aaron02

176

178 creature->setDisplayId(npc.displayid);
179 creature->setFaction(npc.faction);

183 creature->setVirtualItemSlotId(OFFHAND, creatureProperties->itemslot_2);
184 creature->setVirtualItemSlotId(RANGED, creatureProperties->itemslot_3);
185

188
189 creature->mEvent = this;
