17bool ChatCommandHandler::HandleGMActiveCommand(const char* args, WorldSession* m_session)
18{
19 auto player = m_session->GetPlayer();
20 bool toggle_no_notice = std::string(args) == "no_notice" ? true : false;
21 if (player->isGMFlagSet())
22 {
23 if (!toggle_no_notice)
24 {
25 systemMessage(m_session, "GM Flag removed.");
26 blueSystemMessage(m_session, "<GM> Will no longer show in chat messages or above your name until you use this command again.");
27 }
28 player->removePlayerFlags(PLAYER_FLAG_GM);
29 player->setFaction(player->getInitialFactionId());
30 player->updatePvPArea();
31 player->updateVisibility();
32 }
33 else
34 {
35#if VERSION_STRING >= WotLK
36 if (player->hasPlayerFlags(PLAYER_FLAG_DEVELOPER))
37 HandleGMDevTagCommand("no_notice", m_session);
38#endif
39
40 systemMessage(m_session, "GM Flag set.");
41 blueSystemMessage(m_session, "<GM> will now appear above your name and in chat messages until you use this command again.");
42 player->addPlayerFlags(PLAYER_FLAG_GM);
43 player->setFaction(35);
44 player->removePvpFlag();
45 player->updateVisibility();
46 }
47 return true;
48}

149//.gm list
150bool ChatCommandHandler::HandleGMListCommand(const char* /*args*/, WorldSession* m_session)
151{
152 bool print_headline = true;
153
154 bool is_gamemaster = m_session->hasPermissions();
155
156 std::lock_guard guard(sObjectMgr.m_playerLock);
157 for (const auto playerPair : sObjectMgr.getPlayerStorage())
158 {
159 Player* player = playerPair.second;
160 if (player->getSession()->hasPermissions())
161 {
162 if (!worldConfig.gm.listOnlyActiveGms)
163 {
164 if (print_headline)
165 greenSystemMessage(m_session, "The following GMs are on this server:");
166
167 if (worldConfig.gm.hidePermissions && !is_gamemaster)
168 systemMessage(m_session, " - {}", player->getName());
169 else
170 systemMessage(m_session, " - {} [{}]", player->getName(), player->getSession()->GetPermissions().get());
171
172 print_headline = false;
173 }
174 else if (worldConfig.gm.listOnlyActiveGms && player->isGMFlagSet())
175 {
176 if (player->isGMFlagSet())
177 {
178 if (print_headline)
179 greenSystemMessage(m_session, "The following GMs are active on this server:");
180
181 if (worldConfig.gm.hidePermissions && !is_gamemaster)
182 systemMessage(m_session, " - {}", player->getName());
183 else
184 systemMessage(m_session, " - {} [{}]", player->getName(), player->getSession()->GetPermissions().get());
185
186 print_headline = false;
187 }
188 else
189 {
190 systemMessage(m_session, "No GMs are currently logged in on this server.");
191 print_headline = false;
192 }
193 }
194 }
195 }
196
197 if (print_headline)
198 {
199 if (!worldConfig.gm.listOnlyActiveGms)
200 systemMessage(m_session, "No GMs are currently logged in on this server.");
201 else
202 systemMessage(m_session, "No GMs are currently active on this server.");
203 }
204
205 return true;
206}
