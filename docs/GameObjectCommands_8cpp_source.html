19//.gobject damage
20bool ChatCommandHandler::HandleGODamageCommand(const char* args, WorldSession* session)
21{
22 uint32_t damage = 0;
23 uint32_t spellid = 0;
24
25 if (sscanf(args, "%u %u", &damage, &spellid) < 1)
26 {
27 if (damage == 0)
28 {
29 redSystemMessage(session, "You need to specify how much you want to damage the selected GO!");
30 return true;
31 }
32 }
33
34 auto gameobject = session->GetPlayer()->getSelectedGo();
35 if (gameobject == nullptr)
36 {
37 redSystemMessage(session, "You need to select a GO first!");
38 return true;
39 }
40
41 if (gameobject->GetGameObjectProperties()->type != GAMEOBJECT_TYPE_DESTRUCTIBLE_BUILDING)
42 {
43 redSystemMessage(session, "The selected GO must be a destructible building!");
44 return true;
45 }
46
47 if (spellid == 0)
48 spellid = 57609;
49
50 GameObject_Destructible* dgo = static_cast<GameObject_Destructible*>(gameobject);
51 if (dgo->GetHP() == 0)
52 {
53 redSystemMessage(session, "Cannot further damage a destroyed GameObject");
54 return true;
55 }
56
57 uint64_t guid = session->GetPlayer()->getGuid();
58 dgo->Damage(damage, guid, 0, spellid);
59
60 greenSystemMessage(session, "GameObject has been damaged for {} hitpoints", damage);
61 greenSystemMessage(session, "New hitpoints {}", dgo->GetHP());
62
63 return true;
64}

66//.gobject delete
67bool ChatCommandHandler::HandleGODeleteCommand(const char* /*args*/, WorldSession* m_session)
68{
69 GameObject* selected_gobject = m_session->GetPlayer()->getSelectedGo();
70 if (selected_gobject == nullptr)
71 {
72 redSystemMessage(m_session, "No selected GameObject...");
73 return true;
74 }
75
76 if (selected_gobject->IsInBg())
77 {
78 redSystemMessage(m_session, "GameObjects can't be deleted in Battlegrounds");
79 return true;
80 }
81
82 if (selected_gobject->m_spawn != nullptr && selected_gobject->m_spawn->entry == selected_gobject->getEntry())
83 {
84 uint32_t cellx = uint32_t(((Map::Terrain::_maxX - selected_gobject->m_spawn->spawnPoint.x) / Map::Cell::cellSize));
85 uint32_t celly = uint32_t(((Map::Terrain::_maxY - selected_gobject->m_spawn->spawnPoint.y) / Map::Cell::cellSize));
86
87 if (cellx < Map::Cell::_sizeX && celly < Map::Cell::_sizeY)
88 {
89 CellSpawns* cell_spawns = selected_gobject->getWorldMap()->getBaseMap()->getSpawnsList(cellx, celly);
90 if (cell_spawns != nullptr)
91 {
92 for (GameobjectSpawnList::iterator itr = cell_spawns->GameobjectSpawns.begin(); itr != cell_spawns->GameobjectSpawns.end(); ++itr)
93 {
94 if ((*itr) == selected_gobject->m_spawn)
95 {
96 cell_spawns->GameobjectSpawns.erase(itr);
97 break;
98 }
99 }
100 }
101
102 selected_gobject->deleteFromDB();
103
104 delete selected_gobject->m_spawn;
105 selected_gobject->m_spawn = nullptr;
106 }
107 }
108 sGMLog.writefromsession(m_session, "deleted game object entry %u on map %u at X:%f Y:%f Z:%f Name %s", selected_gobject->getEntry(),
109 selected_gobject->GetMapId(), selected_gobject->GetPositionX(), selected_gobject->GetPositionY(), selected_gobject->GetPositionZ(),
110 sMySQLStore.getGameObjectProperties(selected_gobject->getEntry())->name.c_str());
111 selected_gobject->despawn(0, 0);
112
113 m_session->GetPlayer()->setSelectedGo(0);
114
115 return true;
116}

415//.gobject rotate
416bool ChatCommandHandler::HandleGORotateCommand(const char* args, WorldSession* m_session)
417{
418 char Axis;
419 float deg;
420 if (sscanf(args, "%c %f", &Axis, &deg) < 1)
421 return false;
422
423 GameObject* go = m_session->GetPlayer()->getSelectedGo();
424 if (!go)
425 {
426 redSystemMessage(m_session, "No selected GameObject...");
427 return true;
428 }
429
430 float rotation_x = m_session->GetPlayer()->m_goLastXRotation;
431 float rotation_y = m_session->GetPlayer()->m_goLastYRotation;
432 float orientation = go->GetOrientation();
433
434 switch (tolower(Axis))
435 {
436 case 'x':
437 go->setLocalRotationAngles(orientation, rotation_y, deg);
438 m_session->GetPlayer()->m_goLastXRotation = deg;
439 break;
440 case 'y':
441 go->setLocalRotationAngles(orientation, deg, rotation_x);
442 m_session->GetPlayer()->m_goLastYRotation = deg;
443 break;
444 case 'o':
445 go->SetOrientation(m_session->GetPlayer()->GetOrientation());
446 go->setLocalRotationAngles(go->GetOrientation(), rotation_y, rotation_x);
447 break;
448 default:
449 redSystemMessage(m_session, "Invalid Axis, Please use x, y, or o.");
450 return true;
451 }
452
453 greenSystemMessage(m_session, "Gameobject spawn id: {} rotated", go->m_spawn->id);
454
455 uint32_t NewGuid = m_session->GetPlayer()->getWorldMap()->generateGameobjectGuid();
456 go->RemoveFromWorld(true);
457 go->SetNewGuid(NewGuid);
458 go->PushToWorld(m_session->GetPlayer()->getWorldMap());
459 go->saveToDB();
460
461 m_session->GetPlayer()->setSelectedGo(NewGuid);
462 return true;
463}

465//.gobject select
466bool ChatCommandHandler::HandleGOSelectCommand(const char* args, WorldSession* m_session)
467{
468 GameObject* GObj = nullptr;
469 GameObject* GObjs = m_session->GetPlayer()->getSelectedGo();
470
471 float cDist = 9999.0f;
472 float nDist = 0.0f;
473 bool bUseNext = false;
474
475 if (args)
476 {
477 if (args[0] == '1')
478 {
479 if (GObjs == nullptr)
480 bUseNext = true;
481
482 for (const auto& Itr : m_session->GetPlayer()->getInRangeObjectsSet())
483 {
484 if (Itr && Itr->isGameObject() && Itr->GetPhase() == m_session->GetPlayer()->GetPhase())
485 {
486 // Find the current go, move to the next one
487 if (bUseNext)
488 {
489 // Select the first.
490 GObj = static_cast<GameObject*>(Itr);
491 break;
492 }
493
494 if (Itr == GObjs)
495 {
496 // Found him. Move to the next one, or beginning if we're at the end
497 bUseNext = true;
498 }
499 }
500 }
501 }
502 }
503 if (!GObj)
504 {
505 for (const auto& Itr : m_session->GetPlayer()->getInRangeObjectsSet())
506 {
507 if (Itr && Itr->isGameObject() && Itr->GetPhase() == m_session->GetPlayer()->GetPhase())
508 {
509 if ((nDist = m_session->GetPlayer()->CalcDistance(Itr)) < cDist)
510 {
511 cDist = nDist;
512 nDist = 0.0f;
513 GObj = static_cast<GameObject*>(Itr);
514 }
515 }
516 }
517 }
518
519
520 if (GObj == nullptr)
521 {
522 redSystemMessage(m_session, "No inrange GameObject found.");
523 return true;
524 }
525
526 m_session->GetPlayer()->setSelectedGo(GObj->getGuid());
527
528 //reset last rotation values on selecting a new go.
529 m_session->GetPlayer()->m_goLastXRotation = 0.0f;
530 m_session->GetPlayer()->m_goLastYRotation = 0.0f;
531
532 greenSystemMessage(m_session, "Selected GameObject [ {} ] which is {} meters away from you.",
533 sMySQLStore.getGameObjectProperties(GObj->getEntry())->name, m_session->GetPlayer()->CalcDistance(GObj));
534
535 return true;
536}

620{
621 uint32_t go_faction = 0;
622 if (sscanf(args, "%u", &go_faction) < 1)
623 {
624 redSystemMessage(m_session, "Wrong Syntax! Use: .gobject setfaction <faction>");
625 return true;

627
628 auto gameobject = m_session->GetPlayer()->getSelectedGo();
629 if (gameobject == nullptr)
630 {
631 redSystemMessage(m_session, "No GameObject is selected.");

633 }
634
635 auto faction_template = sFactionTemplateStore.lookupEntry(go_faction);

644 auto go_spawn = gameobject->m_spawn;
645

652 greenSystemMessage(m_session, "Faction changed in gameobject_spawns table for spawn ID: {}.", go_spawn->id);
