46void LoadGameObjectModelList(std::string const& dataPath)
47{
48#ifndef NO_CORE_FUNCS
49 auto startTime = Util::TimeNow();
50#endif
51
52 FILE* model_list_file = fopen((dataPath + "/" + VMAP::GAMEOBJECT_MODELS).c_str(), "rb");
53 if (!model_list_file)
54 {
55 sLogger.failure("Unable to open '{}' file.", VMAP::GAMEOBJECT_MODELS);
56 return;
57 }
58
59 uint32_t name_length, displayId;
60 uint8_t isWmo;
61 char buff[500];
62 while (true)
63 {
64 Vector3 v1, v2;
65 if (fread(&displayId, sizeof(uint32_t), 1, model_list_file) != 1)
66 if (feof(model_list_file)) // EOF flag is only set after failed reading attempt
67 break;
68
69 if (fread(&isWmo, sizeof(uint8_t), 1, model_list_file) != 1
70 || fread(&name_length, sizeof(uint32_t), 1, model_list_file) != 1
71 || name_length >= sizeof(buff)
72 || fread(&buff, sizeof(char), name_length, model_list_file) != name_length
73 || fread(&v1, sizeof(Vector3), 1, model_list_file) != 1
74 || fread(&v2, sizeof(Vector3), 1, model_list_file) != 1)
75 {
76 sLogger.failure("File '{}' seems to be corrupted!", VMAP::GAMEOBJECT_MODELS);
77 break;
78 }
79
80 if (v1.isNaN() || v2.isNaN())
81 {
82 sLogger.failure("File '{}' Model '{}' has invalid v1{} v2{} values!", VMAP::GAMEOBJECT_MODELS, std::string(buff, name_length), v1.toString(), v2.toString());
83 continue;
84 }
85
86 model_list.insert
87 (
88 ModelList::value_type(displayId, GameobjectModelData(std::string(buff, name_length), AABox(v1, v2), isWmo != 0))
89 );
90 }
91
92 fclose(model_list_file);
93 sLogger.info("LoadGameObjectModelList : Loaded {} GameObject models in {} ms", uint32_t(model_list.size()), static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)));
94}

102bool GameObjectModel::initialize(std::unique_ptr<GameObjectModelOwnerBase> modelOwner, std::string const& dataPath)
103{
104 ModelList::const_iterator it = model_list.find(modelOwner->GetDisplayId());
105 if (it == model_list.end())
106 return false;
107
108 G3D::AABox mdl_box(it->second.bound);
109 // ignore models with no bounds
110 if (mdl_box == G3D::AABox::zero())
111 {
112 sLogger.failure("GameObject model {} has zero bounds, loading skipped", it->second.name);
113 return false;
114 }
115
116 iModel = ((VMAP::VMapManager2*)VMAP::VMapFactory::createOrGetVMapManager())->acquireModelInstance(dataPath + "vmaps/", it->second.name);
117
118 if (!iModel)
119 return false;
120
121 name = it->second.name;
122 iPos = modelOwner->GetPosition();
123 phasemask = modelOwner->GetPhaseMask();
124 iScale = modelOwner->GetScale();
125 iInvScale = 1.f / iScale;
126
127 G3D::Matrix3 iRotation = G3D::Matrix3::fromEulerAnglesZYX(modelOwner->GetOrientation(), 0, 0);
128 iInvRot = iRotation.inverse();
129 // transform bounding box:
130 mdl_box = AABox(mdl_box.low() * iScale, mdl_box.high() * iScale);
131 AABox rotated_bounds;
132 for (int i = 0; i < 8; ++i)
133 rotated_bounds.merge(iRotation * mdl_box.corner(i));
134
135 iBound = rotated_bounds + iPos;
136#ifdef SPAWN_CORNERS
137 // test:
138 for (int i = 0; i < 8; ++i)
139 {
140 Vector3 pos(iBound.corner(i));
141 modelOwner->DebugVisualizeCorner(pos);
142 }
143#endif
144
145 owner = std::move(modelOwner);
146 isWmo = it->second.isWmo;
147 return true;
148}

161bool GameObjectModel::intersectRay(const G3D::Ray& ray, float& MaxDist, bool StopAtFirstHit, uint32_t ph_mask) const
162{
163 if (!(phasemask & ph_mask) || !owner->IsSpawned())
164 return false;
165
166 float time = ray.intersectionTime(iBound);
167 if (time == G3D::finf())
168 return false;
169
170 // child bounds are defined in object space:
171 Vector3 p = iInvRot * (ray.origin() - iPos) * iInvScale;
172 Ray modRay(p, iInvRot * ray.direction());
173 float distance = MaxDist * iInvScale;
174 bool hit = iModel->IntersectRay(modRay, distance, StopAtFirstHit);
175 if (hit)
176 {
177 distance *= iScale;
178 MaxDist = distance;
179 }
180 return hit;
181}

183void GameObjectModel::intersectPoint(G3D::Vector3 const& point, VMAP::AreaInfo& info, uint32_t ph_mask) const
184{
185 if (!(phasemask & ph_mask) || !owner->IsSpawned() || !isMapObject())
186 return;
187
188 if (!iBound.contains(point))
189 return;
190
191 // child bounds are defined in object space:
192 Vector3 pModel = iInvRot * (point - iPos) * iInvScale;
193 Vector3 zDirModel = iInvRot * Vector3(0.f, 0.f, -1.f);
194 float zDist;
195 if (iModel->IntersectPoint(pModel, zDirModel, zDist, info))
196 {
197 Vector3 modelGround = pModel + zDist * zDirModel;
198 float world_Z = ((modelGround * iInvRot) * iScale + iPos).z;
199 if (info.ground_Z < world_Z)
200 info.ground_Z = world_Z;
201 }
202}

230bool GameObjectModel::getLiquidLevel(G3D::Vector3 const& point, VMAP::LocationInfo& info, float& liqHeight) const
231{
232 // child bounds are defined in object space:
233 Vector3 pModel = iInvRot * (point - iPos) * iInvScale;
234 //Vector3 zDirModel = iInvRot * Vector3(0.f, 0.f, -1.f);
235 float zDist;
236 if (info.hitModel->GetLiquidLevel(pModel, zDist))
237 {
238 // calculate world height (zDist in model coords):
239 // assume WMO not tilted (wouldn't make much sense anyway)
240 liqHeight = zDist * iScale + iPos.z;
241 return true;
242 }
243 return false;
