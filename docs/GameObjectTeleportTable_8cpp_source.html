14//////////////////////////////////////////////////////////////////////////////////////////
15//\brief This script will add support for an SQL table called gameobject_teleports.
16// This table can be used to teleport players when they use a game object such
17// as a door or portal. Any object used in this table should exist in the
18// gameobject_properties table, and be of type 10. Custom portal can use the generic
19// display id of 6831. Portals also have the parameter_1 field set to '1'.
20//////////////////////////////////////////////////////////////////////////////////////////
21
22struct GameobjectTeleport
23{
24 uint32_t mapid;
25 float x, y, z, o;
26 uint32_t req_level;
27 uint8_t req_class;
28 uint32_t req_achievement;
29};

41 void OnActivate(Player* pPlayer) override
42 {
43 std::map<uint32_t, GameobjectTeleport*>::iterator itr = m_teleStorage.find(this->_gameobject->getEntry());
44 if (itr != m_teleStorage.end())
45 {
46 GameobjectTeleport* gt = itr->second;
47 uint32_t required_level = gt->req_level;
48 uint8_t req_class = gt->req_class;
49 uint32_t req_achievement = gt->req_achievement;
50
51 if (required_level > pPlayer->getLevel())
52 {
53 pPlayer->broadcastMessage("You must be at least level %u to use this portal", required_level);
54 return;
55 }
56 else if (req_class != 0 && req_class != pPlayer->getClass())
57 {
58 pPlayer->broadcastMessage("You do not have the required class to use this Portal");
59 return;
60 }
61#if VERSION_STRING > TBC
62 else if (req_achievement != 0 && pPlayer->getAchievementMgr()->hasCompleted(req_achievement))
63 {
64 pPlayer->broadcastMessage("You do not have the required achievement to use this Portal");
65 return;
66 }
67#endif
68 else
69 {
70 uint32_t mapid = gt->mapid;
71 LocationVector location;
72 location.x = gt->x;
73 location.y = gt->y;
74 location.z = gt->z;
75 location.o = gt->o;
76
77 pPlayer->safeTeleport(mapid, 0, location);
78 }
79 }
80 }
81 static GameObjectAIScript* Create(GameObject* GO) { return new CustomTeleport(GO); }
82};

84void InitializeGameObjectTeleportTable(ScriptMgr* mgr)
85{
86 auto result = WorldDatabase.Query("SELECT entry, mapid, x_pos, y_pos, z_pos, orientation, required_level, required_class, required_achievement FROM gameobject_teleports");
87 if (result != NULL)
88 {
89 // Check if the SQL table is setup correctly
90 if (result->GetFieldCount() < 9)
91 {
92 DLLLogDetail("Error: Custom portals disabled, invalid 'gameobject_teleports' table.");
93 return;
94 }
95 do
96 {
97 GameobjectTeleport* gt = new GameobjectTeleport;
98 Field* fields = result->Fetch();
99 uint32_t entry = fields[0].asUint32();
100 gt->mapid = fields[1].asUint32();
101 gt->x = fields[2].asFloat();
102 gt->y = fields[3].asFloat();
103 gt->z = fields[4].asFloat();
104 gt->o = fields[5].asFloat();
105 gt->req_level = fields[6].asUint32();
106 gt->req_class = fields[7].asUint8();
107 gt->req_achievement = fields[8].asUint32();
108 m_teleStorage[entry] = gt;
109 mgr->register_gameobject_script(entry, &CustomTeleport::Create);
110 } while (result->NextRow());
111 }
