314bool GameObject::loadFromDB(MySQLStructure::GameobjectSpawn* spawn, WorldMap* map, bool addToWorld)
315{
316 if (!spawn)
317 {
318 sLogger.failure("Gameobject (spawnId: {}) not found in table gameobject_spawns, cant load.");
319 return false;
320 }
321
322 if (!map || !map->getBaseMap())
323 {
324 sLogger.failure("Gameobject (spawnId: {}) invalid WorldMap or base data Invalid, cant load.");
325 return false;
326 }
327
328 uint32_t entry = spawn->entry;
329
330 m_spawnId = spawn->id;
331 if (!create(entry, map, spawn->phase, spawn->spawnPoint, spawn->rotation, spawn->state))
332 return false;
333
334 m_spawnedByDefault = true;
335
336 if (!GetGameObjectProperties()->getDespawnPossibility() && !GetGameObjectProperties()->isDespawnAtAction())
337 {
338 setFlags(GO_FLAG_NEVER_DESPAWN);
339 m_respawnDelayTime = 0;
340 m_respawnTime = 0;
341 }
342 else
343 {
344 m_respawnDelayTime = spawn->spawntimesecs;
345 m_respawnTime = map->getGORespawnTime(spawn->id);
346
347 // ready to respawn
348 if (m_respawnTime && m_respawnTime <= Util::getTimeNow())
349 {
350 m_respawnTime = 0;
351 map->removeRespawnTime(SPAWN_TYPE_GAMEOBJECT, spawn->id);
352 }
353 }
354
355 m_spawn = spawn;
356
357 // add to insert Pool
358 if (addToWorld)
359 map->AddObject(this);
360
361 return true;
362}

446bool GameObject::create(uint32_t entry, WorldMap* map, uint32_t phase, LocationVector const& position, QuaternionData const& rotation, GameObject_State state, uint32_t spawnId)
447{
448 gameobject_properties = sMySQLStore.getGameObjectProperties(entry);
449 if (gameobject_properties == nullptr)
450 {
451 sLogger.failure("Something tried to create a GameObject with invalid entry {}", entry);
452 return false;
453 }
454
455 if (gameobject_properties->type == GAMEOBJECT_TYPE_MO_TRANSPORT)
456 {
457 sLogger.failure("Gameobject (GUID: {} Entry: {}) not created: gameobject type GAMEOBJECT_TYPE_MO_TRANSPORT cannot be manually created.", getGuidLow(), entry);
458 return false;
459 }
460
461 Object::_Create(map->getBaseMap()->getMapId(), position.x, position.y, position.z, position.o);
462 setEntry(entry);
463 SetPosition(position);
464 setDisplayId(gameobject_properties->display_id);
465 m_phase = phase;
466
467 setLocalRotation(rotation.x, rotation.y, rotation.z, rotation.w);
468
469#if VERSION_STRING > TBC
470 MySQLStructure::GameObjectSpawnExtra const* gameObjectAddon = sMySQLStore.getGameObjectExtra(getSpawnId());
471
472 // For most of gameobjects is (0, 0, 0, 1) quaternion, there are only some transports with not standard rotation
473 QuaternionData parentRotation;
474 if (gameObjectAddon)
475 parentRotation = gameObjectAddon->parentRotation;
476
477 setParentRotation(parentRotation);
478#else
479 write(gameObjectData()->rotation[0], rotation.x);
480 write(gameObjectData()->rotation[1], rotation.y);
481
482 write(gameObjectData()->o, position.o);
483
484 float rotationZ = rotation.z;
485 float rotationW = rotation.w;
486 if (rotationZ == 0.0f && rotationW == 0.0f)
487 {
488 rotationZ = sin(position.o / 2);
489 rotationW = cos(position.o / 2);
490 }
491
492 write(gameObjectData()->rotation[2], rotationZ);
493 write(gameObjectData()->rotation[3], rotationW);
494
495#endif
496
497 setScale(gameobject_properties->size);
498 SetFaction(0);
499 setFlags(0);
500
501 // Spawn Overrides
502 if (MySQLStructure::GameObjectSpawnOverrides const* goOverride = sMySQLStore.getGameObjectOverride(getSpawnId()))
503 {
504 setScale(goOverride->scale);
505 SetFaction(goOverride->faction);
506 setFlags(goOverride->flags);
507 }
508
509 m_model = createModel();
510
511 setGoType(static_cast<uint8_t>(gameobject_properties->type));
512 m_prevGoState = state;
513 setState(state);
514 setArtKit(0);
515
516 switch (gameobject_properties->type)
517 {
518 case GAMEOBJECT_TYPE_FISHINGHOLE:
519 {
520 setAnimationProgress(0);
521 dynamic_cast<GameObject_FishingHole*>(this)->setMaxOpen(Util::getRandomUInt(gameobject_properties->fishinghole.max_success_opens, gameobject_properties->fishinghole.max_success_opens));
522 } break;
523 case GAMEOBJECT_TYPE_DESTRUCTIBLE_BUILDING:
524 {
525 auto* const destructible = dynamic_cast<GameObject_Destructible*>(this);
526 destructible->setHP(gameobject_properties->destructible_building.intact_num_hits + gameobject_properties->destructible_building.damaged_num_hits);
527 destructible->setMaxHP(destructible->GetHP());
528 setAnimationProgress(255);
529 } break;
530 case GAMEOBJECT_TYPE_TRANSPORT:
531 {
532 m_overrides = GAMEOBJECT_INFVIS | GAMEOBJECT_ONMOVEWIDE; //Make it forever visible on the same map;
533
534 #if VERSION_STRING == Classic
535 m_updateFlag = (m_updateFlag | UPDATEFLAG_TRANSPORT);
536 #endif
537 #if VERSION_STRING == TBC
538 m_updateFlag = (m_updateFlag | UPDATEFLAG_TRANSPORT);
539 #endif
540 #if VERSION_STRING == WotLK
541 m_updateFlag = (m_updateFlag | UPDATEFLAG_TRANSPORT) & ~UPDATEFLAG_POSITION;
542 #endif
543 #if VERSION_STRING == Cata
544 m_updateFlag = (m_updateFlag | UPDATEFLAG_TRANSPORT) & ~UPDATEFLAG_POSITION;
545 #endif
546 #if VERSION_STRING == Mop
547 m_updateFlag = (m_updateFlag | UPDATEFLAG_TRANSPORT) & ~UPDATEFLAG_POSITION;
548 #endif
549
550 setLevel(gameobject_properties->transport.pause);
551 setState(gameobject_properties->transport.startOpen ? GO_STATE_OPEN : GO_STATE_CLOSED);
552 setAnimationProgress(0);
553 m_goValue.CurrentSeg = 0;
554 m_goValue.AnimationInfo = sTransportHandler.getTransportAnimInfo(entry);
555 m_goValue.PathProgress = 0;
556 } break;
557 case GAMEOBJECT_TYPE_FISHINGNODE:
558 {
559 setAnimationProgress(0);
560 } break;
561 default:
562 setAnimationProgress(100);
563 break;
564 }
565
566 InitAI();
567
568 if (spawnId)
569 m_spawnId = spawnId;
570
571 if (uint32_t linkedEntry = gameobject_properties->getLinkedGameObjectEntry())
572 {
573 GameObject* linkedGO = map->createGameObject(linkedEntry);
574 if (linkedGO)
575 {
576 if (linkedGO->create(linkedEntry, map, phase, position, rotation, GO_STATE_CLOSED))
577 {
578 setLinkedTrap(linkedGO);
579 linkedGO->PushToWorld(map);
580 }
581 else
582 {
583 delete linkedGO;
584 }
585 }
586 }
587
588 // Check if GameObject is Large
589 if (gameobject_properties->isLargeGameObject())
590 m_overrides = GAMEOBJECT_AREAWIDE; // not implemented yet
591
592 // Check if GameObject is Infinite
593 if (gameobject_properties->isInfiniteGameObject())
594 m_overrides = GAMEOBJECT_MAPWIDE;
595
596 return true;
597}

653QuaternionData GameObject::getWorldRotation() const
654{
655 QuaternionData localRotation = getLocalRotation();
656 if (Transporter* transport = GetTransport())
657 {
658 QuaternionData worldRotation = transport->getWorldRotation();
659
660 G3D::Quat worldRotationQuat(worldRotation.x, worldRotation.y, worldRotation.z, worldRotation.w);
661 G3D::Quat localRotationQuat(localRotation.x, localRotation.y, localRotation.z, localRotation.w);
662
663 G3D::Quat resultRotation = localRotationQuat * worldRotationQuat;
664

670void GameObject::updatePackedRotation()
671{
672 static const int32_t PACK_YZ = 1 << 20;
673 static const int32_t PACK_X = PACK_YZ << 1;
674
675 static const int32_t PACK_YZ_MASK = (PACK_YZ << 1) - 1;
676 static const int32_t PACK_X_MASK = (PACK_X << 1) - 1;
677
678 int8_t w_sign = (m_localRotation.w >= 0.f ? 1 : -1);
679 int64_t x = int32_t(m_localRotation.x * PACK_X) * w_sign & PACK_X_MASK;
680 int64_t y = int32_t(m_localRotation.y * PACK_YZ) * w_sign & PACK_YZ_MASK;
681 int64_t z = int32_t(m_localRotation.z * PACK_YZ) * w_sign & PACK_YZ_MASK;
682 m_packedRotation = z | (y << 21) | (x << 42);
683
684#if VERSION_STRING <= TBC
685 write(gameObjectData()->rotation[0], m_localRotation.x);
686 write(gameObjectData()->rotation[1], m_localRotation.y);
687 write(gameObjectData()->rotation[2], m_localRotation.z);
688 write(gameObjectData()->rotation[3], m_localRotation.w);
689#endif
690}

803void GameObject::Update(unsigned long time_passed)
804{
805 if (m_event_Instanceid != m_instanceId)
806 {
807 event_Relocate();
808 return;
809 }
810
811 if (!IsInWorld() || m_deleted)
812 return;
813
814 // Update Spells
815 _UpdateSpells(time_passed);
816
817 // Respawn Handling
818 if (m_despawnDelay)
819 {
820 if (m_despawnDelay > time_passed)
821 {
822 m_despawnDelay -= time_passed;
823 }
824 else
825 {
826 m_despawnDelay = 0;
827 despawn(0, m_despawnRespawnTime);
828 }
829 }
830
831 switch (m_lootState)
832 {
833 case GO_NOT_READY:
834 {
835 _internalUpdateOnState(time_passed);
836
837 if (getGoType() == GAMEOBJECT_TYPE_CHEST)
838 {
839 if (dynamic_cast<GameObject_Chest*>(this)->getRestockTime() > Util::getTimeNow())
840 return;
841 }
842 } [[fallthrough]];
843 case GO_READY:
844 {
845 if (!m_loadedFromDB)
846 {
847 if (m_respawnTime > 0) // timer on
848 {
849 time_t now = Util::getTimeNow();
850 if (m_respawnTime <= now) // timer expired
851 {
852 m_respawnTime = 0;
853 m_usetimes = 0;
854
855 _internalUpdateOnState(time_passed);
856
857 if (m_lootState == GO_JUST_DEACTIVATED)
858 return;
859
860 // Despawn timer
861 if (!m_spawnedByDefault)
862 {
863 // Can be despawned or destroyed
864 setLootState(GO_JUST_DEACTIVATED);
865 return;
866 }
867
868 // Respawn timer
869 getWorldMap()->AddObject(this);
870 }
871 }
872 }
873
874 // Set respawn timer
875 if (m_loadedFromDB && m_respawnTime > 0)
876 saveRespawnTime();
877
878 if (isSpawned())
879 {
880 GameObjectProperties const* goInfo = GetGameObjectProperties();
881 if (goInfo->type == GAMEOBJECT_TYPE_TRAP)
882 {
883 if (Util::getMSTime() < m_cooldownTime)
884 break;
885
886 // Type 2 (bomb) does not need to be triggered by a unit and despawns after casting its spell.
887 if (goInfo->trap.charges == 2)
888 {
889 setLootState(GO_ACTIVATED);
890 break;
891 }
892
893 // Type 0 despawns after being triggered, type 1 does not.
894 // Trap Activasion Radius, Casting Radius comes from the Uses Spell
895 float radius;
896 if (!goInfo->trap.radius)
897 {
898 // Battleground traps: data2 == 0 && data5 == 3
899 if (goInfo->trap.cooldown != 3)
900 break;
901
902 radius = 3.f;
903 }
904 else
905 radius = goInfo->trap.radius / 2.f;
906
907 // Pointer to appropriate target if found any
908 Unit* target = nullptr;
909
910 // Proper Gameobject Casting is not implemented yet, so lets search for targets
911 if (getUnitOwner())
912 {
913 // Hunter trap: Units which are Unfriendly Gameobject owner
914 for (auto itr : getInRangeObjectsSet())
915 {
916 if (radius > itr->getDistance(this->GetPosition()))
917 {
918 if (itr->isCreatureOrPlayer() && this->isValidTarget(itr))
919 target = itr->ToUnit();
920 }
921 }
922 }
923 else
924 {
925 // Environmental trap: Target Here can be any Player in Distance
926 Player* player = nullptr;
927 for (auto itr : getInRangePlayersSet())
928 {
929 if (radius > itr->getDistance(this->GetPosition()))
930 target = player;
931 }
932 }
933
934 if (target)
935 setLootState(GO_ACTIVATED, target);
936
937 }
938 else if (uint32_t max_charges = goInfo->getCharges())
939 {
940 if (m_usetimes >= max_charges)
941 {
942 m_usetimes = 0;
943 setLootState(GO_JUST_DEACTIVATED); // can be despawned or destroyed
944 }
945 }
946 }
947 } break;
948 case GO_ACTIVATED:
949 {
950 _internalUpdateOnState(time_passed);
951 } break;
952 case GO_JUST_DEACTIVATED:
953 {
954 // If nearby linked trap exists, despawn it
955 if (GameObject* linkedTrap = getLinkedTrap())
956 linkedTrap->despawn(0, 0);
957
958 _internalUpdateOnState(time_passed);
959
960 // Do not delete chests or goobers that are not consumed on loot, while still allowing them to despawn when they expire if summoned
961 bool isSummonedAndExpired = (getUnitOwner() || getSpellId()) && m_respawnTime == 0;
962 if ((getGoType() == GAMEOBJECT_TYPE_CHEST || getGoType() == GAMEOBJECT_TYPE_GOOBER) && !GetGameObjectProperties()->isDespawnAtAction() && !isSummonedAndExpired)
963 {
964 if (getGoType() == GAMEOBJECT_TYPE_CHEST && GetGameObjectProperties()->chest.restock_time > 0)
965 {
966 // Start restock timer when the chest is fully looted
967 dynamic_cast<GameObject_Chest*>(this)->setRestockTime(Util::getTimeNow() + GetGameObjectProperties()->chest.restock_time);
968 setLootState(GO_NOT_READY);
969 }
970 else
971 {
972 setLootState(GO_READY);
973 }
974
975 return;
976 }
977 else if (getCreatedByGuid() || getSpellId())
978 {
979 setRespawnTime(0);
980 Delete();
981 return;
982 }
983
984 setLootState(GO_NOT_READY);
985
986 //burning flags in some battlegrounds, if you find better condition, just add it
987 if (GetGameObjectProperties()->isDespawnAtAction() || getAnimationProgress() > 0)
988 {
989 sendGameobjectDespawnAnim();
990 //reset flags
991 if (MySQLStructure::GameObjectSpawnOverrides const* goOverride = sMySQLStore.getGameObjectOverride(m_spawnId))
992 setFlags(goOverride->flags);
993 }
994
995 if (!m_respawnDelayTime)
996 return;
997
998 if (!m_spawnedByDefault)
999 {
1000 m_respawnTime = 0;
1001
1002 if (m_spawnId)
1003 despawn(0, 0);
1004 else
1005 Delete();
1006
1007 return;
1008 }
1009
1010 uint32_t respawnDelay = m_respawnDelayTime;
1011 m_respawnTime = Util::getTimeNow() + respawnDelay;
1012
1013 // if option not set then object will be saved at grid unload
1014 // Otherwise just save respawn time to map object memory
1015 despawn(0, 0);
1016 } break;
1017 }
1018}

1021{
1022 if (delay > 0)
1023 {
1024 if (!m_despawnDelay || m_despawnDelay > delay)
1025 {
1026 m_despawnDelay = delay;
1027 m_despawnRespawnTime = forceRespawntime;
1028 }
1029 }
1030 else
1031 {
1032 if (!IsInWorld())
1033 return;
1034
1035 if (GetScript())
1036 GetScript()->OnDespawn();
1037
1038 if (m_spawn && m_loadedFromDB)
1039 {
1040 uint32_t const respawnDelay = (forceRespawntime > 0) ? forceRespawntime : m_spawn->spawntimesecs;
1041 saveRespawnTime(respawnDelay);
1042 }
1043 else if (!m_loadedFromDB) // Respawning for non Database Loaded Objects

1046 if (MapCell* pCell = GetMapCell())

1054 RemoveFromWorld(false);
1055 }
1056 else
1057 {
1058 sLogger.failure("GameObject::Despawn tries to respawn go {} without a valid MapCell, return!", this->getEntry());
1059 }
1060 return;
1061 }
1062
1063 RemoveFromWorld(true);
1064 expireAndDelete();
1065 }
