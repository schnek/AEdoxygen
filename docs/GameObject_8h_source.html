534//////////////////////////////////////////////////////////////////////////////////////////
535// Class implementing Type 18 (SUMMONING_RITUAL) GameObjects
536class GameObject_Ritual : public GameObject
537{
538public:
539 GameObject_Ritual(uint64_t GUID);
540 ~GameObject_Ritual();
541
542 void InitAI();
543
544 void onUse(Player* player) override;
545
546 struct RitualStruct
547 {
548 uint64_t CasterGUID = 0;
549 uint64_t TargetGUID = 0;
550 uint32_t SpellID = 0;
551 uint32_t CurrentMembers = 0;
552 uint32_t MaxMembers = 0;
553 std::vector<uint64_t> Members;
554
555 RitualStruct(uint32_t members) : MaxMembers(members), Members(members) {}
556
557 void Setup(uint64_t caster_guid, uint64_t target_guid, uint32_t spell_id)
558 {
559 CasterGUID = caster_guid;
560 TargetGUID = target_guid;
561 SpellID = spell_id;
562
563 AddMember(caster_guid);
564 }
565
566 uint64_t GetCasterGUID() { return CasterGUID; }
567 uint64_t GetTargetGUID() { return TargetGUID; }
568 uint32_t GetSpellID() { return SpellID; }
569
570 bool AddMember(uint64_t GUID)
571 {
572 uint32_t i = 0;
573 for (; i < MaxMembers; i++)
574 if (Members[i] == 0)
575 break;
576
577 if (i == MaxMembers)
578 return false;
579
580 Members[i] = GUID;
581 CurrentMembers++;
582
583 return true;
584 }
585
586 bool RemoveMember(uint64_t GUID)
587 {
588 uint32_t i = 0;
589 for (; i < MaxMembers; i++)
590 {
591 if (Members[i] == GUID)
592 {
593 Members[i] = 0;
594 CurrentMembers--;
595 return true;
596 }
597 }
598
599 return false;
600 }
601
602 bool HasMember(uint64_t GUID)
603 {
604 for (uint32_t i = 0; i < MaxMembers; i++)
605 if (Members[i] == GUID)
606 return true;
607
608 return false;
609 }
610
611 uint64_t GetMemberGUIDBySlot(uint32_t Slot) { return Members[Slot]; }
612
613 bool HasFreeSlots()
614 {
615 if (CurrentMembers < MaxMembers)
616 return true;
617
618 return false;
619 }
620
621 uint32_t GetMaxMembers() { return MaxMembers; }
622
623 void Finish() { SpellID = 0; }
624
625 bool IsFinished()
626 {
627 if (SpellID == 0)
628 return true;
629
630 return false;
631 }
632 };
633
634 RitualStruct* GetRitual() const { return Ritual.get(); }
635
636private:
637 std::unique_ptr<RitualStruct> Ritual;
638};
