23void GenericMovementGenerator::initialize(Unit* /*owner*/)
24{
25 if (hasFlag(MOVEMENTGENERATOR_FLAG_DEACTIVATED) && !hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING)) // Resume spline is not supported
26 {
27 removeFlag(MOVEMENTGENERATOR_FLAG_DEACTIVATED);
28 addFlag(MOVEMENTGENERATOR_FLAG_FINALIZED);
29 return;
30 }
31
32 removeFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING | MOVEMENTGENERATOR_FLAG_DEACTIVATED);
33 addFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED);
34
35 _duration->resetInterval(_splineInit.Launch());
36}

43bool GenericMovementGenerator::update(Unit* owner, uint32_t diff)
44{
45 if (!owner || hasFlag(MOVEMENTGENERATOR_FLAG_FINALIZED))
46 return false;
47
48 // Cyclic splines never expire, so update the duration only if it's not cyclic
49 if (!owner->movespline->isCyclic())
50 _duration->updateTimer(diff);
51
52 if (_duration->isTimePassed() || owner->movespline->Finalized())
53 {
54 addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
55 return false;
56 }
57 return true;
58}
