76void GossipTrainer::onHello(Object* object, Player* player)
77{
78 if (auto creature = dynamic_cast<Creature*>(object))
79 {
80 auto gossipTextId = sMySQLStore.getGossipTextIdForNpc(creature->getEntry());
81 if (!sMySQLStore.getNpcGossipText(gossipTextId))
82 gossipTextId = DefaultGossipTextId;
83
84 GossipMenu menu(creature->getGuid(), gossipTextId, player->getSession()->language);
85
86 if (const auto trainer = creature->GetTrainer())
87 {
88 if (!player->canTrainAt(trainer))
89 {
90 menu.setTextID(trainer->Cannot_Train_GossipTextId);
91 }
92 else
93 {
94 std::string name = creature->GetCreatureProperties()->Name;
95 std::string::size_type pos = name.find(' ');
96
97 if (pos != std::string::npos)

99
100 auto msg = std::string(player->getSession()->LocalizedGossipOption(ISEEK));
101 msg += std::string(player->getSession()->LocalizedGossipOption(TRAINING)) + ", " + name + ".";
102 menu.addItem(GOSSIP_ICON_TRAINER, 0, 1, msg);
103
104 if (creature->isVendor())
105 {
106 const auto vendorRestrictions = sMySQLStore.getVendorRestriction(creature->GetCreatureProperties()->Id);
107 if (player->canBuyAt(vendorRestrictions))
108 menu.addItem(GOSSIP_ICON_VENDOR, VENDOR, 2);
109 }
110 }
111 }
112
113 sQuestMgr.FillQuestMenu(creature, player, menu);
114
115 menu.sendGossipPacket(player);

372void GossipClassTrainer::onHello(Object* object, Player* player)
373{
374 if (auto creature = dynamic_cast<Creature*>(object))
375 {
376 const auto playerSession = player->getSession();
377
378 auto gossipTextId = sMySQLStore.getGossipTextIdForNpc(creature->getEntry());
379 if (!sMySQLStore.getNpcGossipText(gossipTextId))
380 gossipTextId = DefaultGossipTextId;
381
382 GossipMenu menu(object->getGuid(), gossipTextId, playerSession->language);
383
384 const auto playerClass = player->getClass();
385 auto trainer = creature->GetTrainer();
386
387 if (trainer)
388 {
389 if (trainer->RequiredClass != playerClass)
390 {
391 menu.setTextID(trainer->Cannot_Train_GossipTextId);
392 }
393 else
394 {
395 menu.setTextID(trainer->Can_Train_Gossip_TextId);
396
397 std::string menuItemName = playerSession->LocalizedGossipOption(ISEEK);
398 std::string creatureName = creature->GetCreatureProperties()->Name;
399
400 std::string::size_type pos = creatureName.find(' ');
401
402 if (pos != std::string::npos)
403 creatureName = creatureName.substr(0, pos);
404
405 switch (playerClass)
406 {
407 case ::MAGE:
408 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_MAGE));
409 break;
410 case ::SHAMAN:
411 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_SHAMAN));
412 break;
413 case ::WARRIOR:
414 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_WARRIOR));
415 break;
416 case ::PALADIN:
417 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_PALADIN));
418 break;
419 case ::WARLOCK:
420 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_WARLOCK));
421 break;
422 case ::HUNTER:
423 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_HUNTER));
424 break;
425 case ::ROGUE:
426 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_ROGUE));
427 break;
428 case ::DRUID:
429 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_DRUID));
430 break;
431 case ::PRIEST:
432 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_PRIEST));
433 break;
434 #if VERSION_STRING > TBC
435 case ::DEATHKNIGHT:
436 menuItemName += std::string(playerSession->LocalizedGossipOption(GI_DEATHKNIGHT));
437 break;
438 #endif
439 default:
440 break;
441 }
442 menuItemName += " ";
443 menuItemName += std::string(playerSession->LocalizedGossipOption(TRAINING)) + ", " + creatureName + ".";
444
445 menu.addItem(GOSSIP_ICON_TRAINER, 0, 1, menuItemName);
446
447 if (creature->getLevel() > TrainerTalentResetMinLevel && player->getLevel() > worldConfig.player.minTalentResetLevel && creature->GetTrainer()->RequiredClass == playerClass)
448 {
449 menu.addItem(GOSSIP_ICON_CHAT, CLASSTRAINER_TALENTRESET, 2);
450
451 if (player->getLevel() >= worldConfig.player.minDualSpecLevel && player->m_talentSpecsCount < 2)
452 menu.addItem(GOSSIP_ICON_CHAT, LEARN_DUAL_TS, 4);
453 }
454 }
455 }
456
457 sQuestMgr.FillQuestMenu(creature, player, menu);
458
459 menu.sendGossipPacket(player);
460 }
