31void InnkeeperGossip::onHello(Object* pObject, Player* Plr)
32{
33 Creature* pCreature = pObject->isCreature() ? static_cast<Creature*>(pObject) : nullptr;
34 if (pCreature == nullptr)
35 {
36 return;
37 }
38
39 uint32_t TextID = 820;
40 uint32_t Text = sMySQLStore.getGossipTextIdForNpc(pCreature->getEntry());
41 if (Text != 0)
42 {
43 MySQLStructure::NpcGossipText const* text = sMySQLStore.getNpcGossipText(Text);
44 if (text != nullptr)
45 {
46 TextID = Text;
47 }
48 }
49 GossipMenu menu(pCreature->getGuid(), TextID, 0);
50
51 // Halow's End started?
52 auto _now = std::chrono::system_clock::now();
53 auto _time_now = std::chrono::system_clock::to_time_t(_now);
54 tm * ct = std::localtime(&_time_now);
55 if (ct->tm_mon == 9 && (ct->tm_mday > 17 && ct->tm_mday <= 31))
56 {
57 if (!Plr->hasAurasWithId(SPELL_TRICK_OR_TREATED))
58 {
59 menu.addItem(GOSSIP_ICON_CHAT, GI_TRICK_TREAT, 4);
60 }
61 }
62
63 if (pCreature->isVendor())
64 {
65 menu.addItem(GOSSIP_ICON_VENDOR, VENDOR, 1);
66 }
67
68 menu.addItem(GOSSIP_ICON_CHAT, INNKEEPER, 2);
69 menu.addItem(GOSSIP_ICON_CHAT, INNKEEPERASK, 3);
70 sQuestMgr.FillQuestMenu(pCreature, Plr, menu);
71 menu.sendGossipPacket(Plr);
72}

74void InnkeeperGossip::onSelectOption(Object* pObject, Player* Plr, uint32_t Id, const char* /*Code*/, uint32_t /*gossipId*/)
75{
76 Creature* pCreature = pObject->isCreature() ? static_cast<Creature*>(pObject) : nullptr;
77
78 if (pCreature == nullptr)
79 return;
80
81 switch (Id)
82 {
83 case 1: // Vendor
84 {
85 Plr->getSession()->sendInventoryList(pCreature);
86 } break;
87 case 2: // Binder
88 {
89 Plr->getSession()->sendInnkeeperBind(pCreature);
90 } break;
91 case 3: // What can i do?
92 {
93 // Prepare second menu
94 GossipMenu::sendQuickMenu(pCreature->getGuid(), 1853, Plr, 2, GOSSIP_ICON_CHAT, Plr->getSession()->LocalizedGossipOption(INNKEEPER));
95 } break;
96 case 4: // Event of halloween
97 {
98 if (!Plr->hasAurasWithId(SPELL_TRICK_OR_TREATED))
99 {
100 pCreature->castSpell(Plr, SPELL_TRICK_OR_TREATED, true);
101
102 // either trick or treat, 50% chance
103 if (Util::getRandomUInt(1))
104 {
105 Plr->castSpell(Plr, SPELL_TREAT, true);
106 }
107 else
108 {
109 int32_t trickspell = 0;
110 switch (Util::getRandomUInt(5))
111 {
112 case 0:
113 {
114 trickspell = 24753; // cannot cast, random 30sec
115 } break;
116 case 1:
117 {
118 trickspell = 24713; // lepper gnome costume
119 } break;
120 case 2:
121 {
122 if (Plr->getGender() == 0)
123 {
124 trickspell = 24735; // male ghost costume
125 }
126 else
127 {
128 trickspell = 24736; // female ghostcostume
129 }
130 } break;
131 case 3:
132 {
133 if (Plr->getGender() == 0)
134 {
135 trickspell = 24711; // male ninja costume
136 }
137 else
138 {
139 trickspell = 24710; // female ninja costume
140 }
141 } break;
142 case 4:
143 {
144 if (Plr->getGender() == 0)
145 {
146 trickspell = 24708; // male pirate costume
147 }
148 else
149 {
150 trickspell = 24709; // female pirate costume
151 }
152 } break;
153 case 5:
154 {
155 trickspell = 24723; // skeleton costume
156 } break;
157 default:
158 {
159 return;
160 }
161 }
162 pCreature->castSpell(Plr, trickspell, true);
163 }
164 }
165 GossipMenu::senGossipComplete(Plr);
166 } break;
167 }
168}
