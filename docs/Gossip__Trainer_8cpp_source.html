13class MasterHammersmith : public GossipScript
14{
15public:
16 void onHello(Object* pObject, Player* plr) override
17 {
18 GossipMenu menu(pObject->getGuid(), 7245);
19 menu.addItem(GOSSIP_ICON_TRAINER, GI_T_HAMMERSMITH_LEARN, 1);
20 menu.addItem(GOSSIP_ICON_TRAINER, GI_T_HAMMERSMITH_UNLEARN, 2);
21
22 menu.sendGossipPacket(plr);
23 }
24
25 void onSelectOption(Object* pObject, Player* plr, uint32_t Id, const char* /*Code*/, uint32_t /*gossipId*/) override
26 {
27 uint32_t textid;
28 if (1 == Id)
29 {
30 if (!plr->hasSkillLine(164) || plr->getSkillLineCurrent(164, false) < 300)
31 textid = 20001;
32 else if (!plr->hasSpell(9787))
33 textid = 20002;
34 else if (plr->hasSpell(17040))
35 textid = 20003;
36 else if (plr->hasSpell(17041) || plr->hasSpell(17039) || plr->hasSpell(9788))
37 textid = 20004;
38 else
39 {
40 if (!plr->hasEnoughCoinage(600))
41 textid = 20005;
42 else
43 {
44 //pCreature->SendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Make good use of this knowledge.");
45 textid = 20006;
46 dynamic_cast<Creature*>(pObject)->castSpell(plr, 39099, true);
47 plr->modCoinage(-600);
48 }
49 }
50 }
51 else
52 {
53 if (!plr->hasSpell(17040))
54 textid = 20007;
55 else if (!plr->hasEnoughCoinage(250000) && plr->getLevel() <= 50 || !plr->hasEnoughCoinage(500000) && plr->getLevel() > 50 && plr->getLevel() <= 65
56 || !plr->hasEnoughCoinage(1000000) && plr->getLevel() > 65)
57 textid = 20008;
58 else
59 {
60 int32_t unlearnGold = 0;
61 if (plr->getLevel() <= 50)
62 unlearnGold = 250000;
63 if (plr->getLevel() > 50 && plr->getLevel() <= 65)
64 unlearnGold = 500000;
65 if (plr->getLevel() > 65)
66 unlearnGold = 1000000;
67
68 plr->modCoinage(-unlearnGold);
69 plr->removeSpell(17040, false);
70 textid = 20009;
71 }
72 }
73 GossipMenu::sendSimpleMenu(pObject->getGuid(), textid, plr);
74 }
75
76 void destroy() override { delete this; }
77
78};
