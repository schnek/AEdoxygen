129{

131 {
132 std::lock_guard lock(m_groupLock);
133
134 Player* pPlayer = sObjectMgr.getPlayer(info->guid);
135
136 if (m_isqueued)
137 {
138 m_isqueued = false;
139 sBattlegroundManager.removeGroupFromQueues(m_Id);
140 }
141
142 if (!IsFull())
143 {
144 SubGroup* subgroup = (subgroupid > 0) ? m_SubGroups[subgroupid].get() : FindFreeSubGroup();

148 }
149
150 if (subgroup->AddPlayer(info))

154

156 info->m_Group->RemovePlayer(info);
157
158 if (m_Leader == NULL && pPlayer)
159 m_Leader = info;
160
161 info->m_Group = this;
162 info->subGroup = (int8_t)subgroup->GetID();
163
164 ++m_MemberCount;
165 m_dirty = true;
166 Update(); // Send group update
167
168 return true;
169 }
170
171 info->m_Group = NULL;
172 info->subGroup = -1;
173 return false;
174 }
175
176 info->m_Group = NULL;
177 info->subGroup = -1;
178 return false;
179 }
180
181 return false;

199void Group::Update()
200{
201 if (m_updateblock)
202 return;
203
204 Player* pNewLeader = nullptr;
205
206 if (!m_Leader || m_Leader && sObjectMgr.getPlayer(m_Leader->guid))
207 {
208 pNewLeader = FindFirstPlayer();
209 if (pNewLeader)
210 m_Leader = pNewLeader->getPlayerInfo();
211 }
212
213 if (m_Looter && !sObjectMgr.getPlayer(m_Looter->guid))
214 {
215 if (!pNewLeader)
216 pNewLeader = FindFirstPlayer();
217
218 if (pNewLeader)
219 m_Looter = pNewLeader->getPlayerInfo();
220 }
221
222 std::lock_guard lock(m_groupLock);
223
224 for (uint8_t i = 0; i < m_SubGroupCount; i++)
225 {
226 if (SubGroup* sg1 = m_SubGroups[i].get())
227 {
228 for (const auto characterInfo : sg1->getGroupMembers())
229 {
230 // skip offline players
231 if (!sObjectMgr.getPlayer(characterInfo->guid))
232 continue;
233
234 WorldPacket data(SMSG_GROUP_LIST, (50 + (m_MemberCount * 20)));
235 data << uint8_t(m_GroupType);
236 data << uint8_t(characterInfo->subGroup);
237
238 uint8_t flags = 0;
239 if (characterInfo == m_assistantLeader)
240 flags |= 1;
241 if (characterInfo == m_mainTank)
242 flags |= 2;
243 if (characterInfo == m_mainAssist)
244 flags |= 4;
245 data << uint8_t(flags);
246
247 if (m_Leader && sObjectMgr.getPlayer(m_Leader->guid) && sObjectMgr.getPlayer(m_Leader->guid)->IsInBg())
248 data << uint8_t(1); //if the leader is in a BG, then the group is a BG group
249 else
250 data << uint8_t(0);
251
252 if (m_GroupType & GROUP_TYPE_LFD)
253 {
254 data << uint8_t(sLfgMgr.GetState(GetID()) == LFG_STATE_FINISHED_DUNGEON ? 2 : 0);
255 data << uint32_t(sLfgMgr.GetDungeon(GetID()));
256#if VERSION_STRING >= Cata
257 data << uint8_t(0); //unk
258#endif
259 }
260
261 data << uint64_t(GetID()); // Group guid
262 data << uint32_t(updatecounter++); // 3.3 - increments every time a group list update is being sent to client
263 data << uint32_t(m_MemberCount - 1); // we don't include self
264
265 for (uint8_t j = 0; j < m_SubGroupCount; j++)
266 {
267 if (SubGroup* sg2 = m_SubGroups[j].get())
268 {
269 for (const auto characterInfo2 : sg2->getGroupMembers())
270 {
271 if (characterInfo == characterInfo2) // skip self
272 continue;
273
274 // should never happen but just in case
275 if (characterInfo2 == nullptr)
276 continue;
277
278 Player* plr = sObjectMgr.getPlayer(characterInfo2->guid);
279 data << (plr ? plr->getName().c_str() : characterInfo2->name.c_str());
280 if(plr)
281 data << plr->getGuid();
282 else
283 data << characterInfo2->guid << uint32_t(0); // highguid
284
285 if (sObjectMgr.getPlayer(characterInfo2->guid))
286 data << uint8_t(1);
287 else
288 data << uint8_t(0);
289
290 data << uint8_t(characterInfo2->subGroup);
291
292 flags = 0;
293
294 if (characterInfo2 == m_assistantLeader)
295 flags |= 1;
296 if (characterInfo2 == m_mainTank)
297 flags |= 2;
298 if (characterInfo2 == m_mainAssist)
299 flags |= 4;
300
301 data << uint8_t(flags);
302 data << uint8_t(plr ? plr->retRoles() : 0); // Player roles
303 }
304 }
305 }
306
307 if (m_Leader)
308 data << m_Leader->guid << uint32_t(0);
309 else
310 data << uint64_t(0);
311
312 data << uint8_t(m_LootMethod);
313
314 if (m_Looter)
315 data << m_Looter->guid << uint32_t(0);
316 else
317 data << uint64_t(0);
318
319 data << uint8_t(m_LootThreshold);
320 data << uint8_t(m_difficulty);
321 data << uint8_t(m_raiddifficulty);
322 data << uint8_t(0); // 3.3 - unk
323
324 if (Player* loggedInPlayer = sObjectMgr.getPlayer(characterInfo->guid))
325 if (!loggedInPlayer->IsInWorld())
326 loggedInPlayer->copyAndSendDelayedPacket(&data);
327 else
328 loggedInPlayer->getSession()->SendPacket(&data);
329 }
330 }
331 }
332
333 if (m_dirty)
334 {
335 m_dirty = false;
336 SaveToDB();
337 }
338}

422void Group::RemovePlayer(CachedCharacterInfo* info)
423{
424 if (info == nullptr)
425 return;
426
427 Player* pPlayer = sObjectMgr.getPlayer(info->guid);
428
429 m_groupLock.lock();
430
431 if (m_isqueued)
432 {
433 m_isqueued = false;
434 sBattlegroundManager.removeGroupFromQueues(m_Id);
435 }
436
437 SubGroup* sg = nullptr;
438 if (info->subGroup >= 0 && info->subGroup < 8)
439 sg = m_SubGroups[info->subGroup].get();
440
441 if (!sg || sg->m_GroupMembers.find(info) == sg->m_GroupMembers.end())
442 {
443 for (uint8_t i = 0; i < m_SubGroupCount; ++i)
444 {
445 if (m_SubGroups[i])
446 {
447 if (m_SubGroups[i]->m_GroupMembers.find(info) != m_SubGroups[i]->m_GroupMembers.end())
448 {
449 sg = m_SubGroups[i].get();
450 break;
451 }
452 }
453 }
454 }
455
456 info->m_Group = nullptr;
457 info->subGroup = -1;
458
459 if (!sg)
460 {
461 m_groupLock.unlock();
462 return;
463 }
464
465 m_dirty = true;
466 sg->RemovePlayer(info);
467 --m_MemberCount;
468
469 m_groupLock.unlock();
470
471 // remove team member from the instance
472 if (pPlayer)
473 {
474 if (pPlayer->getSession())
475 {
476#if VERSION_STRING < Cata
477 SendNullUpdate(pPlayer);
478#endif
479
480 pPlayer->getSession()->SendPacket(SmsgGroupDestroyed().serialise().get());
481
482 pPlayer->getSession()->SendPacket(SmsgPartyCommandResult(2, pPlayer->getName(), ERR_PARTY_NO_ERROR).serialise().get());
483#if VERSION_STRING >= Cata
484 pPlayer->getSession()->sendEmptyGroupList(pPlayer);
485#endif
486 }
487
488 //Remove some party auras.
489 for (uint16_t i = AuraSlots::POSITIVE_SLOT_START; i < AuraSlots::POSITIVE_SLOT_END; i++)
490 {
491 auto* const aur = pPlayer->getAuraWithAuraSlot(i);
492 if (aur && aur->m_areaAura)
493 {
494 Object* caster = aur->getCaster();
495 if (caster && pPlayer->getGuid() != caster->getGuid())
496 aur->removeAura();
497 }
498 }
499 }
500
501 if (m_MemberCount < 2)
502 {
503 if (m_disbandOnNoMembers)
504 {
505 Disband();
506 return;
507 }
508 }
509
510 m_groupLock.lock();
511
512 /* eek! ;P */
513 Player* newPlayer = nullptr;
514 if (m_Looter == info)
515 {
516 newPlayer = FindFirstPlayer();
517 if (newPlayer)
518 m_Looter = newPlayer->getPlayerInfo();
519 else
520 m_Looter = nullptr;
521 }
522
523 if (m_Leader == info)
524 {
525 if (newPlayer == nullptr)
526 newPlayer = FindFirstPlayer();
527
528 if (newPlayer)
529 SetLeader(newPlayer, false);
530 else
531 m_Leader = nullptr;
532 }
533
534 m_groupLock.unlock();
535
536 Update();
537}

657
658 if (info->subGroup > 0 && info->subGroup < 8)
659 sg = m_SubGroups[info->subGroup].get();
660
661 if (sg == NULL || sg->m_GroupMembers.find(info) == sg->m_GroupMembers.end())
662 {
663 for (uint8_t i = 0; i < m_SubGroupCount; ++i)
664 {
665 if (m_SubGroups[i] != NULL)
666 {
667 if (m_SubGroups[i]->m_GroupMembers.find(info) != m_SubGroups[i]->m_GroupMembers.end())
668 {
669 sg = m_SubGroups[i].get();

671 }
672 }
673 }
674 }
675
676 if (sg == NULL)
677 {
678 return;
679 }
680
681 sg->RemovePlayer(info);

698void Group::LoadFromDB(Field* fields)
699{
700#define LOAD_ASSISTANT(__i, __d) g = fields[__i].asUint32(); if (g != 0) { __d = sObjectMgr.getCachedCharacterInfo(g); }
701
702 std::lock_guard lock(m_groupLock);
703
704 uint32_t g;
705 m_updateblock = true;
706
707 m_Id = fields[0].asUint32();
708
709 m_GroupType = fields[1].asUint8();
710 m_SubGroupCount = fields[2].asUint8();
711 m_LootMethod = fields[3].asUint8();
712 m_LootThreshold = fields[4].asUint8();
713 m_difficulty = fields[5].asUint8();
714 m_raiddifficulty = fields[6].asUint8();
715
716 LOAD_ASSISTANT(7, m_assistantLeader);
717 LOAD_ASSISTANT(8, m_mainTank);
718 LOAD_ASSISTANT(9, m_mainAssist);
719
720 // create groups
721 for (uint8_t i = 1; i < m_SubGroupCount; ++i)
722 m_SubGroups[i] = std::make_unique<SubGroup>(this, i);
723
724 // assign players into groups
725 for (uint8_t i = 0; i < m_SubGroupCount; ++i)
726 {
727 for (uint8_t j = 0; j < 5; ++j)
728 {
729 uint32_t guid = fields[10 + (i * 5) + j].asUint32();
730 if (guid == 0)
731 continue;
732
733 CachedCharacterInfo* inf = sObjectMgr.getCachedCharacterInfo(guid);
734 if (inf == NULL)
735 continue;
736
737 AddMember(inf);
738 }
739 }
740
741 m_updateblock = false;
742}

1159void Group::resetInstances(uint8_t method, bool isRaid, Player* SendMsgTo)
1160{
1161 if (isBGGroup() /* || isBFGroup()*/)
1162 return;
1163
1164 // method can be INSTANCE_RESET_ALL, INSTANCE_RESET_CHANGE_DIFFICULTY, INSTANCE_RESET_GROUP_DISBAND
1165
1166 // we assume that when the difficulty changes, all instances that can be reset will be
1167 InstanceDifficulty::Difficulties diff = getDifficulty(isRaid);
1168
1169 for (BoundInstancesMap::iterator itr = m_boundInstances[diff].begin(); itr != m_boundInstances[diff].end();)
1170 {
1171 InstanceSaved* instanceSave = itr->second.save;
1172 WDB::Structures::MapEntry const* entry = sMapStore.lookupEntry(itr->first);
1173 if (!entry || entry->isRaid() != isRaid || (!instanceSave->canReset() && method != INSTANCE_RESET_GROUP_DISBAND))
1174 {
1175 ++itr;
1176 continue;
1177 }
1178
1179 if (method == INSTANCE_RESET_ALL)
1180 {
1181 // the "reset all instances" method can only reset normal maps
1182 if (entry->map_type == MAP_RAID || diff == InstanceDifficulty::Difficulties::DUNGEON_HEROIC)
1183 {
1184 ++itr;
1185 continue;
1186 }
1187 }
1188
1189 bool isEmpty = true;
1190 // if the map is loaded, reset it
1191 WorldMap* map = sMapMgr.findWorldMap(instanceSave->getMapId(), instanceSave->getInstanceId());
1192 if (map && map->getBaseMap()->isInstanceMap() && !(method == INSTANCE_RESET_GROUP_DISBAND && !instanceSave->canReset()))
1193 {
1194 if (instanceSave->canReset())
1195 isEmpty = ((InstanceMap*)map)->reset(method);
1196 else
1197 isEmpty = !map->getPlayerCount();
1198 }
1199
1200 if (SendMsgTo)
1201 {
1202 if (!isEmpty)
1203 SendMsgTo->sendResetInstanceFailed(0, instanceSave->getMapId());
1204 else
1205 SendMsgTo->sendPacket(SmsgInstanceReset(instanceSave->getMapId()).serialise().get());
1206 }
1207
1208 if (isEmpty || method == INSTANCE_RESET_GROUP_DISBAND || method == INSTANCE_RESET_CHANGE_DIFFICULTY)
1209 {
1210 // do not reset the instance, just unbind if others are permanently bound to it
1211 if (isEmpty && instanceSave->canReset())
1212 {
1213 instanceSave->deleteFromDB();
1214 }
1215 else
1216 {
1217 CharacterDatabase.Execute("DELETE FROM group_instance WHERE instance = %u", instanceSave->getInstanceId());
1218 }
1219
1220 // i don't know for sure if hash_map iterators
1221 m_boundInstances[diff].erase(itr);
1222 itr = m_boundInstances[diff].begin();
1223 // this unloads the instance save unless online players are bound to it
1224 // (eg. permanent binds or GM solo binds)
1225 instanceSave->removeGroup(this);
1226 }
1227 else
1228 {
1229 ++itr;
1230 }
1231 }

1424{

1428 for (item = loot->items.begin(); item != loot->items.end(); ++item, ++itemSlot)

1433 if (item->is_blocked)

1435
