57void GruulTheDragonkillerAI::AIUpdate(unsigned long time_passed)
58{
59 if (_isCasting())
60 return;
61
62 scriptEvents.updateEvents(time_passed, getScriptPhase());
63
64 while (uint32_t eventId = scriptEvents.getFinishedEvent())
65 {
66 switch (eventId)
67 {
68 case EVENT_SHATTER:
69 _castAISpell(mShatter);
70 break;
71 case EVENT_HURTFUL_STRIKE:
72 {
73 Unit* pCurrentTarget = getCreature()->getAIInterface()->getCurrentTarget();
74 if (pCurrentTarget != nullptr)
75 {
76 Unit* pTarget = pCurrentTarget;
77 for (const auto& itr : getCreature()->getInRangePlayersSet())
78 {
79 Player* pPlayer = static_cast<Player*>(itr);
80 if (!pPlayer || !pPlayer->isAlive())
81 continue;
82 if (pPlayer->hasUnitFlags(UNIT_FLAG_FEIGN_DEATH))
83 continue;
84 if (getRangeToObject(pPlayer) > 8.0f)
85 continue;
86 if (getCreature()->getThreatManager().getThreat(pPlayer) >= getCreature()->getThreatManager().getThreat(pCurrentTarget))
87 continue;
88
89 pTarget = static_cast<Unit*>(pPlayer);
90 }
91
92 if (pTarget == pCurrentTarget)
93 _castAISpell(mHurtfulStrike);
94 else
95 getCreature()->castSpell(pTarget, SPELL_HURTFUL_STRIKE, true);
96 }
97 scriptEvents.addEvent(EVENT_HURTFUL_STRIKE, 8000);
98 }
99 break;
100 default:
101 break;
102 }
103 }
104}

137SpellScriptEffectDamage ShatterDamage::doCalculateEffect(Spell* spell, uint8_t effIndex, int32_t* dmg)
138{
139 if (effIndex != EFF_INDEX_0 || spell->getUnitTarget() == nullptr)
140 return SpellScriptEffectDamage::DAMAGE_DEFAULT;
141
142 float radius = spell->getEffectRadius(EFF_INDEX_0);
143 auto distance = spell->getUnitTarget()->GetDistance2dSq(spell->getCaster());
144
145 if (distance < 1.0f)
146 distance = 1.0f;
147
148 *dmg = Util::float2int32(*dmg * ((radius - distance) / radius));
149
150 return SpellScriptEffectDamage::DAMAGE_FULL_RECALCULATION;
151}
