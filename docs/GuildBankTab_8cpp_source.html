86bool GuildBankTab::writeSlotPacket(WorldPacket& data, uint8_t slotId, bool ignoreEmpty) const
87{
88 Item* pItem = getItem(slotId);
89 const uint32_t itemEntry = pItem ? pItem->getEntry() : 0;
90
91 if (!itemEntry && ignoreEmpty)
92 return false;
93
94 data << uint8_t(slotId);
95 data << uint32_t(itemEntry);
96 if (itemEntry)
97 {
98 data << uint32_t(0);
99
100 if (uint32_t random = pItem->getRandomPropertiesId())
101 {
102 data << uint32_t(random);
103 data << uint32_t(pItem->generateRandomSuffixFactor(pItem->getItemProperties()));
104 }
105 else
106 {
107 data << uint32_t(0);
108 }
109
110 data << uint32_t(pItem->getStackCount());
111 data << uint32_t(0);
112 data << uint8_t(abs(pItem->getSpellCharges(0)));
113
114 uint8_t enchCount = 0;
115 size_t enchCountPos = data.wpos();
116
117 data << uint8_t(enchCount);
118 for (uint32_t i = PERM_ENCHANTMENT_SLOT; i < MAX_ENCHANTMENT_SLOT; ++i)
119 {
120 if (uint32_t enchId = pItem->getEnchantmentId(static_cast<uint8_t>(EnchantmentSlot(i))))
121 {
122 data << uint8_t(i);
123 data << uint32_t(enchId);
124 ++enchCount;
125 }
126 }
127 data.put<uint8_t>(enchCountPos, enchCount);
128 }
129 return true;
130}

206bool GuildBankTab::setItem(uint8_t slotId, std::unique_ptr<Item> item)
207{
208 if (slotId >= MAX_GUILD_BANK_SLOTS && slotId != UNDEFINED_TAB_SLOT)
209 return false;
210
211 if (item != nullptr)
212 {
213 uint32_t slot_id = 0;
214 const auto itemGuid = item->getGuidLow();
215 if (slotId == 0 || slotId == UNDEFINED_TAB_SLOT)
216 {
217 for (uint8_t i = 0; i < MAX_GUILD_BANK_SLOTS; ++i)
218 {
219 if (mItems[i] == nullptr)
220 {
221 slot_id = i;
222 break;
223 }
224 }
225 }
226 else
227 {
228 slot_id = slotId;
229 }
230
231 mItems[slot_id] = std::move(item);
232 CharacterDatabase.Execute("INSERT INTO guild_bank_items VALUES (%u, %u, %u, %u)", mGuildId, static_cast<uint32_t>(mTabId), slot_id, itemGuid);
233 }
234 else
235 {
236 mItems[slotId] = nullptr;
237 CharacterDatabase.Execute("DELETE FROM guild_bank_items WHERE guildId = %u AND tabId = %u AND slotId = %u", mGuildId, static_cast<uint32_t>(mTabId), static_cast<uint32_t>(slotId));
238 }
239
240 return true;
