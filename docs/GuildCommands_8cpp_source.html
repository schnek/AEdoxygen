17bool ChatCommandHandler::HandleGuildCreateCommand(const char* args, WorldSession* m_session)
18{
19 Player* selected_player = GetSelectedPlayer(m_session, true, true);
20 if (selected_player == nullptr)
21 return true;
22
23 if (!*args)
24 return false;
25
26 if (selected_player->getGuild())
27 {
28 redSystemMessage(m_session, "{} is already in a guild.", selected_player->getName());
29 return true;
30 }
31
32 if (strlen(args) > 75)
33 {
34 systemMessage(m_session, "The name was too long by {}", uint32_t(strlen(args) - 75));
35 return true;
36 }
37
38 for (uint32_t i = 0; i < strlen(args); i++)
39 {
40 if (!isalpha(args[i]) && args[i] != ' ')
41 {
42 systemMessage(m_session, "Error, name can only contain chars A-Z and a-z.");
43 return true;
44 }
45 }
46
47 Charter tempCharter(0, selected_player->getGuidLow(), CHARTER_TYPE_GUILD);
48 tempCharter.setGuildName(std::string(args));
49
50 auto* guild = sGuildMgr.createGuild(selected_player, std::string(args));
51 if (guild == nullptr)
52 {
53 systemMessage(m_session, "Guild not created");
54 return true;
55 }
56
57 greenSystemMessage(m_session, "Guild created");
58 sGMLog.writefromsession(m_session, "Created guild '%s'", args);
59 return true;
60}

103//.guild join
104bool ChatCommandHandler::HandleGuildJoinCommand(const char* args, WorldSession* m_session)
105{
106 Player* selected_player = GetSelectedPlayer(m_session, true, true);
107 if (selected_player == nullptr)
108 return true;
109
110 if (selected_player->isInGuild())
111 {
112 redSystemMessage(m_session, "{} is already in a guild.", selected_player->getName());
113 return true;
114 }
115
116 if (!*args)
117 return false;
118
119 Guild* guild = sGuildMgr.getGuildByName(std::string(args));
120 if (guild != nullptr)
121 {
122 uint32_t memberCount = guild->getMembersCount();
123
124 if (worldConfig.guild.maxMembers > 0 && memberCount >= worldConfig.guild.maxMembers)
125 {
126 m_session->SystemMessage("That guild is full.");
127 return true;
128 }
129
130 guild->addMember(selected_player->getGuid(), 4);
131 greenSystemMessage(m_session, "You have joined the guild '{}'", guild->getName());
132 sGMLog.writefromsession(m_session, "Force joined guild '%s'", guild->getName().c_str());
133 return true;
134 }
135 redSystemMessage(m_session, "Guild {} is not a valid guildname!", args);
136
137 return false;
138}

159{
160 Player* selected_player = GetSelectedPlayer(m_session, true, true);
161 if (selected_player == nullptr)
162 return true;
163
164 if (selected_player->isInGuild())

181 sGMLog.writefromsession(m_session, "Changed guild name of '%s' to '%s'", selected_player->getGuild()->getName().c_str(), args);
182
183 return true;

186//.guild removeplayer
187bool ChatCommandHandler::HandleGuildRemovePlayerCommand(const char* /*args*/, WorldSession* m_session)
188{
189 Player* selected_player = GetSelectedPlayer(m_session, true, true);
190 if (selected_player == nullptr)
191 return true;
192
193 if (selected_player->getGuild())
194 {
195 redSystemMessage(m_session, "{} not in a guild.", selected_player->getName());
196 return true;
197 }
198
199 if (selected_player->getGuild()->getLeaderGUID() != selected_player->getGuid() || !m_session->GetPlayer()->isGMFlagSet())
200 {
201 redSystemMessage(m_session, "Only guild leaders and gms can remove players from a guild!");
202 return true;
203 }
204
205 greenSystemMessage(m_session, "Kicked {} from Guild: {}", selected_player->getName(), selected_player->getGuild()->getName());
206
207 if (selected_player != m_session->GetPlayer())
208 sGMLog.writefromsession(m_session, "Kicked %s from Guild %s", selected_player->getName().c_str(), selected_player->getGuild()->getName().c_str());
209
210 selected_player->getGuild()->handleRemoveMember(selected_player->getSession(), selected_player->getGuid());
211
212 return true;
213}
