293inline std::string _GetGuildEventString(GuildEvents event)
294{
295 switch (event)
296 {
297 case GE_PROMOTION: { return "Member promotion"; }
298 case GE_DEMOTION: { return "Member demotion"; }
299 case GE_MOTD: { return "Guild MOTD"; }
300 case GE_JOINED: { return "Member joined"; }
301 case GE_LEFT: { return "Member left"; }
302 case GE_REMOVED: { return "Member removed"; }
303 case GE_LEADER_IS: { return "Leader is"; }
304 case GE_LEADER_CHANGED: { return "Leader changed"; }
305 case GE_DISBANDED: { return "Guild disbanded"; }
306 case GE_TABARDCHANGE: { return "Tabard change"; }
307 case GE_RANK_UPDATED: { return "Rank updated"; }
308 case GE_RANK_DELETED: { return "Rank deleted"; }
309 case GE_SIGNED_ON: { return "Member signed on"; }
310 case GE_SIGNED_OFF: { return "Member signed off"; }
311 case GE_GUILDBANKBAGSLOTS_CHANGED: { return "Bank bag slots changed"; }
312 case GE_BANK_TAB_PURCHASED: { return "Bank tab purchased"; }
313 case GE_BANK_TAB_UPDATED: { return "Bank tab updated"; }
314 case GE_BANK_MONEY_SET: { return "Bank money set"; }
315 case GE_BANK_TAB_AND_MONEY_UPDATED: { return "Bank and money updated"; }
316 case GE_BANK_TEXT_CHANGED: { return "Bank tab text changed"; }
317 default:
318 break;
319 }
320 return "None";
321}
