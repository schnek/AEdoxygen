31void GuildFinderMgr::loadGuildSettingsFromDB()
32{
33 sLogger.info("Loading guild finder guild-related settings...");
34
35 CharacterDatabase.Execute("DELETE gfgs FROM guild_finder_guild_settings gfgs LEFT JOIN guilds g ON gfgs.guildId = g.guildId WHERE g.guildId IS NULL");
36
37 // 0 1 2 3 4 5 6 7
38 auto result = CharacterDatabase.Query("SELECT gfgs.guildId, gfgs.availability, gfgs.classRoles, gfgs.interests, gfgs.level, gfgs.listed, gfgs.comment, c.race "
39 "FROM guild_finder_guild_settings gfgs "
40 "LEFT JOIN guild_members gm ON gm.guildId = gfgs.guildId "
41 "LEFT JOIN characters c ON c.guid = gm.playerid LIMIT 1");
42
43 if (result == nullptr)
44 {
45 sLogger.info("Loaded 0 guild finder guild-related settings. Table `guild_finder_guild_settings` is empty.");
46 return;
47 }
48
49 uint32_t count = 0;
50 auto startTime = Util::TimeNow();
51 do
52 {
53 Field* fields = result->Fetch();
54 uint32_t guildId = fields[0].asUint32();
55 uint8_t availability = fields[1].asUint8();
56 uint8_t classRoles = fields[2].asUint8();
57 uint8_t interests = fields[3].asUint8();
58 uint8_t level = fields[4].asUint8();
59 bool listed = (fields[5].asUint8() != 0);
60 std::string comment = fields[6].asCString();
61
62 PlayerTeam guildTeam = TEAM_ALLIANCE;
63 if (auto raceEntry = sChrRacesStore.lookupEntry(fields[7].asUint8()))
64 {
65 if (raceEntry->team_id == 1)
66 {
67 guildTeam = TEAM_HORDE;
68 }
69 }
70
71 LFGuildSettings settings(listed, guildTeam, guildId, classRoles, availability, interests, level, comment);
72 _lfgGuildStore[guildId] = settings;
73
74 ++count;
75 } while(result->NextRow());
76
77 sLogger.info("Loaded {} guild finder guild-related settings in {} ms.", count, Util::GetTimeDifferenceToNow(startTime));
78}

232LFGuildStore GuildFinderMgr::getGuildsMatchingSetting(LFGuildPlayer& settings, PlayerTeam faction)
233{
234 LFGuildStore resultSet;
235 for (LFGuildStore::const_iterator itr = _lfgGuildStore.begin(); itr != _lfgGuildStore.end(); ++itr)
236 {
237 LFGuildSettings const& guildSettings = itr->second;
238
239 if (guildSettings.getTeam() != faction)
240 {
241 continue;
242 }
243
244 if (!(guildSettings.getAvailability() & settings.getAvailability()))
245 {
246 continue;
247 }
248
249 if (!(guildSettings.getClassRoles() & settings.getClassRoles()))
250 {
251 continue;
252 }
253
254 if (!(guildSettings.getInterests() & settings.getInterests()))
255 {
256 continue;
257 }
258
259 if (!(guildSettings.getLevel() & settings.getLevel()))
260 {
261 continue;
262 }
263
264 resultSet.insert(std::make_pair(itr->first, guildSettings));
265 }
266
267 return resultSet;

292void GuildFinderMgr::deleteGuild(uint32_t guildId)
293{
294 std::vector<MembershipRequest>::iterator itr = _membershipRequestStore[guildId].begin();
295 while (itr != _membershipRequestStore[guildId].end())
296 {
297 uint32_t applicant = itr->getPlayerGUID();
298
299 CharacterDatabase.Execute("DELETE FROM guild_finder_applicant WHERE guildId = %u AND playerGuid = %u", itr->getGuildId(), applicant);
300
301 CharacterDatabase.Execute("DELETE FROM guild_finder_guild_settings WHERE guildId = %u", itr->getGuildId());
302
303 if (Player* player = sObjectMgr.getPlayer(applicant))
304 {
305 sendMembershipRequestListUpdate(*player);
306 }
307
308 ++itr;
309 }
310
311 _membershipRequestStore.erase(guildId);
312 _lfgGuildStore.erase(guildId);
313
314 if (Guild* guild = sGuildMgr.getGuildById(guildId))
315 {
316 sendApplicantListUpdate(*guild);
317 }
