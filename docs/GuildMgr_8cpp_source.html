433void GuildMgr::loadGuildXpForLevelFromDB()
434{
435 auto startTime = Util::TimeNow();
436
437 GuildXPperLevel.resize(worldConfig.guild.maxLevel);
438 for (uint8_t level = 0; level < worldConfig.guild.maxLevel; ++level)
439 GuildXPperLevel[level] = 0;
440
441 // 0 1
442 auto result = WorldDatabase.Query("SELECT lvl, xp_for_next_level FROM guild_xp_for_level");
443 if (result == nullptr)
444 {
445 sLogger.debug("Loaded 0 xp for guild level definitions. DB table `guild_xp_for_level` is empty.");
446 return;
447 }
448
449 uint32_t count = 0;
450
451 do
452 {
453 Field* fields = result->Fetch();
454
455 uint32_t level = fields[0].asUint8();
456 uint32_t requiredXP = static_cast<uint32_t>(fields[1].asUint64());
457
458 if (level >= worldConfig.guild.maxLevel)
459 {
460 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "Table `guild_xp_for_level` includes invalid xp definitions for level {} which is higher than the defined levelcap in your config file! <skipped>", level);
461 continue;
462 }
463
464 GuildXPperLevel[level] = requiredXP;
465 ++count;
466
467 } while (result->NextRow());
468
469 // fill level gaps
470 for (uint8_t level = 1; level < worldConfig.guild.maxLevel; ++level)
471 {
472 if (!GuildXPperLevel[level])
473 {
474 sLogger.failure("Level {} does not have XP for guild level data. Using data of level [{}] + 1660000.", level + 1, level);
475 GuildXPperLevel[level] = GuildXPperLevel[level - 1U] + 1660000;
476 }
477 }
478
479 sLogger.debug("Loaded {} xp for guild level definitions in {} ms", count, static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)));
480}

482void GuildMgr::loadGuildRewardsFromDB()
483{
484 auto startTime = Util::TimeNow();
485
486 // 0 1 2 3 4
487 auto result = WorldDatabase.Query("SELECT entry, standing, racemask, price, achievement FROM guild_rewards");
488 if (result == nullptr)
489 {
490 sLogger.debug("Loaded 0 guild reward definitions. DB table `guild_rewards` is empty.");
491 return;
492 }
493
494 uint32_t count = 0;
495
496 do
497 {
498 GuildReward reward;
499 Field* fields = result->Fetch();
500 reward.entry = fields[0].asUint32();
501 reward.standing = fields[1].asUint8();
502 reward.racemask = fields[2].asInt32();
503 reward.price = fields[3].asUint64();
504 reward.achievementId = fields[4].asUint32();
505
506 if (!sItemStore.lookupEntry(reward.entry))
507 {
508 sLogger.failure("Guild rewards constains not existing item entry {}", reward.entry);
509 continue;
510 }
511
512 if (reward.standing >= 8)
513 {
514 sLogger.failure("Guild rewards contains wrong reputation standing {}, max is {}", uint32_t(reward.standing), 8 - 1);
515 continue;
516 }
517
518 GuildRewards.push_back(reward);
519 ++count;
520 } while (result->NextRow());
521
522 sLogger.debug("Loaded {} guild reward definitions in {} ms", count, static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)));
523}
