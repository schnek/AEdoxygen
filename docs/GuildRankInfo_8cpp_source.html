120void GuildRankInfo::createMissingTabsIfNeeded(uint8_t tabs, bool /*_delete*/, bool logOnCreate)
121{
122 for (uint8_t i = 0; i < tabs; ++i)
123 {
124 GuildBankRightsAndSlots& rightsAndSlots = mBankTabRightsAndSlots[i];
125 if (rightsAndSlots.getTabId() == i)
126 continue;
127
128 rightsAndSlots.setTabId(i);
129 if (mRankId == GR_GUILDMASTER)
130 rightsAndSlots.SetGuildMasterValues();
131
132 if (logOnCreate)
133 sLogger.failure("Guild {} has broken Tab {} for rank {}. Created default tab.", mGuildId, i, static_cast<uint32_t>(mRankId));
134
135 CharacterDatabase.Execute("REPLACE INTO guild_bank_rights VALUES(%u, %u, %u, %u, %u);",
136 mGuildId, i, static_cast<uint32_t>(mRankId), static_cast<uint32_t>(rightsAndSlots.getRights()), rightsAndSlots.getSlots());
137 }
138}
