64void Guild::sendGuildInvitePacket(WorldSession* session, std::string invitedName)
65{
66 const auto invitedPlayer = sObjectMgr.getPlayer(invitedName.c_str(), false);
67 const auto guild = session->GetPlayer()->getGuild();
68
69 if (invitedPlayer == nullptr)
70 {
71 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_INVITE, invitedName, GC_ERROR_PLAYER_NOT_FOUND_S).serialise().get());
72 return;
73 }
74
75 if (guild == nullptr)
76 {
77 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_CREATE, "", GC_ERROR_PLAYER_NOT_IN_GUILD).serialise().get());
78 return;
79 }
80
81 if (invitedPlayer->getGuildId())
82 {
83 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_INVITE, invitedPlayer->getName(), GC_ERROR_ALREADY_IN_GUILD_S).serialise().get());
84 return;
85 }
86
87 if (invitedPlayer->getInvitedByGuildId())
88 {
89 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_INVITE, invitedPlayer->getName(), GC_ERROR_ALREADY_INVITED_TO_GUILD).serialise().get());
90 return;
91 }
92
93 if (!session->GetPlayer()->getGuild()->_hasRankRight(session->GetPlayer()->getGuid(), GR_RIGHT_INVITE))
94 {
95 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_INVITE, "", GC_ERROR_PERMISSIONS).serialise().get());
96 return;
97 }
98
99 if (invitedPlayer->getTeam() != session->GetPlayer()->getTeam() && !session->GetPlayer()->getSession()->hasPermissions() && !worldConfig.player.isInterfactionGuildEnabled)
100 {
101 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_INVITE, "", GC_ERROR_NOT_ALLIED).serialise().get());
102 return;
103 }
104
105 const auto memberCount = guild->getMembersCount();
106 if (worldConfig.guild.maxMembers > 0 && memberCount >= worldConfig.guild.maxMembers)
107 {
108 session->SystemMessage("Your guild is full.");
109 return;
110 }
111
112 session->SendPacket(SmsgGuildCommandResult(GC_TYPE_INVITE, invitedName, GC_ERROR_SUCCESS).serialise().get());
113
114 guild->logEvent(GE_LOG_INVITE_PLAYER, session->GetPlayer()->getGuidLow(), invitedPlayer->getGuidLow());
115 invitedPlayer->setInvitedByGuildId(guild->getId());
116
117#if VERSION_STRING < Cata
118 invitedPlayer->getSession()->SendPacket(SmsgGuildInvite(session->GetPlayer()->getName(), guild->getName()).serialise().get());
119
120#else
121 invitedPlayer->getSession()->SendPacket(SmsgGuildInvite(session->GetPlayer()->getName(), guild->getName(), guild->getLevel(),
122 guild->getEmblemInfo(), guild->getId(), guild->getGUID()).serialise().get());
123#endif
124}

128bool Guild::create(Player* pLeader, std::string const& name)
129{
130 if (sGuildMgr.getGuildByName(name))
131 return false;
132
133 WorldSession* pLeaderSession = pLeader->getSession();
134 if (pLeaderSession == nullptr)
135 return false;
136
137 m_id = sGuildMgr.getNextGuildId();
138 m_leaderGuid = pLeader->getGuid();
139 m_name = name;
140 m_info = "No message set.";
141 m_motd = "No message set.";
142 m_bankMoney = 0;
143 m_createdDate = ::time(nullptr);
144 m_level = 1;
145 m_experience = 0;
146 m_todayExperience = 0;
147 createLogHolders();
148
149 sLogger.debug("GUILD: creating guild {} for leader {} ({})", name, pLeader->getName(), WoWGuid::getGuidLowPartFromUInt64(m_leaderGuid));
150
151 CharacterDatabase.Execute("DELETE FROM guild_members WHERE guildId = %u", m_id);
152
153 CharacterDatabase.Execute("INSERT INTO guilds (guildId, guildName, leaderGuid, emblemStyle, emblemColor, borderStyle, borderColor, backgroundColor,"
154 "guildInfo, motd, createdate, bankBalance, guildLevel, guildExperience, todayExperience) "
155 "VALUES('%u', '%s', '%u', '%u', '%u', '%u', '%u', '%u', '%s', '%s', '%u', '%u', '%u', '0', '0')",
156 m_id, name.c_str(), m_leaderGuid, m_emblemInfo.getStyle(), m_emblemInfo.getColor(), m_emblemInfo.getBorderStyle(),
157 m_emblemInfo.getBorderColor(), m_emblemInfo.getBackgroundColor(), m_info.c_str(), m_motd.c_str(), uint32_t(m_createdDate),
158 m_bankMoney, m_level);
159
160 createDefaultGuildRanks();
161 bool ret = addMember(m_leaderGuid, GR_GUILDMASTER);
162
163 if (ret)
164 {
165#if VERSION_STRING >= Cata
166 broadcastEvent(GE_FOUNDER, 0, {});
167#endif
168 sHookInterface.OnGuildCreate(pLeader, this);
169 }
170
171 saveGuildToDB();
172
173 return ret;
174}

253void Guild::handleRoster(WorldSession* session)
254{
255#if VERSION_STRING < Cata
256 WorldPacket data(SMSG_GUILD_ROSTER, (4 + m_motd.length() + 1 + m_info.length() + 1 + 4 + _getRanksSize() * (4 + 4 + MAX_GUILD_BANK_TABS * (4 + 4)) + _guildMembersStore.size() * 50));
257 data << uint32_t(_guildMembersStore.size());
258 data << m_motd;
259 data << m_info;
260
261 data << uint32_t(_getRanksSize());
262 for (auto ritr = _guildRankInfoStore.begin(); ritr != _guildRankInfoStore.end(); ++ritr)
263 {
264 data << uint32_t(ritr->getRights());
265 if (ritr->getBankMoneyPerDay() == sizeof(uint32_t))
266 data << uint32_t(sizeof(uint32_t));
267 else
268 data << uint32_t(ritr->getBankMoneyPerDay());
269
270 for (uint8_t i = 0; i < MAX_GUILD_BANK_TABS; ++i)
271 {
272 data << uint32_t(ritr->getBankTabRights(i));
273 data << uint32_t(ritr->getBankTabSlotsPerDay(i));
274 }
275 }
276
277 for (auto itr = _guildMembersStore.begin(); itr != _guildMembersStore.end(); ++itr)
278 {
279 data << uint64_t(itr->second->getGUID())
280 << uint8_t(itr->second->getFlags())
281 << itr->second->getName()
282 << uint32_t(itr->second->getRankId())
283 << uint8_t(itr->second->getLevel())
284 << uint8_t(itr->second->getClass())
285 << uint8_t(0)
286 << uint32_t(itr->second->getZoneId());
287
288 if (!itr->second->getFlags())
289 data << float(float(::time(nullptr) - itr->second->getLogoutTime()) / static_cast<uint64_t>(DAY));
290
291 data << itr->second->getPublicNote();
292
293 if (_hasRankRight(session->GetPlayer()->getGuid(), GR_RIGHT_VIEWOFFNOTE))
294 data << itr->second->getOfficerNote();
295 else
296 data << "";
297 }
298
299 session->SendPacket(&data);
300#else
301 ByteBuffer memberData(100);
302 WorldPacket data(SMSG_GUILD_ROSTER, 100);
303 data.writeBits(m_motd.length(), 11);
304 data.writeBits(_guildMembersStore.size(), 18);
305
306 for (GuildMembersStore::const_iterator itr = _guildMembersStore.begin(); itr != _guildMembersStore.end(); ++itr)
307 {
308 GuildMember* member = itr->second.get();
309 size_t pubNoteLength = member->getPublicNote().length();
310 size_t offNoteLength = member->getOfficerNote().length();
311
312 ObjectGuid guid = member->getGUID();
313 data.writeBit(guid[3]);
314 data.writeBit(guid[4]);
315 data.writeBit(0);
316 data.writeBit(0);
317 data.writeBits(pubNoteLength, 8);
318 data.writeBits(offNoteLength, 8);
319 data.writeBit(guid[0]);
320 data.writeBits(member->getName().length(), 7);
321 data.writeBit(guid[1]);
322 data.writeBit(guid[2]);
323 data.writeBit(guid[6]);
324 data.writeBit(guid[5]);
325 data.writeBit(guid[7]);
326
327 memberData << uint8_t(member->getClass());
328 memberData << uint32_t(member->getTotalReputation());
329 memberData.WriteByteSeq(guid[0]);
330 memberData << uint64_t(member->getWeekActivity());
331 memberData << uint32_t(member->getRankId());
332 memberData << uint32_t(member->getAchievementPoints());
333
334 memberData << uint32_t(0);
335 memberData << uint32_t(0);
336 memberData << uint32_t(0);
337 memberData << uint32_t(0);
338 memberData << uint32_t(0);
339 memberData << uint32_t(0);
340
341 memberData.WriteByteSeq(guid[2]);
342 memberData << uint8_t(member->getFlags());
343 memberData << uint32_t(member->getZoneId());
344 memberData << uint64_t(member->getTotalActivity());
345 memberData.WriteByteSeq(guid[7]);
346 memberData << uint32_t(worldConfig.guild.maxRepPerWeek - member->getWeekReputation());
347
348 if (pubNoteLength)
349 {
350 memberData.WriteString(member->getPublicNote());
351 }
352
353 memberData.WriteByteSeq(guid[3]);
354 memberData << uint8_t(member->getLevel());
355 memberData << int32_t(0);
356 memberData.WriteByteSeq(guid[5]);
357 memberData.WriteByteSeq(guid[4]);
358 memberData << uint8_t(0);
359 memberData.WriteByteSeq(guid[1]);
360 memberData << float(member->isOnline() ? 0.0f : float(::time(nullptr) - member->getLogoutTime()) / static_cast<uint64_t>(DAY));
361
362 if (offNoteLength)
363 {
364 memberData.WriteString(member->getOfficerNote());
365 }
366
367 memberData.WriteByteSeq(guid[6]);
368 memberData.WriteString(member->getName());
369 }
370
371 size_t infoLength = m_info.length();
372 data.writeBits(infoLength, 12);
373
374 data.flushBits();
375 data.append(memberData);
376
377 if (infoLength)
378 {
379 data.WriteString(m_info);
380 }
381
382 data.WriteString(m_motd);
383 data << uint32_t(mAccountsNumber);
384 data << uint32_t(worldConfig.guild.maxRepPerWeek);
385 data.appendPackedTime(m_createdDate);
386 data << uint32_t(0);
387
388 if (session)
389 {
390 sLogger.debugFlag(AscEmu::Logging::LF_OPCODE, "SMSG_GUILD_ROSTER {}", session->GetPlayer()->getName());
391 session->SendPacket(&data);
392 }
393 else
394 {
395 sLogger.debugFlag(AscEmu::Logging::LF_OPCODE, "SMSG_GUILD_ROSTER [Broadcast]");
396 broadcastPacket(&data);
397 }
398#endif

1268{

1270 uint8_t ranks = _getRanksSize();
1271
1272 if (ranks < MIN_GUILD_RANKS || ranks > MAX_GUILD_RANKS)
1273 {
1274 sLogger.failure("Guild {} has invalid number of ranks, creating...", m_id);
1275 broken_ranks = true;
1276 }
1277 else
1278 {
1279 for (uint8_t rankId = 0; rankId < ranks; ++rankId)
1280 {
1281 GuildRankInfo* rankInfo = getRankInfo(rankId);
1282 if (rankInfo->getId() != rankId)
1283 {
1284 sLogger.failure("Guild {} has invalid rank id {}, creating default set of ranks...", m_id, rankId);
1285 broken_ranks = true;
1286 }
1287 else
1288 {
1289 rankInfo->createMissingTabsIfNeeded(_getPurchasedTabsSize(), false, true);
1290 }
1291 }
1292 }
1293
1294 if (broken_ranks)
1295 {
1296 _guildRankInfoStore.clear();
1297 createDefaultGuildRanks();
1298 }
1299
1300 for (auto itr = _guildMembersStore.begin(); itr != _guildMembersStore.end(); ++itr)
