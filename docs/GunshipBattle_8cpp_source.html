406void GunshipAI::DamageTaken(Unit* /*_attacker*/, uint32_t* damage)
407{
408 if (*damage >= getCreature()->getHealth())
409 {
410 OnDied(nullptr);
411 *damage = getCreature()->getHealth() - 1;
412 return;
413 }
414
415 if (_summonedFirstMage)
416 return;
417
418 if (getCreature()->GetTransport()->getEntry() != uint32_t(_teamInInstance == TEAM_HORDE ? GO_THE_SKYBREAKER_HORDE_ICC : GO_ORGRIM_S_HAMMER_ALLIANCE_ICC))
419 return;
420
421 if (getCreature()->getHealthPct() > 90)
422 return;
423
424 _summonedFirstMage = true;
425 if (Creature* captain = mInstance->getLocalCreatureData(_teamInInstance == TEAM_HORDE ? DATA_GB_MURADIN_BRONZEBEARD : DATA_GB_HIGH_OVERLORD_SAURFANG))
426 captain->GetScript()->DoAction(ACTION_SPAWN_MAGE);
427}

429void GunshipAI::OnDied(Unit* /*pTarget*/)
430{
431 if (_died)
432 return;
433
434 _died = true;
435
436 // Victory??
437 bool isVictory = getCreature()->GetTransport()->getEntry() == GO_THE_SKYBREAKER_HORDE_ICC || getCreature()->GetTransport()->getEntry() == GO_ORGRIM_S_HAMMER_ALLIANCE_ICC;
438 mInstance->setBossState(DATA_ICECROWN_GUNSHIP_BATTLE, isVictory ? Performed : Failed);
439
440 // Disangege
441 if (Creature* creature = mInstance->getLocalCreatureData(getCreature()->getEntry() == NPC_GB_ORGRIMS_HAMMER ? DATA_SKYBREAKER_BOSS : DATA_ORGRIMMAR_HAMMER_BOSS))
442 {
443 mInstance->sendUnitEncounter(EncounterFrameDisengaged, creature);
444 }
445
446 mInstance->sendUnitEncounter(EncounterFrameDisengaged, getCreature());
447
448 // Stopp Playing Music
449 mInstance->setZoneMusic(4812, 0);
450
451 std::list<Creature*> creatures;
452 // Eject All Passengers
453 GetCreatureListWithEntryInGrid(creatures, _teamInInstance == TEAM_HORDE ? NPC_GB_HORDE_CANON : NPC_GB_ALLIANCE_CANON, 900.0f);
454 for (auto itr = creatures.begin(); itr != creatures.end(); ++itr)
455 {
456 Creature* cannon = *itr;
457 if (isVictory)
458 cannon->GetScript()->DoAction(ACTION_BATTLE_DONE);
459 else
460 cannon->GetScript()->DoAction(ACTION_FAIL);
461 }
462 creatures.clear();
463 // Explosion
464 GetCreatureListWithEntryInGrid(creatures, NPC_GB_GUNSHIP_HULL, 900.0f);
465 for (auto itr = creatures.begin(); itr != creatures.end(); ++itr)
466 {
467 Creature* hull = *itr;
468 if (hull->GetTransport() != getCreature()->GetTransport())
469 continue;
470
471 if (isVictory)
472 hull->GetScript()->DoAction(ACTION_BATTLE_DONE);
473 else
474 hull->GetScript()->DoAction(ACTION_FAIL);
475 }
476 creatures.clear();
477
478 if (isVictory)
479 mInstance->DoAction(ACTION_BATTLE_DONE);
480 else
481 mInstance->DoAction(ACTION_FAIL);
482}
