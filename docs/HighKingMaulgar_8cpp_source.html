157void KroshFirehandAI::AIUpdate(unsigned long time_passed)
158{
159 if (_isCasting())
160 return;
161
162 scriptEvents.updateEvents(time_passed, getScriptPhase());
163
164 while (uint32_t eventId = scriptEvents.getFinishedEvent())
165 {
166 switch (eventId)
167 {
168 case EVENT_BLASTWAVE:
169 {
170 Unit* unit = getBestUnitTarget(TargetFilter_Closest);
171 if (unit && getRangeToObject(unit) < 15.0f)
172 _castAISpell(mBlastWave);
173
174 scriptEvents.addEvent(EVENT_BLASTWAVE, 20000);
175 }
176 break;
177 default:
178 break;
179 }
180 }
181}
