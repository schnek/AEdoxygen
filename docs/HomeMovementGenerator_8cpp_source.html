38template<>
39void HomeMovementGenerator<Creature>::setTargetLocation(Creature* owner)
40{
41 // if we are ROOT/STUNNED/DISTRACTED even after aura clear, finalize on next update - otherwise we would get stuck in evade
42 if (owner->hasUnitStateFlag(UNIT_STATE_ROOTED | UNIT_STATE_STUNNED | UNIT_STATE_DISTRACTED))
43 {
44 addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
45 return;
46 }
47
48 owner->removeUnitStateFlag(UNIT_STATE_ALL_ERASABLE & ~UNIT_STATE_EVADING);
49 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
50
51 LocationVector destination = owner->GetSpawnPosition();
52 MovementMgr::MoveSplineInit init(owner);
53
54 owner->updateAllowedPositionZ(destination.x, destination.y, destination.z);
55 init.MoveTo(positionToVector3(destination));
56 init.SetFacing(destination.getOrientation());
57 init.SetWalk(false);
58 init.Launch();
59}
