62bool MastersCall(uint8_t effectIndex, Spell* pSpell)
63{
64 Player* caster = pSpell->getPlayerCaster();
65
66 if (caster == NULL)
67 return true;
68
69 Pet* Summon = caster->getPet();
70 if (Summon == NULL || Summon->isDead())
71 return true;
72
73 switch (effectIndex)
74 {
75 case 0:
76 Summon->castSpell(caster, pSpell->damage, true);
77 return true;
78 case 1:
79 Summon->castSpell(Summon, 62305, true);
80 return true;
81 }
82
83 return true;
84}

86bool TheBeastWithin(uint8_t /*effectIndex*/, Aura* a, bool apply)
87{
88 Unit* m_target = a->getOwner();
89
90 const auto mechanics = sSpellMgr.getCrowdControlMechanicList(false);
91 for (int x = 0; mechanics[x] != MECHANIC_NONE; ++x)
92 {
93 if (apply)
94 m_target->m_mechanicsDispels[mechanics[x]]++;
95 else
96 m_target->m_mechanicsDispels[mechanics[x]]--;
97 }
98
99 if (apply)
100 m_target->removeAllAurasBySpellMechanic(mechanics);
101
102 return true;
