77void IchronAI::DoAction(int32_t actionId)
78{
79 switch (actionId)
80 {
81 case Ichron::ACTION_WATER_GLOBULE_HIT:
82 {
83 if (!getCreature()->isAlive())
84 break;
85
86 getCreature()->modHealth(int32_t(getCreature()->getPctFromMaxHealth(3)));
87 mDehydration = false;
88 } break;
89 case Ichron::ACTION_PROTECTIVE_BUBBLE_SHATTERED:
90 {
91 sendDBChatMessage(Ichron::SAY_SHATTER);
92
93 getCreature()->castSpell(nullptr, Ichron::SPELL_SPLATTER, true);
94 getCreature()->castSpell(nullptr, Ichron::SPELL_BURST, true);
95 getCreature()->castSpell(getCreature(), Ichron::SPELL_DRAINED, true);
96
97 uint32_t damage = getCreature()->getPctFromMaxHealth(30);
98 getCreature()->modHealth(-std::min<int32_t>(damage, getCreature()->getHealth() - 1));
99
100 for (auto spell : mCreatureAISpells)
101 {
102 spell->setCooldownTimer(spell->mCooldownTimer->getExpireTime() + 15 * TimeVarsMs::Second);
103 }
104 break;
105 }
106 case Ichron::ACTION_DRAINED:
107 {
108 if (getCreature()->getHealthPct() > 30)
109 {
110 sendDBChatMessage(Ichron::SAY_BUBBLE);
111 getCreature()->castSpell(getCreature(), Ichron::SPELL_PROTECTIVE_BUBBLE, true);
112 }
113 } break;
114 default:
115 break;
116 }
117}

205{
206 if (spell->getUnitTarget() == nullptr)

208

210 {
211 if (Aura* aura = target->getAuraWithId(Ichron::SPELL_SHRINK))
212 aura->refreshOrModifyStack(false, -1);

214 if (target->GetScript())

216 }

222/// Spell: 54306 - Protective Bubble
223SpellScriptCheckDummy IchronBubble::onAuraDummyEffect(Aura* aur, AuraEffectModifier* /*aurEff*/, bool /*apply*/)
224{
225 if (aur->getCharges() <= 1)
226 {
227 for (AreaAuraList::iterator itr = aur->targets.begin(); itr != aur->targets.end(); ++itr)
228 {
229 auto unit = aur->getOwner()->getWorldMap()->getUnit(*itr);
230 if (unit == nullptr || !unit->ToCreature())
231 return SpellScriptCheckDummy::DUMMY_OK;
232
233 if (CreatureAIScript* targetAI = unit->ToCreature()->GetScript())
234 targetAI->DoAction(Ichron::ACTION_PROTECTIVE_BUBBLE_SHATTERED);
235 }
236 }
237
238 return SpellScriptCheckDummy::DUMMY_OK;
