75bool RotateMovementGenerator::update(Unit* owner, uint32_t diff)
76{
77 if (!owner)
78 return false;
79
80 float angle = owner->GetOrientation();
81 if (_direction == ROTATE_DIRECTION_LEFT)
82 {
83 angle += float(diff) * float(M_PI) * 2.f / float(_maxDuration);
84 while (angle >= float(M_PI) * 2.f)
85 angle -= float(M_PI) * 2.f;
86 }
87 else
88 {
89 angle -= float(diff) * float(M_PI) * 2.f / float(_maxDuration);
90 while (angle < 0.f)
91 angle += float(M_PI) * 2.f;
92 }
93
94 MovementMgr::MoveSplineInit init(owner);
95 init.MoveTo(positionToVector3(owner->GetPosition()), false);
96 if (owner->hasUnitMovementFlag(MOVEFLAG_TRANSPORT) && owner->getTransGuid())
97 init.DisableTransportPathTransformations();
98 init.SetFacing(angle);
99 init.Launch();
100
101 if (_duration > diff)
102 _duration -= diff;
103 else
104 {
105 addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
106 return false;
107 }
108
109 return true;
110}
111#else
112bool RotateMovementGenerator::update(Unit* owner, uint32_t diff)
113{
114 float angle = owner->GetOrientation();
115 angle += (float(diff) * static_cast<float>(M_PI * 2) / _maxDuration) * (_direction == ROTATE_DIRECTION_LEFT ? 1.0f : -1.0f);
116 angle = G3D::wrap(angle, 0.0f, float(G3D::twoPi()));
117
118 owner->SetOrientation(angle); // UpdateSplinePosition does not set orientation with UNIT_STATE_ROTATING
119 owner->setFacingTo(angle); // Send spline movement to clients
120
121 if (_duration > diff)
122 _duration -= diff;
123 else
124 return false;
125
126 return true;
127}
128#endif

173 init.Launch();
