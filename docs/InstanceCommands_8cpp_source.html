23bool ChatCommandHandler::HandleCreateInstanceCommand(const char* args, WorldSession* m_session)
24{
25 Player* plr = GetSelectedPlayer(m_session, true, true);
26 if (plr == nullptr)
27 return true;
28
29 float x, y, z;
30 uint32_t mapid;
31
32 if (sscanf(args, "%u %f %f %f", &mapid, &x, &y, &z) != 4)
33 return false;
34
35 // Create Map Manager
36 WorldMap* mgr = sMapMgr.createInstanceForPlayer(mapid, m_session->GetPlayer());
37 if (mgr == nullptr)
38 {
39 sLogger.failure("call failed for map {}", mapid);
40 return false;
41 }
42 sLogger.info("CreateInstanceGMCommand : GM created instance for map {}", mapid);
43
44 LocationVector vec(x, y, z);
45 m_session->GetPlayer()->safeTeleport(mgr, vec);
46
47 return true;
48}

164//.instance reset
165bool ChatCommandHandler::HandleResetInstanceCommand(const char* args, WorldSession* m_session)
166{
167 uint32_t instanceId;
168 int argc = 1;
169 char* playername = nullptr;
170 char* guidString = (char*)args;
171
172 // Parse arguments
173 char* space = (char*)strchr(args, ' ');
174 if (space)
175 {
176 *space = '\0';
177 playername = space + 1;
178 argc = 2;
179 }
180
181 instanceId = atoi(guidString);
182 if (!instanceId)
183 {
184 redSystemMessage(m_session, "You must specify an instance id.");
185 return true;
186 }
187
188
189 Player* plr;
190
191 if (argc == 1)
192 plr = GetSelectedPlayer(m_session, true, true);
193 else
194 plr = sObjectMgr.getPlayer(playername, false);
195
196 if (!plr)
197 {
198 redSystemMessage(m_session, "Player not found");
199 return true;
200 }
201
202 InstanceMap* instance = sMapMgr.findInstanceMap(instanceId);
203 if (instance == nullptr)
204 {
205 redSystemMessage(m_session, "There's no instance with id {}.", instanceId);
206 return true;
207 }
208
209 if (instance && instance->hasPlayers())
210 {
211 redSystemMessage(m_session, "Failed to reset non-persistent instance with id {}, due to player still inside.", instanceId);
212 return true;
213 }
214
215 instance->reset(INSTANCE_RESET_GLOBAL);
216
217 // tell player the instance was reset
218 plr->getSession()->SendPacket(SmsgInstanceReset(instance->getBaseMap()->getMapId()).serialise().get());
219
220 // redSystemMessage(m_session, "Resetting single non-persistent instances is not available yet.");
221 sGMLog.writefromsession(m_session, "used reset instance command on %s, instance %u,", plr->getName().c_str(), instanceId);
222 return true;
223}

251 player->resetInstances(INSTANCE_RESET_ALL, false);

253

258//.instance shutdown
259bool ChatCommandHandler::HandleShutdownInstanceCommand(const char* args, WorldSession* m_session)
260{
261 uint32_t instanceId = (args ? atoi(args) : 0);
262 if (instanceId == 0)
263 return false;
264
265 InstanceMap* instance = sMapMgr.findInstanceMap(instanceId);
266 if (instance == nullptr)
267 {
268 redSystemMessage(m_session, "There's no instance with id {}.", instanceId);
269 return true;
270 }
271
272 if (instance->isUnloadPending())
273 {
274 redSystemMessage(m_session, "Instance with id {} already shut down.", instanceId);
275 return true;
276 }
277
278 systemMessage(m_session, "Attempting to shutdown instance with id {}...", instanceId);
279
280 // Remove all Players
281 instance->removeAllPlayers();
282 instance->setUnloadPending(true);
283
284 systemMessage(m_session, "...done");
285
286 sGMLog.writefromsession(m_session, "used shutdown instance command on instance %u,", instanceId);
287 return true;
288}
