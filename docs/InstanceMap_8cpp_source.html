70void InstanceMap::permBindAllPlayers()
71{
72 if (!getBaseMap()->isInstanceMap())
73 return;
74
75 InstanceSaved* save = sInstanceMgr.getInstanceSave(getInstanceId());
76 if (!save)
77 {
78 sLogger.failure("Cannot bind players to instance map (Name: {}, Entry: {}, Difficulty: {}, ID: {}) because no instance save is available!", getBaseMap()->getMapName(), getBaseMap()->getMapId(), getDifficulty(), getInstanceId());
79 return;
80 }
81
82 // perm bind all players that are currently inside the instance
83 for (const auto& itr : getPlayers())
84 {
85 Player* player = itr.second;
86 // never instance bind GMs with GM mode enabled
87 if (player->isGMFlagSet())
88 continue;
89
90 InstancePlayerBind* bind = player->getBoundInstance(save->getMapId(), save->getDifficulty());
91 if (bind && bind->perm)
92 {
93 if (bind->save && bind->save->getInstanceId() != save->getInstanceId())
94 {
95 sLogger.failure("Player ({}, Name: {}) is in instance map (Name: {}, Entry: {}, Difficulty: {}, ID: {}) that is being bound, but already has a save for the map on ID {}!", player->getGuidLow(), player->getName(), getBaseMap()->getMapName() , save->getMapId(), save->getDifficulty(), save->getInstanceId(), bind->save->getInstanceId());
96 }
97 else if (!bind->save)
98 {
99 sLogger.failure("Player ({}, Name: {}) is in instance map (Name: {}, Entry: {}, Difficulty: {}, ID: {}) that is being bound, but already has a bind (without associated save) for the map!", player->getGuidLow(), player->getName(), getBaseMap()->getMapName(), save->getMapId(), save->getDifficulty(), save->getInstanceId());
100 }
101 }
102 else
103 {
104 player->bindToInstance(save, true);
105 WorldPacket data(SMSG_INSTANCE_SAVE_CREATED, 4);
106 data << uint32_t(0);
107 player->sendPacket(&data);
108#if VERSION_STRING > TBC
109 player->getSession()->sendCalendarRaidLockout(save, true);
110#endif
111
112 // if group leader is in instance, group also gets bound
113 if (const auto group = player->getGroup())
114 if (group->GetLeader()->guid == player->getGuidLow())
115 group->bindToInstance(save, true);
116 }
117 }
118}

120bool InstanceMap::addPlayerToMap(Player* player)
121{
122 // disable unload Timer
123 m_unloadTimer = 0;
124
125 if (getBaseMap()->isInstanceMap())
126 {
127 const auto group = player->getGroup();
128
129 // increase Instance Hourly Limit
130 if (!group || !group->isLFGGroup())
131 player->addInstanceEnterTime(getInstanceId(), Util::getTimeNow());
132
133 // get or create an instance save for the map
134 InstanceSaved* mapSave = sInstanceMgr.getInstanceSave(getInstanceId());
135 if (!mapSave)
136 {
137 sLogger.debug("creating instance save for map {} spawnmode {} with instance id {}", getBaseMap()->getMapId(), getSpawnMode(), getInstanceId());
138 mapSave = sInstanceMgr.addInstanceSave(getBaseMap()->getMapId(), getInstanceId(), InstanceDifficulty::Difficulties(getSpawnMode()), 0, true);
139 }
140
141 if (mapSave)
142 {
143 // check for existing instance binds
144 InstancePlayerBind* playerBind = player->getBoundInstance(getBaseMap()->getMapId(), InstanceDifficulty::Difficulties(getSpawnMode()));
145 if (playerBind && playerBind->perm)
146 {
147 // cannot enter other instances if bound permanently
148 if (playerBind->save != mapSave)
149 {
150 sLogger.debug("player % s % s is permanently bound to instance % s % d, % d, % d, % d, % d, % d but he is being put into instance % s % d, % d, % d, % d, % d, % d", player->getName(), player->getGuid(), getBaseMap()->getMapName(), playerBind->save->getMapId(), playerBind->save->getInstanceId(), playerBind->save->getDifficulty(), playerBind->save->getPlayerCount(), playerBind->save->getGroupCount(), playerBind->save->canReset(), getBaseMap()->getMapName(), mapSave->getMapId(), mapSave->getInstanceId(), mapSave->getDifficulty(), mapSave->getPlayerCount(), mapSave->getGroupCount(), mapSave->canReset());
151 return false;
152 }
153 }
154 else
155 {
156 if (group)
157 {
158 // solo saves should have been reset when the map was loaded
159 InstanceGroupBind* groupBind = group->getBoundInstance(getBaseMap());
160 if (playerBind && playerBind->save != mapSave)
161 {
162 sLogger.debug("player {} {} is being put into instance {} {}, {}, {}, {}, {}, {} but he is in group {} and is bound to instance {}, {}, {}, {}, {}, {}!", player->getName(), player->getGuid(), getBaseMap()->getMapName(), mapSave->getMapId(), mapSave->getInstanceId(), mapSave->getDifficulty(), mapSave->getPlayerCount(), mapSave->getGroupCount(), mapSave->canReset(), group->GetLeader()->guid, playerBind->save->getMapId(), playerBind->save->getInstanceId(), playerBind->save->getDifficulty(), playerBind->save->getPlayerCount(), playerBind->save->getGroupCount(), playerBind->save->canReset());
163 if (groupBind)
164 sLogger.debug("the group is bound to the instance {} {}, {}, {}, {}, {}, {}", getBaseMap()->getMapName(), groupBind->save->getMapId(), groupBind->save->getInstanceId(), groupBind->save->getDifficulty(), groupBind->save->getPlayerCount(), groupBind->save->getGroupCount(), groupBind->save->canReset());
165 return false;
166 }
167 // bind to the group or keep using the group save
168 if (!groupBind)
169 {
170 group->bindToInstance(mapSave, false);
171 }
172 else
173 {
174 // cannot jump to a different instance without resetting it
175 if (groupBind->save != mapSave)
176 {
177 sLogger.debug("player {} {} is being put into instance {}, {}, {} but he is in group {} which is bound to instance {}, {}, {}! \n", player->getName(), player->getGuid(), mapSave->getMapId(), mapSave->getInstanceId(), mapSave->getDifficulty(), group->GetLeader()->guid, groupBind->save->getMapId(), groupBind->save->getInstanceId(), groupBind->save->getDifficulty());
178 sLogger.debug("MapSave players: {}, group count: {} \n", mapSave->getPlayerCount(), mapSave->getGroupCount());
179 if (groupBind->save)
180 sLogger.debug("GroupBind save players: {}, group count: {}", groupBind->save->getPlayerCount(), groupBind->save->getGroupCount());
181 else
182 sLogger.debug("GroupBind save NULL");
183 return false;
184 }
185 // if the group/leader is permanently bound to the instance
186 // players also become permanently bound when they enter
187 if (groupBind->perm)
188 {
189 WorldPacket data(SMSG_INSTANCE_LOCK_WARNING_QUERY, 9);
190 data << uint32_t(60000);
191 data << uint32_t(getScript() ? getScript()->getCompletedEncounterMask() : 0);
192 data << uint8_t(0);
193 player->sendPacket(&data);
194 player->setPendingBind(mapSave->getInstanceId(), 60000);
195 }
196 }
197 }
198 else
199 {
200 // set up a solo bind or continue using it
201 if (!playerBind)
202 player->bindToInstance(mapSave, false);
203 else
204 // cannot jump to a different instance without resetting it
205 ASSERT(playerBind->save == mapSave);
206 }
207 }
208 }
209
210 // for normal instances cancel the reset schedule when the
211 // first player enters (no players yet)
212 setResetSchedule(false);
213
214 sLogger.info("Player '{}' entered instance '{}' of map '{}' \n", player->getName(), getInstanceId(), getBaseMap()->getMapName());
215 }
216
217 return true;
218}

248EnterState InstanceMap::cannotEnter(Player* player)
249{
250 if (isUnloadPending())
251 return CANNOT_ENTER_INSTANCE_BIND_MISMATCH;
252
253 if (player->getWorldMap() == this)
254 return CANNOT_ENTER_ALREADY_IN_MAP;
255
256 // allow GM's to enter
257 if (player->isGMFlagSet())
258 return WorldMap::cannotEnter(player);
259
260 // cannot enter if the instance is full (player cap)
261 uint32_t maxPlayers = getMaxPlayers();
262 if (getPlayerCount() >= maxPlayers)
263 return CANNOT_ENTER_MAX_PLAYERS;
264
265 // cannot enter while an encounter is in progress (unless this is a relog, in which case it is permitted)
266 if (!player->IsInWorld() && getBaseMap()->isRaid() && isCombatInProgress())
267 return CANNOT_ENTER_ENCOUNTER;
268
269 // cannot enter if player is permanent saved to a different instance id
270 if (InstancePlayerBind* playerBind = player->getBoundInstance(getBaseMap()->getMapId(), getDifficulty()))
271 if (playerBind->perm && playerBind->save)
272 if (playerBind->save->getInstanceId() != getInstanceId())
273 return CANNOT_ENTER_INSTANCE_BIND_MISMATCH;
274
275 return WorldMap::cannotEnter(player);
276}

278void InstanceMap::createInstanceData(bool load)
279{
280 if (getScript() == nullptr)
281 return;
282
283 auto const& encounters = sObjectMgr.getDungeonEncounterList(getBaseMap()->getMapId(), getDifficulty());
284 if (!encounters)
285 return;
286
287 uint32_t bossNumber = static_cast<uint32_t>(encounters->size());
288
289 if (!getScript()->getEncounterCount())
290 getScript()->setBossNumber(bossNumber);
291
292 if (load)
293 {
294 auto result = CharacterDatabase.Query("SELECT data, completedEncounters FROM instance WHERE map = %u AND id = %u", getBaseMap()->getMapId(), getInstanceId());
295
296 if (result)
297 {
298 Field* fields = result->Fetch();
299 std::string data = fields[0].asCString();
300
301 getScript()->setCompletedEncountersMask(fields[1].asUint32());
302 if (!data.empty())
303 {
304 sLogger.debug("Loading instance data for `{}` with id {}", getBaseMap()->getMapName(), getInstanceId());
305 getScript()->loadSavedInstanceData(data.c_str());
306 }
307 }
308 }
309 else
310 {
311 getScript()->generateBossDataState();
312 }

315bool InstanceMap::reset(uint8_t method)
316{
317 if (getPlayerCount())
318 {
319 if (method == INSTANCE_RESET_ALL || method == INSTANCE_RESET_CHANGE_DIFFICULTY)
320 {
321 // notify the players to leave the instance so it can be reset
322 for (const auto& itr : getPlayers())
323 itr.second->sendResetFailedNotify(getBaseMap()->getMapId());
324 }
325 else
326 {
327 bool doUnload = true;
328 if (method == INSTANCE_RESET_GLOBAL)
329 {
330 // set the homebind timer for players inside (1 minute)
331 for (const auto& itr : getPlayers())
332 {
333 InstancePlayerBind* bind = itr.second->getBoundInstance(getBaseMap()->getMapId(), getDifficulty());
334 if (bind && bind->extendState && bind->save->getInstanceId() == getInstanceId())
335 doUnload = false;
336 }
337
338 if (doUnload && hasPermBoundPlayers()) // check if any unloaded players have a nonexpired save to this
339 doUnload = false;
340 }
341
342 if (doUnload)
343 {
344 // the unload timer is not started
345 // instead the map will unload immediately after the players have left
346 m_unloadWhenEmpty = true;
347 m_resetAfterUnload = true;
348 }
349 }
350 }
351 else
352 {
353 // unloaded at next update
354 m_unloadTimer = 1;
355 m_resetAfterUnload = !(method == INSTANCE_RESET_GLOBAL && hasPermBoundPlayers());
356 }
357
358 return (getPlayerCount() == 0);
359}
