166void InstanceScript::generateBossDataState()
167{
168 const auto* encounters = sObjectMgr.getDungeonEncounterList(getWorldMap()->getBaseMap()->getMapId(), getWorldMap()->getDifficulty());
169 uint32_t i = 0;
170
171 for (DungeonEncounterList::const_iterator itr = encounters->begin(); itr != encounters->end(); ++itr, ++i)
172 {
173 const auto encounter = itr->get();
174
175 BossInfo* bossInfo = &bosses[i];
176 bossInfo->entry = encounter->creditEntry;
177 bossInfo->state = NotStarted;
178 }
179
180 // Set States
181 for (i = 0; i < bosses.size(); ++i)
182 setBossState(i, NotStarted);
183}

185bool InstanceScript::setBossState(uint32_t id, EncounterStates state)
186{
187 if (id < bosses.size())
188 {
189 BossInfo* bossInfo = &bosses[id];
190 if (bossInfo->state == InvalidState) // loading
191 {
192 bossInfo->state = state;
193 return false;
194 }
195 else
196 {
197 if (bossInfo->state == state)
198 return false;
199
200 if (bossInfo->state == Performed)
201 {
202 return false;
203 }
204
205 bossInfo->state = state;
206 saveToDB();
207 }
208
209 OnEncounterStateChange(id, state);
210
211 if (state == NotStarted)
212 getInstance()->respawnBossLinkedGroups(bossInfo->entry);
213
214 return true;
215 }
216 return false;
217}

249void InstanceScript::readSaveDataBossStates(std::istringstream& data)
250{
251 const auto* encounters = sObjectMgr.getDungeonEncounterList(getWorldMap()->getBaseMap()->getMapId(), getWorldMap()->getDifficulty());
252 size_t i = 0;
253
254 for (DungeonEncounterList::const_iterator itr = encounters->begin(); itr != encounters->end(); ++itr, ++i)
255 {
256 const auto encounter = itr->get();
257
258 BossInfo* bossInfo = &bosses[i];
259 bossInfo->entry = encounter->creditEntry;
260 }
261
262 uint32_t bossId = 0;
263 for (std::vector<BossInfo>::iterator itr = bosses.begin(); itr != bosses.end(); ++itr, ++bossId)
264 {
265 uint32_t buff;
266 data >> buff;
267 if (buff == InProgress || buff == Failed || buff == PreProgress)
268 buff = NotStarted;
269
270 if (buff < InvalidState)
271 {
272 setBossState(bossId, EncounterStates(buff));
273 }
274 }
275}

294void InstanceScript::updateEncounterState(EncounterCreditType type, uint32_t creditEntry)
295{
296 DungeonEncounterList const* encounters = sObjectMgr.getDungeonEncounterList(mInstance->getBaseMap()->getMapId(), mInstance->getDifficulty());
297 if (!encounters)
298 return;
299
300 uint32_t dungeonId = 0;
301
302 for (auto const& encounter : *encounters)
303 {
304 if (encounter->creditType == type && encounter->creditEntry == creditEntry)
305 {
306 completedEncounters |= 1 << encounter->dbcEntry->encounterIndex;
307 if (encounter->lastEncounterDungeon)
308 {
309 dungeonId = encounter->lastEncounterDungeon;
310 break;
311 }
312 }
313 }
314
315 if (dungeonId)
316 {
317 for (auto const& playersPair : mInstance->getPlayers())
318 {
319 if (Player* player = playersPair.second)
320 {
321 if (const auto group = player->getGroup())
322 {
323 if (group->isLFGGroup())
324 {
325 sLfgMgr.RewardDungeonDoneFor(dungeonId, player);
326 return;
327 }
328 }
329 }
330 }
331 }

515void InstanceScript::setCellForcedStates(float xMin, float xMax, float yMin, float yMax, bool forceActive /*true*/)
516{
517 if (xMin == xMax || yMin == yMax)
518 return;
519
520 float Y = yMin;
521 while (xMin < xMax)
522 {
523 while (yMin < yMax)
524 {
525 MapCell* CurrentCell = mInstance->getCellByCoords(xMin, yMin);
526 if (forceActive && CurrentCell == nullptr)
527 {
528 CurrentCell = mInstance->createByCoords(xMin, yMin);
529 if (CurrentCell != nullptr)
530 CurrentCell->init(mInstance->getPosX(xMin), mInstance->getPosY(yMin), mInstance);
531 }
532
533 if (CurrentCell != nullptr)
534 {
535 if (forceActive)
536 mInstance->addForcedCell(CurrentCell);
537 else
538 mInstance->removeForcedCell(CurrentCell);
539 }
540
541 yMin += 40.0f;
542 }
543
544 yMin = Y;
545 xMin += 40.0f;
546 }
547}

646GameObject* InstanceScript::getClosestGameObjectForPosition(uint32_t entry, float posX, float posY, float posZ)
647{
648 GameObjectSet gameObjectSet = getGameObjectsSetForEntry(entry);
649
650 if (gameObjectSet.size() == 0)
651 return nullptr;
652
653 if (gameObjectSet.size() == 1)
654 return *(gameObjectSet.begin());
655
656 float distance = 99999;
657 float nearestDistance = 99999;
658
659 for (auto gameobject : gameObjectSet)
660 {
661 if (gameobject != nullptr)
662 {
663 distance = getRangeToObjectForPosition(gameobject, posX, posY, posZ);
664 if (distance < nearestDistance)
665 {
666 nearestDistance = distance;
667 return gameobject;
668 }
669 }
670 }
671
672 return nullptr;
673}

690float InstanceScript::getRangeToObjectForPosition(Object* object, float posX, float posY, float posZ)
691{
692 if (object == nullptr)
693 return 0.0f;
694
695 LocationVector pos = object->GetPosition();
696 float dX = pos.x - posX;
697 float dY = pos.y - posY;
698 float dZ = pos.z - posZ;
699
700 return sqrtf(dX * dX + dY * dY + dZ * dZ);
701}
