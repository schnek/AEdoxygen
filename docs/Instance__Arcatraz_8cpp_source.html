89 void VoidZoneArc()
90 {
91 _resetTimer(VoidTimer, (Util::getRandomUInt(10) + 30) * 1000);
92
93 std::vector<Player*> TargetTable;
94 for (const auto& itr : getCreature()->getInRangePlayersSet())
95 {
96 if (!itr || !itr->isPlayer())
97 continue;
98
99 Player* RandomTarget = static_cast<Player*>(itr);
100 if (RandomTarget->isAlive() && itr->isHostileTo(getCreature()))
101 TargetTable.push_back(RandomTarget);
102 }
103
104 if (!TargetTable.size())
105 return;
106
107 auto random_index = Util::getRandomUInt(0, uint32_t(TargetTable.size() - 1));
108 auto random_target = TargetTable[random_index];
109
110 if (random_target == nullptr)
111 return;
112
113 float vzX = Util::getRandomUInt(5) * cos(Util::getRandomFloat(6.28f)) + random_target->GetPositionX();
114 float vzY = Util::getRandomUInt(5) * cos(Util::getRandomFloat(6.28f)) + random_target->GetPositionY();
115 float vzZ = random_target->GetPositionZ();
116
117 Creature* VoidZone = spawnCreature(CN_VOIDZONEARC, vzX, vzY, vzZ, 0.0f);
118 VoidZone->addUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
119 VoidZone->m_noRespawn = true;
120 if (!VoidZone->IsInWorld())
121 {
122 VoidZone->Despawn(0, 0);
123 return;
124 }
125 random_target = NULL;
126 VoidZone->Despawn(60000, 0);
127 }
128
129 void AIUpdate() override
130 {
131 if (_isTimerFinished(SpeechTimer))
132 Speech();
133
134 if (_isTimerFinished(VoidTimer))
135 VoidZoneArc();
136 }
