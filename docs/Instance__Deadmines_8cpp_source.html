139class MrSmiteAI : public CreatureAIScript
140{
141public:
142 static CreatureAIScript* Create(Creature* c) { return new MrSmiteAI(c); }
143 explicit MrSmiteAI(Creature* pCreature) : CreatureAIScript(pCreature)
144 {
145 auto smiteSlam = addAISpell(Deadmines::SMITE_SLAM, 25.0f, TARGET_ATTACKING, 0, 15, false, true);
146 smiteSlam->setMinMaxDistance(0.0f, 8.0f);
147
148 mStomp = addAISpell(Deadmines::SMITE_STOMP, 0.0f, TARGET_SELF);
149 mWaitAtChest = 0;
150 _setWieldWeapon(true);
151 }
152
153 void OnCombatStop(Unit* /*pTarget*/) override
154 {
155 if (isScriptPhase(4))
156 _removeAura(Deadmines::SMITES_HAMMER);
157
158 if (!isAlive())
159 _setWieldWeapon(false);
160
161 setScriptPhase(1);
162 SwitchWeapons();
163 _removeTimer(mWaitAtChest);
164 }
165
166 void AIUpdate() override
167 {
168 if (_getHealthPercent() <= 66 && isScriptPhase(1))
169 setScriptPhase(2);
170 else if (_getHealthPercent() <= 33 && isScriptPhase(3))
171 setScriptPhase(4);
172
173 if (isScriptPhase(2) || isScriptPhase(4))
174 {
175 if (NearChest())
176 SwitchWeapons();
177 else
178 MoveToChest();
179 }
180
181 if (_isTimerFinished(mWaitAtChest))
182 MoveToPlayer();
183 }
184
185 void OnScriptPhaseChange(uint32_t phaseId) override
186 {
187 switch (phaseId)
188 {
189 case 2:
190 sendChatMessage(CHAT_MSG_MONSTER_YELL, 5778, "You landlubbers are tougher than I thought. I'll have to improvise!");
191 _castAISpell(mStomp);
192 break;
193 case 4:
194 sendChatMessage(CHAT_MSG_MONSTER_YELL, 5779, "D'ah! Now you're making me angry!");
195 _castAISpell(mStomp);
196 break;
197 default:
198 break;
199 }
200 }
201
202 void MoveToChest()
203 {
204 if (canEnterCombat())
205 getCreature()->getAIInterface()->setAllowedToEnterCombat(false);
206
207 stopMovement();
208 moveTo(1.100060f, -780.026367f, 9.811194f);
209 }
210
211 void MoveToPlayer()
212 {
213 getCreature()->getAIInterface()->setAllowedToEnterCombat(true);
214 }
215
216 bool NearChest()
217 {
218 if (getCreature()->GetPositionX() == 1.100060f && getCreature()->GetPositionY() == -780.026367f)
219 return true;
220
221 float XDiff = getCreature()->GetPositionX() - 1.100060f;
222 float YDiff = getCreature()->GetPositionY() + 780.026367f;
223 float Distance = std::sqrt(XDiff * XDiff + YDiff * YDiff);
224 if (Distance <= 5.0f)
225 return true;
226
227 return false;
228 }
229
230 void SwitchWeapons()
231 {
232 // CREDITS to Skyboat on ascentemu.com/forums he had some of this info on one of his releases
233 switch (getScriptPhase())
234 {
235 case 1: // Phase 1 (Default)
236 _setDisplayWeaponIds(5192, 0);
237 getCreature()->setBaseAttackTime(MELEE, getCreature()->getBaseAttackTime(MELEE)); // 1483 is taken from NCDB creature_proto
238 break;
239 case 2: // Phase 2
240 _setDisplayWeaponIds(5196, 5196);
241 getCreature()->setBaseAttackTime(MELEE, getCreature()->getBaseAttackTime(MELEE) / 2);
242 break;
243 case 4: // Phase 4
244 // Is base attack time change needed if we use aura ?
245 _setDisplayWeaponIds(7230, 0);
246 getCreature()->setBaseAttackTime(MELEE, getCreature()->getBaseAttackTime(MELEE) * 2);
247 _applyAura(Deadmines::SMITES_HAMMER);
248 break;
249 }
250
251 // Wait at the chest for 4.5seconds -- Still needs work
252 getCreature()->setAttackTimer(MELEE, 4500);
253 mWaitAtChest = _addTimer(4500);
254 setScriptPhase(getScriptPhase() + 1);
255 }
256
257protected:
258 CreatureAISpells* mStomp;
259 uint32_t mWaitAtChest;
260};
