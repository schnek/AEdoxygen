221 Player* GetRandomPlayerTarget()
222 {
223 std::vector< uint32_t > possible_targets;
224 for (const auto& iter : getCreature()->getInRangePlayersSet())
225 {
226 if (iter && static_cast<Player*>(iter)->isAlive())
227 possible_targets.push_back(static_cast<uint32_t>(iter->getGuid()));
228 }
229 if (possible_targets.size() > 0)
230 {
231 uint32_t random_player = possible_targets[Util::checkChance(uint32_t(possible_targets.size() - 1))];
232 return getCreature()->getWorldMap()->getPlayer(random_player);
233 }
234 return nullptr;
235 }
