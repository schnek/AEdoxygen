133 void OnGameObjectActivate(GameObject* pGameObject, Player* /*pPlayer*/) override
134 {

137 case GO_ALTAR1_SLADRAN:
138 {
139 GameObject* pStatue = GetGameObjectByGuid(mSladranStatueGUID);

142
143 pGameObject->setFlags(GO_FLAG_NOT_SELECTABLE);
144 mStatueCount++;
145 }
146 break;
147 case GO_ALTAR2_COLOSSUS:
148 {
149 GameObject* pStatue = GetGameObjectByGuid(mColossusStatueGUID);
150 if (pStatue)

152
153 pGameObject->setFlags(GO_FLAG_NOT_SELECTABLE);
154 mStatueCount++;
155 }
156 break;
157 case GO_ALTAR3_MOORABI:
158 {
159 GameObject* pStatue = GetGameObjectByGuid(mMoorabiStatueGUID);
160 if (pStatue)
161 pStatue->setState(pStatue->getState() == 1 ? 0 : 1);
162
163 pGameObject->setFlags(GO_FLAG_NOT_SELECTABLE);
164 mStatueCount++;
165 }
166 break;
167 }
168
169 if (mStatueCount < 3)
170 return;
171
172 GameObject* pTrapDoors = GetGameObjectByGuid(mTrapDoorGUID);
173 GameObject* pCoilision = GetGameObjectByGuid(mCoilisionGUID);
174 if (pTrapDoors)
175 pTrapDoors->setState(pTrapDoors->getState() == 1 ? 0 : 1);
176 if (pCoilision)
177 pCoilision->setState(pCoilision->getState() == 1 ? 0 : 1);
178 }

180 void OnCreatureDeath(Creature* pVictim, Unit* /*pKiller*/) override
181 {
182 GameObject* pDoors = NULL;
183 GameObject* pAltar = NULL;
184 switch (pVictim->getEntry())
185 {
186 case CN_MOORABI:
187 {
188 pAltar = GetGameObjectByGuid(mMoorabiAltarGUID);
189 if (pAltar)
190 pAltar->removeFlags(GO_FLAG_NOT_SELECTABLE);
191
192 if (mInstance->getDifficulty() != InstanceDifficulty::DUNGEON_HEROIC)
193 return;
194
195 pDoors = GetGameObjectByGuid(mEckDoorsGUID);
196 if (pDoors)
197 pDoors->setState(GO_STATE_OPEN);
198 }
199 break;
200 case CN_GAL_DARAH:
201 {
202 pDoors = GetGameObjectByGuid(mDoor1GUID);
203 if (pDoors)
204 pDoors->setState(GO_STATE_OPEN);
205
206 pDoors = GetGameObjectByGuid(mDoor2GUID);
207 if (pDoors)
208 pDoors->setState(GO_STATE_OPEN);
209 }
210 break;
211 case CN_SLADRAN:
212 {
213 pAltar = GetGameObjectByGuid(mSladranAltarGUID);
214 if (pAltar)
215 pAltar->removeFlags(GO_FLAG_NOT_SELECTABLE);
216 }
217 break;
218 case CN_DRAKKARI_COLOSSUS:
219 {
220 pAltar = GetGameObjectByGuid(mColossusAltarGUID);
221 if (pAltar)
222 pAltar->removeFlags(GO_FLAG_NOT_SELECTABLE);
223 }
224 break;
225 }
226 }
