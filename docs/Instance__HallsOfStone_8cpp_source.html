207 void OnCombatStart(Unit* /*pTarget*/) override
208 {
209 mStompTimerId = _addTimer(STOMP_TIMER);
210 }
211
212 void AIUpdate() override
213 {
214 if (_isTimerFinished(mStompTimerId))
215 {
216 _castAISpell(mStomp);
217 setRooted(true);
218 _resetTimer(mStompTimerId, (STOMP_TIMER + SHATTER_TIMER));
219 mShatterTimerId = _addTimer(SHATTER_TIMER);
220
221 }
222 if (_isTimerFinished(mShatterTimerId))
223 {
224 _castAISpell(mShatter);
225 setRooted(false);
226 _removeTimer(mShatterTimerId);
227 }
228 }
229
230protected:
231 CreatureAISpells* mStomp;
232 CreatureAISpells* mShatter;
233
234 uint32_t mStompTimerId;
235 uint32_t mShatterTimerId;
