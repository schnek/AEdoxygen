31class SelinFireheartAI : public CreatureAIScript
32{
33public:
34 static CreatureAIScript* Create(Creature* c) { return new SelinFireheartAI(c); }
35 explicit SelinFireheartAI(Creature* pCreature) : CreatureAIScript(pCreature)
36 {
37 addAISpell(SF_DRAINLIFE, 8.0f, TARGET_RANDOM_SINGLE, 0, 35);
38
39 if (isHeroic())
40 addAISpell(SF_DRAINMANA, 8.0f, TARGET_RANDOM_SINGLE, 0, 35);
41
42 ManaRage = sSpellMgr.getSpellInfo(FC_MANARAGE);
43 ManaRageTrigger = addAISpell(FC_MANARAGE_TRIGGER, 0.0f, TARGET_SELF, 0, 0);
44 FelExplosion = addAISpell(SF_FELEXPLOSION, 0.0f, TARGET_SELF, 0, 0);
45 mEnableFelExplosion = false;
46 }
47
48 void OnCombatStart(Unit* /*pTarget*/) override
49 {
50 // Selin Fireheart starts with 0 mana and drains it from the felcrystals in the room
51 // \todo Set it so mana regen is off
52 getCreature()->setPower(POWER_TYPE_MANA, 0);
53 }
54
55 // During the AIUpdate() Selin will spam FelExplosion until hes out of mana
56 // He will then attempt to gain mana from a FelCrystal thats in the room by running to them
57 void AIUpdate() override
58 {
59 // 10% of his mana according to wowhead is 3231 which is whats needed to cast FelExplosion
60 if (_getManaPercent() < 10 || mEnableFelExplosion == false)
61 Mana();
62 else if (!_isCasting())// Mana is greater than 10%
63 CastFelExplosion();
64 }
65
66 void Mana()
67 {
68 // Attempt to get a Fel Crystal and move to it if not in range.
69 // Once in range we get the FelCrystal to cast Mana Rage on Selin
70 Unit* FelCrystal = nullptr;
71 PreventActions(false);
72
73 FelCrystal = FindFelCrystal();
74
75 if (!FelCrystal || !FelCrystal->isAlive())
76 {
77 PreventActions(true);
78 FelCrystal = nullptr;
79 return;
80 }
81
82 // Not in range
83 if (getCreature()->GetDistance2dSq(FelCrystal) > 100)
84 {
85 moveTo(FelCrystal->GetPositionX(), FelCrystal->GetPositionY(), FelCrystal->GetPositionZ());
86 FelCrystal = nullptr;
87 return;
88 }
89
90 getCreature()->stopMoving();
91
92 if (!FelCrystal->isCastingSpell())
93 FelCrystal->castSpell(getCreature(), ManaRage, false);
94
95 // Mana Rage giving of mana doesnt work so we give 10%(3231) / AIUpdate() Event.
96 _castAISpell(ManaRageTrigger);
97 uint32_t mana = getCreature()->getPower(POWER_TYPE_MANA) + 3231;
98 if (mana >= getCreature()->getMaxPower(POWER_TYPE_MANA))
99 mana = getCreature()->getMaxPower(POWER_TYPE_MANA);
100
101 getCreature()->setPower(POWER_TYPE_MANA, mana);
102
103 // Re-Enable FelExplosion
104 if (_getManaPercent() >= 100)
105 PreventActions(true);
106
107 FelCrystal = nullptr;
108 }
109
110 void PreventActions(bool Allow)
111 {
112 mEnableFelExplosion = Allow;
113 _setMeleeDisabled(!Allow);
114 _setRangedDisabled(!Allow);
115 _setCastDisabled(!Allow);
116 _setTargetingDisabled(Allow);
117 }
118
119 Unit* FindFelCrystal()
120 {
121 // Find a FelCrystal
122 Unit* FC = nullptr;
123 for (uint8_t x = 0; x < 5; x++)
124 {
125 FC = getNearestCreature(FelCrystals[x].x, FelCrystals[x].y, FelCrystals[x].z, CN_TRASH_FELCRYSTALS);
126 if (!FC || !FC->isAlive() || FC->GetInstanceID() != getCreature()->GetInstanceID())
127 FC = nullptr;
128 else
129 break;
130 }
131 return FC;
132 }
133
134 void CastFelExplosion()
135 {
136 _castAISpell(FelExplosion);
137
138 // No Idea why the mana isnt taken when the spell is cast so had to manually take it -_-
139 getCreature()->setPower(POWER_TYPE_MANA, getCreature()->getPower(POWER_TYPE_MANA) - 3231);
140 }
141
142 SpellInfo const* ManaRage;
143 CreatureAISpells* ManaRageTrigger;
144 CreatureAISpells* FelExplosion;
145 bool mEnableFelExplosion;
146};

167

170 addEmoteForEvent(Event_OnDied, 4861); // It is... not over.
171 }
172
173 void AIUpdate() override
174 {
175 if ((_getHealthPercent() <= 85 && mSummon == 0) ||
176 (_getHealthPercent() <= 70 && mSummon == 1) ||
177 (_getHealthPercent() <= 55 && mSummon == 2) ||
178 (_getHealthPercent() <= 40 && mSummon == 3) ||
179 (_getHealthPercent() <= 25 && mSummon == 4))
180 {
181 _castAISpell(mPureEnergy);
182 ++mSummon;
183 //spawnCreature(CN_PURE_ENERGY, 231, -207, 6, 0, true);
184 }
185
186 if (_getHealthPercent() <= 10 && isScriptPhase(1))
187 setScriptPhase(2);
188 }

190 CreatureAISpells* mPureEnergy;
191 uint8_t mSummon;
