42 void OnPlayerEnter(Player* pPlayer)
43 {
44 if (pPlayer->getGender() == 0)

50 void SetData(uint32_t pIndex, uint32_t pData)
51 {
52 if (pIndex >= OHF_END)
53 return;
54
55 if (pIndex == OHF_PHASE_2)
56 mInstance->getWorldStatesHandler().SetWorldStateForZone(2367, 0, WORLDSTATE_OLD_HILLSBRAD_BARRELS, 0);
57
58 m_phaseData[pIndex] = pData;
59 }
60
61 uint32_t GetData(uint32_t pIndex)

63 if (pIndex >= OHF_END)
64 return 0;

67 }
68
69 void OnGameObjectActivate(GameObject* pGameObject, Player* /*pPlayer*/)
70 {
71 if (pGameObject->getEntry() != GO_LODGE_ABLAZE || GetData(OHF_PHASE_1) == OHF_DATA_DONE)
72 return;
73
74 pGameObject->despawn(1000, 0);
75 m_numBarrel++;
76 pGameObject->getWorldMap()->getWorldStatesHandler().SetWorldStateForZone(2367, 0, WORLDSTATE_OLD_HILLSBRAD_BARRELS, m_numBarrel);
77 if (m_numBarrel != 5)
78 return;
79
80 SetData(OHF_PHASE_1, OHF_DATA_DONE);
81
82 for (auto& itr : mInstance->getPlayers())
83 {
84 if (auto* player = itr.second)
85 {
86 if (auto* questLog = player->getQuestLogByQuestId(10283))
87 {
88 questLog->setMobCountForIndex(0, questLog->getMobCountByIndex(0) + 1);
89 questLog->sendUpdateAddKill(0);
90 questLog->updatePlayerFields();
91 }
92 }
93 }
94
95 for (uint8_t i = 0; i < 21; ++i)
96 {
97 GameObject* pGO = spawnGameObject(GO_FIRE, Fires[i].x, Fires[i].y, Fires[i].z, Fires[i].o);
98 if (pGO != nullptr)
99 pGO->despawn(10 * 60 * 1000, 0);
100 }
101
102 spawnCreature(CN_LIEUTENANT_DRAKE, 2118.310303f, 89.565969f, 52.453037f, 2.027089f);
103 }
104};
