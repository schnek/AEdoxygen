175class WhitemaneAI : public CreatureAIScript

180 {
181 sleep = addAISpell(ScarletMonastery::SP_WHITEMANE_SLEEP, 0.0f, TARGET_ATTACKING, 0, 0, false, true);
182 sleep->setAttackStopTimer(1000);
183
184 resurrection = addAISpell(ScarletMonastery::SP_WHITEMANE_RESURRECTION, 0.0f, TARGET_VARIOUS, 0, 0, false, true);
185 resurrection->setAttackStopTimer(1000);
186 resurrection->addEmote("Arise, my champion!", CHAT_MSG_MONSTER_YELL, ScarletMonastery::SAY_SOUND_RESTALK2);
187
188 addEmoteForEvent(Event_OnCombatStart, ScarletMonastery::SAY_WHITEMANE_01);
189 addEmoteForEvent(Event_OnTargetDied, ScarletMonastery::SAY_WHITEMANE_02);
190 addEmoteForEvent(Event_OnDied, ScarletMonastery::SAY_WHITEMANE_03);
191 }
192
193 void OnDamageTaken(Unit* /*mAttacker*/, uint32_t fAmount) override
194 {
195 if (fAmount < 5)
196 return;
197
198 if (getCreature()->getHealthPct() <= 50 && getScriptPhase() == 1)
199 ChangeToPhase1();
200 }
201
202 void ChangeToPhase1()
203 {
204 setScriptPhase(2);
205
206 sendDBChatMessage(2106);
207 CastSleep();
208 CastRes();
209 }
210
211 void CastSleep()
212 {
213 getCreature()->castSpell(getCreature(), sleep->mSpellInfo, true);
214 }
215
216 void CastRes()
217 {
218 auto morgrain = getNearestCreature(ScarletMonastery::CN_COMMANDER_MOGRAINE);
219 if (morgrain != nullptr)
220 getCreature()->castSpell(morgrain, resurrection->mSpellInfo, true);
221 }
222
223protected:
224 CreatureAISpells* sleep;
225 CreatureAISpells* resurrection;
