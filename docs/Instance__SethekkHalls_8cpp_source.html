215public:
216 static CreatureAIScript* Create(Creature* c) { return new LakkaAI(c); }
217 explicit LakkaAI(Creature* pCreature) : CreatureAIScript(pCreature)
218 {
219 stopMovement();
220
221 //WPs
222 for (uint8_t i = 1; i < 4; ++i)

224 addWaypoint(1, createWaypoint(i, 0, WAYPOINT_MOVE_TYPE_WALK, LakkaWaypoint[i]));

227
228 void OnReachWP(uint32_t type, uint32_t iWaypointId) override
229 {
230 if (type != WAYPOINT_MOTION_TYPE)
231 return;
232
233 switch (iWaypointId)
234 {
235 case 1:
236 {
237 setWaypointToMove(1, 2);
238 for (const auto& itr : getCreature()->getInRangeObjectsSet())
239 {
240 if (itr && itr->isPlayer())
241 {
242 Player* pPlayer = static_cast<Player*>(itr);
243 if (pPlayer != nullptr)
244 {
245 QuestLogEntry* pQuest = pPlayer->getQuestLogByQuestId(10097);
246 if (pQuest != nullptr && pQuest->getMobCountByIndex(1) < 1)
247 {
248 pQuest->setMobCountForIndex(1, 1);
249 pQuest->sendUpdateAddKill(1);
250 pQuest->updatePlayerFields();
251 }
252 }
253 }
254 }
255 }
256 break;
257 case 3:
258 {
259 despawn(100, 0);
260 }
261 break;
262 default:
263 {
264 setWaypointToMove(1, 1);
265 }
266 }
267 }
