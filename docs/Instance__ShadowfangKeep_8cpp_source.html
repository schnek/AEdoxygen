646 distance = std::sqrt(distanceX - distanceY);
647 }
648 else if (i != ShadowfangKeep::ashcrombeWpCount-1)
649 {
650 distanceX = (ShadowfangKeep::SorcererAshcrombeWPS[i].x - ShadowfangKeep::SorcererAshcrombeWPS[i-1].x)*(ShadowfangKeep::SorcererAshcrombeWPS[i].x - ShadowfangKeep::SorcererAshcrombeWPS[i-1].x);
651 distanceY = (ShadowfangKeep::SorcererAshcrombeWPS[i].y - ShadowfangKeep::SorcererAshcrombeWPS[i-1].y)*(ShadowfangKeep::SorcererAshcrombeWPS[i].y - ShadowfangKeep::SorcererAshcrombeWPS[i-1].y);
652 distance = std::sqrt(distanceX + distanceY);
653 }
