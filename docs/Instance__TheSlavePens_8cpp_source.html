23class TotemsAI : public CreatureAIScript
24{
25public:
26 static CreatureAIScript* Create(Creature* c) { return new TotemsAI(c); }
27 explicit TotemsAI(Creature* pCreature) : CreatureAIScript(pCreature)
28 {
29 uint32_t Despawn = 30000;
30 uint32_t AIUpdate = 1000;
31
32 SpellID = 1;
33 switch (getCreature()->getEntry())
34 {
35 case CN_MENNUS_HEALING_WARD:
36 SpellID = 34977;
37 break;
38 case CN_TAINED_EARTHGRAB_TOTEM:
39 SpellID = 20654;
40 AIUpdate = 5000;
41 break;
42 case CN_TAINED_STONESKIN_TOTEM:
43 Despawn = 60000;
44 SpellID = 25509; // temporary spell
45 AIUpdate = 0;
46 break;
47 default: // for Corrupted Nova Totem and it's also safe case
48 {
49 getCreature()->addUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
50 Despawn = 6000;
51 SpellID = 33132;
52 AIUpdate = 5000;
53 }
54 }
55
56 if (AIUpdate != 0)
57 RegisterAIUpdateEvent(AIUpdate);
58
59 setAIAgent(AGENT_SPELL);
60 getCreature()->Despawn(Despawn, 0);
61 getCreature()->m_noRespawn = true;
62
63 getCreature()->castSpell(getCreature(), SpellID, true);
64 }
65
66 void AIUpdate() override
67 {
68 if (getCreature()->hasUnitFlags(UNIT_FLAG_NON_ATTACKABLE))
69 getCreature()->removeUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
70
71 getCreature()->castSpell(getCreature(), SpellID, true);
72 }
73
74protected:
75 uint32_t SpellID;
76};

105 novaTotem->setAttackStopTimer(1000);
106
107 TotemCounter = 0;
108
109 addEmoteForEvent(Event_OnCombatStart, SAY_MENNU_BETRAYER_01);

112 addEmoteForEvent(Event_OnTargetDied, SAY_MENNU_BETRAYER_04);

117 void OnCombatStart(Unit* /*mTarget*/) override
118 {
119 for (uint8_t i = 0; i < 4; i++)
120 SummonedTotems[i] = false;
121
122 TotemCounter = 0;
123 }
124
125 void AIUpdate() override
126 {
127 if (getScriptPhase() == 1 && TotemCounter != 0 && !getCreature()->isCastingSpell())
128 {
129 TotemSpawning();
130 }
131 }
132
133 // Random totem spawning
134 void TotemSpawning()
135 {
136 getCreature()->setAttackTimer(MELEE, 1500);
137
138 bool Spawned = false;
139 uint32_t Counter = 0;
140 while(!Spawned)
141 {
142 if (Counter >= 2)
143 {
144 for (uint8_t i = 0; i < 4; i++)
145 {
146 if (!SummonedTotems[i])
147 {
148 spawnCreature(Totems[i], getCreature()->GetPosition());
149 getCreature()->castSpell(getCreature(), earthgrabTotem->mSpellInfo, earthgrabTotem->mIsTriggered);
150 getCreature()->castSpell(getCreature(), stoneskinTotem->mSpellInfo, stoneskinTotem->mIsTriggered);
151 getCreature()->castSpell(getCreature(), novaTotem->mSpellInfo, novaTotem->mIsTriggered);
152 SummonedTotems[i] = true;
153 TotemCounter++;
154 break;
155 }
156 }
157
158 Spawned = true;
159 }
160
161 uint32_t i = Util::getRandomUInt(3);
162 if (SummonedTotems[i])
163 Counter++;
164 else
165 {
166 spawnCreature(Totems[i], getCreature()->GetPosition());
167 switch (i)
168 {
169 case 1:
170 getCreature()->castSpell(getCreature(), earthgrabTotem->mSpellInfo, earthgrabTotem->mIsTriggered);
171 break;
172 case 2:
173 getCreature()->castSpell(getCreature(), stoneskinTotem->mSpellInfo, stoneskinTotem->mIsTriggered);
174 break;
175 case 3:
176 getCreature()->castSpell(getCreature(), novaTotem->mSpellInfo, novaTotem->mIsTriggered);
177 break;
178 default:
179 break;
180 }
181
182 SummonedTotems[i] = true;
183 TotemCounter++;
184 Spawned = true;
185 }
186 }
187
188 if (TotemCounter == 4)
189 {
190 for (uint8_t i = 0; i < 4; i++)
191 SummonedTotems[i] = false;
192
193 TotemCounter = 0;
194 }
195 }
196
197protected:
198 uint32_t TotemCounter;
199
200 CreatureAISpells* earthgrabTotem;
201 CreatureAISpells* stoneskinTotem;
202 CreatureAISpells* novaTotem;
203};
