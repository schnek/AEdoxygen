158 void AIUpdate() override
159 {
160 if (Gnomes.size())
161 {
162 Unit* Gnome = NULL;
163 for (std::vector<Unit*>::iterator itr = Gnomes.begin(); itr < Gnomes.end(); ++itr)
164 {
165 Gnome = *itr;
166 if (!Gnome->isAlive() || !Gnome->IsInWorld())
167 {
168 itr = Gnomes.erase(itr);
169 continue;
170 }
171
172 if (Gnome->getAIInterface()->getCurrentTarget())
173 continue;
174
175 if (!Gnome->getAIInterface()->getUnitToFollow())
176 {
177 Gnome->getMovementManager()->moveFollow(getCreature(), 15.0f, Gnome->getFollowAngle());
178 }
179
180 if (getCreature()->GetDistance2dSq(Gnome) > 250.0f)
181 {
182 if (Gnome->isCastingSpell())
183 Gnome->interruptSpell();
184
185 continue;
186 }
187
188 if (!Gnome->isCastingSpell())
189 {
190 Gnome->pauseMovement(1);
191 Gnome->castSpell(getCreature(), REPAIR, false); // core problem? casted on self (and effect is applied on caster instead of _unit)
192 }
193 }
194 }
195
196 if ((getCreature()->getHealthPct() <= 75 && GnomeCounter == 0) || (getCreature()->getHealthPct() <= 50 && GnomeCounter == 1) || (getCreature()->getHealthPct() <= 25 && GnomeCounter == 2))
197 {
198 Unit* Gnome = NULL;

203 {
204 Gnome->getMovementManager()->moveFollow(getCreature(), 15.0f, 2.0f);
205 Gnomes.push_back(Gnome);
206 }
207 }
208
209 sendDBChatMessage(SAY_MEKGINEER_STEAMRIGGER_01);
210
211 GnomeCounter++;
212 }

412 Unit* GetClosestDistiller()
413 {
414 float distance = 50.0f;
415 Unit* pDistiller = NULL;
416 Unit* Unit2Check = NULL;
417
418 for (uint8_t i = 1; i < 5; i++)
419 {
420 Unit2Check = getNearestCreature(Distiller[i].x, Distiller[i].y, Distiller[i].z, 17954);
421 if (!Unit2Check)
422 continue;
423
424 if (!Unit2Check->isAlive())
425 continue;
426
427 float Dist2Unit = getCreature()->GetDistance2dSq(Unit2Check);
428 if (Dist2Unit > distance * distance)
429 continue;
430
431 pDistiller = Unit2Check;
432 distance = std::sqrt(Dist2Unit);
433 DistillerNumber = i;
434 }
435
436 return pDistiller;
437 }
