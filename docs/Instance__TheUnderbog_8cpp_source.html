135 void OnCombatStart(Unit* mTarget) override
136 {
137 Unit* Bear = getNearestCreature(getCreature()->GetPositionX(), getCreature()->GetPositionY(), getCreature()->GetPositionZ(), 17827);
138 if (Bear && Bear->isAlive())
139 Bear->getAIInterface()->onHostileAction(mTarget);

142 void AIUpdate() override
143 {
144 if (getCreature()->getThreatManager().getCurrentVictim())
145 {
146 Unit* target = getCreature()->getThreatManager().getCurrentVictim();
147 if (getCreature()->GetDistance2dSq(target) >= 100.0f && getCreature()->getDistanceSq(target) <= 900.0f && Util::getRandomUInt(3) != 1)
148 {
149 getCreature()->pauseMovement(2000);
150 if (!getCreature()->isCastingSpell())
151 {
152 uint32_t RangedSpell = Util::getRandomUInt(100);
153 if (RangedSpell <= 20 && _isTimerFinished(aimedShot->mCooldownTimer->isTimePassed()))
154 {
155 getCreature()->castSpell(target, aimedShot->mSpellInfo, true);
156 getCreature()->setAttackTimer(MELEE, aimedShot->getAttackStopTimer());
157 aimedShot->mCooldownTimer->resetInterval(aimedShot->mCooldown);
158 }
159
160 if (RangedSpell > 20 && RangedSpell <= 40 && _isTimerFinished(multiShot->mCooldownTimer->isTimePassed()))
161 {
162 getCreature()->castSpell(target, multiShot->mSpellInfo, true);
163 getCreature()->setAttackTimer(MELEE, multiShot->getAttackStopTimer());
164 multiShot->mCooldownTimer->resetInterval(multiShot->mCooldown);
165 }
166 else
167 {
168 if (_isTimerFinished(shot->mCooldownTimer->isTimePassed()))
169 {
170 getCreature()->castSpell(target, shot->mSpellInfo, true);
171 getCreature()->setAttackTimer(MELEE, shot->getAttackStopTimer());
172 shot->mCooldownTimer->resetInterval(shot->mCooldown);
173 }
174 }
175 }
176 }
177 }
178 }

180private:
181 CreatureAISpells* aimedShot;
182 CreatureAISpells* multiShot;
183 CreatureAISpells* shot;
