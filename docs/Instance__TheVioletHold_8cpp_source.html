60bool TheVioletHoldScript::setBossState(uint32_t type, EncounterStates state)
61{
62 if (!InstanceScript::setBossState(type, state))
63 return false;
64
65 switch (type)
66 {
67 case DATA_1ST_BOSS:
68 if (state == Performed)
69 updateEncountersStateForCreature(NPC_EREKEM, getWorldMap()->getDifficulty());
70 break;
71 case DATA_2ND_BOSS:
72 if (state == Performed)
73 updateEncountersStateForCreature(NPC_MORAGG, getWorldMap()->getDifficulty());
74 break;
75 case DATA_CYANIGOSA:
76 if (state == Performed)
77 setLocalData(DATA_MAIN_EVENT_STATE, Performed);
78 break;
79 case DATA_MORAGG:
80 case DATA_EREKEM:
81 case DATA_ICHORON:
82 case DATA_LAVANTHOR:
83 case DATA_XEVOZZ:
84 case DATA_ZURAMAT:
85 if (WaveCount == 6)
86 setBossState(DATA_1ST_BOSS, state);
87 else if (WaveCount == 12)
88 setBossState(DATA_2ND_BOSS, state);
89
90 if (state == Performed)
91 setLocalData(DATA_WAVE_COUNT, WaveCount + 1);
92 break;
93 default:
94 break;
95 }
96
97 return true;
98}

377void TheVioletHoldScript::setLocalData(uint32_t type, uint32_t data)
378{
379 switch (type)
380 {
381 case DATA_WAVE_COUNT:
382 {
383 WaveCount = data;
384 if (WaveCount)
385 {
386 // Add Next Wave
387 scriptEvents.addEvent(EVENT_NEXT_WAVE, (isBossWave(WaveCount - 1) ? 45 : 5) * TimeVarsMs::Second);
388 }
389 } break;
390 case DATA_DOOR_INTEGRITY:
391 {
392 DoorIntegrity = data;
393 Defenseless = false;
394 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_PRISON_STATE, DoorIntegrity);
395 } break;
396 case DATA_MAIN_EVENT_STATE:
397 {
398 EventState = data;
399
400 switch (EventState)
401 {
402 case EncounterStates::NotStarted: // Reset the Encounter
403 {
404 // Open Door
405 if (GameObject* mainDoor = getGameObjectFromData(DATA_MAIN_DOOR))
406 {
407 mainDoor->setState(GO_STATE_OPEN);
408 mainDoor->removeFlags(GO_FLAG_LOCKED);
409 }
410
411 // Reset Worldstates
412 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_WAVE_COUNT, WaveCount);
413 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_PRISON_STATE, DoorIntegrity);
414 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_SHOW, 0);
415
416 // Make Activation Crystalls not Selectable
417 for (uint8_t i = 0; i < ActivationCrystalCount; ++i)
418 if (GameObject* crystal = GetGameObjectByGuid(ActivationCrystalGUIDs[i]))
419 crystal->setFlags(GO_FLAG_NOT_SELECTABLE);
420 } break;
421 case EncounterStates::InProgress: // We Started the Encounter
422 {
423 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_WAVE_COUNT, WaveCount);
424 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_PRISON_STATE, DoorIntegrity);
425 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_SHOW, 1);
426
427 WaveCount = 1;
428 scriptEvents.addEvent(EVENT_NEXT_WAVE, 1 * TimeVarsMs::Second);
429 scriptEvents.addEvent(EVENT_STATE_CHECK, 3 * TimeVarsMs::Second);
430
431 for (uint8_t i = 0; i < ActivationCrystalCount; ++i)
432 if (GameObject* crystal = GetGameObjectByGuid(ActivationCrystalGUIDs[i]))
433 crystal->removeFlags(GO_FLAG_NOT_SELECTABLE);
434
435 } break;
436 case EncounterStates::Performed: // Encounter Done
437 {
438 // Open Door
439 if (GameObject* mainDoor = getGameObjectFromData(DATA_MAIN_DOOR))
440 {
441 mainDoor->setState(GO_STATE_OPEN);
442 mainDoor->removeFlags(GO_FLAG_LOCKED);
443 }
444
445 // Hide Worldstates
446 getInstance()->getWorldStatesHandler().SetWorldStateForZone(4415, 0, WORLD_STATE_VH_SHOW, 0);
447
448 // Play Outro
449 if (Creature* sinclari = getCreatureFromData(DATA_SINCLARI))
450 sinclari->GetScript()->DoAction(ACTION_SINCLARI_OUTRO);
451 } break;
452 }
453 } break;
454 case DATA_START_BOSS_ENCOUNTER:
455 {
456 switch (WaveCount)
457 {
458 case 6:
459 startBossEncounter(FirstBossId);
460 break;
461 case 12:
462 startBossEncounter(SecondBossId);
463 break;
464 }
465 } break;
466 case DATA_HANDLE_CELLS:
467 {
468 handleCells(data, false);
469 } break;
470 default:
471 break;
472 }

527void TheVioletHoldScript::StateCheck()
528{
529 // if main event is in progress and players have wiped then reset instance
530 if ((EventState == EncounterStates::InProgress && WipeCheck()) || EventState == EncounterStates::Failed)
531 {
532 resetBossEncounter(FirstBossId);
533 resetBossEncounter(SecondBossId);
534 resetBossEncounter(DATA_CYANIGOSA);
535
536 WaveCount = 0;
537 DoorIntegrity = 100;
538 Defenseless = true;
539 setLocalData(DATA_MAIN_EVENT_STATE, EncounterStates::NotStarted);
540
541 scriptEvents.resetEvents();
542
543 if (Creature* sinclari = getCreatureFromData(DATA_SINCLARI))
544 {
545 sinclari->getAIInterface()->enterEvadeMode();
546 sinclari->setNpcFlags(UNIT_NPC_FLAG_GOSSIP);
547 }
548 }

551bool TheVioletHoldScript::WipeCheck()
552{
553 for (auto player : getWorldMap()->getPlayers())
554 {
555 if (Player* plr = player.second)
556 {
557 // Skip Gamemasters
558 if (plr->hasPlayerFlags(PLAYER_FLAG_GM))
559 continue;
560
561 // Players still Alive
562 if (plr->isAlive())
563 return false;
564 }
565 }

575void TheVioletHoldScript::resetBossEncounter(uint8_t bossId)
576{
577 if (bossId < DATA_CYANIGOSA || bossId > DATA_ZURAMAT)
578 return;
579
580 Creature* boss = getCreatureFromData(bossId);
581 if (!boss)
582 return;
583
584 switch (bossId)
585 {
586 case DATA_CYANIGOSA:
587 {
588 boss->ToSummon()->unSummon();
589 } break;
590 case DATA_EREKEM:
591 {
592 for (uint32_t i = DATA_EREKEM_GUARD_1; i <= DATA_EREKEM_GUARD_2; ++i)
593 {
594 if (Creature* guard = GetCreatureByGuid(getLocalData(i)))
595 {
596 if (guard->isDead())
597 guard->Despawn(1000, 1);
598
599 if (getBossState(bossId) == EncounterStates::Performed)
600 {
601 guard->SetPosition(guard->GetSpawnPosition());
602 updateKilledBoss(guard);
603 }
604
605 guard->getMovementManager()->moveTargetedHome();
606 guard->getAIInterface()->setIgnoreCreatureCombat(true);
607 guard->getAIInterface()->setIgnorePlayerCombat(true);
608 guard->addUnitFlags(UNIT_FLAG_IGNORE_CREATURE_COMBAT);
609 }
610 }
611 } [[fallthrough]];
612 default:
613 {
614 if (boss->isDead())
615 {
616 // respawn and update to a placeholder npc to avoid be looted again
617 boss->SetPosition(boss->GetSpawnPosition());
618 updateKilledBoss(boss);
619 boss->Despawn(1000, 1);
620 }
621 } break;
622 }
623}

723 scriptEvents.updateEvents(time_passed, getScriptPhase());
724

726 {

728 {
729 case 1:
730 {
731 getCreature()->setMoveWalk(true);
732 getCreature()->getMovementManager()->movePoint(0, SinclariPositions[0]);
733 } break;
734 case 2:
735 {
736 getCreature()->emote(EMOTE_ONESHOT_USESTANDING);
737 // Spawn Defense System From WorldMap so it wont get Despawned later on
738 getCreature()->getWorldMap()->summonCreature(NPC_DEFENSE_SYSTEM, DefenseSystemLocation);

740 case 3:
741 {
742 getCreature()->setFacingTo(SinclariPositions[0].getOrientation());
743 sendDBChatMessage(SAY_SINCLARI_INTRO_1);
744
745 std::list<Creature*> guardList;
746 GetCreatureListWithEntryInGrid(guardList, NPC_VIOLET_HOLD_GUARD, 100.0f);
747 for (Creature* guard : guardList)
748 {
749 if (!guard->isAlive())
750 continue;
751
752 guard->getAIInterface()->setReactState(REACT_PASSIVE);
753 guard->setMoveWalk(false);
754 guard->getMovementManager()->movePoint(0, GuardsMovePosition);
755 }
756 } break;
757 case 4:
758 {
759 getCreature()->getMovementManager()->movePoint(0, SinclariPositions[1]);
760 summons.despawnAll();
761 } break;
762 case 5:
763 {
764 getCreature()->setFacingTo(SinclariPositions[1].getOrientation());
765
766 std::list<Creature*> guardList;
767 GetCreatureListWithEntryInGrid(guardList, NPC_VIOLET_HOLD_GUARD, 100.0f);
768 for (Creature* guard : guardList)
769 guard->setVisible(false);
770 } break;
771 case 6:
772 {
773 sendDBChatMessage(SAY_SINCLARI_INTRO_2);
774 } break;
775 case 7:
776 {
777 getCreature()->emote(EMOTE_ONESHOT_TALK_NOSHEATHE);
778 } break;
779 case 8:
780 {
781 // Close the Door and Lock it

783 {
784 mainDoor->setState(GO_STATE_CLOSED);
785 mainDoor->setFlags(GO_FLAG_LOCKED);

789 {
790 // Tell the Instance Script we are Started
791 mInstance->setLocalData(DATA_MAIN_EVENT_STATE, EncounterStates::InProgress);
792 } break;
793 case 10:
794 {
795 getCreature()->setNpcFlags(UNIT_NPC_FLAG_NONE);

798 break;
799 }
