118 void SetLocaleInstanceData(uint32_t /*pType*/, uint32_t pIndex, uint32_t pData)
119 {
120 switch (pIndex)
121 {
122 case UTGARDE_FORGE_MASTER:
123 {
124 mUtgardeData[UTGARDE_FORGE_MASTER]++;
125 HandleForge();
126 } break;
127 case UTGARDE_INGVAR:
128 {
129 mUtgardeData[UTGARDE_INGVAR] = pData;
130
131 if (pData == Performed)
132 {
133 GameObject* pGO = nullptr;
134 for (uint8_t i = 0; i < 2; ++i)
135 {
136 pGO = GetGameObjectByGuid(mIngvarDoors[i]);
137 if (pGO)
138 {
139 pGO->setState(pGO->getState() == 1 ? 0 : 1);
140 }
141 }
142 }
143 } break;
144 }
145 }
146
147 void HandleForge()
148 {
149 GameObject* pGO = nullptr;
150 pGO = GetGameObjectByGuid(m_fmData[mUtgardeData[UTGARDE_FORGE_MASTER] - 1].mBellow);
151 if (pGO)
152 {
153 pGO->setState(pGO->getState() == 1 ? 0 : 1);
154 }
155
156 pGO = GetGameObjectByGuid(m_fmData[mUtgardeData[UTGARDE_FORGE_MASTER] - 1].mFire);
157 if (pGO)
158 {
159 pGO->setState(pGO->getState() == 1 ? 0 : 1);
160 }
161
162 pGO = GetGameObjectByGuid(m_fmData[mUtgardeData[UTGARDE_FORGE_MASTER] - 1].mAnvil);
163 if (pGO)
164 {
165 pGO->setState(pGO->getState() == 1 ? 0 : 1);
166 }
