60 void AIUpdate() override
61 {
62 plaguecount++;
63 randomplague = 16 + Util::getRandomUInt(3);
64 if (plaguecount >= randomplague && getCreature()->getThreatManager().getCurrentVictim())
65 {
66 plaguecount = 0;
67 Unit* target = getCreature()->getThreatManager().getCurrentVictim();
68 getCreature()->castSpell(target, plague, true);
69 }
70 else if (getCreature()->getHealthPct() <= 30 && morphcheck)
71 {
72 morphcheck = false;
73
74 getCreature()->castSpell(getCreature(), morph, false);
75 }
76 }
