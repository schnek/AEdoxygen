32 std::lock_guard lock(ipBanMutex);
33 _ipBanList.clear();

35 auto result = sLogonSQL->Query("SELECT ip, expire FROM ipbans");
36 if (result)
37 {

40 std::string ipString = result->Fetch()[0].asCString();
41 const uint32_t expireTime = result->Fetch()[1].asUint32();
42
43 std::string smask = "32";
44
45 std::string::size_type i = ipString.find('/');
46 std::string stmp = ipString.substr(0, i);
47 if (i == std::string::npos)
48 sLogger.info("IP ban '{}' netmask not specified. assuming /32", ipString);

50 smask = ipString.substr(i + 1);
51
52 const unsigned int ipraw = Util::makeIP(stmp.c_str());
53 const unsigned int ipmask = atoi(smask.c_str());

56 sLogger.failure("IP ban '{}' could not be parsed. Ignoring", ipString);
57 continue;
58 }
59
60 IPBan ipBan;

62 ipBan.Mask = ipraw;
63 ipBan.Expire = expireTime;
64 ipBan.db_ip = ipString;
65 _ipBanList.push_back(ipBan);
66

118IpBanStatus IpBanMgr::getBanStatus(in_addr ip_address)
119{

121
122 for (auto itr2 = _ipBanList.begin(); itr2 != _ipBanList.end();)
123 {
124 const auto bannedIp = itr2;

131
132 if (static_cast<uint32_t>(UNIXTIME) >= bannedIp->Expire)
133 {
134 sLogonSQL->Execute("DELETE FROM ipbans WHERE expire = %u AND ip = \"%s\"", bannedIp->Expire, sLogonSQL->EscapeString(bannedIp->db_ip).c_str());
135 _ipBanList.erase(bannedIp);
136 }
137 else
138 {
139 return BAN_STATUS_TIME_LEFT_ON_BAN;
