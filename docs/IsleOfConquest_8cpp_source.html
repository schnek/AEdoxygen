449{

451
452 if (m_hasStarted)

454

902void IsleOfConquest::AssaultControlPoint(Player *player, uint32_t id)
903{
904 ControlPointTypes state = controlpoint[id].state;
905 uint32_t team = player->getTeam();
906
907 if (state > IOC_SPAWN_TYPE_HORDE_CONTROLLED)
908 {
909 DLLLogDetail("IOC control point %u is in an invalid state, cannot be assaulted.", id);
910 return;
911 }
912
913 sEventMgr.RemoveEvents(this, EVENT_IOC_CAPTURE_CP_1 + id);
914
915 switch (state)
916 {
917 case IOC_SPAWN_TYPE_NEUTRAL:
918 if (team == TEAM_ALLIANCE)
919 SpawnControlPoint(id, IOC_SPAWN_TYPE_ALLIANCE_ASSAULT);
920 else
921 SpawnControlPoint(id, IOC_SPAWN_TYPE_HORDE_ASSAULT);
922 break;
923
924 case IOC_SPAWN_TYPE_ALLIANCE_ASSAULT:
925 case IOC_SPAWN_TYPE_ALLIANCE_CONTROLLED:
926 if (team == TEAM_HORDE)
927 SpawnControlPoint(id, IOC_SPAWN_TYPE_HORDE_ASSAULT);
928 break;
929
930 case IOC_SPAWN_TYPE_HORDE_ASSAULT:
931 case IOC_SPAWN_TYPE_HORDE_CONTROLLED:
932 if (team == TEAM_ALLIANCE)
933 SpawnControlPoint(id, IOC_SPAWN_TYPE_ALLIANCE_ASSAULT);
934 break;
935 }
936
937 switch (state)
938 {
939 case IOC_SPAWN_TYPE_NEUTRAL:
940 case IOC_SPAWN_TYPE_ALLIANCE_CONTROLLED:
941 case IOC_SPAWN_TYPE_HORDE_CONTROLLED:
942 player->m_bgScore.MiscData[BattlegroundDef::IOC_BASES_ASSAULTED]++;
943 break;
944

946 case IOC_SPAWN_TYPE_HORDE_ASSAULT:
947 player->m_bgScore.MiscData[BattlegroundDef::IOC_BASES_DEFENDED]++;
948 break;
949 }
950
951 if (player->getTeam() == TEAM_ALLIANCE)
952 sendChatMessage(CHAT_MSG_BG_EVENT_ALLIANCE, 0, "%s has assaulted the %s! If it remains uncontested, the alliance will take it within a minute!", player->getName().c_str(), ControlPointNames[id]);
953 else
954 if (player->getTeam() == TEAM_HORDE)
955 sendChatMessage(CHAT_MSG_BG_EVENT_HORDE, 0, "%s has assaulted the %s! If it remains uncontested, the horde will take it within a minute!", player->getName().c_str(), ControlPointNames[id]);
956
957 sEventMgr.AddEvent(this, &IsleOfConquest::CaptureControlPoint, id, EVENT_IOC_CAPTURE_CP_1 + id, 60 * 1 * 1000, 1, 0);
958}
