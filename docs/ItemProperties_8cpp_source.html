42uint32_t ItemProperties::getBuyPriceForItem(uint32_t count, uint32_t factionStanding) const
43{
44 const float pricemod[9] =
45 {
46 1.0f, // HATED
47 1.0f, // HOSTILE
48 1.0f, // UNFRIENDLY
49 1.0f, // NEUTRAL
50 0.95f, // FRIENDLY
51 0.90f, // HONORED
52 0.85f, // REVERED
53 0.80f // EXHALTED
54 };
55
56 int32_t cost = BuyPrice;
57
58 if (factionStanding && factionStanding <= 8)
59 cost = Util::float2int32(ceilf(BuyPrice * pricemod[factionStanding]));
60
61 return cost * count;
62}
