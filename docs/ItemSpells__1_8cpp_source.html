38 How to add a new item spell to the dummy spell handler:
39
40 1) Add a new function to handle the spell at the end of this file but before the
41 SetupItemSpells_1() function. SetupItemSpells_1() must always be the last function.
42
43 2) Register the dummy spell by adding a new line to the end of the list in the
44 SetupItemSpells_1() function.
45
46 Please look at how the other spells are handled and try to use the
47 same variable names and formatting style in your new spell handler.

509bool ExtractGas(uint8_t /*effectIndex*/, Spell* s)
510{
511 bool check = false;
512 uint32_t cloudtype = 0;
513 Creature* creature = nullptr;
514
515 if (!s->getPlayerCaster())
516 return false;
517
518 for (const auto& itr : s->getPlayerCaster()->getInRangeObjectsSet())
519 {
520 if (itr && itr->isCreature())
521 {
522 creature = static_cast<Creature*>(itr);
523 cloudtype = creature->getEntry();
524
525 if (cloudtype == 24222 || cloudtype == 17408 || cloudtype == 17407 || cloudtype == 17378)
526 {
527 if (s->getPlayerCaster()->GetDistance2dSq(itr) < 400)
528 {
529 s->getPlayerCaster()->setTargetGuid(creature->getGuid());
530 check = true;
531 break;
532 }
533 }
534 }
535 }
536
537 if (!check)
538 return false;
539
540 uint32_t item = 0;
541 uint32_t count = 0;
542
543 count = 3 + (Util::getRandomUInt(3));
544
545 if (cloudtype == 24222)
546 item = 22572; //-air
547 if (cloudtype == 17408)
548 item = 22576; //-mana
549 if (cloudtype == 17407)
550 item = 22577; //-shadow
551 if (cloudtype == 17378)
552 item = 22578; //-water
553
554 s->getPlayerCaster()->getItemInterface()->AddItemById(item, count, 0);
555 creature->Despawn(3500, creature->GetCreatureProperties()->RespawnTime);
556
557 return true;
558}

604//////////////////////////////////////////////////////////////////////////////////////////
605//Gnomish Shrink Ray dummy effect handler
606//
607//Preconditions
608// Casted by Player
609//
610//Effect
611// Normally casts spell 13010 on the targeted unit. Shrinking it, and
612// making it weaker
613//
614// On malfunction it either shrinks, or enlarges, the the caster,
615// the caster's party, all enemies, or enlarges the target
616//
617//
618//////////////////////////////////////////////////////////////////////////////////////////
619bool ShrinkRay(uint8_t /*effectIndex*/, Spell* s)
620{
621 if (s->getPlayerCaster() == NULL)
622 return true;
623
624 uint32_t spellids[] =
625 {
626 13004, // grow
627 13010 // shrink
628 };
629
630 uint32_t chance = Util::getRandomUInt(5);
631 bool malfunction = false;
632
633 if (chance == 5)
634 malfunction = true;
635
636 if (!malfunction)
637 {
638
639 s->getPlayerCaster()->castSpell(s->getUnitTarget(), spellids[1], true);
640
641 }
642 else
643 {
644 uint32_t spellindex = Util::getRandomUInt(1);
645 uint32_t who = Util::getRandomUInt(3);
646
647 switch (who)
648 {
649
650 case 0: // us
651 {
652 s->getPlayerCaster()->castSpell(s->getPlayerCaster(), spellids[spellindex], true);
653 }
654 break;
655
656 case 1: // them
657 {
658 // if it's a malfunction it will only grow the target, since shrinking is normal
659 s->getPlayerCaster()->castSpell(s->getUnitTarget(), spellids[0], true);
660 }
661 break;
662
663 case 2: // our party
664 {
665 for (const auto& itr : s->getPlayerCaster()->getInRangePlayersSet())
666 {
667 if (!itr)
668 continue;
669
670 Player* p = static_cast<Player*>(itr);
671 if ((p->GetPhase() & s->getPlayerCaster()->GetPhase()) == 0)
672 continue;
673
674 if (p->getGroup()->GetID() != s->getPlayerCaster()->getGroup()->GetID())
675 continue;
676
677 s->getPlayerCaster()->castSpell(p, spellids[spellindex], true);
678 }
679 }
680 break;
681
682 case 3: // every attacking enemy
683 {
684 for (const auto& itr : s->getPlayerCaster()->getInRangeOppositeFactionSet())
685 {
686 if(!itr)
687 continue;
688
689 Object* o = itr;
690
691 if ((o->GetPhase() & s->getPlayerCaster()->GetPhase()) == 0)
692 continue;
693
694 if (!o->isCreature())
695 continue;
696
697 Unit* u = static_cast<Unit*>(o);
698
699 if (u->getTargetGuid() != s->getPlayerCaster()->getGuid())
700 continue;
701
702 if (!s->getPlayerCaster()->isValidTarget(u))
703 continue;
704
705 s->getPlayerCaster()->castSpell(u, spellids[spellindex], true);
706 }
707 }
708 break;
709 }
710 }
711
712 return true;
713}
