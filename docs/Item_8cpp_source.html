94void Item::create(uint32_t itemId, Player* owner)
95{
96 setEntry(itemId);
97
98 if (owner != nullptr)
99 {
100 setOwner(owner);
101 setContainerGuid(owner->getGuid());
102 }
103
104 setStackCount(1);
105
106 m_itemProperties = sMySQLStore.getItemProperties(itemId);
107 if (!m_itemProperties)
108 {
109 sLogger.failure("Item::create: Can't create item {} missing properties!", itemId);
110 return;
111 }
112
113 for (uint8_t i = 0; i < MAX_ITEM_PROTO_SPELLS; ++i)
114 setSpellCharges(i, m_itemProperties->Spells[i].Charges);
115
116 setDurability(m_itemProperties->MaxDurability);
117 setMaxDurability(m_itemProperties->MaxDurability);
118
119 m_owner = owner;
120 if (m_itemProperties->LockId > 1)
121 m_isLocked = true;
122 else
123 m_isLocked = false;
124}

286bool Item::addEnchantment(uint32_t enchantmentId, EnchantmentSlot slot, uint32_t duration, bool removedAtLogout/* = false*/, uint32_t randomSuffix/* = 0*/)
287{
288 m_isDirty = true;
289
290 WDB::Structures::SpellItemEnchantmentEntry const* Enchantment = nullptr;
291#if VERSION_STRING >= Cata
292 std::unique_ptr<WDB::Structures::SpellItemEnchantmentEntry> custom_enchant = nullptr;
293 switch (slot)
294 {
295
296 case TRANSMOGRIFY_ENCHANTMENT_SLOT:
297 case REFORGE_ENCHANTMENT_SLOT:
298 {
299 custom_enchant = std::make_unique<WDB::Structures::SpellItemEnchantmentEntry>();
300 custom_enchant->Id = enchantmentId;
301
302 Enchantment = custom_enchant.get();
303 } break;
304
305 default:
306 {
307#endif
308 const auto spell_item_enchant = sSpellItemEnchantmentStore.lookupEntry(enchantmentId);
309 if (spell_item_enchant == nullptr)
310 return false;
311
312 Enchantment = spell_item_enchant;
313#if VERSION_STRING >= Cata
314 } break;
315 }
316#endif
317
318 EnchantmentInstance enchantInstance;
319 enchantInstance.BonusApplied = false;
320 enchantInstance.Slot = slot;
321 enchantInstance.Enchantment = Enchantment;
322#if VERSION_STRING >= Cata
323 enchantInstance.customEnchantmentHolder = std::move(custom_enchant);
324#endif
325 enchantInstance.RemoveAtLogout = removedAtLogout;
326 enchantInstance.RandomSuffix = randomSuffix;
327
328 // Set enchantment to item's wowdata fields
329 _setEnchantmentDataFields(slot, Enchantment->Id, duration, 0);
330
331 m_enchantments.try_emplace(slot, std::move(enchantInstance));
332
333 if (m_owner == nullptr)
334 return true;
335
336 // Add the removal event
337 if (duration)
338 m_owner->getItemInterface()->addTemporaryEnchantedItem(this, slot);
339
340 // Do not send log packet if owner is not yet in world
341 if (!m_owner->IsInWorld())
342 return true;
343
344#if VERSION_STRING >= Cata
345 if (slot == TRANSMOGRIFY_ENCHANTMENT_SLOT)
346 return true;
347#endif
348
349 m_owner->sendPacket(SmsgEnchantmentLog(m_owner->getGuid(), m_owner->getGuid(), getEntry(), Enchantment->Id).serialise().get());
350
351 // Apply enchantment bonus only if the item is equipped
352 // but send enchant time update packet for items in inventory as well
353 const auto equipSlot = m_owner->getItemInterface()->GetInventorySlotByGuid(getGuid());
354 if (equipSlot >= EQUIPMENT_SLOT_START && equipSlot < EQUIPMENT_SLOT_END)
355 applyEnchantmentBonus(slot, true);
356 else if (duration)
357 sendEnchantTimeUpdate(slot, duration / 1000);
358
359 return true;
360}

362void Item::removeEnchantment(EnchantmentSlot slot, bool timerExpired/* = false*/)
363{
364 const auto itr = m_enchantments.find(slot);
365 if (itr == m_enchantments.end())
366 return;
367
368 // Remove enchantment bonus
369 if (itr->second.BonusApplied)
370 applyEnchantmentBonus(slot, false);
371
372 _setEnchantmentDataFields(slot, 0, 0, 0);
373 m_enchantments.erase(itr);
374
375 if (!timerExpired)
376 m_owner->getItemInterface()->removeTemporaryEnchantedItem(this, slot);
377}

441void Item::applyEnchantmentBonus(EnchantmentSlot slot, bool apply)
442{
443 if (m_owner == nullptr)
444 return;
445
446 const auto enchantment = m_enchantments.find(slot);
447 if (enchantment == m_enchantments.end())
448 return;
449
450 WDB::Structures::SpellItemEnchantmentEntry const* Entry = enchantment->second.Enchantment;
451 const uint32_t RandomSuffixAmount = enchantment->second.RandomSuffix;
452
453 if (enchantment->second.BonusApplied == apply)
454 return;
455
456 enchantment->second.BonusApplied = apply;
457
458 if (apply)
459 {
460 if (const auto duration = getEnchantmentDuration(enchantment->second.Slot) / 1000)
461 sendEnchantTimeUpdate(enchantment->second.Slot, duration);
462 }
463
464 uint32_t itemSlot = m_owner->getItemInterface()->GetInventorySlotByGuid(getGuid());
465 if (itemSlot < EQUIPMENT_SLOT_END)
466 {
467#if VERSION_STRING >= WotLK
468 if (slot == PERM_ENCHANTMENT_SLOT || slot == TEMP_ENCHANTMENT_SLOT)
469#else
470 if (slot < MAX_INSPECTED_ENCHANTMENT_SLOT)
471#endif
472 m_owner->setVisibleItemEnchantment(itemSlot, slot, (apply ? static_cast<uint16_t>(Entry->Id) : 0));
473 }
474 else if (apply)
475 {
476 sLogger.failure("Item::applyEnchantmentBonus : Tried to apply visual enchantment but equipment slot {} is invalid", itemSlot);
477 }
478
479#if VERSION_STRING >= Cata
480 if (slot == REFORGE_ENCHANTMENT_SLOT)
481 m_owner->applyReforgeEnchantment(this, apply);
482#endif
483
484 for (uint32_t c = 0; c < 3; c++)
485 {
486 if (Entry->type[c])
487 {
488 switch (Entry->type[c])
489 {
490 case ITEM_ENCHANTMENT_TYPE_COMBAT_SPELL:
491 {
492 if (apply)
493 {
494 if (Entry->spell[c] != 0)
495 {
496 const auto procChance = Entry->min[c] == 0 ? Util::float2int32(static_cast<float>(getItemProperties()->Delay) * 0.001f / 60.0f * 100.0f) : Entry->min[c];
497 switch (m_owner->getItemInterface()->GetInventorySlotByGuid(getGuid()))
498 {
499 case EQUIPMENT_SLOT_MAINHAND:
500 m_owner->addProcTriggerSpell(Entry->spell[c], 0, m_owner->getGuid(), procChance, static_cast<SpellProcFlags>(PROC_ON_DONE_MELEE_HIT | PROC_ON_DONE_MELEE_SPELL_HIT), EXTRA_PROC_ON_MAIN_HAND_HIT_ONLY, nullptr, nullptr, nullptr, this);
501 break;
502 case EQUIPMENT_SLOT_OFFHAND:
503 m_owner->addProcTriggerSpell(Entry->spell[c], 0, m_owner->getGuid(), procChance, static_cast<SpellProcFlags>(PROC_ON_DONE_MELEE_HIT | PROC_ON_DONE_MELEE_SPELL_HIT | PROC_ON_DONE_OFFHAND_ATTACK), EXTRA_PROC_ON_OFF_HAND_HIT_ONLY, nullptr, nullptr, nullptr, this);
504 break;
505 case EQUIPMENT_SLOT_RANGED:
506 m_owner->addProcTriggerSpell(Entry->spell[c], 0, m_owner->getGuid(), procChance, static_cast<SpellProcFlags>(PROC_ON_DONE_RANGED_HIT | PROC_ON_DONE_RANGED_SPELL_HIT), EXTRA_PROC_NULL, nullptr, nullptr, nullptr, this);
507 break;
508 default:
509 m_owner->addProcTriggerSpell(Entry->spell[c], 0, m_owner->getGuid(), procChance, static_cast<SpellProcFlags>(PROC_ON_DONE_MELEE_HIT | PROC_ON_DONE_MELEE_SPELL_HIT), EXTRA_PROC_NULL, nullptr, nullptr, nullptr, this);
510 break;
511 }
512 }
513 }
514 else
515 {
516 m_owner->removeProcTriggerSpell(Entry->spell[c], m_owner->getGuid(), getGuid());
517 }
518 }
519 break;
520
521 case ITEM_ENCHANTMENT_TYPE_DAMAGE:
522 {
523 int32_t val = Entry->min[c];
524 if (RandomSuffixAmount)
525 val = RANDOM_SUFFIX_MAGIC_CALCULATION(RandomSuffixAmount, getPropertySeed());
526
527 if (apply)
528 m_owner->modModDamageDonePositive(SCHOOL_NORMAL, val);
529 else
530 m_owner->modModDamageDonePositive(SCHOOL_NORMAL, -val);
531
532 m_owner->calculateDamage();
533 }
534 break;
535
536 case ITEM_ENCHANTMENT_TYPE_EQUIP_SPELL:
537 {
538 if (apply)
539 {
540 SpellCastTargets targets(m_owner->getGuid());
541
542 if (Entry->spell[c] != 0)
543 {
544 SpellInfo const* sp = sSpellMgr.getSpellInfo(Entry->spell[c]);
545 if (sp == nullptr)
546 continue;
547
548 Spell* spell = sSpellMgr.newSpell(m_owner, sp, true, nullptr);
549 spell->setItemCaster(this);
550 spell->prepare(&targets);
551 }
552 }
553 else
554 {
555 if (Entry->spell[c] != 0)
556 m_owner->removeAuraByItemGuid(Entry->spell[c], getGuid());
557 }
558 }
559 break;
560
561 case ITEM_ENCHANTMENT_TYPE_RESISTANCE:
562 {
563 int32_t val = Entry->min[c];
564 if (RandomSuffixAmount)
565 val = RANDOM_SUFFIX_MAGIC_CALCULATION(RandomSuffixAmount, getPropertySeed());
566
567 if (apply)
568 m_owner->m_flatResistanceModifierPos[Entry->spell[c]] += val;
569 else
570 m_owner->m_flatResistanceModifierPos[Entry->spell[c]] -= val;
571
572 m_owner->calcResistance(static_cast<uint8_t>(Entry->spell[c]));
573 }
574 break;
575
576 case ITEM_ENCHANTMENT_TYPE_STAT:
577 {
578 int32_t val = Entry->min[c];
579 if (RandomSuffixAmount)
580 val = RANDOM_SUFFIX_MAGIC_CALCULATION(RandomSuffixAmount, getPropertySeed());
581
582 m_owner->modifyBonuses(Entry->spell[c], val, apply);
583 m_owner->updateStats();
584 }
585 break;
586
587 case ITEM_ENCHANTMENT_TYPE_TOTEM:
588 {
589 if (apply)
590 {
591 int32_t val = Entry->min[c];
592 if (RandomSuffixAmount)
593 val = RANDOM_SUFFIX_MAGIC_CALCULATION(RandomSuffixAmount, getPropertySeed());
594
595 int32_t value = static_cast<int32_t>(getItemProperties()->Delay) * val / 1000;
596 m_owner->modModDamageDonePositive(SCHOOL_NORMAL, value);
597 }
598 else
599 {
600 int32_t val = Entry->min[c];
601 if (RandomSuffixAmount)
602 val = RANDOM_SUFFIX_MAGIC_CALCULATION(RandomSuffixAmount, getPropertySeed());
603
604 int32_t value = -static_cast<int32_t>(getItemProperties()->Delay * val / 1000);
605 m_owner->modModDamageDonePositive(SCHOOL_NORMAL, value);
606 }
607 m_owner->calculateDamage();
608 }
609 break;
610
611 case ITEM_ENCHANTMENT_TYPE_USE_SPELL:
612 {
613 for (uint8_t i = 0; i < 3; ++i)
614 {
615 if (apply)
616 m_onUseSpellIds[i] = Entry->spell[i];
617 else
618 m_onUseSpellIds[i] = 0;
619 }
620 break;
621 }
622
623 case ITEM_ENCHANTMENT_TYPE_PRISMATIC_SOCKET:
624 {
625 break;
626 }
627
628 default:
629 sLogger.failure("Unknown enchantment type: {} ({})", Entry->type[c], Entry->Id);
630 break;
631 }
632 }
633 }
634}

688 continue;

698 {

701 }
702 }
703 }
704 else
705 {
706 auto item_random_suffix = sItemRandomSuffixStore.lookupEntry(abs(int(getRandomPropertiesId())));
707 for (uint8_t k = 0; k < 3; ++k)
708 {
709 if (item_random_suffix == nullptr)
710 continue;
711

715 if (spell_item_enchant == nullptr)

722 if (_findFreeRandomEnchantmentSlot(&newSlot, RandomEnchantmentType::SUFFIX))
