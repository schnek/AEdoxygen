149 {
150 if (!spellProc->getProcOwner()->isPlayer())
151 return false;
152
153 // Check for correct class
154 const auto plr = static_cast<Player*>(spellProc->getProcOwner());
155 if (plr->getClass() != WARRIOR &&
156 plr->getClass() != PALADIN &&
157 plr->getClass() != HUNTER &&
158 plr->getClass() != ROGUE &&
159 plr->getClass() != DEATHKNIGHT &&
160 plr->getClass() != SHAMAN &&
161 plr->getClass() != DRUID)
162 return false;
163
164 return true;
165 }

168 {

170
171 // Cast random effect
172 if (isHeroic)
173 plr->castSpell(plr, heroicProcs[plr->getClass()][Util::getRandomUInt(2)], true);
174 else
175 plr->castSpell(plr, normalProcs[plr->getClass()][Util::getRandomUInt(2)], true);
176
177 return SpellScriptExecuteState::EXECUTE_PREVENT;
178 }
179

254// Shadowmourne grants stackable Soul Fragments on hits and on 10th stack an aoe and a buff
255class Shadowmourne : public SpellScript
256{
257public:
258 SpellScriptCheckDummy onAuraDummyEffect(Aura* aur, AuraEffectModifier* /*aurEff*/, bool apply) override
259 {
260 // On dummy aura apply make it proc self
261 if (apply)
262 aur->getOwner()->addProcTriggerSpell(aur->getSpellInfo(), aur->getCasterGuid(), aur);
263 else
264 aur->getOwner()->removeProcTriggerSpell(aur->getSpellId(), aur->getCasterGuid());
265
266 return SpellScriptCheckDummy::DUMMY_OK;
267 }
268
269 void onAuraRemove(Aura* aur, AuraRemoveMode /*mode*/) override
270 {
271 // Also remove Soul Fragments
272 aur->getOwner()->removeAllAurasById(SPELL_SOUL_FRAGMENT);
273 }
274
275 bool canProc(SpellProc* spellProc, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, DamageInfo /*damageInfo*/) override
276 {
277 if (!spellProc->getProcOwner()->isPlayer())
278 return false;
279
280 // Check if caster has Chaos Bane buff active which prevents gathering Soul Fragments
281 if (spellProc->getProcOwner()->hasAurasWithId(SPELL_CHAOS_BANE_BUFF))
282 return false;
283
284 return true;
285 }
286
287 SpellScriptExecuteState onDoProcEffect(SpellProc* spellProc, Unit* victim, SpellInfo const* /*castingSpell*/, DamageInfo /*damageInfo*/) override
288 {
289 // Check stacks
290 const auto aur = spellProc->getProcOwner()->getAuraWithId(SPELL_SOUL_FRAGMENT);
291 if (aur != nullptr && aur->getStackCount() >= 9)
292 {
293 // Cast aoe spell
294 spellProc->getProcOwner()->castSpell(victim, SPELL_CHAOS_BANE_AOE, true);
295 // Cast self buff
296 spellProc->getProcOwner()->castSpell(spellProc->getProcOwner(), SPELL_CHAOS_BANE_BUFF, true);
297 // Remove Soul Fragments
298 spellProc->getProcOwner()->removeAllAurasById(SPELL_SOUL_FRAGMENT);
299 }
300 else
301 {
302 // Add new Soul Fragment
303 spellProc->getProcOwner()->castSpell(spellProc->getProcOwner(), SPELL_SOUL_FRAGMENT, true);
304 }
305
306 return SpellScriptExecuteState::EXECUTE_PREVENT;
307 }
