397 template <typename T>
398 class ArcLuna
399 {
400 public:
401 typedef int(*mfp)(lua_State* L, T* ptr);
402 typedef struct { const char* name; mfp mfunc; } RegType;
403
404 static void Register(lua_State* L)
405 {
406 lua_newtable(L);
407 int methods = lua_gettop(L);
408
409 luaL_newmetatable(L, GetTClassName<T>());
410 int metatable = lua_gettop(L);
411
412 luaL_newmetatable(L, "DO NOT TRASH");
413 lua_pop(L, 1);
414
415 // store method table in globals so that
416 // scripts can add functions written in Lua.
417 lua_pushvalue(L, methods);
418 lua_setglobal(L, GetTClassName<T>());
419
420 // hide metatable from Lua getmetatable()
421 lua_pushvalue(L, methods);
422 lua_setfield(L, metatable, "__metatable");
423
424 lua_pushcfunction(L, index);
425 lua_setfield(L, metatable, "__index");
426
427 lua_pushcfunction(L, tostring_T);
428 lua_setfield(L, metatable, "__tostring");
429
430 lua_pushcfunction(L, gc_T);
431 lua_setfield(L, metatable, "__gc");
432
433 lua_newtable(L); // mt for method table
434 lua_setmetatable(L, methods);
435
436 // fill method table with methods from class T
437 for (RegType* l = ((RegType*)GetMethodTable<T>()); l->name; l++)
438 {
439 lua_pushstring(L, l->name);
440 lua_pushlightuserdata(L, (void*)l);
441 lua_pushcclosure(L, thunk, 1);
442 lua_settable(L, methods);
443 }
444 lua_pop(L, 2); // drop metatable and method table
445 }
446
447 // push onto the Lua stack a userdata containing a pointer to T object
448 static int push(lua_State* L, T* obj, bool gc = false)
449 {
450 if (!obj)
451 {
452 lua_pushnil(L); return lua_gettop(L);
453 }
454
455 luaL_getmetatable(L, GetTClassName<T>()); // lookup metatable in Lua registry
456 if (lua_isnil(L, -1))
457 luaL_error(L, "%s missing metatable", GetTClassName<T>());
458
459 int mt = lua_gettop(L);
460 T** ptrHold = (T**)lua_newuserdata(L, sizeof(T**));
461 int ud = lua_gettop(L);
462 if (ptrHold != nullptr)
463 {
464 *ptrHold = obj;
465 lua_pushvalue(L, mt);
466 lua_setmetatable(L, -2);
467 char name[32];
468 tostring(name, obj);
469 lua_getfield(L, LUA_REGISTRYINDEX, "DO NOT TRASH");
470 if (lua_isnil(L, -1))
471 {
472 luaL_newmetatable(L, "DO NOT TRASH");
473 lua_pop(L, 1);
474 }
475 lua_getfield(L, LUA_REGISTRYINDEX, "DO NOT TRASH");
476 if (gc == false)
477 {
478 lua_pushboolean(L, 1);
479 lua_setfield(L, -2, name);
480 }
481 lua_pop(L, 1);
482 }
483 lua_settop(L, ud);
484 lua_replace(L, mt);
485 lua_settop(L, mt);
486 return mt; // index of userdata containing pointer to T object
487 }
488
489 // get userdata from Lua stack and return pointer to T object
490 static T* check(lua_State* L, int narg)
491 {
492 T** ptrHold = static_cast<T**>(lua_touserdata(L, narg));
493 if (ptrHold == nullptr)
494 return nullptr;
495 return *ptrHold;
496 }
497
498 private:
499 ArcLuna(); // hide default constructor
500
501 // member function dispatcher
502 static int thunk(lua_State* L)
503 {
504 // stack has userdata, followed by method args
505 T* obj = check(L, 1); // get 'self', or if you prefer, 'this'
506 lua_remove(L, 1); // remove self so member function args start at index 1
507 // get member function from upvalue
508 RegType* l = static_cast<RegType*>(lua_touserdata(L, lua_upvalueindex(1)));
509 //return (obj->*(l->mfunc))(L); // call member function
510 return l->mfunc(L, obj);
511 }
512
513 // garbage collection metamethod
514 static int gc_T(lua_State* L)
515 {
516 T* obj = check(L, 1);
517 if (obj == nullptr)
518 return 0;
519
520 lua_getfield(L, LUA_REGISTRYINDEX, "DO NOT TRASH");
521 if (lua_istable(L, -1))
522 {
523 char name[32];
524 tostring(name, obj);
525 lua_getfield(L, -1, std::string(name).c_str());
526 if (lua_isnil(L, -1))
527 {
528 delete obj;
529 obj = nullptr;
530 }
531 }
532 lua_pop(L, 3);
533 return 0;
534 }
535 static int tostring_T(lua_State* L)
536 {
537 char buff[32];
538 T** ptrHold = (T**)lua_touserdata(L, 1);
539 T* obj = *ptrHold;
540 sprintf(buff, "%p", obj);
541 lua_pushfstring(L, "%s (%s)", GetTClassName<T>(), buff);
542 return 1;
543 }
544
545 static void tostring(char* buff, void* obj)
546 {
547 sprintf(buff, "%p", obj);
548 }
549 static int index(lua_State* L)
550 {
551 /*Paroxysm : the table obj and the missing key are currently on the stack(index 1 & 2) */
552 lua_getglobal(L, GetTClassName<T>());
553 // string form of the key.
554 const char* key = lua_tostring(L, 2);
555 if (lua_istable(L, -1))
556 {
557 lua_pushvalue(L, 2);
558 lua_rawget(L, -2);
559 // If the key were looking for is not in the table, retrieve its' metatables' index value.
560 if (lua_isnil(L, -1))
561 {
562 lua_getmetatable(L, -2);
563 if (lua_istable(L, -1))
564 {
565 lua_getfield(L, -1, "__index");
566 if (lua_isfunction(L, -1))
567 {
568 lua_pushvalue(L, 1);
569 lua_pushvalue(L, 2);
570 lua_pcall(L, 2, 1, 0);
571 }
572 else if (lua_istable(L, -1))
573 lua_getfield(L, -1, key);
574 else
575 lua_pushnil(L);
576 }
577 else
578 lua_pushnil(L);
579 }
580 else if (lua_istable(L, -1))
581 {
582 lua_pushvalue(L, 2);
583 lua_rawget(L, -2);
584 }
585 }
586 else
587 lua_pushnil(L);
588
589 lua_insert(L, 1);
590 lua_settop(L, 1);
591 return 1;
592 }
593 };

634 static int push(lua_State* L, uint64_t guid)
635 {
636 int index = 0;
637 if (guid == 0)
638 {
639 lua_pushnil(L);
640 index = lua_gettop(L);
641 }
642 else
643 {
644 luaL_getmetatable(L, GetName());
645 if (lua_isnoneornil(L, -1))

648 }

658 else
659 {
660 (*guidHold) = guid;
661 lua_pushvalue(L, mt);
662 lua_setmetatable(L, ud);
663 lua_replace(L, mt);
664 lua_settop(L, mt);
665 index = mt;
666 }
667 }
