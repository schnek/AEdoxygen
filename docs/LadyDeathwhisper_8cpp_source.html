153void LadyDeathwhisperAI::AIUpdate(unsigned long time_passed)
154{
155 if (!_isInCombat() && !getScriptPhase() == PHASE_INTRO)
156 return;
157
158 scriptEvents.updateEvents(time_passed, getScriptPhase());
159
160 if (_isCasting())
161 return;
162
163 while (uint32_t eventId = scriptEvents.getFinishedEvent())
164 {
165 switch (eventId)
166 {
167 case EVENT_INTRO_2:
168 {
169 sendDBChatMessage(SAY_LADY_INTRO_2);
170 break;
171 }
172 case EVENT_INTRO_3:
173 {
174 sendDBChatMessage(SAY_LADY_INTRO_3);
175 break;
176 }
177 case EVENT_INTRO_4:
178 {
179 sendDBChatMessage(SAY_LADY_INTRO_4);
180 break;
181 }
182 case EVENT_INTRO_5:
183 {
184 sendDBChatMessage(SAY_LADY_INTRO_5);
185 break;
186 }
187 case EVENT_INTRO_6:
188 {
189 sendDBChatMessage(SAY_LADY_INTRO_6);
190 break;
191 }
192 case EVENT_INTRO_7:
193 {
194 sendDBChatMessage(SAY_LADY_INTRO_7);
195 setScriptPhase(PHASE_ONE);
196 break;
197 }
198 case EVENT_DEATH_AND_DECAY:
199 {
200 if (Unit* target = getBestPlayerTarget())
201 {
202 deathAndDecaySpell->setCustomTarget(target);
203 _castAISpell(deathAndDecaySpell);
204 }
205 scriptEvents.addEvent(EVENT_DEATH_AND_DECAY, Util::getRandomUInt(22000, 30000));
206 break;
207 }
208 case EVENT_DOMINATE_MIND_H:
209 {

211 for (uint8_t i = 0; i < dominateMindCount; i++)
212 if (Unit* target = getBestPlayerTarget(TargetFilter_NotCurrent))

242 {
243 EmpowerCultist();

249 _castAISpell(frostBoltSpell);

260 {
261 _castAISpell(touchOfInsignifcanceSpell);
262 scriptEvents.addEvent(EVENT_P2_TOUCH_OF_INSIGNIFICANCE, Util::getRandomUInt(9000, 13000), PHASE_TWO);

264 }

284 _castAISpell(berserkSpell);

381void LadyDeathwhisperAI::EmpowerCultist()
382{
383 if (summons.empty())
384 return;
385
386 std::list<Creature*> temp;
387 for (auto itr = summons.begin(); itr != summons.end(); ++itr)
388 if ((*itr)->isAlive() && ((*itr)->getEntry() == NPC_CULT_FANATIC || (*itr)->getEntry() == NPC_CULT_ADHERENT))
389 temp.push_back((*itr));
390
391 if (temp.empty())
392 return;
393
394 // select random cultist
395 uint8_t i = static_cast<uint8_t>(Util::getRandomUInt(0, static_cast<uint32_t>(temp.size() - 1)));
396 auto it = temp.begin();
397 std::advance(it, i);
398
399 Creature* cultist = (*it);
400 if (cultist->getEntry() == NPC_CULT_FANATIC)
401 {
402 darkTransformationSpell->setCustomTarget(cultist);
403 _castAISpell(darkTransformationSpell);
404 }
405 else
406 {
407 darkEmpowermentSpell->setCustomTarget(cultist);
408 _castAISpell(darkEmpowermentSpell);
409 }

428/// Spell: Mana Barrier
429SpellScriptExecuteState ManaBarrier::onAuraPeriodicTick(Aura* aur, AuraEffectModifier* /*aurEff*/, float_t* /*damage*/)
430{
431 // Aura should periodically trigger spell every 1 sec but that spell is serverside so we don't have it in DBC
432 // Overwrite default periodic tick with converting mana to hp
433
434 const auto auraOwner = aur->getOwner();
435 // If aura owner doesn't use mana, remove aura
436 if (auraOwner->getMaxPower(POWER_TYPE_MANA) == 0)
437 {
438 aur->removeAura();
439 return SpellScriptExecuteState::EXECUTE_PREVENT;
440 }
441
442 const auto healthDelta = auraOwner->getMaxHealth() - auraOwner->getHealth();
443 if (healthDelta == 0)
444 {
445 // Unit is already at max health, so do nothing
446 return SpellScriptExecuteState::EXECUTE_PREVENT;
447 }
448
449 const auto currentMana = auraOwner->getPower(POWER_TYPE_MANA);
450 if (healthDelta < currentMana)
451 {
452 // Restore health to max and remove equal amount of mana
453 auraOwner->setPower(POWER_TYPE_MANA, currentMana - healthDelta);
454 auraOwner->setHealth(auraOwner->getMaxHealth());
455 return SpellScriptExecuteState::EXECUTE_PREVENT;
456 }
457
458 // Unit takes more damage than it has mana left => remove aura
459 aur->removeAura();
460
461 // Aura is used by Lady Deathwhisper, change phase when all mana is drained
462 if (auraOwner->isCreature())
463 {
464 const auto creatureOwner = static_cast<Creature*>(auraOwner);
465 if (creatureOwner->GetScript() != nullptr)
466 creatureOwner->GetScript()->DoAction(ACTION_MANABARRIER_DOWN);
467 }
468
469 return SpellScriptExecuteState::EXECUTE_PREVENT;
470}
