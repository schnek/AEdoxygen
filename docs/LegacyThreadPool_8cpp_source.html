52bool CThreadPool::ThreadExit(Thread* t)
53{
54 _mutex.acquire();
55
56 // we're definitely no longer active
57 m_activeThreads.erase(t);
58
59 // do we have to kill off some threads?
60 if(_threadsToExit > 0)
61 {
62 // kill us.
63 --_threadsToExit;
64 ++_threadsExitedSinceLastCheck;
65 if(t->DeleteAfterExit)
66 m_freeThreads.erase(t);
67
68 _mutex.release();
69 delete t;
70 return false;
71 }
72
73 // enter the "suspended" pool
74 ++_threadsExitedSinceLastCheck;
75 ++_threadsEaten;
76 std::set<Thread*>::iterator itr = m_freeThreads.find(t);
77
78 if(itr != m_freeThreads.end())
79 {
80 sLogger.failure("Thread {} duplicated with thread {}", (*itr)->ControlInterface.GetId(), t->ControlInterface.GetId());
81 }
82 m_freeThreads.insert(t);
83
84 sLogger.debug("Thread {} entered the free pool.", t->ControlInterface.GetId());
85 _mutex.release();
86 return true;
87}

89void CThreadPool::ExecuteTask(ThreadBase* ExecutionTarget)
90{
91 Thread* t;
92 _mutex.acquire();
93 ++_threadsRequestedSinceLastCheck;
94 --_threadsEaten;
95
96 // grab one from the pool, if we have any.
97 if(m_freeThreads.size())
98 {
99 t = *m_freeThreads.begin();
100 m_freeThreads.erase(m_freeThreads.begin());
101
102 // execute the task on this thread.
103 t->ExecutionTarget = ExecutionTarget;
104
105 // resume the thread, and it should start working.
106 t->ControlInterface.Resume();
107 sLogger.debug("Thread {} left the thread pool.", t->ControlInterface.GetId());
108 }
109 else
110 {
111 // creating a new thread means it heads straight to its task.
112 // no need to resume it :)
113 t = StartThread(ExecutionTarget);
114 }
115
116 // add the thread to the active set
117 sLogger.debug("Thread {} is now executing task at {}", t->ControlInterface.GetId(), fmt::ptr(ExecutionTarget));
118
119 m_activeThreads.insert(t);
120 _mutex.release();
121}

144void CThreadPool::IntegrityCheck()
145{
146 _mutex.acquire();
147 int32_t gobbled = _threadsEaten;
148
149 if(gobbled < 0)
150 {
151 // this means we requested more threads than we had in the pool last time.
152 // spawn "gobbled" + THREAD_RESERVE extra threads.
153 uint32_t new_threads = abs(gobbled) + THREAD_RESERVE;
154 _threadsEaten = 0;
155
156 for(uint32_t i = 0; i < new_threads; ++i)
157 StartThread(NULL);
158
159 sLogger.debug("ThreadPool : (gobbled < 0) Spawning {} threads.", new_threads);
160 }
161 else if(gobbled < THREAD_RESERVE)
162 {
163 // this means while we didn't run out of threads, we were getting damn low.
164 // spawn enough threads to keep the reserve amount up.
165 uint32_t new_threads = (THREAD_RESERVE - gobbled);
166 for(uint32_t i = 0; i < new_threads; ++i)
167 StartThread(NULL);
168
169 sLogger.debug("ThreadPool : (gobbled <= 5) Spawning {} threads.", new_threads);
170 }
171 else if(gobbled > THREAD_RESERVE)
172 {
173 // this means we had "excess" threads sitting around doing nothing.
174 // lets kill some of them off.
175 uint32_t kill_count = (gobbled - THREAD_RESERVE);
176 KillFreeThreads(kill_count);
177 _threadsEaten -= kill_count;
178 sLogger.debug("ThreadPool : (gobbled > 5) Killing {} threads.", kill_count);
179 }
180 else
181 {
182 // perfect! we have the ideal number of free threads.
183 sLogger.debug("ThreadPool : Perfect!");
184 }
185
186 _threadsExitedSinceLastCheck = 0;
187 _threadsRequestedSinceLastCheck = 0;
188 _threadsFreedSinceLastCheck = 0;
189
190 _mutex.release();
191}

193void CThreadPool::KillFreeThreads(uint32_t count)
194{
195 sLogger.debug("ThreadPool : Killing {} excess threads.", count);
196 _mutex.acquire();
197 Thread* t;
198 ThreadSet::iterator itr;
199 uint32_t i;
200 for(i = 0, itr = m_freeThreads.begin(); i < count && itr != m_freeThreads.end(); ++i, ++itr)
201 {
202 t = *itr;
203 t->ExecutionTarget = NULL;
204 t->DeleteAfterExit = true;
205 ++_threadsToExit;
206 t->ControlInterface.Resume();
207 }
208 _mutex.release();
209}

211void CThreadPool::Shutdown()
212{
213 _mutex.acquire();
214 size_t tcount = m_activeThreads.size() + m_freeThreads.size(); // exit all
215 sLogger.debug("ThreadPool : Shutting down {} threads.", tcount);
216 KillFreeThreads((uint32_t)m_freeThreads.size());
217 _threadsToExit += (uint32_t)m_activeThreads.size();
218
219 for(std::set< Thread* >::iterator itr = m_activeThreads.begin(); itr != m_activeThreads.end(); ++itr)
220 {
221
222 Thread* t = *itr;
223
224 if(t->ExecutionTarget)
225 t->ExecutionTarget->onShutdown();
226 else
227 t->ControlInterface.Resume();
228 }
229 _mutex.release();
230
231 for(int i = 0;; i++)
232 {
233 _mutex.acquire();
234 if(m_activeThreads.size() || m_freeThreads.size())
235 {
236 if(i != 0 && m_freeThreads.size() != 0)
237 {
238 /*if we are here then a thread in the free pool checked if it was being shut down just before CThreadPool::Shutdown() was called,
239 but called Suspend() just after KillFreeThreads(). All we need to do is to resume it.*/
240 Thread* t;
241 ThreadSet::iterator itr;
242 for(itr = m_freeThreads.begin(); itr != m_freeThreads.end(); ++itr)
243 {
244 t = *itr;
245 t->ControlInterface.Resume();
246 }
247 }
248 sLogger.debug("ThreadPool : {} active and {} free threads remaining...", m_activeThreads.size(), m_freeThreads.size());
249 _mutex.release();
250 Arcemu::Sleep(1000);
251 continue;
252 }
253
254 _mutex.release();
255 break;
256 }
257}

262static unsigned long WINAPI thread_proc(void* param)
263{
264 Thread* t = (Thread*)param;
265 t->SetupMutex.acquire();
266 uint32_t tid = t->ControlInterface.GetId();
267 bool ht = (t->ExecutionTarget != NULL);
268 t->SetupMutex.release();
269 sLogger.debug("Thread {} started.", t->ControlInterface.GetId());
270
271 for(;;)
272 {
273 if(t->ExecutionTarget != NULL)
274 {
275 if(t->ExecutionTarget->runThread())
276 delete t->ExecutionTarget;
277
278 t->ExecutionTarget = NULL;
279 }
280
281 if(!ThreadPool.ThreadExit(t))
282 {
283 sLogger.debug("Thread {} exiting.", tid);
284 break;
285 }
286 else
287 {
288 if(ht)
289 sLogger.debug("Thread {} waiting for a new task.", tid);
290 // enter "suspended" state. when we return, the threadpool will either tell us to fuk off, or to execute a new task.
291 t->ControlInterface.Suspend();
292 // after resuming, this is where we will end up. start the loop again, check for tasks, then go back to the threadpool.
293 }
294 }
295
296 // at this point the t pointer has already been freed, so we can just cleanly exit.
297 ExitThread(0);
298}

300Thread* CThreadPool::StartThread(ThreadBase* ExecutionTarget)
301{
302 HANDLE h;
303 Thread* t = new Thread;
304
305 t->DeleteAfterExit = false;
306 t->ExecutionTarget = ExecutionTarget;
307 //h = (HANDLE)_beginthreadex(NULL, 0, &thread_proc, (void*)t, 0, NULL);
308 t->SetupMutex.acquire();
309 h = CreateThread(NULL, 0, &thread_proc, (LPVOID)t, 0, (LPDWORD)&t->ControlInterface.thread_id);
310 t->ControlInterface.Setup(h);
311 t->SetupMutex.release();
312
313 return t;

318static void* thread_proc(void* param)
319{
320 Thread* t = (Thread*)param;
321 t->SetupMutex.acquire();
322 sLogger.debug("Thread {} started.", t->ControlInterface.GetId());
323 t->SetupMutex.release();
324
325 for(;;)
326 {
327 if(t->ExecutionTarget != NULL)
328 {
329 if(t->ExecutionTarget->runThread())
330 delete t->ExecutionTarget;
331
332 t->ExecutionTarget = NULL;
333 }
334
335 if(!ThreadPool.ThreadExit(t))
336 break;
337 else
338 {
339 // enter "suspended" state. when we return, the threadpool will either tell us to fuk off, or to execute a new task.
340 t->ControlInterface.Suspend();
341 // after resuming, this is where we will end up. start the loop again, check for tasks, then go back to the threadpool.
342 }
343 }
344
345 pthread_exit(0);

348Thread* CThreadPool::StartThread(ThreadBase* ExecutionTarget)
349{
350 pthread_t target;
351 Thread* t = new Thread;
352 t->ExecutionTarget = ExecutionTarget;
353 t->DeleteAfterExit = false;
354
355 // lock the main mutex, to make sure id generation doesn't get messed up
356 _mutex.acquire();
357 t->SetupMutex.acquire();
358 pthread_create(&target, NULL, &thread_proc, (void*)t);
359 t->ControlInterface.Setup(target);
360 pthread_detach(target);
361 t->SetupMutex.release();
362 _mutex.release();
363 return t;
364}
