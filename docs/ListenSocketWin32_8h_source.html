24 ListenSocket(const char* ListenAddress, uint32_t Port)

26 m_socket = WSASocketW(AF_INET, SOCK_STREAM, 0, NULL, 0, WSA_FLAG_OVERLAPPED);
27 SocketOps::ReuseAddr(m_socket);
28 SocketOps::Blocking(m_socket);
29 SocketOps::SetTimeout(m_socket, 60);
30
31 m_address.sin_family = AF_INET;
32 m_address.sin_port = ntohs((u_short)Port);
33 m_address.sin_addr.s_addr = htonl(INADDR_ANY);
34 m_opened = false;
35
36 if (strcmp(ListenAddress, "0.0.0.0"))
37 {
38 struct hostent* hostname = gethostbyname(ListenAddress);
39 if (hostname != 0)
40 memcpy(&m_address.sin_addr.s_addr, hostname->h_addr_list[0], hostname->h_length);
41 }
42
43 // bind.. well attempt to.
44 int ret = ::bind(m_socket, (const sockaddr*)&m_address, sizeof(m_address));
45 if (ret != 0)
46 {
47 sLogger.failure("Bind unsuccessful on port {}.", Port);
48 return;
49 }
50
51 ret = listen(m_socket, 5);
52 if (ret != 0)
53 {
54 sLogger.failure("Unable to listen on port {}.", Port);
55 return;
56 }
57
58 m_opened = true;
59 len = sizeof(sockaddr_in);

61 }
