58bool LocationVector::isWithinBox(LocationVector const& center, float xradius, float yradius, float zradius) const
59{
60 // rotate the Object position instead of rotating the whole cube, that way we can make a simplified
61 // is-in-cube check and we have to calculate only one point instead of 4
62
63 // 2PI = 360*, keep in mind that ingame orientation is counter-clockwise
64 double rotation = 2 * M_PI - center.getOrientation();
65 double sinVal = std::sin(rotation);
66 double cosVal = std::cos(rotation);
67
68 float BoxDistX = getPositionX() - center.getPositionX();
69 float BoxDistY = getPositionY() - center.getPositionY();
70
71 float rotX = float(center.getPositionX() + BoxDistX * cosVal - BoxDistY * sinVal);
72 float rotY = float(center.getPositionY() + BoxDistY * cosVal + BoxDistX * sinVal);
73
74 // box edges are parallel to coordiante axis, so we can treat every dimension independently :D
75 float dz = getPositionZ() - center.getPositionZ();
76 float dx = rotX - center.getPositionX();
77 float dy = rotY - center.getPositionY();
78 if ((std::fabs(dx) > xradius) ||
79 (std::fabs(dy) > yradius) ||
80 (std::fabs(dz) > zradius))
81 return false;
82
83 return true;
84}
