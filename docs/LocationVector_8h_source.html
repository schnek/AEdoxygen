109 // std::sqrt(dx * dx + dy * dy + dz * dz)
110 float Distance(const LocationVector & comp)
111 {
112 float delta_x = comp.x - x;
113 float delta_y = comp.y - y;
114 float delta_z = comp.z - z;
115
116 return sqrtf(delta_x * delta_x + delta_y * delta_y + delta_z * delta_z);
117 }

169 y = getPositionY() + (offset.getPositionY() * std::cos(getOrientation()) + offset.getPositionX() * std::sin(getOrientation()));
170 z = getPositionZ() + offset.getPositionZ();
