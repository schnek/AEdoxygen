43 void Logger::initalizeLogger(std::string file_prefix)
44 {
45#ifdef _WIN32
46 handle_stdout = GetStdHandle(STD_OUTPUT_HANDLE);
47#endif
48
49 std::string normal_filename = file_prefix + "-normal.log";
50 std::string error_filename = file_prefix + "-error.log";
51
52 std::string current_date_time = Util::GetCurrentDateTimeString();
53
54 char logMessage[32768];
55 sprintf(logMessage, "=================[%s]=================", current_date_time.c_str());
56
57 this->normalLogFile = fopen(normal_filename.c_str(), "a");
58 if (this->normalLogFile == nullptr)
59 std::cerr << __FUNCTION__ << " : Error opening file " << normal_filename << std::endl;
60 else
61 writeFile(this->normalLogFile, logMessage);
62
63 this->errorLogFile = fopen(error_filename.c_str(), "a");
64 if (this->errorLogFile == nullptr)
65 std::cerr << __FUNCTION__ << " : Error opening file " << error_filename << std::endl;
66 else
67 writeFile(this->errorLogFile, logMessage);
68 }

80 void Logger::log(Severity severity, MessageType messageType, std::string_view message)
81 {
82#if defined(_MSC_VER)
83 SetConsoleOutputCP(65001);
84#endif
85 if (this->minimumMessageType > messageType)
86 return;
87
88 auto logMessage = fmt::format("{} {}{}: {}", Util::GetCurrentTimeString(), getSeverityText(severity), getMessageTypeText(messageType), message);
89
90 setSeverityConsoleColor(severity);
91 std::cout << logMessage << "\n";
92 setConsoleColor(CONSOLE_COLOR_NORMAL);
93
94 if (severity >= Severity::FAILURE)
95 writeFile(this->errorLogFile, logMessage);
96
97 writeFile(this->normalLogFile, logMessage);
98 }

212 std::string getFormattedFileName(const std::string& path_prefix, const std::string& file_prefix, bool use_date_time)
213 {
214 std::stringstream path_name;
215 path_name << path_prefix;
216
217 if (use_date_time)
218 {
219 std::string current_date_time = Util::GetCurrentDateTimeString();
220 //replace time seperator with valid character for file name
221 std::replace(current_date_time.begin(), current_date_time.end(), ':', '-');
222 std::replace(current_date_time.begin(), current_date_time.end(), ' ', '_');
223
224 path_name << current_date_time << "_";
225 }
226
227 path_name << file_prefix << ".log";
228
