13// Required for fmt lib 10.0+ because enums are not formatted automatically anymore -Appled
14template <typename EnumType>
15 requires std::is_enum_v<EnumType>
16struct fmt::formatter<EnumType> : fmt::formatter<std::underlying_type_t<EnumType>>
17{
18 // Forwards the formatting by casting the enum to it's underlying type
19 auto format(const EnumType& enumValue, format_context& ctx) const
20 {
21 return fmt::formatter<std::underlying_type_t<EnumType>>::format(
22 static_cast<std::underlying_type_t<EnumType>>(enumValue), ctx);
23 }
24};
