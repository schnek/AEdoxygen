53void LogonCommClientSocket::OnRead()
54{
55 while (true)
56 {
57 if (!remaining)
58 {
59 if (readBuffer.GetSize() < 6)
60 return; // no header
61
62 // read header
63 readBuffer.Read(&opcode, 2);
64 readBuffer.Read(&remaining, 4);
65
66 // decrypt the first two bytes
67 if (use_crypto)
68 {
69 _rwCrypto.process((uint8_t*)&opcode, (uint8_t*)&opcode, 2);
70 _rwCrypto.process((uint8_t*)&remaining, (uint8_t*)&remaining, 4);
71 }
72
73 // convert network byte order
74 byteSwapUInt32(&remaining);
75 }
76
77 // do we have a full packet?
78 if (readBuffer.GetSize() < remaining)
79 return;
80
81 // create the buffer
82 WorldPacket buff(opcode, remaining);
83 if (remaining)
84 {
85 buff.resize(remaining);
86 //Read(remaining, (uint8_t*)buff.contents());
87 readBuffer.Read(buff.contents(), remaining);
88 }
89
90 // decrypt the rest of the packet
91 if (use_crypto && remaining)
92 _rwCrypto.process(buff.contents(), buff.contents(), remaining);
93
94 // handle the packet
95 HandlePacket(buff);
96
97 remaining = 0;
98 opcode = 0;
99 }
100}

103{

124 &LogonCommClientSocket::HandleModifyDatabaseResult, // LRSMSG_ACCOUNT_DB_MODIFY_RESULT
125 &LogonCommClientSocket::HandlePopulationRequest, // LRSMSG_REALM_POPULATION_REQUEST
126 NULL, // LRCMSG_REALM_POPULATION_RESULT

130 &LogonCommClientSocket::HandleResultAllAccount, // LRSMSG_ALL_ACCOUNT_RESULT
131 };
132
133 if (recvData.GetOpcode() >= LRMSG_MAX_OPCODES || Handlers[recvData.GetOpcode()] == 0)

136 return;
137 }
138

170void LogonCommClientSocket::HandleSessionInfo(WorldPacket& recvData)
171{
172 uint32_t request_id;
173 recvData >> request_id;
174
175 Mutex & m = sLogonCommHandler.getPendingLock();
176 m.acquire();
177
178 // find the socket with this request
179 WorldSocket* sock = sLogonCommHandler.getWorldSocketForClientRequestId(request_id);
180 if (sock == nullptr || sock->Authed || !sock->IsConnected()) // Expired/Client disconnected
181 {
182 m.release();
183 return;
184 }
185
186 // extract sessionkey / account information (done by WS)
187 sock->Authed = true;
188 sLogonCommHandler.removeUnauthedClientSocket(request_id);
189 sock->InformationRetreiveCallback(recvData, request_id);
190 m.release();

209void LogonCommClientSocket::SendPacket(WorldPacket* data, bool no_crypto)
210{
211 LogonWorldPacket header;
212 if (!IsConnected() || IsDeleted())
213 return;
214
215 BurstBegin();
216
217 header.opcode = data->GetOpcode();
218 //header.size = ntohl((u_long)data->size());
219 header.size = (uint32_t)data->size();
220
221 byteSwapUInt32(&header.size);
222
223 if (use_crypto && !no_crypto)
224 _sendCrypto.process((unsigned char*)&header, (unsigned char*)&header, 6);
225
226 bool rv = BurstSend((const uint8_t*)&header, 6);
227
228 if (data->size() > 0 && rv)
229 {
230 if (use_crypto && !no_crypto)
231 _sendCrypto.process(data->contents(), data->contents(), (unsigned int)data->size());
232
233 rv = BurstSend(data->contents(), (uint32_t)data->size());
234 }
235
236 if (rv) BurstPush();
237 BurstEnd();
238}

297void LogonCommClientSocket::HandleRequestAccountMapping(WorldPacket& recvData)
298{
299 auto startTime = Util::TimeNow();
300 uint32_t realm_id;
301 uint32_t account_id;
302 std::map<uint32_t, uint8_t> mapping_to_send;
303 std::map<uint32_t, uint8_t>::iterator itr;
304
305 // grab the realm id
306 recvData >> realm_id;
307
308 // fetch the character mapping
309 auto result = CharacterDatabase.Query("SELECT acct FROM characters");
310
311 if (result)
312 {
313 do
314 {
315 account_id = result->Fetch()[0].asUint32();
316 itr = mapping_to_send.find(account_id);
317 if (itr != mapping_to_send.end())
318 itr->second++;
319 else
320 mapping_to_send.insert(std::make_pair(account_id, static_cast<uint8_t>(1)));
321 }
322 while (result->NextRow());
323 }
324
325 if (!mapping_to_send.size())
326 {
327 // no point sending empty shit
328 return;
329 }
330
331 ByteBuffer uncompressed(40000 * 5 + 8);
332 //uint32_t Count = 0;
333 uint32_t Remaining = (uint32_t)mapping_to_send.size();
334 itr = mapping_to_send.begin();
335 for (;;)
336 {
337 // Send no more than 40000 characters at once.
338 uncompressed << realm_id;
339
340 if (Remaining > 40000)
341 uncompressed << uint32_t(40000);
342 else
343 uncompressed << Remaining;
344
345 for (uint32_t i = 0; i < 40000; ++i, ++itr)
346 {
347 uncompressed << uint32_t(itr->first) << uint8_t(itr->second);
348 if (!--Remaining)
349 break;
350 }
351
352 CompressAndSend(uncompressed);
353 if (!Remaining)
354 break;
355
356 uncompressed.clear();
357 }
358 sLogger.info("LogonCommClient : Build character mapping in {} ms. ({})", static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)), static_cast<uint32_t>(mapping_to_send.size()));
359}

361void LogonCommClientSocket::CompressAndSend(ByteBuffer& uncompressed)
362{
363 // I still got no idea where this came from :p
364 size_t destsize = uncompressed.size() + uncompressed.size() / 10 + 16;
365
366 // w000t w000t kat000t for gzipped packets
367 WorldPacket data(LRCMSG_ACC_CHAR_MAPPING_RESULT, destsize + 4);
368 data.resize(destsize + 4);
369
370 z_stream stream;
371 stream.zalloc = 0;
372 stream.zfree = 0;
373 stream.opaque = 0;
374
375 if (deflateInit(&stream, 1) != Z_OK)
376 {
377 sLogger.failure("deflateInit failed.");
378 return;
379 }
380
381 // set up stream pointers
382 stream.next_out = data.contents() + 4;
383 stream.avail_out = (uInt)destsize;
384 stream.next_in = uncompressed.contents();
385 stream.avail_in = (uInt)uncompressed.size();
386
387 // call the actual process
388 if (deflate(&stream, Z_NO_FLUSH) != Z_OK ||
389 stream.avail_in != 0)
390 {
391 sLogger.failure("deflate failed.");
392 return;
393 }
394
395 // finish the deflate
396 if (deflate(&stream, Z_FINISH) != Z_STREAM_END)
397 {
398 sLogger.failure("deflate failed: did not end stream");
399 return;
400 }
401
402 // finish up
403 if (deflateEnd(&stream) != Z_OK)
404 {
405 sLogger.failure("deflateEnd failed.");
406 return;
407 }
408
409 *(uint32_t*)data.contents() = (uint32_t)uncompressed.size();
410
411 data.resize(stream.total_out + 4);
412 SendPacket(&data, false);
413}

450void LogonCommClientSocket::HandleModifyDatabaseResult(WorldPacket& recvData)
451{
452 uint32_t method_id;
453 uint8_t result_id;
454 //Get the result/method id for further processing
455 recvData >> method_id;
456 recvData >> result_id;
457
458 switch (method_id)
459 {
460 case Method_Account_Change_PW:
461 {
462 std::string account_name;
463 recvData >> account_name;
464 const char* account_string = account_name.c_str();
465
466 WorldSession* pSession = sWorld.getSessionByAccountName(account_string);
467 if (pSession == nullptr)
468 {
469 sLogger.failure("No session found!");
470 return;
471 }
472
473 if (result_id == Result_Account_PW_wrong)
474 {
475 pSession->SystemMessage("Your entered old password did not match database password!");
476 }
477 else if (result_id == Result_Account_SQL_error)
478 {
479 pSession->SystemMessage("Something went wrong by updating mysql data!");
480 }
481 else if (result_id == Result_Account_Finished)
482 {
483 pSession->SystemMessage("Your password is now updated");
484 }
485 else
486 {
487 sLogger.failure("HandleModifyDatabaseResult: Unknown logon result in Method_Account_Change_PW");
488 }
489
490 }break;
491 case Method_Account_Create:
492 {
493 std::string account_name;
494 std::string created_name;
495
496 recvData >> account_name;
497 recvData >> created_name;
498
499 const char* account_string = account_name.c_str();
500 const char* created_string = created_name.c_str();
501
502 if (account_name.compare("none") != 0)
503 {
504 WorldSession* pSession = sWorld.getSessionByAccountName(account_string);
505 if (pSession == nullptr)
506 {
507 sLogger.failure("No session found!");
508 return;
509 }
510
511 if (result_id == Result_Account_Exists)
512 {
513 pSession->SystemMessage("Account name: '%s' already in use!", created_string);
514 }
515 else if (result_id == Result_Account_Finished)
516 {
517 pSession->SystemMessage("Account: '%s' created", created_string);
518 }
519 else
520 {
521 sLogger.failure("HandleModifyDatabaseResult: Unknown logon result in Method_Account_Create");
522 }
523 }
524
525 }break;
526 }

529void LogonCommClientSocket::HandleResultCheckAccount(WorldPacket& recvData)
530{
531 uint8_t result_id;
532 std::string account_name;
533 std::string request_name;
534
535 recvData >> result_id; //Get the result id for further processing
536 recvData >> account_name;
537 recvData >> request_name;
538
539 //transform std::string to const char
540 const char* request_string = request_name.c_str();
541 const char* account_string = account_name.c_str();
542
543 WorldSession* session_name = sWorld.getSessionByAccountName(request_string);
544 if (session_name == nullptr)
545 {
546 if (request_name.compare("none") != 0)
547 {
548 sLogger.failure("Receiver {} not found!", request_string);
549 return;
550 }
551 }
552
553 switch (result_id)
554 {
555 case 1: // Account not available
556 {
557 if (request_name.compare("none") != 0)
558 session_name->SystemMessage("Account: %s not found in database!", account_string);
559 } break;
560 case 2: // No additional data set
561 {
562 if (request_name.compare("none") != 0)
563 session_name->SystemMessage("No gmlevel set for account: %s !", account_string);
564 } break;
565 case 3: // Everything is okay
566 {
567 std::string gmlevel;
568 recvData >> gmlevel;
569
570 uint32_t accountId;
571 recvData >> accountId;
572
573 if (gmlevel.compare("0") != 0)
574 {
575 //Update account_permissions
576 CharacterDatabase.Execute("REPLACE INTO account_permissions (`id`, `permissions`, `name`) VALUES (%u, '%s', '%s')", accountId, gmlevel.c_str(), account_string);
577 if (request_name.compare("none") != 0)
578 session_name->SystemMessage("Account permissions has been updated to '%s' for account '%s' (%u). The change will be effective immediately.", gmlevel.c_str(), account_string, accountId);
579
580 //Update forcedpermission map
581 sLogonCommHandler.setAccountPermission(accountId, gmlevel);
582
583 //Write info to gmlog
584 if (request_name.compare("none") != 0)
585 sGMLog.writefromsession(session_name, "set account %s (%u) permissions to %s", account_string, accountId, gmlevel.c_str());
586
587 //Send information to updated account
588 WorldSession* updated_account_session = sWorld.getSessionByAccountName(account_string);
589 if (updated_account_session != nullptr)
590 {
591 updated_account_session->SystemMessage("Your permissions has been updated! Please reconnect your account.");
592 }
593 }
594 else
595 {
596 //Update account_permissions
597 CharacterDatabase.Execute("DELETE FROM account_permissions WHERE id = %u", accountId);
598 if (request_name.compare("none") != 0)
599 session_name->SystemMessage("Account permissions removed for account '%s' (%u). The change will be effective immediately.", account_string, accountId);
600
601 //Update forcedpermission map
602 sLogonCommHandler.removeAccountPermission(accountId);
603
604 //Write info to gmlog
605 if (request_name.compare("none") != 0)
606 sGMLog.writefromsession(session_name, "removed permissions for account %s (%u)", account_string, accountId);
607
608 //Send information to updated account
609 WorldSession* updated_account_session = sWorld.getSessionByAccountName(account_string);
610 if (updated_account_session != nullptr)
611 {
612 sWorld.disconnectSessionByAccountName(account_string, session_name);
613 }
614 }
615
616 } break;
617 case 4: // Account ID
618 {
619 uint32_t account_id;
620 recvData >> account_id;
621
622 if (request_name.compare("none") != 0)
623 session_name->SystemMessage("Account '%s' has account ID %u.", account_string, account_id);
624 }
625 }
