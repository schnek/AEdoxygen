8//////////////////////////////////////////////////////////////////////////////////////////
9/// \brief Logon Realm communication
10///
11/// These opcodes are used for logon/realm communication. All opcodes used by realm is marked
12/// as C (client). All opcodes used by logon is marked as S (server). Our logonserver is
13/// our communication server, the realm is our communication client.
14///
15/// \note work in progress.
16//////////////////////////////////////////////////////////////////////////////////////////
17enum LogonRealmOpcodes
18{
19 LRCMSG_REALM_REGISTER_REQUEST = 0x001, // try register our realm
20 LRSMSG_REALM_REGISTER_RESULT = 0x002, // register result from logonserver
21 LRCMSG_ACC_SESSION_REQUEST = 0x003,
22 LRSMSG_ACC_SESSION_RESULT = 0x004,
23 LRCMSG_LOGON_PING_STATUS = 0x005, // request logon online
24 LRSMSG_LOGON_PING_RESULT = 0x006, // send result if logon is online
25 LRCMSG_FREE_01 = 0x007, // unused
26 LRSMSG_FREE_02 = 0x008, // unused
27 LRCMSG_AUTH_REQUEST = 0x009, // try authenticate our realm
28 LRSMSG_AUTH_RESPONSE = 0x00A, // authentication result from logonserver
29 LRSMSG_ACC_CHAR_MAPPING_REQUEST = 0x00B,
30 LRCMSG_ACC_CHAR_MAPPING_RESULT = 0x00C,
31 LRCMSG_ACC_CHAR_MAPPING_UPDATE = 0x00D,
32 LRSMSG_SEND_ACCOUNT_DISCONNECT = 0x00E, // send when account is disconnected
33 LRCMSG_LOGIN_CONSOLE_REQUEST = 0x00F,
34 LRSMSG_LOGIN_CONSOLE_RESULT = 0x010,
35 LRCMSG_ACCOUNT_DB_MODIFY_REQUEST = 0x011, // request logon db change
36 LRSMSG_ACCOUNT_DB_MODIFY_RESULT = 0x012,
37 LRSMSG_REALM_POPULATION_REQUEST = 0x013,
38 LRCMSG_REALM_POPULATION_RESULT = 0x014,
39 LRCMSG_ACCOUNT_REQUEST = 0x015, // request account data
40 LRSMSG_ACCOUNT_RESULT = 0x016, // send account information to realm
41 LRCMSG_ALL_ACCOUNT_REQUEST = 0x017, // request all account data
42 LRSMSG_ALL_ACCOUNT_RESULT = 0x018, // send id, name, rank of all accounts to realm
43
44 LRMSG_MAX_OPCODES // max opcodes
45};
