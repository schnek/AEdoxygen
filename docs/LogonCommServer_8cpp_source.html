79void LogonCommServerSocket::OnRead()
80{
81 while (true)
82 {
83 if (!remaining)
84 {
85 if (readBuffer.GetSize() < 6)
86 return; // no header
87
88 // read header
89 readBuffer.Read((uint8_t*)&opcode, 2);
90 readBuffer.Read((uint8_t*)&remaining, 4);
91
92 if (use_crypto)
93 {
94 // decrypt the packet
95 _rwCrypto.process((unsigned char*)&opcode, (unsigned char*)&opcode, 2);
96 _rwCrypto.process((unsigned char*)&remaining, (unsigned char*)&remaining, 4);
97 }
98
99 /* reverse byte order */
100 byteSwapUInt32(&remaining);
101 }
102
103 // do we have a full packet?
104 if (readBuffer.GetSize() < remaining)
105 return;
106
107 // create the buffer
108 WorldPacket buff(opcode, remaining);
109 if (remaining)
110 {
111 buff.resize(remaining);
112 //Read(remaining, (uint8_t*)buff.contents());
113 readBuffer.Read((uint8_t*)buff.contents(), remaining);
114 }
115
116 if (use_crypto && remaining)
117 _rwCrypto.process((unsigned char*)buff.contents(), (unsigned char*)buff.contents(), remaining);
118
119 // handle the packet
120 HandlePacket(buff);
121
122 remaining = 0;
123 opcode = 0;
124 }
125}

127void LogonCommServerSocket::HandlePacket(WorldPacket & recvData)
128{
129 if (authenticated == 0 && recvData.GetOpcode() != LRCMSG_AUTH_REQUEST)
130 {
131 // invalid
132 Disconnect();
133 return;
134 }
135
136 static logonpacket_handler Handlers[LRMSG_MAX_OPCODES] =
137 {
138 NULL,
139 &LogonCommServerSocket::HandleRegister, // LRCMSG_REALM_REGISTER_REQUEST
140 NULL, // LRSMSG_REALM_REGISTER_RESULT
141 &LogonCommServerSocket::HandleSessionRequest, // LRCMSG_ACC_SESSION_REQUEST
142 NULL, // LRSMSG_ACC_SESSION_RESULT
143 &LogonCommServerSocket::HandlePing, // LRCMSG_LOGON_PING_STATUS
144 NULL, // LRSMSG_LOGON_PING_RESULT
145 NULL, // LRCMSG_FREE_01
146 NULL, // LRCMSG_FREE_02
147 &LogonCommServerSocket::HandleAuthChallenge, // LRCMSG_AUTH_REQUEST
148 NULL, // LRSMSG_AUTH_RESPONSE
149 NULL, // LRSMSG_ACC_CHAR_MAPPING_REQUEST
150 &LogonCommServerSocket::HandleMappingReply, // LRCMSG_ACC_CHAR_MAPPING_RESULT
151 &LogonCommServerSocket::HandleUpdateMapping, // LRCMSG_ACC_CHAR_MAPPING_UPDATE
152 NULL, // LRSMSG_SEND_ACCOUNT_DISCONNECT
153 &LogonCommServerSocket::HandleTestConsoleLogin, // LRCMSG_LOGIN_CONSOLE_REQUEST
154 NULL, // LRSMSG_LOGIN_CONSOLE_RESULT
155 &LogonCommServerSocket::HandleDatabaseModify, // LRCMSG_ACCOUNT_DB_MODIFY_REQUEST
156 NULL, // LRSMSG_ACCOUNT_DB_MODIFY_RESULT
157 NULL, // LRSMSG_REALM_POPULATION_REQUEST
158 &LogonCommServerSocket::HandlePopulationRespond, // LRCMSG_REALM_POPULATION_RESULT
159 &LogonCommServerSocket::HandleRequestCheckAccount, // LRCMSG_ACCOUNT_REQUEST
160 NULL, // LRSMSG_ACCOUNT_RESULT
161 &LogonCommServerSocket::HandleRequestAllAccounts, // LRCMSG_ALL_ACCOUNT_REQUEST
162 NULL, // LRSMSG_ALL_ACCOUNT_RESULT
163 };
164
165 if (recvData.GetOpcode() >= LRMSG_MAX_OPCODES || Handlers[recvData.GetOpcode()] == 0)
166 {
167 sLogger.failure("Got unknwon packet from logoncomm: {}", recvData.GetOpcode());
168 return;
169 }
170
171 (this->*(Handlers[recvData.GetOpcode()]))(recvData);
172}

174void LogonCommServerSocket::HandleRegister(WorldPacket & recvData)
175{
176 std::string realmName;
177 uint32_t realmId;
178
179 recvData >> realmId;
180 recvData >> realmName;
181
182 sLogger.info("Registering realm `{}` with ID {}.", realmName, realmId);
183
184 // check Realms if realmId is valid! Otherwise send back error.
185 auto realm = sRealmManager.getRealmById(realmId);
186 if (realm == nullptr)
187 {
188 WorldPacket data(LRSMSG_REALM_REGISTER_RESULT, 4);
189 data << uint32_t(1); // 1 = realm not known by logonserver - failed
190 data << realmId;
191 data << realmName;
192 SendPacket(&data);
193
194 return;
195 }
196
197 realm->name = realmName;
198
199 recvData >> realm->address;
200 recvData >> realm->flags;
201 recvData >> realm->icon;
202 recvData >> realm->timeZone;
203 recvData >> realm->population;
204 recvData >> realm->lock;
205 recvData >> realm->gameBuild;
206
207 sRealmManager.setStatusForRealm(realmId, 1);
208
209 WorldPacket data(LRSMSG_REALM_REGISTER_RESULT, 4);
210 data << uint32_t(0); // 0 = everything ok - success
211 data << realmId;
212 data << realm->name;
213 SendPacket(&data);
214
215 server_ids.insert(realmId);
216
217 // request character mapping for this realm
218 data.Initialize(LRSMSG_ACC_CHAR_MAPPING_REQUEST);
219 data << realmId;
220 SendPacket(&data);
221}

223void LogonCommServerSocket::HandleSessionRequest(WorldPacket & recvData)
224{
225 uint32_t request_id;
226 std::string account_name;
227
228 recvData >> request_id;
229 recvData >> account_name;
230
231 // get sessionkey!
232 uint32_t error = 0;
233 const auto* acct = sAccountMgr.getAccountByName(account_name);
234 if (acct == nullptr || acct->SessionKey == NULL)
235 error = 1; // Unauthorized user.
236
237 // build response packet
238 WorldPacket data(LRSMSG_ACC_SESSION_RESULT, 150);
239 data << request_id;
240 data << error;
241
242 if (!error)
243 {
244 // Append account information.
245 data << acct->AccountId;
246 data << acct->UsernamePtr->c_str();
247 if (!acct->GMFlags)
248 data << uint8_t(0);
249 else
250 data << acct->GMFlags.get();
251
252 data << acct->AccountFlags;
253 data.append(acct->SessionKey.get(), 40);
254 data << acct->forcedLanguage;
255 data << acct->Muted;
256 }
257
258 SendPacket(&data);
259}

273void LogonCommServerSocket::SendPacket(WorldPacket* data)
274{
275 bool rv;
276 BurstBegin();
277
278 LogonWorldPacket header;
279 header.opcode = data->GetOpcode();
280 //header.size = ntohl((u_long)data->size());
281 header.size = (uint32_t)data->size();
282
283 byteSwapUInt32(&header.size);
284
285 if (use_crypto)
286 _sendCrypto.process((unsigned char*)&header, (unsigned char*)&header, 6);
287
288 rv = BurstSend((uint8_t*)&header, 6);
289
290 if (data->size() > 0 && rv)
291 {
292 if (use_crypto)
293 _sendCrypto.process((unsigned char*)data->contents(), (unsigned char*)data->contents(), (uint32_t)data->size());
294
295 rv = BurstSend(data->contents(), (uint32_t)data->size());
296 }
297
298 if (rv) BurstPush();
299 BurstEnd();
300}

302void LogonCommServerSocket::HandleAuthChallenge(WorldPacket & recvData)
303{
304 unsigned char key[20];
305 uint8_t realmId = 0;
306
307 recvData.read(key, 20);
308 recvData >> realmId;
309
310 const auto realm = sRealmManager.getRealmById(realmId);
311 if (realm == nullptr)
312 {
313 sLogger.failure("Realm {} is missing in table realms. Please add the server to your realms table.", static_cast<uint32_t>(realmId));
314 return;
315 }
316
317 Sha1Hash hash;
318 hash.updateData(realm->password);
319 hash.finalize();
320
321 // check if we have the correct password
322 uint32_t result = 1;
323
324 if (memcmp(key, hash.getDigest(), 20) != 0)
325 result = 0;
326
327 sLogger.info("Authentication request from {}, id {} - result {}.", GetRemoteIP(), uint32_t(realmId), result ? "OK" : "FAIL");
328
329 std::stringstream sstext;
330 sstext << "Key: ";
331 char buf[3];
332 for (int i = 0; i < 20; ++i)
333 {
334 snprintf(buf, 3, "%.2X", key[i]);
335 sstext << buf;
336 }
337 //sLogger.info(sstext); FIX fmt
338
339 _rwCrypto.setup(key, 20);
340 _sendCrypto.setup(key, 20);
341
342 // packets are encrypted from now on
343 use_crypto = true;
344
345 // send the response packet
346 WorldPacket data(LRSMSG_AUTH_RESPONSE, 1);
347 data << result;
348 SendPacket(&data);
349
350 // set our general var
351 authenticated = result;
352}

354void LogonCommServerSocket::HandleMappingReply(WorldPacket & recvData)
355{
356 // this packet is gzipped, whee! :D
357 uint32_t real_size;
358 recvData >> real_size;
359 uLongf rsize = real_size;
360
361 ByteBuffer buf(real_size);
362 buf.clear();
363 buf.resize(real_size);
364
365 if (uncompress((uint8_t*)buf.contents(), &rsize, recvData.contents() + 4, (u_long)recvData.size() - 4) != Z_OK)
366 {
367 sLogger.failure("Uncompress of mapping failed.");
368 return;
369 }
370
371 uint32_t account_id;
372 uint8_t number_of_characters;
373 uint32_t count;
374 uint32_t realm_id;
375 buf >> realm_id;
376 auto realm = sRealmManager.getRealmById(realm_id);
377 if (!realm)
378 return;
379
380 sRealmManager.getRealmLock().lock();
381
382 std::unordered_map<uint32_t, uint8_t>::iterator itr;
383 buf >> count;
384 sLogger.info("Got mapping packet for realm {}, total of {} entries.", (unsigned int)realm_id, (unsigned int)count);
385 for (uint32_t i = 0; i < count; ++i)
386 {
387 buf >> account_id >> number_of_characters;
388 itr = realm->_characterMap.find(account_id);
389 if (itr != realm->_characterMap.end())
390 itr->second = number_of_characters;
391 else
392 realm->_characterMap.insert(std::make_pair(account_id, number_of_characters));
393 }
394
395 sRealmManager.getRealmLock().unlock();
396}

665void LogonCommServerSocket::HandleRequestCheckAccount(WorldPacket & recvData)
666{
667 uint32_t method;
668 recvData >> method;
669
670
671 std::string account_name; // account to check
672 std::string request_name; // account request the check
673
674 WorldPacket data(LRSMSG_ACCOUNT_RESULT, 300);
675
676 switch (method)
677 {
678 case 1: // account exist?
679 {
680 std::string additional; // additional data
681
682 recvData >> account_name;
683 recvData >> request_name;
684 recvData >> additional;
685
686 const char* additional_data = additional.c_str();
687
688 std::string account_name_save = account_name; // save original account_name to check
689
690 // remember we expect this in uppercase
691 AscEmu::Util::Strings::toUpperCase(account_name);
692
693 const auto* account_check = sAccountMgr.getAccountByName(account_name);
694 if (account_check == nullptr)
695 {
696 // Send packet "account not available"
697 data << uint8_t(1);
698 data << account_name_save; // requested account
699 data << request_name; // account_name for receive the session
700 }
701 else if (!additional_data)
702 {
703 // Send packet "No additional data set"
704 data << uint8_t(2);
705 data << account_name_save; // requested account
706 data << request_name; // account_name for receive the session
707 }
708 else
709 {
710 // Send packet "everything okay"
711 data << uint8_t(3);
712 data << account_name_save; // requested account
713 data << request_name; // account_name for receive the session
714 data << additional; // additional data
715 data << uint32_t(account_check->AccountId);
716 }
717 } break;
718 case 2: // get account id
719 {
720 recvData >> account_name;
721 recvData >> request_name;
722
723 std::string account_name_save = account_name; // save original account_name to check
724
725 // remember we expect this in uppercase
726 AscEmu::Util::Strings::toUpperCase(account_name);
727
728 const auto* account_check = sAccountMgr.getAccountByName(account_name);
729 if (account_check == nullptr)
730 {
731 // Send packet "account not available"
732 data << uint8_t(1);
733 data << account_name_save; // requested account
734 data << request_name; // account_name for receive the session
735 }
736 else
737 {
738 // Send packet "account id"
739 data << uint8_t(4);
740 data << account_name_save; // requested account
741 data << request_name; // account_name for receive the session
742 data << uint32_t(account_check->AccountId);
743 }
744 } break;
745 }
746
747 SendPacket(&data);
748}
