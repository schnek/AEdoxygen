26//////////////////////////////////////////////////////////////////////////////////////////
27//Handles .lookup achievement
28//\brief GM achievement lookup command usage:
29//.lookup achievement string : searches for "string" in achievement name
30//.lookup achievement desc string : searches for "string" in achievement description
31//.lookup achievement reward string : searches for "string" in achievement reward name
32//.lookup achievement criteria string : searches for "string" in achievement criteria name
33//.lookup achievement all string : searches for "string" in achievement name, description, reward, and critiera
34//////////////////////////////////////////////////////////////////////////////////////////
35bool ChatCommandHandler::HandleLookupAchievementCommand([[maybe_unused]]const char* args, [[maybe_unused]]WorldSession* m_session)
36{
37#if VERSION_STRING > TBC
38 if (!*args)
39 return false;
40
41 std::string x;
42 bool lookupname = true, lookupdesc = false, lookupcriteria = false, lookupreward = false;
43 std::string name(args, 5);
44 std::string desc(args, 5);
45 std::string criteria(args, 9);
46 std::string reward(args, 7);
47 std::string all(args, 4);
48 if (AscEmu::Util::Strings::isEqual(name, "name "))
49 {
50 x = std::string(args + 5);
51 }
52 else if (AscEmu::Util::Strings::isEqual(desc, "desc "))
53 {
54 lookupname = false;
55 lookupdesc = true;
56 x = std::string(args + 5);
57 }
58 else if (AscEmu::Util::Strings::isEqual(criteria, "criteria "))
59 {
60 lookupname = false;
61 lookupcriteria = true;
62 x = std::string(args + 9);
63 }
64 else if (AscEmu::Util::Strings::isEqual(reward, "reward "))
65 {
66 lookupname = false;
67 lookupreward = true;
68 x = std::string(args + 7);
69 }
70 else if (AscEmu::Util::Strings::isEqual(all, "all "))
71 {
72 lookupdesc = true;
73 lookupcriteria = true;
74 lookupreward = true;
75 x = std::string(args + 4);
76 }
77 else
78 {
79 x = std::string(args);
80 }
81
82 if (x.length() < 4)
83 {
84 redSystemMessage(m_session, "Your search string must be at least 4 characters long.");
85 return true;
86 }
87
88 AscEmu::Util::Strings::toLowerCase(x);
89 greenSystemMessage(m_session, "Starting search of achievement `{}`...", x);
90 auto startTime = Util::TimeNow();
91 uint32_t i, j, numFound = 0;
92 std::string y, recout;
93 char playerGUID[17];
94 snprintf(playerGUID, 17, "%llu", m_session->GetPlayer()->getGuid());
95 if (lookupname || lookupdesc || lookupreward)
96 {
97 std::set<uint32_t> foundList;
98 j = sAchievementStore.getNumRows();
99 bool foundmatch;
100 for (i = 0; i < j && numFound < 25; ++i)
101 {
102 auto achievement = sAchievementStore.lookupEntry(i);
103 if (achievement)
104 {
105 if (foundList.find(achievement->ID) != foundList.end())
106 {
107 // already listed this achievement (some achievements have multiple entries in dbc)
108 continue;
109 }
110 foundmatch = false;
111 if (lookupname)
112 {
113#if VERSION_STRING < Cata
114 y = std::string(achievement->name[sWorld.getDbcLocaleLanguageId()]);
115#else
116 y = std::string(achievement->name[0]);
117#endif
118 AscEmu::Util::Strings::toLowerCase(y);
119 foundmatch = AscEmu::Util::Strings::contains(x, y);
120 }
121 if (!foundmatch && lookupdesc)
122 {
123#if VERSION_STRING < Cata
124 y = std::string(achievement->description[sWorld.getDbcLocaleLanguageId()]);
125#else
126 y = std::string(achievement->description[0]);
127#endif
128 AscEmu::Util::Strings::toLowerCase(y);
129 foundmatch = AscEmu::Util::Strings::contains(x, y);
130 }
131 if (!foundmatch && lookupreward)
132 {
133#if VERSION_STRING < Cata
134 y = std::string(achievement->rewardName[sWorld.getDbcLocaleLanguageId()]);
135#else
136 y = std::string(achievement->rewardName[0]);
137#endif
138 AscEmu::Util::Strings::toLowerCase(y);
139 foundmatch = AscEmu::Util::Strings::contains(x, y);
140 }
141 if (!foundmatch)
142 {
143 continue;
144 }
145 foundList.insert(achievement->ID);
146 std::stringstream strm;
147 strm << achievement->ID;
148 // create achievement link
149 recout = "|cffffffffAchievement ";
150 recout += strm.str();
151 recout += ": |cfffff000|Hachievement:";
152 recout += strm.str();
153 recout += ":";
154 recout += playerGUID;
155 time_t completetime = m_session->GetPlayer()->getAchievementMgr()->getCompletedTime(achievement);
156 if (completetime)
157 {
158 // achievement is completed
159 struct tm* ct;
160 ct = localtime(&completetime);
161 strm.str("");
162 strm << ":1:" << ct->tm_mon + 1 << ":" << ct->tm_mday << ":" << ct->tm_year - 100 << ":-1:-1:-1:-1|h[";
163 recout += strm.str();
164 }
165 else
166 {
167 // achievement is not completed
168 recout += ":0:0:0:-1:0:0:0:0|h[";
169 }
170#if VERSION_STRING < Cata
171 recout += achievement->name[sWorld.getDbcLocaleLanguageId()];
172#else
173 recout += achievement->name[0];
174#endif
175 if (!lookupreward)
176 {
177 recout += "]|h|r";
178 }
179 else
180 {
181 recout += "]|h |cffffffff";
182#if VERSION_STRING < Cata
183 recout += achievement->rewardName[sWorld.getDbcLocaleLanguageId()];
184#else
185 recout += achievement->rewardName[0];
186#endif
187 recout += "|r";
188 }
189 strm.str("");
190 SendMultilineMessage(m_session, recout.c_str());
191 if (++numFound >= 25)
192 {
193 redSystemMessage(m_session, "More than 25 results found.");
194 break;
195 }
196 }
197 } // for loop (number of rows, up to 25)
198 } // lookup name or description
199 if (lookupcriteria && numFound < 25)
200 {
201 std::set<uint32_t> foundList;
202 j = sAchievementCriteriaStore.getNumRows();
203 for (i = 0; i < j && numFound < 25; ++i)
204 {
205 auto criteria = sAchievementCriteriaStore.lookupEntry(i);
206 if (criteria)
207 {
208 if (foundList.find(criteria->ID) != foundList.end())
209 {
210 // already listed this achievement (some achievements have multiple entries in dbc)
211 continue;
212 }
213#if VERSION_STRING < Cata
214 y = std::string(criteria->name[sWorld.getDbcLocaleLanguageId()]);
215#else
216 y = std::string(criteria->name[0]);
217#endif
218 AscEmu::Util::Strings::toLowerCase(y);
219 if (AscEmu::Util::Strings::contains(x, y) == false)
220 {
221 continue;
222 }
223
224 foundList.insert(criteria->ID);
225 std::stringstream strm;
226 strm << criteria->ID;
227 recout = "|cffffffffCriteria ";
228 recout += strm.str();
229 recout += ": |cfffff000";
230#if VERSION_STRING < Cata
231 recout += criteria->name[sWorld.getDbcLocaleLanguageId()];
232#else
233 recout += criteria->name[0];
234#endif
235 strm.str("");
236 auto achievement = sAchievementStore.lookupEntry(criteria->referredAchievement);
237 if (achievement)
238 {
239 // create achievement link
240 recout += " |cffffffffAchievement ";
241 strm << achievement->ID;
242 recout += strm.str();
243 recout += ": |cfffff000|Hachievement:";
244 recout += strm.str();
245 recout += ":";
246 recout += playerGUID;
247 time_t completetime = m_session->GetPlayer()->getAchievementMgr()->getCompletedTime(achievement);
248 if (completetime)
249 {
250 // achievement is completed
251 struct tm* ct;
252 ct = localtime(&completetime);
253 strm.str("");
254 strm << ":1:" << ct->tm_mon + 1 << ":" << ct->tm_mday << ":" << ct->tm_year - 100 << ":-1:-1:-1:-1|h[";
255 recout += strm.str();
256 }
257 else
258 {
259 // achievement is not completed
260 recout += ":0:0:0:-1:0:0:0:0|h[";
261 }
262#if VERSION_STRING < Cata
263 recout += achievement->name[sWorld.getDbcLocaleLanguageId()];
264#else
265 recout += achievement->name[0];
266#endif
267 if (!lookupreward)
268 {
269 recout += "]|h|r";
270 }
271 else
272 {
273 recout += "]|h |cffffffff";
274#if VERSION_STRING < Cata
275 recout += achievement->rewardName[sWorld.getDbcLocaleLanguageId()];
276#else
277 recout += achievement->rewardName[0];
278#endif
279 recout += "|r";
280 }
281 strm.str("");
282 }
283 SendMultilineMessage(m_session, recout.c_str());
284 if (++numFound >= 25)
285 {
286 redSystemMessage(m_session, "More than 25 results found.");
287 break;
288 }
289 }
290 } // for loop (number of rows, up to 25)
291 } // lookup criteria
292 if (numFound == 0)
293 {
294 recout = "|cff00ccffNo matches found.";
295 SendMultilineMessage(m_session, recout.c_str());
296 }
297 blueSystemMessage(m_session, "Search completed in {} ms.", static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)));
298#endif
299 return true;
300}

305 if (!*args)
306 return false;
307
308 std::string x = std::string(args);
309 AscEmu::Util::Strings::toLowerCase(x);
310 if (x.length() < 4)
311 {
312 redSystemMessage(m_session, "Your search string must be at least 4 characters long.");
313 return true;
314 }
315
316 blueSystemMessage(m_session, "Starting search of creature `{}`...", x);
317 auto startTime = Util::TimeNow();
318
319 uint32_t count = 0;
320
321 MySQLDataStore::CreaturePropertiesContainer const* its = sMySQLStore.getCreaturePropertiesStore();
322 for (MySQLDataStore::CreaturePropertiesContainer::const_iterator itr = its->begin(); itr != its->end(); ++itr)
323 {
324 CreatureProperties const* it = sMySQLStore.getCreatureProperties(itr->second.Id);
325 if (it != nullptr)
326 {
327 MySQLStructure::LocalesCreature const* lit = (m_session->language > 0) ? sMySQLStore.getLocalizedCreature(it->Id, m_session->language) : nullptr;
328
329 std::string litName = std::string( lit ? lit->name : "");
330
331 AscEmu::Util::Strings::toLowerCase(litName);
332
333 bool localizedFound = false;
334 if (AscEmu::Util::Strings::contains(x, litName))
335 localizedFound = true;
336
337 std::string names_lower = it->lowercase_name;
338 if (AscEmu::Util::Strings::contains(x, names_lower) || localizedFound)
339 {
340 systemMessage(m_session, "ID: {} {}{}", it->Id, MSG_COLOR_YELLOW, it->Name);
341 ++count;
342
343 if (count == 25)
344 {
345 redSystemMessage(m_session, "More than 25 results returned. aborting.");
346 break;
347 }
348 }
349 }
350 }
351
352 if (count == 0)
353 redSystemMessage(m_session, "No results returned. aborting.");
354
355 blueSystemMessage(m_session, "Search completed in {} ms.", static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)));
356 return true;

408void ChatCommandHandler::sendItemLinkToPlayer(ItemProperties const* iProto, WorldSession* pSession, bool ItemCount, Player* owner, uint32_t language)
409{
410 if (!iProto || !pSession)
411 return;
412
413 if (ItemCount && owner == NULL)
414 return;
415
416 if (ItemCount)
417 {
418 int8_t count = static_cast<int8_t>(owner->getItemInterface()->GetItemCount(iProto->ItemId, true));
419 //int8_t slot = owner->getItemInterface()->GetInventorySlotById(iProto->ItemId); //DISABLED due to being a retarded concept
420 if (iProto->ContainerSlots > 0)
421 systemMessage(pSession, "Item {} {} Count {} ContainerSlots {}", iProto->ItemId, sMySQLStore.getItemLinkByProto(iProto, language), count, iProto->ContainerSlots);
422 else
423 systemMessage(pSession, "Item {} {} Count {}", iProto->ItemId, sMySQLStore.getItemLinkByProto(iProto, language), count);
424 }
425 else
426 {
427 if (iProto->ContainerSlots > 0)
428 systemMessage(pSession, "Item {} {} ContainerSlots {}", iProto->ItemId, sMySQLStore.getItemLinkByProto(iProto, language), iProto->ContainerSlots);
429 else
430 systemMessage(pSession, "Item {} {}", iProto->ItemId, sMySQLStore.getItemLinkByProto(iProto, language));
431 }
432}

547bool ChatCommandHandler::HandleLookupQuestCommand(const char* args, WorldSession* m_session)
548{
549 if (!*args)
550 return false;
551
552 std::string search_string = std::string(args);
553 AscEmu::Util::Strings::toLowerCase(search_string);
554 if (search_string.length() < 4)
555 {
556 redSystemMessage(m_session, "Your search string must be at least 4 characters long.");
557 return true;
558 }
559
560 blueSystemMessage(m_session, "Starting search of quests `{}`...", search_string.c_str());
561 auto startTime = Util::TimeNow();
562 std::string recout;
563 uint32_t count = 0;
564
565 MySQLDataStore::QuestPropertiesContainer const* its = sMySQLStore.getQuestPropertiesStore();
566 for (MySQLDataStore::QuestPropertiesContainer::const_iterator itr = its->begin(); itr != its->end(); ++itr)
567 {
568 QuestProperties const* quest = sMySQLStore.getQuestProperties(itr->second.id);
569 if (quest == nullptr)
570 continue;
571
572 std::string lower_quest_title = quest->title;
573
574 MySQLStructure::LocalesQuest const* li = (m_session->language > 0) ? sMySQLStore.getLocalizedQuest(quest->id, m_session->language) : nullptr;
575
576 std::string liName = std::string(li ? li->title : "");
577
578 AscEmu::Util::Strings::toLowerCase(liName);
579 AscEmu::Util::Strings::toLowerCase(lower_quest_title);
580
581 bool localizedFound = false;
582 if (AscEmu::Util::Strings::contains(search_string, liName))
583 localizedFound = true;
584
585 if (AscEmu::Util::Strings::contains(search_string, lower_quest_title) || localizedFound)
586 {
587 std::string questid = std::to_string(quest->id);
588 std::string questtitle = localizedFound ? (li ? li->title : "") : quest->title;
589 // send quest link
590 recout = questid;
591 recout += ": |cff00ccff|Hquest:";
592 recout += questid;
593 recout += ":";
594 recout += std::to_string(quest->min_level);
595 recout += "|h[";
596 recout += questtitle;
597 recout += "]|h|r";
598 SendMultilineMessage(m_session, recout.c_str());
599
600 ++count;
601 if (count == 25)
602 {
603 redSystemMessage(m_session, "More than 25 results returned. aborting.");
604 break;
605 }
606 }
607 }
608
609 if (count == 0)
610 {
611 recout = "|cff00ccffNo matches found.\n\n";
612 SendMultilineMessage(m_session, recout.c_str());
613 }
614
615 blueSystemMessage(m_session, "Search completed in {} ms.", static_cast<uint32_t>(Util::GetTimeDifferenceToNow(startTime)));

618}
