60bool LootItem::isAllowedForPlayer(Player const* player) const
61{
62 if (itemproto == nullptr)
63 return false;
64
65 // not show loot for players without profession or those who already know the recipe
66 if (itemproto->Flags & ITEM_FLAG_SMART_LOOT && (!player->hasSkillLine(itemproto->RequiredSkill) || player->hasSpell(itemproto->Spells[1].Id)))
67 return false;
68
69 // not show loot for not own team
70 if (itemproto->Flags2 & ITEM_FLAG2_HORDE_ONLY && player->GetTeam() != TEAM_HORDE)
71 return false;
72 if (itemproto->Flags2 & ITEM_FLAG2_ALLIANCE_ONLY && player->GetTeam() != TEAM_ALLIANCE)
73 return false;
74
75 // If item starts quest do not show item if player is already on the quest
76 // or if player has finished the quest
77 if (starts_quest)
78 {
79 if (player->hasQuestInQuestLog(itemproto->QuestId) || player->hasQuestFinished(itemproto->QuestId))
80 return false;
81 }
82 // or if item requires a quest player needs to have the quest active
83 else if (needs_quest)
84 {
85 if (!player->hasQuestForItem(itemId))
86 return false;
87 }
88
89 return true;
90}
