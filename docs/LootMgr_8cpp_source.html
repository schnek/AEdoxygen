18template <class T> // works for anything that has the field 'chance' and is stored in plain array
19T* randomChoiceVector(std::vector<std::pair<T*, float>> const& variant)
20{
21 typename std::vector<std::pair<T*, float>>::const_iterator itr;
22
23 if (variant.empty())
24 return nullptr;
25
26 float totalChance = 0.0f;
27 for (itr = variant.cbegin(); itr != variant.cend(); ++itr)
28 totalChance += itr->second;
29
30 auto val = Util::getRandomFloat(totalChance);
31 for (itr = variant.cbegin(); itr != variant.cend(); ++itr)
32 {
33 val -= itr->second;
34 if (val <= 0)
35 return itr->first;
36 }
37
38 // should not come here, buf if it does, we should return something reasonable
39 return variant.cbegin()->first;
40}

134void LootMgr::loadLootProp()
135{
136 auto result = WorldDatabase.Query("SELECT * FROM item_randomprop_groups");
137 if (result != nullptr)
138 {
139 do
140 {
141 uint32_t id = result->Fetch()[0].asUint32();
142 uint32_t eid = result->Fetch()[1].asUint32();
143 float ch = result->Fetch()[2].asFloat();
144 auto item_random_properties = sItemRandomPropertiesStore.lookupEntry(eid);
145 if (item_random_properties == nullptr)
146 {
147 sLogger.failure("LootMgr::loadLootProp : RandomProperty group {} references non-existent randomproperty {}.", id, eid);
148 continue;
149 }
150
151 const auto itr = _randomprops.find(id);
152 if (itr == _randomprops.cend())
153 {
154 RandomPropertyVector v;
155 v.push_back(std::make_pair(item_random_properties, ch));
156 _randomprops.insert(std::make_pair(id, v));
157 }
158 else
159 {
160 itr->second.push_back(std::make_pair(item_random_properties, ch));
161 }
162 } while (result->NextRow());
163 }
164
165 result = WorldDatabase.Query("SELECT * FROM item_randomsuffix_groups");
166 if (result != nullptr)
167 {
168 do
169 {
170 uint32_t id = result->Fetch()[0].asUint32();
171 uint32_t eid = result->Fetch()[1].asUint32();
172 float ch = result->Fetch()[2].asFloat();
173 auto item_random_suffix = sItemRandomSuffixStore.lookupEntry(eid);
174 if (item_random_suffix == nullptr)
175 {
176 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "LootMgr::loadLootProp : RandomSuffix group {} references non-existent randomsuffix {}.", id, eid);
177 continue;
178 }
179
180 auto itr = _randomsuffix.find(id);
181 if (itr == _randomsuffix.cend())
182 {
183 RandomSuffixVector v;
184 v.push_back(std::make_pair(item_random_suffix, ch));
185 _randomsuffix.insert(std::make_pair(id, v));
186 }
187 else
188 {
189 itr->second.push_back(std::make_pair(item_random_suffix, ch));
190 }
191 } while (result->NextRow());
192 }
193}

195void LootMgr::loadLootTables(std::string const& szTableName, LootTemplateMap* LootTable)
196{
197 auto result = sMySQLStore.getWorldDBQuery("SELECT * FROM %s ORDER BY entryid ASC", szTableName.c_str());
198 if (result == nullptr)
199 {
200 sLogger.failure("LootMgr::loadLootTables : Loading loot from table {} failed.", szTableName);
201 return;
202 }
203
204 LootTemplateMap::const_iterator tab;
205 uint32_t count = 0;
206 do
207 {
208 Field* fields = result->Fetch();
209
210 std::vector<float> chance;
211 chance.reserve(4);
212
213 uint32_t entry = fields[0].asUint32();
214 uint32_t itemId = fields[1].asUint32();
215 chance.push_back(fields[2].asFloat());
216 chance.push_back(fields[3].asFloat());
217 chance.push_back(fields[4].asFloat());
218 chance.push_back(fields[5].asFloat());
219 uint32_t mincount = fields[6].asUint32();
220 uint32_t maxcount = fields[7].asUint8();
221
222 const auto itemProto = sMySQLStore.getItemProperties(itemId);
223 if (itemProto == nullptr)
224 {
225 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "LootMgr::loadLootTables : Invalid Item with entry {} set in {}", itemId, szTableName);
226 continue;
227 }
228
229 LootStoreItem storeitem = LootStoreItem(itemProto, chance, mincount, maxcount);
230
231 // Looking for the template of the entry
232 if (LootTable->empty() || tab->first != entry)
233 {
234 // Searching the template (in case template Id changed)
235 const auto [tabItr, _] = LootTable->try_emplace(entry, Util::LazyInstanceCreator([] {
236 return std::make_unique<LootTemplate>();
237 }));
238 tab = tabItr;
239 }
240
241 // Add Item to our Tempelate
242 tab->second->addEntry(storeitem);
243 count++;
244 } while (result->NextRow());
245
246 sLogger.info("{} loot templates loaded from {}", count, szTableName);
247}

274void LootMgr::addLoot(Loot* loot, uint32_t itemid, std::vector<float> chance, uint32_t mincount, uint32_t maxcount, uint8_t lootDifficulty)
275{
276 if (loot->items.size() > 16)
277 {
278 sLogger.debug("LootMgr::addLoot cannot add item {} to Loot, Maximum drops reached", itemid);
279 return;
280 }
281
282 const auto itemprop = sMySQLStore.getItemProperties(itemid);
283 if (itemprop == nullptr)
284 return;
285
286 LootStoreItem item = LootStoreItem(itemprop, chance, mincount, maxcount);
287
288 // check difficulty level
289 if (item.chance[lootDifficulty] < 0.0f)
290 {
291 return;
292 }
293
294 // Bad luck for the entry
295 if (!item.roll(0))
296 {
297 return;
298 }
299
300 // add the Item to our Loot List
301 loot->addLootItem(item);
302}
