38void LootRoll::finalize()
39{
40 sEventMgr.RemoveEvents(this);
41 // this we will have to finalize with groups types.. for now
42 // we'll just assume need before greed. person with highest roll
43 // in need gets the item.
44 uint8_t highest = 0;
45 int8_t hightype = -1;
46 uint32_t playerLowGuid = 0;
47
48 for (const auto& [lowPlrGuid, roll] : m_NeedRolls)
49 {
50 if (roll > highest)
51 {
52 highest = roll;
53 playerLowGuid = lowPlrGuid;
54 hightype = ROLL_NEED;
55 }
56 }
57
58 if (highest == 0)
59 {
60 for (const auto& [lowPlrGuid, roll] : m_GreedRolls)
61 {
62 if (roll > highest)
63 {
64 highest = roll;
65 playerLowGuid = lowPlrGuid;
66 hightype = ROLL_GREED;
67 }
68 }
69 }
70
71 Creature* creature = nullptr;
72 GameObject* gameObject = nullptr;
73
74 Loot* pLoot = nullptr;
75 WoWGuid wowGuid;
76 wowGuid.Init(_guid);
77
78 if (wowGuid.isUnit())
79 {
80 if (creature = _mgr->getCreature(wowGuid.getGuidLowPart()))
81 pLoot = &creature->loot;
82 }
83 else if (wowGuid.isGameObject())
84 {
85 if (gameObject = _mgr->getGameObject(wowGuid.getGuidLowPart()))
86 {
87 if (gameObject->IsLootable())
88 {
89 GameObject_Lootable* pLGO = static_cast<GameObject_Lootable*>(gameObject);
90 pLoot = &pLGO->loot;
91 }
92 }
93 }
94
95 if (pLoot == nullptr)
96 {
97 return;
98 }
99
100 if (_slotid >= pLoot->items.size())
101 {
102 return;
103 }
104
105 const auto amt = pLoot->items.at(_slotid).count;
106 if (amt == 0)
107 {
108 pLoot->items.at(_slotid).roll = nullptr;
109 return;
110 }
111
112 Player* _player = playerLowGuid != 0 ? _mgr->getPlayer(playerLowGuid) : nullptr;
113 if (_player == nullptr)
114 {
115 /* all passed */
116 auto pitr = m_passRolls.cbegin();
117 while (_player == nullptr && pitr != m_passRolls.cend())
118 _player = _mgr->getPlayer((*(pitr++)));
119
120 if (_player != nullptr)
121 {
122 if (_player->isInGroup())
123 _player->getGroup()->SendPacketToAll(SmsgLootAllPassed(_guid, _groupcount, _itemid, _randomsuffixid, _randompropertyid).serialise().get());
124 else
125 _player->getSession()->SendPacket(SmsgLootAllPassed(_guid, _groupcount, _itemid, _randomsuffixid, _randompropertyid).serialise().get());
126 }
127
128 /* item can now be looted by anyone :) */
129 pLoot->items.at(_slotid).is_passed = true;
130 pLoot->items.at(_slotid).is_blocked = false;
131 pLoot->items.at(_slotid).roll = nullptr;
132 return;
133 }
134
135 if (_player->isInGroup())
136 _player->getGroup()->SendPacketToAll(SmsgLootRollWon(_guid, _slotid, _itemid, _randomsuffixid, _randompropertyid, _player->getGuid(), highest, hightype).serialise().get());
137 else
138 _player->getSession()->SendPacket(SmsgLootRollWon(_guid, _slotid, _itemid, _randomsuffixid, _randompropertyid, _player->getGuid(), highest, hightype).serialise().get());
139
140 LootItem const& item = _slotid >= pLoot->items.size() ? pLoot->quest_items[_slotid - pLoot->items.size()] : pLoot->items[_slotid];
141
142 // Add Item to Player
143 Item* newItem = _player->storeItem(&item);
144 if (newItem == nullptr)
145 return;
146
147 if (creature != nullptr)
148 {
149 if (creature->IsInWorld() && creature->GetScript())
150 creature->GetScript()->OnLootTaken(_player, item.itemproto);
151 }
152 else if (gameObject != nullptr)
153 {
154 if (gameObject->GetScript())
155 gameObject->GetScript()->OnLootTaken(_player, item.itemproto);
156 }
157 // mark as looted
158 pLoot->items.at(_slotid).count = 0;
159 pLoot->items.at(_slotid).is_looted = true;
160
161 pLoot->itemRemoved(_slotid);
162 --pLoot->unlootedCount;
163 pLoot->items.at(_slotid).roll = nullptr;
164}

166bool LootRoll::playerRolled(Player* player, uint8_t choice)
167{
168 // don't allow cheaters
169 if (m_NeedRolls.find(player->getGuidLow()) != m_NeedRolls.cend() || m_GreedRolls.find(player->getGuidLow()) != m_GreedRolls.cend())
170 return false;
171
172 auto roll = static_cast<uint8_t>(Util::getRandomUInt(99) + 1);
173 switch (choice)
174 {
175 case ROLL_PASS:
176 m_passRolls.insert(player->getGuidLow());
177 roll = 128;
178 break;
179 case ROLL_GREED:
180 m_GreedRolls.insert({ player->getGuidLow(), roll });
181 break;
182 case ROLL_NEED:
183 m_NeedRolls.insert({ player->getGuidLow(), roll });
184 break;
185 case ROLL_DISENCHANT:
186 roll = 128;
187 break;
188 }
189
190 if (player->isInGroup())
191 player->getGroup()->SendPacketToAll(SmsgLootRoll(_guid, _slotid, player->getGuid(), _itemid, _randomsuffixid, _randompropertyid, roll, choice).serialise().get());
192 else
193 player->getSession()->SendPacket(SmsgLootRoll(_guid, _slotid, player->getGuid(), _itemid, _randomsuffixid, _randompropertyid, roll, choice).serialise().get());
194
195 // check for early completion
196 --_remaining;
197 if (_remaining == 0)
198 {
199 // kill event early
200 finalize();
201 return true;
202 }
203
204 return false;
205}
