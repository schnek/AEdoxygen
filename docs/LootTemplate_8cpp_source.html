22void LootTemplate::generateLoot(Loot& loot, uint8_t lootDifficulty) const
23{
24 // Randomize our Loot
25 LootStoreItemList lootEntries = Entries;
26 Util::randomShuffleVector(lootEntries);
27
28 // Rolling items
29 for (const auto& lootStoreItem : lootEntries)
30 {
31 // check difficulty level
32 if (lootStoreItem.chance[lootDifficulty] < 0.0f)
33 continue;
34
35 // Bad luck for the entry
36 if (!lootStoreItem.roll(lootDifficulty))
37 continue;
38
39 if (lootStoreItem.itemproto != nullptr)
40 {
41 uint8_t item_counterEquipable = 0;
42 uint8_t item_counternotEquipable = 0;
43 for (auto _item = loot.items.cbegin(); _item != loot.items.cend(); ++_item)
44 {
45 // Non-equippable items are limited to 3 drops
46 if (lootStoreItem.itemproto->InventoryType == INVTYPE_NON_EQUIP && item_counternotEquipable < MAX_NR_LOOT_NONEQUIPABLE)
47 item_counternotEquipable++;
48 // Equippable item are limited to 2 drops
49 else if (lootStoreItem.itemproto->InventoryType != INVTYPE_NON_EQUIP && item_counterEquipable < MAX_NR_LOOT_EQUIPABLE)
50 item_counterEquipable++;
51 }
52
53 // Stopp adding Items
54 if ((item_counterEquipable == MAX_NR_LOOT_EQUIPABLE && lootStoreItem.itemproto->InventoryType != INVTYPE_NON_EQUIP) ||
55 (item_counternotEquipable == MAX_NR_LOOT_NONEQUIPABLE && lootStoreItem.itemproto->InventoryType == INVTYPE_NON_EQUIP))
56 continue;
57 }
58
59 // add the item to our Loot list
60 loot.addLootItem(lootStoreItem);
61 }
62}
