30bool Loot::fillLoot(uint32_t lootId, LootTemplateMap const& tempelateStore, Player* lootOwner, bool personal, uint8_t lootMode /*= InstanceDifficulty::DUNGEON_NORMAL*/)
31{
32 // Must be provided
33 if (lootOwner == nullptr)
34 return false;
35
36 auto temp = tempelateStore.find(lootId);
37 if (temp == tempelateStore.cend())
38 return false;
39
40 const auto& tempelate = temp->second;
41
42 items.reserve(MAX_NR_LOOT_ITEMS);
43 quest_items.reserve(MAX_NR_LOOT_QUESTITEMS);
44
45 tempelate->generateLoot(*this, lootMode);
46
47 // If Player is in a Group add Personal loot to them
48 const auto group = lootOwner->getGroup();
49 if (!personal && group != nullptr)
50 {
51 // Player allowed to Loot this Round
52 roundRobinPlayer = lootOwner->getGuid();
53
54 for (uint8_t i = 0; i < group->GetSubGroupCount(); ++i)
55 {
56 if (group->GetSubGroup(i) != nullptr)
57 {
58 for (const auto& itr : group->GetSubGroup(i)->getGroupMembers())
59 {
60 if (auto* const loggedInPlayer = sObjectMgr.getPlayer(itr->guid))
61 fillNotNormalLootFor(loggedInPlayer, loggedInPlayer->isAtGroupRewardDistance(lootOwner));
62 }
63 }
64 }
65
66 for (uint8_t i = 0; i < items.size(); ++i)
67 {
68 if (items[i].itemproto)
69 if (items[i].itemproto->Quality < group->GetThreshold())
70 items[i].is_underthreshold = true;
71 }
72 }
73 // ... for personal loot
74 else
75 {
76 fillNotNormalLootFor(lootOwner, true);
77 }
78
79 return true;
80}

82void Loot::generateGold(CreatureProperties const* property, uint8_t difficulty)
83{
84 uint32_t amount = 0;
85
86 // Base Gold
87 amount = property->money;
88
89 // Difficulty Gold
90 if (difficulty != 0)
91 {
92 uint32_t creature_difficulty_entry = sMySQLStore.getCreatureDifficulty(property->Id, difficulty);
93 if (const auto properties_difficulty = sMySQLStore.getCreatureProperties(creature_difficulty_entry))
94 {
95 if (properties_difficulty->money != property->money)
96 amount = properties_difficulty->money;
97 }
98 }
99
100 // Gold rates
101 amount = static_cast<uint32_t>(amount * worldConfig.getFloatRate(RATE_MONEY));
102
103 if (amount)
104 gold = amount;
105}

114 else if (items.size() < MAX_NR_LOOT_ITEMS) // Non-quest drop

121 ++unlootedCount;

137LootItem* Loot::lootItemInSlot(uint32_t lootSlot, Player const* player, Personaltem* *qitem, Personaltem* *ffaitem)
138{
139 LootItem* item = nullptr;
140 bool is_looted = true;
141 if (lootSlot >= items.size())
142 {
143 const uint32_t questSlot = lootSlot - static_cast<uint32_t>(items.size());
144 const auto itr = PlayerQuestItems.find(player->getGuidLow());
145 if (itr != PlayerQuestItems.cend() && questSlot < itr->second->size())
146 {
147 Personaltem* qitem2 = &itr->second->at(questSlot);
148 if (qitem != nullptr)
149 *qitem = qitem2;
150 item = &quest_items[qitem2->index];
151 is_looted = qitem2->is_looted;
152 }
153 }
154 else
155 {
156 item = &items[lootSlot];
157 is_looted = item->is_looted;
158 if (item->is_ffa)
159 {
160 const auto itr = PlayerFFAItems.find(player->getGuidLow());
161 if (itr != PlayerFFAItems.cend())
162 {
163 for (auto iter = itr->second->begin(); iter != itr->second->cend(); ++iter)
164 {
165 if (iter->index == lootSlot)
166 {
167 Personaltem* ffaitem2 = &(*iter);
168 if (ffaitem != nullptr)
169 *ffaitem = ffaitem2;
170 is_looted = ffaitem2->is_looted;
171 break;
172 }
173 }
174 }
175 }
176 }
177
178 if (is_looted)
179 return nullptr;
180
181 return item;
182}

211bool Loot::hasItemFor(Player const* player) const
212{
213 // Personal Quest Items
214 auto itr = PlayerQuestItems.find(player->getGuidLow());
215 if (itr != PlayerQuestItems.cend())
216 {
217 const auto& quest_list = itr->second;
218 for (auto personalItem = quest_list->cbegin(); personalItem != quest_list->cend(); ++personalItem)
219 {
220 LootItem const& item = quest_items[personalItem->index];
221 if (!personalItem->is_looted && !item.is_looted && item.isAllowedForPlayer(player))
222 return true;
223 }
224 }
225
226 // Personal Free for all Items
227 itr = PlayerFFAItems.find(player->getGuidLow());
228 if (itr != PlayerFFAItems.cend())
229 {
230 const auto& ffa_list = itr->second;
231 for (auto personalFfaItem = ffa_list->cbegin(); personalFfaItem != ffa_list->cend(); ++personalFfaItem)
232 {
233 LootItem const& item = items[personalFfaItem->index];
234 if (!personalFfaItem->is_looted && !item.is_looted && item.isAllowedForPlayer(player))
235 return true;
236 }
237 }
238
239 return false;
240}

282void Loot::fillNotNormalLootFor(Player* player, bool presentAtLooting)
283{
284 uint32_t plguid = player->getGuidLow();
285
286 // Add Quest Items
287 auto personaltem = std::as_const(PlayerQuestItems).find(plguid);
288 if (personaltem == PlayerQuestItems.cend())
289 {
290 if (items.size() == MAX_NR_LOOT_ITEMS)
291 return;
292
293 auto personalList = std::make_unique<PersonaltemList>();
294
295 for (uint8_t i = 0; i < quest_items.size(); ++i)
296 {
297 LootItem& item = quest_items[i];
298
299 if (item.is_looted)
300 continue;
301
302 if (item.isAllowedForPlayer(player) || (player->getGroup() && ((player->getGroup()->GetMethod() == PARTY_LOOT_MASTER_LOOTER && player->getGroup()->GetLooter() && player->getGroup()->GetLooter()->guid == player->getGuidLow()) || player->getGroup()->GetMethod() != PARTY_LOOT_MASTER_LOOTER)))
303 {
304 personalList->push_back(Personaltem(i));
305
306 // increase once if one looter only, looter-times if free for all
307 if (item.is_ffa || !item.is_blocked)
308 ++unlootedCount;
309 if (!player->getGroup() || (player->getGroup()->GetMethod() != PARTY_LOOT_GROUP && player->getGroup()->GetMethod() != PARTY_LOOT_ROUND_ROBIN))
310 item.is_blocked = true;
311
312 if ((items.size() + personalList->size()) >= MAX_NR_LOOT_ITEMS)
313 break;
314 }
315 }
316
317 if (!personalList->empty())
318 PlayerQuestItems[plguid] = std::move(personalList);
319 }
320
321 // Add Free For All Items
322 personaltem = std::as_const(PlayerFFAItems).find(plguid);
323 if (personaltem == PlayerFFAItems.cend())
324 {
325 auto personalList = std::make_unique<PersonaltemList>();
326
327 for (uint8_t i = 0; i < items.size(); ++i)
328 {
329 LootItem const& item = items[i];
330 if (!item.is_looted && item.is_ffa && item.isAllowedForPlayer(player))
331 {
332 personalList->push_back(Personaltem(i));
333 ++unlootedCount;
334 }
335 }
336
337 if (!personalList->empty())
338 PlayerFFAItems[plguid] = std::move(personalList);
339 }
340
341 // Add NonQuest and Non FFA Items
342 for (uint8_t i = 0; i < items.size(); ++i)
343 {
344 LootItem& item = items[i];
345 if (!item.is_looted && !item.is_ffa && (item.isAllowedForPlayer(player) ||
346 (player->getGroup() && ((player->getGroup()->GetMethod() == PARTY_LOOT_MASTER_LOOTER &&
347 player->getGroup()->GetLooter()->guid == player->getGuidLow()) ||
348 player->getGroup()->GetMethod() != PARTY_LOOT_MASTER_LOOTER))))
349 {
350 if (presentAtLooting)
351 item.addAllowedLooter(player);
352 }
353 }
354
355 // if not present at looting player has to pick it up manually
356 if (!presentAtLooting)
357 return;
358
359 // Process currency items
360 processCurrencyItems(player);
361}

363void Loot::processCurrencyItems(Player* player)
364{
365 const auto max_slot = getMaxSlotInLootFor(player);
366 const auto itemsCount = static_cast<uint8_t>(items.size());
367
368 for (uint8_t i = 0; i < max_slot; ++i)
369 {
370 LootItem const* item = nullptr;
371 if (i < itemsCount)
372 {
373 item = &items[i];
374 }
375 else
376 {
377 const uint8_t questIndex = i - itemsCount;
378 if (questIndex < quest_items.size())
379 item = &quest_items[questIndex];
380 }
381
382 if (item != nullptr && !item->is_looted && item->is_ffa && item->isAllowedForPlayer(player))
383 if (item->itemproto != nullptr && item->itemproto->isCurrencyToken())
384 player->storeNewLootItem(i, this);
385 }
