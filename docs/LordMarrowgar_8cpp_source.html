91void LordMarrowgarAI::AIUpdate(unsigned long time_passed)
92{
93 if (!_isInCombat())
94 return;
95
96 scriptEvents.updateEvents(time_passed, getScriptPhase());
97
98 if (_isCasting())
99 return;
100
101 while (uint32_t eventId = scriptEvents.getFinishedEvent())
102 {
103 switch (eventId)
104 {
105 case EVENT_BONE_SPIKE_GRAVEYARD:
106 {
107 if (isHeroic() || !getCreature()->hasAurasWithId(SPELL_BONE_STORM))
108 _castAISpell(boneSpikeGraveyardSpell);
109
110 scriptEvents.addEvent(EVENT_BONE_SPIKE_GRAVEYARD, Util::getRandomInt(15000, 20000));
111 break;
112 }
113 case EVENT_COLDFLAME:
114 {
115 coldflameLastPos = getCreature()->GetPosition();
116
117 if (!getCreature()->hasAurasWithId(SPELL_BONE_STORM))
118 _castAISpell(coldflameNormalSpell);
119 else
120 _castAISpell(coldflameBoneStormSpell);
121
122 scriptEvents.addEvent(EVENT_COLDFLAME, 5000);
123 break;
124 }
125 case EVENT_WARN_BONE_STORM:
126 {
127 boneSlice = false;
128 _castAISpell(boneStormSpell);
129
130 scriptEvents.delayEvent(EVENT_BONE_SPIKE_GRAVEYARD, 3000);
131 scriptEvents.delayEvent(EVENT_COLDFLAME, 3000);
132
133 scriptEvents.addEvent(EVENT_BONE_STORM_BEGIN, 3050);
134 scriptEvents.addEvent(EVENT_WARN_BONE_STORM, Util::getRandomInt(90000, 95000));
135 }
136 case EVENT_BONE_STORM_BEGIN:
137 {
138 getCreature()->setSpeedRate(TYPE_RUN, baseSpeed * 3.0f, true);
139 sendDBChatMessage(SAY_MARR_BONE_STORM); // BONE STORM!
140
141 scriptEvents.addEvent(EVENT_BONE_STORM_END, boneStormDuration + 1);
142 }
143 [[fallthrough]];
144 case EVENT_BONE_STORM_MOVE:
145 {
146 scriptEvents.addEvent(EVENT_BONE_STORM_MOVE, boneStormDuration / 3);
147
148 boneStormtarget = getBestPlayerTarget(TargetFilter_NotCurrent);
149 if (!boneStormtarget)
150 boneStormtarget = getBestPlayerTarget(TargetFilter_Current);
151
152 if (boneStormtarget)
153 getCreature()->getMovementManager()->movePoint(POINT_TARGET_BONESTORM_PLAYER, boneStormtarget->GetPosition());
154
155 break;
156 }
157 case EVENT_BONE_STORM_END:
158 {
159 if (MovementGenerator* movement = getCreature()->getMovementManager()->getMovementGenerator([](MovementGenerator const* a) -> bool
160 {
161 if (a->getMovementGeneratorType() == POINT_MOTION_TYPE)
162 {
163 PointMovementGenerator<Creature> const* pointMovement = dynamic_cast<PointMovementGenerator<Creature> const*>(a);
164 return pointMovement && pointMovement->getId() == POINT_TARGET_BONESTORM_PLAYER;
165 }
166 return false;
167 }))
168
169 getCreature()->getMovementManager()->remove(movement);
170
171 getCreature()->getMovementManager()->moveChase(getCreature()->getAIInterface()->getCurrentTarget());
172
173 getCreature()->setSpeedRate(TYPE_RUN, baseSpeed, true);
174 scriptEvents.removeEvent(EVENT_BONE_STORM_MOVE);
175 scriptEvents.addEvent(EVENT_ENABLE_BONE_SLICE, 10000);
176
177 if (!isHeroic())
178 scriptEvents.addEvent(EVENT_BONE_SPIKE_GRAVEYARD, Util::getRandomInt(15000, 20000));
179 break;
180 }
181 case EVENT_ENABLE_BONE_SLICE:
182 {
183 boneSlice = true;
184 break;
185 }
186 case EVENT_ENRAGE:
187 {
188 _castAISpell(berserkSpell);
189 break;
190 }
191 default:
192 break;
193 }
194 }
195
196 // We should not melee attack when storming
197 if (getCreature()->hasAurasWithId(SPELL_BONE_STORM))
198 {
199 _setMeleeDisabled(true);
200 return;
201 }
202
203 _setMeleeDisabled(false);
204
205 // 10 seconds since encounter start Bone Slice replaces melee attacks
206 if (boneSlice)
207 {
208 _castAISpell(boneSliceSpell);
209 }
210
211}

301void ColdflameAI::OnSummon(Unit* summoner)
302{
303 if (!mInstance || !summoner->isCreature())
304 return;
305
306 if (summoner->hasAurasWithId(SPELL_BONE_STORM))
307 {
308 // Bonestorm X Pattern
309 if (LordMarrowgarAI* marrowgarAI = static_cast<LordMarrowgarAI*>(static_cast<Creature*>(summoner)->GetScript()))
310 {
311 LocationVector const* ownerPos = marrowgarAI->GetLastColdflamePosition();
312 float angle = ownerPos->o / M_PI_FLOAT * 180.0f;
313 MoveTeleport(ownerPos->x , ownerPos->y, ownerPos->z, ownerPos->o);
314 // Store last Coldflame Position and add 90 degree to create x pattern
315 LocationVector nextPos;
316 nextPos.x = ownerPos->x;
317 nextPos.y = ownerPos->y;
318 nextPos.z = ownerPos->z;
319 nextPos.o = angle + 90 * M_PI_FLOAT / 180.0f;
320
321 marrowgarAI->SetLastColdflamePosition(nextPos);
322 }
323 }
324 // Random target Case
325 else
326 {
327 Unit* target = mInstance->getInstance()->getUnit(static_cast<Creature*>(summoner)->GetScript()->GetCreatureData64(DATA_COLDFLAME_GUID));
328 if (!target)
329 {
330 getCreature()->Despawn(100, 0);
331 return;
332 }
333 MoveTeleport(summoner->GetPosition());
334
335 getCreature()->SetOrientation(getCreature()->calcAngle(summoner->GetPositionX(), summoner->GetPositionY(), target->GetPositionX(), target->GetPositionY()) * M_PI_FLOAT / 180.0f);
336 }
337 MoveTeleport(summoner->GetPositionX(), summoner->GetPositionY(),summoner->GetPositionZ(), getCreature()->GetOrientation());
338 scriptEvents.addEvent(EVENT_COLDFLAME_TRIGGER, 500);
339}

341void ColdflameAI::AIUpdate(unsigned long time_passed)
342{
343 scriptEvents.updateEvents(time_passed, getScriptPhase());
344
345 if (scriptEvents.getFinishedEvent() == EVENT_COLDFLAME_TRIGGER)
346 {
347 LocationVector newPos;
348 newPos = getCreature()->GetPosition();
349
350 float angle = newPos.o;
351 float destx, desty, destz;
352 destx = newPos.x + 5.0f * std::cos(angle);
353 desty = newPos.y + 5.0f * std::sin(angle);
354 destz = newPos.z;
355
356 newPos.x = destx;
357 newPos.y = desty;
358 newPos.z = destz;
359
360 MoveTeleport(newPos.x, newPos.y, newPos.z, newPos.o);
361 _castAISpell(coldflameTriggerSpell);
362 scriptEvents.addEvent(EVENT_COLDFLAME_TRIGGER, 500);

431/// Spell: Bone Storm Damage
432SpellScriptEffectDamage BoneStormDamage::doCalculateEffect(Spell* spell, uint8_t effIndex, int32_t* dmg)
433{
434 if (effIndex != EFF_INDEX_0 || spell->getUnitTarget() == nullptr)
435 return SpellScriptEffectDamage::DAMAGE_DEFAULT;
436
437 auto distance = spell->getUnitTarget()->GetDistance2dSq(spell->getCaster());
438 // If target is closer than 5 yards, do full damage
439 if (distance <= 5.0f)
440 distance = 1.0f;
441
442 *dmg = Util::float2int32(*dmg / distance);
443 return SpellScriptEffectDamage::DAMAGE_FULL_RECALCULATION;
444}

446//////////////////////////////////////////////////////////////////////////////////////////
447/// Spell: Bone Spike Graveyard
448void BoneSpikeGraveyard::onAuraApply(Aura* aur)
449{
450 aur->removeAuraEffect(EFF_INDEX_1);
451
452 if (Creature* marrowgar = static_cast<Creature*>(aur->GetUnitCaster()))
453 {
454 CreatureAIScript* marrowgarAI = marrowgar->GetScript();
455 if (!marrowgarAI)
456 return;
457
458 uint8_t boneSpikeCount = uint8_t(aur->GetUnitCaster()->getWorldMap()->getSpawnMode() & 1 ? 3 : 1);
459 std::list<Unit*> targets;
460
461 targets.clear();
462
463 for (uint8_t target = 0; target < boneSpikeCount; ++target)
464 targets.push_back(GetRandomTargetNotMainTank(marrowgar));
465
466 if (targets.empty())
467 return;
468
469 uint32_t i = 0;
470 for (std::list<Unit*>::const_iterator itr = targets.begin(); itr != targets.end(); ++itr, ++i)
471 {
472 Unit* target = *itr;
473 target->castSpell(target, BoneSpikeSummonId[i], true);
474 }
475
476
477 std::vector<uint32_t> emoteVector;
478 emoteVector.clear();
479 emoteVector.push_back(SAY_MARR_BONESPIKE_1);// Stick Around
480 emoteVector.push_back(SAY_MARR_BONESPIKE_2);// The only Escape is Darkness
481 emoteVector.push_back(SAY_MARR_BONESPIKE_3);// More Bones for the offering
482
483 marrowgarAI->sendRandomDBChatMessage(emoteVector, nullptr);
484 }
485}

497Unit* BoneSpikeGraveyard::GetRandomTargetNotMainTank(Creature* caster)
498{
499 Unit* target = nullptr;
500 std::vector<Player*> players;
501
502 Unit* mt = caster->getAIInterface()->getCurrentTarget();
503 if (mt == nullptr || !mt->isPlayer())
504 return 0;
505
506 for (const auto& itr : caster->getInRangePlayersSet())
507 {
508 Player* obj = static_cast<Player*>(itr);
509 if (obj != mt && CheckTarget(obj, caster->GetScript()))
510 players.push_back(obj);
511 }
512
513 if (players.size())
514 target = players[Util::getRandomUInt(static_cast<uint32_t>(players.size() - 1))];
515
516 return target;
517}

519bool BoneSpikeGraveyard::CheckTarget(Unit* target, CreatureAIScript* creatureAI)
520{
521 if (target->getObjectTypeId() != TYPEID_PLAYER)
522 return false;
523
524 if (target->hasAurasWithId(SPELL_IMPALED))
525 return false;
526
527 // Check if it is one of the tanks soaking Bone Slice
528 for (uint32_t i = 0; i < MAX_BONE_SPIKE_IMMUNE; ++i)
529 if (target->getGuid() == creatureAI->GetCreatureData64(DATA_SPIKE_IMMUNE + i))
530 return false;
531
532 return true;
533}
