140void JaraxxusAI::DoAction(int32_t action)
141{
142 if (action == ACTION_JARAXXUS_INTRO)
143 {
144 setReactState(REACT_PASSIVE);
145 setScriptPhase(jaraxxus::PHASE_INTRO);
146 addAIFunction(&JaraxxusAI::intro, DoOnceScheduler(1s));
147 }
148 else if (action == ACTION_JARAXXUS_ENGAGE)
149 {
150 _removeAura(SPELL_JARAXXUS_CHAINS);
151 setIgnorePlayerCombat(false);
152 setReactState(REACT_AGGRESSIVE);
153 setZoneWideCombat();
154 }
