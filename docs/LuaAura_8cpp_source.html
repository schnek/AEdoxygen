125int LuaAura::SetVar(lua_State* L, Aura* aura)
126{
127 const char* var = luaL_checkstring(L, 1);
128 int subindex = 0;
129 if (lua_gettop(L) == 3)
130 {
131 subindex = static_cast<int>(luaL_optinteger(L, 2, 0));
132 }
133 if (!aura || !var || subindex < 0)
134 {
135 lua_pushboolean(L, 0);
136 return 1;
137 }
138 int valindex = 2;
139 if (subindex)
140 valindex++;
141 SpellInfo const* proto = aura->getSpellInfo();
142 LuaSpellEntry l = GetLuaSpellEntryByName(var);
143 if (!l.name)
144 RET_BOOL(false);
145 switch (l.typeId) //0: int, 1: char*, 2: bool, 3: float
146 {
147 case 0:
148 GET_SPELLVAR_INT(proto, l.offset, subindex) = static_cast<int>(luaL_checkinteger(L, valindex));
149 lua_pushboolean(L, 1);
150 break;
151 case 1:
152 strcpy(GET_SPELLVAR_CHAR(proto, l.offset, subindex), luaL_checkstring(L, valindex));
153 lua_pushboolean(L, 1);
154 break;
155 case 2:
156 GET_SPELLVAR_BOOL(proto, l.offset, subindex) = CHECK_BOOL(L, valindex);
157 lua_pushboolean(L, 1);
158 break;
159 case 3:
160 GET_SPELLVAR_FLOAT(proto, l.offset, subindex) = (float)luaL_checknumber(L, valindex);
161 lua_pushboolean(L, 1);
162 break;
163 }
164 return 1;
165}
