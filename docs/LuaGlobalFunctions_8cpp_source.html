33int LuaGlobalFunctions::PerformIngameSpawn(lua_State* L)
34{
35 uint32_t spawntype = static_cast<uint32_t>(luaL_checkinteger(L, 1));
36 uint32_t entry = static_cast<uint32_t>(luaL_checkinteger(L, 2));
37 uint32_t map = static_cast<uint32_t>(luaL_checkinteger(L, 3));
38 float x = CHECK_FLOAT(L, 4);
39 float y = CHECK_FLOAT(L, 5);
40 float z = CHECK_FLOAT(L, 6);
41 float o = CHECK_FLOAT(L, 7);
42 uint32_t faction = static_cast<uint32_t>(luaL_checkinteger(L, 8)); //also scale as percentage
43 uint32_t duration = static_cast<uint32_t>(luaL_checkinteger(L, 9));
44 uint32_t equip1 = static_cast<uint32_t>(luaL_optinteger(L, 10, 1));
45 uint32_t equip2 = static_cast<uint32_t>(luaL_optinteger(L, 11, 1));
46 uint32_t equip3 = static_cast<uint32_t>(luaL_optinteger(L, 12, 1));
47 //13: instance id
48 uint32_t save = static_cast<uint32_t>(luaL_optinteger(L, 14, 0));
49 if (x && y && z && entry)
50 {
51 if (spawntype == 1) //Unit
52 {
53 CreatureProperties const* p = sMySQLStore.getCreatureProperties(entry);
54 if (p == nullptr)
55 return 0;
56
57 WorldMap* mapMgr = sMapMgr.findWorldMap(map);
58 if (!mapMgr)
59 return 0;
60
61 //int32_t instanceid = static_cast<int32_t>(luaL_optinteger(L, 13, mapMgr->GetInstanceID()));
62 Creature* pCreature = mapMgr->createCreature(entry);
63 pCreature->Load(p, x, y, z, o);
64 pCreature->setFaction(faction);
65 pCreature->setVirtualItemSlotId(MELEE, equip1);
66 pCreature->setVirtualItemSlotId(OFFHAND, equip2);
67 pCreature->setVirtualItemSlotId(RANGED, equip3);
68 pCreature->setPhase(PHASE_SET, 1);
69 pCreature->m_noRespawn = true;
70 pCreature->AddToWorld(mapMgr);
71 if (duration > 0)
72 pCreature->Despawn(duration, 0);
73 if (save)
74 pCreature->SaveToDB();
75 PUSH_UNIT(L, pCreature);
76 }
77 else if (spawntype == 2) //GO
78 {
79 auto gameobject_info = sMySQLStore.getGameObjectProperties(entry);
80 if (gameobject_info == nullptr)
81 return 0;
82
83 WorldMap* mapMgr = sMapMgr.findWorldMap(map);
84 if (!mapMgr)
85 return 0;
86
87 GameObject* go = mapMgr->createGameObject(entry);
88 go->create(entry, mapMgr, 0, LocationVector(x, y, z, o), QuaternionData(), GO_STATE_CLOSED);
89 go->Phase(PHASE_SET, 1);
90 go->setScale(((float)faction) / 100.0f);
91
92 go->AddToWorld(mapMgr);
93
94 if (duration)
95 go->despawn(duration, 0);
96 if (save)
97 go->saveToDB();
98 PUSH_GO(L, go);
99 }
100 else
101 lua_pushnil(L);
102 }
103 else
104 lua_pushnil(L);
105 return 1;
106}

387int LuaGlobalFunctions::SetDBCSpellVar(lua_State* L)
388{
389 uint32_t entry = static_cast<uint32_t>(luaL_checkinteger(L, 1));
390 const char* var = luaL_checkstring(L, 2);
391 int subindex = 0;
392 int valindex = 3;
393 if (lua_gettop(L) == 4)
394 {
395 subindex = static_cast<int>(luaL_optinteger(L, 3, 0));
396 valindex++;
397 }
398 SpellInfo const* proto = sSpellMgr.getSpellInfo(entry);
399 if (!entry || !var || subindex < 0 || !proto)
400 {
401 lua_pushboolean(L, 0);
402 return 1;
403 }
404 LuaSpellEntry l = GetLuaSpellEntryByName(var);
405 if (!l.name)
406 {
407 lua_pushnil(L);
408 return 1;
409 }
410 switch (l.typeId) //0: int, 1: char*, 2: bool, 3: float
411 {
412 case 0:
413 GET_SPELLVAR_INT(proto, l.offset, subindex) = static_cast<int>(luaL_checkinteger(L, valindex));
414 lua_pushboolean(L, 1);
415 break;
416 case 1:
417 strcpy(GET_SPELLVAR_CHAR(proto, l.offset, subindex), luaL_checkstring(L, valindex));
418 lua_pushboolean(L, 1);
419 break;
420 case 2:
421 GET_SPELLVAR_BOOL(proto, l.offset, subindex) = CHECK_BOOL(L, valindex);
422 lua_pushboolean(L, 1);
423 break;
424 case 3:
425 GET_SPELLVAR_FLOAT(proto, l.offset, subindex) = (float)luaL_checknumber(L, valindex);
426 lua_pushboolean(L, 1);
427 break;
428 }
429 return 1;
430}

472 {
473 for (int i = 2; i <= top; ++i)
