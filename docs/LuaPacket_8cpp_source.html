17int LuaPacket::CreatePacket(lua_State* L, WorldPacket* /*packet*/)
18{
19 const uint16_t opcode = static_cast<uint16_t>(luaL_checkinteger(L, 1));
20 const size_t size = static_cast<size_t>(luaL_checkinteger(L, 2));
21 if (opcode >= NUM_OPCODES)
22 luaL_error(L, "CreatePacket got opcode %d greater than max opcode %d.", opcode, NUM_OPCODES);
23 else
24 {
25 WorldPacket* npacket = new WorldPacket(opcode, size);
26 PUSH_PACKET(L, npacket);
27 }
28 return 1;
29}
