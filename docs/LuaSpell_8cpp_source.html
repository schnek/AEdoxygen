209int LuaSpell::SetVar(lua_State* L, Spell* sp)
210{
211 const char* var = luaL_checkstring(L, 1);
212 int subindex = 0;
213 int valindex = 2;
214 if (lua_gettop(L) == 3)
215 {
216 subindex = static_cast<int>(luaL_optinteger(L, 2, 0));
217 valindex++;
218 }
219 if (!sp || !var || subindex < 0)
220 {
221 lua_pushboolean(L, 0);
222 return 1;
223 }
224 sp->InitProtoOverride();
225 SpellInfo const* proto = sp->getSpellInfo();
226 LuaSpellEntry l = GetLuaSpellEntryByName(var);
227 if (!l.name)
228 RET_BOOL(false);
229 switch (l.typeId) //0: int, 1: char*, 2: bool, 3: float
230 {
231 case 0:
232 GET_SPELLVAR_INT(proto, l.offset, subindex) = static_cast<int>(luaL_checkinteger(L, valindex));
233 lua_pushboolean(L, 1);
234 break;
235 case 1:
236 strcpy(GET_SPELLVAR_CHAR(proto, l.offset, subindex), luaL_checkstring(L, valindex));
237 lua_pushboolean(L, 1);
238 break;
239 case 2:
240 GET_SPELLVAR_BOOL(proto, l.offset, subindex) = CHECK_BOOL(L, valindex);
241 lua_pushboolean(L, 1);
242 break;
243 case 3:
244 GET_SPELLVAR_FLOAT(proto, l.offset, subindex) = (float)luaL_checknumber(L, valindex);
245 lua_pushboolean(L, 1);
246 break;
247 }
248 return 1;
249}
