150inline LuaSpellEntry GetLuaSpellEntryByName(const char* name)
151{
152 for (uint32_t itr = 0; luaSpellVars[itr].name != NULL; itr++)
153 {
154 LuaSpellEntry l = luaSpellVars[itr];
155 if (strcmp(l.name, name) == 0) //they entered a correct var name
156 return l;
157 }
158 int lElem = sizeof(luaSpellVars) / sizeof(luaSpellVars[0]) - 1;
159 return luaSpellVars[lElem];
160}
