15int LuaSqlApi::GetColumn(lua_State* L, QueryResult* res)
16{
17 if (res != nullptr)
18 {
19 const uint32_t column = CHECK_ULONG(L, 1);
20 const uint32_t fields = res->GetFieldCount();
21 if (column > fields)
22 luaL_error(L, "GetColumn, Column %d bigger than max column %d", column, res->GetFieldCount());
23 else
24 {
25 Field* field = &(res->Fetch()[column]);
26 PUSH_SQLFIELD(L, field);
27 }
28 }
29 else
30 lua_pushnil(L);
31 return 1;
32}
