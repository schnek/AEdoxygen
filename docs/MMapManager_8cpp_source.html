56 bool MMapManager::loadMapData(uint32_t mapId)
57 {
58 // we already have this map loaded?
59 MMapDataSet::iterator itr = loadedMMaps.find(mapId);
60 if (itr != loadedMMaps.end())
61 {
62 if (itr->second)
63 return true;
64 }
65 else
66 {
67 if (thread_safe_environment)
68 itr = loadedMMaps.insert(MMapDataSet::value_type(mapId, nullptr)).first;
69 else
70 {
71 sLogger.failure("Invalid mapId {} passed to MMapManager after startup in thread unsafe environment", mapId);
72 ASSERT(false);
73 }
74 }
75
76 // load and init dtNavMesh - read parameters from file
77 std::string dataDir = worldConfig.server.dataDir + "mmaps/";
78 uint32_t pathLen = static_cast<uint32_t>(dataDir.length() + strlen("%04i.mmap") + 1);
79 auto fileName = std::make_unique<char[]>(pathLen);
80 snprintf(fileName.get(), pathLen, (dataDir + "%04i.mmap").c_str(), mapId);
81
82 FILE* file = fopen(fileName.get(), "rb");
83 if (!file)
84 {
85 sLogger.debug("MMAP:loadMapData: Error: Could not open mmap file '{}'", fileName.get());
86 return false;
87 }
88
89 dtNavMeshParams params;
90 int count = static_cast<int>(fread(&params, sizeof(dtNavMeshParams), 1, file));
91 fclose(file);
92 if (count != 1)
93 {
94 sLogger.failure("Error: Could not read params from file '{}'", fileName.get());
95 return false;
96 }
97
98 dtNavMesh* mesh = dtAllocNavMesh();
99 ASSERT(mesh);
100 if (dtStatusFailed(mesh->init(&params)))
101 {
102 dtFreeNavMesh(mesh);
103 sLogger.failure("Failed to initialize dtNavMesh for mmap {:04} from file {}", mapId, fileName.get());
104 return false;
105 }
106
107 sLogger.debug("MMAP:loadMapData: Loaded {:04}.mmap", mapId);
108
109 // store inside our map list
110 itr->second = std::make_unique<MMapData>(mesh);
111 itr->second->mmapLoadedTiles.clear();
112 return true;
113 }

120 bool MMapManager::loadMap(const std::string& basePath, uint32_t mapId, int32_t x, int32_t y)
121 {
122 // make sure the mmap is loaded and ready to load tiles
123 if (!loadMapData(mapId))
124 return false;
125
126 // get this mmap data
127 MMapData* mmap = loadedMMaps[mapId].get();
128 ASSERT(mmap->navMesh);
129
130 // check if we already have this tile loaded
131 uint32_t packedGridPos = packTileID(x, y);
132 if (mmap->mmapLoadedTiles.find(packedGridPos) != mmap->mmapLoadedTiles.end())
133 return false;
134
135 // load this tile :: /MMMXXYY.mmtile
136 uint32_t pathLen = static_cast<uint32_t>(basePath.length() + strlen("/%04i%02i%02i.mmtile") + 1);
137 auto fileName = std::make_unique<char[]>(pathLen);
138
139 snprintf(fileName.get(), pathLen, (basePath + "/%04i%02i%02i.mmtile").c_str(), mapId, x, y);
140
141 FILE* file = fopen(fileName.get(), "rb");
142 if (!file)
143 {
144 sLogger.debug("Could not open mmtile file '{}'", fileName.get());
145 return false;
146 }
147
148 // read header
149 MmapTileHeader fileHeader;
150 if (fread(&fileHeader, sizeof(MmapTileHeader), 1, file) != 1 || fileHeader.mmapMagic != MMAP_MAGIC)
151 {
152 sLogger.failure("Bad header in mmap {:04}{:02}{:02}.mmtile", mapId, x, y);
153 fclose(file);
154 return false;
155 }
156
157 if (fileHeader.mmapVersion != MMAP_VERSION)
158 {
159 sLogger.failure("{:04}{:02}{:02}.mmtile was built with generator v{}, expected v{}",
160 mapId, x, y, fileHeader.mmapVersion, MMAP_VERSION);
161 fclose(file);
162 return false;
163 }
164
165 unsigned char* data = (unsigned char*)dtAlloc(fileHeader.size, DT_ALLOC_PERM);
166 ASSERT(data);
167
168 size_t result = fread(data, fileHeader.size, 1, file);
169 if (!result)
170 {
171 sLogger.failure("Bad header or data in mmap {:04}{:02}{:02}.mmtile", mapId, x, y);
172 fclose(file);
173 return false;
174 }
175
176 fclose(file);
177
178 dtMeshHeader* header = (dtMeshHeader*)data;
179 dtTileRef tileRef = 0;
180
181 // memory allocated for data is now managed by detour, and will be deallocated when the tile is removed
182 if (dtStatusSucceed(mmap->navMesh->addTile(data, fileHeader.size, DT_TILE_FREE_DATA, 0, &tileRef)))
183 {
184 mmap->mmapLoadedTiles.insert(std::pair<uint32_t, dtTileRef>(packedGridPos, tileRef));
185 ++loadedTiles;
186 sLogger.debug("MMAP:loadMap: Loaded mmtile {:04}[{:02}, {:02}] into {:04}[{:02}, {:02}]", mapId, x, y, mapId, header->x, header->y);
187 return true;
188 }
189 else
190 {
191 sLogger.debug("MMAP:loadMap: Could not load {:04}{:02}{:02}.mmtile into navmesh", mapId, x, y);
192 dtFree(data);
193 return false;
194 }
195 }

197 bool MMapManager::unloadMap(uint32_t mapId, int32_t x, int32_t y)
198 {
199 // check if we have this map loaded
200 MMapDataSet::const_iterator itr = GetMMapData(mapId);
201 if (itr == loadedMMaps.end())
202 {
203 // file may not exist, therefore not loaded
204 sLogger.debug("MMAP:unloadMap: Asked to unload not loaded navmesh map. {:04}{:02}{:02}.mmtile", mapId, x, y);
205 return false;
206 }
207
208 MMapData* mmap = itr->second.get();
209
210 // check if we have this tile loaded
211 uint32_t packedGridPos = packTileID(x, y);
212 if (mmap->mmapLoadedTiles.find(packedGridPos) == mmap->mmapLoadedTiles.end())
213 {
214 // file may not exist, therefore not loaded
215 sLogger.debug("MMAP:unloadMap: Asked to unload not loaded navmesh tile. {:04}{:02}{:02}.mmtile", mapId, x, y);
216 return false;
217 }
218
219 dtTileRef tileRef = mmap->mmapLoadedTiles[packedGridPos];
220
221 // unload, and mark as non loaded
222 if (dtStatusFailed(mmap->navMesh->removeTile(tileRef, nullptr, nullptr)))
223 {
224 // this is technically a memory leak
225 // if the grid is later reloaded, dtNavMesh::addTile will return error but no extra memory is used
226 // we cannot recover from this error - assert out
227 sLogger.failure("Could not unload {:04}{:02}{:02}.mmtile from navmesh", mapId, x, y);
228 }
229 else
230 {
231 mmap->mmapLoadedTiles.erase(packedGridPos);
232 --loadedTiles;
233 sLogger.debug("MMAP:unloadMap: Unloaded mmtile {:04}[{:02}, {:02}] from {:04}", mapId, x, y, mapId);
234 return true;
235 }
236
237 return false;
238 }

240 bool MMapManager::unloadMap(uint32_t mapId)
241 {
242 MMapDataSet::iterator itr = loadedMMaps.find(mapId);
243 if (itr == loadedMMaps.end() || !itr->second)
244 {
245 // file may not exist, therefore not loaded
246 sLogger.debug("MMAP:unloadMap: Asked to unload not loaded navmesh map. {:04}.mmtile", mapId);
247 return false;
248 }
249
250 // unload all tiles from given map
251 MMapData* mmap = itr->second.get();
252 for (MMapTileSet::iterator i = mmap->mmapLoadedTiles.begin(); i != mmap->mmapLoadedTiles.end(); ++i)
253 {
254 uint32_t x = (i->first >> 16);
255 uint32_t y = (i->first & 0x0000FFFF);
256 if (dtStatusFailed(mmap->navMesh->removeTile(i->second, nullptr, nullptr)))
257 sLogger.failure("Could not unload {:04}{:02}{:02}.mmtile from navmesh", mapId, x, y);
258 else
259 {
260 --loadedTiles;
261 sLogger.debug("MMAP:unloadMap: Unloaded mmtile {:04}[{:02}, {:02}] from {:04}", mapId, x, y, mapId);
262 }
263 }
264
265 itr->second = nullptr;
266 sLogger.debug("MMAP:unloadMap: Unloaded {:04}.mmap", mapId);
267
268 return true;
269 }

271 bool MMapManager::unloadMapInstance(uint32_t mapId, uint32_t instanceId)
272 {
273 // check if we have this map loaded
274 MMapDataSet::const_iterator itr = GetMMapData(mapId);
275 if (itr == loadedMMaps.end())
276 {
277 // file may not exist, therefore not loaded
278 sLogger.debug("MMAP:unloadMapInstance: Asked to unload not loaded navmesh map {:04}", mapId);
279 return false;
280 }
281
282 MMapData* mmap = itr->second.get();
283 if (mmap->navMeshQueries.find(instanceId) == mmap->navMeshQueries.end())
284 {
285 sLogger.debug("MMAP:unloadMapInstance: Asked to unload not loaded dtNavMeshQuery mapId {:04} instanceId {}", mapId, instanceId);
286 return false;
287 }
288
289 dtNavMeshQuery* query = mmap->navMeshQueries[instanceId];
290
291 dtFreeNavMeshQuery(query);
292 mmap->navMeshQueries.erase(instanceId);
293 sLogger.debug("MMAP:unloadMapInstance: Unloaded mapId {:04} instanceId {}", mapId, instanceId);
294
295 return true;
296 }

307 dtNavMeshQuery const* MMapManager::GetNavMeshQuery(uint32_t mapId, uint32_t instanceId)
308 {
309 MMapDataSet::const_iterator itr = GetMMapData(mapId);
310 if (itr == loadedMMaps.end())
311 return nullptr;
312
313 MMapData* mmap = itr->second.get();
314 if (mmap->navMeshQueries.find(instanceId) == mmap->navMeshQueries.end())
315 {
316 // allocate mesh query
317 dtNavMeshQuery* query = dtAllocNavMeshQuery();
318 ASSERT(query);
319 if (dtStatusFailed(query->init(mmap->navMesh, 1024)))
320 {
321 dtFreeNavMeshQuery(query);
322 sLogger.failure("Failed to initialize dtNavMeshQuery for mapId {:04} instanceId {}", mapId, instanceId);
323 return nullptr;
324 }
325
326 sLogger.debug("MMAP:GetNavMeshQuery: created dtNavMeshQuery for mapId {:04} instanceId {}", mapId, instanceId);
327 mmap->navMeshQueries.insert(std::pair<uint32_t, dtNavMeshQuery*>(instanceId, query));
328 }
329
330 return mmap->navMeshQueries[instanceId];
331 }
