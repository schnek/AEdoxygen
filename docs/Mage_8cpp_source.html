229class MasterOfElements : public SpellScript
230{
231public:
232 void onCreateSpellProc(SpellProc* spellProc, Object* /*obj*/) override
233 {
234 spellProc->setCastedOnProcOwner(true);
235 }
236
237 bool canProc(SpellProc* /*spellProc*/, Unit* /*victim*/, SpellInfo const* castingSpell, DamageInfo damageInfo) override
238 {
239 if (castingSpell == nullptr || damageInfo.weaponType == RANGED)
240 return false;
241#if VERSION_STRING < WotLK
242 // Can proc only from fire or frost school
243 if (castingSpell->getSchoolMask() & SCHOOL_MASK_FIRE || castingSpell->getSchoolMask() & SCHOOL_MASK_FROST)
244 return true;
245 return false;
246#else
247 // Reset aura before continuing
248 originalAura = nullptr;
249 return true;
250#endif
251 }
252
253 bool canProcOnTriggered(SpellProc* /*spellProc*/, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, Aura* triggeredFromAura) override
254 {
255 if (triggeredFromAura == nullptr)
256 return false;
257
258 originalAura = triggeredFromAura;
259 return true;
260 }
261
262 SpellScriptExecuteState onDoProcEffect(SpellProc* spellProc, Unit* /*victim*/, SpellInfo const* castingSpell, DamageInfo /*damageInfo*/) override
263 {
264 if (originalAura != nullptr && !originalAura->isDeleted() && originalAura->getSpellInfo()->isChanneled())
265 {
266 // "For channeled spells that deal damage in ticks such as Arcane Missiles and Blizzard, the mana returned is based on the tick that critically hit, and not the entire spell."
267 // i.e. Arcane Missiles deals damage in 5 ticks; it will only calculate mana return for 1/5th of the cost of Arcane Missiles
268
269 uint16_t ticks = 1;
270
271 // Find the aura effect
272 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
273 {
274 const auto aurEff = originalAura->getAuraEffect(i);
275 if (aurEff->getAuraEffectType() == SPELL_AURA_NONE)
276 continue;
277
278#if VERSION_STRING == Classic
279 if (aurEff->getAuraEffectType() == SPELL_AURA_PERIODIC_TRIGGER_SPELL)
280#else
281 if (aurEff->getAuraEffectType() == SPELL_AURA_PERIODIC_TRIGGER_SPELL ||
282 aurEff->getAuraEffectType() == SPELL_AURA_PERIODIC_TRIGGER_SPELL_WITH_VALUE)
283#endif
284 {
285 ticks = originalAura->getPeriodicTickCountForEffect(aurEff->getEffectIndex());
286 break;
287 }
288 }
289
290 // Divide mana return with tick count
291 manaReturn = originalAura->getSpellInfo()->getBasePowerCost(spellProc->getProcOwner()) / ticks;
292 originalAura = nullptr;
293 }
294 else
295 {
296 manaReturn = castingSpell->getBasePowerCost(spellProc->getProcOwner());
297 }
298
299 if (manaReturn <= 0)
300 return SpellScriptExecuteState::EXECUTE_PREVENT;
301
302 manaReturn = static_cast<int32_t>(std::round(manaReturn * spellProc->getOverrideEffectDamage(EFF_INDEX_0) / 100.0f));
303 return SpellScriptExecuteState::EXECUTE_OK;
304 }
305
306 SpellScriptExecuteState onCastProcSpell(SpellProc* /*spellProc*/, Unit* /*caster*/, Unit* /*victim*/, Spell* spell) override
307 {
308 spell->forced_basepoints->set(EFF_INDEX_0, manaReturn);
309 manaReturn = 0;
310 return SpellScriptExecuteState::EXECUTE_OK;
311 }
312
313private:
314 Aura* originalAura = nullptr;
315 int32_t manaReturn = 0;
316};
317#endif

319class Polymorph : public SpellScript
320{
321public:
322 SpellScriptExecuteState beforeAuraEffect(Aura* aur, AuraEffectModifier* aurEff, bool apply) override
323 {
324 if (aurEff->getAuraEffectType() != SPELL_AURA_TRANSFORM)
325 return SpellScriptExecuteState::EXECUTE_OK;
326
327 if (apply)
328 {
329 // Dismount player on aura apply
330 // TODO: should also dismount creatures
331 if (aur->getPlayerOwner() != nullptr)
332 aur->getPlayerOwner()->dismount();
333
334 // Add this unitstate only for player polymorph spells
335 // Mostly polymorphs casted by creatures won't regenerate health
336 aur->getOwner()->addUnitStateFlag(UNIT_STATE_POLYMORPHED);
337
338 // Glyph of the Penguin
339 const auto caster = aur->GetUnitCaster();
340 if (caster != nullptr && caster->hasAurasWithId(SPELL_GLYPH_OF_THE_PENGUIN) && aur->getSpellInfo()->getSpellIconID() == ICON_POLYMORPH_SHEEP)
341 {
342 // Override misc value (Sheep) with Penguin npc
343 aurEff->setEffectMiscValue(CREATURE_CHILLY);
344 }
345 }
346 else
347 {
348 aur->getOwner()->removeUnitStateFlag(UNIT_STATE_POLYMORPHED);
349 }
350
351 return SpellScriptExecuteState::EXECUTE_OK;
352 }
