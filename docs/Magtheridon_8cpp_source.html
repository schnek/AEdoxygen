68void MagtheridonAI::AIUpdate(unsigned long time_passed)
69{
70 if(isScriptPhase(PHASE_BANISH))
71 _castAISpell(shadowCage);
72
73 if (!isScriptPhase(PHASE_BANISH) && !isScriptPhase(PHASE_1) && !getCreature()->getAIInterface()->getCurrentTarget())
74 return;
75
76 scriptEvents.updateEvents(time_passed, getScriptPhase());
77
78 if (_isCasting())
79 return;
80
81 while (uint32_t eventId = scriptEvents.getFinishedEvent())
82 {
83 switch (eventId)
84 {
85 case EVENT_BERSERK :
86 _castAISpell(berserk);
87 break;
88 case EVENT_CLEAVE:
89 _castAISpell(cleave);
90 scriptEvents.addEvent(EVENT_CLEAVE, 10000);
91 break;
92 case EVENT_BLAZE:
93 _castAISpell(blaseTarget);
94 scriptEvents.addEvent(EVENT_BLAZE, 20000);
95 break;
96 case EVENT_QUAKE:
97 _castAISpell(quake);
98 scriptEvents.addEvent(EVENT_QUAKE, 60000);
99 break;
100 case EVENT_START_FIGHT:
101 CombatStart();
102 break;
103 case EVENT_RELEASED:
104 getCreature()->getAIInterface()->setAllowedToEnterCombat(true);
105 getCreature()->getAIInterface()->setIgnorePlayerCombat(false);
106 getCreature()->getAIInterface()->setIgnoreCreatureCombat(false);
107 getCreature()->removeUnitFlags(UNIT_FLAG_NOT_SELECTABLE);
108 setScriptPhase(PHASE_2);
109
110 getInstanceScript()->setLocalData(DATA_MANTICRON_CUBE, ACTION_ENABLE);
111 scriptEvents.addEvent(EVENT_CLEAVE, 10000);
112 scriptEvents.addEvent(EVENT_BLAST_NOVA, 60000);
113 scriptEvents.addEvent(EVENT_BLAZE, 20000);
114 scriptEvents.addEvent(EVENT_QUAKE, 35000);
115 scriptEvents.addEvent(EVENT_BERSERK, 12000000);
116 break;
117 case EVENT_COLLAPSE:
118 getInstanceScript()->setLocalData(DATA_COLLAPSE_2, ACTION_ENABLE);
119 scriptEvents.addEvent(EVENT_DEBRIS_KNOCKDOWN, 4000);
120 break;
121 case EVENT_DEBRIS_KNOCKDOWN:
122 {
123 if (Creature* trigger = ((MagtheridonsLairInstanceScript*)getInstanceScript())->worldTrigger)
124 {
125 trigger->castSpell(trigger, SPELL_DEBRIS_KNOCKDOWN, true);
126 getCreature()->getAIInterface()->setReactState(REACT_AGGRESSIVE);
127 }
128 scriptEvents.addEvent(EVENT_DEBRIS, 20000);
129 }
130 break;
131 case EVENT_DEBRIS:
132 {
133 Unit* target = getBestPlayerTarget(TargetFilter_NotCurrent);
134 if(!target)
135 target = getBestPlayerTarget(TargetFilter_Current);
136
137 if (target)
138 {
139 /*Creature* debris = */spawnCreature(NPC_MAGTHERIDON_ROOM, target->GetPosition());
140 target->castSpell(target, SPELL_DEBRIS_VISUAL, true); // hackfix invis creatures cant have visual spells atm -_-
141 scriptEvents.addEvent(EVENT_DEBRIS, 20000);
142 }
143 }
144 break;
145 case EVENT_NEARLY_EMOTE:
146 getCreature()->sendChatMessageAlternateEntry(17257, CHAT_MSG_MONSTER_EMOTE, LANG_UNIVERSAL, "%s is nearly free of his bonds!");
147 break;
148 case EVENT_BLAST_NOVA:
149 getCreature()->sendChatMessageAlternateEntry(17257, CHAT_MSG_MONSTER_EMOTE, LANG_UNIVERSAL, "%s begins to cast Blast Nova!");
150 _castAISpell(blastNova);
151 scriptEvents.addEvent(EVENT_BLAST_NOVA, 55000);
152 break;
153 case EVENT_TAUNT:
154 switch (Util::getRandomUInt(0, 5))
155 {
156 case 0:
157 sendDBChatMessage(SAY_TAUNT01);
158 break;
159 case 1:
160 sendDBChatMessage(SAY_TAUNT02);
161 break;
162 case 2:
163 sendDBChatMessage(SAY_TAUNT03);
164 break;
165 case 3:
166 sendDBChatMessage(SAY_TAUNT04);
167 break;
168 case 4:
169 sendDBChatMessage(SAY_TAUNT05);
170 break;
171 case 5:
172 sendDBChatMessage(SAY_TAUNT06);
173 break;
174 }
175
176 scriptEvents.addEvent(EVENT_TAUNT, Util::getRandomUInt(240000, 300000));
177 break;
178 default:
179 break;
180 }
181 }
182
183 // When we have 5 Auras from the Cubes Banish us
184 if (getCreature()->getAuraCountForId(SPELL_SHADOW_GRASP_VISUAL) == 5)
185 _castAISpell(shadowCage);
186}
