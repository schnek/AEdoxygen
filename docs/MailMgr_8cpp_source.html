40MailError MailSystem::DeliverMessage(uint64_t recipent, MailMessage* message)
41{
42 // assign a new id
43 message->message_id = sObjectMgr.generateMailId();
44
45 Player* plr = sObjectMgr.getPlayer((uint32_t)recipent);
46 if (plr != NULL)
47 {
48 plr->m_mailBox->AddMessage(message);
49 if ((uint32_t)UNIXTIME >= message->delivery_time)
50 plr->sendPacket(AscEmu::Packets::SmsgReceivedMail().serialise().get());
51 }
52
53 SaveMessageToSQL(message);
54 return MAIL_OK;
55}

84void MailSystem::SaveMessageToSQL(MailMessage* message)
85{
86 std::stringstream ss;
87
88 ss << "DELETE FROM mailbox WHERE message_id = ";
89 ss << message->message_id;
90 ss << ";";
91
92 CharacterDatabase.ExecuteNA(ss.str().c_str());
93
94 ss.rdbuf()->str("");
95
96 std::vector< uint32_t >::iterator itr;
97 ss << "INSERT INTO mailbox VALUES("
98 << message->message_id << ","
99 << message->message_type << ","
100 << message->player_guid << ","
101 << message->sender_guid << ",\'"
102 << CharacterDatabase.EscapeString(message->subject) << "\',\'"
103 << CharacterDatabase.EscapeString(message->body) << "\',"
104 << message->money << ",'";
105
106 for (itr = message->items.begin(); itr != message->items.end(); ++itr)
107 ss << (*itr) << ",";
108
109 ss << "',"
110 << message->cod << ","
111 << message->stationery << ","
112 << message->expire_time << ","
113 << message->delivery_time << ","
114 << message->checked_flag << ","
115 << message->deleted_flag << ");";
116
117 CharacterDatabase.ExecuteNA(ss.str().c_str());

129void MailSystem::SendAutomatedMessage(uint32_t type, uint64_t sender, uint64_t receiver, std::string subject, std::string body,
130 uint32_t money, uint32_t cod, std::vector<uint64_t> &item_guids, uint32_t stationery, MailCheckMask checked, uint32_t deliverdelay)
131{
132 // This is for sending automated messages, for example from an auction house.
133 MailMessage msg;
134 msg.message_type = type;
135 msg.sender_guid = sender;
136 msg.player_guid = receiver;
137 msg.subject = subject;
138 msg.body = body;
139 msg.money = money;
140 msg.cod = cod;
141 for (std::vector<uint64_t>::iterator itr = item_guids.begin(); itr != item_guids.end(); ++itr)
142 msg.items.push_back(WoWGuid::getGuidLowPartFromUInt64(*itr));
143
144 msg.stationery = stationery;
145 msg.delivery_time = (uint32_t)UNIXTIME + deliverdelay;
146
147 // 30 days expiration time for unread mail + possible delivery delay.
148 if (!sMailSystem.MailOption(MAIL_FLAG_NO_EXPIRY))
149 msg.expire_time = (uint32_t)UNIXTIME + deliverdelay + (TimeVars::Day * MAIL_DEFAULT_EXPIRATION_TIME);
150 else
151 msg.expire_time = 0;
152
153 msg.deleted_flag = false;
154 msg.checked_flag = checked;
155
156 // Send the message.
157 DeliverMessage(receiver, &msg);
158}

179void Mailbox::Load(QueryResult* result)
180{
181 if (!result)
182 return;
183
184 Field* fields;
185 MailMessage msg;
186 uint32_t i;
187 char* str;
188 char* p;
189 uint32_t itemguid;
190 uint32_t now = (uint32_t)UNIXTIME;
191
192 do
193 {
194 fields = result->Fetch();
195 uint32_t expiry_time = fields[10].asUint32();
196
197 // Do not load expired mails!
198 if (expiry_time < now)
199 continue;
200
201 // Create message struct
202 i = 0;
203 msg.items.clear();
204 msg.message_id = fields[i++].asUint32();
205 msg.message_type = fields[i++].asUint32();
206 msg.player_guid = fields[i++].asUint32();
207 msg.sender_guid = fields[i++].asUint32();
208 msg.subject = fields[i++].asCString();
209 msg.body = fields[i++].asCString();
210 msg.money = fields[i++].asUint32();
211 str = (char*)fields[i++].asCString();
212 p = strchr(str, ',');
213 if (p == NULL)
214 {
215 itemguid = atoi(str);
216 if (itemguid != 0)
217 msg.items.push_back(itemguid);
218 }
219 else
220 {
221 while (p)
222 {
223 *p = 0;
224 p++;
225
226 itemguid = atoi(str);
227 if (itemguid != 0)
228 msg.items.push_back(itemguid);
229
230 str = p;
231 p = strchr(str, ',');
232 }
233 }
234
235 msg.cod = fields[i++].asUint32();
236 msg.stationery = fields[i++].asUint32();
237 msg.expire_time = fields[i++].asUint32();
238 msg.delivery_time = fields[i++].asUint32();
239 msg.checked_flag = fields[i++].asUint32();
240 msg.deleted_flag = fields[i++].asBool();
241
242 // Add to the mailbox
243 AddMessage(&msg);
244
245 }
246 while (result->NextRow());
247}
