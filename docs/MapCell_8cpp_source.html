82void MapCell::setActivity(bool state)
83{
84 uint32_t mapId = _map->getBaseMap()->getMapId();
85 uint32_t tileX = _x / 8;
86 uint32_t tileY = _y / 8;
87
88 if (!state)
89 _idlepending = false;
90
91 if (!_active && state)
92 {
93 // Move all objects to active set.
94 for (ObjectSet::iterator itr = _objects.begin(); itr != _objects.end(); ++itr)
95 {
96 if (!(*itr)->IsActive() && (*itr)->CanActivate())
97 (*itr)->Activate(_map);
98 }
99
100 if (_unloadpending)
101 cancelPendingUnload();
102
103 if (worldConfig.terrainCollision.isCollisionEnabled)
104 {
105 const auto mgr = VMAP::VMapFactory::createOrGetVMapManager();
106 MMAP::MMapManager* mmgr = MMAP::MMapFactory::createOrGetMMapManager();
107
108 std::string vmapPath = worldConfig.server.dataDir + "vmaps";
109 std::string mmapPath = worldConfig.server.dataDir + "mmaps";
110
111 std::lock_guard lock(m_cellloadLock);
112
113 if (m_celltilesLoaded[mapId][tileX][tileY] == 0)
114 {
115 mgr->loadMap(vmapPath.c_str(), mapId, tileX, tileY);
116 mmgr->loadMap(mmapPath, mapId, tileX, tileY);
117 }
118 ++m_celltilesLoaded[mapId][tileX][tileY];
119 }
120 }
121 else if (_active && !state)
122 {
123 // Move all objects from active set.
124 for (ObjectSet::iterator itr = _objects.begin(); itr != _objects.end(); ++itr)
125 {
126 if ((*itr)->IsActive())
127 (*itr)->deactivate(_map);
128 }
129
130 if (!_unloadpending && canUnload())
131 queueUnloadPending();
132
133 if (worldConfig.terrainCollision.isCollisionEnabled)
134 {
135 const auto mgr = VMAP::VMapFactory::createOrGetVMapManager();
136 MMAP::MMapManager* mmgr = MMAP::MMapFactory::createOrGetMMapManager();
137 std::lock_guard lock(m_cellloadLock);
138
139 if (!(--m_celltilesLoaded[mapId][tileX][tileY]))
140 {
141 mgr->unloadMap(mapId, tileX, tileY);
142 mmgr->unloadMap(mapId, tileX, tileY);
143 }
144 }
145 }
146
147 _active = state;
148}

150void MapCell::removeObjects()
151{
152 ObjectSet::iterator itr;
153
154 //Zack : we are delaying cell removal so transports can see objects far away. We are waiting for the event to remove us
155 if (_unloadpending == true)
156 return;
157
158 /* delete objects in pending respawn state */
159 for (itr = _respawnObjects.begin(); itr != _respawnObjects.end(); ++itr)
160 {
161 switch ((*itr)->getObjectTypeId())
162 {
163 case TYPEID_UNIT:
164 if (!(*itr)->isPet())
165 {
166 _map->_reusable_guids_creature.push_back((*itr)->GetUIdFromGUID());
167 reinterpret_cast<Creature*>(*itr)->m_respawnCell = nullptr;
168 delete static_cast<Creature*>(*itr);
169 }
170 break;
171 case TYPEID_GAMEOBJECT:
172 _map->_reusable_guids_gameobject.push_back((*itr)->GetUIdFromGUID());
173 reinterpret_cast<GameObject*>(*itr)->m_respawnCell = nullptr;
174 delete static_cast<GameObject*>(*itr);
175 break;
176 }
177 }
178 _respawnObjects.clear();
179
180 //This time it's simpler! We just remove everything :)
181 for (objects_iterator = _objects.begin(); objects_iterator != _objects.end();)
182 {
183 Object* obj = (*objects_iterator);
184
185 ++objects_iterator;
186
187 //If MapUnloadTime is non-zero, a transport could get deleted here (when it arrives to a cell that's scheduled to be unloaded because players left from it), so don't delete it! - By: VLack aka. VLsoft
188 if (!bServerShutdown && obj->isGameObject() && static_cast<GameObject*>(obj)->GetGameObjectProperties()->type == GAMEOBJECT_TYPE_MO_TRANSPORT)
189 continue;
190
191 if (obj->IsActive())
192 obj->deactivate(_map);
193
194 if (obj->IsInWorld())
195 obj->RemoveFromWorld(true);
196
197 delete obj;
198 }
199 _objects.clear();
200 _corpses.clear();
201 _playerCount = 0;
202 _transportCount = 0;
203 _loaded = false;
204}

206void MapCell::loadObjects(CellSpawns* sp)
207{
208 //we still have mobs loaded on cell. There is no point of loading them again
209 if (_loaded == true)
210 return;
211
212 _loaded = true;
213 if (sp->CreatureSpawns.size()) //got creatures
214 {
215 for (CreatureSpawnList::iterator i = sp->CreatureSpawns.begin(); i != sp->CreatureSpawns.end(); ++i)
216 {
217 auto spawnGroupData = sMySQLStore.getSpawnGroupDataBySpawn((*i)->id);
218 bool onRespawn = false;
219
220 Creature* c = _map->createCreature((*i)->entry);
221
222 c->m_loadedFromDB = true;
223
224 if (c->Load(*i, _map->getDifficulty(), _map->getBaseMap()->getMapInfo()) && c->CanAddToWorld())
225 {
226 // Respawn Handling
227 if (auto info = _map->getRespawnInfo(SPAWN_TYPE_CREATURE, c->spawnid))
228 {
229 onRespawn = true;
230
231 // get the cell with our SPAWN location. if we've moved cell this might break :P
232 MapCell* pCell = _map->getCellByCoords(c->GetSpawnX(), c->GetSpawnY());
233 if (pCell == nullptr)
234 pCell = c->GetMapCell();
235
236 if (pCell != nullptr)
237 {
238 pCell->_respawnObjects.insert(c);
239
240 c->SetPosition(c->GetSpawnPosition(), true);
241 c->m_respawnCell = pCell;
242 sEventMgr.RemoveEvents(c);
243
244 RespawnInfo ri;
245 ri.type = SPAWN_TYPE_CREATURE;
246 ri.spawnId = c->getSpawnId();
247 ri.entry = c->getEntry();
248 ri.time = info->time;
249 ri.cellX = c->GetSpawnX();
250 ri.cellY = c->GetSpawnY();
251 ri.obj = c;
252
253 bool success = _map->addRespawn(ri);
254 if (success)
255 _map->saveRespawnDB(ri);
256 }
257 }
258
259 if (spawnGroupData)
260 spawnGroupData->spawns[(*i)->id] = c;
261
262 // Creatures Spawning
263 if (!onRespawn)
264 c->PushToWorld(_map);
265 }
266 else
267 {
268 MySQLStructure::CreatureSpawn* spawn = (*i);
269 sLogger.failure("Failed spawning Creature {} with spawnId {} MapId {}", spawn->entry, spawn->id, _map->getBaseMap()->getMapId());
270 delete c; //missing proto or something of that kind
271 }
272 }
273 }
274
275 if (sp->GameobjectSpawns.size()) //got GOs
276 {
277 for (GameobjectSpawnList::iterator i = sp->GameobjectSpawns.begin(); i != sp->GameobjectSpawns.end(); ++i)
278 {
279 GameObject* go = _map->createGameObject((*i)->entry);
280 bool onRespawn = false;
281
282 if (go->loadFromDB(*i, _map, false))
283 {
284 go->m_loadedFromDB = true;
285
286 // Respawn Handling
287 if (auto info = _map->getRespawnInfo(SPAWN_TYPE_GAMEOBJECT, go->getSpawnId()))
288 {
289 onRespawn = true;
290 RespawnInfo ri;
291 ri.type = SPAWN_TYPE_GAMEOBJECT;
292 ri.spawnId = go->getSpawnId();
293 ri.entry = go->getEntry();
294 ri.time = info->time;
295 ri.cellX = go->GetSpawnX();
296 ri.cellY = go->GetSpawnY();
297 ri.obj = nullptr;
298
299 bool success = _map->addRespawn(ri);
300 if (success)
301 _map->saveRespawnDB(ri);
302
303 go->expireAndDelete();
304 }
305
306 if (!onRespawn)
307 go->PushToWorld(_map);
308 }
309 else
310 {
311 MySQLStructure::GameobjectSpawn* spawn = (*i);
312 sLogger.failure("Failed spawning GameObject {} with spawnId {} MapId {}", spawn->entry, spawn->id, _map->getBaseMap()->getMapId());
313 delete go; //missing proto or something of that kind
314 }
315 }
316 }

370void MapCell::unload()
371{
372 if (_unloadpending)
373 {
374 sLogger.debug("Unloading cell {} {}", _x, _y);
375
376 if (_active)
377 {
378 sLogger.failure("MapCell::Unload tried to unload an active MapCell, return!");
379 return;
380 }
381
382 _unloadpending = false;
383
384 /*in ~MapCell RemoveObjects() can delete an Object without removing it from the MapCell.cpp
385 Example:
386 Creature A has guardian B. MapCell is unloaded, _mapmgr->Remove(_x, _y) is called, nullifying the reference to the cell
387 in CellHandler. ~MapCell is called, RemoveObjects() is called and despawns A which despawns B, calling Object::RemoveFromWorld()
388 which calls MapMgr::RemoveObject(B) which calls cell->RemoveObject(obj) ONLY if cell is not NULL, but in this case is NULL, leaving
389 a reference to a deleted Object in MapCell::_objects, iterated in RemoveObjects(). Calling it here fixes this issue.
390 Note: RemoveObjects() is still called in ~MapCell, due to fancy ArcEmu behaviors, like the in-game command ".mapcell delete <x> <y>*/
391
392 sLogger.debug("Unloading cell {} {}", _x, _y);
393
394 removeObjects();
395 _map->remove(_x, _y);
396 }
397 else
398 {
399 sLogger.failure("MapCell::Unload tried to unload MapCell which is not marked for unload");
400 }
