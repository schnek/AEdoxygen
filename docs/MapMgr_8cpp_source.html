36void MapMgr::initialize()
37{
38 // Generate Instances based on WorldMapInfo Table
39 const auto mapInfoStore = sMySQLStore.getWorldMapInfoStore();
40 for (auto mapInfo = mapInfoStore->begin(); mapInfo != mapInfoStore->end(); ++mapInfo)
41 {
42 if (mapInfo->second.mapid >= MAX_NUM_MAPS)
43 {
44 sLogger.failure("InstanceMgr : One or more of your worldmap_info rows specifies an invalid map: {}", mapInfo->second.mapid);
45 continue;
46 }
47
48 createBaseMap(mapInfo->second.mapid);
49 }

58void MapMgr::shutdown()
59{
60 std::scoped_lock<std::mutex> lock(m_mapsLock);
61
62 // Continents
63 for (auto map = m_WorldMaps.cbegin(); map != m_WorldMaps.cend();)
64 {
65 map->second->unloadAll(true);
66 map = m_WorldMaps.erase(map);
67 }
68
69 // Instances
70 for (auto ini = m_InstancedMaps.cbegin(); ini != m_InstancedMaps.cend();)
71 {
72 ini->second->unloadAll(true);
73 ini = m_InstancedMaps.erase(ini);
74 }
75
76 m_pendingRemoveMaps.clear();
77}

79void MapMgr::removeInstance(uint32_t instanceId)
80{
81 // get the Lock so we cant Update and Delete at the same time :)
82 std::scoped_lock<std::mutex> lock(m_mapsLock);
83
84 auto ini = m_InstancedMaps.find(instanceId);
85 if (ini != m_InstancedMaps.end())
86 {
87 if (ini->second->isUnloadPending())
88 {
89 auto&& mapHolder = std::move(m_InstancedMaps.extract(ini));
90 mapHolder.mapped()->shutdownMapThread();
91 // Wait for thread to finish its work before freeing memory
92 m_pendingRemoveMaps.emplace_back(std::move(mapHolder.mapped()));
93 }
94 }
95}

110void MapMgr::update()
111{
112 std::scoped_lock<std::mutex> lock(m_mapsLock);
113 for (auto itr = m_pendingRemoveMaps.cbegin(); itr != m_pendingRemoveMaps.cend();)
114 {
115 if ((*itr)->isMapReadyForDelete())
116 itr = m_pendingRemoveMaps.erase(itr);
117 else
118 ++itr;
119 }

122void MapMgr::createBaseMap(uint32_t mapId)
123{
124 BaseMap* map = findBaseMap(mapId);
125
126 if (map == nullptr)
127 {
128 std::scoped_lock<std::mutex> lock(m_mapsLock);
129
130 // Only Create Valid Maps
131 const auto mapEntry = sMapStore.lookupEntry(mapId);
132 if (mapEntry == nullptr)
133 return;
134
135 const auto mapInfo = sMySQLStore.getWorldMapInfo(mapId);
136 if (mapInfo == nullptr)
137 return;
138
139 m_BaseMaps.insert_or_assign(mapId, std::make_unique<BaseMap>(mapId, mapInfo, mapEntry));
140
141 if (!mapEntry->isInstanceableMap())
142 {
143 m_WorldMaps.insert_or_assign(mapId, createWorldMap(mapId, worldConfig.server.mapUnloadTime * 1000));
144 }
145 }

219WorldMap* MapMgr::createInstanceForPlayer(uint32_t mapId, Player* player, uint32_t loginInstanceId /*= 0*/)
220{
221 const auto& baseMap = findBaseMap(mapId);
222 if (baseMap == nullptr)
223 return nullptr;
224
225 WorldMap* map = nullptr;
226 uint32_t newInstanceId = 0;
227
228 if (baseMap->isBattlegroundOrArena())
229 {
230 return nullptr;
231 }
232 else
233 {
234 InstancePlayerBind* pBind = player->getBoundInstance(baseMap->getMapId(), player->getDifficulty(baseMap->isRaid()));
235 InstanceSaved* pSave = pBind ? pBind->save : nullptr;
236
237 if (!pBind || !pBind->perm)
238 {
239 // if the player has a saved instance id on login,
240 // we use this instance
241 // or port him to the entrance
242 if (loginInstanceId)
243 {
244 map = findInstanceMap(loginInstanceId);
245 if (!map && pSave && pSave->getInstanceId() == loginInstanceId)
246 map = createInstance(mapId, loginInstanceId, pSave, pSave->getDifficulty(), player->getTeam());
247 return map;
248 }
249
250 if (const auto group = player->getGroup())
251 {
252 if (const InstanceGroupBind* groupBind = group->getBoundInstance(baseMap))
253 {
254 // solo instance saves should be reset when entering a group's instance
255 player->unbindInstance(baseMap->getMapId(), player->getDifficulty(baseMap->isRaid()));
256 pSave = groupBind->save;
257 }
258 }
259 }
260
261 if (pSave)
262 {
263 // solo/permanent/group saves
264 newInstanceId = pSave->getInstanceId();
265 map = findInstanceMap(newInstanceId);
266 if (!map)
267 map = createInstance(mapId, newInstanceId, pSave, pSave->getDifficulty(), player->getTeam());
268 }
269 else
270 {
271 // when we land here the Instance gets Created for the first Time
272 newInstanceId = instanceIdPool.generateId();
273 InstanceDifficulty::Difficulties diff = player->getGroup() ? player->getGroup()->getDifficulty(baseMap->isRaid()) : player->getDifficulty(baseMap->isRaid());
274 map = findInstanceMap(newInstanceId);
275 if (!map)
276 map = createInstance(mapId, newInstanceId, nullptr, diff, player->getTeam());
277 }
278 }
279
280 if (map)
281 sLogger.debug("MapMgr::createInstanceForPlayer Create Instance {} for Map {}", baseMap->getMapName(), mapId);
282
283 return map;
284}

286InstanceMap* MapMgr::createInstance(uint32_t mapId, uint32_t InstanceId, InstanceSaved* save, InstanceDifficulty::Difficulties difficulty, PlayerTeam InstanceTeam)
287{
288 // load/create a map
289 std::scoped_lock<std::mutex> lock(m_mapsLock);
290
291 // make sure we have a valid BaseMap
292 const auto& baseMap = findBaseMap(mapId);
293 if (!baseMap)
294 {
295 sLogger.failure("MapMgr::createInstance: no BaseMap for map {}", mapId);
296 return nullptr;
297 }
298
299 // make sure we have a valid map id
300 WDB::Structures::MapEntry const* entry = sMapStore.lookupEntry(mapId);
301 if (!entry)
302 {
303 sLogger.failure("MapMgr::createInstance: no entry for map {}", mapId);
304 return nullptr;
305 }
306
307 const auto mapInfo = sMySQLStore.getWorldMapInfo(mapId);
308 if (mapInfo == nullptr)
309 {
310 sLogger.failure("MapMgr::createInstance: no WorldMapInfo for map {}", mapId);
311 return nullptr;
312 }
313
314 // some instances only have one difficulty
315#if VERSION_STRING > TBC
316 getDownscaledMapDifficultyData(mapId, difficulty);
317#endif
318 sLogger.debug("MapMgr::createInstance Create {} map instance {} for {} created with difficulty {}", save ? "" : "new ", InstanceId, mapId, difficulty ? "heroic" : "normal");
319
320 auto map = std::make_unique<InstanceMap>(baseMap, mapId, worldConfig.server.mapUnloadTime * 1000, InstanceId, difficulty, InstanceTeam);
321
322 // Load Saved Respawns when existing
323 map->loadRespawnTimes();
324
325 // Initialize Map Script and Load Static Spawns
326 map->initialize();
327
328 // Load Saved Data when possible
329 bool load_data = save != nullptr;
330 map->createInstanceData(load_data);
331
332 // In Instances we load all Cells
333 map->updateAllCells(true);
334
335 // Save pointer to InstanceMap to avoid casting later -Appled
336 auto* instMap = map.get();
337 // Add current Instance to our Active Instances
338 const auto [_, emplaced] = m_InstancedMaps.try_emplace(InstanceId, std::move(map));
339
340 if (!emplaced)
341 return nullptr;
342
343 // Scheduling the new map for running
344 instMap->startMapThread();
345
346 return instMap;
347}

409EnterState MapMgr::canPlayerEnter(uint32_t mapid, uint32_t minLevel, Player* player, bool loginCheck)
410{
411 WDB::Structures::MapEntry const* entry = sMapStore.lookupEntry(mapid);
412 if (!entry)
413 return CANNOT_ENTER_NO_ENTRY;
414
415 if (!entry->isInstanceMap())
416 return CAN_ENTER;
417
418 MySQLStructure::MapInfo const* mapInfo = sMySQLStore.getWorldMapInfo(mapid);
419 if (!mapInfo && mapInfo->isWorldMap())
420 return CANNOT_ENTER_UNINSTANCED_DUNGEON;
421
422 InstanceDifficulty::Difficulties targetDifficulty, requestedDifficulty;
423 targetDifficulty = requestedDifficulty = player->getDifficulty(entry->isRaid());
424
425#if VERSION_STRING > TBC
426 // Get the highest available difficulty if current setting is higher than the instance allows
427 WDB::Structures::MapDifficulty const* mapDiff = getDownscaledMapDifficultyData(entry->id, targetDifficulty);
428 if (!mapDiff)
429 return CANNOT_ENTER_DIFFICULTY_UNAVAILABLE;
430#endif
431
432 //Bypass checks for GMs
433 if (player->isGMFlagSet())
434 return CAN_ENTER;
435
436 //Other requirements
437 if (!mapInfo || !mapInfo->hasFlag(WMI_INSTANCE_ENABLED))
438 return CANNOT_ENTER_UNSPECIFIED_REASON;
439
440 if (mapInfo->hasFlag(WMI_INSTANCE_XPACK_01) && !player->getSession()->HasFlag(ACCOUNT_FLAG_XPACK_01) && !player->getSession()->HasFlag(ACCOUNT_FLAG_XPACK_02))
441 return CANNOT_ENTER_XPACK01;
442
443 if (mapInfo->hasFlag(WMI_INSTANCE_XPACK_02) && !player->getSession()->HasFlag(ACCOUNT_FLAG_XPACK_02))
444 return CANNOT_ENTER_XPACK02;
445
446 if (minLevel && player->getLevel() < minLevel)
447 return CANNOT_ENTER_MIN_LEVEL;
448
449 if (mapInfo->required_quest_A && (player->getTeam() == TEAM_ALLIANCE) && !player->hasQuestFinished(mapInfo->required_quest_A))
450 return CANNOT_ENTER_ATTUNE_QA;
451
452 if (mapInfo->required_quest_H && (player->getTeam() == TEAM_HORDE) && !player->hasQuestFinished(mapInfo->required_quest_H))
453 return CANNOT_ENTER_ATTUNE_QH;
454
455 if (mapInfo->required_item && !player->getItemInterface()->GetItemCount(mapInfo->required_item, true))
456 return CANNOT_ENTER_ATTUNE_ITEM;
457
458 if (player->getDungeonDifficulty() >= InstanceDifficulty::DUNGEON_HEROIC &&
459 mapInfo->isMultimodeDungeon()
460 && ((mapInfo->heroic_key_1 > 0 && !player->getItemInterface()->GetItemCount(mapInfo->heroic_key_1, false))
461 && (mapInfo->heroic_key_2 > 0 && !player->getItemInterface()->GetItemCount(mapInfo->heroic_key_2, false))
462 )
463 )
464 return CANNOT_ENTER_KEY;
465
466 if (!entry->isWorldMap() && player->getDungeonDifficulty() >= InstanceDifficulty::DUNGEON_HEROIC && player->getLevel() < mapInfo->minlevel_heroic)
467 return CANNOT_ENTER_MIN_LEVEL_HC;
468
469 const auto group = player->getGroup();
470 if (entry->isRaid()) // can only enter in a raid group
471 if ((!group || !group->isRaidGroup()) && !player->m_cheats.hasTriggerpassCheat)
472 return CANNOT_ENTER_NOT_IN_RAID;
473
474 if (!player->isAlive())
475 {
476 // only let us enter when its the instance our corpse is in
477 uint32_t corpseInstance = player->getCorpseInstanceId();
478
479 const auto instance = sMapMgr.findWorldMap(mapid, corpseInstance);
480 if (instance == nullptr || instance->getBaseMap()->getMapId() != mapid)
481 return CANNOT_ENTER_CORPSE_IN_DIFFERENT_INSTANCE;
482 }
483
484 //Get instance where player's group is bound & its map
485 if (!loginCheck && group)
486 {
487 InstanceGroupBind* boundInstance = group->getBoundInstance(entry);
488 if (boundInstance && boundInstance->save)
489 if (WorldMap* boundMap = sMapMgr.findWorldMap(mapid, boundInstance->save->getInstanceId()))
490 if (EnterState denyReason = boundMap->cannotEnter(player))
491 return denyReason;
492 }
493
494 // players are only allowed to enter 5 instances per hour
495 if (entry->isInstanceMap() && (!player->getGroup() || (player->getGroup() && !player->getGroup()->isLFGGroup())))
496 {
497 uint32_t instanceIdToCheck = 0;
498 if (InstanceSaved* save = player->getInstanceSave(mapid, entry->isRaid()))
499 instanceIdToCheck = save->getInstanceId();
500
501 if (!player->checkInstanceCount(instanceIdToCheck) && !player->isDead())
502 return CANNOT_ENTER_TOO_MANY_INSTANCES;
503 }
504
505 return CAN_ENTER;
