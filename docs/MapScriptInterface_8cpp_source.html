78GameObject* MapScriptInterface::findNearestGoWithType(Object* o, uint32_t type)
79{
80 GameObject* go = nullptr;
81 float r = FLT_MAX;
82
83 for (const auto& itr : o->getInRangeObjectsSet())
84 {
85 Object* iro = itr;
86 if (!iro || !iro->isGameObject())
87 continue;
88
89 GameObject* irgo = static_cast<GameObject*>(iro);
90
91 if (irgo->getGoType() != type)
92 continue;
93
94 if ((irgo->GetPhase() & o->GetPhase()) == 0)
95 continue;
96
97 float range = o->getDistanceSq(iro);
98
99 if (range < r)
100 {
101 r = range;
102 go = irgo;
103 }
104 }
105
106 return go;
107}

109Creature* MapScriptInterface::findNearestCreature(Object* pObject, uint32_t entry, float maxSearchRange /*= 250.0f*/) const
110{
111 if (!pObject || maxSearchRange <= 0.0f)
112 return nullptr;
113
114 const float ox = pObject->GetPositionX();
115 const float oy = pObject->GetPositionY();
116 const float r2 = maxSearchRange * maxSearchRange;
117
118 const int cx = m_worldMap.getPosX(ox);
119 const int cy = m_worldMap.getPosY(oy);
120
121 const float cellSizeX = Map::Cell::_sizeX;
122 const float cellSizeY = Map::Cell::_sizeY;
123
124 const int rx = cellSizeX > 0.0f ? static_cast<int>(std::ceil(maxSearchRange / cellSizeX)) : 1;
125 const int ry = cellSizeY > 0.0f ? static_cast<int>(std::ceil(maxSearchRange / cellSizeY)) : 1;
126
127 Creature* target = nullptr;
128 float bestD2 = std::numeric_limits<float>::infinity();
129
130 for (int ix = cx - rx; ix <= cx + rx; ++ix)
131 {
132 for (int iy = cy - ry; iy <= cy + ry; ++iy)
133 {
134 MapCell* cell = m_worldMap.getCell(ix, iy);
135 if (!cell)
136 continue;
137
138 for (auto it = cell->Begin(); it != cell->End(); ++it)
139 {
140 Object* obj = *it;
141 if (!obj || !obj->isCreature())
142 continue;
143
144 if (obj->getEntry() != entry)
145 continue;
146
147 Creature* go = static_cast<Creature*>(obj);
148
149 const float dx = go->GetPositionX() - ox;
150 const float dy = go->GetPositionY() - oy;
151 const float d2 = dx * dx + dy * dy;
152
153 if (d2 > r2)
154 continue;
155
156 if (d2 < bestD2)
157 {
158 bestD2 = d2;
159 target = go;
160
161 if (bestD2 <= 1e-6f)
162 return target;
163 }
164 }
165 }
166 }
167
168 return target;
169}

290uint32_t MapScriptInterface::getPlayerCountInRadius(float x, float y, float z /* = 0.0f */, float radius /* = 5.0f */)
291{
292 // use a cell radius of 2
293 uint32_t PlayerCount = 0;
294 uint32_t cellX = m_worldMap.getPosX(x);
295 uint32_t cellY = m_worldMap.getPosY(y);
296
297 uint32_t endX = cellX < Map::Cell::_sizeX ? cellX + 1 : Map::Cell::_sizeX;
298 uint32_t endY = cellY < Map::Cell::_sizeY ? cellY + 1 : Map::Cell::_sizeY;
299 uint32_t startX = cellX > 0 ? cellX - 1 : 0;
300 uint32_t startY = cellY > 0 ? cellY - 1 : 0;
301 MapCell* pCell;
302 ObjectSet::iterator iter, iter_end;
303
304 for (uint32_t cx = startX; cx < endX; ++cx)
305 {
306 for (uint32_t cy = startY; cy < endY; ++cy)
307 {
308 pCell = m_worldMap.getCell(cx, cy);
309 if (pCell == nullptr || pCell->getPlayerCount() == 0)
310 continue;
311
312 iter = pCell->Begin();
313 iter_end = pCell->End();
314
315 for (; iter != iter_end; ++iter)
316 {
317 if ((*iter)->isPlayer() &&
318 (*iter)->CalcDistance(x, y, (z == 0.0f ? (*iter)->GetPositionZ() : z)) < radius)
319 {
320 ++PlayerCount;
321 }
322 }
323 }
324 }
325
326 return PlayerCount;
327}
