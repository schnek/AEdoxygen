37 {

138 If intersection is found within pMaxDist, sets pMaxDist to intersection distance and returns true.
139 Else, pMaxDist is not modified and returns false;
140 */
141
142 bool StaticMapTree::getIntersectionTime(const G3D::Ray& pRay, float &pMaxDist, bool pStopAtFirstHit) const
143 {
144 float distance = pMaxDist;
145 MapRayCallback intersectionCallBack(iTreeValues.get());
146 iTree.intersectRay(pRay, intersectionCallBack, distance, pStopAtFirstHit);
147 if (intersectionCallBack.didHit())
148 pMaxDist = distance;
149 return intersectionCallBack.didHit();
150 }

153 bool StaticMapTree::isInLineOfSight(const Vector3& pos1, const Vector3& pos2) const
154 {
155 float maxDist = (pos2 - pos1).magnitude();
156 // return false if distance is over max float, in case of cheater teleporting to the end of the universe
157 if (maxDist == std::numeric_limits<float>::max() || !std::isfinite(maxDist))
158 return false;
159
160 // valid map coords should *never ever* produce float overflow, but this would produce NaNs too
161 ASSERT(maxDist < std::numeric_limits<float>::max());
162 // prevent NaN values which can cause BIH intersection to enter infinite loop
163 if (maxDist < 1e-10f)
164 return true;
165 // direction with length of 1
166 G3D::Ray ray = G3D::Ray::fromOriginAndDirection(pos1, (pos2 - pos1)/maxDist);
167 if (getIntersectionTime(ray, maxDist, true))
168 return false;
169
170 return true;
171 }

173 /**
174 When moving from pos1 to pos2 check if we hit an object. Return true and the position if we hit one
175 Return the hit pos or the original dest pos
176 */
177
178 bool StaticMapTree::getObjectHitPos(const Vector3& pPos1, const Vector3& pPos2, Vector3& pResultHitPos, float pModifyDist) const
179 {
180 bool result=false;
181 float maxDist = (pPos2 - pPos1).magnitude();
182 // valid map coords should *never ever* produce float overflow, but this would produce NaNs too
183 ASSERT(maxDist < std::numeric_limits<float>::max());
184 // prevent NaN values which can cause BIH intersection to enter infinite loop
185 if (maxDist < 1e-10f)
186 {
187 pResultHitPos = pPos2;
188 return false;
189 }
190 Vector3 dir = (pPos2 - pPos1)/maxDist; // direction with length of 1
191 G3D::Ray ray(pPos1, dir);
192 float dist = maxDist;
193 if (getIntersectionTime(ray, dist, false))
194 {
195 pResultHitPos = pPos1 + dir * dist;
196 if (pModifyDist < 0)
197 {
198 if ((pResultHitPos - pPos1).magnitude() > -pModifyDist)
199 {
200 pResultHitPos = pResultHitPos + dir*pModifyDist;
201 }
202 else
203 {
204 pResultHitPos = pPos1;
205 }
206 }
207 else
208 {
209 pResultHitPos = pResultHitPos + dir*pModifyDist;
210 }
211 result = true;
212 }
213 else
214 {
215 pResultHitPos = pPos2;
216 result = false;
217 }
218 return result;
219 }

223 float StaticMapTree::getHeight(const Vector3& pPos, float maxSearchDist) const
224 {
225 float height = G3D::finf();
226 Vector3 dir = Vector3(0, 0, -1);
227 G3D::Ray ray(pPos, dir); // direction with length of 1
228 float maxDist = maxSearchDist;
229 if (getIntersectionTime(ray, maxDist, false))
230 {
231 height = pPos.z - maxDist;
232 }
233 return(height);
234 }

238 bool StaticMapTree::CanLoadMap(const std::string &vmapPath, uint32_t mapID, uint32_t tileX, uint32_t tileY)
239 {
240 std::string basePath = vmapPath;
241 if (basePath.length() > 0 && basePath[basePath.length()-1] != '/' && basePath[basePath.length()-1] != '\\')
242 basePath.push_back('/');
243 std::string fullname = basePath + VMapManager2::getMapFileName(mapID);
244 bool success = true;
245 FILE* rf = fopen(fullname.c_str(), "rb");
246 if (!rf)
247 return false;
248 /// @todo check magic number when implemented...
249 char tiled;
250 char chunk[8];
251 if (!readChunk(rf, chunk, VMAP_MAGIC, 8) || fread(&tiled, sizeof(char), 1, rf) != 1)
252 {
253 fclose(rf);
254 return false;
255 }
256 if (tiled)
257 {
258 std::string tilefile = basePath + getTileFileName(mapID, tileX, tileY);
259 FILE* tf = fopen(tilefile.c_str(), "rb");
260 if (!tf)
261 success = false;
262 else
263 {
264 if (!readChunk(tf, chunk, VMAP_MAGIC, 8))
265 success = false;
266 fclose(tf);
267 }
268 }
269 fclose(rf);
270 return success;
271 }

275 bool StaticMapTree::InitMap(const std::string &fname, VMapManager2* vm)
276 {
277 sLogger.debug("initializing StaticMapTree '{}'", fname);
278 bool success = false;
279 std::string fullname = iBasePath + fname;
280 FILE* rf = fopen(fullname.c_str(), "rb");
281 if (!rf)
282 return false;
283
284 char chunk[8];
285 char tiled = '\0';
286
287 if (readChunk(rf, chunk, VMAP_MAGIC, 8) && fread(&tiled, sizeof(char), 1, rf) == 1 &&
288 readChunk(rf, chunk, "NODE", 4) && iTree.readFromFile(rf))
289 {
290 iNTreeValues = iTree.primCount();
291 iTreeValues = std::make_unique<ModelInstance[]>(iNTreeValues);
292 success = readChunk(rf, chunk, "GOBJ", 4);
293 }
294
295 iIsTiled = tiled != '\0';
296
297 // global model spawns
298 // only non-tiled maps have them, and if so exactly one (so far at least...)
299 ModelSpawn spawn;
300#ifdef VMAP_DEBUG
301 sLogger.debug("map isTiled: {}", static_cast<uint32_t>(iIsTiled));
302#endif
303 if (!iIsTiled && ModelSpawn::readFromFile(rf, spawn))
304 {
305 WorldModel* model = vm->acquireModelInstance(iBasePath, spawn.name);
306 sLogger.debug("loading {}", spawn.name);
307 if (model)
308 {
309 // assume that global model always is the first and only tree value (could be improved...)
310 iTreeValues[0] = ModelInstance(spawn, model);
311 iLoadedSpawns[0] = 1;
312 }
313 else
314 {
315 success = false;
316 sLogger.failure("could not acquire WorldModel pointer for '{}'", spawn.name);
317 }
318 }
319
320 fclose(rf);
321 return success;
322 }

340 bool StaticMapTree::LoadMapTile(uint32_t tileX, uint32_t tileY, VMapManager2* vm)
341 {
342 if (!iIsTiled)
343 {
344 // currently, core creates grids for all maps, whether it has terrain tiles or not
345 // so we need "fake" tile loads to know when we can unload map geometry
346 iLoadedTiles[packTileID(tileX, tileY)] = false;
347 return true;
348 }
349 if (!iTreeValues)
350 {
351 sLogger.failure("tree has not been initialized [{}, {}]", tileX, tileY);
352 return false;
353 }
354 bool result = true;
355
356 std::string tilefile = iBasePath + getTileFileName(iMapID, tileX, tileY);
357 FILE* tf = fopen(tilefile.c_str(), "rb");
358 if (tf)
359 {
360 char chunk[8];
361
362 if (!readChunk(tf, chunk, VMAP_MAGIC, 8))
363 result = false;
364 uint32_t numSpawns = 0;
365 if (result && fread(&numSpawns, sizeof(uint32_t), 1, tf) != 1)
366 result = false;
367 for (uint32_t i=0; i<numSpawns && result; ++i)
368 {
369 // read model spawns
370 ModelSpawn spawn;
371 result = ModelSpawn::readFromFile(tf, spawn);
372 if (result)
373 {
374 // acquire model instance
375 WorldModel* model = vm->acquireModelInstance(iBasePath, spawn.name);
376 if (!model)
377 sLogger.failure("could not acquire WorldModel pointer [{}, {}]", tileX, tileY);
378
379 // update tree
380 uint32_t referencedVal;
381
382 if (fread(&referencedVal, sizeof(uint32_t), 1, tf) == 1)
383 {
384 if (!iLoadedSpawns.count(referencedVal))
385 {
386 if (referencedVal > iNTreeValues)
387 {
388 sLogger.failure("invalid tree element ({}/{}) referenced in tile {}", referencedVal, iNTreeValues, tilefile);
389 continue;
390 }
391
392 iTreeValues[referencedVal] = ModelInstance(spawn, model);
393 iLoadedSpawns[referencedVal] = 1;
394 }
395 else
396 {
397 ++iLoadedSpawns[referencedVal];
398#ifdef VMAP_DEBUG
399 if (iTreeValues[referencedVal].ID != spawn.ID)
400 sLogger.debug("trying to load wrong spawn in node");
401 else if (iTreeValues[referencedVal].name != spawn.name)
402 sLogger.debug("name collision on GUID={}", spawn.ID);
403#endif
404 }
405 }
406 else
407 result = false;
408 }
409 }
410 iLoadedTiles[packTileID(tileX, tileY)] = true;
411 fclose(tf);
412 }
413 else
414 iLoadedTiles[packTileID(tileX, tileY)] = false;
415 return result;
416 }

420 void StaticMapTree::UnloadMapTile(uint32_t tileX, uint32_t tileY, VMapManager2* vm)
421 {
422 uint32_t tileID = packTileID(tileX, tileY);
423 loadedTileMap::iterator tile = iLoadedTiles.find(tileID);
424 if (tile == iLoadedTiles.end())
425 {
426 sLogger.failure("trying to unload non-loaded tile - Map:{} X:{} Y:{}", iMapID, tileX, tileY);
427 return;
428 }
429 if (tile->second) // file associated with tile
430 {
431 std::string tilefile = iBasePath + getTileFileName(iMapID, tileX, tileY);
432 FILE* tf = fopen(tilefile.c_str(), "rb");
433 if (tf)
434 {
435 bool result=true;
436 char chunk[8];
437 if (!readChunk(tf, chunk, VMAP_MAGIC, 8))
438 result = false;
439 uint32_t numSpawns;
440 if (fread(&numSpawns, sizeof(uint32_t), 1, tf) != 1)
441 result = false;
442 for (uint32_t i=0; i<numSpawns && result; ++i)
443 {
444 // read model spawns
445 ModelSpawn spawn;
446 result = ModelSpawn::readFromFile(tf, spawn);
447 if (result)
448 {
449 // release model instance
450 vm->releaseModelInstance(spawn.name);
451
452 // update tree
453 uint32_t referencedNode;
454
455 if (fread(&referencedNode, sizeof(uint32_t), 1, tf) != 1)
456 result = false;
457 else
458 {
459 if (!iLoadedSpawns.count(referencedNode))
460 sLogger.failure("trying to unload non-referenced model '{}' (ID:{})", spawn.name, spawn.ID);
461 else if (--iLoadedSpawns[referencedNode] == 0)
462 {
463 iTreeValues[referencedNode].setUnloaded();
464 iLoadedSpawns.erase(referencedNode);
465 }
466 }
467 }
468 }
469 fclose(tf);
470 }
471 }
472 iLoadedTiles.erase(tile);
473 }
