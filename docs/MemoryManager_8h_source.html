29public:
30
31 typedef shared_ptr<class MemoryManager> Ref;

33 /** Return a pointer to \a s bytes of memory that are unused by
34 the rest of the program. The contents of the memory are
35 undefined */
36 virtual void* alloc(size_t s);
37
38 /** Invoke to declare that this memory will no longer be used by
39 the program. The memory manager is not required to actually
40 reuse or release this memory. */
41 virtual void free(void* ptr);
42
43 /** Returns true if this memory manager is threadsafe (i.e., alloc
44 and free can be called asychronously) */
45 virtual bool isThreadsafe() const;
46
47 /** Return the instance. There's only one instance of the default
48 MemoryManager; it is cached after the first creation. */
49 static MemoryManager::Ref create();
50};
