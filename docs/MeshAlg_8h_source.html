32/**
33 Indexed <B>mesh alg</B>orithms. You have to build your own mesh class.
34 <P>
35 No mesh class is provided with G3D because there isn't an "ideal"
36 mesh format-- one application needs keyframed animation, another
37 skeletal animation, a third texture coordinates, a fourth
38 cannot precompute information, etc. Instead of compromising, this
39 class implements the hard parts of mesh computation and you can write

50 /** Adjacency information for a vertex.
51 Does not contain the vertex position or normal,
52 which are stored in the MeshAlg::Geometry object.
53 <CODE>Vertex</CODE>s must be stored in an array
54 parallel to (indexed in the same way as)

61 /**
62 Array of edges adjacent to this vertex.
63 Let e = edgeIndex[i].
64 edge[(e >= 0) ? e : ~e].vertexIndex[0] == this
65 vertex index.
66
67 Edges may be listed multiple times if they are
68 degenerate.
69 */
70 SmallArray<int, 6> edgeIndex;
71
72 /**
73 Returns true if e or ~e is in the edgeIndex list.
74 */
75 inline bool inEdge(int e) const {
76 return edgeIndex.contains(~e) || edgeIndex.contains(e);
77 }

80 Array of faces containing this vertex. Faces
81 may be listed multiple times if they are degenerate.

92 /**
93 Oriented, indexed triangle.
94 */
95 class Face {
96 public:
97 Face();
98
99 /**
100 Used by Edge::faceIndex to indicate a missing face.
101 This is a large negative value.
102 */
103 static const int NONE;
104
105
106 /**
107 Vertices in the face in counter-clockwise order.
108 Degenerate faces may include the same vertex multiple times.
109 */
110 int vertexIndex[3];
111
112 inline bool containsVertex(int v) const {
113 return contains(vertexIndex, 3, v);
114 }
115
116 /**
117 Edge indices in counter-clockwise order. Edges are
118 undirected, so it is important to know which way
119 each edge is pointing in a face. This is encoded
120 using negative indices.
121
122 If <CODE>edgeIndex[i] >= 0</CODE> then this face
123 contains the directed edge
124 between vertex indices
125 <CODE>edgeArray[face.edgeIndex[i]].vertexIndex[0]</CODE>
126 and
127 <CODE>edgeArray[face.edgeIndex[i]].vertexIndex[1]</CODE>.
128
129 If <CODE>edgeIndex[i] < 0</CODE> then
130 <CODE>~edgeIndex[i]</CODE> (i.e. the two's
131 complement of) is used and this face contains the directed
132 edge between vertex indices
133 <CODE>edgeArray[~face.edgeIndex[i]].vertexIndex[0]</CODE>
134 and
135 <CODE>edgeArray[~face.edgeIndex[i]].vertexIndex[1]</CODE>.
136
137 Degenerate faces may include the same edge multiple times.
138 */
139 // Temporarily takes on the value Face::NONE during adjacency
140 // computation to indicate an edge that has not yet been assigned.
141 int edgeIndex[3];
142
143 inline bool containsEdge(int e) const {
144 if (e < 0) {
145 e = ~e;
146 }
147 return contains(edgeIndex, 3, e) || contains(edgeIndex, 3, ~e);
148 }
149
150 /** Contains the forward edge e if e >= 0 and the backward edge
151 ~e otherwise. */
152 inline bool containsDirectedEdge(int e) const {
153 return contains(edgeIndex, 3, e);
154 }
155 };

158 /** Oriented, indexed edge */
159 class Edge {
160 public:
161 Edge();
162
163 /** Degenerate edges may include the same vertex times. */
164 int vertexIndex[2];
165
166 inline bool containsVertex(int v) const {
167 return contains(vertexIndex, 2, v);
168 }
169
170 /**
171 The edge is directed <B>forward</B> in face 0
172 <B>backward</B> in face 1. Face index of MeshAlg::Face::NONE
173 indicates a boundary (a.k.a. crack, broken) edge.
174 */
175 int faceIndex[2];
176
177 /** Returns true if f is contained in the faceIndex array in either slot.
178 To see if it is forward in that face, just check edge.faceIndex[0] == f.*/
179 inline bool inFace(int f) const {
180 return contains(faceIndex, 2, f);
181 }
182
183 /**
184 Returns true if either faceIndex is NONE.
185 */
186 inline bool boundary() const {
187 return (faceIndex[0] == Face::NONE) ||
188 (faceIndex[1] == Face::NONE);
189 }
190
191 /**
192 Returns the reversed edge.
193 */
194 inline Edge reverse() const {
195 Edge e;
196 e.vertexIndex[0] = vertexIndex[1];
197 e.vertexIndex[1] = vertexIndex[0];
198 e.faceIndex[0] = faceIndex[1];
199 e.faceIndex[1] = faceIndex[0];
200 return e;
201 }
202 };

399 /**
400 Welds nearby and colocated elements of the <I>oldVertexArray</I> together so that
401 <I>newVertexArray</I> contains no vertices within <I>radius</I> of one another.
402 Every vertex in newVertexPositions also appears in oldVertexPositions.
403 This is useful for downsampling meshes and welding cracks created by artist errors
404 or numerical imprecision.
405
406 The two integer arrays map indices back and forth between the arrays according to:
407 <PRE>
408 oldVertexArray[toOld[ni]] == newVertexArray[ni]
409 oldVertexArray[oi] == newVertexArray[toNew[ni]]
410 </PRE>
411
412 Note that newVertexPositions is never longer than oldVertexPositions
413 and is shorter when vertices are welded.
414
415 Welding with a large radius will effectively compute a lower level of detail for
416 the mesh.
417
418 The welding method runs in roughly linear time in the length of oldVertexArray--
419 a uniform spatial grid is used to achieve nearly constant time vertex collapses
420 for uniformly distributed vertices.
421
422 It is sometimes desirable to keep the original vertex ordering but
423 identify the unique vertices. The following code computes
424 array canonical s.t. canonical[v] = first occurance of
425 a vertex near oldVertexPositions[v] in oldVertexPositions.
426
427 <PRE>
428 Array<int> canonical(oldVertexPositions.size()), toNew, toOld;
429 computeWeld(oldVertexPositions, Array<Vector3>(), toNew, toOld, radius);
430 for (int v = 0; v < canonical.size(); ++v) {
431 canonical[v] = toOld[toNew[v]];
432 }
433 </PRE>
434
435 See also G3D::MeshAlg::weldAdjacency.
436
437 @cite The method is that described as the 'Grouper' in Baum, Mann, Smith, and Winget,
438 Making Radiosity Usable: Automatic Preprocessing and Meshing Techniques for
439 the Generation of Accurate Radiosity Solutions, Computer Graphics vol 25, no 4, July 1991.
440
441 @deprecated Use weld.
442 */
443 static void computeWeld(
444 const Array<Vector3>& oldVertexPositions,
445 Array<Vector3>& newVertexPositions,
446 Array<int>& toNew,
447 Array<int>& toOld,
448 float radius = fuzzyEpsilon32);

561 /** Converts quadlist (QUADS),
562 triangle fan (TRIANGLE_FAN),
563 tristrip(TRIANGLE_STRIP), and quadstrip (QUAD_STRIP) indices into
564 triangle list (TRIANGLES) indices and appends them to outIndices. */
565 template<class IndexType>
566 static void toIndexedTriList(
567 const Array<IndexType>& inIndices,
568 MeshAlg::Primitive inType,
569 Array<IndexType>& outIndices) {
570
571 debugAssert(
572 inType == PrimitiveType::TRIANGLE_STRIP ||
573 inType == PrimitiveType::TRIANGLE_FAN ||
574 inType == PrimitiveType::QUADS ||
575 inType == PrimitiveType::QUAD_STRIP);
576
577 const int inSize = inIndices.size();
578
579 switch(inType) {
580 case PrimitiveType::TRIANGLE_FAN:
581 {
582 debugAssert(inSize >= 3);
583
584 int N = outIndices.size();
585 outIndices.resize(N + (inSize - 2) * 3);
586
587 for (IndexType i = 1, outIndex = N; i <= (inSize - 2); ++i, outIndex += 3) {
588 outIndices[outIndex] = inIndices[0];
589 outIndices[outIndex + 1] = inIndices[i];
590 outIndices[outIndex + 2] = inIndices[i + 1];
591 }
592
593 break;
594 }
595
596 case PrimitiveType::TRIANGLE_STRIP:
597 {
598 debugAssert(inSize >= 3);
599
600 int N = outIndices.size();
601 outIndices.resize(N + (inSize - 2) * 3);
602
603 bool atEven = false;
604 for (IndexType i = 0, outIndex = N; i < (inSize - 2); ++i, outIndex += 3) {
605 if (atEven) {
606 outIndices[outIndex] = inIndices[i + 1];
607 outIndices[outIndex + 1] = inIndices[i];
608 outIndices[outIndex + 2] = inIndices[i + 2];
609 atEven = false;
610 } else {
611 outIndices[outIndex] = inIndices[i];
612 outIndices[outIndex + 1] = inIndices[i + 1];
613 outIndices[outIndex + 2] = inIndices[i + 2];
614 atEven = true;
615 }
616 }
617
618 break;
619 }
620
621 case PrimitiveType::QUADS:
622 {
623 debugAssert(inIndices.size() >= 4);
624
625 int N = outIndices.size();
626 outIndices.resize(N + (inSize / 4) * 3);
627
628 for (IndexType i = 0, outIndex = N; i <= (inSize - 4); i += 4, outIndex += 6) {
629 outIndices[outIndex] = inIndices[i];
630 outIndices[outIndex + 1] = inIndices[i + 1];
631 outIndices[outIndex + 2] = inIndices[i + 3];
632 outIndices[outIndex + 3] = inIndices[i + 1];
633 outIndices[outIndex + 4] = inIndices[i + 2];
634 outIndices[outIndex + 5] = inIndices[i + 3];
635 }
636
637 break;
638 }
639
640 case PrimitiveType::QUAD_STRIP:
641 {
642 debugAssert(inIndices.size() >= 4);
643
644 int N = outIndices.size();
645 outIndices.resize(N + (inSize - 2) * 3);
646
647 for (IndexType i = 0, outIndex = N; i <= (inSize - 2); i += 2, outIndex += 6) {
648 outIndices[outIndex] = inIndices[i];
649 outIndices[outIndex + 1] = inIndices[i + 1];
650 outIndices[outIndex + 2] = inIndices[i + 2];
651 outIndices[outIndex + 3] = inIndices[i + 2];
652 outIndices[outIndex + 4] = inIndices[i + 1];
653 outIndices[outIndex + 5] = inIndices[i + 3];
654 }
655 break;
656 }
657 default:
658 alwaysAssertM(false, "Illegal argument");
659 }
660 }
