30bool ChatCommandHandler::handleCommandsCommand(const char* args, WorldSession* m_session)
31{
32 std::string output;
33 uint32_t count = 0;
34
35 output = "Available commands: \n\n";
36
37 for (const auto table : sCommandTableStorage.getCommandRegistry())
38 {
39 std::string top;
40 if (*args && !resolveTopLevelAbbrev(args, m_session, top))
41 continue;
42
43 if (table.commandPermission == "z")
44 {
45 output += "|cffff6060";
46 output += table.command;
47 output += "|r, ";
48 }
49 else if (table.commandPermission == "m")
50 {
51 output += "|cff00ffff";
52 output += table.command;
53 output += "|r, ";
54 }
55 else if (table.commandPermission == "c")
56 {
57 output += "|cff00ff00";
58 output += table.command;
59 output += "|r, ";
60 }
61 else
62 {
63 output += "|cff00ccff";
64 output += table.command;
65 output += "|r, ";
66 }
67
68 count++;
69 if (count == 5) // 5 per line
70 {
71 output += "\n";
72 count = 0;
73 }
74 }
75 if (count)
76 output += "\n";
77
78 SendMultilineMessage(m_session, output.c_str());
79
80 return true;
81}

83bool ChatCommandHandler::showHelpForCommand(WorldSession* m_session, const char* args)
84{
85 auto &reg = sCommandTableStorage.getCommandRegistry();
86
87 // normalize input: strip leading spaces and an optional dot
88 std::string_view sv = args ? std::string_view(args) : std::string_view{};
89 size_t i = 0;
90
91 while (i < sv.size() && std::isspace(static_cast<unsigned char>(sv[i])))
92 ++i;
93
94 // be consequent if you allow '.' or '!' in the normalization, you need to allow it here too
95 if (i < sv.size() && (sv[i] == '.' || sv[i] == '!'))
96 ++i;
97
98 while (i < sv.size() && std::isspace(static_cast<unsigned char>(sv[i])))
99 ++i;
100
101 sv.remove_prefix(i);
102 if (sv.empty())
103 return false;
104
105 // split input into tokens (views)
106 std::vector<std::string_view> tokens;
107 {
108 size_t p = 0;
109 while (p < sv.size())
110 {
111 while (p < sv.size() && std::isspace(static_cast<unsigned char>(sv[p])))
112 ++p;
113
114 if (p >= sv.size())
115 break;
116
117 size_t q = p;
118 while (q < sv.size() && !std::isspace(static_cast<unsigned char>(sv[q])))
119 ++q;
120
121 tokens.emplace_back(sv.substr(p, q - p));
122 p = q;
123 }
124 }
125
126 if (tokens.empty())
127 return false;
128
129 // resolve ONLY the first token to a top-level command
130 std::string top;
131 if (!resolveTopLevelAbbrev(tokens[0], m_session, top))
132 return false; // unknown/ambiguous top-level
133
134 auto lo = [](char c) {
135 return static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
136 };
137
138 auto istarts_with = [&](std::string_view a, std::string_view b) {
139 if (b.size() > a.size())
140 return false;
141
142 for (size_t k = 0; k < b.size(); ++k)
143 if (lo(a[k]) != lo(b[k]))
144 return false;
145
146 return true;
147 };
148
149 // exact depth we want help for (do NOT auto-descend)
150 const size_t wantedDepth = tokens.size();
151
152 // Try to find an entry exactly at that depth: first word == resolved top-level,
153 // the remaining words are matched by abbreviation.
154 const ChatCommand* match = nullptr;
155 std::string matchCmd;
156
157 for (const auto& e : reg)
158 {
159 if (e.command.empty())
160 continue;
161
162 // split command into words
163 std::vector<std::string> words;
164 {
165 std::istringstream is(e.command);
166 for (std::string w; is >> w; )
167 words.push_back(std::move(w));
168 }
169
170 if (words.size() != wantedDepth)
171 continue; // exact depth only
172
173 if (words.empty())
174 continue;
175
176 // first word must equal resolved top (case-insensitive)
177 if (words[0].size() != top.size())
178 {
179 // handled below by per-char compare anyway
180 }
181 {
182 bool eq = true;
183 for (size_t k=0; k<top.size(); ++k)
184 if (lo(words[0][k]) != lo(top[k])) { eq = false; break; }
185 if (!eq) continue;
186 }
187
188 // remaining words matched by abbreviation
189 bool ok = true;
190 for (size_t w = 1; w < wantedDepth; ++w)
191 {
192 if (!istarts_with(words[w], tokens[w]))
193 {
194 ok = false;
195 break;
196 }
197 }
198
199 if (!ok)
200 continue;
201
202 const char perm = e.commandPermission.empty() ? '0' : e.commandPermission[0];
203 if (perm != '0' && !m_session->CanUseCommand(perm))
204 continue;
205
206 match = &e;
207 matchCmd = e.command;
208 break;
209 }
210
211 if (match)
212 {
213 // exact-depth entry exists → show its help (or say there is none)
214 if (!match->help.empty())
215 SendMultilineMessage(m_session, match->help.c_str());
216 else
217 systemMessage(m_session, "There is no help for that command");
218
219 return true;
220 }
221
222 // No exact-depth entry → treat as a node: list children under that prefix.
223 // Build the typed prefix string: <top> + the typed (abbrev) tokens joined by spaces.
224 std::string typedPrefix = top;
225 for (size_t t = 1; t < tokens.size(); ++t)
226 {
227 typedPrefix.push_back(' ');
228 typedPrefix.append(tokens[t].data(), tokens[t].size());
229 }
230
231 // But for listing children we need the *resolved* prefix (first word = resolved top),
232 // without auto-resolving further words. So children start with "<top> ".
233 const std::string listPrefix = top + ' ';
234
235 bool any = false;
236 for (const auto &cmd : reg)
237 {
238 if (cmd.command.size() <= listPrefix.size())
239 continue;
240
241 // starts with "<top> " (case-insensitive)
242 bool starts = true;
243 for (size_t k = 0; k < listPrefix.size(); ++k)
244 {
245 if (lo(cmd.command[k]) != lo(listPrefix[k]))
246 {
247 starts = false;
248 break;
249 }
250 }
251
252 if (!starts)
253 continue;
254
255 const char perm = cmd.commandPermission.empty() ? '0' : cmd.commandPermission[0];
256 if (perm != '0' && !m_session->CanUseCommand(perm))
257 continue;
258
259 // show only the remainder after "<top> "
260 auto help = cmd.help.empty() ? "No Help Available" : cmd.help;
261
262 blueSystemMessage(m_session, " {} - {}", cmd.command.substr(listPrefix.size()), help);
263
264 any = true;
265 }
266
267 if (!any)
268 systemMessage(m_session, "There is no help for that command");
269 else
270 greenSystemMessage(m_session, "Available Subcommands:");
271
272 return true;
273}

501//.kill - kills target or player with <name>
502bool ChatCommandHandler::HandleKillCommand(const char* args, WorldSession* m_session)
503{
504 bool is_name_set = false;
505 auto unit_target = GetSelectedUnit(m_session);
506
507 if (*args)
508 is_name_set = true;
509
510 if (is_name_set)
511 {
512 auto named_player = sObjectMgr.getPlayer(args, false);
513 if (named_player == nullptr)
514 {
515 redSystemMessage(m_session, "Player {} is not online or does not exist!", args);
516 return true;
517 }
518 named_player->die(nullptr, 0, 0);
519 redSystemMessage(named_player->getSession(), "You were killed by {} with a GM command.", m_session->GetPlayer()->getName());
520 greenSystemMessage(m_session, "Killed player {}.", args);
521 sGMLog.writefromsession(m_session, "used kill command on Player Name: %s Guid: %s ", named_player->getName().c_str(), std::to_string(named_player->getGuid()).c_str());
522 }
523 else
524 {
525 if (unit_target != nullptr)
526 {
527 switch (unit_target->getObjectTypeId())
528 {
529 case TYPEID_UNIT:
530 {
531 auto creature = static_cast<Creature*>(unit_target);
532 auto kill_spell = sSpellMgr.getSpellInfo(5);
533 if (kill_spell == nullptr)
534 return true;
535
536 SpellCastTargets targets(unit_target->getGuid());
537 Spell* spell = sSpellMgr.newSpell(m_session->GetPlayer(), kill_spell, true, 0);
538 spell->prepare(&targets);
539
540 greenSystemMessage(m_session, "Killed Creature {}.", creature->GetCreatureProperties()->Name);
541 sGMLog.writefromsession(m_session, "used kill command on Creature %u [%s], spawn ID: %u", creature->getEntry(), creature->GetCreatureProperties()->Name.c_str(), creature->spawnid);
542 break;
543 }
544 case TYPEID_PLAYER:
545 {
546 auto player = static_cast<Player*>(unit_target);
547
548 player->setHealth(0);
549 player->die(nullptr, 0, 0);
550 redSystemMessage(player->getSession(), "You were killed by {} with a GM command.", m_session->GetPlayer()->getName());
551 greenSystemMessage(m_session, "Killed player {}.", player->getName());
552 sGMLog.writefromsession(m_session, "used kill command on Player Name: %s Guid: %s", m_session->GetPlayer()->getName().c_str(), std::to_string(player->getGuid()).c_str());
553 break;
554 }
555 default:
556 redSystemMessage(m_session, "Something went wrong while killing a unit with this command!");
557 break;
558
559 }
560 }
561 else
562 {
563 redSystemMessage(m_session, "No selected target or player name set.");
564 redSystemMessage(m_session, "Use .kill on an selected unit (player/creature)");
565 redSystemMessage(m_session, "Use .kill <playername> to kill players");
566 return true;
567 }
568
569 }
570
571 return true;
572}
