24// Generic feign death spells, can be added with/without dynamic flag and prevent chat emote flag
25class GenericFeignDeath : public SpellScript
26{
27public:
28 GenericFeignDeath(bool preventChatEmotes = true, bool withDynamicFlag = true) : _preventChatEmotes(preventChatEmotes), _withDynamicFlag(withDynamicFlag) {}
29
30 SpellScriptCheckDummy onAuraDummyEffect(Aura* aur, AuraEffectModifier* /*aurEff*/, bool apply) override
31 {
32 if (apply)
33 {
34#if VERSION_STRING >= TBC
35 aur->getOwner()->addUnitFlags2(UNIT_FLAG2_FEIGN_DEATH);
36#endif
37 if (_withDynamicFlag)
38 aur->getOwner()->addDynamicFlags(U_DYN_FLAG_DEAD);
39 if (_preventChatEmotes)
40 aur->getOwner()->addUnitFlags(UNIT_FLAG_FEIGN_DEATH);
41 }
42 else
43 {
44#if VERSION_STRING >= TBC
45 aur->getOwner()->removeUnitFlags2(UNIT_FLAG2_FEIGN_DEATH);
46#endif
47 if (_withDynamicFlag)
48 aur->getOwner()->removeDynamicFlags(U_DYN_FLAG_DEAD);
49 if (_preventChatEmotes)
50 aur->getOwner()->removeUnitFlags(UNIT_FLAG_FEIGN_DEATH);
51 }
52
53 return SpellScriptCheckDummy::DUMMY_OK;
54 }
55
56private:
57 bool _preventChatEmotes = true;
58 bool _withDynamicFlag = true;
59};

61// Polymorph spells used by creatures
62class Polymorph : public SpellScript
63{
64public:
65 Polymorph(bool requiresDismount, bool regeneratesHealth) : _dismount(requiresDismount), _health(regeneratesHealth) {}
66
67 SpellScriptExecuteState beforeAuraEffect(Aura* aur, AuraEffectModifier* aurEff, bool apply) override
68 {
69 if (aurEff->getAuraEffectType() != SPELL_AURA_TRANSFORM)
70 return SpellScriptExecuteState::EXECUTE_OK;
71
72 if (apply)
73 {
74 if (_dismount)
75 {
76 // todo: creatures
77 if (aur->getPlayerOwner() != nullptr)
78 aur->getPlayerOwner()->dismount();
79 }
80
81 if (_health)
82 aur->getOwner()->addUnitStateFlag(UNIT_STATE_POLYMORPHED);
83 }
84 else
85 {
86 if (_health)
87 aur->getOwner()->removeUnitStateFlag(UNIT_STATE_POLYMORPHED);
88 }
89
90 return SpellScriptExecuteState::EXECUTE_OK;
91 }
92
93private:
94 bool _dismount = false;
95 bool _health = false;
96};
