20UnitSpeedType SelectSpeedType(uint32_t moveFlags)
21{
22 if (moveFlags & MOVEFLAG_FLYING)
23 {
24 if (moveFlags & MOVEFLAG_MOVE_BACKWARD )
25 return TYPE_FLY_BACK;
26 else
27 return TYPE_FLY;
28 }
29 else if (moveFlags & MOVEFLAG_SWIMMING)
30 {
31 if (moveFlags & MOVEFLAG_MOVE_BACKWARD)
32 return TYPE_SWIM_BACK;
33 else
34 return TYPE_SWIM;
35 }
36 else if (moveFlags & MOVEFLAG_WALK)
37 {
38 return TYPE_WALK;
39 }
40 else if (moveFlags & MOVEFLAG_MOVE_BACKWARD)
41 {
42 return TYPE_RUN_BACK;
43 }
44
45 // Flying creatures use MOVEFLAG_CAN_FLY or MOVEFLAG_DISABLE_GRAVITY
46 // Run speed is their default flight speed.
47 return TYPE_RUN;
48}

50int32_t MoveSplineInit::Launch()
51{
52 MoveSpline& move_spline = *unit->movespline;
53
54 // Elevators also use MOVEFLAG_TRANSPORT but we do not keep track of their position changes
55#if VERSION_STRING <= WotLK
56 bool transport = unit->hasUnitMovementFlag(MOVEFLAG_TRANSPORT) && unit->getTransGuid();
57#else
58 bool transport = unit->getTransGuid() != 0;
59#endif
60 Location real_position;
61 // there is a big chance that current position is unknown if current state is not finalized, need compute it
62 // this also allows CalculatePath spline position and update map position in much greater intervals
63 // Don't compute for transport movement if the unit is in a motion between two transports
64 if (!move_spline.Finalized() && move_spline.onTransport == transport)
65 {
66 real_position = move_spline.ComputePosition();
67 }
68 else
69 {
70 LocationVector const pos = transport ? unit->obj_movement_info.transport_position : unit->GetPosition();
71 real_position.x = pos.getPositionX();
72 real_position.y = pos.getPositionY();
73 real_position.z = pos.getPositionZ();
74 real_position.orientation = unit->GetOrientation();
75 }
76
77 // should i do the things that user should do? - no.
78 if (args.path.empty())
79 return 0;
80
81 // corrent first vertex
82 args.path[0] = real_position;
83 args.initialOrientation = real_position.orientation;
84 move_spline.onTransport = transport;
85
86#if VERSION_STRING == WotLK
87 args.flags.enter_cycle = args.flags.cyclic;
88#endif
89
90 uint32_t moveFlags = unit->obj_movement_info.getMovementFlags();
91#if VERSION_STRING <= TBC
92 moveFlags |= MOVEFLAG_SPLINE_ENABLED;
93 moveFlags = (moveFlags & ~(MOVEFLAG_MOVE_BACKWARD)) | MOVEFLAG_MOVE_FORWARD;
94#elif VERSION_STRING == WotLK
95 moveFlags |= MOVEFLAG_SPLINE_ENABLED;
96
97 if (!args.flags.backward)
98 moveFlags = (moveFlags & ~(MOVEFLAG_MOVE_BACKWARD)) | MOVEFLAG_MOVE_FORWARD;
99 else
100 moveFlags = (moveFlags & ~(MOVEFLAG_MOVE_FORWARD)) | MOVEFLAG_MOVE_BACKWARD;
101#else
102 moveFlags |= MOVEFLAG_MOVE_FORWARD;
103#endif
104
105 if (moveFlags & MOVEFLAG_ROOTED)
106 moveFlags &= ~MOVEFLAG_MOVING_MASK;
107
108 if (!args.HasVelocity)
109 {
110 // If spline is initialized with SetWalk method it only means we need to select
111 // walk move speed for it but not add walk flag to unit
112 uint32_t moveFlagsForSpeed = moveFlags;
113 if (args.walk)
114 moveFlagsForSpeed |= MOVEFLAG_WALK;
115 else
116 moveFlagsForSpeed &= ~MOVEFLAG_WALK;
117
118 args.velocity = unit->getSpeedRate(SelectSpeedType(moveFlagsForSpeed), true);
119
120 // TODO: move this in Unit::updateSpeed, very hacky to do in splines -Appled
121 if (unit->isCreature() && !unit->isSummon() && unit->m_playerControler == nullptr && !unit->isVehicle() && !unit->isInEvadeMode())
122 {
123 // Bosses that can be slowed will also slow down when hp drops low
124 if (!unit->hasSpellImmunity(SPELL_IMMUNITY_SLOW))
125 {
126 // Units that are below 30% hp will start gradually slow their speed
127 const auto speedReductionPct = std::min(0.0f, (1.66f * (unit->getHealthPct() - 30.0f)));
128 if (speedReductionPct < 0.0f)
129 args.velocity *= (1.0f + (speedReductionPct / 100.0f));
130 }
131 }
132 }
133
134#if VERSION_STRING > TBC
135 // limit the speed in the same way the client does
136 args.velocity = std::min(args.velocity, args.flags.catmullrom || args.flags.flying ? 50.0f : std::max(28.0f, unit->getSpeedRate(TYPE_RUN, true) * 4.0f));
137#endif
138
139 if (!args.Validate(unit))
140 return 0;
141
142 unit->obj_movement_info.flags = moveFlags;
143 move_spline.Initialize(args);
144
145 WorldPacket data(SMSG_MONSTER_MOVE, 64);
146 data << WoWGuid(unit->getGuid());
147 if (transport)
148 {
149 data.SetOpcode(SMSG_MONSTER_MOVE_TRANSPORT);
150 data << WoWGuid(unit->getTransGuid());
151#if VERSION_STRING >= WotLK
152 data << int8_t(unit->GetTransSeat());
153#endif
154 }
155
156 PacketBuilder::WriteMonsterMove(move_spline, data);
157 unit->sendMessageToSet(&data, true);
158
159 return move_spline.Duration();
160}

162void MoveSplineInit::Stop()
163{
164 MoveSpline& move_spline = *unit->movespline;
165
166 // No need to stop if we are not moving
167 if (move_spline.Finalized())
168 return;
169
170#if VERSION_STRING <= WotLK
171 bool transport = unit->hasUnitMovementFlag(MOVEFLAG_TRANSPORT) && unit->getTransGuid();
172#else
173 bool transport = unit->getTransGuid() != 0;
174#endif
175
176 Location loc;
177 if (move_spline.onTransport == transport)
178 {
179 loc = move_spline.ComputePosition();
180 }
181 else
182 {
183 LocationVector const pos = transport ? unit->obj_movement_info.transport_position : unit->GetPosition();
184 loc.x = pos.getPositionX();
185 loc.y = pos.getPositionY();
186 loc.z = pos.getPositionZ();
187 loc.orientation = unit->GetOrientation();
188 }
189
190 args.flags = MoveSplineFlag::Done;
191
192#if VERSION_STRING <= WotLK
193 unit->obj_movement_info.removeMovementFlag(MOVEFLAG_SPLINE_FORWARD_ENABLED);
194#else
195 unit->obj_movement_info.removeMovementFlag(MOVEFLAG_MOVE_FORWARD);
196#endif
197 move_spline.onTransport = transport;
198 move_spline.Initialize(args);
199
200 WorldPacket data(SMSG_MONSTER_MOVE, 64);
201 data << WoWGuid(unit->getGuid());
202 if (transport)
203 {
204 data.SetOpcode(SMSG_MONSTER_MOVE_TRANSPORT);
205 data << WoWGuid(unit->getTransGuid());
206#if VERSION_STRING >= WotLK
207 data << int8_t(unit->GetTransSeat());
208#endif
209 }
210
211 PacketBuilder::WriteStopMovement(loc, args.splineId, data);
212 unit->sendMessageToSet(&data, true);
213}

302void MoveSplineInit::MoveTo(Vector3 const& dest, bool generatePath, bool forceDestination)
303{
304 if (generatePath)
305 {
306 PathGenerator path(unit);
307 bool result = path.calculatePath(dest.x, dest.y, dest.z, forceDestination);
308 if (result && !(path.getPathType() & PATHFIND_NOPATH))
309 {
310 MovebyPath(path.getPath());
311 return;
312 }
313 }
314
315 args.path_Idx_offset = 0;
316 args.path.resize(2);
317 TransportPathTransform transform(unit, args.TransformForTransport);
318 args.path[1] = transform(dest);
319}
