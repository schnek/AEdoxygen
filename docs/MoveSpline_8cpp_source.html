15Location MoveSpline::ComputePosition() const
16{
17 ASSERT(Initialized());
18
19 float u = 1.f;
20 int32_t seg_time = spline.length(point_Idx, point_Idx + 1);
21 if (seg_time > 0)
22 u = (time_passed - spline.length(point_Idx)) / (float)seg_time;
23 Location c;
24 c.orientation = initialOrientation;
25 spline.evaluate_percent(point_Idx, u, c);
26
27 if (splineflags.animation)
28 ;// MoveSplineFlag::Animation disables falling or parabolic movement
29 else if (splineflags.parabolic)
30 computeParabolicElevation(c.z);
31 else if (splineflags.falling)
32 computeFallElevation(c.z);
33
34 if (splineflags.done && splineflags.isFacing())
35 {
36 if (splineflags.final_angle)
37 c.orientation = facing.angle;
38 else if (splineflags.final_point)
39 c.orientation = std::atan2(facing.f.y - c.y, facing.f.x - c.x);
40 //nothing to do for MoveSplineFlag::Final_Target flag
41 }
42 else
43 {
44 if (!splineflags.hasFlag(MoveSplineFlag::OrientationFixed | MoveSplineFlag::Falling))
45 {
46 Vector3 hermite;
47 spline.evaluate_derivative(point_Idx, u, hermite);
48 c.orientation = std::atan2(hermite.y, hermite.x);
49 }
50
51 if (splineflags.orientationInversed)
52 c.orientation = -c.orientation;
53 }
54 return c;
55}

135void MoveSpline::computeParabolicElevation(float& el) const
136{
137 if (time_passed > effect_start_time)
138 {
139 float t_passedf = MSToSec(time_passed - effect_start_time);
140 float t_durationf = MSToSec(Duration() - effect_start_time); // client use not modified duration here
141
142 // -a*x*x + bx + c:
143 //(dur * v3->z_acceleration * dt)/2 - (v3->z_acceleration * dt * dt)/2 + Z;
144 el += (t_durationf - t_passedf) * 0.5f * vertical_acceleration * t_passedf;
145 }
146}

148void MoveSpline::computeFallElevation(float& el) const
149{
150 float z_now = spline.getPoint(spline.first()).z - MovementMgr::computeFallElevation(MSToSec(time_passed), false);
151 float final_z = FinalDestination().z;
152 el = std::max(z_now, final_z);
153}

187void MoveSpline::init_spline(MoveSplineInitArgs const& args)
188{
189 static SplineBase::EvaluationMode const modes[2] = { SplineBase::ModeLinear, SplineBase::ModeCatmullrom };
190 if (args.flags.cyclic)
191 {
192 uint32_t cyclic_point = 0;
193 if (splineflags.enter_cycle)
194 cyclic_point = 1; // shouldn't be modified, came from client
195 spline.init_cyclic_spline(&args.path[0], static_cast<int32_t>(args.path.size()), modes[args.flags.isSmooth()], cyclic_point, args.initialOrientation);
196 }
197 else
198 {
199 spline.init_spline(&args.path[0], static_cast<int32_t>(args.path.size()), modes[args.flags.isSmooth()], args.initialOrientation);
200 }
201
202 // init spline timestamps
203 if (splineflags.falling)
204 {
205 FallInitializer init(spline.getPoint(spline.first()).z);
206 spline.initLengths(init);
207 }
208 else
209 {
210 CommonInitializer init(args.velocity);
211 spline.initLengths(init);
212 }
213
214 /// @todo what to do in such cases? problem is in input data (all points are at same coords)
215 if (spline.length() < minimal_duration)
216 {
217 sLogger.failure("MoveSpline::init_spline: zero length spline, wrong input data?");
218 spline.set_length(spline.last(), spline.isCyclic() ? 1000 : 1);
219 }
220 point_Idx = spline.first();
221}

223void MoveSpline::Initialize(MoveSplineInitArgs const& args)
224{
225 splineflags = args.flags;
226 facing = args.facing;
227 m_Id = args.splineId;
228 point_Idx_offset = args.path_Idx_offset;
229 initialOrientation = args.initialOrientation;
230
231 time_passed = 0;
232 vertical_acceleration = 0.f;
233 effect_start_time = 0;
234
235 velocity = args.velocity;
236
237 // Check if its a stop spline
238 if (args.flags.done)
239 {
240 spline.clear();
241 return;
242 }
243
244 init_spline(args);
245#if VERSION_STRING > TBC
246 // init parabolic / animation
247 // spline initialized, duration known and i able to compute parabolic acceleration
248 if (args.flags & (MoveSplineFlag::Parabolic | MoveSplineFlag::Animation))
249 {
250 effect_start_time = static_cast<int32_t>(Duration() * args.time_perc);
251 if (args.flags.parabolic && effect_start_time < Duration())
252 {
253 float f_duration = MSToSec(Duration() - effect_start_time);
254 vertical_acceleration = args.parabolic_amplitude * 8.f / (f_duration * f_duration);
255 }
256 }
257#endif
258}

288// MONSTER_MOVE packet format limitation for not CatmullRom movement:
289// each vertex offset packed into 11 bytes
290bool MoveSplineInitArgs::_checkPathBounds() const
291{
292 if (!(flags & MoveSplineFlag::Mask_CatmullRom) && path.size() > 2)
293 {
294 enum{
295 MAX_OFFSET = (1 << 11) / 2
296 };
297 Vector3 middle = (path.front()+path.back()) / 2;
298 Vector3 offset;
299 for (uint32_t i = 1; i < path.size()-1; ++i)
300 {
301 offset = path[i] - middle;
302 if (std::fabs(offset.x) >= static_cast<float>(MAX_OFFSET) || std::fabs(offset.y) >= static_cast<float>(MAX_OFFSET) || std::fabs(offset.z) >= static_cast<float>(MAX_OFFSET))
303 {
304 sLogger.failure("MoveSplineInitArgs::_checkPathBounds check failed");
305 return false;
306 }
307 }
308 }
309 return true;
310}

324MoveSpline::UpdateResult MoveSpline::_updateState(int32_t& ms_time_diff)
325{
326 if (Finalized())
327 {
328 ms_time_diff = 0;
329 return Result_Arrived;
330 }
331
332 UpdateResult result = Result_None;
333
334 int32_t minimal_diff = std::min(ms_time_diff, segment_time_elapsed());
335 ASSERT(minimal_diff >= 0);
336 time_passed += minimal_diff;
337 ms_time_diff -= minimal_diff;
338
339 if (time_passed >= next_timestamp())
340 {
341 ++point_Idx;
342 if (point_Idx < spline.last())
343 {
344 result = Result_NextSegment;
345 }
346 else
347 {
348 if (spline.isCyclic())
349 {
350 point_Idx = spline.first();
351 time_passed = time_passed % Duration();
352 result = Result_NextCycle;
353
354 // Remove first point from the path after one full cycle.
355 // That point was the position of the unit prior to entering the cycle and it shouldn't be repeated with continuous cycles.
356 if (splineflags.enter_cycle)
357 {
358 splineflags.enter_cycle = false;
359
360 MoveSplineInitArgs args{spline.getPointCount()};
361 args.path.assign(spline.getPoints().begin() + spline.first() + 1, spline.getPoints().begin() + spline.last());
362 args.facing = facing;
363 args.flags = splineflags;
364 args.path_Idx_offset = point_Idx_offset;
365 // MoveSplineFlag::Parabolic | MoveSplineFlag::Animation not supported currently
366 //args.parabolic_amplitude = ?;
367 //args.time_perc = ?;
368 args.splineId = m_Id;
369 args.initialOrientation = initialOrientation;
370 args.velocity = 1.0f; // Calculated below
371 args.HasVelocity = true;
372 args.TransformForTransport = onTransport;
373 if (args.Validate(nullptr))
374 {
375 // New cycle should preserve previous cycle's duration for some weird reason, even though
376 // the path is really different now. Blizzard is weird. Or this was just a simple oversight.
377 // Since our splines precalculate length with velocity in mind, if we want to find the desired
378 // velocity, we have to make a fake spline, calculate its duration and then compare it to the
379 // desired duration, thus finding out how much the velocity has to be increased for them to match.
380 MoveSpline tempSpline;
381 tempSpline.Initialize(args);
382 args.velocity = (float)tempSpline.Duration() / Duration();
383
384 if (args.Validate(nullptr))
385 init_spline(args);
386 }
387 }
388 }
389 else
390 {
391 _Finalize();
392 ms_time_diff = 0;
393 result = Result_Arrived;
394 }
395 }
396 }
397
398 return result;
399}
