195inline float normalizeOrientation(float orientation)
196{
197 if (orientation < 0)
198 {
199 float mod = orientation *-1;
200 mod = fmod(mod, 2.0f * static_cast<float>(M_PI));
201 mod = -mod + 2.0f * static_cast<float>(M_PI);
202 return mod;
203 }
204
205 return fmod(orientation, 2.0f * static_cast<float>(M_PI));
206}
