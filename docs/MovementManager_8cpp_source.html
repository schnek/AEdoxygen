147std::vector<MovementGeneratorInformation> MovementManager::getMovementGeneratorsInformation() const
148{
149 std::vector<MovementGeneratorInformation> list;
150
151 if (_defaultGenerator)
152 list.emplace_back(_defaultGenerator->getMovementGeneratorType(), 0, std::string());
153
154 for (auto itr = _generators.begin(); itr != _generators.end(); ++itr)
155 {
156 MovementGenerator* movement = *itr;
157 MovementGeneratorType const type = movement->getMovementGeneratorType();
158 switch (type)
159 {
160 case CHASE_MOTION_TYPE:
161 case FOLLOW_MOTION_TYPE:
162 if (AbstractFollower* followInformation = dynamic_cast<AbstractFollower*>(movement))
163 {
164 if (Unit* target = followInformation->getTarget())
165 list.emplace_back(type, target->getGuid(), target->ToCreature()->GetCreatureProperties()->Name.c_str());
166 else
167 list.emplace_back(type, 0, std::string());
168 }
169 else
170 list.emplace_back(type, 0, std::string());
171 break;
172 default:
173 list.emplace_back(type, 0, std::string());
174 break;
175 }
176 }
177
178 return list;
179}

243MovementGenerator* MovementManager::getMovementGenerator(std::function<bool(MovementGenerator const*)> const& filter, MovementSlot slot) const
244{
245 if (empty() || isInvalidMovementSlot(slot))
246 return nullptr;
247
248 MovementGenerator* movement = nullptr;
249 switch (slot)
250 {
251 case MOTION_SLOT_DEFAULT:
252 if (_defaultGenerator && filter(_defaultGenerator.get()))
253 movement = _defaultGenerator.get();
254 break;
255 case MOTION_SLOT_ACTIVE:
256 if (!_generators.empty())
257 {
258 auto itr = std::find_if(_generators.begin(), _generators.end(), std::ref(filter));
259 if (itr != _generators.end())
260 movement = *itr;
261 }
262 break;
263 default:
264 break;
265 }
266
267 return movement;
268}

270bool MovementManager::hasMovementGenerator(std::function<bool(MovementGenerator const*)> const& filter, MovementSlot slot) const
271{
272 if (empty() || isInvalidMovementSlot(slot))
273 return false;
274
275 bool value = false;

279 if (_defaultGenerator && filter(_defaultGenerator.get()))
280 value = true;
281 break;
282 case MOTION_SLOT_ACTIVE:
283 if (!_generators.empty())
284 {
285 auto itr = std::find_if(_generators.begin(), _generators.end(), std::ref(filter));
286 value = itr != _generators.end();
287 }
288 break;
289 default:
290 break;
291 }
292
293 return value;

296void MovementManager::update(uint32_t diff)
297{
298 if (!_owner)
299 return;
300
301 if (hasFlag(MOTIONMASTER_FLAG_INITIALIZATION_PENDING | MOTIONMASTER_FLAG_INITIALIZING))
302 return;
303
304 if (empty())
305 {
306 sLogger.failure("MovementManager: update called without Initializing! ({})", _owner->getGuid());
307 return;
308 }
309
310 addFlag(MOTIONMASTER_FLAG_UPDATE);
311
312 MovementGenerator* top = getCurrentMovementGenerator();
313 if (hasFlag(MOTIONMASTER_FLAG_STATIC_INITIALIZATION_PENDING) && isStatic(top))
314 {
315 removeFlag(MOTIONMASTER_FLAG_STATIC_INITIALIZATION_PENDING);
316 top->initialize(_owner);
317 }
318 if (top->hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING))
319 top->initialize(_owner);
320 if (top->hasFlag(MOVEMENTGENERATOR_FLAG_DEACTIVATED))
321 top->reset(_owner);
322
323 if (!top->update(_owner, diff))
324 {
325 // Since all the actions that modify any slot are delayed, this movement is guaranteed to be top
326 pop(true, true); // Natural, and only, call to MovementInform
327 }
328
329 removeFlag(MOTIONMASTER_FLAG_UPDATE);
330
331 resolveDelayedActions();
332}

334void MovementManager::add(MovementGenerator* movement, MovementSlot slot/* = MOTION_SLOT_ACTIVE*/)
335{
336 if (!movement)
337 return;
338
339 if (isInvalidMovementSlot(slot))
340 {
341 delete movement;
342 return;
343 }
344
345 if (hasFlag(MOTIONMASTER_FLAG_DELAYED))
346 {
347 DelayedActionDefine action = [this, movement, slot]()
348 {
349 add(movement, slot);
350 };
351 _delayedActions.emplace_back(std::move(action), MOTIONMASTER_DELAYED_ADD);
352 }
353 else
354 directAdd(movement, slot);
355}

357void MovementManager::remove(MovementGenerator* movement, MovementSlot slot/* = MOTION_SLOT_ACTIVE*/)
358{
359 if (!movement || isInvalidMovementSlot(slot))
360 return;
361
362 if (hasFlag(MOTIONMASTER_FLAG_DELAYED))
363 {
364 DelayedActionDefine action = [this, movement, slot]()
365 {
366 remove(movement, slot);
367 };
368 _delayedActions.emplace_back(std::move(action), MOTIONMASTER_DELAYED_REMOVE);
369 return;
370 }
371
372 if (empty())
373 return;
374
375 switch (slot)
376 {
377 case MOTION_SLOT_DEFAULT:
378 if (_defaultGenerator && _defaultGenerator.get() == movement)
379 directClearDefault();
380 break;
381 case MOTION_SLOT_ACTIVE:
382 if (!_generators.empty())
383 {
384 auto bounds = _generators.equal_range(movement);
385 auto itr = std::find(bounds.first, bounds.second, movement);
386 if (itr != _generators.end())
387 remove(itr, getCurrentMovementGenerator() == *itr, false);
388 }
389 break;
390 default:
391 break;
392 }
393}

395void MovementManager::remove(MovementGeneratorType type, MovementSlot slot/* = MOTION_SLOT_ACTIVE*/)
396{
397 if (isInvalidMovementGeneratorType(type) || isInvalidMovementSlot(slot))
398 return;
399
400 if (hasFlag(MOTIONMASTER_FLAG_DELAYED))
401 {
402 DelayedActionDefine action = [this, type, slot]()
403 {
404 remove(type, slot);
405 };
406 _delayedActions.emplace_back(std::move(action), MOTIONMASTER_DELAYED_REMOVE_TYPE);
407 return;
408 }
409
410 if (empty())
411 return;
412
413 switch (slot)
414 {
415 case MOTION_SLOT_DEFAULT:
416 if (_defaultGenerator && _defaultGenerator->getMovementGeneratorType() == type)
417 directClearDefault();
418 break;
419 case MOTION_SLOT_ACTIVE:
420 if (!_generators.empty())
421 {
422 auto itr = std::find_if(_generators.begin(), _generators.end(), [type](MovementGenerator const* a) -> bool
423 {
424 return a->getMovementGeneratorType() == type;
425 });
426
427 if (itr != _generators.end())
428 remove(itr, getCurrentMovementGenerator() == *itr, false);
429 }
430 break;
431 default:
432 break;
433 }
434}

648void MovementManager::moveCloserAndStop(uint32_t id, Unit* target, float distance)
649{
650 float distanceToTravel = _owner->getDistanceSq(target) - distance;
651 if (distanceToTravel > 0.0f)
652 {
653 float angle = _owner->getAbsoluteAngle(target);
654 float destx = _owner->GetPositionX() + distanceToTravel * std::cos(angle);
655 float desty = _owner->GetPositionY() + distanceToTravel * std::sin(angle);
656 movePoint(id, destx, desty, target->GetPositionZ());
657 }
658 else
659 {
660 // We are already close enough. We just need to turn toward the target without changing position.
661 MovementMgr::MoveSplineInit init(_owner);
662 init.MoveTo(_owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ());
663 init.SetFacing(target);
664 add(new GenericMovementGenerator(std::move(init), EFFECT_MOTION_TYPE, id));
665 }

723void MovementManager::moveKnockbackFrom(float srcX, float srcY, float speedXY, float speedZ)
724{
725 // This function may make players fall below map
726 if (_owner->isPlayer())
727 return;
728
729 if (speedXY < 0.01f)
730 return;
731
732 LocationVector dest = _owner->GetPosition();
733 float moveTimeHalf = speedZ / MovementMgr::gravity;
734 float dist = 2 * moveTimeHalf * speedXY;
735 float max_height = -MovementMgr::computeFallElevation(moveTimeHalf, false, -speedZ);
736
737 // Use a mmap raycast to get a valid destination.
738 _owner->movePositionToFirstCollision(dest, dist, _owner->getRelativeAngle(srcX, srcY) + float(M_PI));
739
740 MovementMgr::MoveSplineInit init(_owner);
741 init.MoveTo(dest.getPositionX(), dest.getPositionY(), dest.getPositionZ(), false);
742 init.SetParabolic(max_height, 0);
743#if VERSION_STRING >= WotLK
744 init.SetOrientationFixed(true);
745#endif
746 init.SetVelocity(speedXY);
747
748 GenericMovementGenerator* movement = new GenericMovementGenerator(std::move(init), EFFECT_MOTION_TYPE, 0);
749 movement->Priority = MOTION_PRIORITY_HIGHEST;
750 add(movement);
751}

775void MovementManager::moveJump(float x, float y, float z, float o, float speedXY, float speedZ, uint32_t id, bool hasOrientation /* = false*/)
776{
777 if (speedXY < 0.01f)
778 return;
779
780 float moveTimeHalf = speedZ / MovementMgr::gravity;
781 float max_height = -MovementMgr::computeFallElevation(moveTimeHalf, false, -speedZ);
782
783 MovementMgr::MoveSplineInit init(_owner);

796void MovementManager::moveCirclePath(float x, float y, float z, float radius, bool clockwise, uint8_t stepCount)
797{
798 float step = 2 * float(M_PI) / stepCount * (clockwise ? -1.0f : 1.0f);
799 LocationVector const& pos = { x, y, z, 0.0f };
800 float angle = pos.getAbsoluteAngle(_owner->GetPositionX(), _owner->GetPositionY());
801
802 MovementMgr::MoveSplineInit init(_owner);
803
804 // add the owner's current position as starting point as it gets removed after entering the cycle
805 init.Path().push_back(G3D::Vector3(_owner->GetPositionX(), _owner->GetPositionY(), _owner->GetPositionZ()));
806
807 for (uint8_t i = 0; i < stepCount; angle += step, ++i)
808 {
809 G3D::Vector3 point;
810 point.x = x + radius * cosf(angle);
811 point.y = y + radius * sinf(angle);
812
813 if (_owner->IsFlying())
814 {
815 point.z = z;
816 }
817 else
818 {
819 point.z = _owner->getWorldMap()->getHeight(LocationVector(point.x, point.y, z));
820#if VERSION_STRING >= WotLK
821 point.z += _owner->getHoverHeight();
822#endif
823 }
824
825 init.Path().push_back(point);
826 }
827
828 if (_owner->IsFlying())
829 {
830 init.SetFly();
831 init.SetCyclic();
832#if VERSION_STRING >= WotLK
833 init.SetAnimation(ANIMATION_FLAG_HOVER);
834#endif
835 }
836 else
837 {
838 init.SetWalk(true);
839 init.SetCyclic();
840 }
841
842 add(new GenericMovementGenerator(std::move(init), EFFECT_MOTION_TYPE, 0));
843}

898void MovementManager::moveFall(uint32_t id/* = 0*/)
899{
900 // Use larger distance for vmap height search than in most other cases
901 float tz = _owner->getWorldMap()->getHeight(_owner->GetPosition());
902 if (tz <= INVALID_HEIGHT)
903 return;
904
905 // Abort too if the ground is very near
906 if (std::fabs(_owner->GetPositionZ() - tz) < 0.1f)
907 return;
908
909 // rooted units don't move (also setting falling+root flag causes client freezes)
910 if (_owner->hasUnitStateFlag(UNIT_STATE_ROOTED | UNIT_STATE_STUNNED))
911 return;
912
913 _owner->addUnitMovementFlag(MOVEFLAG_FALLING);
914 _owner->obj_movement_info.setFallTime(0);
915
916 // Don't run spline movement for players
917 if (_owner->isPlayer())
918 return;
919
920#if VERSION_STRING >= WotLK
921 tz += _owner->getHoverHeight();
922#endif
923
924 MovementMgr::MoveSplineInit init(_owner);
925 init.MoveTo(_owner->GetPositionX(), _owner->GetPositionY(), tz, false);
926 init.SetFall();
927
928 GenericMovementGenerator* movement = new GenericMovementGenerator(std::move(init), EFFECT_MOTION_TYPE, id);
929 movement->Priority = MOTION_PRIORITY_HIGHEST;
930 add(movement);
931}
