25// Custom Functions to make par a keyvariable in an unordered map
26struct ChainKeyTypeHash
27{
28 std::size_t operator()(const ChainKeyType& key) const
29 {
30 std::size_t hash = std::hash<uint32_t>{}(key.first);
31 hash ^= std::hash<uint16_t>{}(key.second) + 0x9e3779b9 + (hash << 6) + (hash >> 2);
32 return hash;
33 }
34};
35
36// Custom Functions to make par a keyvariable in an unordered map
37struct ChainKeyTypeEqual
38{
39 bool operator()(const ChainKeyType& lhs, const ChainKeyType& rhs) const
40 {
41 return lhs.first == rhs.first && lhs.second == rhs.second;
42 }
43};
