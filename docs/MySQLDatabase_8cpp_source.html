51bool MySQLDatabase::Initialize(const char* Hostname, unsigned int port, const char* Username, const char* Password, const char* DatabaseName, uint32_t ConnectionCount, uint32_t /*BufferSize*/, bool useLegacyAuth)
52{
53 uint32_t i;
54 MYSQL* temp = NULL;
55 MYSQL* temp2 = NULL;
56 bool my_true = true;
57
58 mHostname = std::string(Hostname);
59 mConnectionCount = ConnectionCount;
60 mUsername = std::string(Username);
61 mPassword = std::string(Password);
62 mDatabaseName = std::string(DatabaseName);
63
64 sLogger.info("MySQLDatabase : Connecting to `{}`, database `{}`...", Hostname, DatabaseName);
65
66 Connections.reserve(ConnectionCount);
67 for(i = 0; i < ConnectionCount; ++i)
68 {
69 temp = mysql_init(NULL);
70 if(temp == NULL)
71 continue;
72
73 if(mysql_options(temp, MYSQL_SET_CHARSET_NAME, "utf8"))
74 sLogger.failure("Could not set utf8 character set.");
75
76 if(mysql_options(temp, MYSQL_OPT_RECONNECT, &my_true))
77 sLogger.failure("MYSQL_OPT_RECONNECT could not be set, connection drops may occur but will be counteracted.");
78
79 // Check if we want to use MySQL 8 legacy authentication otherwise use new auth
80 if (!useLegacyAuth)
81 {
82 // Set authentication to caching_sha2_password (new method)
83 if (mysql_options(temp, MYSQL_DEFAULT_AUTH, "caching_sha2_password"))
84 {
85 sLogger.failure("Could not set default authentication plugin to caching_sha2_password.");
86 mysql_close(temp);
87 return false;
88 }
89 sLogger.info("Using new MySQL 8 authentication method (caching_sha2_password).");
90 }
91 else
92 {
93 // No explicit authentication setting; use legacy authentication
94 sLogger.info("Using legacy MySQL authentication method.");
95 }
96
97 temp2 = mysql_real_connect(temp, Hostname, Username, Password, DatabaseName, port, NULL, 0);
98 if(temp2 == NULL)
99 {
100 sLogger.failure("Connection failed due to: `{}`", mysql_error(temp));
101 mysql_close(temp);
102 return false;
103 }
104
105 Connections.emplace_back(std::make_unique<MySQLDatabaseConnection>(temp2));
106 }
107
108 Database::_Initialize();
109 return true;
110}

160bool MySQLDatabase::_SendQuery(DatabaseConnection* con, const char* Sql, bool Self)
161{
162 //dunno what it does ...leaving untouched
163 int result = mysql_query(static_cast<MySQLDatabaseConnection*>(con)->MySql, Sql);
164 if(result > 0)
165 {
166 if(Self == false && _HandleError(static_cast<MySQLDatabaseConnection*>(con), mysql_errno(static_cast<MySQLDatabaseConnection*>(con)->MySql)))
167 {
168 // Re-send the query, the connection was successful.
169 // The true on the end will prevent an endless loop here, as it will
170 // stop after sending the query twice.
171 result = _SendQuery(con, Sql, true);
172 }
173 else
174 sLogger.failure("Sql query failed due to [{}], Query: [{}]", mysql_error(static_cast<MySQLDatabaseConnection*>(con)->MySql), Sql);
175 }
176
177 return (result == 0 ? true : false);
178}

180bool MySQLDatabase::_HandleError(MySQLDatabaseConnection* con, uint32_t ErrorNumber)
181{
182 // Handle errors that should cause a reconnect to the Database.
183 switch(ErrorNumber)
184 {
185 case 2006: // Mysql server has gone away
186 case 2008: // Client ran out of memory
187 case 2013: // Lost connection to sql server during query
188 case 2055: // Lost connection to sql server - system error
189 {
190 // Let's instruct a reconnect to the db when we encounter these errors.
191 return _Reconnect(con);
192 }
193 break;
194 }
195
196 return false;
197}

221std::unique_ptr<QueryResult> MySQLDatabase::_StoreQueryResult(DatabaseConnection* con)
222{
223 MySQLDatabaseConnection* db = static_cast<MySQLDatabaseConnection*>(con);
224 MYSQL_RES* pRes = mysql_store_result(db->MySql);
225 uint32_t uRows = (uint32_t)mysql_affected_rows(db->MySql);
226 uint32_t uFields = (uint32_t)mysql_field_count(db->MySql);
227
228 if(uRows == 0 || uFields == 0 || pRes == 0)
229 {
230 if(pRes != NULL)
231 mysql_free_result(pRes);
232
233 return NULL;
234 }
235
236 auto res = std::make_unique<MySQLQueryResult>(pRes, uFields, uRows);
237 res->NextRow();
238
239 return res;
240}
