45void NazanAI::AIUpdate(unsigned long time_passed)
46{
47 if (getScriptPhase() == AIR_PHASE)
48 {
49 Creature* Vazruden = getInstanceScript()->GetCreatureByGuid(VazrudenGUID);
50 if (Fly_Timer < time_passed || !(Vazruden && Vazruden->isAlive() && Vazruden->getHealthPct() > 20))
51 {
52 getCreature()->setMoveDisableGravity(false);
53 getCreature()->setMoveWalk(true);
54 getCreature()->getMovementManager()->clear();
55
56 if (Unit* victim = getBestPlayerTarget(TargetFilter_Closest))
57 {
58 getCreature()->getAIInterface()->onHostileAction(victim);
59 getCreature()->getMovementManager()->moveChase(victim);
60 }
61
62 sendDBChatMessage(NAZAN_EMOTE);
63 setScriptPhase(GROUND_PHASE);
64 return;
65 }
66
67 Fly_Timer -= time_passed;
68 }

161void VazrudenTheHeraldAI::AIUpdate(unsigned long /*time_passed*/)
162{
163 switch (getScriptPhase())
164 {
165 case CombatStart:
166 {
167 getCreature()->getMovementManager()->clear();
168 getCreature()->getMovementManager()->movePoint(0, VazrudenMiddle[0], VazrudenMiddle[1], VazrudenMiddle[2]);
169 } break;
170 case SUMMON_PHASE:
171 {
172 Creature* Nazan = getInstanceScript()->GetCreatureByGuid(nazanGUID);
173 Creature* Vazruden = getInstanceScript()->GetCreatureByGuid(vazrudenGUID);
174 if ((Nazan && Nazan->isAlive()) || (Vazruden && Vazruden->isAlive()))
175 {
176 if ((Nazan && Nazan->getAIInterface()->getCurrentTarget()) || (Vazruden && Vazruden->getAIInterface()->getCurrentTarget()))
177 return;
178
179 despawnAdds();
180 getCreature()->getAIInterface()->enterEvadeMode();
181 return;
182 }
183
184 if (!(Nazan && Nazan->isAlive()) && !(Vazruden && Vazruden->isAlive()))
185 {
186 getCreature()->Despawn(100, 0);
187 }
188 } break;
189 }
190}
