31static void logSocketInfo(const SOCKET& sock) {
32 uint32 val;
33 socklen_t sz = 4;
34
35 getsockopt(sock, SOL_SOCKET, SO_RCVBUF, (char*)&val, (socklen_t*)&sz);

52/** Invokes select on one socket. Returns SOCKET_ERROR on error, 0 if
53 there is no read pending, sock if there a read pending. */
54static int selectOneReadSocket(const SOCKET& sock) {
55 // 0 time timeout is specified to poll and return immediately
56 struct timeval timeout;
57 timeout.tv_sec = 0;
58 timeout.tv_usec = 0;
59
60 // Create a set that contains just this one socket
61 fd_set socketSet;
62 FD_ZERO(&socketSet);
63 FD_SET(sock, &socketSet);
64
65 int ret = select((int)sock + 1, &socketSet, NULL, NULL, &timeout);
66
67 return ret;
68}

71/** Returns true if the socket has a read pending */
72static bool readWaiting(const SOCKET& sock) {
73 int ret = selectOneReadSocket(sock);
74
75 switch (ret) {
76 case SOCKET_ERROR:
77 logPrintf("ERROR: selectOneReadSocket returned "
78 "SOCKET_ERROR in readWaiting(). %s", socketErrorCode().c_str());
79 // Return true so that we'll force an error on read and close
80 // the socket.
81 return true;
82
83 case 0:
84 return false;
85
86 default:
87 return true;
88 }
89}

336bool NetworkDevice::init() {
337 debugAssert(! initialized);
338
339 // Used for combining the MAC and ip information
340 typedef Table<std::string, EthernetAdapter> AdapterTable;
341
342 AdapterTable table;
343
344 // Head of a linked list of network interfaces on this machine
345 ifaddrs* ifap = NULL;
346
347 int r = getifaddrs(&ifap);
348
349 if (r != 0) {
350 logPrintf("ERROR: getifaddrs returned %d\n", r);
351 return false;
352 }
353
354 ifaddrs* current = ifap;
355
356 if (current == NULL) {
357 logPrintf("WARNING: No network interfaces found\n");
358 EthernetAdapter a;
359 a.name = "fallback";
360 a.hostname = "localhost";
361 a.ip = (127 << 24) | 1;
362 a.broadcast = 0xFFFFFFFF;
363 a.subnet = 0x000000FF;
364 addAdapter(a);
365
366 } else {
367
368 while (current != NULL) {
369
370 bool up = (current->ifa_flags & IFF_UP);
371 bool loopback = (current->ifa_flags & IFF_LOOPBACK);
372
373 if (! up || loopback) {
374 // Skip this adapter; it is offline or is a loopback
375 current = current->ifa_next;
376 continue;
377 }
378
379 if (! table.containsKey(current->ifa_name)) {
380 EthernetAdapter a;
381 a.name = current->ifa_name;
382 table.set(a.name, a);
383 }
384
385 // This adapter must exist because it was created above
386 EthernetAdapter& adapter = table[current->ifa_name];
387
388 const sockaddr_in* interfaceAddress = castToIP4(current->ifa_addr);
389 const sockaddr_in* broadcastAddress = castToIP4(current->ifa_dstaddr);
390 const sockaddr_in* subnetMask = castToIP4(current->ifa_netmask);
391
392 uint32 ip = getIP(interfaceAddress);
393 uint32 ba = getIP(broadcastAddress);
394 uint32 sn = getIP(subnetMask);
395
396 if (ip != 0) {
397 adapter.ip = ip;
398 }
399
400 if (ba != 0) {
401 adapter.broadcast = ba;
402 }
403
404 if (sn != 0) {
405 adapter.subnet = sn;
406 }
407
408 uint8_t* MAC = NULL;
409 // Extract MAC address
410 if ((current->ifa_addr != NULL) && (current->ifa_addr->sa_family == AF_LINK)) {
411# ifdef __linux__
412 {
413 // Linux
414 struct ifreq ifr;
415
416 int fd = socket(AF_INET, SOCK_DGRAM, 0);
417
418 ifr.ifr_addr.sa_family = AF_INET;
419 strcpy(ifr.ifr_name, current->ifa_name);
420 ioctl(fd, SIOCGIFHWADDR, &ifr);
421 close(fd);
422
423 MAC = reinterpret_cast<uint8_t*>(ifr.ifr_hwaddr.sa_data);
424 }
425# else
426 {
427 // The MAC address and the interfaceAddress come in as
428 // different interfaces with the same name.
429
430 // Posix/FreeBSD/Mac OS
431 sockaddr_dl* sdl = (struct sockaddr_dl *)current->ifa_addr;
432 MAC = reinterpret_cast<uint8_t*>(LLADDR(sdl));
433 }
434# endif
435
436 // See if there was a MAC address
437 if (MAC != NULL) {
438 bool anyNonZero = false;
439 for (int i = 0; i < 6; ++i) {
440 anyNonZero = anyNonZero || (MAC[i] != 0);
441 }
442 if (anyNonZero) {
443 System::memcpy(adapter.mac, MAC, 6);
444 }
445 }
446 }
447
448 current = current->ifa_next;
449 }
450
451 freeifaddrs(ifap);
452 ifap = NULL;
453 }
454
455 // Extract all interesting adapters from the table
456 for (AdapterTable::Iterator it = table.begin(); it.isValid(); ++it) {
457 const EthernetAdapter& adapter = it->value;
458
459 // Only add adapters that have IP addresses
460 if (adapter.ip != 0) {
461 addAdapter(adapter);
462 } else {
463 logPrintf("NetworkDevice: Ignored adapter %s because ip = 0\n", adapter.name.c_str());
464 }
465 }
466
467 initialized = true;
468
469 return true;
470}

515void NetworkDevice::localHostAddresses(Array<NetAddress>& array) const {
516 array.resize(0);
517
518 char ac[256];
519
520 if (gethostname(ac, sizeof(ac)) == SOCKET_ERROR) {
521 Log::common()->printf("Error while getting local host name\n");
522 return;
523 }
524
525 struct hostent* phe = gethostbyname(ac);
526 if (phe == 0) {
527 Log::common()->printf("Error while getting local host address\n");
528 return;
529 }
530
531 for (int i = 0; (phe->h_addr_list[i] != 0); ++i) {
532 struct in_addr addr;
533 memcpy(&addr, phe->h_addr_list[i], sizeof(struct in_addr));
534 array.append(NetAddress(addr));
535 }

584/**
585 Increases the send and receive sizes of a socket to 2 MB from 8k
586 */
587static void increaseBufferSize(SOCKET sock) {
588
589 // Increase the buffer size; the default (8192) is too easy to
590 // overflow when the network latency is high.
591 {
592 uint32 val = 1024 * 1024 * 2;
593 if (setsockopt(sock, SOL_SOCKET, SO_RCVBUF,
594 (char*)&val, sizeof(val)) == SOCKET_ERROR) {
595 Log::common()->printf("WARNING: Increasing socket "
596 "receive buffer to %d failed.\n", val);
597 Log::common()->println(socketErrorCode());
598 }
599
600 if (setsockopt(sock, SOL_SOCKET, SO_SNDBUF,
601 (char*)&val, sizeof(val)) == SOCKET_ERROR) {
602 Log::common()->printf("WARNING: Increasing socket "
603 "send buffer to %d failed.\n", val);
604 Log::common()->println(socketErrorCode());
605 }
606 }
607}

616ReliableConduit::ReliableConduit
617 (const NetAddress& _addr) :
618 state(NO_MESSAGE),
619 receiveBuffer(NULL),
620 receiveBufferTotalSize(0),
621 receiveBufferUsedSize(0) {
622
623 NetworkDevice* nd = NetworkDevice::instance();
624
625 messageType = 0;
626
627 addr = _addr;
628 Log::common()->print("Creating a TCP socket ");
629 sock = socket(AF_INET, SOCK_STREAM, IPPROTO_IP);
630
631 if (sock == SOCKET_ERROR) {
632 Log::common()->println("FAIL");
633 Log::common()->println(socketErrorCode());
634 nd->closesocket(sock);
635 return;
636 }
637
638 Log::common()->println("Ok");
639
640 // Setup socket options (both constructors should set the same options)
641
642 // Disable Nagle's algorithm (we send lots of small packets)
643 const int T = true;
644 if (setsockopt(sock, IPPROTO_TCP, TCP_NODELAY,
645 (const char*)&T, sizeof(T)) == SOCKET_ERROR) {
646
647 Log::common()->println("WARNING: Disabling Nagel's "
648 "algorithm failed.");
649 Log::common()->println(socketErrorCode());
650 } else {
651 Log::common()->println("Disabled Nagel's algorithm.");
652 }
653
654 // Set the NO LINGER option so the socket doesn't hang around if
655 // there is unsent data in the queue when it closes.
656 struct linger ling;
657 ling.l_onoff = 0;
658 ling.l_linger = 0;
659 if (setsockopt(sock, SOL_SOCKET, SO_LINGER,
660 (const char*)&ling, sizeof(ling)) == SOCKET_ERROR) {
661
662 Log::common()->println("WARNING: Setting socket no linger failed.");
663 Log::common()->println(socketErrorCode());
664 } else {
665 Log::common()->println("Set socket option no_linger.");
666 }
667
668 // Set reuse address so that a new server can start up soon after
669 // an old one has closed.
670 if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
671 (const char*)&T, sizeof(T)) == SOCKET_ERROR) {
672
673 Log::common()->println("WARNING: Setting socket reuseaddr failed.");
674 Log::common()->println(socketErrorCode());
675 } else {
676 Log::common()->println("Set socket option reuseaddr.");
677 }
678
679 // Ideally, we'd like to specify IPTOS_LOWDELAY as well.
680
681 logSocketInfo(sock);
682
683 increaseBufferSize(sock);
684
685 Log::common()->printf("Created TCP socket %d\n", sock);
686
687 std::string x = addr.toString();
688 Log::common()->printf("Connecting to %s on TCP socket %d ", x.c_str(), sock);
689
690 int ret = connect(sock, (struct sockaddr *) &(addr.addr), sizeof(addr.addr));
691
692 if (ret == WSAEWOULDBLOCK) {
693 RealTime t = System::time() + 5.0;
694 // Non-blocking; we must wait until select returns non-zero
695 while ((selectOneWriteSocket(sock) == 0) && (System::time() < t)) {
696 System::sleep(0.02);
697 }
698
699 // TODO: check for failure on the select call
700
701 } else if (ret != 0) {
702 sock = (SOCKET)SOCKET_ERROR;
703 Log::common()->println("FAIL");
704 Log::common()->println(socketErrorCode());
705 return;
706 }
707
708 Log::common()->println("Ok");
709}

776bool ReliableConduit::messageWaiting() {
777 switch (state) {
778 case HOLDING:
779 // We've already read the message and are waiting
780 // for a receive call.
781 return true;
782
783 case RECEIVING:
784
785 if (! ok()) {
786 return false;
787 }
788 // We're currently receiving the message. Read a little more.
789 receiveIntoBuffer();
790
791 if (messageSize == receiveBufferUsedSize) {
792 // We've read the whole mesage. Switch to holding state
793 // and return true.
794 state = HOLDING;
795 return true;
796 } else {
797 // There are more bytes left to read. We'll read them on
798 // the next call. Because the *entire* message is not ready,
799 // return false.
800 return false;
801 }
802 break;
803
804 case NO_MESSAGE:
805 if (Conduit::messageWaiting()) {
806 // Message incoming. Read the header.
807
808 state = RECEIVING;
809 receiveHeader();
810
811 // Loop back around now that we're in the receive state; we
812 // may be able to read the whole message before returning
813 // to the caller.
814 return messageWaiting();
815 } else {
816 // No message incoming.
817 return false;
818 }
819 }
820
821 debugAssertM(false, "Should not reach this point");
822 return false;
823}

836void ReliableConduit::sendBuffer(const BinaryOutput& b) {
837 NetworkDevice* nd = NetworkDevice::instance();
838 int ret = ::send(sock, (const char*)b.getCArray(), (int)b.size(), 0);
839

875void ReliableConduit::receiveHeader() {
876 NetworkDevice* nd = NetworkDevice::instance();
877 debugAssert(state == RECEIVING);
878
879 // Read the type
880 uint32 tmp;
881 int ret = recv(sock, (char*)&tmp, sizeof(tmp), 0);
882
883 // The type is the first four bytes. It is little endian.
884 if (System::machineEndian() == G3D_LITTLE_ENDIAN) {
885 messageType = tmp;
886 } else {
887 // Swap the byte order
888 for (int i = 0; i < 4; ++i) {
889 ((char*)&messageType)[i] = ((char*)&tmp)[3 - i];
890 }
891 }
892
893 if ((ret == SOCKET_ERROR) || (ret != sizeof(messageType))) {
894 Log::common()->printf("Call to recv failed. ret = %d,"
895 " sizeof(messageType) = %d\n",
896 (int)ret, (int)sizeof(messageType));
897 Log::common()->println(socketErrorCode());
898 nd->closesocket(sock);
899 messageType = 0;
900 return;
901 }
902
903 // Read the size
904 ret = recv(sock, (char*)&messageSize, sizeof(messageSize), 0);
905
906 if ((ret == SOCKET_ERROR) || (ret != sizeof(messageSize))) {
907 Log::common()->printf("Call to recv failed. ret = %d,"
908 " sizeof(len) = %d\n", (int)ret,
909 (int)sizeof(messageSize));
910 Log::common()->println(socketErrorCode());
911 nd->closesocket(sock);
912 messageType = 0;
913 return;
914 }
915
916 messageSize = ntohl(messageSize);
917 debugAssert(messageSize < 6e7);
918
919 debugAssert(receiveBufferUsedSize == 0);
920
921 // Extend the size of the buffer.
922 if (messageSize > receiveBufferTotalSize) {
923 receiveBuffer = realloc(receiveBuffer, messageSize);
924 receiveBufferTotalSize = messageSize;
925 }
926
927 if (receiveBuffer == NULL) {
928 Log::common()->println("Could not allocate a memory buffer "
929 "during receivePacket.");
930 nd->closesocket(sock);
931 }
932
933 bReceived += 4;
934}

937void ReliableConduit::receiveIntoBuffer() {
938 NetworkDevice* nd = NetworkDevice::instance();
939
940 debugAssert(state == RECEIVING);
941 debugAssert(messageType != 0);
942 debugAssertM(receiveBufferUsedSize < messageSize, "Message already received.");
943 debugAssertM(messageSize >= receiveBufferUsedSize, "Message size overflow.");
944
945 // Read the data itself
946 int ret = 0;
947 uint32 left = messageSize - (uint32)receiveBufferUsedSize;
948 int count = 0;
949 while ((ret != SOCKET_ERROR) && (left > 0) && (count < 100)) {
950
951 ret = recv(sock, ((char*)receiveBuffer) + (uint32)receiveBufferUsedSize, left, 0);
952
953 if (ret > 0) {
954 left -= ret;
955 receiveBufferUsedSize += ret;
956 bReceived += ret;
957
958 if (left > 0) {
959 // There's still more. Give the machine a chance to read
960 // more data, but don't wait forever.
961
962 ++count;
963 System::sleep(0.001);
964 }
965 } else {
966 // Something went wrong; our blocking read returned nothing.
967 break;
968 }
969 }
970
971 if ((ret == 0) || (ret == SOCKET_ERROR)) {
972
973 if (ret == SOCKET_ERROR) {
974 Log::common()->printf("Call to recv failed. ret = %d,"
975 " sizeof(messageSize) = %d\n", ret, messageSize);
976 Log::common()->println(socketErrorCode());
977 } else {
978 Log::common()->printf("recv returned 0\n");
979 }
980 nd->closesocket(sock);
981 return;
982 }
983
984 ++mReceived;
985}

1120 messageSender = NetAddress(remote_addr);
1121
1122 ++mReceived;
1123 bReceived += ret;
1124
1125 messageBuffer.resize(ret, DONT_SHRINK_UNDERLYING_ARRAY);
1126
