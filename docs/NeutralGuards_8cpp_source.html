38 // Generates 3 random waypoints around the NPC
39 void setGuardWaypoints()
40 {
41 if (!getCreature()->getWorldMap())
42 return;
43
44 float x = 0.0f;
45 float y = 0.0f;
46 float z = 0.0f;
47 float o = 0.0f;
48
49 for (uint8_t i = 1; i <= 4; i++)
50 {
51 float ang = Util::getRandomFloat(100.0f) / 100.0f;
52 float ran = Util::getRandomFloat(100.0f) / 10.0f;
53 while (ran < 1)
54 ran = Util::getRandomFloat(100.0f) / 10.0f;
55
56 x = getCreature()->GetSpawnX() + ran * sin(ang);
57 y = getCreature()->GetSpawnY() + ran * cos(ang);
58 z = getCreature()->getMapHeight(LocationVector(x, y, getCreature()->GetSpawnZ() + 2));
59
60 addWaypoint(1, createWaypoint(i, 800, WAYPOINT_MOVE_TYPE_WALK, LocationVector(x, y, z, o)));
61 }
62 }
