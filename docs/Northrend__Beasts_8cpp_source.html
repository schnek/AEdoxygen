125{
126 getInstanceScript()->setLocalData(DATA_DESPAWN_SNOBOLDS, 0);

373void SnoboldAI::mountOnBoss()
374{
375 Unit* gormok = getInstanceScript()->getCreatureFromData(DATA_GORMOK_THE_IMPALER);
376 if (gormok && gormok->isAlive())
377 {
378 attackStop();
379 targetGUID = 0;
380 mountedOnPlayer = false;
381
382 cancelFunctionFromScheduler(func_Batter);
383 cancelFunctionFromScheduler(func_HeadCrack);
384
385 for (uint8_t i = 0; i < Beasts::MAX_SNOBOLDS; i++)
386 {
387 if (!gormok->getVehicleKit()->getPassenger(i))
388 {
389 getCreature()->callEnterVehicle(gormok, i);
390 DoAction(Beasts::ACTION_ENABLE_FIRE_BOMB);
391 break;
392 }
393 }
394 }
395 else
396 {
397 // Disable Spells we Only Use Mounted
398 cancelFunctionFromScheduler(func_CheckMount);
399 cancelFunctionFromScheduler(func_FireBomb);
400
401 removeUnitFlags(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE);
402 attackStop();
403
404 setRooted(false);
405 gormokDead = true;
406
407 if (Unit* target = selectUnitTarget(FilterArgs(TargetFilter_Player)))
408 {
409 attackStart(target);
410 moveChase(target);
411
412 targetGUID = target->getGuid();
413
414 func_Batter = addAIFunction(&SnoboldAI::Batter, DoOnceScheduler(5s, 50.0f));
415 func_HeadCrack = addAIFunction(&SnoboldAI::HeadCrack, DoOnceScheduler(1s));
416 }
417 }
418}
