24bool ChatCommandHandler::HandleNpcAddAgentCommand(const char* args, WorldSession* m_session)
25{
26 //new
27 auto creature_target = GetSelectedCreature(m_session, true);
28 if (creature_target == nullptr)
29 return true;
30
31 uint32_t ai_type;
32 uint32_t procEvent;
33 uint32_t procChance;
34 uint32_t maxcount;
35 uint32_t spellId;
36 uint32_t spellType;
37 uint32_t spelltargetType;
38 uint32_t spellCooldown;
39 float floatMisc1;
40 uint32_t Misc2;
41
42 if (sscanf(args, "%u %u %u %u %u %u %u %u %f %u", &ai_type, &procEvent, &procChance, &maxcount, &spellId, &spellType, &spelltargetType, &spellCooldown, &floatMisc1, &Misc2) != 10)
43 {
44 redSystemMessage(m_session, "Command must be in format: .npc add trainerspell <ai_type> <procEvent> <procChance> <maxcount> <spellId> <spellType> <spelltarget_overwrite> <spellCooldown> <floatMisc1> <Misc2>.");
45 return true;
46 }
47
48 auto spell_entry = sSpellMgr.getSpellInfo(spellId);
49 if (spell_entry == nullptr)
50 {
51 redSystemMessage(m_session, "Spell {} is not invalid!", spellId);
52 return true;
53 }
54
55 systemMessage(m_session, "Added agent_type {} for spell {} to creature {} ({}).", ai_type, spellId, creature_target->GetCreatureProperties()->Name, creature_target->getEntry());
56 sGMLog.writefromsession(m_session, "added agent_type %u for spell %u to creature %s (%u).", ai_type, spellId, creature_target->GetCreatureProperties()->Name.c_str(), creature_target->getEntry());
57 WorldDatabase.Execute("INSERT INTO ai_agents VALUES(%u, 4, %u, %u, %u, %u, %u, %u, %u, %u, %f, %u",
58 creature_target->getEntry(), ai_type, procEvent, procChance, maxcount, spellId, spellType, spelltargetType, spellCooldown, floatMisc1, Misc2);
59
60 auto ai_spell = std::make_unique<AI_Spell>();
61 ai_spell->agent = static_cast<uint16_t>(ai_type);
62 ai_spell->procChance = procChance;
63 ai_spell->procCount = maxcount;
64 ai_spell->spell = spell_entry;
65 ai_spell->spellType = static_cast<uint8_t>(spellType);
66 ai_spell->spelltargetType = static_cast<uint8_t>(spelltargetType);
67 ai_spell->floatMisc1 = floatMisc1;
68 ai_spell->Misc2 = Misc2;
69 ai_spell->cooldown = spellCooldown;
70 ai_spell->procCounter = 0;
71 ai_spell->cooldowntime = 0;
72 ai_spell->minrange = spell_entry->getMinRange();
73 ai_spell->maxrange = spell_entry->getMaxRange();
74
75 if (auto* creatureProperties = const_cast<CreatureProperties*>(creature_target->GetCreatureProperties()))
76 {
77 if (!spell_entry->isPassive())
78 creatureProperties->castable_spells.emplace_back(spell_entry->getId());
79 else
80 creatureProperties->start_auras.emplace(spell_entry->getId());
81 }
82
83 switch (ai_type)
84 {
85 case AGENT_MELEE:
86 creature_target->getAIInterface()->setMeleeDisabled(false);
87 break;
88 case AGENT_RANGED:
89 creature_target->getAIInterface()->setRangedDisabled(false);
90 break;
91 case AGENT_FLEE:
92 creature_target->getAIInterface()->setCanFlee(true);
93 break;
94 case AGENT_SPELL:
95 creature_target->getAIInterface()->addSpellToList(std::move(ai_spell));
96 break;
97 case AGENT_CALLFORHELP:
98 creature_target->getAIInterface()->setCanCallForHelp(true);
99 break;
100 default:
101 {
102 redSystemMessage(m_session, "Invalid ai_type {}", ai_type);
103 break;
104 }
105 }
106 return true;
107}

219bool ChatCommandHandler::HandleNpcDeleteCommand(const char* /*args*/, WorldSession* m_session)
220{
221 auto creature_target = GetSelectedCreature(m_session, true);
222 if (creature_target == nullptr)
223 return true;
224
225 if (creature_target->isPet())
226 {
227 systemMessage(m_session, "You can't delete a pet.");
228 return true;
229 }
230
231 if (creature_target->isSummon())
232 {
233 creature_target->Delete();
234 }
235 else
236 {
237 //creature_target->getAIInterface()->hideWayPoints(m_session->GetPlayer());
238
239 uint32_t spawn_id = creature_target->spawnid;
240 if (spawn_id != 0)
241 {
242 blueSystemMessage(m_session, "Creature {} ({}) deleted from creature_spawn table.", creature_target->GetCreatureProperties()->Name, spawn_id);
243 sGMLog.writefromsession(m_session, "used npc delete on creature %s (%u), pos %f %f %f", creature_target->GetCreatureProperties()->Name.c_str(), spawn_id, creature_target->GetPositionX(), creature_target->GetPositionY(), creature_target->GetPositionZ());
244 creature_target->DeleteFromDB();
245 }
246
247 if (creature_target->m_spawn)
248 {
249 uint32_t cellx = uint32_t(((Map::Terrain::_maxX - creature_target->m_spawn->x) / Map::Cell::cellSize));
250 uint32_t celly = uint32_t(((Map::Terrain::_maxY - creature_target->m_spawn->y) / Map::Cell::cellSize));
251
252 if (cellx <= Map::Cell::_sizeX && celly <= Map::Cell::_sizeY)
253 {
254 CellSpawns* sp = creature_target->getWorldMap()->getBaseMap()->getSpawnsList(cellx, celly);
255 if (sp != nullptr)
256 {
257 for (CreatureSpawnList::iterator itr = sp->CreatureSpawns.begin(); itr != sp->CreatureSpawns.end(); ++itr)
258 if ((*itr) == creature_target->m_spawn)
259 {
260 sp->CreatureSpawns.erase(itr);
261 break;
262 }
263 }
264 delete creature_target->m_spawn;
265 creature_target->m_spawn = nullptr;
266 }
267 }
268
269 creature_target->RemoveFromWorld(false, true);
270 }
271
272 return true;
273}

1108
1109 if (sscanf(args, "%hhu %u", &equipment_slot, &item_id) != 2)
1110 {
