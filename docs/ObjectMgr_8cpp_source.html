231void ObjectMgr::updateArenaTeamRankings() const
232{
233 std::lock_guard guard(m_arenaTeamLock);
234 for (auto& arenaTeams : m_arenaTeamMap)
235 {
236 std::vector<ArenaTeam*> ranking;
237 ranking.reserve(arenaTeams.size());
238
239 for (auto& arenaTeamPair : arenaTeams)
240 ranking.push_back(arenaTeamPair.second);
241
242 std::ranges::sort(ranking, ArenaSorter());
243 uint32_t rank = 1;
244
245 for (const auto& arenaTeam : ranking)
246 {
247 if (arenaTeam->m_stats.ranking != rank)
248 {
249 arenaTeam->m_stats.ranking = rank;
250 arenaTeam->saveToDB();
251 }
252
253 ++rank;
254 }
255 }
256}

610{
611 m_vendors.clear();
612
613 auto result = sMySQLStore.getWorldDBQuery("SELECT * FROM vendors");
614 if (result != nullptr)
615 {
616 std::vector<CreatureItem>* items = nullptr;
617
618 if (result->GetFieldCount() < 6 + 2)
619 {
620 sLogger.failure("Invalid format in vendors ({}/8) columns, not enough data to proceed.", result->GetFieldCount());
621 return;
622 }
623
624 do
625 {
626 Field* fields = result->Fetch();
627 items = createVendorList(fields[0].asUint32());
628
629 CreatureItem itm{};
630 itm.itemid = fields[1].asUint32();
631 itm.amount = fields[2].asUint32();
632 itm.available_amount = fields[3].asUint32();
633 itm.max_amount = fields[3].asUint32();
634 itm.incrtime = fields[4].asUint32();
635 itm.extended_cost = nullptr;
636 if (fields[5].asUint32() > 0)
637 {
638 const auto item_extended_cost = sItemExtendedCostStore.lookupEntry(fields[5].asUint32());
639 if (item_extended_cost == nullptr)
640 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "LoadVendors : Extendedcost for item {} references nonexistent EC {}", fields[1].asUint32(), fields[5].asUint32());
641 else
642 itm.extended_cost = item_extended_cost;
643 }
644
645 items->push_back(itm);
646 } while (result->NextRow());
647 }
648 sLogger.info("ObjectMgr : {} vendors loaded.", static_cast<uint32_t>(m_vendors.size()));
649}

700 {
701 Field* fields = result->Fetch();
702 uint32_t entry = fields[0].asUint32();
703
704 if (sAchievementStore.lookupEntry(entry) == nullptr)
705 {
706 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : Achievement reward entry {} has wrong achievement, ignore", entry);
707 continue;
708 }
709
710 AchievementReward reward;
711 reward.gender = fields[1].asUint8();
712 reward.titel_A = fields[2].asUint32();
713 reward.titel_H = fields[3].asUint32();
714 reward.itemId = fields[4].asUint32();
715 reward.sender = fields[5].asUint32();
716 reward.subject = fields[6].asCString() != nullptr ? fields[6].asCString() : "";
717 reward.text = fields[7].asCString() != nullptr ? fields[7].asCString() : "";
718
719 if (reward.gender > GENDER_NONE)
720 {
721 sLogger.debug("ObjectMgr : achievement reward {} has wrong gender {}.", entry, static_cast<uint32_t>(reward.gender));
722 }
723
724 bool dup = false;
725 auto bounds = m_achievementRewards.equal_range(entry);
726 for (const auto& iter : std::ranges::subrange(bounds.first, bounds.second))
727 {
728 if (iter.second.gender == GENDER_NONE || reward.gender == GENDER_NONE)
729 {
730 dup = true;
731 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : Achievement reward {} must have single GENDER_NONE ({}), ignore duplicate case", entry, GENDER_NONE);
732 break;
733 }
734 }
735
736 if (dup)
737 continue;
738
739 // must be title or mail at least
740 if (reward.titel_A == 0 && reward.titel_H == 0 && reward.sender == 0)
741 {
742 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} not have any rewards, ignore.", entry);
743 continue;
744 }
745
746 //check mail data before item for report including wrong item case
747 if (reward.sender != 0)
748 {
749 if (sMySQLStore.getCreatureProperties(reward.sender) == nullptr)
750 {
751 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} has invalid creature entry {} as sender, ignore.", entry, reward.sender);
752 continue;
753 }
754
755 if (reward.subject.empty() || reward.text.empty())
756 {
757 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} has invalid mail text data (subject, text), ignored", entry);
758 continue;
759 }
760 }
761
762 if (reward.itemId != 0)
763 {
764 if (sMySQLStore.getItemProperties(reward.itemId) == nullptr)
765 {
766 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} has invalid item id {}, ignore", entry, reward.itemId);
767 continue;
768 }
769
770 if (reward.sender == 0)
771 {
772 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} has item id {} but has no sender, ignore", entry, reward.itemId);
773 continue;
774 }
775 }
776
777 if (reward.titel_A != 0 && sCharTitlesStore.lookupEntry(reward.titel_A) == nullptr)
778 {
779 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} has invalid title id ({}) in `title_A`, ignore", entry, reward.titel_A);
780 continue;
781 }
782
783 if (reward.titel_H != 0 && sCharTitlesStore.lookupEntry(reward.titel_H) == nullptr)
784 {
785 sLogger.debugFlag(AscEmu::Logging::LF_DB_TABLES, "ObjectMgr : achievement_reward {} has invalid title id ({}) in `title_H`, ignore", entry, reward.titel_H);
786 continue;
787 }
788
789 m_achievementRewards.emplace(entry, reward);
790 ++count;
791
792 } while (result->NextRow());
793
794 sLogger.info("ObjectMgr : Loaded {} achievement rewards", count);
795}

933bool ObjectMgr::handleInstanceReputationModifiers(Player* _player, Unit* _unitVictim)
934{
935 const uint32_t team = _player->getTeam();
936
937 if (!_unitVictim->isCreature())
938 return false;
939
940 const auto itr = m_reputationInstance.find(_unitVictim->GetMapId());
941 if (itr == m_reputationInstance.end())
942 return false;
943
944 bool isBoss = false;
945 if (dynamic_cast<Creature*>(_unitVictim)->GetCreatureProperties()->isBoss)
946 isBoss = true;
947
948 int32_t repLimit;
949 int32_t value;
950
951 for (const auto& instanceRepMod : itr->second->mods)
952 {
953 if (!instanceRepMod->faction[team])
954 continue;
955
956 if (isBoss)
957 {
958 value = instanceRepMod->boss_rep_reward;
959 repLimit = instanceRepMod->boss_rep_limit;
960 }
961 else
962 {
963 value = instanceRepMod->mob_rep_reward;
964 repLimit = instanceRepMod->mob_rep_limit;
965 }
966
967 if (!value || (repLimit && _player->getFactionStanding(instanceRepMod->faction[team]) >= repLimit))
968 continue;
969
970 value = Util::float2int32(value * worldConfig.getFloatRate(RATE_KILLREPUTATION));
971 _player->modFactionStanding(instanceRepMod->faction[team], value);
972 }
973
974 return true;
975}
