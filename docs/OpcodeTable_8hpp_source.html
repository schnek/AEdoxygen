93 uint32_t getInternalIdForHex(uint16_t hex)
94 {
95 auto versionId = getVersionIdForAEVersion();
96
97 for (const auto table : _versionHexTable[versionId])
98 if (table.hexValue == hex)
99 return table.internalId;
100
101 return 0;
102 }

104 std::string getNameForOpcode(uint16_t hex)
105 {
106 const auto internalId = getInternalIdForHex(hex);
107
108 auto multiversionTable = multiversionOpcodeStore.find(internalId);
109 if (multiversionTable != multiversionOpcodeStore.end())
110 return multiversionTable->second.name + " [" + getNameForAEVersion() + "]";
111
112 return "Unknown internal id!";
113 }

115 std::string getNameForInternalId(uint32_t id)
116 {
117 auto multiversionTable = multiversionOpcodeStore.find(id);
118 if (multiversionTable != multiversionOpcodeStore.end())

124 uint16_t getHexValueForVersionId(uint32_t internalId)
125 {
126 auto versionId = getVersionIdForAEVersion();
127
128 if (versionId >= 0 && versionId < MAX_VERSION_INDEX)
129 {
130 auto multiversionTable = multiversionOpcodeStore.find(internalId);
131 if (multiversionTable != multiversionOpcodeStore.end())
132 return multiversionTable->second.hexValues[versionId];
133 }
134
135 return 0;
136 }
