251bool RighteousDefense(uint8_t /*effectIndex*/, Spell* s)
252{
253 //we will try to lure 3 enemies from our target
254
255 Unit* unitTarget = s->getUnitTarget();
256
257 if (!unitTarget || !s->getUnitCaster())
258 return false;
259
260 Unit* targets[3];
261 uint32_t targets_got = 0;
262
263 for (const auto& itr : unitTarget->getInRangeObjectsSet())
264 {
265 if (itr)
266 {
267 // don't add objects that are not creatures and that are dead
268 if (!itr->isCreature() || !static_cast<Creature*>(itr)->isAlive())
269 continue;
270
271 Creature* cr = static_cast<Creature*>(itr);
272 if (cr->getAIInterface()->getCurrentTarget() == unitTarget)
273 targets[targets_got++] = cr;
274
275 if (targets_got == 3)
276 break;
277 }
278 }
279
280 for (uint32_t j = 0; j < targets_got; j++)
281 {
282 //set threat to this target so we are the msot hated
283 float threat_to_him = targets[j]->getThreatManager().getThreat(unitTarget);
284 float threat_to_us = targets[j]->getThreatManager().getThreat(s->getUnitCaster());
285 float threat_dif = threat_to_him - threat_to_us;
286 if (threat_dif > 0) //should nto happen
287 targets[j]->getThreatManager().matchUnitThreatToHighestThreat(s->getUnitCaster());
288
289 targets[j]->getAIInterface()->onHostileAction(s->getUnitCaster());
290 targets[j]->getAIInterface()->setCurrentTarget(s->getUnitCaster());
291 }
292
293 return true;
294}
