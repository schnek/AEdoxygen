82class EyeForAnEye : public SpellScript
83{
84public:
85 SpellScriptExecuteState onDoProcEffect(SpellProc* spellProc, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, DamageInfo damageInfo) override
86 {
87 damage = damageInfo.realDamage * spellProc->getOverrideEffectDamage(EFF_INDEX_0) / 100;
88 if (damage == 0)
89 return SpellScriptExecuteState::EXECUTE_PREVENT;
90
91 return SpellScriptExecuteState::EXECUTE_OK;
92 }
93
94 SpellScriptExecuteState onCastProcSpell(SpellProc* /*spellProc*/, Unit* /*caster*/, Unit* /*victim*/, Spell* spell) override
95 {
96 spell->forced_basepoints->set(EFF_INDEX_0, damage);
97 damage = 0;
98 return SpellScriptExecuteState::EXECUTE_OK;
99 }
100
101 SpellScriptEffectDamage doCalculateEffect(Spell* spell, uint8_t /*effIndex*/, int32_t* damage) override
102 {
103 if (spell->getUnitCaster() == nullptr)
104 return SpellScriptEffectDamage::DAMAGE_DEFAULT;
105
106 // Damage cannot exceed 50% of paladin's health
107 const int32_t maxDmg = spell->getUnitCaster()->getMaxHealth() / 2;
108 if (*damage > maxDmg)
109 *damage = maxDmg;
110
111 return SpellScriptEffectDamage::DAMAGE_NO_BONUSES;
112 }
113
114private:
115 uint32_t damage = 0;
116};

270class SealOfVengeanceAndCorruptionDirect : public SpellScript
271{
272public:
273 bool canProc(SpellProc* spellProc, Unit* victim, SpellInfo const* /*castingSpell*/, DamageInfo /*damageInfo*/) override
274 {
275 if (victim == nullptr)
276 return false;
277
278 uint32_t auraId = SPELL_HOLY_VENGEANCE;
279 if (spellProc->getSpell()->getId() == SPELL_SEAL_OF_CORRUPTION_DIRECT)
280 auraId = SPELL_BLOOD_CORRUPTION;
281
282 const auto aur = victim->getAuraWithId(auraId);
283 if (aur == nullptr)
284 return false;
285
286#if VERSION_STRING == TBC
287 // As of Patch 2.2.0, Seal of Vengeance deals a small amount of Holy Damage if it strikes a target with 5 stacks of the debuff already applied.
288 if (aur->getStackCount() < 5)
289 return false;
290#elif VERSION_STRING >= Cata
291 // Once Censure is stacked to 5 times, each of the Paladin's attacks also deals 15% weapon damage
292 if (aur->getStackCount() < 5)
293 return false;
294#endif
295
296 return true;
297 }

299#if VERSION_STRING == WotLK
300 SpellScriptEffectDamage doCalculateEffect(Spell* spell, uint8_t /*effIndex*/, int32_t* damage) override
301 {
302 // According to WoWhead from 3.2.0 patch, all auto attacks and special attacks can proc this
303 // Weapon damage starts at 6.6% and goes up to 33% if target has five stacks of debuff
304 if (spell->getUnitTarget() == nullptr)
305 return SpellScriptEffectDamage::DAMAGE_DEFAULT;
306
307 uint32_t auraId = SPELL_HOLY_VENGEANCE;
308 if (spell->getSpellInfo()->getId() == SPELL_SEAL_OF_CORRUPTION_DIRECT)
309 auraId = SPELL_BLOOD_CORRUPTION;
310
311 float_t dmgPercent = 6.6f;
312 const auto aur = spell->getUnitTarget()->getAuraWithId(auraId);
313 if (aur != nullptr)
314 dmgPercent *= aur->getStackCount();
315
316 *damage = static_cast<uint32_t>(std::round(dmgPercent));
317 return SpellScriptEffectDamage::DAMAGE_NO_BONUSES;
318 }
