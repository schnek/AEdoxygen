18enum PathType
19{
20 PATHFIND_BLANK = 0x00, // path not built yet
21 PATHFIND_NORMAL = 0x01, // normal path
22 PATHFIND_SHORTCUT = 0x02, // travel through obstacles, terrain, air, etc (old behavior)
23 PATHFIND_INCOMPLETE = 0x04, // we have partial path to follow - getting closer to target
24 PATHFIND_NOPATH = 0x08, // no valid path at all or error in generating one
25 PATHFIND_NOT_USING_PATH = 0x10, // used when we are either flying/swiming or on map w/o mmaps
26 PATHFIND_SHORT = 0x20, // path is longer or equal to its limited path length
27 PATHFIND_FARFROMPOLY_START = 0x40, // start position is far from the mmap poligon
28 PATHFIND_FARFROMPOLY_END = 0x80, // end positions is far from the mmap poligon
29 PATHFIND_FARFROMPOLY = PATHFIND_FARFROMPOLY_START | PATHFIND_FARFROMPOLY_END, // start or end positions are far from the mmap poligon
30};
