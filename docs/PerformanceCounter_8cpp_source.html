25 float PerformanceCounter::GetCurrentCPUUsage()
26 {
27 unsigned long long now = Ascemu::SysInfo::GetTickCount();
28 unsigned long long now_cpu_usage = Ascemu::SysInfo::GetCPUUsage();
29 unsigned long long cpu_usage = now_cpu_usage - last_cpu_usage; // micro seconds
30 unsigned long long time_elapsed = now - last_update; // milli seconds
31
32 // converting to micro seconds
33 time_elapsed *= 1000;
34
35 float cpu_usage_percent = static_cast< float >(double(cpu_usage) / double(time_elapsed));
36
37 cpu_usage_percent *= 100;
38
39 // If we have more than 1 CPUs/cores,
40 // without dividing here we could get over 100%
41 cpu_usage_percent /= cpu_count;
42
43 last_update = now;
44 last_cpu_usage = now_cpu_usage;
45
46 return static_cast< float >(cpu_usage_percent);
