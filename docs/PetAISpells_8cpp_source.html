66class ShadowFiendAI : public CreatureAIScript
67{
68public:
69 static CreatureAIScript* Create(Creature* c) { return new ShadowFiendAI(c); }
70 explicit ShadowFiendAI(Creature* pCreature) : CreatureAIScript(pCreature) {}
71
72 void OnLoad() override
73 {
74 if (getCreature()->isPet())
75 {
76 if (auto pet = dynamic_cast<Pet*>(getCreature()))
77 {
78 auto playerOwner = pet->getUnitOwner();
79
80 const auto ownerBonus = static_cast<float>(playerOwner->GetDamageDoneMod(SCHOOL_SHADOW) * 0.375f); // 37.5%
81 pet->BaseAttackType = SCHOOL_SHADOW; // Melee hits are supposed to do damage with the shadow school
82 pet->setBaseAttackTime(MELEE, 1500); // Shadowfiend is supposed to do 10 attacks, sometimes it can be 11
83 pet->setMinDamage(pet->getMinDamage() + ownerBonus);
84 pet->setMaxDamage(pet->getMaxDamage() + ownerBonus);
85 pet->m_baseDamage[0] += ownerBonus;
86 pet->m_baseDamage[1] += ownerBonus;
87
88 const auto unitTarget = pet->getWorldMap()->getUnit(playerOwner->getTargetGuid());
89 if (unitTarget != nullptr && playerOwner->isValidTarget(unitTarget))
90 {
91 pet->getAIInterface()->onHostileAction(unitTarget);
92 pet->getAIInterface()->setCurrentTarget(unitTarget);
93 }
94 }
95 }
96 }

167class DancingRuneWeaponAI : public CreatureAIScript
168{
169public:
170 static CreatureAIScript* Create(Creature* c) { return new DancingRuneWeaponAI(c); }
171 explicit DancingRuneWeaponAI(Creature* pCreature) : CreatureAIScript(pCreature)
172 {
173 dpsCycle = 0;
174 dpsSpell = 0;
175 }
176
177 void OnLoad() override
178 {
179 getCreature()->setDisplayId(getCreature()->GetCreatureProperties()->Female_DisplayID);
180 getCreature()->setBaseAttackTime(MELEE, 2000);
181
182 if (getCreature()->isSummon())
183 {
184 if (auto summon = dynamic_cast<Summon*>(getCreature()))
185 {
186 auto unitOwner = summon->getUnitOwner();
187
188 if (unitOwner->isPlayer())
189 {
190 if (auto playerOwner = dynamic_cast<Player*>(unitOwner))
191 {
192 const auto item = playerOwner->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_MAINHAND);
193 if (item != nullptr)
194 {
195 for (uint8_t si = 0; si < 5; si++)
196 {
197 if (item->getItemProperties()->Spells[si].Id == 0)
198 continue;
199
200 if (item->getItemProperties()->Spells[si].Trigger == CHANCE_ON_HIT)
201 procSpell[si] = item->getItemProperties()->Spells[si].Id;
202 }
203
204 summon->setVirtualItemSlotId(MELEE, item->getEntry());
205 summon->setBaseAttackTime(MELEE, item->getItemProperties()->Delay);
206 }
207
208#if VERSION_STRING == WotLK
209 playerOwner->setPower(POWER_TYPE_RUNIC_POWER, 0);
210#endif
211 }
212 }
213
214 summon->setMinDamage(float(unitOwner->GetDamageDoneMod(SCHOOL_NORMAL)));
215 summon->setMaxDamage(float(unitOwner->GetDamageDoneMod(SCHOOL_NORMAL)));
216 }
217 }
218 }
219
220 void OnCombatStart(Unit* /*mTarget*/) override
221 {
222 RegisterAIUpdateEvent(getCreature()->getBaseAttackTime(MELEE));
223 }
224
225 void OnCombatStop(Unit* /*mTarget*/) override
226 {
227 RemoveAIUpdateEvent();
228 dpsCycle = 0;
229 }
230
231 void AIUpdate() override
232 {
233 const auto currentTarget = getCreature()->getThreatManager().getCurrentVictim();
234 if (!getCreature()->isCastingSpell() && currentTarget)
235 {
236 switch (dpsCycle)
237 {
238 case 0:
239 dpsSpell = 49921; // Plague Strike
240 break;
241 case 1:
242 dpsSpell = 49909; // Icy Touch
243 break;
244 case 2:
245 case 3:
246 dpsSpell = 55262; // Heart Strike x 2
247 break;
248 case 4:
249 dpsSpell = 51425; // Obliterate
250 break;
251 case 5:
252 dpsSpell = 49895; // Death Coil
253 break;
254 case 6:
255 case 7:
256 dpsSpell = 51425; // Obliterate x 2
257 break;
258 case 8:
259 case 9:
260 dpsSpell = 55262; // Heart Strike x 2
261 break;
262 case 10:
263 case 11:
264 dpsSpell = 49895; // Death Coil x 2
265 break;
266 default:
267 break;
268 }
269 dpsCycle++;
270
271 if (dpsCycle > 11)
272 dpsCycle = 0;
273
274 const auto nextSpell = sSpellMgr.getSpellInfo(dpsSpell);
275 if (nextSpell)
276 getCreature()->castSpell(currentTarget, nextSpell, true);
277 }
278 }
279
280 void OnHit(Unit* mTarget, float /*fAmount*/) override
281 {
282 for (auto p : procSpell)
283 {
284 if (p != 0)
285 {
286 const auto spellProc = sSpellMgr.getSpellInfo(p);
287 if (!spellProc)
288 return;
289
290 const auto randomProcChance = Util::getRandomUInt(99);
291 uint32_t spellProcChance = spellProc->getProcChance();
292 if (spellProcChance < 1)
293 spellProcChance = 10; // Got to be fair :P
294
295 if (randomProcChance <= spellProcChance)
296 {
297 const auto victim = spellProc->custom_self_cast_only ? getCreature() : mTarget;
298 getCreature()->castSpell(victim, spellProc, true);
299 }
300 }
301 }
302 }
303private:
304 int dpsCycle;
305 int dpsSpell;
306 int procSpell[5];
307};
