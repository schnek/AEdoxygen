15//.pet create
16bool ChatCommandHandler::HandlePetCreateCommand(const char* args, WorldSession* m_session)
17{
18 Player* selected_player = GetSelectedPlayer(m_session, true, true);
19 if (selected_player == nullptr)
20 return true;
21
22 if (!*args)
23 return false;
24
25 uint32_t entry = std::stoul(args);
26 CreatureProperties const* creature_proto = sMySQLStore.getCreatureProperties(entry);
27 if (creature_proto == nullptr)
28 {
29 redSystemMessage(m_session, "Creature Entry: {} is not a valid endtry!", entry);
30 return true;
31 }
32
33 if (selected_player->getPet() != nullptr)
34 {
35 redSystemMessage(m_session, "You must dismiss your current pet.");
36 return true;
37 }
38
39 if (!selected_player->findFreeActivePetSlot().has_value())
40 {
41#if VERSION_STRING < Cata
42 if (selected_player->isClassHunter())
43 redSystemMessage(m_session, "You must put your current pet to stables.");
44 else
45 redSystemMessage(m_session, "You must dismiss your inactive pet.");
46#else
47 redSystemMessage(m_session, "You must have one free active pet slot.");
48#endif
49 return true;
50 }
51
52 float followangle = -M_PI_FLOAT * 2;
53 LocationVector vector(selected_player->GetPosition());
54 vector.x += (3 * (cosf(followangle + selected_player->GetOrientation())));
55 vector.y += (3 * (sinf(followangle + selected_player->GetOrientation())));
56
57 const auto pet = sObjectMgr.createPet(entry, nullptr);
58 if (!pet->createAsSummon(creature_proto, nullptr, selected_player, vector, 0, nullptr, 0, PET_TYPE_HUNTER))
59 {
60 pet->DeleteMe();
61 return true;
62 }
63
64 if (selected_player != m_session->GetPlayer())
65 {
66 sGMLog.writefromsession(m_session, "used created pet with entry %u for player %s", entry, selected_player->getName().c_str());
67 blueSystemMessage(m_session, "Pet with entry {} created for player {}.", entry, selected_player->getName());
68 blueSystemMessage(selected_player->getSession(), "{} created a pet with entry {} for you.", m_session->GetPlayer()->getName(), entry);
69
70 }
71 else
72 {
73 blueSystemMessage(m_session, "Pet with entry {} created.", entry);
74 }
75
76 return true;
77}

79//.pet dismiss
80bool ChatCommandHandler::HandlePetDismissCommand(const char* /*args*/, WorldSession* m_session)
81{
82 Player* selected_player = GetSelectedPlayer(m_session, false, true);
83 Pet* selected_pet = nullptr;
84 Creature* selected_creature = nullptr;
85 if (selected_player != nullptr)
86 {
87 if (selected_player->getPet() == nullptr)
88 {
89 redSystemMessage(m_session, "Player has no pet.");
90 return true;
91 }
92 selected_player->getPet()->unSummon();
93 }
94 else
95 {
96 // no player selected, see if it is a pet or if creature has pet
97 selected_creature = GetSelectedCreature(m_session, false);
98 if (selected_creature == nullptr)
99 return false;
100
101 if (selected_creature->isPet())
102 {
103 selected_pet = dynamic_cast<Pet*>(selected_creature);
104 auto* const petOwner = selected_pet->getUnitOwner();
105 if (petOwner == nullptr)
106 return false;
107
108 selected_player = petOwner->isPlayer() ? selected_pet->getPlayerOwner() : nullptr;
109 selected_creature = petOwner->isCreature() ? dynamic_cast<Creature*>(petOwner) : nullptr;
110 }
111 else
112 {
113 if (selected_creature->getPet() == nullptr)
114 return false;
115
116 // Target is creature but not a pet
117 selected_pet = selected_creature->getPet();
118 selected_player = selected_pet->getPlayerOwner();
119 }
120
121 if (selected_pet == nullptr)
122 return false;
123
124 selected_pet->unSummon();
125 }
126
127 if (selected_creature != nullptr)
128 {
129 greenSystemMessage(m_session, "Dismissed {}'s pet.", selected_creature->GetCreatureProperties()->Name);
130 sGMLog.writefromsession(m_session, "used dismiss pet command on creature %s", selected_creature->GetCreatureProperties()->Name.c_str());
131 }
132 else if (selected_player != nullptr && selected_player != m_session->GetPlayer())
133 {
134 greenSystemMessage(m_session, "Dismissed {}'s pet.", selected_player->getName());
135 systemMessage(selected_player->getSession(), "{} dismissed your pet.", m_session->GetPlayer()->getName());
136 sGMLog.writefromsession(m_session, "used dismiss pet command on player %s", selected_player->getName().c_str());
137 }
138 else
139 {
140 greenSystemMessage(m_session, "Your pet is dismissed.");
141 }
142
143 return true;
144}

247bool ChatCommandHandler::HandlePetSetLevelCommand(const char* args, WorldSession* m_session)
248{
249 if (!*args)
250 return false;
251
252
253 int32_t newLevel = std::stoul(args);
254 if (newLevel < 1)
255 return false;
256
257 Player* selected_player = GetSelectedPlayer(m_session, false, true);
258 Pet* selected_pet = nullptr;
259 if (selected_player != nullptr)
260 {
261 selected_pet = selected_player->getPet();
262 if (selected_pet == nullptr)
263 {
264 redSystemMessage(m_session, "Player has no pet.");
265 return true;
266 }
267 }
268 else
269 {
270 Creature* selected_creature = GetSelectedCreature(m_session, false);
271 if (selected_creature == nullptr)
272 return false;
273
274 if (!selected_creature->isPet())
275 return false;
276
277 selected_pet = dynamic_cast< Pet* >(selected_creature);
278
279 selected_player = selected_pet->getPlayerOwner();
280 }
281
282 if (static_cast<uint32_t>(newLevel) > selected_player->getLevel())
283 {
284 redSystemMessage(m_session, "You can not set a pet level higher than thew player level!");
285 newLevel = selected_player->getLevel();
286 }
287
288 selected_pet->setLevel(newLevel);
289 selected_pet->setPetExperience(0);
290 selected_pet->setPetNextLevelExperience(selected_pet->getNextLevelXp(newLevel));
291 selected_pet->applyStatsForLevel();
292 selected_pet->updateSpellList();
293
294 if (selected_player != m_session->GetPlayer())
295 {
296 greenSystemMessage(m_session, "Set {}'s pet to level {}.", selected_player->getName(), static_cast<uint32_t>(newLevel));
297 systemMessage(selected_player->getSession(), "{} set your pet to level {}.", m_session->GetPlayer()->getName(), newLevel);
298 sGMLog.writefromsession(m_session, "leveled %s's pet to %u", selected_player->getName().c_str(), static_cast<uint32_t>(newLevel));
299 }
300 else
301 {
302 greenSystemMessage(m_session, "You set your pet to level {}.", static_cast<uint32_t>(newLevel));
303 }
304
305 return true;
