85void Pet::Update(unsigned long time_passed)
86{
87 Summon::Update(time_passed);
88
89#if VERSION_STRING < Cata
90 // Update happiness
91 if (isAlive() && m_petType == PET_TYPE_HUNTER)
92 {
93 if (m_happinessTimer == 0)
94 {
95 // According to Wowpedia pet loses 50 happiness in 6 minutes so 1042 in 7.5 seconds
96 int16_t happinessLoss = 1042;
97 // In combat reduce burn by half (guessed)
98 if (isInCombat())
99 happinessLoss >>= 1;
100
101 modPower(POWER_TYPE_HAPPINESS, -happinessLoss);
102 m_happinessTimer = PET_HAPPINESS_UPDATE_TIMER;
103 }
104 else if (!IsInBg())
105 {
106 // Pets won't lose happiness in battlegrounds
107 if (time_passed > m_happinessTimer)
108 m_happinessTimer = 0;
109 else
110 m_happinessTimer -= static_cast<uint16_t>(time_passed);
111 }
112 }
113#endif
114}

125void Pet::PrepareForRemove()
126{
127 Summon::PrepareForRemove();
128
129 auto* const plrOwner = getPlayerOwner();
130 if (plrOwner != nullptr)
131 plrOwner->eventDismissPet();
132
133 if (!m_petExpires)
134 {
135 // Remove dead minions from pet cache map
136 if (!isAlive() && m_petType != PET_TYPE_HUNTER)
137 {
138 if (plrOwner != nullptr)
139 plrOwner->removePetCache(m_petId);
140 }
141 // If pet/minion is alive and not being deleted, update and save it before removing
142 else if (!m_isScheduledForDeletion)
143 {
144 updatePetInfo(!m_isScheduledForTemporaryUnsummon);
145 // Save is not required when unsummoning temporarily
146 if (plrOwner != nullptr && !m_isScheduledForTemporaryUnsummon)
147 plrOwner->_savePet(nullptr, false, this);
148 }
149 }
150
151 m_isScheduledForDeletion = false;
152 m_isScheduledForTemporaryUnsummon = false;
153
154 if (IsInWorld() && IsActive())
155 deactivate(m_WorldMap);
156}

168void Pet::sendSpellsToController(Unit* controller, uint32_t duration)
169{
170 if (controller != m_unitOwner)
171 return;
172
173 // TODO: find out values for this
174 const uint16_t petFlags = 0;
175 const auto familyId = static_cast<uint16_t>(myFamily != nullptr ? myFamily->ID : 0);
176
177 // Action bar
178 AscEmu::Packets::SmsgPetActionsArray petActions{};
179 std::ranges::transform(m_actionBar, petActions.begin(), [](const auto& action) { return action.raw; });
180
181 // Send other spells in spell book for permanent pets
182 AscEmu::Packets::SmsgPetSpellsVector petSpells;
183 if (!m_petExpires)
184 {
185 petSpells.reserve(mSpells.size());
186 for (const auto& spell : mSpells)

188 }
189
190 controller->sendPacket(AscEmu::Packets::SmsgPetSpells(getGuid(), familyId, duration, getAIInterface()->getReactState(),
191 m_petAction, petFlags, std::move(petActions), std::move(petSpells)).serialise().get());

220bool Pet::createAsSummon(CreatureProperties const* creatureProperties, Creature* createdFromCreature, Unit* unitOwner, LocationVector const& position, uint32_t duration, SpellInfo const* createdBySpell, uint8_t effIndex, PetTypes type)
221{
222 // The caller has to delete us
223 if (creatureProperties == nullptr || unitOwner == nullptr || type == PET_TYPE_NONE)
224 return false;
225
226 Summon::load(creatureProperties, unitOwner, position, duration, createdBySpell != nullptr ? createdBySpell->getId() : 0);
227 const auto plrOwner = getPlayerOwner();
228
229 m_petType = type;
230 m_petExpires = duration > 0;
231 // Generate id only for permanent pets
232 if (plrOwner != nullptr && !m_petExpires)
233 m_petId = plrOwner->getFreePetNumber();
234
235 creature_properties = creatureProperties;
236 myFamily = sCreatureFamilyStore.lookupEntry(creatureProperties->Family);
237
238 Create(m_unitOwner->GetMapId(), position.x, position.y, position.z, m_unitOwner->GetOrientation());
239
240 const auto ownerLevel = m_unitOwner->getLevel();
241 auto petLevel = ownerLevel;
242 if (m_unitOwner->isPlayer() && createdFromCreature != nullptr)
243 {
244 petLevel = std::min(createdFromCreature->getLevel(), ownerLevel);
245 }
246 else if (createdBySpell != nullptr)
247 {
248 const auto intLevel = static_cast<int32_t>(ownerLevel) + static_cast<int32_t>(createdBySpell->getEffectMultipleValue(effIndex));
249 petLevel = static_cast<uint32_t>(std::max(1, intLevel));
250 }
251 setLevel(petLevel);
252
253 // Use the actual model from the tamed creature
254 const auto modelId = createdFromCreature != nullptr
255 ? createdFromCreature->getNativeDisplayId()
256 : creatureProperties->Male_DisplayID;
257 setDisplayId(modelId);
258 setNativeDisplayId(modelId);
259 eventModelChange();
260
261 if (m_petType == PET_TYPE_HUNTER && m_unitOwner->isPlayer())
262 {
263 // Newly tamed hunter pet
264 m_petName = "Pet";
265 if (myFamily != nullptr)
266 {
267#if VERSION_STRING < Cata
268 m_petName.assign(myFamily->name[sWorld.getDbcLocaleLanguageId()]);
269#else
270 m_petName.assign(myFamily->name[0]);
271#endif
272 }
273 }
274 else
275 {
276 // Other summoned pets (warlock minions, water elemental, npc pets etc)
277 _setNameForEntry(creatureProperties->Id, createdBySpell);
278 }
279
280 // Save permanent player pets (rest are updated in updatePetInfo)
281 if (plrOwner != nullptr && !m_petExpires)
282 {
283 auto petCache = std::make_unique<PetCache>();
284 petCache->number = m_petId;
285 petCache->type = m_petType;
286 petCache->model = getNativeDisplayId();
287 // Check if player has a free slot for new pet
288 const auto petSlot = plrOwner->findFreeActivePetSlot();
289 if (!petSlot.has_value())
290 return false;
291
292 petCache->slot = petSlot.value();
293 petCache->spellid = createdBySpell != nullptr ? createdBySpell->getId() : 0;
294 petCache->alive = true;
295
296 plrOwner->addPetCache(std::move(petCache), petCache->number);
297 }
298
299 return _preparePetForPush(nullptr);
300}

302bool Pet::loadFromDB(Player* owner, PetCache const* petCache)
303{
304 // The caller has to delete us
305 if (petCache == nullptr || owner == nullptr)
306 return false;
307
308 creature_properties = sMySQLStore.getCreatureProperties(petCache->entry);
309 // The caller has to delete us
310 if (creature_properties == nullptr)
311 return false;
312
313 auto pos = owner->GetPosition();
314 pos.x += 2.0f, pos.y += 2.0f;
315 Summon::load(creature_properties, owner, pos, 0, petCache->spellid);
316
317 myFamily = sCreatureFamilyStore.lookupEntry(creature_properties->Family);
318 Create(owner->GetMapId(), pos.x, pos.y, pos.z, owner->GetOrientation());
319
320 m_petType = static_cast<PetTypes>(petCache->type);
321 m_petId = petCache->number;
322 m_petName.assign(petCache->name);
323
324#if VERSION_STRING < Mop
325 m_talentResetTime = petCache->reset_time;
326 m_talentResetCost = petCache->reset_cost;
327#endif
328
329 // _preparePetForPush will validate level
330 setLevel(petCache->level);
331
332 const auto modelId = petCache->model != 0 ? petCache->model : creature_properties->Male_DisplayID;
333 setDisplayId(modelId);
334 setNativeDisplayId(modelId);
335 eventModelChange();
336
337 return _preparePetForPush(petCache);
338}

340void Pet::updatePetInfo(bool setPetOffline) const
341{
342 // Update only permanent pets
343 if (m_petExpires)
344 return;
345
346 const auto* plrOwner = getPlayerOwner();
347 if (plrOwner == nullptr)
348 return;
349
350 auto* const petCache = plrOwner->getModifiablePetCache(m_petId);
351 if (petCache == nullptr)
352 return;
353
354 petCache->entry = getEntry();
355 petCache->model = getNativeDisplayId();
356 petCache->name.assign(m_petName);
357 petCache->number = m_petId;
358 petCache->xp = getPetExperience();
359 petCache->level = getLevel();
360 petCache->active = !setPetOffline;
361
362 // Save actionbar
363 // TODO: copied from legacy method, rewrite this later
364 std::stringstream ss;
365 ss.rdbuf()->str("");
366 for (uint8_t i = 0; i < 10; ++i)
367 {
368 if (m_actionBar[i].raw != 0)
369 ss << m_actionBar[i].parts.spellId << " " << m_actionBar[i].parts.state;
370 else
371 ss << "0 0";
372
373 ss << ",";
374 }
375
376 petCache->actionbar = ss.str();
377#if VERSION_STRING < Mop
378 petCache->reset_cost = m_talentResetCost;
379 petCache->reset_time = m_talentResetTime;
380#else
381 petCache->reset_cost = 0;
382 petCache->reset_time = 0;
383#endif
384#if VERSION_STRING == WotLK || VERSION_STRING == Cata
385 petCache->talentpoints = getPetTalentPoints();
386#else
387 petCache->talentpoints = 0;
388#endif
389 petCache->petstate = getAIInterface()->getReactState();
390 petCache->alive = isAlive();
391 petCache->current_power = getPower(getPowerType());
392 petCache->current_hp = getHealth();
393#if VERSION_STRING < Cata
394 petCache->current_happiness = getPower(POWER_TYPE_HAPPINESS);
395#else
396 petCache->current_happiness = 0;
397#endif
398#if VERSION_STRING == Classic
399 petCache->renamable = hasUnitFlags(UNIT_FLAG_PET_CAN_BE_RENAMED);
400#else
401 petCache->renamable = getPetFlags() & PET_FLAG_CAN_BE_RENAMED ? true : false;
402#endif
403}

423{
424 if (m_petType != PET_TYPE_HUNTER)
425 {
426 // Summons and minions should always be deleted on unsummon
427 // unless they are just temporarily unsummoned
428 if (!m_isScheduledForDeletion && !m_isScheduledForTemporaryUnsummon)
429 {
430 abandonPet();
431 return;
432 }
433 }
434
435 if (auto* const plrOwner = getPlayerOwner())
436 {
437 plrOwner->addGroupUpdateFlag(GROUP_UPDATE_PET);
438
439 if (const auto battleground = plrOwner->getBattleground())

446 plrOwner->setLastBattlegroundPetId(m_petId);
447 else

449 }
450 }
451 }

453 Summon::unSummon();

490void Pet::giveXp(uint32_t xp)
491{
492 if (!m_unitOwner->isPlayer())
493 return;
494
495 auto newXP = xp + getPetExperience();
496 auto nextLevelXp = getPetNextLevelExperience();
497 auto petLevel = getLevel();
498
499 // Check if pet levels up
500 auto levelUp = false;
501 while (newXP >= nextLevelXp && newXP > 0)
502 {
503 // Check owner's level
504 if (m_unitOwner->getLevel() <= petLevel)
505 {
506 // Pet is at the same level as owner, do not level up
507 newXP = nextLevelXp - 1;
508 break;
509 }
510
511 levelUp = true;
512 ++petLevel;
513
514 newXP -= nextLevelXp;
515 nextLevelXp = getNextLevelXp(petLevel);
516 setPetNextLevelExperience(nextLevelXp);
517 }
518
519 if (levelUp)
520 {
521 setLevel(petLevel);
522#if VERSION_STRING == WotLK || VERSION_STRING == Cata
523 setPetTalentPoints(getTotalTalentPoints() - getSpentTalentPoints());
524#endif
525 applyStatsForLevel();
526 updateSpellList();
527#if VERSION_STRING == WotLK || VERSION_STRING == Cata
528 SendTalentsToOwner();
529#endif
530 }
531
532 setPetExperience(newXP);
533}

556void Pet::applyStatsForLevel()
557{
558 if (m_petType == PET_TYPE_HUNTER)
559 ApplyPetLevelAbilities();
560 else
561 ApplySummonLevelAbilities();
562
563 // Apply scale for this hunter pet family
564 // Size scaling is affected by level of the pet
565 if (m_petType == PET_TYPE_HUNTER && m_unitOwner->isPlayer() && myFamily != nullptr && myFamily->minsize > 0.0f)
566 {
567 float_t scaleValue = 0.0f;
568 const auto petLevel = getLevel();
569 if (petLevel >= myFamily->maxlevel)
570 {
571 scaleValue = myFamily->maxsize;
572 }
573 else if (petLevel < myFamily->minlevel)
574 {
575 scaleValue = myFamily->minsize;
576 }
577 else
578 {
579 const float_t levelDiff = static_cast<float_t>(petLevel - myFamily->minlevel) / myFamily->maxlevel;
580 const auto scaleDiff = static_cast<float_t>(myFamily->maxsize - myFamily->minsize);
581 scaleValue = scaleDiff * levelDiff + myFamily->minsize;
582 }
583
584 setScale(scaleValue);
585 }
586 else
587 {
588 setScale(1.0f);
589 }
590
591 setFullHealth();
592 setPower(POWER_TYPE_MANA, getMaxPower(POWER_TYPE_MANA));
593 setPower(POWER_TYPE_FOCUS, getMaxPower(POWER_TYPE_FOCUS));
594}

630bool Pet::_preparePetForPush(PetCache const* petCache)
631{
632 const auto isNewSummon = petCache == nullptr;
633 setEntry(creature_properties->Id);
634 setPhase(PHASE_SET, m_unitOwner->GetPhase());
635
636 // Validate pet level if owner is player
637 if (m_unitOwner->isPlayer())
638 {
639 const auto ownerLevel = m_unitOwner->getLevel();
640 // Pet level should never be higher
641 auto petLevel = std::min(getLevel(), ownerLevel);
642#if VERSION_STRING == WotLK
643 // Hunter pets should be max 5 levels below owner in wotlk
644 if (m_petType == PET_TYPE_HUNTER)
645 petLevel = ownerLevel > 5 && petLevel < (ownerLevel - 5) ? ownerLevel - 5 : petLevel;
646#elif VERSION_STRING >= Cata
647 // todo: confirm this
648 petLevel = ownerLevel;
649#endif
650 setLevel(petLevel);
651 }
652
653 if (m_petType == PET_TYPE_HUNTER)
654 {
655 setClass(WARRIOR);
656#if VERSION_STRING < Cata
657 setMaxPower(POWER_TYPE_HAPPINESS, 1000000);
658 setPower(POWER_TYPE_HAPPINESS, PET_HAPPINESS_UPDATE_VALUE / 2);
659#endif
660 setMaxPower(POWER_TYPE_FOCUS, 100);
661 setPower(POWER_TYPE_FOCUS, 100);
662 setPowerType(POWER_TYPE_FOCUS);
663
664 setPetExperience(isNewSummon ? 0 : petCache->xp);
665 setPetNextLevelExperience(getNextLevelXp(getLevel()));
666
667#if VERSION_STRING >= TBC
668 // Since TBC all hunter pets have same base attack speed
669 setBaseAttackTime(MELEE, 2000);
670 setBaseAttackTime(OFFHAND, 2000);
671 setBaseAttackTime(RANGED, 2000);
672#endif
673 }
674 else
675 {
676 setClass(MAGE);
677 setPowerType(POWER_TYPE_MANA);
678
679#if VERSION_STRING >= TBC
680 // todo: should warlock minions also use properties?
681 setBaseAttackTime(MELEE, creature_properties->AttackTime);
682 setBaseAttackTime(OFFHAND, creature_properties->AttackTime);
683 setBaseAttackTime(RANGED, creature_properties->RangedAttackTime);
684#endif
685 }
686
687#if VERSION_STRING == Classic
688 setBaseAttackTime(MELEE, creature_properties->AttackTime);
689 setBaseAttackTime(OFFHAND, creature_properties->AttackTime);
690 setBaseAttackTime(RANGED, creature_properties->RangedAttackTime);
691#endif
692 setAttackPower(0);
693 setAttackPowerMods(0);
694 setModCastSpeed(1.0f);
695 setBoundingRadius(creature_properties->BoundingRadius);
696 setCombatReach(creature_properties->CombatReach);
697 setSheathType(SHEATH_STATE_MELEE);
698
699 // Set base damage values
700 for (uint8_t i = 0; i < 2; ++i)
701 {
702 m_baseDamage[i] = 0;
703 m_baseOffhandDamage[i] = 0;
704 m_baseRangedDamage[i] = 0;
705 }
706
707 applyStatsForLevel();
708 resetCurrentSpeeds();
709
710 if (!isNewSummon)
711 {
712 setHealth(petCache->current_hp);
713 setPower(getPowerType(), petCache->current_power);
714
715 if (m_petType == PET_TYPE_HUNTER)
716 {
717#if VERSION_STRING < Cata
718 setPower(POWER_TYPE_HAPPINESS, petCache->current_happiness);
719#endif
720#if VERSION_STRING == WotLK || VERSION_STRING == Cata
721 setPetTalentPoints(petCache->talentpoints);
722#endif
723
724#if VERSION_STRING == Classic
725 if (!petCache->renamable)

1173void Pet::_addSpell(SpellInfo const* spellInfo, bool silently, uint8_t spellState)
1174{
1175 if (spellInfo == nullptr)
1176 return;
1177
1178 if (sSpellMgr.isSpellDisabled(spellInfo->getId()))
1179 return;
1180
1181 // Check if pet already knows this spell
1182 if (hasSpell(spellInfo->getId()))
1183 return;
1184
1185 uint32_t supercededSpellId = 0;
1186 if (spellInfo->hasSpellRanks())
1187 {
1188 // Check if pet already knows a higher rank of this spell
1189 const auto* higherRankInfo = spellInfo->getRankInfo()->getLastSpell();
1190 do
1191 {
1192 // If pet can know only one rank of this spell rank chain, try find an existing higher rank
1193 // Possible lower ranks are removed when a higher rank is added to spell map
1194 if (hasSpell(higherRankInfo->getId()))
1195 return;
1196
1197 if (higherRankInfo->getId() == spellInfo->getId())
1198 break;
1199
1200 higherRankInfo = higherRankInfo->getRankInfo()->getPreviousSpell();
1201 } while (higherRankInfo != nullptr);
1202
1203 // Pets can have only a single rank known from spell rank chain, remove all previous ranks
1204 // TODO: confirm if true for all pets and spells
1205 const auto* previousRank = spellInfo->getRankInfo()->getPreviousSpell();
1206 const auto sendRemovePacket = !spellInfo->isTalent();
1207 while (previousRank != nullptr)
1208 {
1209 const auto [removedSpell, tmpSpellState] = _removeSpell(previousRank->getId(), sendRemovePacket, true);
1210 if (removedSpell)
1211 {
1212 if (!spellInfo->isTalent())
1213 {
1214 supercededSpellId = previousRank->getId();
1215 spellState = tmpSpellState;
1216 }
1217 break;
1218 }
1219
1220 previousRank = previousRank->getRankInfo()->getPreviousSpell();
1221 }
1222 }
1223
1224 if (spellInfo->isPassive())
1225 spellState = PET_SPELL_STATE_PASSIVE;
1226
1227 mSpells.try_emplace(spellInfo->getId(), spellState);
1228
1229 if (spellState == PET_SPELL_STATE_AUTOCAST)
1230 _createAISpell(spellInfo);
1231
1232 // Update action bar
1233 if (!spellInfo->isPassive())
1234 {
1235 if (supercededSpellId != 0)
1236 {
1237 for (auto& button : m_actionBar)
1238 {
1239 if (button.parts.spellId == supercededSpellId)
1240 button = PetActionButtonData{ .parts{.spellId = spellInfo->getId(), .state = spellState} };
1241 }
1242 }
1243 else
1244 {
1245 // Try find an empty slot
1246 for (auto& button : m_actionBar)
1247 {
1248 if (button.raw == 0)
1249 {
1250 button = PetActionButtonData{ .parts{.spellId = spellInfo->getId(), .state = spellState} };
1251 break;
1252 }
1253 }
1254 }
1255 }
1256
1257 if (!IsInWorld())
1258 return;
1259
1260#if VERSION_STRING >= WotLK
1261 if (!silently && !(spellInfo->getAttributes() & ATTRIBUTES_NO_CAST))
1262 {
1263 if (auto* const plrOwner = getPlayerOwner())
1264 plrOwner->sendPacket(AscEmu::Packets::SmsgPetLearnedSpell(spellInfo->getId()).serialise().get());
1265 }
1266#endif
1267
1268 // Cast talents and auto castable spells with learn spell effect
1269 if ((spellInfo->isTalent() || spellInfo->getAttributesEx() & ATTRIBUTESEX_AUTOCASTED_AT_SPELL_LEARN) && spellInfo->hasEffect(SPELL_EFFECT_LEARN_SPELL))
1270 castSpell(getGuid(), spellInfo, true);
1271 // Cast passive spells
1272 else if (spellInfo->isPassive())
1273 castSpell(getGuid(), spellInfo, true);
1274
1275 sendSpellsToController(m_unitOwner, m_duration);
1276}

1278std::tuple<bool, uint8_t> Pet::_removeSpell(uint32_t spellId, [[maybe_unused]]bool silently/* = false*/, bool removingPreviousRanks/* = false*/)
1279{
1280 const auto itr = std::as_const(mSpells).find(spellId);
1281 if (itr == mSpells.cend())
1282 return { false, PET_SPELL_STATE_DEFAULT };
1283
1284 const auto spellState = itr->second;
1285
1286 mSpells.erase(itr);
1287 removeAllAurasByIdForGuid(spellId, getGuid());
1288
1289 const auto* const spellInfo = sSpellMgr.getSpellInfo(spellId);
1290 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
1291 {
1292 switch (spellInfo->getEffect(i))
1293 {
1294 case SPELL_EFFECT_LEARN_SPELL:
1295 // If spell teaches another spell, remove it recursively as well
1296 if (const auto taughtSpellId = spellInfo->getEffectTriggerSpell(i))
1297 removeSpell(taughtSpellId);
1298 break;
1299 case SPELL_EFFECT_TRIGGER_SPELL:
1300 if (const auto triggerSpellId = spellInfo->getEffectTriggerSpell(i))
1301 removeAllAurasByIdForGuid(triggerSpellId, getGuid());
1302 break;
1303 default:
1304 break;
1305 }
1306 }
1307
1308#if VERSION_STRING >= WotLK
1309 if (!silently)
1310 {
1311 if (auto* const plrOwner = getPlayerOwner())
1312 {
1313 if (plrOwner->IsInWorld())
1314 plrOwner->sendPacket(AscEmu::Packets::SmsgPetUnlearnedSpell(spellId).serialise().get());
1315 }
1316 }
1317#endif
1318
1319 if (!removingPreviousRanks)
1320 {
1321 // Remove spell from action bar as well
1322 for (auto& button : m_actionBar)
1323 {
1324 if (button.parts.spellId == spellId)
1325 button.raw = 0;
1326 }
1327 }
1328
1329 // Remove AI spell
1330 _removeAISpell(spellId);
1331
1332 return { true, spellState };
1333}

1335CreatureAISpells* Pet::_createAISpell(SpellInfo const* spellInfo)
1336{
1337 if (spellInfo == nullptr)
1338 return nullptr;
1339
1340 uint8_t targetType = TARGET_SELF;
1341 std::function<Unit* ()> targetFunc = nullptr;
1342
1343 // Set AI target by spell target mask
1344 const auto spellTargetMask = spellInfo->getRequiredTargetMask(true);
1345 if (spellTargetMask & SPELL_TARGET_AREA_SELF)
1346 {
1347 targetType = TARGET_SOURCE;
1348 }
1349 else if (spellTargetMask & SPELL_TARGET_AREA_CURTARGET)
1350 {
1351 targetType = TARGET_DESTINATION;
1352 }
1353 else if (spellTargetMask & SPELL_TARGET_OBJECT_PETOWNER)
1354 {
1355 targetType = TARGET_FUNCTION;
1356 targetFunc = [this]() { return getUnitOwner(); };
1357 }
1358 else if (spellTargetMask & SPELL_TARGET_REQUIRE_ATTACKABLE)
1359 {
1360 targetType = TARGET_ATTACKING;
1361 }
1362
1363 // Autocasted imp Fire Shield should be casted on party members in combat
1364 if (spellInfo->hasEffectApplyAuraName(SPELL_AURA_DAMAGE_SHIELD))
1365 {
1366 targetType = TARGET_FUNCTION;
1367 targetFunc = [this, spellId = spellInfo->getId()]() {
1368 std::vector<Unit*> targetVector;
1369 if (auto* const plrOwner = getPlayerOwner())
1370 {
1371 // Try group or just owner itself
1372 if (auto* const group = plrOwner->getGroup())
1373 {
1374 // Get group members only from same subgroup
1375 if (const auto* subGroup = group->GetSubGroup(plrOwner->getSubGroupSlot()))
1376 {
1377 group->Lock();
1378 for (const auto& member : subGroup->getGroupMembers())
1379 {
1380 if (auto* const plrMember = sObjectMgr.getPlayer(member->guid))
1381 targetVector.emplace_back(plrMember);
1382 }
1383 group->Unlock();
1384 }
1385 }
1386 else
1387 {
1388 targetVector.emplace_back(plrOwner);
1389 }
1390 }
1391 else if (auto* const owner = getUnitOwner())
1392 {
1393 targetVector.emplace_back(owner);
1394 }
1395
1396 if (targetVector.size() > 1)
