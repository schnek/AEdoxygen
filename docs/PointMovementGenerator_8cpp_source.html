32template<class T>
33void PointMovementGenerator<T>::doInitialize(T* owner)
34{
35 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING | MOVEMENTGENERATOR_FLAG_TRANSITORY | MOVEMENTGENERATOR_FLAG_DEACTIVATED);
36 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED);
37
38 if (_movementId == EVENT_CHARGE_PREPATH)
39 {
40 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
41 return;
42 }
43
44 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
45 {
46 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
47 owner->stopMoving();
48 return;
49 }
50
51 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
52
53 MovementMgr::MoveSplineInit init(owner);
54 init.MoveTo(_x, _y, _z , _generatePath);
55 if (_speed > 0.0f)
56 init.SetVelocity(_speed);
57
58 if (_finalOrient)
59 init.SetFacing(*_finalOrient);
60
61 init.Launch();
62
63 // Call for creature group update
64 if (Creature* creature = owner->ToCreature())
65 creature->signalFormationMovement();
66}

76template<class T>
77bool PointMovementGenerator<T>::doUpdate(T* owner, uint32_t /*diff*/)
78{
79 if (!owner)
80 return false;
81
82 if (_movementId == EVENT_CHARGE_PREPATH)
83 {
84 if (owner->movespline->Finalized())
85 {
86 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
87 return false;
88 }
89 return true;
90 }
91
92 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->ToUnit()->isCastingSpell())
93 {
94 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
95 owner->stopMoving();
96 return true;
97 }
98
99 if ((MovementGenerator::hasFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED) && owner->movespline->Finalized()) || (MovementGenerator::hasFlag(MOVEMENTGENERATOR_FLAG_SPEED_UPDATE_PENDING) && !owner->movespline->Finalized()))
100 {
101 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED | MOVEMENTGENERATOR_FLAG_SPEED_UPDATE_PENDING);
102
103 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
104
105 MovementMgr::MoveSplineInit init(owner);
106 init.MoveTo(_x, _y, _z, _generatePath);
107 if (_speed > 0.0f) // Default value for point motion type is 0.0, if 0.0 spline will use GetSpeed on unit
108 init.SetVelocity(_speed);
109 init.Launch();
110
111 // Call for creature group update
112 if (Creature* creature = owner->ToCreature())
113 creature->signalFormationMovement();
114 }
115
116 if (owner->movespline->Finalized())
117 {
118 MovementGenerator::removeFlag(MOVEMENTGENERATOR_FLAG_TRANSITORY);
119 MovementGenerator::addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
120 return false;
121 }
122 return true;
123}

170void AssistanceMovementGenerator::finalize(Unit* owner, bool active, bool movementInform)
171{
172 addFlag(MOVEMENTGENERATOR_FLAG_FINALIZED);
173 if (active)
174 owner->removeUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
175
176 if (movementInform && hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED) && owner->isAlive())
177 {
178 // Make sound and wait 1.5 seconds still before calling for friends
179 owner->SendAIReaction();
180 owner->getMovementManager()->moveSeekAssistanceDistract(1500);
181 }
182}
