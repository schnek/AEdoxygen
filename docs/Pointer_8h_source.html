20/**
21 Acts like a pointer to a value of type ValueType (i.e.,
22 ValueType*), but can operate through accessor methods as well as on
23 a value in memory. This is useful for implementing scripting
24 languages and other applications that need to connect existing APIs
25 by reference.
26
27 Because the accessors require values to be passed by value (instead of by reference)
28 this is primarily useful for objects whose memory size is small.
29
30 \code
31 class Foo {
32 public:
33 void setEnabled(bool b);
34 bool getEnabled() const;
35 };
36
37 Foo f;
38 bool b;
39
40 Pointer<bool> p1(&b);
41 Pointer<bool> p2(&f, &Foo::getEnabled, &Foo::setEnabled);
42
43 *p1 = true;
44 *p2 = false;
45 *p2 = *p1; \/\/ Value assignment
46 p2 = p1; \/\/ Pointer aliasing
47
48 \/\/ Or, equivalently:
49 p1.setValue(true);
50 p2.setValue(false);
51
52 p2.setValue(p1.getValue());
53 p2 = p1;
54 \endcode
55
56 <i>Note:</i> Because of the way that dereference is implemented, you cannot pass <code>*p</code> through a function
57 that takes varargs (...), e.g., <code>printf("%d", *p)</code> will produce a compile-time error. Instead use
58 <code>printf("%d",(bool)*p)</code> or <code>printf("%d", p.getValue())</code>.
59
60 \cite McGuire, GUIs for Real-time Programs, using Universal Pointers, SIGGRAPH 2008 Poster.
61 */
