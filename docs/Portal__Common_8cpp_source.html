29void CommonPortalAI::AIUpdate(unsigned long time_passed)
30{
31 if (!mInstance || mInstance->getLocalData(DATA_MAIN_EVENT_STATE) == EncounterStates::NotStarted)
32 {
33 scriptEvents.resetEvents();
34 summons.despawnAll();
35 despawn();
36 return;
37 }
38
39 scriptEvents.updateEvents(time_passed, getScriptPhase());
40
41 while (uint32_t eventId = scriptEvents.getFinishedEvent())
42 {
43 switch (eventId)
44 {
45 case 1:
46 {
47 if (Util::getRandomInt(0, 1))
48 {
49 uint32_t entrys[] = { NPC_PORTAL_GUARDIAN, NPC_PORTAL_KEEPER };
50 uint32_t entry = entrys[Util::getRandomInt(0, 1)];
51 if (Creature* portalKeeper = summonCreature(entry, getCreature()->GetPosition(), CORPSE_TIMED_DESPAWN, 20 * TimeVarsMs::Second))
52 getCreature()->castSpell(portalKeeper, SPELL_PORTAL_CHANNEL, false);
53
54 if (Creature* sinclariTrigger = mInstance->getCreatureFromData(DATA_SINCLARI_TRIGGER))
55 {
56 if (entry == NPC_PORTAL_GUARDIAN)
57 sinclariTrigger->GetScript()->sendDBChatMessage(SAY_SINCLARI_PORTAL_GUARDIAN);
58 else if (entry == NPC_PORTAL_KEEPER)
59 sinclariTrigger->GetScript()->sendDBChatMessage(SAY_SINCLARI_PORTAL_KEEPER);
60 }
61 }
62 else
63 {
64 uint8_t k = mInstance->getLocalData(DATA_WAVE_COUNT) < 12 ? 3 : 4;
65 while (k--)
66 {
67 uint32_t entrys[] = { NPC_AZURE_INVADER_1, NPC_AZURE_INVADER_2, NPC_AZURE_SPELLBREAKER_1, NPC_AZURE_SPELLBREAKER_2, NPC_AZURE_MAGE_SLAYER_1, NPC_AZURE_MAGE_SLAYER_2, NPC_AZURE_BINDER_1, NPC_AZURE_BINDER_2 };
68 uint32_t entry = entrys[Util::getRandomInt(0, 6)];
69 summonCreature(entry, getCreature()->GetPosition(), CORPSE_TIMED_DESPAWN, 20 * TimeVarsMs::Second);
70 }
71 }
72
73 scriptEvents.addEvent(2, 1 * TimeVarsMs::Second);
74 } break;
75 case 2:
76 {
77 getCreature()->setVisible(false);
78 } break;
79 default:
80 break;
81 }
82 }
83}
