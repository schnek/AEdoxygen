29void ElitePortalAI::AIUpdate(unsigned long time_passed)
30{
31 if (!mInstance || mInstance->getLocalData(DATA_MAIN_EVENT_STATE) == EncounterStates::NotStarted)
32 {
33 scriptEvents.resetEvents();
34 summons.despawnAll();
35 despawn();
36 return;
37 }
38
39 scriptEvents.updateEvents(time_passed, getScriptPhase());
40
41 while (uint32_t eventId = scriptEvents.getFinishedEvent())
42 {
43 switch (eventId)
44 {
45 case 1:
46 {
47 uint8_t k = mInstance->getLocalData(DATA_WAVE_COUNT) < 12 ? 3 : 4;
48 while (k--)
49 {
50 uint32_t entrys[] = { NPC_AZURE_CAPTAIN_1, NPC_AZURE_RAIDER_1, NPC_AZURE_STALKER_1, NPC_AZURE_SORCEROR_1 };
51 summonCreature(entrys[Util::getRandomInt(0, 3)], getCreature()->GetPosition(), CORPSE_TIMED_DESPAWN, 20 * TimeVarsMs::Second);
52 }
53
54 if (Creature* sinclariTrigger = mInstance->getCreatureFromData(DATA_SINCLARI_TRIGGER))
55 if (sinclariTrigger->GetScript())
56 sinclariTrigger->GetScript()->sendDBChatMessage(SAY_SINCLARI_ELITE_SQUAD);
57
58 scriptEvents.addEvent(2, 1 * TimeVarsMs::Second);
59 } break;
60 case 2:
61 {
62 getCreature()->setVisible(false);
63 } break;
64 default:
65 break;
66 }
67 }
68}
