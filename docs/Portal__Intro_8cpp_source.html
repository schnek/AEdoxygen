33void IntroPortalAI::AIUpdate(unsigned long time_passed)
34{
35 if (!mInstance || mInstance->getLocalData(DATA_MAIN_EVENT_STATE) != EncounterStates::NotStarted)
36 {
37 scriptEvents.resetEvents();
38 summons.despawnAll();
39 despawn();
40 return;
41 }
42
43 scriptEvents.updateEvents(time_passed, getScriptPhase());
44
45 while (uint32_t eventId = scriptEvents.getFinishedEvent())
46 {
47 switch (eventId)
48 {
49 case 1:
50 {
51 // Limit the number of current summons
52 if (summons.size() < 3)
53 {
54 uint32_t entrys[] = { NPC_AZURE_INVADER_1, NPC_AZURE_MAGE_SLAYER_1, NPC_AZURE_BINDER_1 };
55 summonCreature(entrys[Util::getRandomInt(0, 2)], getCreature()->GetPosition(), CORPSE_TIMED_DESPAWN, 20 * TimeVarsMs::Second);
56 }
57
58 scriptEvents.addEvent(1, 15 * TimeVarsMs::Second);
59 } break;
60 default:
61 break;
62 }
63 }
64}
