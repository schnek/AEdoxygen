23template<typename Type>
24class PreallocatedQueue
25{
26 public:
27
28 PreallocatedQueue(uint32_t size, uint32_t realloc_size = 100)
29 {
30 // create buffer
31 Buffer = (uint8_t*)malloc(size * sizeof(Type));
32 m_readpos = m_writepos = 0;
33 m_buffersize = size * sizeof(Type);
34 type_size = sizeof(Type);
35 m_reallocsize = realloc_size;
36 }
37
38 ~PreallocatedQueue()
39 {
40 free(Buffer);
41 }
42
43 void push_back(Type ptr)
44 {
45 if ((type_size + m_writepos) > m_buffersize)
46 reallocate(m_buffersize + (m_reallocsize * type_size));
47
48 memcpy(&Buffer[m_writepos], &ptr, type_size);
49 m_writepos += type_size;
50 }
51
52 Type pop_front()
53 {
54 assert((m_readpos + type_size) <= m_writepos);
55 Type returner;
56 memcpy(&returner, &Buffer[m_readpos], type_size);
57 m_readpos += type_size;
58
59 // clear buffer completely if we're at the end of the buffer
60 if (m_readpos == m_writepos)
61 clear();
62
63 return returner;
64 }
65
66 void clear()
67 {
68 m_readpos = m_writepos = 0;
69 }
70
71 bool empty()
72 {
73 if (m_readpos == m_writepos)
74 return true;
75 return false;
76 }
77
78 void reallocate(uint32_t size)
79 {
80 Buffer = (uint8_t*)realloc(Buffer, size);
81 assert(Buffer);
82 m_buffersize = size;
83 }
84
85 protected:

91 uint32_t m_writesize;

95};
