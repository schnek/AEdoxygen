27bool Penance(uint8_t /*effectIndex*/, Spell* pSpell)
28{
29 if (!pSpell->getPlayerCaster() || !pSpell->getPlayerCaster()->isAlive() ||
30 !pSpell->getUnitTarget() || !pSpell->getUnitTarget()->isAlive())
31 return true;
32
33 Unit* target = pSpell->getUnitTarget();
34 Player* player = pSpell->getPlayerCaster();
35
36 // index 0 contains the spell for the first tick, index 1 is the peroidic cast spell.
37 uint32_t hostileSpell[] = { 0, 0 };
38 uint32_t friendlySpell[] = { 0, 0 };
39
40 switch (pSpell->getSpellInfo()->getId())
41 {
42 case 47540: //Rank 1
43 hostileSpell[0] = 47666;
44 hostileSpell[1] = 47758;
45
46 friendlySpell[0] = 47750;
47 friendlySpell[1] = 47757;
48 break;
49 case 53005:
50 hostileSpell[0] = 52998;
51 hostileSpell[1] = 53001;
52
53 friendlySpell[0] = 52983;
54 friendlySpell[1] = 52986;
55 break;
56 case 53006:
57 hostileSpell[0] = 52999;
58 hostileSpell[1] = 53002;
59
60 friendlySpell[0] = 52984;
61 friendlySpell[1] = 52987;
62 break;
63 case 53007:
64 hostileSpell[0] = 53000;
65 hostileSpell[1] = 53003;
66
67 friendlySpell[0] = 52985;
68 friendlySpell[1] = 52988;
69 break;
70 }
71
72 if (player->isValidTarget(target)) // Do holy damage
73 {
74 // First tick is instant.
75 player->castSpell(target, hostileSpell[0], true);
76 player->castSpell(target, hostileSpell[1], false);
77 }
78 else // Heal
79 {
80 player->castSpell(target, friendlySpell[0], true);
81 player->castSpell(target, friendlySpell[1], false);
82 }
83 return true;
84}
