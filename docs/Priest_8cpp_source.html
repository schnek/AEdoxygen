151class DivineAegis : public SpellScript
152{
153public:
154 bool canProc(SpellProc* /*spellProc*/, Unit* /*victim*/, SpellInfo const* castingSpell, DamageInfo /*damageInfo*/) override
155 {
156 // Should proc only on heal spells
157 if (castingSpell == nullptr || !castingSpell->hasEffect(SPELL_EFFECT_HEAL))
158 return false;
159 return true;
160 }
161
162 SpellScriptExecuteState onDoProcEffect(SpellProc* spellProc, Unit* victim, SpellInfo const* /*castingSpell*/, DamageInfo damageInfo) override
163 {
164 if (victim == nullptr)
165 return SpellScriptExecuteState::EXECUTE_PREVENT;
166
167 absorbAmount = static_cast<uint32_t>(std::ceil(damageInfo.realDamage * spellProc->getOverrideEffectDamage(EFF_INDEX_0) / 100.0f));
168 // Patch 3.1.0: Divine Aegis effects will now stack, however the amount absorbed cannot exceed 125 * level(of the target).
169 if (absorbAmount > (victim->getLevel() * 125))
170 absorbAmount = victim->getLevel() * 125;
171
172 return SpellScriptExecuteState::EXECUTE_OK;
173 }
174
175 SpellScriptExecuteState onCastProcSpell(SpellProc* /*spellProc*/, Unit* /*caster*/, Unit* /*victim*/, Spell* spell) override
176 {
177 spell->forced_basepoints->set(EFF_INDEX_0, absorbAmount);
178 absorbAmount = 0;
179 return SpellScriptExecuteState::EXECUTE_OK;
180 }
181
182private:
183 uint32_t absorbAmount = 0;

363class VampiricEmbrace : public SpellScript
364{
365public:
366 void onCreateSpellProc(SpellProc* spellProc, Object* /*obj*/) override
367 {
368#if VERSION_STRING < WotLK
369 spellProc->setCastedByProcCreator(true);
370#else
371 spellProc->setCastedOnProcOwner(true);
372#endif
373 }
374
375 bool canProc([[maybe_unused]]SpellProc* spellProc, [[maybe_unused]]Unit* victim, SpellInfo const* castingSpell, DamageInfo damageInfo) override
376 {
377 if (castingSpell == nullptr)
378 return false;
379
380#if VERSION_STRING < WotLK
381 // Should proc only for the priest who casted Vampiric Embrace
382 if (victim == nullptr || victim->getGuid() != spellProc->getCasterGuid())
383 return false;
384#endif
385
386 // Only on shadow damage
387 // Check also that it's not procing from wand
388 if (!(damageInfo.schoolMask & SCHOOL_MASK_SHADOW) || damageInfo.isHeal || damageInfo.weaponType == RANGED)
389 return false;
390
391 // Only on single target spells
392 //\ todo: not sure on this
393 if (!(castingSpell->hasTargetType(EFF_TARGET_SINGLE_ENEMY) || castingSpell->hasTargetType(EFF_TARGET_SELECTED_ENEMY_CHANNELED)))
394 return false;
395
396 // Check spell's effects
397 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
398 {
399 if (castingSpell->getEffect(i) == SPELL_EFFECT_NULL)
400 continue;
401
402 // Procs on normal shadow damage
403 if (castingSpell->getEffect(i) == SPELL_EFFECT_SCHOOL_DAMAGE)
404 return true;
405
406 // Procs on periodic shadow damage
407 if (damageInfo.isPeriodic &&
408 (castingSpell->getEffectApplyAuraName(i) == SPELL_AURA_PERIODIC_DAMAGE || castingSpell->getEffectApplyAuraName(i) == SPELL_AURA_PERIODIC_LEECH))
409 return true;
410 }
411
412 return false;
413 }
414
415 SpellScriptExecuteState onDoProcEffect(SpellProc* spellProc, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, DamageInfo damageInfo) override
416 {
417 const auto healPct = spellProc->getOverrideEffectDamage(EFF_INDEX_0);
418 selfHeal = static_cast<uint32_t>(std::ceil(damageInfo.realDamage * healPct / 100.0f));
419
420#if VERSION_STRING >= Mop
421 // TODO: Mop
422#elif VERSION_STRING == Cata
423 const auto partyHealPct = std::round(healPct / 2.0f);
424 partyHeal = static_cast<uint32_t>(std::ceil(damageInfo.realDamage * partyHealPct / 100.0f));
425#elif VERSION_STRING == WotLK
426 const auto partyHealPct = std::round(healPct / 5.0f);
427 partyHeal = static_cast<uint32_t>(std::ceil(damageInfo.realDamage * partyHealPct / 100.0f));
428#endif
429
430 if (selfHeal == 0)
431 return SpellScriptExecuteState::EXECUTE_PREVENT;
432
433 return SpellScriptExecuteState::EXECUTE_OK;
434 }
435
436 SpellScriptExecuteState onCastProcSpell(SpellProc* /*spellProc*/, Unit* /*caster*/, Unit* /*victim*/, Spell* spell) override
437 {
438#if VERSION_STRING < WotLK
439 // Same amount for party and self in Classic and TBC
440 spell->forced_basepoints->set(EFF_INDEX_0, selfHeal);
441#else
442 spell->forced_basepoints->set(EFF_INDEX_0, partyHeal);
443 spell->forced_basepoints->set(EFF_INDEX_1, selfHeal);
444#endif
445 selfHeal = 0;
446 partyHeal = 0;
447 return SpellScriptExecuteState::EXECUTE_OK;
448 }
449
450#if VERSION_STRING >= WotLK && VERSION_STRING < Mop
451 void filterEffectTargets(Spell* spell, uint8_t effIndex, std::vector<uint64_t>* effTargets) override
452 {
453 if (effIndex != EFF_INDEX_0)
454 return;
455
456 // Remove caster from party effect
457 effTargets->erase(std::remove(effTargets->begin(), effTargets->end(), spell->getCaster()->getGuid()), effTargets->end());
458 }
459#endif
460
461private:
462 uint32_t selfHeal = 0;
463 uint32_t partyHeal = 0;
464};
