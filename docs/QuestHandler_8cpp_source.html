41std::unique_ptr<WorldPacket> WorldSession::buildQuestQueryResponse(QuestProperties const* qst)
42{
43 // 2048 bytes should be more than enough. The fields cost ~200 bytes.
44 // better to allocate more at startup than have to realloc the buffer later on.
45
46 auto data = std::make_unique<WorldPacket>(SMSG_QUEST_QUERY_RESPONSE, 248);
47 MySQLStructure::LocalesQuest const* lci = (language > 0) ? sMySQLStore.getLocalizedQuest(qst->id, language) : nullptr;
48 uint32_t i;
49
50 *data << uint32_t(qst->id); // Quest ID
51 *data << uint32_t(2); // Unknown, always seems to be 2
52
53#if VERSION_STRING > TBC
54 *data << int32_t(qst->questlevel); // Quest level
55#endif
56
57 *data << uint32_t(qst->min_level); // Quest required level
58
59
60 if (qst->quest_sort > 0)
61 *data << int32_t(-(int32_t)qst->quest_sort); // Negative if pointing to a sort.
62 else
63 *data << uint32_t(qst->zone_id); // Positive if pointing to a zone.
64
65 *data << uint32_t(qst->type); // Info ID / Type
66 *data << qst->suggestedplayers; // suggested players
67
68 *data << uint32_t(qst->required_rep_faction); // Faction ID
69 *data << uint32_t(qst->required_rep_value); // Faction Amount
70
71 *data << uint32_t(0); // Unknown (always 0)
72 *data << uint32_t(0); // Unknown (always 0)
73
74 *data << uint32_t(qst->next_quest_id); // Next Quest ID
75#if VERSION_STRING > TBC
76 *data << uint32_t(0); // Column id +1 from QuestXp.dbc, entry is quest level
77
78 *data << uint32_t(sQuestMgr.GenerateRewardMoney(_player, qst)); // Copper reward
79 *data << uint32_t(qst->reward_money < 0 ? -qst->reward_money : 0); // Required Money
80
81 *data << uint32_t(qst->reward_spell); // Spell added to spellbook upon completion
82 *data << uint32_t(qst->effect_on_player); // Spell casted on player upon completion
83
84 *data << uint32_t(qst->bonushonor); // 2.3.0 - bonus honor
85 *data << float(0); // 3.3.0 - some multiplier for honor
86 *data << uint32_t(qst->srcitem); // Item given at the start of a quest (srcitem)
87 *data << uint32_t(qst->quest_flags); // Quest Flags
88 *data << qst->rewardtitleid; // 2.4.0 unk
89 *data << uint32_t(0); // playerkillcount
90 *data << qst->rewardtalents;
91 *data << uint32_t(0); // 3.3.0 Unknown
92 *data << uint32_t(0); // 3.3.0 Unknown
93#else
94 *data << uint32_t(sQuestMgr.GenerateRewardMoney(_player, qst)); // Copper reward
95 *data << uint32_t(qst->reward_money < 0 ? -qst->reward_money : 0); // Required Money
96
97 *data << uint32_t(qst->effect_on_player); // Spell casted on player upon completion
98 *data << uint32_t(qst->reward_spell); // Spell added to spellbook upon completion
99
100 *data << uint32_t(qst->bonushonor); // 2.3.0 - bonus honor
101
102 *data << uint32_t(qst->srcitem); // Item given at the start of a quest (srcitem)
103 *data << uint32_t(qst->quest_flags); // Quest Flags
104 *data << qst->rewardtitleid; // 2.4.0 unk
105#endif
106
107 // (loop 4 times)
108 for (i = 0; i < 4; ++i)
109 {
110 *data << qst->reward_item[i]; // Forced Reward Item [i]
111 *data << qst->reward_itemcount[i]; // Forced Reward Item Count [i]
112 }
113
114 // (loop 6 times)
115 for (i = 0; i < 6; ++i)
116 {
117 *data << qst->reward_choiceitem[i]; // Choice Reward Item [i]
118 *data << qst->reward_choiceitemcount[i]; // Choice Reward Item Count [i]
119 }
120
121#if VERSION_STRING > TBC
122 // (loop 5 times) - these 3 loops are here to allow displaying rep rewards in client (not handled in core yet)
123 for (i = 0; i < 5; ++i)
124 *data << uint32_t(qst->reward_repfaction[i]); // reward factions ids
125
126 for (i = 0; i < 5; ++i)
127 *data << uint32_t(0); // column index in QuestFactionReward.dbc but use unknown
128
129 for (i = 0; i < 5; ++i) // Unknown
130 *data << uint32_t(0);
131#endif
132
133 *data << qst->point_mapid;
134 *data << qst->point_x;
135 *data << qst->point_y;
136 *data << qst->point_opt;
137
138 if (lci != nullptr)
139 {
140 *data << lci->title;
141 *data << lci->objectives;
142 *data << lci->details;
143 *data << lci->endText;
144#if VERSION_STRING > TBC
145 *data << uint8_t(0);
146#endif
147 }
148 else
149 {
150 *data << qst->title; // Title / name of quest
151 *data << qst->objectives; // Objectives / description
152 *data << qst->details; // Details
153 *data << qst->endtext; // Subdescription
154#if VERSION_STRING > TBC
155 *data << uint8_t(0); // most 3.3.0 quests i seen have something like "Return to NPCNAME"
156#endif
157 }
158
159 for (i = 0; i < 4; ++i)
160 {
161 *data << qst->required_mob_or_go[i]; // Kill mob entry ID [i]
162 *data << qst->required_mob_or_go_count[i]; // Kill mob count [i]
163#if VERSION_STRING > TBC
164 *data << uint32_t(0); // Unknown
165 *data << uint32_t(0); // 3.3.0 Unknown
166#else
167 *data << qst->required_item[i]; // Collect item [i]
168 *data << qst->required_itemcount[i]; // Collect item count [i]
169#endif
170 }
171
172#if VERSION_STRING > TBC
173 for (i = 0; i < MAX_REQUIRED_QUEST_ITEM; ++i)
174 {
175 *data << qst->required_item[i]; // Collect item [i]
176 *data << qst->required_itemcount[i]; // Collect item count [i]
177 }
178#endif
179
180 if (lci != nullptr)
181 {
182 *data << lci->objectiveText[0];
183 *data << lci->objectiveText[1];
184 *data << lci->objectiveText[2];
185 *data << lci->objectiveText[3];
186 }
187 else
188 {
189 *data << qst->objectivetexts[0]; // Objective 1 - Used as text if mob not set
190 *data << qst->objectivetexts[1]; // Objective 2 - Used as text if mob not set
191 *data << qst->objectivetexts[2]; // Objective 3 - Used as text if mob not set
192 *data << qst->objectivetexts[3]; // Objective 4 - Used as text if mob not set
193 }
194
195 return data;
196}

334void WorldSession::handleQuestPushResultOpcode(WorldPacket& recvPacket)
335{
336 MsgQuestPushResult srlPacket;
337 if (!srlPacket.deserialise(recvPacket))
338 return;
339
340 sLogger.debugFlag(AscEmu::Logging::LF_OPCODE, "Received CMSG_QUEST_PUSH_RESULT");
341
342 if (_player->getQuestSharerByDbId())
343 {
344 const auto questSharerPlayer = sObjectMgr.getPlayer(_player->getQuestSharerByDbId());
345 if (questSharerPlayer)
346 {
347 const uint64_t guid = recvPacket.size() >= 13 ? _player->getGuid() : srlPacket.giverGuid;
348 questSharerPlayer->getSession()->SendPacket(MsgQuestPushResult(guid, 0, srlPacket.pushResult).serialise().get());
349 _player->setQuestSharerDbId(0);
350 }
351 }
352}

364{
365 CmsgQuestQuery srlPacket;
366 if (!srlPacket.deserialise(recvPacket))
367 return;
368
369 if (const auto questProperties = sMySQLStore.getQuestProperties(srlPacket.questId))

402 SendPacket(&data);

437void WorldSession::handleQuestgiverStatusQueryOpcode(WorldPacket& recvPacket)
438{
439 CmsgQuestgiverStatusQuery srlPacket;
440 if (!srlPacket.deserialise(recvPacket))
441 return;
442
443 if (_player->IsInBg())
444 return;
445
446 Object* qst_giver = nullptr;
447
448 if (srlPacket.questGiverGuid.isUnit())
449 {
450 Creature* quest_giver = _player->getWorldMap()->getCreature(srlPacket.questGiverGuid.getGuidLowPart());
451 if (quest_giver)
452 qst_giver = quest_giver;
453 else
454 return;
455
456 if (!quest_giver->isQuestGiver())
457 {
458 sLogger.debug("Creature is not a questgiver.");
459 return;
460 }
461 }
462 else if (srlPacket.questGiverGuid.isItem())
463 {
464 Item* quest_giver = _player->getItemInterface()->GetItemByGUID(srlPacket.questGiverGuid.getRawGuid());
465 if (quest_giver)
466 qst_giver = quest_giver;
467 else
468 return;
469 }
470 else if (srlPacket.questGiverGuid.isGameObject())
471 {
472 GameObject* quest_giver = _player->getWorldMap()->getGameObject(srlPacket.questGiverGuid.getGuidLowPart());
473 if (quest_giver)
474 qst_giver = quest_giver;
475 else
476 return;
477 }
478
479 if (!qst_giver)
480 {
481 sLogger.debug("Invalid questgiver GUID {}.", std::to_string(srlPacket.questGiverGuid.getRawGuid()));
482 return;
483 }
484
485#if VERSION_STRING < Cata
486 const auto questStatus = static_cast<uint8_t>(sQuestMgr.CalcStatus(qst_giver, _player));
487#else
488 const auto questStatus = sQuestMgr.CalcStatus(qst_giver, _player);
489#endif
490 SendPacket(SmsgQuestgiverStatus(srlPacket.questGiverGuid.getRawGuid(), questStatus).serialise().get());
491}

493void WorldSession::handleQuestGiverQueryQuestOpcode(WorldPacket& recvPacket)
494{
495 CmsgQuestgiverQueryQuest srlPacket;
496 if (!srlPacket.deserialise(recvPacket))
497 return;
498
499 Object* qst_giver = nullptr;
500
501 bool bValid = false;
502
503 QuestProperties const* qst = sMySQLStore.getQuestProperties(srlPacket.questId);
504 if (!qst)
505 {
506 sLogger.debug("Invalid quest with id {}", srlPacket.questId);
507 return;
508 }
509
510 uint32_t status = QuestStatus::NotAvailable;
511
512 if (srlPacket.guid.isUnit())
513 {
514 Creature* quest_giver = _player->getWorldMap()->getCreature(srlPacket.guid.getGuidLowPart());
515 if (quest_giver)
516 qst_giver = quest_giver;
517 else
518 return;
519 if (quest_giver->isQuestGiver())
520 {
521 bValid = true;
522 status = sQuestMgr.CalcQuestStatus(qst_giver, _player, qst, static_cast<uint8_t>(quest_giver->GetQuestRelation(qst->id)), false);
523 }
524 }
525 else if (srlPacket.guid.isGameObject())
526 {
527 GameObject* quest_giver = _player->getWorldMap()->getGameObject(srlPacket.guid.getGuidLowPart());
528 if (quest_giver)
529 qst_giver = quest_giver;
530 else
531 return;
532 bValid = false;
533 if (quest_giver->getGoType() == GAMEOBJECT_TYPE_QUESTGIVER)
534 {
535 bValid = true;
536 if (auto go_quest_giver = dynamic_cast<GameObject_QuestGiver*>(quest_giver))
537 status = sQuestMgr.CalcQuestStatus(qst_giver, _player, qst, static_cast<uint8_t>(go_quest_giver->GetQuestRelation(qst->id)), false);
538 }
539 }
540 else if (srlPacket.guid.isItem())
541 {
542 Item* quest_giver = _player->getItemInterface()->GetItemByGUID(srlPacket.guid.getRawGuid());
543 if (quest_giver)
544 qst_giver = quest_giver;
545 else
546 return;
547
548 const auto itemProto = quest_giver->getItemProperties();
549 if (itemProto->Bonding != ITEM_BIND_ON_USE || quest_giver->isSoulbound())
550 {
551 if (sScriptMgr.CallScriptedItem(quest_giver, _player))
552 return;
553 }
554
555 if (itemProto->Bonding == ITEM_BIND_ON_USE)
556 quest_giver->addFlags(ITEM_FLAG_SOULBOUND);
557
558 bValid = true;
559 status = sQuestMgr.CalcQuestStatus(qst_giver, _player, qst, 1, false);
560 }
561
562 if (!qst_giver)
563 {
564 sLogger.debug("Invalid questgiver GUID.");
565 return;
566 }
567
568 if (!bValid)
569 {
570 sLogger.debug("object is not a questgiver.");
571 return;
572 }
573
574 WorldPacket data;
575
576 if ((status == QuestStatus::Available) || (status == QuestStatus::Repeatable) || (status == QuestStatus::AvailableChat))
577 {
578 sQuestMgr.BuildQuestDetails(&data, qst, qst_giver, 1, language, _player);
579 SendPacket(&data);
580 sLogger.debug("Sent SMSG_QUESTGIVER_QUEST_DETAILS.");
581
582 if (qst->HasFlag(QUEST_FLAGS_AUTO_ACCEPT))
583 _player->acceptQuest(qst_giver->getGuid(), qst->id);
584 }
585 else if (status == QuestStatus::NotFinished || status == QuestStatus::Finished)
586 {
587 sQuestMgr.BuildRequestItems(&data, qst, qst_giver, status, language);
588 SendPacket(&data);
589 sLogger.debug("Sent SMSG_QUESTGIVER_REQUEST_ITEMS.");
590 }
591}

593void WorldSession::handleQuestlogRemoveQuestOpcode(WorldPacket& recvPacket)
594{
595 CmsgQuestlogRemoveQuest srlPacket;
596 if (!srlPacket.deserialise(recvPacket))
597 return;
598
599 if (srlPacket.questLogSlot >= 25)
600 return;
601
602 QuestLogEntry* qEntry = _player->getQuestLogBySlotId(srlPacket.questLogSlot);
603 if (!qEntry)
604 {
605 sLogger.debug(" No quest in slot {}.", srlPacket.questLogSlot);
606 return;
607 }
608 QuestProperties const* qPtr = qEntry->getQuestProperties();
609
610 if (const auto questScript = qEntry->getQuestScript())
611 questScript->OnQuestCancel(_player);
612
613 qEntry->finishAndRemove();
614
615 for (uint8_t i = 0; i < 4; ++i)
616 {
617 if (qPtr->receive_items[i])
618 _player->getItemInterface()->RemoveItemAmt(qPtr->receive_items[i], 1);
619 }
620
621 if (qPtr->srcitem && qPtr->srcitem != qPtr->receive_items[0])
622 {
623 ItemProperties const* itemProto = sMySQLStore.getItemProperties(qPtr->srcitem);

888void WorldSession::handlePushQuestToPartyOpcode(WorldPacket& recvPacket)
889{
890 CmsgPushquesttoparty srlPacket;
891 if (!srlPacket.deserialise(recvPacket))
892 return;
893
894 QuestProperties const* pQuest = sMySQLStore.getQuestProperties(srlPacket.questId);
895 if (pQuest)
896 {
897 if (auto group = _player->getGroup())
898 {
899 uint32_t pguid = _player->getGuidLow();
900 SubGroup* sgr = group ? group->GetSubGroup(_player->getSubGroupSlot()) : nullptr;
901
902 if (sgr)
903 {
904 group->Lock();

906 {
907 Player* pPlayer = sObjectMgr.getPlayer(cachedCharacterInfo->guid);
908 if (pPlayer && pPlayer->getGuid() != pguid)
909 {
910 _player->getSession()->SendPacket(MsgQuestPushResult(pPlayer->getGuid(), 0, QUEST_SHARE_MSG_SHARING_QUEST).serialise().get());
911
912 uint8_t response = QUEST_SHARE_MSG_SHARING_QUEST;
913 uint32_t status = sQuestMgr.PlayerMeetsReqs(pPlayer, pQuest, false);
914
915 if (pPlayer->hasQuestInQuestLog(srlPacket.questId))
916 {
917 response = QUEST_SHARE_MSG_HAVE_QUEST;
918 }
919 else if (pPlayer->hasQuestFinished(srlPacket.questId))
920 {
921 response = QUEST_SHARE_MSG_FINISH_QUEST;
922 }
923 else if (status != QuestStatus::Available && status != QuestStatus::AvailableChat)
924 {
925 response = QUEST_SHARE_MSG_CANT_TAKE_QUEST;
926 }
927 else if (pPlayer->getFreeQuestSlot() > MAX_QUEST_SLOT)
928 {
929 response = QUEST_SHARE_MSG_LOG_FULL;
930 }
931 else if (pPlayer->m_duelPlayer)
932 {
933 response = QUEST_SHARE_MSG_BUSY;
934 }
935
936 if (response == QUEST_SHARE_MSG_SHARING_QUEST && !pPlayer->isVisibleObject(_player->getGuid()))
937 {
938 response = QUEST_SHARE_MSG_BUSY;
939 }
940
941 if (response != QUEST_SHARE_MSG_SHARING_QUEST)
942 {
943 sQuestMgr.SendPushToPartyResponse(_player, pPlayer, response);
944 continue;
945 }
946
947 WorldPacket data;
948 sQuestMgr.BuildQuestDetails(&data, pQuest, _player, 1, pPlayer->getSession()->language, pPlayer);
949 pPlayer->setQuestSharerDbId(pguid);
950 pPlayer->getSession()->SendPacket(&data);
951 }
952 }
953 group->Unlock();
954 }
955 }
