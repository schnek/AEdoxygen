322void OnEmote(Player* pPlayer, uint32_t Emote, Unit* pUnit)
323{
324 pUnit = pPlayer->getWorldMap()->getUnit(pPlayer->getTargetGuid());
325 if(!pUnit || !pUnit->isAlive() || pUnit->getAIInterface()->getCurrentTarget())
326 return;
327
328 switch(Emote)
329 {
330 case EMOTE_ONESHOT_FLEX:
331 InnkeeperFlex(pPlayer, pUnit);
332 break;
333
334 case EMOTE_STATE_DANCE:
335 InnkeeperDance(pPlayer, pUnit);
336 break;
337
338 case EMOTE_ONESHOT_TRAIN:
339 InnkeeperTrain(pPlayer, pUnit);
340 break;
341
342 case EMOTE_ONESHOT_CHICKEN:
343 InnkeeperChicken(pPlayer, pUnit);
344 break;
345 }
346}
