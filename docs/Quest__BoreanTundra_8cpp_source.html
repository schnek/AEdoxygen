614 return true;
615

619 return true;

627 pCreature->castSpell(pCreature, 46800, true);
628 }
629
630 pPlayer->addQuestKill(11897, 2, 0);
631 }
632
633 if (pCreature->getEntry() == 26249)
634 {
635 if (questLog->getMobCountByIndex(1) == 0)

637 pCreature->castSpell(pCreature, 46798, true);
