199 void DoAction(int32_t /*action*/) override
200 {
201 wait_timer = 5000;
202 setScriptPhase(PHASE_TO_EQUIP);
203
204 getCreature()->setStandState(STANDSTATE_STAND);
205 getCreature()->removeAllAurasById(SPELL_SOUL_PRISON_CHAIN);
206
207 float z;
208 if (Creature* anchor = getCreature()->getWorldMapCreature(anchorGUID))
209 {
210 anchor->interruptSpell();
211 anchor->getNearPoint(getCreature(), anchorX, anchorY, z, 1.0f, anchor->getAbsoluteAngle(getCreature()));

218 void OnReachWP(uint32_t type, uint32_t iWaypointId) override
219 {
220 if (type != POINT_MOTION_TYPE)
221 return;
222
223 if (iWaypointId == 1)
224 {
225 wait_timer = 5000;
226
227 getCreature()->castSpell(getCreature(), SPELL_DK_INITIATE_VISUAL, true);
228
229 if (Player* starter = getCreature()->getWorldMapPlayer(playerGUID))
230 sendRandomDBChatMessage(emoteVectorAttack, starter);
231
232 setScriptPhase(PHASE_TO_ATTACK);
233 }

236 void AIUpdate(unsigned long time_passed) override
237 {
238 switch (getScriptPhase())
239 {
240 case PHASE_CHAINED:
241 {
242 if (!anchorGUID)
243 {
244 if (Creature* anchor = findNearestCreature(29521, 30.0f))
245 {
246 anchor->GetScript()->SetCreatureData64(0, getCreature()->getGuid());
247 anchor->castSpell(getCreature(), SPELL_SOUL_PRISON_CHAIN);
248 anchorGUID = anchor->getGuid();
249 }
250
251 GameObject* prison = nullptr;
252
253 for (uint8_t i = 0; i < 12; ++i)
254 {
255 if (GameObject* temp_prison = findNearestGameObject(acherus_soul_prison[i], 30))
256 {
257 if (getCreature()->isInDist(temp_prison, 99.0f))
258 prison = temp_prison;
259 }
260 }
261
262 if (prison)
263 {
264 prison->setState(GO_STATE_OPEN);
265 prison->removeFlags(GO_FLAG_NONSELECTABLE);
266 }
267 }
268 } break;
269 case PHASE_TO_EQUIP:
270 {
271 if (wait_timer)
272 {
273 if (wait_timer > time_passed)
274 {
275 wait_timer -= time_passed;
276 }
277 else
278 {
279 getCreature()->getMovementManager()->movePoint(1, anchorX, anchorY, getCreature()->GetPositionZ());
280 setScriptPhase(PHASE_EQUIPING);
281 wait_timer = 0;
282 }
283 }
284 } break;
285 case PHASE_TO_ATTACK:
286 {
287 if (wait_timer)
288 {
289 if (wait_timer > time_passed)
290 {
291 wait_timer -= time_passed;
292 }
293 else
294 {
295 getCreature()->setFaction(14);
296 getCreature()->getAIInterface()->setAllowedToEnterCombat(true);
297 getCreature()->getAIInterface()->setIgnorePlayerCombat(false);
298 getCreature()->getAIInterface()->setReactState(REACT_AGGRESSIVE);
299 setScriptPhase(PHASE_ATTACKING);
300
301 if (Player* target = getCreature()->getWorldMapPlayer(playerGUID))
302 {
303 getCreature()->getAIInterface()->setCurrentTarget(target);
304 getCreature()->getAIInterface()->onHostileAction(target);
305 }
306
307 wait_timer = 0;
308 }
309 }
310 } break;
311 }
312 }
313
314protected:
315 std::vector<uint32_t> emoteVectorStart;

317 uint64_t anchorGUID = 0;
318 uint64_t playerGUID = 0;
319 uint32_t wait_timer = 0;
320 float anchorX, anchorY = 0.0f;
