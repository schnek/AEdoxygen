32 void OnReachWP(uint32_t type, uint32_t iWaypointId) override
33 {
34 if (type != WAYPOINT_MOTION_TYPE)
35 return;
36
37 if (iWaypointId == 13)
38 {
39 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Thanks, these orcs wanted me to weld in the boiler");
40 getCreature()->Despawn(5000, 1000);
41 getCreature()->stopMoving();
42 if (getCreature()->m_escorter == nullptr)
43 return;
44
45 Player* player = getCreature()->m_escorter;
46 getCreature()->m_escorter = nullptr;
47
48 if (auto* questLog = player->getQuestLogByQuestId(1440))
49 questLog->sendQuestComplete();
50 }
51 }
52};
