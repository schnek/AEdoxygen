38class WoodlandWalkerGossip : public GossipScript
39{
40public:
41 void onHello(Object* pObject, Player* plr) override
42 {
43 Creature* pCreature = (pObject->isCreature()) ? (static_cast<Creature*>(pObject)) : nullptr;
44 if (pCreature == nullptr)
45 return;
46
47 uint32_t chance = Util::getRandomUInt(1);
48 if (chance == 0)
49 {
50 pCreature->setFaction(14);
51 pCreature->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "The Woodlands Walker is angered by your request and attacks!");
52 }
53 else
54 {
55 plr->getItemInterface()->AddItemById(36786, 1, 0);
56 pCreature->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Breaking off a piece of its bark, the Woodlands Walker hands it to you before departing.");
57 }
58 }
