41 void InitOrReset() override
42 {
43 getCreature()->setEmoteState(EMOTE_ONESHOT_NONE);
44 // Hackfix for Updatemgr
45 addAIFunction([this](CreatureAIFunc pThis) { startSleeping(); }, DoOnceScheduler(1s));
46 }
47
48 void OnReachWP(uint32_t /*type*/, uint32_t id) override
49 {
50 if (id == ToWoodpile)
51 {
52 // Woodpile Reached
53 getCreature()->setEmoteState(EMOTE_STATE_WORK_CHOPWOOD);
54 addAIFunction([this](CreatureAIFunc pThis) {handleMoveRandom(pThis); }, DoOnceScheduler(3min));
55 }
56 else if (id == Random)
57 {
58 // Random Position Reached go Back to Sleep
59 startSleeping();
60 }
61 }
62
63 void OnHitBySpell(uint32_t spellId, Unit* caster) override
64 {
65 if (!caster || !caster->isPlayer())
66 return;
67
68 if (spellId != SPELL_AWAKEN_PEON)
69 return;
70
71 removeAllFunctionsFromScheduler();
72 handleWakeUp(caster);
73 }
74
75 void startSleeping()
76 {
77 castSpellOnSelf(SPELL_BUFF_SLEEP);
78 // Start working after aura expires by itself
79 addAIFunction([this](CreatureAIFunc PThis) { handleWakeUp(nullptr); }, DoOnceScheduler(2min));
80 }
81
82 void handleMoveRandom(CreatureAIFunc pThis)
83 {
84 getCreature()->setEmoteState(EMOTE_ONESHOT_NONE);
85
86 // Move to a random point near closest wood pile
87 if (GameObject* Lumberpile = findNearestGameObject(GO_LUMBERPILE, SEARCH_RADIUS))
88 {
89 LocationVector randomPosition = Lumberpile->GetPosition();
90
91 float distance = Util::getRandomFloat(RANDOM_MIN_DIST, RANDOM_MAX_DIST);
92 float angle = Util::getRandomFloat(0.f, std::numbers::pi_v<float>);
93
94 getCreature()->movePositionToFirstCollision(randomPosition, distance, angle);
95
96 movePoint(Random, randomPosition);
97 }
98 }
99
100 void handleWakeUp(Unit* caster)
101 {
102 // Remove Zzz aura
103 _removeAura(SPELL_BUFF_SLEEP);
104
105 // Caster is nullptr when peon wakes by itself
106 if (caster != nullptr)
107 {
108 // Send chat Message
109 sendDBChatMessageByIndex(0, caster);
110
111 // Hackfix, clear AffectedUnits for Player. Quest Increment happens trough CastedSpell in QuestMgr.
112 if (auto* questLog = caster->ToPlayer()->getQuestLogByQuestId(QUEST_LAZY_PEONS))
113 questLog->clearAffectedUnits();
114 }
115
116 // Move to closest wood pile
117 if (GameObject* Lumberpile = findNearestGameObject(GO_LUMBERPILE, SEARCH_RADIUS))
118 {
119 LocationVector dest
120 {
121 Lumberpile->GetPositionX() + CHOP_OFFSET_X,
122 Lumberpile->GetPositionY(),
123 Lumberpile->GetPositionZ()
124 };
125 const float orientation = faceFromTo(getCreature(), dest.x, dest.y, Lumberpile->GetPositionX(), Lumberpile->GetPositionY());
126
127 movePoint(ToWoodpile, dest, true, orientation);
128 }
129 }

131 static constexpr float toRad(float deg) noexcept
132 {
133 return deg * std::numbers::pi_v<float> / 180.0f;
134 }
135
136 static float faceFromTo(Creature* c, float fromX, float fromY, float toX, float toY)
137 {
138 if (!c) return 0.0f;
139 const float deg = c->calcAngle(fromX, fromY, toX, toY);
140 return deg * std::numbers::pi_v<float> / 180.0f;
141 }
142};
