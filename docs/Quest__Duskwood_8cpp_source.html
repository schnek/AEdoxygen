34class ElizaAI : public CreatureAIScript
35{
36public:

38 explicit ElizaAI(Creature* pCreature) : CreatureAIScript(pCreature)
39 {
40 mElizaCombatTimer = 0;
41 setCanEnterCombat(false);
42
43 addAISpell(ELIZA_FROST_NOVA, 10.0f, TARGET_ATTACKING, 0, 1, false, true);
44 addAISpell(ELIZA_FROSTBOLT, 20.0f, TARGET_ATTACKING, 3, 1);
45
46 mSummonGuard = addAISpell(ELIZA_SUMMON_GUARD, 0.0f, TARGET_SELF);
47
48 mElizaGuard = nullptr;
49 }
50
51 void OnCombatStart(Unit*) override
52 {
53 sendChatMessage(CHAT_MSG_MONSTER_SAY, 0, "Wait...you are not my husband. But he must have sent you. And you...look..delicious!");
54 mElizaCombatTimer = _addTimer(4000);
55 }
56
57 void AIUpdate() override
58 {
59 if (_isTimerFinished(mElizaCombatTimer))
60 {
61 setCanEnterCombat(true);
62 _removeTimer(mElizaCombatTimer);
63 }
64 if (_getHealthPercent() >= 10 && _getHealthPercent() <= 98 && !_isCasting())
65 {
66 mElizaGuard = getCreature()->getWorldMap()->getInterface()->getCreatureNearestCoords(getCreature()->GetPositionX(), getCreature()->GetPositionY(), getCreature()->GetPositionZ(), 1871);
67 if (mElizaGuard == nullptr)
68 {
69 _castAISpell(mSummonGuard);
70 }
71 }
72 }
73
74 uint32_t mElizaCombatTimer;
75 CreatureAISpells* mSummonGuard;
76 Creature* mElizaGuard;
77};
