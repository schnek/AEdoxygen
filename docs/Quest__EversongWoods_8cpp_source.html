55void ProspectorAnvilwardGossip::onSelectOption(Object* pObject, Player* Plr, uint32_t Id, const char* /*Code*/, uint32_t /*gossipId*/)
56{
57 switch (Id)
58 {
59 case 1:
60 {
61 GossipMenu menu(pObject->getGuid(), 8240, Plr->getSession()->language);
62 menu.addItem(GOSSIP_ICON_CHAT, 461, 2); // Why... yes, of course. I've something to show you right inside this building. Mr. Anvilward.
63 menu.sendGossipPacket(Plr);
64 }break;
65 case 2:
66 {
67 Creature* pCreature = static_cast<Creature*>(pObject);
68
69 pCreature->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Very well. Let's see what you have to show me.");
70 GossipMenu::senGossipComplete(Plr);
71
72 pCreature->getMovementManager()->movePath(pCreature->getWaypointPath(), false);
73 pCreature->pauseMovement(10);
74 }break;
75 }
76};
