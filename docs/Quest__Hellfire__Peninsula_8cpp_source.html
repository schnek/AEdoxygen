163{
164public:
165 void onHello(Object* pObject, Player* pPlayer) override
166 {
167 int8_t i = -1;
168 Creature* pPrisoner = static_cast<Creature*>(pObject);
169 switch (pPrisoner->getEntry())
170 {
171 case 20677:
172 i = 0;
173 break;
174 case 20678:
175 i = 1;
176 break;
177 case 20679:
178 i = 2;
179 break;
180 }
181
182 if (i == -1)
183 return;
184
185 if (auto* questLog = pPlayer->getQuestLogByQuestId(10368))
186 {
187 if (questLog->getMobCountByIndex(i) < questLog->getQuestProperties()->required_mob_or_go_count[i])
188 {
189 if (pPlayer->getItemInterface()->GetItemCount(29501) > 0)
190 {
191 GossipMenu menu(pObject->getGuid(), 10104, pPlayer->getSession()->language);
192 menu.addItem(GOSSIP_ICON_CHAT, 463, 1); // Walk free, Elder. Bring the spirits back to your tribe.
193 menu.sendGossipPacket(pPlayer);
194 }
195 }
196 }
197 }
