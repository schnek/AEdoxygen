31class ScryingOrb : public GameObjectAIScript
32{
33public:
34 explicit ScryingOrb(GameObject* goinstance) : GameObjectAIScript(goinstance) {}
35 static GameObjectAIScript* Create(GameObject* GO) { return new ScryingOrb(GO); }
36
37 void OnActivate(Player* pPlayer) override
38 {
39 if (auto* questLog = pPlayer->getQuestLogByQuestId(11490))
40 {
41 GameObject* pOrb = pPlayer->getWorldMap()->getInterface()->getGameObjectNearestCoords(pPlayer->GetPositionX(), pPlayer->GetPositionY(), pPlayer->GetPositionZ(), 187578);
42 if (pOrb)
43 {
44 pOrb->setState(GO_STATE_OPEN);
45 questLog->setMobCountForIndex(0, 1);
46 questLog->sendUpdateAddKill(0);
47 questLog->updatePlayerFields();
48 }
49 }
50 else
51 {
52 pPlayer->broadcastMessage("Missing required quest : The Scryer's Scryer");
53 }
54 }
55};
