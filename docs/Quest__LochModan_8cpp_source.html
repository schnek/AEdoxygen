29class ProtectingtheShipment : public QuestScript
30{
31public:
32 void OnQuestStart(Player* mTarget, QuestLogEntry* /*qLogEntry*/) override
33 {
34 float SSX = mTarget->GetPositionX();
35 float SSY = mTarget->GetPositionY();
36 float SSZ = mTarget->GetPositionZ();
37
38
39 Creature* creat = mTarget->getWorldMap()->getInterface()->getCreatureNearestCoords(SSX, SSY, SSZ, 1379);
40 if (creat == nullptr)
41 return;
42 creat->m_escorter = mTarget;
43
44 auto path = creat->GetScript()->getCustomPath(1);
45 creat->getMovementManager()->movePath(*path, false);
46 creat->pauseMovement(3000);
47 creat->getAIInterface()->setAllowedToEnterCombat(false);
48 creat->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Okay let's do!");
49 creat->setNpcFlags(UNIT_NPC_FLAG_NONE);
50 }
51};

77 void OnReachWP(uint32_t type, uint32_t iWaypointId) override
