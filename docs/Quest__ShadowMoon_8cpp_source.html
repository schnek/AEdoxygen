102 mJovaanTimer = 0;

109 void AIUpdate() override
110 {
111 if (_isTimerFinished(mJovaanTimer))
112 {
113 switch (mJovaanPhase)
114 {
115 case 0:
116 {
117 CreatureAIScript* pRazuunAI = spawnCreatureAndGetAIScript(21502, -3300.47f, 2927.22f, 173.870f, 2.42924f); // Spawn Razuun
118 if (pRazuunAI != nullptr)
119 {

121 pRazuunAI->setCanEnterCombat(false);
122 pRazuunAI->setRooted(true);
123 }
124 getCreature()->setStandState(STANDSTATE_KNEEL);
125 getCreature()->emote(EMOTE_ONESHOT_TALK);
126 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Everything is in readiness, warbringer.");
127 mJovaanPhase = 1;
128 _resetTimer(mJovaanTimer, 6000);
129 }
130 break;
131 case 1:
132 {
133 getCreature()->emote(EMOTE_ONESHOT_TALK);
134 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Warbringer, that will require the use of all the hold's infernals. It may leave us vulnerable to a counterattack.");
135 mJovaanPhase = 2;
136 _resetTimer(mJovaanTimer, 11000);
137 }
138 break;
139 case 2:
140 {
141 getCreature()->setStandState(STANDSTATE_STAND);
142 mJovaanPhase = 3;
143 _resetTimer(mJovaanTimer, 1000);
144 }
145 break;
146 case 3:
147 {
148 getCreature()->emote(EMOTE_ONESHOT_SALUTE);
149 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "It shall be as you say, warbringer. One last question, if I may...");
150 mJovaanPhase = 4;
151 _resetTimer(mJovaanTimer, 10000);
152 }
153 break;
154 case 4:
155 {
156 getCreature()->emote(EMOTE_ONESHOT_QUESTION);
157 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "What's in the crate?");
158 mJovaanPhase = 5;
159 _resetTimer(mJovaanTimer, 10000);
160 }
161 break;
162 case 5:
163 {
164 getCreature()->emote(EMOTE_ONESHOT_SALUTE);
165 mJovaanPhase = -1;
166 _removeTimer(mJovaanTimer);
167 }
168 break;
169 }
170 }
171 }
