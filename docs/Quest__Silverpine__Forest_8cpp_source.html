35 void OnReachWP(uint32_t type, uint32_t iWaypointId) override
36 {
37 if (type != WAYPOINT_MOTION_TYPE)
38 return;
39
40 if (iWaypointId == 9)
41 {
42 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Thanks, you helped me to overcome this obstacle");
43 getCreature()->Despawn(5000, 1000);
44 getCreature()->stopMoving();
45 if (getCreature()->m_escorter == nullptr)
46 return;
47
48 Player* player = getCreature()->m_escorter;
49 getCreature()->m_escorter = nullptr;
50
51 if (auto* questLog = player->getQuestLogByQuestId(435))
52 questLog->sendQuestComplete();
53 }
54 }
