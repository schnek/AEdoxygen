30{

34
35 void OnLoad() override
36 {
37 getCreature()->setFaction(12);
38 getCreature()->setStandState(STANDSTATE_STAND);
39 }
40
41 void OnDamageTaken(Unit* mAttacker, uint32_t fAmount) override
42 {
43 if (getCreature()->getHealth() - fAmount <= getCreature()->getMaxHealth() * 0.2f)
44 {
45 if (mAttacker->isPlayer())
46 {
47 getCreature()->addUnitFlags(UNIT_FLAG_NOT_SELECTABLE);
48 RegisterAIUpdateEvent(1000);
49
50 if (auto* questLog = static_cast<Player*>(mAttacker)->getQuestLogByQuestId(1447))
51 questLog->sendQuestComplete();
52 }
53 }
54 }
55
56 void AIUpdate() override
57 {
58 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Okay, okay! Enough fighting. No one else needs to get hurt.");
59 getCreature()->removeAllNegativeAuras();
60 getCreature()->setFaction(12);
61 getCreature()->setHealthPct(100);
62 getCreature()->getThreatManager().clearAllThreat();
63 getCreature()->getThreatManager().removeMeFromThreatLists();
64 getCreature()->getAIInterface()->handleEvent(EVENT_LEAVECOMBAT, getCreature(), 0);
65 _setMeleeDisabled(true);
66 getCreature()->getAIInterface()->setAllowedToEnterCombat(false);
67 getCreature()->removeUnitFlags(UNIT_FLAG_NOT_SELECTABLE);
68 RemoveAIUpdateEvent();
69 }
70};

81 Creature* Dashel = mTarget->getWorldMap()->getInterface()->getCreatureNearestCoords(SSX, SSY, SSZ, 4961);
82
83 if (Dashel == nullptr)
84 return;
85
86 Dashel->setFaction(72);
87 Dashel->getAIInterface()->setMeleeDisabled(false);
88 Dashel->getAIInterface()->setAllowedToEnterCombat(true);
89
90 uint32_t chance = Util::getRandomUInt(100);
91 if (chance < 15)
92 {
93 std::string say = "Now you're gonna get it good, ";
94 say += mTarget->getName();
95 say += "!";
96 Dashel->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, say.c_str());
97 }
98 Creature* ct1 = mTarget->getWorldMap()->createAndSpawnCreature(4969, LocationVector(-8686.803711f, 445.267792f, 99.789223f, 5.461184f));
99 if (ct1 != nullptr)
100 ct1->Despawn(300000, 0);
101
102 Creature* ct2 = mTarget->getWorldMap()->createAndSpawnCreature(4969, LocationVector(-8675.571289f, 444.162262f, 99.644737f, 3.834103f));
103 if (ct2 != nullptr)
104 ct2->Despawn(300000, 0);
105 }
