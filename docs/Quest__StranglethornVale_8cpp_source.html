161class BloodscalpClanHeads : public QuestScript
162{
163public:
164 void OnQuestComplete(Player* mTarget, QuestLogEntry* /*qLogEntry*/) override
165 {
166 GameObject* skull1 = mTarget->getWorldMap()->getInterface()->getGameObjectNearestCoords(mTarget->GetPositionX(), mTarget->GetPositionY(), mTarget->GetPositionZ(), 2551);
167 if (skull1 == nullptr)
168 return;
169
170 Creature* Kin_weelay = mTarget->getWorldMap()->getInterface()->getCreatureNearestCoords(mTarget->GetPositionX(), mTarget->GetPositionY(), mTarget->GetPositionZ(), 2519);
171 if (Kin_weelay == nullptr)
172 return;
173
174 std::string msg1 = "Ah. Good ";
175 msg1 += mTarget->getName();
176 msg1 += ". Now let us see what tale these heads tell...";
177 Kin_weelay->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, msg1.c_str());
178 Kin_weelay->castSpell(Kin_weelay, sSpellMgr.getSpellInfo(3644), false);
179 skull1->despawn(5000, 0);
180 GameObject* skull2 = mTarget->getWorldMap()->getInterface()->getGameObjectNearestCoords(mTarget->GetPositionX(), mTarget->GetPositionY(), mTarget->GetPositionZ(), 2551);
181 if (skull2)
182 skull2->despawn(5000, 0);
183
184 std::string msg = "There, ";
185 msg += mTarget->getName();
186 msg += ". You may now speak to the Bloodscalp chief and his witchdoctor.";
187 Kin_weelay->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, msg.c_str(), 500);
188 }
