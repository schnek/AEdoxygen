51class ScreecherSpirit : public CreatureAIScript
52{

55 explicit ScreecherSpirit(Creature* pCreature) : CreatureAIScript(pCreature) {}
56
57 void OnLoad() override
58 {
59 if (!getCreature())
60 return;
61
62 Creature* cialo = getCreature()->getWorldMap()->getInterface()->getCreatureNearestCoords(getCreature()->GetPositionX(), getCreature()->GetPositionY(), getCreature()->GetPositionZ(), 5307);
63 if (!cialo)
64 return;
65
66 if (!cialo->isAlive())
67 cialo->Despawn(1, 6 * 60 * 1000);
68
69 getCreature()->Despawn(60 * 1000, 0);
70 }
