76

86 Creature * NewCreature = getCreature()->getWorldMap()->getInterface()->spawnCreature(22419, LocationVector(SSX + Util::getRandomFloat(3.0f), SSY + Util::getRandomFloat(3.0f), SSZ, SSO + Util::getRandomFloat(1.0f)), true, false, 0, 0);

115 if (!mKiller->isPlayer())

117
118 Player* plr = static_cast<Player*>(mKiller);
119 uint8_t chance = (uint8_t)Util::getRandomUInt(5);
120 uint32_t spawn = 0;
121
122 switch (chance)
123 {
124 case 0:
125 case 1:
126 spawn = 22459; //Freed Shat'ar Warrior
127 break;
128
129 case 2:
130 spawn = 21661; //Cabal Skirmisher
131 break;
132
133 case 3:
134 spawn = 16805; //Broken Skeleton
135 break;
136
137 case 4:
138 spawn = 18470; //Bonelasher
139 break;
140
141 case 5:
142 spawn = 22045; //Vengeful Husk
143 break;
144 }
145
146 Creature* creat = plr->getWorldMap()->createAndSpawnCreature(spawn, getCreature()->GetPosition());
147 if (creat == nullptr)
148 return;
149
150 creat->Despawn(1 * 60 * 1000, 0);
151
152 if (spawn != 22459)
153 return;
154
155 creat->setControlled(true, UNIT_STATE_ROOTED);
156 creat->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Finally! I'm free!");
157
158 plr->addQuestKill(10873, 0, 0);
159 }
