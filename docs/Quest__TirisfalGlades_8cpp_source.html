65 void DamageTaken(Unit* /*_attacker*/, uint32_t* damage) override
66 {
67 if (getCreature()->getHealthPct() < 15 || *damage > getCreature()->getHealth())
68 {
69 // Dont let us die
70 *damage = 0;
71
72 // Stopp Sttacking Me
73 removeCombat();
74
75 // Evade
76 getCreature()->getAIInterface()->enterEvadeMode();
77
78 // Outro Event
79 scriptEvents.addEvent(EVENT_EMOTE_RUDE, 3000);
80 }

82
83 void AIUpdate(unsigned long time_passed) override
84 {
85 scriptEvents.updateEvents(time_passed, getScriptPhase());
86
87 if (uint32_t eventId = scriptEvents.getFinishedEvent())
88 {
89 switch (eventId)
90 {
91 case EVENT_EMOTE_RUDE:
92 {
93 getCreature()->emote(EMOTE_ONESHOT_RUDE);
94 scriptEvents.addEvent(EVENT_TALK, 2000);
95 } break;
96 case EVENT_TALK:
97 {
98 sendDBChatMessage(SAY_COMPLETE);
99 scriptEvents.addEvent(EVENT_DRINK, 5000);
100 } break;
101 case EVENT_DRINK:
102 {
103 // Finish the Quest
104 if (Player* plr = getCreature()->getWorldMapPlayer(_playerGuid))
105 {
106 if (auto* questLog = static_cast<Player*>(plr)->getQuestLogByQuestId(QUEST_590))
107 questLog->sendQuestComplete();
108 }
109 _playerGuid = 0;
110 getCreature()->castSpell(nullptr, SPELL_DRINK);
111 scriptEvents.addEvent(EVENT_SET_QUESTGIVER_FLAG, 12000);
112 } break;
113 case EVENT_SET_QUESTGIVER_FLAG:
114 {
115 getCreature()->addNpcFlags(UNIT_NPC_FLAG_QUESTGIVER);
116 scriptEvents.addEvent(EVENT_STAND, 3000);
117 } break;
118 case EVENT_STAND:
119 {
120 getCreature()->setStandState(STANDSTATE_STAND);
121 } break;
122 }
123 }
124 }

126 void onQuestAccept(Player* player, QuestProperties const* qst) override
127 {
128 if (qst->id == QUEST_590)
129 {
130 // setPlayer
131 _playerGuid = player->getGuid();
132
133 getCreature()->removeNpcFlags(UNIT_NPC_FLAG_QUESTGIVER);
134 getCreature()->setFaction(28);
135 getCreature()->getAIInterface()->setMeleeDisabled(false);
136 getCreature()->getAIInterface()->setAllowedToEnterCombat(true);
137 getCreature()->getAIInterface()->setIgnorePlayerCombat(false);
138 getCreature()->getAIInterface()->setIgnoreCreatureCombat(false);
139 }
140 }
