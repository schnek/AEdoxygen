60 void SpawnHighborneLamenter(Player* pThis, uint32_t entry, float posX, float posY, float posZ, float posO)
61 {
62 CreatureProperties const* p = sMySQLStore.getCreatureProperties(entry);
63 if (p == nullptr)
64 return;
65
66 Creature* creature = pThis->getWorldMap()->createCreature(entry);
67 creature->m_spawn = nullptr;
68 creature->Load(p, posX, posY, posZ);
69 creature->SetOrientation(posO);
70 creature->getAIInterface()->setCombatDisabled(true);
71 creature->getAIInterface()->setMeleeDisabled(true);
72 creature->getAIInterface()->setTargetingDisabled(true);
73 creature->PushToWorld(pThis->getWorldMap());
74 creature->Despawn(180000, 0);
75 creature->setFactionTemplate(35);
76 creature->setServersideFaction();
77 }
