32 { 269.29f, -1433.32f, 50.31f }, //1

38class TheSummoning : public QuestScript
39{
40public:
41 void OnQuestStart(Player* pPlayer, QuestLogEntry* /*qLogEntry*/) override
42 {
43 Creature* windwatcher = pPlayer->getWorldMap()->getInterface()->getCreatureNearestCoords(pPlayer->GetPositionX(), pPlayer->GetPositionY(), pPlayer->GetPositionZ(), 6176);
44 if (windwatcher == nullptr)
45 return;
46
47 // questgiver will walk to the place where Cyclonian is spawned only walk when we are at home
48 if (windwatcher->CalcDistance(250.839996f, -1470.579956f, 55.4491f) > 1) return;
49 {
50 windwatcher->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Follow me");
51
52 MovementMgr::PointsArray path;
53 path.reserve(pathSize);
54 std::transform(std::begin(WaypointTheSummoning), std::end(WaypointTheSummoning), std::back_inserter(path), [](LocationVector const& pos)
55 {
56 return G3D::Vector3(pos.x, pos.y, pos.z);
57 });
58 MovementMgr::MoveSplineInit init(windwatcher);
59 init.SetWalk(true);
60 init.MovebyPath(path);
61 windwatcher->getMovementManager()->launchMoveSpline(std::move(init), 0, MOTION_PRIORITY_NORMAL, POINT_MOTION_TYPE);
62
63 }
64 windwatcher->Despawn(15 * 60 * 1000, 0);
65
66 // spawn cyclonian if not spawned already
67 Creature* cyclonian = pPlayer->getWorldMap()->getInterface()->getCreatureNearestCoords(323.947f, -1483.68f, 43.1363f, 6239);
68 if (cyclonian == nullptr)
69 {
70 cyclonian = pPlayer->getWorldMap()->createAndSpawnCreature(6239, LocationVector(323.947f, -1483.68f, 43.1363f, 0.682991f));
71
72 // if spawning cyclonian failed, we have to return.
73 if (cyclonian == nullptr)
74 return;
75 }
76
77 // queue cyclonian for despawn
78 cyclonian->Despawn(15 * 60 * 1000, 0);
79 }
80};

205 void SetCreatureData64(uint32_t Type, uint64_t Data) override
206 {
207 switch (Type)
208 {
209 case 1:
210 if (!eventInProgress)

221 void AIUpdate() override
222 {
223 if (eventInProgress)
224 {
225 Player* warrior = nullptr;
226
227 if (pPlayer)
228 warrior = pPlayer;
229
230 if (!warrior || !warrior->hasQuestInQuestLog(1719))
231 {
232 despawnAndUnsummon();
233 reset();
234 return;
235 }
236
237 if (!warrior->isAlive() && warrior->getQuestLogByQuestId(1719)->getQuestState() == QUEST_INCOMPLETE)
238 {
239 sendDBChatMessage(SAY_TWIGGY_FLATHEAD_DOWN);
240 warrior->getQuestLogByQuestId(1719)->sendQuestFailed(true);
241 despawnAndUnsummon();
242 reset();
243 }
244
245 if (!eventGrate && eventInProgress)
246 {
247 float x, y, z;
248 warrior->getPosition(x, y, z);
249
250 if (x >= -1684 && x <= -1674 && y >= -4334 && y <= -4324)
251 {
252 warrior->areaExploredQuestEvent(1719);
253 sendDBChatMessage(SAY_TWIGGY_FLATHEAD_BEGIN, warrior);
254
255 for (uint8_t i = 0; i < 6; ++i)
256 {
257 Creature* creature = spawnCreature(NPC_AFFRAY_CHALLENGER, AffrayChallengerLoc[i]);
258 if (!creature)
259 continue;
260
261 creature->setFaction(35);
262 creature->addUnitFlags(UNIT_FLAG_NOT_SELECTABLE);
263 creature->addUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
264 creature->emote(EMOTE_ONESHOT_ROAR);
265 creature->Despawn(600000, 0);
266 affrayChallenger[i] = creature;
267 }
268 WaveTimer = 5000;
269 ChallengerChecker = 1000;
270 eventGrate = true;
271 }
272 }
273 else if (eventInProgress)
274 {
275 if (ChallengerChecker <= GetAIUpdateFreq())
276 {
277 for (uint8_t i = 0; i < 6; ++i)
278 {
279 if (affrayChallenger[i])
280 {
281 Creature* creature = affrayChallenger[i];
282 if ((!creature || (!creature->isAlive())) && !challengerDown[i])
283 {
284 sendDBChatMessage(3378);
285 challengerDown[i] = true;
286 }
287 }
288 }
289 ChallengerChecker = 1000;
290 }
291 else ChallengerChecker -= GetAIUpdateFreq();
292
293 if (WaveTimer <= GetAIUpdateFreq())
294 {
295 if (Wave < 6 && affrayChallenger[Wave] && !eventBigWill)
296 {
297 sendDBChatMessage(3377);
298 Creature* creature = affrayChallenger[Wave];
299 if (creature && (creature->isAlive()))
300 {
301 creature->removeUnitFlags(UNIT_FLAG_NOT_SELECTABLE);
302 creature->removeUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
303 creature->emote(EMOTE_ONESHOT_ROAR);
304 creature->setFaction(14);
305
306 creature->getAIInterface()->onHostileAction(warrior);
307 ++Wave;
308 WaveTimer = 20000;
309 }
310 }
311 else if (Wave >= 6 && !eventBigWill)
312 {
313 if (Creature* creature = spawnCreature(NPC_BIG_WILL, -1722, -4341, 6.12f, 6.26f))
314 {
315 bigWill = creature;
316 creature->getMovementManager()->movePoint(0, -1682, -4329, 2.79f);
317 creature->emote(EMOTE_STATE_READYUNARMED);
318 creature->Despawn(480000, 0);
319 eventBigWill = true;
320 WaveTimer = 1000;
321 }
322 }
323 else if (Wave >= 6 && eventBigWill && bigWill)
324 {
325 Creature* creature = bigWill;
326 if (!creature || !creature->isAlive())
327 {
328 sendDBChatMessage(SAY_TWIGGY_FLATHEAD_OVER);
329 reset();
330 }
331 else if(!creature->getAIInterface()->isEngaged()) // Makes BIG WILL attackable.
332 {
333 creature->removeUnitFlags(UNIT_FLAG_NOT_SELECTABLE);
334 creature->removeUnitFlags(UNIT_FLAG_NON_ATTACKABLE);
335 creature->SendScriptTextChatMessage(SAY_BIG_WILL_READY);
336 creature->emote(EMOTE_ONESHOT_ROAR);
337 creature->setFaction(14);
338 creature->getAIInterface()->onHostileAction(warrior);
339 }
340 }
341 }
342 else WaveTimer -= GetAIUpdateFreq();
343 }
344 }
345 }
346
347 void despawnAndUnsummon()
348 {
349 for (uint8_t i = 0; i < 6; ++i) // unsummon challengers
350 {
351 if (affrayChallenger[i])
352 {
353 if (affrayChallenger[i] && affrayChallenger[i]->isAlive())
354 affrayChallenger[i]->Despawn(0, 0);
355 }
356 }
357
358 if (bigWill) // unsummon bigWill
359 {
360 if (bigWill && bigWill->isAlive())
361 bigWill->Despawn(0, 0);
362 }
363 }

369 eventInProgress = true;
370 }
