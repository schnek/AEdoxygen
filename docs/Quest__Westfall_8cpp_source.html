32 void OnReachWP(uint32_t type, uint32_t iWaypointId) override
33 {
34 if (type != WAYPOINT_MOTION_TYPE)
35 return;
36
37 if (iWaypointId == 19)
38 {
39 getCreature()->sendChatMessage(CHAT_MSG_MONSTER_SAY, LANG_UNIVERSAL, "Tell your master that this is where Van Cleef is hiding. I'm outta here!");
40 getCreature()->Despawn(5000, 1000);
41 getCreature()->stopMoving();
42
43 if (getCreature()->m_escorter == nullptr)
44 return;
45
46 Player* player = getCreature()->m_escorter;
47 getCreature()->m_escorter = nullptr;
48
49 if (auto* questLog = player->getQuestLogByQuestId(155))
50 questLog->sendQuestComplete();
51 }
52 }
53};
