31class AncientMarks : public GossipScript
32{
33public:
34 void onHello(Object* pObject, Player* plr) override
35 {
36 uint32_t entry = pObject->getEntry();
37 const char* text = "";
38 uint32_t TextId = 0;
39
40 if (entry == 17900)
41 {
42 text = "Grant me your mark, wise ancient.";
43 TextId = 9176;
44 }
45 else if (entry == 17901)
46 {
47 text = "Grant me your mark, mighty ancient.";
48 TextId = 9177;
49 }
50
51 if (plr->hasQuestFinished(9785) || plr->hasQuestInQuestLog(9785))
52 {
53 GossipMenu menu(pObject->getGuid(), TextId, plr->getSession()->language);
54 menu.addItem(GOSSIP_ICON_CHAT, 0, 1, text);
55 menu.sendGossipPacket(plr);
56 }
57 }

60 {
61 Creature* casta = static_cast<Creature*>(pObject);
62 switch (pObject->getEntry())
63 {
64 case 17900:
65 {
66 plr->addQuestKill(9785, 0, 0);
67
68 if (plr->getFactionStandingRank(942) == 4)
69 casta->castSpell(plr, 31808, true);

71 casta->castSpell(plr, 31810, true);
72 else if (plr->getFactionStandingRank(942) == 6)
73 casta->castSpell(plr, 31811, true);
74 else if (plr->getFactionStandingRank(942) == 7)
75 casta->castSpell(plr, 31815, true);
76
77 } break;
78 case 17901:
79 {
80 plr->addQuestKill(9785, 1, 0);
81
82 if (plr->getFactionStandingRank(942) == 4)

84 else if (plr->getFactionStandingRank(942) == 5)
