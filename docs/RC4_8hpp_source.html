25 {
26 uint8_t* state = _key->state;

28
29 // Initialize the state with the identity permutation

31 {
32 state[i] = i;
33 }
34
35 // Key-scheduling algorithm (KSA)
36 for (int i = 0; i < keyLength; i++)
37 {
38 j = (j + state[i] + _data[i % _len]) % keyLength;
39 std::swap(state[i], state[j]);
40 }
41
42 _key->x = 0;
43 _key->y = 0;
44 }

46 static void process(KeyStruct* _key, const size_t _len, const uint8_t* _inData, uint8_t* _outData)
47 {
48 uint8_t* state = _key->state;
49 uint8_t x = _key->x;
50 uint8_t y = _key->y;
51
52 for (size_t i = 0; i < _len; i++)
53 {
54 x = (x + 1) % keyLength;
55 y = (y + state[x]) % keyLength;
56
57 std::swap(state[x], state[y]);
58
59 const uint8_t xor_index = (state[x] + state[y]) % keyLength;
60 _outData[i] = _inData[i] ^ state[xor_index];
61 }
62
63 _key->x = x;
64 _key->y = y;
65 }
