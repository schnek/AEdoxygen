184 void AIUpdate() override
185 {
186 DespawnTimer++;
187 if (DespawnTimer >= 27)
188 {
189 getCreature()->Despawn(0, 0);
190 return;
191 }
192 // After 4 sec of last direction change, doomfire has 33% chance to change direction
193 DirChange++;
194 if (DirChange == 4 && Util::getRandomUInt(3) == 1 || DirChange >= 5)
195 {
196 if (getCreature()->getAIInterface()->getUnitToFollow())
197 {
198 if (Util::getRandomUInt(3) == 1 || getCreature()->GetDistance2dSq(getCreature()->getAIInterface()->getUnitToFollow()) <= 2.0f)
199 {
200 getCreature()->getAIInterface()->setUnitToFollow(nullptr);
201 }
202 }
203
204 if (!getCreature()->getAIInterface()->getUnitToFollow())
205 {
206 if (Util::getRandomUInt(3) == 1)
207 {
208 Unit* NewTarget = NULL;
209 NewTarget = FindTarget();
210 if (NewTarget)
211 {
212 getCreature()->getAIInterface()->setUnitToFollow(NewTarget);
213 getCreature()->getMovementManager()->moveFollow(NewTarget, 2.0f, 0.0f);
214 }
215 }
216
217 if (!getCreature()->getAIInterface()->getUnitToFollow())
218 {
219 float movedist = 10.0f;
220 float x = 0.0f;
221 float y = 0.0f;
222
223 float xchange = Util::getRandomFloat(movedist);
224 float ychange = std::sqrt(movedist * movedist - xchange * xchange);
225
226 if (Util::getRandomUInt(2) == 1)
227 xchange *= -1;
228 if (Util::getRandomUInt(2) == 1)
229 ychange *= -1;
230
231 x = getCreature()->GetPositionX() + xchange;
232 y = getCreature()->GetPositionY() + ychange;
233
234 getCreature()->getAIInterface()->moveTo(x, y, getCreature()->GetPositionZ());
235 }
236 }
237
238 DirChange = 0;
239 }
240 }
241 // A bit rewritten FindTarget function
242 Unit* FindTarget()
243 {
244 Unit* target = NULL;
245 float distance = 15.0f;
246 float z_diff;
247
248 Unit* pUnit;
249 float dist;
250
251 for (const auto& itr : getCreature()->getInRangeOppositeFactionSet())
252 {
253 if (!itr || !itr->isCreatureOrPlayer())
254 continue;
255
256 pUnit = static_cast<Unit*>(itr);
257
258 if (pUnit->hasUnitFlags(UNIT_FLAG_FEIGN_DEATH))
259 continue;
260
261 z_diff = fabs(getCreature()->GetPositionZ() - pUnit->GetPositionZ());
262 if (z_diff > 2.5f)
263 continue;
264
265 if (!pUnit->isAlive() || getCreature() == pUnit)
266 continue;
267
268 dist = getCreature()->GetDistance2dSq(pUnit);
269
270 if (dist > distance * distance)
271 continue;
272
273 if (dist < 3.0f)
274 continue;
275
276 distance = dist;
277 target = pUnit;
278 }
279
280 return target;
281 }

284 uint32_t DespawnTimer;
