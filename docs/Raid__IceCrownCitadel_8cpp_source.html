549void IceCrownCitadelScript::UpdateEvent()
550{
551 scriptEvents.updateEvents(getUpdateFrequency(), 0);
552
553 while (uint32_t eventId = scriptEvents.getFinishedEvent())
554 {
555 switch (eventId)
556 {
557 case EVENT_SPAWN_GUNSHIPS:
558 {
559 if (getBossState(DATA_ICECROWN_GUNSHIP_BATTLE) == Performed)
560 return;
561
562 if (!isPrepared)
563 {
564 if (getInstance()->getTeamIdInInstance() == TEAM_ALLIANCE)
565 skybreaker = sTransportHandler.createTransport(GO_THE_SKYBREAKER_ALLIANCE_ICC, mInstance);
566
567 if (getInstance()->getTeamIdInInstance() == TEAM_HORDE)
568 orgrimmar = sTransportHandler.createTransport(GO_ORGRIM_S_HAMMER_HORDE_ICC, mInstance);
569
570 isPrepared = true;
571 }
572 break;
573 }
574 case EVENT_SPAWN_ZEPPELIN_ALLIANCE:
575 {
576 HordeZeppelinAlliance = sTransportHandler.createTransport(GO_MIGHTY_WIND, mInstance);
577 HordeZeppelinAlliance->EnableMovement(true, mInstance);
578 break;
579 }
580 case EVENT_WIPE_CHECK:
581 {
582 if (getInstance()->getTeamIdInInstance() == TEAM_ALLIANCE)
583 {
584 DoCheckFallingPlayer(GetCreatureByGuid(MuradinBronzebeardGbGUID));
585 if (DoWipeCheck(skybreaker))
586 scriptEvents.addEvent(EVENT_WIPE_CHECK, 3000);
587 else
588 DoAction(ACTION_FAIL);
589 }
590 else
591 {
592 DoCheckFallingPlayer(GetCreatureByGuid(DeathbringerSaurfangGbGUID));
593 if (DoWipeCheck(orgrimmar))
594 scriptEvents.addEvent(EVENT_WIPE_CHECK, 3000);
595 else
596 DoAction(ACTION_FAIL);
597 }
598 break;
599 }
600 case EVENT_START_FLY:
601 {
602 if (getInstance()->getTeamIdInInstance() == TEAM_ALLIANCE && skybreaker)
603 skybreaker->EnableMovement(true, mInstance);
604
605 if (getInstance()->getTeamIdInInstance() == TEAM_HORDE && orgrimmar)
606 orgrimmar->EnableMovement(true, mInstance);
607 break;
608 }
609 default:
610 break;
611 }
612 }

615void IceCrownCitadelScript::DoAction(int32_t const action)
616{
617 switch (action)
618 {
619 case ACTION_STARTINTRO:
620 {
621 introDone = true;
622 Creature* Tirion = getLocalCreatureData(NPC_INTRO_TIRION);
623 Creature* LichKing = getLocalCreatureData(NPC_INTRO_LICH_KING);
624 Creature* Bolvar = getLocalCreatureData(NPC_INTRO_BOLVAR);
625
626 if (!Tirion)
627 return;
628
629 Creature* Commander = findNearestCreature(Tirion, getInstance()->getTeamIdInInstance() ? NPC_SE_HIGH_OVERLORD_SAURFANG : NPC_SE_MURADIN_BRONZEBEARD, 30.0f);
630
631 if (Tirion && LichKing && Bolvar && Commander)
632 {
633 Tirion->SendTimedScriptTextChatMessage(EVENT_INTRO01, 1000);
634 Tirion->SendTimedScriptTextChatMessage(EVENT_INTRO02, 15000);
635 Tirion->SendTimedScriptTextChatMessage(EVENT_INTRO03, 31000);
636
637 LichKing->SendTimedScriptTextChatMessage(EVENT_INTRO04, 35000);
638 Tirion->SendTimedScriptTextChatMessage(EVENT_INTRO05, 50000);
639 LichKing->SendTimedScriptTextChatMessage(EVENT_INTRO06, 59000);
640 LichKing->SendTimedScriptTextChatMessage(EVENT_INTRO07, 74000);
641 LichKing->SendTimedScriptTextChatMessage(EVENT_INTRO08, 84000);
642
643 Bolvar->SendTimedScriptTextChatMessage(EVENT_INTRO09, 105000);
644 LichKing->SendTimedScriptTextChatMessage(EVENT_INTRO10, 111000);
645
646 if (getInstance()->getTeamIdInInstance() == TEAM_ALLIANCE)
647 {
648 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO20, 117000);
649 Tirion->SendTimedScriptTextChatMessage(EVENT_INTRO21, 129000);
650 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO22, 138000);
651 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO23, 143000);
652 }
653 else
654 {
655 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO30, 117000);
656 Tirion->SendTimedScriptTextChatMessage(EVENT_INTRO31, 127000);
657 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO32, 136000);
658 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO33, 147000);
659 Commander->SendTimedScriptTextChatMessage(EVENT_INTRO34, 157000);
660 }
661 }
662 break;
663 }
664 case ACTION_INTRO_START:
665 {
666 scriptEvents.addEvent(EVENT_START_FLY, 2500);
667 break;
668 }
669 case ACTION_BATTLE_EVENT:
670 {
671 scriptEvents.addEvent(EVENT_WIPE_CHECK, 5000);
672 setBossState(DATA_ICECROWN_GUNSHIP_BATTLE, InProgress);
673 break;
674 }
675 case ACTION_BATTLE_DONE:
676 {
677 if (getBossState(DATA_ICECROWN_GUNSHIP_BATTLE) == Performed)
678 {
679 if (getInstance()->getTeamIdInInstance() == TEAM_ALLIANCE && skybreaker)
680 {
681 skybreaker->EnableMovement(true, mInstance);
682
683 if (orgrimmar)
684 orgrimmar->EnableMovement(true, mInstance);
685 }
686
687 if (getInstance()->getTeamIdInInstance() == TEAM_HORDE && orgrimmar)
688 {
689 orgrimmar->EnableMovement(true, mInstance);
690
691 if (skybreaker)
692 skybreaker->EnableMovement(true, mInstance);
693 }
694 }
695 break;
696 }
697 case ACTION_FAIL:
698 {
699 // Handle Wipe Here
