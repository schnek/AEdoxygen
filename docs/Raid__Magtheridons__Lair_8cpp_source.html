71void MagtheridonsLairInstanceScript::setLocalData(uint32_t type, uint32_t data)
72{
73 switch (type)
74 {
75 case DATA_MANTICRON_CUBE:
76 for (auto cube : cubes)
77 {
78 if (cube)
79 {
80 if (data == ACTION_ENABLE)
81 cube->removeFlags(GO_FLAG_NOT_SELECTABLE);
82 else
83 cube->setFlags(GO_FLAG_NOT_SELECTABLE);
84 }
85 }
86 break;
87 case DATA_COLLAPSE:
88 if (hall)
89 {
90 if (data == ACTION_ENABLE)
91 hall->setState(GO_STATE_OPEN);
92 else
93 hall->setState(GO_STATE_CLOSED);
94 }
95 break;
96 case DATA_COLLAPSE_2:
97 for (auto column : columns)
98 {
99 if (column)
100 {
101 if (data == ACTION_ENABLE)
102 column->setState(GO_STATE_OPEN);
103 else
104 column->setState(GO_STATE_CLOSED);
105 }
106 }
107 break;
108 case DATA_DOOR:
109 if (door)
110 {
111 if (data == ACTION_ENABLE)
112 door->setState(GO_STATE_OPEN);
113 else
114 door->setState(GO_STATE_CLOSED);
115 }
116 break;
117 default:
118 break;
119 }
120}

142void ManticronCubeGO::OnActivate(Player* pPlayer)
143{
144 Instance = (MagtheridonsLairInstanceScript*)_gameobject->getWorldMap()->getScript();
145
146 // We check if player has aura that prevents anyone from using this GO
147 if (pPlayer->getAuraWithId(SPELL_MIND_EXHAUSTION) || pPlayer->getAuraWithId(SPELL_SHADOW_GRASP))
148 return;
149
150 if (!Instance && !Instance->magtheridon)
151 return;
152
153 if (Creature* trigger = _gameobject->getWorldMap()->getInterface()->findNearestCreature(pPlayer, NPC_HELFIRE_RAID_TRIGGER, 10.0f))
154 trigger->castSpell(Instance->magtheridon, SPELL_SHADOW_GRASP_VISUAL, true);
155
156 pPlayer->setTargetGuid(Instance->magtheridon->getGuid());
157 pPlayer->castSpell(pPlayer, SPELL_SHADOW_GRASP, true);
158}

268void HellfireChannelerAI::AIUpdate(unsigned long time_passed)
269{
270 if(!_isInCombat() && !getCreature()->hasAurasWithId(SPELL_SHADOW_GRASP_C))
271 _castAISpell(shadowGasp);
272
273 scriptEvents.updateEvents(time_passed, getScriptPhase());
274
275 if (_isCasting())
276 return;
277
278 while (uint32_t eventId = scriptEvents.getFinishedEvent())
279 {
280 switch (eventId)
281 {
282 case EVENT_SHADOWBOLT:
283 _castAISpell(shadowBolt);
284 scriptEvents.addEvent(EVENT_SHADOWBOLT, Util::getRandomUInt(15000, 20000));
285 break;
286 case EVENT_FEAR1:
287 _castAISpell(fear);
288 scriptEvents.addEvent(EVENT_FEAR1, Util::getRandomUInt(25000, 40000));
289 break;
290 case EVENT_ABYSSAL:
291 _castAISpell(summonAbyssal);
292 scriptEvents.addEvent(EVENT_ABYSSAL, 60000);
293 break;
294 default:
295 break;
296 }
297 }
