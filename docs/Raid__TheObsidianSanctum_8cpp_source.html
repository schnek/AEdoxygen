232 void CheckDrakes()
233 {
234 if (mInstance == NULL)
235 return;
236
237 m_iDrakeCount = 0;
238
239 for (uint8_t i = 0; i < (OS_DATA_END - 1); ++i)
240 {
241 m_cDrakes[i] = mInstance->getCreature(i);
242 if (m_cDrakes[i] != NULL && m_cDrakes[i]->isAlive())
243 {
244 m_bDrakes[i] = true;
245 m_iDrakeCount++;
246 mInstance->DoDrakeAura(i);
247 }
248 }
