118bool TrialOfTheCrusaderInstanceScript::setBossState(uint32_t type, EncounterStates state)
119{
120 if (!InstanceScript::setBossState(type, state))
121 return false;
122
123 switch (state)
124 {
125 case EncounterStates::InProgress:
126 DoAction(ACTION_ENCOUNTER_STARTED);
127 break;
128 case EncounterStates::Performed:
129 case EncounterStates::Failed:
130 DoAction(ACTION_ENCOUNTER_STOPPED);
131 break;
132 }
133
134 switch (type)
135 {
136 case DATA_JARAXXUS:
137 {
138 if (state == EncounterStates::Failed)
139 {
140 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
141 fordring->GetScript()->DoAction(ACTION_JARAXXUS_WIPE);
142
143 MistressOfPainCount = 0;
144 }
145 else if (state == EncounterStates::Performed)
146 {
147 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
148 fordring->GetScript()->DoAction(ACTION_JARAXXUS_DEFEATED);
149 }
150 } break;
151 case DATA_FACTION_CRUSADERS:
152 {
153 switch (state)
154 {
155 case EncounterStates::InProgress:
156 {
157 ResilienceWillFixItTimer = 0;
158 } break;
159 case EncounterStates::Failed:
160 {
161 CrusadersSpecialState = false;
162 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
163 fordring->GetScript()->DoAction(ACTION_FACTION_WIPE);
164 } break;
165 case EncounterStates::Performed:
166 {
167 updateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_DEFEAT_FACTION_CHAMPIONS);
168 if (ResilienceWillFixItTimer > 0)
169 updateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_CHAMPIONS_KILLED_IN_MINUTE);
170
171 //DoRespawnGameObject(GetGuidData(DATA_CRUSADERS_CHEST), 7_days);
172
173 if (GameObject* cache = getGameObjectFromData(DATA_CRUSADERS_CHEST))
174 cache->removeFlags(GO_FLAG_NOT_SELECTABLE);
175
176 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
177 fordring->GetScript()->DoAction(ACTION_CHAMPIONS_DEFEATED);
178 } break;
179 default:
180 break;
181 }
182 } break;
183 case DATA_TWIN_VALKIRIES:
184 {
185 // Cleanup chest
186 if (GameObject* cache = getGameObjectFromData(DATA_CRUSADERS_CHEST))
187 cache->Delete();
188
189 switch (state)
190 {
191 case EncounterStates::Failed:
192 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
193 fordring->GetScript()->DoAction(ACTION_VALKYR_WIPE);
194 break;
195 case EncounterStates::Performed:
196 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
197 fordring->GetScript()->DoAction(ACTION_VALKYR_DEFEATED);
198 break;
199 default:
200 break;
201 }
202 } break;
203 case DATA_ANUBARAK:
204 {
205 switch (state)
206 {
207 case EncounterStates::Performed:
208 {
209 uint32_t tributeChest = 0;
210 if (getInstance()->getDifficulty() == InstanceDifficulty::RAID_10MAN_HEROIC)
211 {
212 if (TrialCounter >= 50)
213 tributeChest = GO_TRIBUTE_CHEST_10H_99;
214 else
215 {
216 if (TrialCounter >= 45)
217 tributeChest = GO_TRIBUTE_CHEST_10H_50;
218 else
219 {
220 if (TrialCounter >= 25)
221 tributeChest = GO_TRIBUTE_CHEST_10H_45;
222 else
223 tributeChest = GO_TRIBUTE_CHEST_10H_25;
224 }
225 }
226 }
227 else if (getInstance()->getDifficulty() == InstanceDifficulty::RAID_25MAN_HEROIC)
228 {
229 if (TrialCounter >= 50)
230 tributeChest = GO_TRIBUTE_CHEST_25H_99;
231 else
232 {
233 if (TrialCounter >= 45)
234 tributeChest = GO_TRIBUTE_CHEST_25H_50;
235 else
236 {
237 if (TrialCounter >= 25)
238 tributeChest = GO_TRIBUTE_CHEST_25H_45;
239 else
240 tributeChest = GO_TRIBUTE_CHEST_25H_25;
241 }
242 }
243 }
244
245 if (tributeChest)
246 if (Creature* tirion = getCreatureFromData(DATA_FORDRING))
247 if (GameObject* chest = tirion->summonGameObject(tributeChest, LocationVector(805.62f, 134.87f, 142.16f, 3.27f), QuaternionData(), 7 * TimeVarsMs::Day))
248 chest->setRespawnTime(chest->getRespawnDelay());
249 } break;
250 default:
251 break;
252 }
253 } break;
254 default:
255 break;
256 }
257
258 // Trial Counting
259 if (state == EncounterStates::Failed)
260 {
261 if (getInstance()->isHeroic())
262 {
263 // decrease attempt counter at wipe
264 --TrialCounter;
265 getInstance()->getWorldStatesHandler().SetWorldStateForZone(0, 0, UPDATE_STATE_UI_COUNT, getLocalData(TYPE_COUNTER));
266
267 // if theres no more attemps allowed
268 if (!TrialCounter)
269 {
270 if (Creature* anubarak = getCreatureFromData(DATA_ANUBARAK))
271 anubarak->Despawn(0, 0);
272 }
273 }
274 }
275
276 return true;
277}

289void TrialOfTheCrusaderInstanceScript::setLocalData(uint32_t type, uint32_t data)
290{
291 switch (type)
292 {
293 case TYPE_NORTHREND_BEASTS:
294 {
295 NorthrendBeasts = data;
296 switch (data)
297 {
298 case GORMOK_IN_PROGRESS:
299 {
300 setBossState(DATA_NORTHREND_BEASTS, EncounterStates::InProgress);
301 } break;
302 case GORMOK_DONE:
303 {
304 if (Creature* tirion = getCreatureFromData(DATA_FORDRING))
305 tirion->GetScript()->DoAction(ACTION_START_JORMUNGARS);
306 } break;
307 case SNAKES_IN_PROGRESS:
308 {
309 NotOneButTwoJormungarsTimer = 0;
310 } break;
311 case SNAKES_SPECIAL:
312 {
313 NotOneButTwoJormungarsTimer = 10 * TimeVarsMs::Second;
314 } break;
315 case SNAKES_DONE:
316 {
317 if (NotOneButTwoJormungarsTimer > 0)
318 updateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, SPELL_WORMS_KILLED_IN_10_SECONDS);
319 if (Creature* tirion = getCreatureFromData(DATA_FORDRING))
320 tirion->GetScript()->DoAction(ACTION_START_ICEHOWL);
321 } break;
322 case ICEHOWL_DONE:
323 {
324 NorthrendBeasts = EncounterStates::Performed;
325 setBossState(DATA_NORTHREND_BEASTS, EncounterStates::Performed);
326 setLocalData(DATA_DESPAWN_SNOBOLDS, 0);
327
328 if (Creature* combatStalker = getCreatureFromData(DATA_BEASTS_COMBAT_STALKER))
329 combatStalker->Despawn(2000, 0);
330
331 if (Creature* fordring = getCreatureFromData(DATA_FORDRING))
332 fordring->GetScript()->DoAction(ACTION_NORTHREND_BEASTS_DEFEATED);
333 } break;
334 case EncounterStates::Failed:
335 {
336 setBossState(DATA_NORTHREND_BEASTS, EncounterStates::Failed);
337 if (Creature* tirion = getCreatureFromData(DATA_FORDRING))
338 tirion->GetScript()->DoAction(ACTION_NORTHREND_BEASTS_WIPE);
339 SnoboldCount = 0;
340 } break;
341 default:
342 break;
343 }
344 } break;
345 case TYPE_COUNTER:
346 {
347 TrialCounter = data;
348 } break;
349 case DATA_DESPAWN_SNOBOLDS:
350 {
351 for (uint32_t const guid : snoboldGUIDS)
352 if (Creature* snobold = GetCreatureByGuid(guid))
353 snobold->Despawn(2000, 0);
354
355 snoboldGUIDS.clear();
356 } break;
357 //Achievements
358 case DATA_FACTION_CRUSADERS: // Achivement Resilience will Fix
359 {
360 ResilienceWillFixItTimer = 60 * TimeVarsMs::Second;
361 CrusadersSpecialState = true;
362 } break;
363 default:
364 break;
365 }
