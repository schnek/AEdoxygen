225 void AIUpdate() override
226 {
227 // Every 25% Halazzi calls on the lynx
228 if (!mLynx && _getHealthPercent() <= (100 - SplitCount * 25))
229 Split();
230
231 // Lynx OR Halazzi is at 20% HP Merge them together again
232 if (mLynx && (mLynx->getHealthPct() <= 20 || _getHealthPercent() <= 20))
233 Merge();
234
235 // At <25% Phase 3 begins
236 if (_getHealthPercent() < 25 && isScriptPhase(1))
237 {
238 _resetTimer(mTotemTimer, 30000);
239 setScriptPhase(3);
240 }
241
242 if (isScriptPhase(2) || isScriptPhase(3))
243 {
244 if (_isTimerFinished(mTotemTimer))
245 {
246 CreatureAIScript* Totem = spawnCreatureAndGetAIScript(CN_TOTEM, (getCreature()->GetPositionX() + Util::getRandomFloat(3) - 3), (getCreature()->GetPositionY() + Util::getRandomFloat(3) - 3), getCreature()->GetPositionZ(), 0, getCreature()->getFactionTemplate());
247 if (Totem)
248 {
249 Totem->despawn(60000); // Despawn in 60 seconds
250 }
251
252 switch (getScriptPhase())
253 {
254 case 2:
255 _resetTimer(mTotemTimer, 60000);
256 break;
257 case 3:
258 _resetTimer(mTotemTimer, 30000);
259 break; // Spawn them faster then phase 2
260 }
261 }
262 }
