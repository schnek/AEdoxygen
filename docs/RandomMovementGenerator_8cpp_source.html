65void RandomMovementGenerator<Creature>::doInitialize(Creature* owner)
66{
67 removeFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING | MOVEMENTGENERATOR_FLAG_TRANSITORY | MOVEMENTGENERATOR_FLAG_DEACTIVATED | MOVEMENTGENERATOR_FLAG_TIMED_PAUSED);
68 addFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED);
69
70 if (!owner || !owner->isAlive())
71 return;
72
73 _reference = owner->GetPosition();
74 owner->stopMoving();
75
76 if (_maxWanderDistance == 0.f)
77 _maxWanderDistance = owner->getMaxWanderDistance();
78
79 // Retail seems to let a creature walk 2 up to 10 splines before triggering a pause
80 _wanderSteps = static_cast<uint8_t>(Util::getRandomUInt(2, 10));
81
82 _timer->resetInterval(0);
83 _path = nullptr;

100template<>
101void RandomMovementGenerator<Creature>::setRandomLocation(Creature* owner)
102{
103 if (!owner)
104 return;
105
106 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE | UNIT_STATE_LOST_CONTROL) || owner->isCastingSpell())
107 {
108 addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
109 owner->stopMoving();
110 _path = nullptr;
111 return;
112 }
113
114 LocationVector position(_reference);
115 float distance = Util::getRandomFloat(0.f, _maxWanderDistance);
116 float angle = Util::getRandomFloat(0.f, float(M_PI * 2));
117 owner->movePositionToFirstCollision(position, distance, angle);
118
119 // Check if the destination is in LOS
120 if (!owner->IsWithinLOS(position))
121 {
122 // Retry later on
123 _timer->resetInterval(200);
124 return;
125 }
126
127 if (!_path)
128 {
129 _path = std::make_unique<PathGenerator>(owner);
130 _path->setPathLengthLimit(30.0f);
131 }
132
133 bool result = _path->calculatePath(position.getPositionX(), position.getPositionY(), position.getPositionZ());
134 // PATHFIND_FARFROMPOLY shouldn't be checked as creatures in water are most likely far from poly
135 if (!result || (_path->getPathType() & PATHFIND_NOPATH)
136 || (_path->getPathType() & PATHFIND_SHORTCUT)
137 /*|| (_path->GetPathType() & PATHFIND_FARFROMPOLY)*/)
138 {
139 _timer->resetInterval(100);
140 return;
141 }
142
143 removeFlag(MOVEMENTGENERATOR_FLAG_TRANSITORY | MOVEMENTGENERATOR_FLAG_TIMED_PAUSED);
144
145 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
146
147 bool walk = true;
148 switch (owner->getMovementTemplate().getRandom())
149 {
150 case CreatureRandomMovementType::CanRun:
151 walk = owner->isWalking();
152 break;
153 case CreatureRandomMovementType::AlwaysRun:
154 walk = false;
155 break;
156 default:
157 break;
158 }
159
160 MovementMgr::MoveSplineInit init(owner);
161 init.MovebyPath(_path->getPath());
162 init.SetWalk(walk);
163 int32_t splineDuration = init.Launch();
164
165 --_wanderSteps;
166 if (_wanderSteps) // Creature has yet to do steps before pausing
167 _timer->resetInterval(splineDuration);
168 else
169 {
170 // Creature has made all its steps, time for a little break
171 _timer->resetInterval(splineDuration + Util::getRandomUInt(4, 10) * IN_MILLISECONDS); // Retails seems to use rounded numbers so we do as well
172 _wanderSteps = static_cast<uint8_t>(Util::getRandomUInt(2, 10));
173 }
174
175 // Call for creature group update
176 owner->signalFormationMovement();
177}

185template<>
186bool RandomMovementGenerator<Creature>::doUpdate(Creature* owner, uint32_t diff)
187{
188 if (!owner || !owner->isAlive())
189 return true;
190
191 if (hasFlag(MOVEMENTGENERATOR_FLAG_FINALIZED | MOVEMENTGENERATOR_FLAG_PAUSED))
192 return true;
193
194 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell())
195 {
196 addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
197 owner->stopMoving();
198 _path = nullptr;
199 return true;
200 }
201 else
202 removeFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
203
204 _timer->updateTimer(diff);
205 if ((hasFlag(MOVEMENTGENERATOR_FLAG_SPEED_UPDATE_PENDING) && !owner->movespline->Finalized()) || (_timer->isTimePassed() && owner->movespline->Finalized()))
206 setRandomLocation(owner);
207
208 return true;
209}
