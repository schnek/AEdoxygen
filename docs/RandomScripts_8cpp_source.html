56void OnEmote(Player* pPlayer, uint32_t Emote, Unit* pUnit)
57{
58 if (!pUnit || !pUnit->isAlive() || pUnit->getAIInterface()->getCurrentTarget())
59 return;
60
61 // Switch For Emote Name (You do EmoteName to Script Name link).
62 switch (Emote)
63 {
64 case EMOTE_ONESHOT_SALUTE: // <- Its EMOTE name.
65 GuardsOnSalute(pPlayer, pUnit); // <- Its Link.
66 break;
67
68 case EMOTE_ONESHOT_KISS:
69 GaurdsOnKiss(pPlayer, pUnit);
70 break;
71
72 case EMOTE_ONESHOT_WAVE:
73 GuardsOnWave(pPlayer, pUnit);
74 break;
75 }
76}
