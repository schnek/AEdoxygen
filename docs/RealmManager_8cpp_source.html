45 void RealmManager::loadRealms()
46 {
47 const auto result = sLogonSQL->Query("SELECT id, password, status FROM realms");
48 if (result != nullptr)
49 {
50 do
51 {
52 const auto field = result->Fetch();
53 const uint32_t realmCount = result->GetRowCount();
54 this->realms.reserve(realmCount);
55
56 auto realm = std::make_unique<Realm>();
57 realm->id = field[0].asUint32();
58 realm->password = field[1].asCString();
59 realm->status = field[2].asUint8();
60 realm->lastPing = ::Util::TimeNow();
61
62 this->realms.emplace_back(std::move(realm));
63 } while (result->NextRow());
64 }
65 sLogger.info("[RealmManager] Loaded {} realms.", static_cast<uint32_t>(this->realms.size()));
66 }

114 void RealmManager::checkRealmStatus(bool silent)
115 {
116 if (!silent)
117 sLogger.info("[RealmManager] Checking Realm status...");
118
119 for (auto& realm : this->realms)
120 {
121 // if there was no ping in the last 2 minutes (in miliseconds) we set the status to the realm to offline.
122 if (::Util::GetTimeDifferenceToNow(realm->lastPing) > 2 * 60 * 1000 && realm->status != 0)
123 {
124 realm->status = 0;
125 sLogger.info("Realm {} status gets set to 0 (offline) since there was no ping the last 2 minutes ({}).", uint32_t(realm->id), ::Util::GetTimeDifferenceToNow(realm->lastPing));
126 sLogonSQL->Query("UPDATE realms SET status = 0 WHERE id = %u", uint32_t(realm->id));
127 }

131 void RealmManager::sendRealms(AuthSocket* authSocket)
132 {
133 sLogger.trace("[RealmManager] Sending realms to a socket.");
134 realmLock.lock();
135
136 ByteBuffer data(this->realms.size() * 150 + 20);
137 data << uint8_t(0x10);
138 data << uint16_t(0);
139 data << uint32_t(0);
140
141 if (authSocket->GetChallenge()->build == 5875)
142 data << uint8_t(this->realms.size());
143 else
144 data << uint16_t(this->realms.size());
145
146 std::unordered_map<uint32_t, uint8_t>::iterator it;
147 for (const auto& realm : this->realms)
148 {
149 if (realm->gameBuild == authSocket->GetChallenge()->build)
150 {
151 if (realm->gameBuild == 5875)
152 {
153 data << uint32_t(realm->icon);
154 data << uint8_t(realm->flags);
155
156 data << realm->name;
157 data << realm->address;
158 data << float(realm->population);
159
160 it = realm->_characterMap.find(authSocket->GetAccountID());
161 data << uint8_t((it == realm->_characterMap.end()) ? 0 : it->second);
162 data << uint8_t(realm->timeZone);
163 data << uint8_t(0);
164 }
165 else
166 {
167 data << uint8_t(realm->icon);
168 data << uint8_t(realm->lock);
169 data << uint8_t(realm->flags);
170
171 data << realm->name;
172 data << realm->address;
173 data << float(realm->population);
174
175 it = realm->_characterMap.find(authSocket->GetAccountID());
176 data << uint8_t(it == realm->_characterMap.end() ? 0 : it->second);
177 data << uint8_t(realm->timeZone);
178 data << uint8_t(realm->id);
179
180 if (realm->flags & AscEmu::Realm::RealmFlag::SPECIFIC_BUILD)
181 {
182 data << authSocket->GetChallenge()->version1;
183 data << authSocket->GetChallenge()->version2;
184 data << authSocket->GetChallenge()->version3;
185 data << authSocket->GetChallenge()->build;
186 }
187 }
188 }
189 else // send empty packet for other gameserver which not supports client_build
190 {
191 data << uint8_t(0);
192 data << uint8_t(0);
193 data << uint8_t(0);
194
195 data << "";
196 data << "";
197 data << float(0);
198
199 data << uint8_t(0);
200 data << uint8_t(0);
201 data << uint8_t(realm->id);
202 }
203 }
204 data << uint8_t(0x17);
205 data << uint8_t(0);
206
207 realmLock.unlock();
208
209 *reinterpret_cast<uint16_t*>(&data.contents()[1]) = uint16_t(data.size() - 3);
210
211 authSocket->Send(static_cast<const uint8_t*>(data.contents()), uint32_t(data.size()));
212
213 std::list<LogonCommServerSocket*> server_sockets;
214
215 serverSocketLock.lock();
216
217 if (this->serverSockets.empty())
218 {
219 serverSocketLock.unlock();
220 return;
221 }
222
223 for (const auto& serverSocket : this->serverSockets)
224 server_sockets.push_back(serverSocket);
225
226 serverSocketLock.unlock();
227
228 for (const auto& serverSocket : server_sockets)
229 serverSocket->RefreshRealmsPop();
230 }

232 void RealmManager::timeoutSockets()
233 {
234 if (!this->usePings)
235 return;
236
237 const auto now = uint32_t(time(nullptr));
238
239 std::lock_guard lock(serverSocketLock);
240
241 for (auto logonCommServerSocket = this->serverSockets.begin(); logonCommServerSocket != this->serverSockets.end();)
242 {
243 auto commServerSocket = *logonCommServerSocket;
244 ++logonCommServerSocket;
245
246 const uint32_t last_ping = commServerSocket->last_ping;
247 if (last_ping < now && ((now - last_ping) > 300))
248 {
249 for (auto realmIds : commServerSocket->server_ids)
250 setRealmOffline(realmIds);
251
252 commServerSocket->removed = true;
253 this->serverSockets.erase(commServerSocket);
254 commServerSocket->Disconnect();
255 }
256 }

291 void RealmManager::setRealmPopulation(uint32_t realm_id, float population)
292 {
293 std::lock_guard lock(realmLock);
294
295 auto realm = getRealmById(realm_id);
296 if (realm != nullptr)
297 {
298 uint8_t flags;
299 if (population >= 3)
300 flags = RealmFlag::FULL | RealmFlag::INVALID;
301 else if (population >= 2)
302 flags = RealmFlag::INVALID;
303 else if (population >= 1.25)
304 flags = 0;
305 else if (population >= 0.5)
306 flags = RealmFlag::RECOMMENDED;
307 else
308 flags = RealmFlag::NEW_PLAYERS;
309
310 realm->population = (population > 0) ? (population >= 1) ? (population >= 2) ? 2.0f : 1.0f : 0.0f : 0.0f;
311 realm->flags = flags;
312 }
