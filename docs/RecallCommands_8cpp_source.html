116bool ChatCommandHandler::HandleRecallListCommand(const char* args, WorldSession* m_session)
117{
118 uint32_t count = 0;
119
120 std::string recout = MSG_COLOR_GREEN;
121 recout += "Recall locations|r:\n\n";
122
123 std::string search(args);
124 AscEmu::Util::Strings::toLowerCase(search);
125
126 for (const auto& recall : sMySQLStore.getRecallStore())
127 {
128 std::string recallName(recall->name);
129 AscEmu::Util::Strings::toLowerCase(recallName);
130 if (recallName.find(search) == 0)
131 {
132 recout += MSG_COLOR_LIGHTBLUE;
133 recout += recall->name;
134 recout += "|r, ";
135 count++;
136
137 if (count == 5)
138 {
139 recout += "\n";
140 count = 0;
141 }
142 }
143 }
144
145 SendMultilineMessage(m_session, recout.c_str());
146
147 return true;
148}

151bool ChatCommandHandler::HandleRecallPortPlayerCommand(const char* args, WorldSession* m_session)
152{
153 if (!args || !*args)
154 return false;
155
156 std::istringstream iss(std::string{ args });
157 std::string playerName;
158 std::string location;
159
160 if (!(iss >> playerName >> location))
161 return false;
162
163 Player* player = sObjectMgr.getPlayer(playerName.c_str(), false);
164 if (!player)
165 return false;
166
167 if (const auto recall = sMySQLStore.getRecallByName(location))
168 {
169 sGMLog.writefromsession(m_session, "ported %s to %s ( map: %u, x: %f, y: %f, z: %f, 0: %f )", player->getName().c_str(), recall->name.c_str(), recall->mapId, recall->location.x, recall->location.y, recall->location.z, recall->location.o);
170 if (player->getSession() && (player->getSession()->CanUseCommand('a') || !m_session->GetPlayer()->m_isGmInvisible))
171 player->getSession()->SystemMessage("%s teleported you to location %s!", m_session->GetPlayer()->getName().c_str(), recall->name.c_str());
172
173 if (player->GetInstanceID() != m_session->GetPlayer()->GetInstanceID())
174 sEventMgr.AddEvent(player, &Player::eventTeleport, recall->mapId, recall->location, uint32_t(0), EVENT_PLAYER_TELEPORT, 1, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
175 else
176 player->safeTeleport(recall->mapId, 0, recall->location);
177
178 return true;
179 }
180
181 return false;
