64/// An implementation of operator new usable for placement new. The default one is part of STL (which we don't use).

86/// Variable-sized storage type. Mimics the interface of std::vector<T> with some notable differences:
87/// * Uses rcAlloc()/rcFree() to handle storage.
88/// * No support for a custom allocator.
89/// * Uses signed size instead of size_t to avoid warnings in for loops: "for (int i = 0; i < foo.size(); i++)"
90/// * Omits methods of limited utility: insert/erase, (bad performance), at (we don't use exceptions), operator=.
91/// * assign() and the pre-sizing constructor follow C++11 semantics -- they don't construct a temporary if no value is provided.
92/// * push_back() and resize() support adding values from the current vector. Range-based constructors and assign(begin, end) do not.
93/// * No specialization for bool.
94template <typename T, rcAllocHint H>
95class rcVectorBase {
96 rcSizeType m_size;
97 rcSizeType m_cap;
98 T* m_data;
99 // Constructs a T at the give address with either the copy constructor or the default.
100 static void construct(T* p, const T& v) { ::new(rcNewTag(), (void*)p) T(v); }
101 static void construct(T* p) { ::new(rcNewTag(), (void*)p) T; }
102 static void construct_range(T* begin, T* end);
103 static void construct_range(T* begin, T* end, const T& value);
104 static void copy_range(T* dst, const T* begin, const T* end);
105 void destroy_range(rcSizeType begin, rcSizeType end);
106 // Creates an array of the given size, copies all of this vector's data into it, and returns it.
107 T* allocate_and_copy(rcSizeType size);
108 void resize_impl(rcSizeType size, const T* value);
109 // Requires: min_capacity > m_cap.
110 rcSizeType get_new_capacity(rcSizeType min_capacity);
111 public:
112 typedef rcSizeType size_type;
113 typedef T value_type;
114
115 rcVectorBase() : m_size(0), m_cap(0), m_data(0) {}
116 rcVectorBase(const rcVectorBase<T, H>& other) : m_size(0), m_cap(0), m_data(0) { assign(other.begin(), other.end()); }
117 explicit rcVectorBase(rcSizeType count) : m_size(0), m_cap(0), m_data(0) { resize(count); }
118 rcVectorBase(rcSizeType count, const T& value) : m_size(0), m_cap(0), m_data(0) { resize(count, value); }
119 rcVectorBase(const T* begin, const T* end) : m_size(0), m_cap(0), m_data(0) { assign(begin, end); }
120 ~rcVectorBase() { destroy_range(0, m_size); rcFree(m_data); }
121
122 // Unlike in std::vector, we return a bool to indicate whether the alloc was successful.
123 bool reserve(rcSizeType size);
124
125 void assign(rcSizeType count, const T& value) { clear(); resize(count, value); }
126 void assign(const T* begin, const T* end);
127
128 void resize(rcSizeType size) { resize_impl(size, NULL); }
129 void resize(rcSizeType size, const T& value) { resize_impl(size, &value); }
130 // Not implemented as resize(0) because resize requires T to be default-constructible.
131 void clear() { destroy_range(0, m_size); m_size = 0; }
132
133 void push_back(const T& value);
134 void pop_back() { rcAssert(m_size > 0); back().~T(); m_size--; }
135
136 rcSizeType size() const { return m_size; }
137 rcSizeType capacity() const { return m_cap; }
138 bool empty() const { return size() == 0; }
139
140 const T& operator[](rcSizeType i) const { rcAssert(i >= 0 && i < m_size); return m_data[i]; }
141 T& operator[](rcSizeType i) { rcAssert(i >= 0 && i < m_size); return m_data[i]; }
142
143 const T& front() const { rcAssert(m_size); return m_data[0]; }
144 T& front() { rcAssert(m_size); return m_data[0]; }
145 const T& back() const { rcAssert(m_size); return m_data[m_size - 1]; }
146 T& back() { rcAssert(m_size); return m_data[m_size - 1]; }
147 const T* data() const { return m_data; }
148 T* data() { return m_data; }
149
150 T* begin() { return m_data; }
151 T* end() { return m_data + m_size; }
152 const T* begin() const { return m_data; }
153 const T* end() const { return m_data + m_size; }
154
155 void swap(rcVectorBase<T, H>& other);
156
157 // Explicitly deleted.
158 rcVectorBase& operator=(const rcVectorBase<T, H>& other);
159};

161template<typename T, rcAllocHint H>
162bool rcVectorBase<T, H>::reserve(rcSizeType count) {
163 if (count <= m_cap) {
164 return true;
165 }
166 T* new_data = allocate_and_copy(count);
167 if (!new_data) {
168 return false;
169 }
170 destroy_range(0, m_size);
171 rcFree(m_data);
172 m_data = new_data;
173 m_cap = count;
174 return true;
175}
176template <typename T, rcAllocHint H>
177T* rcVectorBase<T, H>::allocate_and_copy(rcSizeType size) {
178 rcAssert(RC_SIZE_MAX / static_cast<rcSizeType>(sizeof(T)) >= size);
179 T* new_data = static_cast<T*>(rcAlloc(sizeof(T) * size, H));
180 if (new_data) {
181 copy_range(new_data, m_data, m_data + m_size);
182 }
183 return new_data;
184}
185template <typename T, rcAllocHint H>
186void rcVectorBase<T, H>::assign(const T* begin, const T* end) {
187 clear();
188 reserve(end - begin);
189 m_size = end - begin;
190 copy_range(m_data, begin, end);
191}
192template <typename T, rcAllocHint H>
193void rcVectorBase<T, H>::push_back(const T& value) {
194 // rcLikely increases performance by ~50% on BM_rcVector_PushPreallocated,
195 // and by ~2-5% on BM_rcVector_Push.
196 if (rcLikely(m_size < m_cap)) {
197 construct(m_data + m_size++, value);
198 return;
199 }
200
201 const rcSizeType new_cap = get_new_capacity(m_cap + 1);
202 T* data = allocate_and_copy(new_cap);
203 // construct between allocate and destroy+free in case value is
204 // in this vector.
205 construct(data + m_size, value);
206 destroy_range(0, m_size);
207 m_size++;
208 m_cap = new_cap;
209 rcFree(m_data);
210 m_data = data;
211}

213template <typename T, rcAllocHint H>
214rcSizeType rcVectorBase<T, H>::get_new_capacity(rcSizeType min_capacity) {
215 rcAssert(min_capacity <= RC_SIZE_MAX);
216 if (rcUnlikely(m_cap >= RC_SIZE_MAX / 2))
217 return RC_SIZE_MAX;
218 return 2 * m_cap > min_capacity ? 2 * m_cap : min_capacity;
219}

221template <typename T, rcAllocHint H>
222void rcVectorBase<T, H>::resize_impl(rcSizeType size, const T* value) {
223 if (size < m_size) {
224 destroy_range(size, m_size);
225 m_size = size;
226 } else if (size > m_size) {
227 if (size <= m_cap) {
228 if (value) {
229 construct_range(m_data + m_size, m_data + size, *value);
230 } else {
231 construct_range(m_data + m_size, m_data + size);
232 }
233 m_size = size;
234 } else {
235 const rcSizeType new_cap = get_new_capacity(size);
236 T* new_data = allocate_and_copy(new_cap);
237 // We defer deconstructing/freeing old data until after constructing
238 // new elements in case "value" is there.
239 if (value) {
240 construct_range(new_data + m_size, new_data + size, *value);
241 } else {
242 construct_range(new_data + m_size, new_data + size);
243 }
244 destroy_range(0, m_size);
245 rcFree(m_data);
246 m_data = new_data;
247 m_cap = new_cap;
248 m_size = size;
249 }
250 }

253void rcVectorBase<T, H>::swap(rcVectorBase<T, H>& other) {
254 // TODO: Reorganize headers so we can use rcSwap here.
255 rcSizeType tmp_cap = other.m_cap;
256 rcSizeType tmp_size = other.m_size;
257 T* tmp_data = other.m_data;
258
259 other.m_cap = m_cap;
260 other.m_size = m_size;
261 other.m_data = m_data;
262
263 m_cap = tmp_cap;
264 m_size = tmp_size;
265 m_data = tmp_data;
