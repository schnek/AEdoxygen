29/// @par
30///
31/// Basically, any spans that are closer to a boundary or obstruction than the specified radius
32/// are marked as unwalkable.
33///
34/// This method is usually called immediately after the heightfield has been built.
35///
36/// @see rcCompactHeightfield, rcBuildCompactHeightfield, rcConfig::walkableRadius
37bool rcErodeWalkableArea(rcContext* ctx, int radius, rcCompactHeightfield& chf)
38{
39 rcAssert(ctx);
40
41 const int w = chf.width;
42 const int h = chf.height;
43
44 rcScopedTimer timer(ctx, RC_TIMER_ERODE_AREA);
45
46 unsigned char* dist = (unsigned char*)rcAlloc(sizeof(unsigned char)*chf.spanCount, RC_ALLOC_TEMP);
47 if (!dist)
48 {
49 ctx->log(RC_LOG_ERROR, "erodeWalkableArea: Out of memory 'dist' (%d).", chf.spanCount);
50 return false;
51 }
52
53 // Init distance.
54 memset(dist, 0xff, sizeof(unsigned char)*chf.spanCount);
55
56 // Mark boundary cells.
57 for (int y = 0; y < h; ++y)
58 {
59 for (int x = 0; x < w; ++x)
60 {
61 const rcCompactCell& c = chf.cells[x+y*w];
62 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
63 {
64 if (chf.areas[i] == RC_NULL_AREA)
65 {
66 dist[i] = 0;
67 }
68 else
69 {
70 const rcCompactSpan& s = chf.spans[i];
71 int nc = 0;
72 for (int dir = 0; dir < 4; ++dir)
73 {
74 if (rcGetCon(s, dir) != RC_NOT_CONNECTED)
75 {
76 const int nx = x + rcGetDirOffsetX(dir);
77 const int ny = y + rcGetDirOffsetY(dir);
78 const int nidx = (int)chf.cells[nx+ny*w].index + rcGetCon(s, dir);
79 if (chf.areas[nidx] != RC_NULL_AREA)
80 {
81 nc++;
82 }
83 }
84 }
85 // At least one missing neighbour.
86 if (nc != 4)
87 dist[i] = 0;
88 }
89 }
90 }
91 }
92
93 unsigned char nd;
94
95 // Pass 1
96 for (int y = 0; y < h; ++y)
97 {
98 for (int x = 0; x < w; ++x)
99 {
100 const rcCompactCell& c = chf.cells[x+y*w];
101 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
102 {
103 const rcCompactSpan& s = chf.spans[i];
104
105 if (rcGetCon(s, 0) != RC_NOT_CONNECTED)
106 {
107 // (-1,0)
108 const int ax = x + rcGetDirOffsetX(0);
109 const int ay = y + rcGetDirOffsetY(0);
110 const int ai = (int)chf.cells[ax+ay*w].index + rcGetCon(s, 0);
111 const rcCompactSpan& as = chf.spans[ai];
112 nd = (unsigned char)rcMin((int)dist[ai]+2, 255);
113 if (nd < dist[i])
114 dist[i] = nd;
115
116 // (-1,-1)
117 if (rcGetCon(as, 3) != RC_NOT_CONNECTED)
118 {
119 const int aax = ax + rcGetDirOffsetX(3);
120 const int aay = ay + rcGetDirOffsetY(3);
121 const int aai = (int)chf.cells[aax+aay*w].index + rcGetCon(as, 3);
122 nd = (unsigned char)rcMin((int)dist[aai]+3, 255);
123 if (nd < dist[i])
124 dist[i] = nd;
125 }
126 }
127 if (rcGetCon(s, 3) != RC_NOT_CONNECTED)
128 {
129 // (0,-1)
130 const int ax = x + rcGetDirOffsetX(3);
131 const int ay = y + rcGetDirOffsetY(3);
132 const int ai = (int)chf.cells[ax+ay*w].index + rcGetCon(s, 3);
133 const rcCompactSpan& as = chf.spans[ai];
134 nd = (unsigned char)rcMin((int)dist[ai]+2, 255);
135 if (nd < dist[i])
136 dist[i] = nd;
137
138 // (1,-1)
139 if (rcGetCon(as, 2) != RC_NOT_CONNECTED)
140 {
141 const int aax = ax + rcGetDirOffsetX(2);
142 const int aay = ay + rcGetDirOffsetY(2);
143 const int aai = (int)chf.cells[aax+aay*w].index + rcGetCon(as, 2);
144 nd = (unsigned char)rcMin((int)dist[aai]+3, 255);
145 if (nd < dist[i])
146 dist[i] = nd;
147 }
148 }
149 }
150 }
151 }
152
153 // Pass 2
154 for (int y = h-1; y >= 0; --y)
155 {
156 for (int x = w-1; x >= 0; --x)
157 {
158 const rcCompactCell& c = chf.cells[x+y*w];
159 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
160 {
161 const rcCompactSpan& s = chf.spans[i];
162
163 if (rcGetCon(s, 2) != RC_NOT_CONNECTED)
164 {
165 // (1,0)
166 const int ax = x + rcGetDirOffsetX(2);
167 const int ay = y + rcGetDirOffsetY(2);
168 const int ai = (int)chf.cells[ax+ay*w].index + rcGetCon(s, 2);
169 const rcCompactSpan& as = chf.spans[ai];
170 nd = (unsigned char)rcMin((int)dist[ai]+2, 255);
171 if (nd < dist[i])
172 dist[i] = nd;
173
174 // (1,1)
175 if (rcGetCon(as, 1) != RC_NOT_CONNECTED)
176 {
177 const int aax = ax + rcGetDirOffsetX(1);
178 const int aay = ay + rcGetDirOffsetY(1);
179 const int aai = (int)chf.cells[aax+aay*w].index + rcGetCon(as, 1);
180 nd = (unsigned char)rcMin((int)dist[aai]+3, 255);
181 if (nd < dist[i])
182 dist[i] = nd;
183 }
184 }
185 if (rcGetCon(s, 1) != RC_NOT_CONNECTED)
186 {
187 // (0,1)
188 const int ax = x + rcGetDirOffsetX(1);
189 const int ay = y + rcGetDirOffsetY(1);
190 const int ai = (int)chf.cells[ax+ay*w].index + rcGetCon(s, 1);
191 const rcCompactSpan& as = chf.spans[ai];
192 nd = (unsigned char)rcMin((int)dist[ai]+2, 255);
193 if (nd < dist[i])
194 dist[i] = nd;
195
196 // (-1,1)
197 if (rcGetCon(as, 0) != RC_NOT_CONNECTED)
198 {
199 const int aax = ax + rcGetDirOffsetX(0);
200 const int aay = ay + rcGetDirOffsetY(0);
201 const int aai = (int)chf.cells[aax+aay*w].index + rcGetCon(as, 0);
202 nd = (unsigned char)rcMin((int)dist[aai]+3, 255);
203 if (nd < dist[i])
204 dist[i] = nd;
205 }
206 }
207 }
208 }
209 }
210
211 const unsigned char thr = (unsigned char)(radius*2);
212 for (int i = 0; i < chf.spanCount; ++i)
213 if (dist[i] < thr)
214 chf.areas[i] = RC_NULL_AREA;
215
216 rcFree(dist);
217
218 return true;
219}

221static void insertSort(unsigned char* a, const int n)
222{
223 int i, j;
224 for (i = 1; i < n; i++)
225 {
226 const unsigned char value = a[i];
227 for (j = i - 1; j >= 0 && a[j] > value; j--)
228 a[j+1] = a[j];
229 a[j+1] = value;
230 }
231}

233/// @par
234///
235/// This filter is usually applied after applying area id's using functions
236/// such as #rcMarkBoxArea, #rcMarkConvexPolyArea, and #rcMarkCylinderArea.
237///
238/// @see rcCompactHeightfield
239bool rcMedianFilterWalkableArea(rcContext* ctx, rcCompactHeightfield& chf)
240{
241 rcAssert(ctx);
242
243 const int w = chf.width;
244 const int h = chf.height;
245
246 rcScopedTimer timer(ctx, RC_TIMER_MEDIAN_AREA);
247
248 unsigned char* areas = (unsigned char*)rcAlloc(sizeof(unsigned char)*chf.spanCount, RC_ALLOC_TEMP);
249 if (!areas)
250 {
251 ctx->log(RC_LOG_ERROR, "medianFilterWalkableArea: Out of memory 'areas' (%d).", chf.spanCount);
252 return false;
253 }
254
255 // Init distance.
256 memset(areas, 0xff, sizeof(unsigned char)*chf.spanCount);
257
258 for (int y = 0; y < h; ++y)
259 {
260 for (int x = 0; x < w; ++x)
261 {
262 const rcCompactCell& c = chf.cells[x+y*w];
263 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
264 {
265 const rcCompactSpan& s = chf.spans[i];
266 if (chf.areas[i] == RC_NULL_AREA)
267 {
268 areas[i] = chf.areas[i];
269 continue;
270 }
271
272 unsigned char nei[9];
273 for (int j = 0; j < 9; ++j)
274 nei[j] = chf.areas[i];
275
276 for (int dir = 0; dir < 4; ++dir)
277 {
278 if (rcGetCon(s, dir) != RC_NOT_CONNECTED)
279 {
280 const int ax = x + rcGetDirOffsetX(dir);
281 const int ay = y + rcGetDirOffsetY(dir);
282 const int ai = (int)chf.cells[ax+ay*w].index + rcGetCon(s, dir);
283 if (chf.areas[ai] != RC_NULL_AREA)
284 nei[dir*2+0] = chf.areas[ai];
285
286 const rcCompactSpan& as = chf.spans[ai];
287 const int dir2 = (dir+1) & 0x3;
288 if (rcGetCon(as, dir2) != RC_NOT_CONNECTED)
289 {
290 const int ax2 = ax + rcGetDirOffsetX(dir2);
291 const int ay2 = ay + rcGetDirOffsetY(dir2);
292 const int ai2 = (int)chf.cells[ax2+ay2*w].index + rcGetCon(as, dir2);
293 if (chf.areas[ai2] != RC_NULL_AREA)
294 nei[dir*2+1] = chf.areas[ai2];
295 }
296 }
297 }
298 insertSort(nei, 9);
299 areas[i] = nei[4];
300 }
301 }
302 }
303
304 memcpy(chf.areas, areas, sizeof(unsigned char)*chf.spanCount);
305
306 rcFree(areas);
307
308 return true;
309}

311/// @par
312///
313/// The value of spacial parameters are in world units.
314///
315/// @see rcCompactHeightfield, rcMedianFilterWalkableArea
316void rcMarkBoxArea(rcContext* ctx, const float* bmin, const float* bmax, unsigned char areaId,
317 rcCompactHeightfield& chf)
318{
319 rcAssert(ctx);
320
321 rcScopedTimer timer(ctx, RC_TIMER_MARK_BOX_AREA);
322
323 int minx = (int)((bmin[0]-chf.bmin[0])/chf.cs);
324 int miny = (int)((bmin[1]-chf.bmin[1])/chf.ch);
325 int minz = (int)((bmin[2]-chf.bmin[2])/chf.cs);
326 int maxx = (int)((bmax[0]-chf.bmin[0])/chf.cs);
327 int maxy = (int)((bmax[1]-chf.bmin[1])/chf.ch);
328 int maxz = (int)((bmax[2]-chf.bmin[2])/chf.cs);
329
330 if (maxx < 0) return;
331 if (minx >= chf.width) return;
332 if (maxz < 0) return;
333 if (minz >= chf.height) return;
334
335 if (minx < 0) minx = 0;
336 if (maxx >= chf.width) maxx = chf.width-1;
337 if (minz < 0) minz = 0;
338 if (maxz >= chf.height) maxz = chf.height-1;
339
340 for (int z = minz; z <= maxz; ++z)
341 {
342 for (int x = minx; x <= maxx; ++x)
343 {
344 const rcCompactCell& c = chf.cells[x+z*chf.width];
345 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
346 {
347 rcCompactSpan& s = chf.spans[i];
348 if ((int)s.y >= miny && (int)s.y <= maxy)
349 {
350 if (chf.areas[i] != RC_NULL_AREA)
351 chf.areas[i] = areaId;
352 }
353 }
354 }
355 }

359static int pointInPoly(int nvert, const float* verts, const float* p)
360{
361 int i, j, c = 0;
362 for (i = 0, j = nvert-1; i < nvert; j = i++)
363 {
364 const float* vi = &verts[i*3];
365 const float* vj = &verts[j*3];
366 if (((vi[2] > p[2]) != (vj[2] > p[2])) &&
367 (p[0] < (vj[0]-vi[0]) * (p[2]-vi[2]) / (vj[2]-vi[2]) + vi[0]) )
368 c = !c;
369 }
370 return c;
371}

373/// @par
374///
375/// The value of spacial parameters are in world units.
376///
377/// The y-values of the polygon vertices are ignored. So the polygon is effectively
378/// projected onto the xz-plane at @p hmin, then extruded to @p hmax.
379///
380/// @see rcCompactHeightfield, rcMedianFilterWalkableArea
381void rcMarkConvexPolyArea(rcContext* ctx, const float* verts, const int nverts,
382 const float hmin, const float hmax, unsigned char areaId,
383 rcCompactHeightfield& chf)
384{
385 rcAssert(ctx);
386
387 rcScopedTimer timer(ctx, RC_TIMER_MARK_CONVEXPOLY_AREA);
388
389 float bmin[3], bmax[3];
390 rcVcopy(bmin, verts);
391 rcVcopy(bmax, verts);
392 for (int i = 1; i < nverts; ++i)
393 {
394 rcVmin(bmin, &verts[i*3]);
395 rcVmax(bmax, &verts[i*3]);
396 }
397 bmin[1] = hmin;
398 bmax[1] = hmax;
399
400 int minx = (int)((bmin[0]-chf.bmin[0])/chf.cs);
401 int miny = (int)((bmin[1]-chf.bmin[1])/chf.ch);
402 int minz = (int)((bmin[2]-chf.bmin[2])/chf.cs);
403 int maxx = (int)((bmax[0]-chf.bmin[0])/chf.cs);
404 int maxy = (int)((bmax[1]-chf.bmin[1])/chf.ch);
405 int maxz = (int)((bmax[2]-chf.bmin[2])/chf.cs);
406
407 if (maxx < 0) return;
408 if (minx >= chf.width) return;
409 if (maxz < 0) return;
410 if (minz >= chf.height) return;
411
412 if (minx < 0) minx = 0;
413 if (maxx >= chf.width) maxx = chf.width-1;
414 if (minz < 0) minz = 0;
415 if (maxz >= chf.height) maxz = chf.height-1;
416
417
418 // TODO: Optimize.
419 for (int z = minz; z <= maxz; ++z)
420 {
421 for (int x = minx; x <= maxx; ++x)
422 {
423 const rcCompactCell& c = chf.cells[x+z*chf.width];
424 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
425 {
426 rcCompactSpan& s = chf.spans[i];
427 if (chf.areas[i] == RC_NULL_AREA)
428 continue;
429 if ((int)s.y >= miny && (int)s.y <= maxy)
430 {
431 float p[3];
432 p[0] = chf.bmin[0] + (x+0.5f)*chf.cs;
433 p[1] = 0;
434 p[2] = chf.bmin[2] + (z+0.5f)*chf.cs;
435
436 if (pointInPoly(nverts, verts, p))
437 {
438 chf.areas[i] = areaId;
439 }
440 }
441 }
442 }
443 }

446int rcOffsetPoly(const float* verts, const int nverts, const float offset,
447 float* outVerts, const int maxOutVerts)
448{
449 const float MITER_LIMIT = 1.20f;
450
451 int n = 0;
452
453 for (int i = 0; i < nverts; i++)
454 {
455 const int a = (i+nverts-1) % nverts;
456 const int b = i;
457 const int c = (i+1) % nverts;
458 const float* va = &verts[a*3];
459 const float* vb = &verts[b*3];
460 const float* vc = &verts[c*3];
461 float dx0 = vb[0] - va[0];
462 float dy0 = vb[2] - va[2];
463 float d0 = dx0*dx0 + dy0*dy0;
464 if (d0 > 1e-6f)
465 {
466 d0 = 1.0f/rcSqrt(d0);
467 dx0 *= d0;
468 dy0 *= d0;
469 }
470 float dx1 = vc[0] - vb[0];
471 float dy1 = vc[2] - vb[2];
472 float d1 = dx1*dx1 + dy1*dy1;
473 if (d1 > 1e-6f)
474 {
475 d1 = 1.0f/rcSqrt(d1);
476 dx1 *= d1;
477 dy1 *= d1;
478 }
479 const float dlx0 = -dy0;
480 const float dly0 = dx0;
481 const float dlx1 = -dy1;
482 const float dly1 = dx1;
483 float cross = dx1*dy0 - dx0*dy1;
484 float dmx = (dlx0 + dlx1) * 0.5f;
485 float dmy = (dly0 + dly1) * 0.5f;
486 float dmr2 = dmx*dmx + dmy*dmy;
487 bool bevel = dmr2 * MITER_LIMIT*MITER_LIMIT < 1.0f;
488 if (dmr2 > 1e-6f)
489 {
490 const float scale = 1.0f / dmr2;
491 dmx *= scale;
492 dmy *= scale;
493 }
494
495 if (bevel && cross < 0.0f)
496 {
497 if (n+2 >= maxOutVerts)
498 return 0;
499 float d = (1.0f - (dx0*dx1 + dy0*dy1))*0.5f;
500 outVerts[n*3+0] = vb[0] + (-dlx0+dx0*d)*offset;
501 outVerts[n*3+1] = vb[1];
502 outVerts[n*3+2] = vb[2] + (-dly0+dy0*d)*offset;
503 n++;
504 outVerts[n*3+0] = vb[0] + (-dlx1-dx1*d)*offset;
505 outVerts[n*3+1] = vb[1];
506 outVerts[n*3+2] = vb[2] + (-dly1-dy1*d)*offset;
507 n++;
508 }
509 else
510 {
511 if (n+1 >= maxOutVerts)
512 return 0;
513 outVerts[n*3+0] = vb[0] - dmx*offset;
514 outVerts[n*3+1] = vb[1];
515 outVerts[n*3+2] = vb[2] - dmy*offset;
516 n++;
517 }
518 }
519
520 return n;
521}

524/// @par
525///
526/// The value of spacial parameters are in world units.
527///
528/// @see rcCompactHeightfield, rcMedianFilterWalkableArea
529void rcMarkCylinderArea(rcContext* ctx, const float* pos,
530 const float r, const float h, unsigned char areaId,
531 rcCompactHeightfield& chf)
532{
533 rcAssert(ctx);
534
535 rcScopedTimer timer(ctx, RC_TIMER_MARK_CYLINDER_AREA);
536
537 float bmin[3], bmax[3];
538 bmin[0] = pos[0] - r;
539 bmin[1] = pos[1];
540 bmin[2] = pos[2] - r;
541 bmax[0] = pos[0] + r;
542 bmax[1] = pos[1] + h;
543 bmax[2] = pos[2] + r;
544 const float r2 = r*r;
545
546 int minx = (int)((bmin[0]-chf.bmin[0])/chf.cs);
547 int miny = (int)((bmin[1]-chf.bmin[1])/chf.ch);
548 int minz = (int)((bmin[2]-chf.bmin[2])/chf.cs);
549 int maxx = (int)((bmax[0]-chf.bmin[0])/chf.cs);
550 int maxy = (int)((bmax[1]-chf.bmin[1])/chf.ch);
551 int maxz = (int)((bmax[2]-chf.bmin[2])/chf.cs);
552
553 if (maxx < 0) return;
554 if (minx >= chf.width) return;
555 if (maxz < 0) return;
556 if (minz >= chf.height) return;
557
558 if (minx < 0) minx = 0;
559 if (maxx >= chf.width) maxx = chf.width-1;
560 if (minz < 0) minz = 0;
561 if (maxz >= chf.height) maxz = chf.height-1;
562
563
564 for (int z = minz; z <= maxz; ++z)
565 {
566 for (int x = minx; x <= maxx; ++x)
567 {
568 const rcCompactCell& c = chf.cells[x+z*chf.width];
569 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
570 {
571 rcCompactSpan& s = chf.spans[i];
572
573 if (chf.areas[i] == RC_NULL_AREA)
574 continue;
575
576 if ((int)s.y >= miny && (int)s.y <= maxy)
577 {
578 const float sx = chf.bmin[0] + (x+0.5f)*chf.cs;
579 const float sz = chf.bmin[2] + (z+0.5f)*chf.cs;
580 const float dx = sx - pos[0];
581 const float dz = sz - pos[2];
582
583 if (dx*dx + dz*dz < r2)
584 {
585 chf.areas[i] = areaId;
586 }
587 }
588 }
589 }
590 }
