26inline bool overlapBounds(const float* amin, const float* amax, const float* bmin, const float* bmax)
27{
28 bool overlap = true;
29 overlap = (amin[0] > bmax[0] || amax[0] < bmin[0]) ? false : overlap;
30 overlap = (amin[1] > bmax[1] || amax[1] < bmin[1]) ? false : overlap;
31 overlap = (amin[2] > bmax[2] || amax[2] < bmin[2]) ? false : overlap;
32 return overlap;
33}

40 return true;

44static rcSpan* allocSpan(rcHeightfield& hf)
45{
46 // If running out of memory, allocate new page and update the freelist.
47 if (!hf.freelist || !hf.freelist->next)
48 {
49 // Create new page.
50 // Allocate memory for the new pool.
51 rcSpanPool* pool = (rcSpanPool*)rcAlloc(sizeof(rcSpanPool), RC_ALLOC_PERM);
52 if (!pool) return 0;
53
54 // Add the pool into the list of pools.
55 pool->next = hf.pools;
56 hf.pools = pool;
57 // Add new items to the free list.
58 rcSpan* freelist = hf.freelist;
59 rcSpan* head = &pool->items[0];
60 rcSpan* it = &pool->items[RC_SPANS_PER_POOL];
61 do
62 {
63 --it;
64 it->next = freelist;
65 freelist = it;
66 }
67 while (it != head);
68 hf.freelist = it;
69 }
70
71 // Pop item from in front of the free list.
72 rcSpan* it = hf.freelist;
73 hf.freelist = hf.freelist->next;
74 return it;
75}

77static void freeSpan(rcHeightfield& hf, rcSpan* ptr)
78{
79 if (!ptr) return;
80 // Add the node in front of the free list.
81 ptr->next = hf.freelist;
82 hf.freelist = ptr;
83}

85static bool addSpan(rcHeightfield& hf, const int x, const int y,
86 const unsigned short smin, const unsigned short smax,
87 const unsigned char area, const int flagMergeThr)
88{
89
90 int idx = x + y*hf.width;
91
92 rcSpan* s = allocSpan(hf);
93 if (!s)
94 return false;
95 s->smin = smin;
96 s->smax = smax;
97 s->area = area;
98 s->next = 0;
99
100 // Empty cell, add the first span.
101 if (!hf.spans[idx])
102 {
103 hf.spans[idx] = s;
104 return true;
105 }
106 rcSpan* prev = 0;
107 rcSpan* cur = hf.spans[idx];
108
109 // Insert and merge spans.
110 while (cur)
111 {
112 if (cur->smin > s->smax)
113 {
114 // Current span is further than the new span, break.
115 break;
116 }
117 else if (cur->smax < s->smin)
118 {
119 // Current span is before the new span advance.
120 prev = cur;
121 cur = cur->next;
122 }
123 else
124 {
125 // Merge spans.
126 if (cur->smin < s->smin)
127 s->smin = cur->smin;
128 if (cur->smax > s->smax)
129 s->smax = cur->smax;
130
131 // Merge flags.
132 if (rcAbs((int)s->smax - (int)cur->smax) <= flagMergeThr)
133 s->area = rcMax(s->area, cur->area);
134
135 // Remove current span.
136 rcSpan* next = cur->next;
137 freeSpan(hf, cur);
138 if (prev)
139 prev->next = next;
140 else
141 hf.spans[idx] = next;
142 cur = next;
143 }
144 }
145
146 // Insert new span.
147 if (prev)
148 {
149 s->next = prev->next;
150 prev->next = s;
151 }
152 else
153 {
154 s->next = hf.spans[idx];
155 hf.spans[idx] = s;
156 }
157
158 return true;
159}

178 }
179
180 return true;
181}

183// divides a convex polygons into two convex polygons on both sides of a line
184static void dividePoly(const float* in, int nin,
185 float* out1, int* nout1,
186 float* out2, int* nout2,
187 float x, int axis)
188{
189 float d[12];
190 for (int i = 0; i < nin; ++i)
191 d[i] = x - in[i*3+axis];
192
193 int m = 0, n = 0;
194 for (int i = 0, j = nin-1; i < nin; j=i, ++i)
195 {
196 bool ina = d[j] >= 0;
197 bool inb = d[i] >= 0;
198 if (ina != inb)
199 {
200 float s = d[j] / (d[j] - d[i]);
201 out1[m*3+0] = in[j*3+0] + (in[i*3+0] - in[j*3+0])*s;
202 out1[m*3+1] = in[j*3+1] + (in[i*3+1] - in[j*3+1])*s;
203 out1[m*3+2] = in[j*3+2] + (in[i*3+2] - in[j*3+2])*s;
204 rcVcopy(out2 + n*3, out1 + m*3);
205 m++;
206 n++;
207 // add the i'th point to the right polygon. Do NOT add points that are on the dividing line
208 // since these were already added above
209 if (d[i] > 0)
210 {
211 rcVcopy(out1 + m*3, in + i*3);
212 m++;
213 }
214 else if (d[i] < 0)
215 {
216 rcVcopy(out2 + n*3, in + i*3);
217 n++;
218 }
219 }
220 else // same side
221 {
222 // add the i'th point to the right polygon. Addition is done even for points on the dividing line
223 if (d[i] >= 0)
224 {
225 rcVcopy(out1 + m*3, in + i*3);
226 m++;
227 if (d[i] != 0)
228 continue;
229 }
230 rcVcopy(out2 + n*3, in + i*3);
231 n++;
232 }
233 }
234
235 *nout1 = m;
236 *nout2 = n;
237}

241static bool rasterizeTri(const float* v0, const float* v1, const float* v2,
242 const unsigned char area, rcHeightfield& hf,
243 const float* bmin, const float* bmax,
244 const float cs, const float ics, const float ich,
245 const int flagMergeThr)
246{
247 const int w = hf.width;
248 const int h = hf.height;
249 float tmin[3], tmax[3];
250 const float by = bmax[1] - bmin[1];
251
252 // Calculate the bounding box of the triangle.
253 rcVcopy(tmin, v0);
254 rcVcopy(tmax, v0);
255 rcVmin(tmin, v1);
256 rcVmin(tmin, v2);
257 rcVmax(tmax, v1);
258 rcVmax(tmax, v2);
259
260 // If the triangle does not touch the bbox of the heightfield, skip the triagle.
261 if (!overlapBounds(bmin, bmax, tmin, tmax))
262 return true;
263
264 // Calculate the footprint of the triangle on the grid's y-axis
265 int y0 = (int)((tmin[2] - bmin[2])*ics);
266 int y1 = (int)((tmax[2] - bmin[2])*ics);
267 // use -1 rather than 0 to cut the polygon properly at the start of the tile
268 y0 = rcClamp(y0, -1, h-1);
269 y1 = rcClamp(y1, 0, h-1);
270
271 // Clip the triangle into all grid cells it touches.
272 float buf[7*3*4];
273 float *in = buf, *inrow = buf+7*3, *p1 = inrow+7*3, *p2 = p1+7*3;
274
275 rcVcopy(&in[0], v0);
276 rcVcopy(&in[1*3], v1);
277 rcVcopy(&in[2*3], v2);
278 int nvrow, nvIn = 3;
279
280 for (int y = y0; y <= y1; ++y)
281 {
282 // Clip polygon to row. Store the remaining polygon as well
283 const float cz = bmin[2] + y*cs;
284 dividePoly(in, nvIn, inrow, &nvrow, p1, &nvIn, cz+cs, 2);
285 rcSwap(in, p1);
286 if (nvrow < 3) continue;
287 if (y < 0) continue;
288 // find the horizontal bounds in the row
289 float minX = inrow[0], maxX = inrow[0];
290 for (int i=1; i<nvrow; ++i)
291 {
292 if (minX > inrow[i*3]) minX = inrow[i*3];
293 if (maxX < inrow[i*3]) maxX = inrow[i*3];
294 }
295 int x0 = (int)((minX - bmin[0])*ics);
296 int x1 = (int)((maxX - bmin[0])*ics);
297 if (x1 < 0 || x0 >= w) {
298 continue;
299 }
300 x0 = rcClamp(x0, -1, w-1);
301 x1 = rcClamp(x1, 0, w-1);
302
303 int nv, nv2 = nvrow;
304
305 for (int x = x0; x <= x1; ++x)
306 {
307 // Clip polygon to column. store the remaining polygon as well
308 const float cx = bmin[0] + x*cs;
309 dividePoly(inrow, nv2, p1, &nv, p2, &nv2, cx+cs, 0);
310 rcSwap(inrow, p2);
311 if (nv < 3) continue;
312 if (x < 0) continue;
313 // Calculate min and max of the span.
314 float smin = p1[1], smax = p1[1];
315 for (int i = 1; i < nv; ++i)
316 {
317 smin = rcMin(smin, p1[i*3+1]);
318 smax = rcMax(smax, p1[i*3+1]);
319 }
320 smin -= bmin[1];
321 smax -= bmin[1];
322 // Skip the span if it is outside the heightfield bbox
323 if (smax < 0.0f) continue;
324 if (smin > by) continue;
325 // Clamp the span to the heightfield bbox.
326 if (smin < 0.0f) smin = 0;
327 if (smax > by) smax = by;
328
329 // Snap the span to the heightfield height grid.
330 unsigned short ismin = (unsigned short)rcClamp((int)floorf(smin * ich), 0, RC_SPAN_MAX_HEIGHT);
331 unsigned short ismax = (unsigned short)rcClamp((int)ceilf(smax * ich), (int)ismin+1, RC_SPAN_MAX_HEIGHT);
332
333 if (!addSpan(hf, x, y, ismin, ismax, area, flagMergeThr))
334 return false;
335 }
336 }
337
338 return true;
339}

365/// @par
366///
367/// Spans will only be added for triangles that overlap the heightfield grid.
368///
369/// @see rcHeightfield
370bool rcRasterizeTriangles(rcContext* ctx, const float* verts, const int /*nv*/,
371 const int* tris, const unsigned char* areas, const int nt,
372 rcHeightfield& solid, const int flagMergeThr)
373{
374 rcAssert(ctx);
375
376 rcScopedTimer timer(ctx, RC_TIMER_RASTERIZE_TRIANGLES);
377
378 const float ics = 1.0f/solid.cs;
379 const float ich = 1.0f/solid.ch;
380 // Rasterize triangles.
381 for (int i = 0; i < nt; ++i)
382 {
383 const float* v0 = &verts[tris[i*3+0]*3];
384 const float* v1 = &verts[tris[i*3+1]*3];
385 const float* v2 = &verts[tris[i*3+2]*3];
386 // Rasterize.
387 if (!rasterizeTri(v0, v1, v2, areas[i], solid, solid.bmin, solid.bmax, solid.cs, ics, ich, flagMergeThr))
388 {
389 ctx->log(RC_LOG_ERROR, "rcRasterizeTriangles: Out of memory.");
390 return false;
391 }
392 }
393
394 return true;
395}
