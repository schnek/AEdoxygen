59/// @class rcContext
60/// @par
61///
62/// This class does not provide logging or timer functionality on its
63/// own. Both must be provided by a concrete implementation
64/// by overriding the protected member functions. Also, this class does not
65/// provide an interface for extracting log messages. (Only adding them.)
66/// So concrete implementations must provide one.
67///
68/// If no logging or timers are required, just pass an instance of this
69/// class through the Recast build process.
70///
71
72/// @par
73///
74/// Example:
75/// @code
76/// // Where ctx is an instance of rcContext and filepath is a char array.
77/// ctx->log(RC_LOG_ERROR, "buildTiledNavigation: Could not load '%s'", filepath);
78/// @endcode
79void rcContext::log(const rcLogCategory category, const char* format, ...)
80{
81 if (!m_logEnabled)
82 return;
83 static const int MSG_SIZE = 512;
84 char msg[MSG_SIZE];
85 va_list ap;
86 va_start(ap, format);
87 int len = vsnprintf(msg, MSG_SIZE, format, ap);
88 if (len >= MSG_SIZE)
89 {
90 len = MSG_SIZE-1;
91 msg[MSG_SIZE-1] = '\0';
92 }
93 va_end(ap);
94 doLog(category, msg, len);
95}

322static void calcTriNormal(const float* v0, const float* v1, const float* v2, float* norm)
323{
324 float e0[3], e1[3];
325 rcVsub(e0, v1, v0);
326 rcVsub(e1, v2, v0);
327 rcVcross(norm, e0, e1);
328 rcVnormalize(norm);
329}

331/// @par
332///
333/// Only sets the area id's for the walkable triangles. Does not alter the
334/// area id's for unwalkable triangles.
335///
336/// See the #rcConfig documentation for more information on the configuration parameters.
337///
338/// @see rcHeightfield, rcClearUnwalkableTriangles, rcRasterizeTriangles
339void rcMarkWalkableTriangles(rcContext* ctx, const float walkableSlopeAngle,
340 const float* verts, int nv,
341 const int* tris, int nt,
342 unsigned char* areas)
343{
344 rcIgnoreUnused(ctx);
345 rcIgnoreUnused(nv);
346
347 const float walkableThr = cosf(walkableSlopeAngle/180.0f*RC_PI);
348
349 float norm[3];
350
351 for (int i = 0; i < nt; ++i)
352 {
353 const int* tri = &tris[i*3];
354 calcTriNormal(&verts[tri[0]*3], &verts[tri[1]*3], &verts[tri[2]*3], norm);
355 // Check if the face is walkable.
356 if (norm[1] > walkableThr)
357 areas[i] = RC_WALKABLE_AREA;
358 }
359}

390int rcGetHeightFieldSpanCount(rcContext* ctx, rcHeightfield& hf)
391{
392 rcIgnoreUnused(ctx);
393
394 const int w = hf.width;
395 const int h = hf.height;
396 int spanCount = 0;
397 for (int y = 0; y < h; ++y)
398 {
399 for (int x = 0; x < w; ++x)
400 {
401 for (rcSpan* s = hf.spans[x + y*w]; s; s = s->next)
402 {
403 if (s->area != RC_NULL_AREA)
404 spanCount++;
405 }
406 }
407 }
408 return spanCount;
409}

411/// @par
412///
413/// This is just the beginning of the process of fully building a compact heightfield.
414/// Various filters may be applied, then the distance field and regions built.
415/// E.g: #rcBuildDistanceField and #rcBuildRegions
416///
417/// See the #rcConfig documentation for more information on the configuration parameters.
418///
419/// @see rcAllocCompactHeightfield, rcHeightfield, rcCompactHeightfield, rcConfig
420bool rcBuildCompactHeightfield(rcContext* ctx, const int walkableHeight, const int walkableClimb,
421 rcHeightfield& hf, rcCompactHeightfield& chf)
422{
423 rcAssert(ctx);
424
425 rcScopedTimer timer(ctx, RC_TIMER_BUILD_COMPACTHEIGHTFIELD);
426
427 const int w = hf.width;
428 const int h = hf.height;
429 const int spanCount = rcGetHeightFieldSpanCount(ctx, hf);
430
431 // Fill in header.
432 chf.width = w;
433 chf.height = h;
434 chf.spanCount = spanCount;
435 chf.walkableHeight = walkableHeight;
436 chf.walkableClimb = walkableClimb;
437 chf.maxRegions = 0;
438 rcVcopy(chf.bmin, hf.bmin);
439 rcVcopy(chf.bmax, hf.bmax);
440 chf.bmax[1] += walkableHeight*hf.ch;
441 chf.cs = hf.cs;
442 chf.ch = hf.ch;
443 chf.cells = (rcCompactCell*)rcAlloc(sizeof(rcCompactCell)*w*h, RC_ALLOC_PERM);
444 if (!chf.cells)
445 {
446 ctx->log(RC_LOG_ERROR, "rcBuildCompactHeightfield: Out of memory 'chf.cells' (%d)", w*h);
447 return false;
448 }
449 memset(chf.cells, 0, sizeof(rcCompactCell)*w*h);
450 chf.spans = (rcCompactSpan*)rcAlloc(sizeof(rcCompactSpan)*spanCount, RC_ALLOC_PERM);
451 if (!chf.spans)
452 {
453 ctx->log(RC_LOG_ERROR, "rcBuildCompactHeightfield: Out of memory 'chf.spans' (%d)", spanCount);
454 return false;
455 }
456 memset(chf.spans, 0, sizeof(rcCompactSpan)*spanCount);
457 chf.areas = (unsigned char*)rcAlloc(sizeof(unsigned char)*spanCount, RC_ALLOC_PERM);
458 if (!chf.areas)
459 {
460 ctx->log(RC_LOG_ERROR, "rcBuildCompactHeightfield: Out of memory 'chf.areas' (%d)", spanCount);
461 return false;
462 }
463 memset(chf.areas, RC_NULL_AREA, sizeof(unsigned char)*spanCount);
464
465 const int MAX_HEIGHT = 0xffff;
466
467 // Fill in cells and spans.
468 int idx = 0;
469 for (int y = 0; y < h; ++y)
470 {
471 for (int x = 0; x < w; ++x)
472 {
473 const rcSpan* s = hf.spans[x + y*w];
474 // If there are no spans at this cell, just leave the data to index=0, count=0.
475 if (!s) continue;
476 rcCompactCell& c = chf.cells[x+y*w];
477 c.index = idx;
478 c.count = 0;
479 while (s)
480 {
481 if (s->area != RC_NULL_AREA)
482 {
483 const int bot = (int)s->smax;
484 const int top = s->next ? (int)s->next->smin : MAX_HEIGHT;
485 chf.spans[idx].y = (unsigned short)rcClamp(bot, 0, 0xffff);
486 chf.spans[idx].h = (unsigned char)rcClamp(top - bot, 0, 0xff);
487 chf.areas[idx] = s->area;
488 idx++;
489 c.count++;
490 }
491 s = s->next;
492 }
493 }
494 }
495
496 // Find neighbour connections.
497 const int MAX_LAYERS = RC_NOT_CONNECTED-1;
498 int tooHighNeighbour = 0;
499 for (int y = 0; y < h; ++y)
500 {
501 for (int x = 0; x < w; ++x)
502 {
503 const rcCompactCell& c = chf.cells[x+y*w];
504 for (int i = (int)c.index, ni = (int)(c.index+c.count); i < ni; ++i)
505 {
506 rcCompactSpan& s = chf.spans[i];
507
508 for (int dir = 0; dir < 4; ++dir)
509 {
510 rcSetCon(s, dir, RC_NOT_CONNECTED);
511 const int nx = x + rcGetDirOffsetX(dir);
512 const int ny = y + rcGetDirOffsetY(dir);
513 // First check that the neighbour cell is in bounds.
514 if (nx < 0 || ny < 0 || nx >= w || ny >= h)
515 continue;
516
517 // Iterate over all neighbour spans and check if any of the is
518 // accessible from current cell.
519 const rcCompactCell& nc = chf.cells[nx+ny*w];
520 for (int k = (int)nc.index, nk = (int)(nc.index+nc.count); k < nk; ++k)
521 {
522 const rcCompactSpan& ns = chf.spans[k];
523 const int bot = rcMax(s.y, ns.y);
524 const int top = rcMin(s.y+s.h, ns.y+ns.h);
525
526 // Check that the gap between the spans is walkable,
527 // and that the climb height between the gaps is not too high.
528 if ((top - bot) >= walkableHeight && rcAbs((int)ns.y - (int)s.y) <= walkableClimb)
529 {
530 // Mark direction as walkable.
531 const int lidx = k - (int)nc.index;
532 if (lidx < 0 || lidx > MAX_LAYERS)
533 {
534 tooHighNeighbour = rcMax(tooHighNeighbour, lidx);
535 continue;
536 }
537 rcSetCon(s, dir, lidx);
538 break;
539 }
540 }
541
542 }
543 }
544 }
545 }
546
547 if (tooHighNeighbour > MAX_LAYERS)
548 {
549 ctx->log(RC_LOG_ERROR, "rcBuildCompactHeightfield: Heightfield has too many layers %d (max: %d)",
550 tooHighNeighbour, MAX_LAYERS);
551 }
552
553 return true;
554}
