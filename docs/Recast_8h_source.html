101/// Provides an interface for optional logging and performance tracking of the Recast
102/// build process.
103/// @ingroup recast
104class rcContext
105{
106public:
107 /// Contructor.
108 /// @param[in] state TRUE if the logging and performance timers should be enabled. [Default: true]
109 inline rcContext(bool state = true) : m_logEnabled(state), m_timerEnabled(state) {}
110 virtual ~rcContext() {}
111
112 /// Enables or disables logging.
113 /// @param[in] state TRUE if logging should be enabled.
114 inline void enableLog(bool state) { m_logEnabled = state; }
115
116 /// Clears all log entries.
117 inline void resetLog() { if (m_logEnabled) doResetLog(); }
118
119 /// Logs a message.
120 /// @param[in] category The category of the message.
121 /// @param[in] format The message.
122 void log(const rcLogCategory category, const char* format, ...);
123
124 /// Enables or disables the performance timers.
125 /// @param[in] state TRUE if timers should be enabled.
126 inline void enableTimer(bool state) { m_timerEnabled = state; }
127
128 /// Clears all peformance timers. (Resets all to unused.)
129 inline void resetTimers() { if (m_timerEnabled) doResetTimers(); }
130
131 /// Starts the specified performance timer.
132 /// @param label The category of the timer.
133 inline void startTimer(const rcTimerLabel label) { if (m_timerEnabled) doStartTimer(label); }
134
135 /// Stops the specified performance timer.
136 /// @param label The category of the timer.
137 inline void stopTimer(const rcTimerLabel label) { if (m_timerEnabled) doStopTimer(label); }
138
139 /// Returns the total accumulated time of the specified performance timer.
140 /// @param label The category of the timer.
141 /// @return The accumulated time of the timer, or -1 if timers are disabled or the timer has never been started.
142 inline int getAccumulatedTime(const rcTimerLabel label) const { return m_timerEnabled ? doGetAccumulatedTime(label) : -1; }
143
144protected:
145 /// Clears all log entries.
146 virtual void doResetLog();
147
148 /// Logs a message.
149 /// @param[in] category The category of the message.
150 /// @param[in] msg The formatted message.
151 /// @param[in] len The length of the formatted message.
152 virtual void doLog(const rcLogCategory category, const char* msg, const int len) { rcIgnoreUnused(category); rcIgnoreUnused(msg); rcIgnoreUnused(len); }
153
154 /// Clears all timers. (Resets all to unused.)
155 virtual void doResetTimers() {}
156
157 /// Starts the specified performance timer.
158 /// @param[in] label The category of timer.
159 virtual void doStartTimer(const rcTimerLabel label) { rcIgnoreUnused(label); }
160
161 /// Stops the specified performance timer.
162 /// @param[in] label The category of the timer.
163 virtual void doStopTimer(const rcTimerLabel label) { rcIgnoreUnused(label); }
164
165 /// Returns the total accumulated time of the specified performance timer.
166 /// @param[in] label The category of the timer.
167 /// @return The accumulated time of the timer, or -1 if timers are disabled or the timer has never been started.
168 virtual int doGetAccumulatedTime(const rcTimerLabel label) const { rcIgnoreUnused(label); return -1; }
169
170 /// True if logging is enabled.
171 bool m_logEnabled;
172
173 /// True if the performance timers are enabled.
174 bool m_timerEnabled;
175};
