124 template<typename RayCallback>
125 void intersectRay(const G3D::Ray& ray, RayCallback& intersectCallback, float& max_dist, const G3D::Vector3& end)
126 {
127 Cell cell = Cell::ComputeCell(ray.origin().x, ray.origin().y);
128 if (!cell.isValid())
129 return;
130
131 Cell last_cell = Cell::ComputeCell(end.x, end.y);
132
133 if (cell == last_cell)
134 {
135 if (Node* node = nodes[cell.x][cell.y].get())
136 node->intersectRay(ray, intersectCallback, max_dist);
137 return;
138 }
139
140 float voxel = (float)CELL_SIZE;
141 float kx_inv = ray.invDirection().x, bx = ray.origin().x;
142 float ky_inv = ray.invDirection().y, by = ray.origin().y;
143
144 int stepX, stepY;
145 float tMaxX, tMaxY;
146 if (kx_inv >= 0)
147 {
148 stepX = 1;
149 float x_border = (cell.x+1) * voxel;
150 tMaxX = (x_border - bx) * kx_inv;
151 }
152 else
153 {
154 stepX = -1;
155 float x_border = (cell.x-1) * voxel;
156 tMaxX = (x_border - bx) * kx_inv;
157 }
158
159 if (ky_inv >= 0)
160 {
161 stepY = 1;
162 float y_border = (cell.y+1) * voxel;
163 tMaxY = (y_border - by) * ky_inv;
164 }
165 else
166 {
167 stepY = -1;
168 float y_border = (cell.y-1) * voxel;
169 tMaxY = (y_border - by) * ky_inv;
170 }
171
172 //int Cycles = std::max((int)ceilf(max_dist/tMaxX),(int)ceilf(max_dist/tMaxY));
173 //int i = 0;
174
175 float tDeltaX = voxel * std::fabs(kx_inv);
176 float tDeltaY = voxel * std::fabs(ky_inv);
177 do
178 {
179 if (Node* node = nodes[cell.x][cell.y].get())
180 {
181 //float enterdist = max_dist;
182 node->intersectRay(ray, intersectCallback, max_dist);
183 }
184 if (cell == last_cell)
185 break;
186 if (tMaxX < tMaxY)
187 {
188 tMaxX += tDeltaX;
189 cell.x += stepX;
190 }
191 else
192 {
193 tMaxY += tDeltaY;
194 cell.y += stepY;
195 }
196 //++i;
197 } while (cell.isValid());
198 }
