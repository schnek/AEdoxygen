59bool Shiv(uint8_t /*effectIndex*/, Spell* pSpell)
60{
61 Unit* pTarget = pSpell->getUnitTarget();
62 if (!pSpell->getPlayerCaster() || !pTarget) return true;
63
64 pSpell->getPlayerCaster()->castSpell(pTarget->getGuid(), 5940, true);
65
66 Item* it = pSpell->getPlayerCaster()->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_OFFHAND);
67 if (!it)
68 return true;
69
70 EnchantmentInstance* ench = it->getEnchantment(TEMP_ENCHANTMENT_SLOT);
71 if (ench)
72 {
73 WDB::Structures::SpellItemEnchantmentEntry const* Entry = ench->Enchantment;
74 for (uint8_t c = 0; c < 3; c++)
75 {
76 if (Entry->type[c] && Entry->spell[c])
77 {
78 SpellInfo const* sp = sSpellMgr.getSpellInfo(Entry->spell[c]);
79 if (!sp) return true;
80
81 if (sp->custom_c_is_flags & SPELL_FLAG_IS_POISON)
82 {
83 pSpell->getPlayerCaster()->castSpell(pTarget->getGuid(), Entry->spell[c], true);
84 }
85 }
86 }
87 }
88 return true;
89}

112bool CloakOfShadows(uint8_t /*effectIndex*/, Spell* s)
113{
114 Unit* unitTarget = s->getUnitTarget();
115
116 if (!unitTarget || !unitTarget->isAlive())
117 return false;
118
119 for (uint16_t j = AuraSlots::NEGATIVE_SLOT_START; j < AuraSlots::NEGATIVE_SLOT_END; ++j)
120 {
121 auto pAura = unitTarget->getAuraWithAuraSlot(j);
122 if (pAura != NULL
123 && !(pAura->getSpellInfo()->getAttributes() & ATTRIBUTES_IGNORE_INVULNERABILITY)
124 && pAura->getSpellInfo()->getFirstSchoolFromSchoolMask() != 0
125 )
126 pAura->removeAura();
127 }
128
129 return true;
130}

132bool CheatDeath(uint8_t /*effectIndex*/, Aura* a, bool apply)
133{
134 Unit* u_target = a->getOwner();
135 Player* p_target = NULL;
136
137 if (u_target->isPlayer())
138 p_target = static_cast<Player*>(u_target);
139
140 if (p_target != NULL)
141 {
142 int32_t m = (int32_t)(8.0f * p_target->calcRating(CR_HIT_TAKEN_MELEE));
143 if (m > 90)
144 m = 90;
145
146 float val;
147
148 if (apply)
149 {
150 val = -m / 100.0f;
151 }
152 else
153 {
154 val = m / 100.0f;
155 }
156
157 for (uint32_t x = 0; x < 7; x++)
158 p_target->m_damageTakenPctMod[x] += val;
159 }
160
161 return true;
162}

217bool KillingSpreePeriodicDummy(uint8_t /*effectIndex*/, Aura* a, bool /*apply*/)
218{
219 Unit* m_target = a->getOwner();
220 if (!m_target->isPlayer())
221 return true;
222
223 Player* p_target = static_cast<Player*>(m_target);
224
225 //Find targets around aura's target in range of 10 yards.
226 //It can hit same target multiple times.
227 for (const auto& itr : p_target->getInRangeObjectsSet())
228 {
229 if (itr)
230 {
231 //Get the range of 10 yards from Effect 1
232 float r = static_cast<float>(a->getSpellInfo()->getEffectRadiusIndex(1));
233
234 //Get initial position of aura target (caster)
235 LocationVector source = p_target->GetPosition();
236
237 //Calc distance to the target
238 float dist = itr->CalcDistance(source);
239
240 //Radius check
241 if (dist <= r)
242 {
243 //Avoid targeting anything that is not unit and not alive
244 if (!itr->isCreatureOrPlayer() || !static_cast<Unit*>(itr)->isAlive())
245 continue;
246
247 uint64_t spellTarget = itr->getGuid();
248 //SPELL_EFFECT_TELEPORT
249 p_target->castSpell(spellTarget, 57840, true);
250 //SPELL_EFFECT_NORMALIZED_WEAPON_DMG and triggering 57842 with the same effect
251 p_target->castSpell(spellTarget, 57841, true);
252 }
253
254 }
255 }
256 return true;
257}
