22#if VERSION_STRING >= WotLK
23class CutToTheChase : public SpellScript
24{
25public:
26 SpellScriptCheckDummy onAuraDummyEffect(Aura* aur, AuraEffectModifier* /*aurEff*/, bool apply) override
27 {
28 if (apply)
29 {
30 // Make it proc self on Eviscerate and Envenom
31 const uint32_t procFlags[3] = { 0x800000, 0x8, 0 };
32 aur->getOwner()->addProcTriggerSpell(aur->getSpellInfo(), aur->getCasterGuid(), aur, procFlags);
33 }
34 else
35 {
36 aur->getOwner()->removeProcTriggerSpell(aur->getSpellId(), aur->getCasterGuid());
37 }
38
39 return SpellScriptCheckDummy::DUMMY_OK;
40 }
41
42 bool canProc(SpellProc* spellProc, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, DamageInfo /*damageInfo*/) override
43 {
44 // Find Slice and Dice aura
45 for (const auto& aurEff : spellProc->getProcOwner()->getAuraEffectList(SPELL_AURA_MOD_HASTE))
46 {
47 auto* const aur = aurEff->getAura();
48 if (aur->getCasterGuid() != spellProc->getCasterGuid())
49 continue;
50
51 const auto spinfo = aur->getSpellInfo();
52 if (spinfo->getSpellFamilyName() != SPELLFAMILY_ROGUE)
53 continue;
54
55 // Slice and Dice
56 if (spinfo->getSpellFamilyFlags(0) == 0x40000 &&
57 spinfo->getSpellFamilyFlags(1) == 0 &&
58 spinfo->getSpellFamilyFlags(2) == 0)
59 {
60 sliceAura = aur;
61 break;
62 }
63 }
64
65 return sliceAura != nullptr;
66 }
67
68 SpellScriptExecuteState onDoProcEffect(SpellProc* /*spellProc*/, Unit* victim, SpellInfo const* /*castingSpell*/, DamageInfo /*damageInfo*/) override
69 {
70 if (victim == nullptr)
71 return SpellScriptExecuteState::EXECUTE_PREVENT;
72
73 // Recalculate duration for aura per 5 combo points
74 uint32_t maxDuration = 0;
75 const auto durEntry = sSpellDurationStore.lookupEntry(sliceAura->getSpellInfo()->getDurationIndex());
76 if (durEntry != nullptr)
77 maxDuration = durEntry->Duration3;
78
79 // Override the original duration and refresh aura
80 sliceAura->setNewMaxDuration(maxDuration);
81
82 sliceAura = nullptr;
83 return SpellScriptExecuteState::EXECUTE_PREVENT;
84 }
85
86private:
87 Aura* sliceAura = nullptr;
88};
89#endif
