258//
259// Note: Implementation of this function in WorldEdit.exe and storm.dll
260// incorrectly treats the character as signed, which leads to the
261// a buffer underflow if the character in the file name >= 0x80:
262// The following steps happen when *pbKey == 0xBF and dwHashType == 0x0000
263// (calculating hash index)
264//
265// 1) Result of AsciiToUpperTable_Slash[*pbKey++] is sign-extended to 0xffffffbf
266// 2) The "ch" is added to dwHashType (0xffffffbf + 0x0000 => 0xffffffbf)
267// 3) The result is used as index to the StormBuffer table,
268// thus dereferences a random value BEFORE the begin of StormBuffer.
269//
270// As result, MPQs containing files with non-ANSI characters will not work between
271// various game versions and localizations. Even WorldEdit, after importing a file
272// with Korean characters in the name, cannot open the file back.
273//
274DWORD HashString(const char * szFileName, DWORD dwHashType)
275{
276 LPBYTE pbKey = (BYTE *)szFileName;
277 DWORD dwSeed1 = 0x7FED7FED;
278 DWORD dwSeed2 = 0xEEEEEEEE;
279 DWORD ch;
280
281 while(*pbKey != 0)
282 {
283 // Convert the input character to uppercase
284 // Convert slash (0x2F) to backslash (0x5C)
285 ch = AsciiToUpperTable[*pbKey++];
286
287 dwSeed1 = StormBuffer[dwHashType + ch] ^ (dwSeed1 + dwSeed2);
288 dwSeed2 = ch + dwSeed1 + dwSeed2 + (dwSeed2 << 5) + 3;
289 }
290
291 return dwSeed1;
292}

368// Calculates a Jenkin's Encrypting and decrypting MPQ file data
369
370ULONGLONG HashStringJenkins(const char * szFileName)
371{
372 LPBYTE pbFileName = (LPBYTE)szFileName;
373 char szNameBuff[0x108];
374 size_t nLength = 0;
375 unsigned int primary_hash = 1;
376 unsigned int secondary_hash = 2;
377
378 // Normalize the file name - convert to uppercase, and convert "/" to "\\".
379 if(pbFileName != NULL)
380 {
381 char * szNamePtr = szNameBuff;
382 char * szNameEnd = szNamePtr + sizeof(szNameBuff);
383
384 // Normalize the file name. Doesn't have to be zero terminated for hashing
385 while(szNamePtr < szNameEnd && pbFileName[0] != 0)
386 *szNamePtr++ = (char)AsciiToLowerTable[*pbFileName++];
387 nLength = szNamePtr - szNameBuff;
388 }
389
390 // Thanks Quantam for finding out what the algorithm is.
391 // I am really getting old for reversing large chunks of assembly
392 // that does hashing :-)
393 hashlittle2(szNameBuff, nLength, &secondary_hash, &primary_hash);
394
395 // Combine those 2 together
396 return ((ULONGLONG)primary_hash << 0x20) | (ULONGLONG)secondary_hash;
397}

416static DWORD EncryptUInt32Unaligned(LPDWORD DataPointer, DWORD i, DWORD dwXorKey)
417{
418 LPBYTE pbDataPointer = (LPBYTE)(DataPointer + i);
419 LPBYTE pbXorKey = (LPBYTE)(&dwXorKey);
420 DWORD dwValue32;
421
422 // Retrieve the value
423 dwValue32 = ((DWORD)pbDataPointer[0] << 0x00) |

425 ((DWORD)pbDataPointer[2] << 0x10) |
426 ((DWORD)pbDataPointer[3] << 0x18);

430 pbDataPointer[1] = (pbDataPointer[1] ^ pbXorKey[1]);
431 pbDataPointer[2] = (pbDataPointer[2] ^ pbXorKey[2]);
432 pbDataPointer[3] = (pbDataPointer[3] ^ pbXorKey[3]);
433 return dwValue32;
434}

465 dwKey2 += StormBuffer[MPQ_HASH_KEY2_MIX + (dwKey1 & 0xFF)];
466
467 // The data are unaligned. Make sure we don't cause data misalignment error
468 dwValue32 = EncryptUInt32Unaligned(DataPointer, i, (dwKey1 + dwKey2));
469
470 dwKey1 = ((~dwKey1 << 0x15) + 0x11111111) | (dwKey1 >> 0x0B);
471 dwKey2 = dwValue32 + dwKey2 + (dwKey2 << 5) + 3;
472 }
473 }
474}
475
476static DWORD DecryptUInt32Unaligned(LPDWORD DataPointer, DWORD i, DWORD dwXorKey)
477{
478 LPBYTE pbDataPointer = (LPBYTE)(DataPointer + i);
479 LPBYTE pbXorKey = (LPBYTE)(&dwXorKey);
480
481 // Perform unaligned XOR
482 pbDataPointer[0] = (pbDataPointer[0] ^ pbXorKey[0]);
483 pbDataPointer[1] = (pbDataPointer[1] ^ pbXorKey[1]);
484 pbDataPointer[2] = (pbDataPointer[2] ^ pbXorKey[2]);
485 pbDataPointer[3] = (pbDataPointer[3] ^ pbXorKey[3]);
486
487 // Retrieve the value
488 return ((DWORD)pbDataPointer[0] << 0x00) |
489 ((DWORD)pbDataPointer[1] << 0x08) |
490 ((DWORD)pbDataPointer[2] << 0x10) |
491 ((DWORD)pbDataPointer[3] << 0x18);
492}

494void DecryptMpqBlock(void * pvDataBlock, DWORD dwLength, DWORD dwKey1)
495{
496 LPDWORD DataPointer = (LPDWORD)pvDataBlock;
497 DWORD dwValue32;
498 DWORD dwKey2 = 0xEEEEEEEE;
499
500 // Round to DWORDs
501 dwLength >>= 2;
502
503 // We need different approach on non-aligned buffers
504 if(STORMLIB_DWORD_ALIGNED(DataPointer))
505 {
506 for(DWORD i = 0; i < dwLength; i++)
507 {
508 // Modify the second key
509 dwKey2 += StormBuffer[MPQ_HASH_KEY2_MIX + (dwKey1 & 0xFF)];
510
511 // We can use 32-bit approach, when the buffer is aligned
512 DataPointer[i] = dwValue32 = DataPointer[i] ^ (dwKey1 + dwKey2);
513
514 dwKey1 = ((~dwKey1 << 0x15) + 0x11111111) | (dwKey1 >> 0x0B);
515 dwKey2 = dwValue32 + dwKey2 + (dwKey2 << 5) + 3;
516 }
517 }
518 else
519 {
520 for(DWORD i = 0; i < dwLength; i++)
521 {
522 // Modify the second key
523 dwKey2 += StormBuffer[MPQ_HASH_KEY2_MIX + (dwKey1 & 0xFF)];
524
525 // The data are unaligned. Make sure we don't cause data misalignment error
526 dwValue32 = DecryptUInt32Unaligned(DataPointer, i, (dwKey1 + dwKey2));
527
528 dwKey1 = ((~dwKey1 << 0x15) + 0x11111111) | (dwKey1 >> 0x0B);
529 dwKey2 = dwValue32 + dwKey2 + (dwKey2 << 5) + 3;
530 }
531 }
532}

534/**
535 * Functions tries to get file decryption key. This comes from these facts
536 *
537 * - We know the decrypted value of the first DWORD in the encrypted data
538 * - We know the decrypted value of the second DWORD (at least aproximately)
539 * - There is only 256 variants of how the second key is modified
540 *
541 * The first iteration of dwKey1 and dwKey2 is this:
542 *
543 * dwKey2 = 0xEEEEEEEE + StormBuffer[MPQ_HASH_KEY2_MIX + (dwKey1 & 0xFF)]
544 * dwDecrypted0 = DataBlock[0] ^ (dwKey1 + dwKey2);
545 *
546 * This means:
547 *
548 * (dwKey1 + dwKey2) = DataBlock[0] ^ dwDecrypted0;
549 *
550 */
551
552DWORD DetectFileKeyBySectorSize(LPDWORD EncryptedData, DWORD dwSectorSize, DWORD dwDecrypted0)
553{
554 // We must have at least 2 DWORDs there to be able to decrypt something
555 if(dwSectorSize >= 0x08)
556 {
557 // Also try subsequent three values. This is because the value of the sector offset[0]
558 // could be higher than the total size of the sector table.
559 // Example MPQ: MPQ_2021_v1_CantExtractCHK.scx
560 for(DWORD dwDecrypted4 = dwDecrypted0 + 4; dwDecrypted0 < dwDecrypted4; dwDecrypted0++)
561 {
562 DWORD dwDecrypted1Max = dwSectorSize + dwDecrypted0;
563 DWORD dwKey1PlusKey2;
564 DWORD DataBlock[2];
565
566 // Get the value of the combined encryption key
567 dwKey1PlusKey2 = (EncryptedData[0] ^ dwDecrypted0) - 0xEEEEEEEE;
568
569 // Try all 256 combinations of dwKey1
570 for(DWORD i = 0; i < 0x100; i++)
571 {
572 DWORD dwSaveKey1;
573 DWORD dwKey1 = dwKey1PlusKey2 - StormBuffer[MPQ_HASH_KEY2_MIX + i];
574 DWORD dwKey2 = 0xEEEEEEEE;
575
576 // Modify the second key and decrypt the first DWORD
577 dwKey2 += StormBuffer[MPQ_HASH_KEY2_MIX + (dwKey1 & 0xFF)];
578 DataBlock[0] = EncryptedData[0] ^ (dwKey1 + dwKey2);
579
580 // Did we obtain the same value like dwDecrypted0?
581 if(DataBlock[0] == dwDecrypted0)
582 {
583 // Save this key value. Increment by one because
584 // we are decrypting sector offset table
585 dwSaveKey1 = dwKey1 + 1;
586
587 // Rotate both keys
588 dwKey1 = ((~dwKey1 << 0x15) + 0x11111111) | (dwKey1 >> 0x0B);
589 dwKey2 = DataBlock[0] + dwKey2 + (dwKey2 << 5) + 3;
590
591 // Modify the second key again and decrypt the second DWORD
592 dwKey2 += StormBuffer[MPQ_HASH_KEY2_MIX + (dwKey1 & 0xFF)];
593 DataBlock[1] = EncryptedData[1] ^ (dwKey1 + dwKey2);
594
595 // Now compare the results
596 if(DataBlock[1] <= dwDecrypted1Max)
597 return dwSaveKey1;
598 }
599 }
600 }
601 }
602
603 // Key not found
604 return 0;
605}

735//-----------------------------------------------------------------------------
736// Hash table and block table manipulation
737
738// Attempts to search a free hash entry, or an entry whose names and locale matches
739TMPQHash * FindFreeHashEntry(TMPQArchive * ha, DWORD dwStartIndex, DWORD dwName1, DWORD dwName2, LCID lcFileLocale)
740{
741 TMPQHash * pDeletedEntry = NULL; // If a deleted entry was found in the continuous hash range
742 TMPQHash * pFreeEntry = NULL; // If a free entry was found in the continuous hash range
743 DWORD dwHashIndexMask = HASH_INDEX_MASK(ha);
744 DWORD dwIndex;
745 USHORT Locale = SFILE_LOCALE(lcFileLocale);
746
747 // Set the initial index
748 dwStartIndex = dwIndex = (dwStartIndex & dwHashIndexMask);
749
750 // Search the hash table and return the found entries in the following priority:
751 // 1) <MATCHING_ENTRY>
752 // 2) <DELETED-ENTRY>
753 // 3) <FREE-ENTRY>
754 // 4) NULL
755 for(;;)
756 {
757 TMPQHash * pHash = ha->pHashTable + dwIndex;
758
759 // If we found a matching entry, return that one
760 if(pHash->dwName1 == dwName1 && pHash->dwName2 == dwName2 && pHash->Locale == Locale)
761 return pHash;
762
763 // If we found a deleted entry, remember it but keep searching
764 if(pHash->dwBlockIndex == HASH_ENTRY_DELETED && pDeletedEntry == NULL)
765 pDeletedEntry = pHash;
766
767 // If we found a free entry, we need to stop searching
768 if(pHash->dwBlockIndex == HASH_ENTRY_FREE)
769 {
770 pFreeEntry = pHash;
771 break;
772 }
773
774 // Move to the next hash entry.
775 // If we reached the starting entry, it's failure.
776 dwIndex = (dwIndex + 1) & dwHashIndexMask;
777 if(dwIndex == dwStartIndex)
778 break;
779 }
780
781 // If we found a deleted entry, return that one preferentially
782 return (pDeletedEntry != NULL) ? pDeletedEntry : pFreeEntry;
783}

785// Retrieves the first hash entry for the given file.
786// Every locale version of a file has its own hash entry
787TMPQHash * GetFirstHashEntry(TMPQArchive * ha, const char * szFileName)
788{
789 DWORD dwHashIndexMask = HASH_INDEX_MASK(ha);
790 DWORD dwStartIndex = ha->pfnHashString(szFileName, MPQ_HASH_TABLE_INDEX);
791 DWORD dwName1 = ha->pfnHashString(szFileName, MPQ_HASH_NAME_A);
792 DWORD dwName2 = ha->pfnHashString(szFileName, MPQ_HASH_NAME_B);
793 DWORD dwIndex;
794
795 // Set the initial index
796 dwStartIndex = dwIndex = (dwStartIndex & dwHashIndexMask);
797
798 // Search the hash table
799 for(;;)
800 {
801 TMPQHash * pHash = ha->pHashTable + dwIndex;
802
803 // If the entry matches, we found it.
804 if(pHash->dwName1 == dwName1 && pHash->dwName2 == dwName2 && MPQ_BLOCK_INDEX(pHash) < ha->dwFileTableSize)
805 return pHash;
806
807 // If that hash entry is a free entry, it means we haven't found the file
808 if(pHash->dwBlockIndex == HASH_ENTRY_FREE)
809 return NULL;
810
811 // Move to the next hash entry. Stop searching
812 // if we got reached the original hash entry
813 dwIndex = (dwIndex + 1) & dwHashIndexMask;
814 if(dwIndex == dwStartIndex)
815 return NULL;
816 }
817}

819TMPQHash * GetNextHashEntry(TMPQArchive * ha, TMPQHash * pFirstHash, TMPQHash * pHash)
820{
821 DWORD dwHashIndexMask = HASH_INDEX_MASK(ha);
822 DWORD dwStartIndex = (DWORD)(pFirstHash - ha->pHashTable);
823 DWORD dwName1 = pHash->dwName1;
824 DWORD dwName2 = pHash->dwName2;
825 DWORD dwIndex = (DWORD)(pHash - ha->pHashTable);
826
827 // Now go for any next entry that follows the pHash,
828 // until either free hash entry was found, or the start entry was reached
829 for(;;)
830 {
831 // Move to the next hash entry. Stop searching
832 // if we got reached the original hash entry
833 dwIndex = (dwIndex + 1) & dwHashIndexMask;
834 if(dwIndex == dwStartIndex)
835 return NULL;
836 pHash = ha->pHashTable + dwIndex;
837
838 // If the entry matches, we found it.
839 if(pHash->dwName1 == dwName1 && pHash->dwName2 == dwName2 && MPQ_BLOCK_INDEX(pHash) < ha->dwFileTableSize)
840 return pHash;
841
842 // If that hash entry is a free entry, it means we haven't found the file
843 if(pHash->dwBlockIndex == HASH_ENTRY_FREE)
844 return NULL;
845 }
846}

875// Finds a free space in the MPQ where to store next data
876// The free space begins beyond the file that is stored at the fuhrtest
877// position in the MPQ. (listfile), (attributes) and (signature) are ignored,
878// unless the MPQ is being flushed.
879ULONGLONG FindFreeMpqSpace(TMPQArchive * ha)
880{
881 TMPQHeader * pHeader = ha->pHeader;
882 TFileEntry * pFileTableEnd = ha->pFileTable + ha->dwFileTableSize;
883 TFileEntry * pFileEntry;
884 ULONGLONG FreeSpacePos = ha->pHeader->dwHeaderSize;
885 DWORD dwChunkCount;
886
887 // Parse the entire block table
888 for(pFileEntry = ha->pFileTable; pFileEntry < pFileTableEnd; pFileEntry++)
889 {
890 // Only take existing files with nonzero size
891 if((pFileEntry->dwFlags & MPQ_FILE_EXISTS) && (pFileEntry->dwCmpSize != 0))
892 {
893 // If we are not saving MPQ tables, ignore internal MPQ files
894 if((ha->dwFlags & MPQ_FLAG_SAVING_TABLES) == 0 && IsInternalMpqFileName(pFileEntry->szFileName))
895 continue;
896
897 // If the end of the file is bigger than current MPQ table pos, update it
898 if((pFileEntry->ByteOffset + pFileEntry->dwCmpSize) > FreeSpacePos)
899 {
900 // Get the end of the file data
901 FreeSpacePos = pFileEntry->ByteOffset + pFileEntry->dwCmpSize;
902
903 // Add the MD5 chunks, if present
904 if(pHeader->dwRawChunkSize != 0 && pFileEntry->dwCmpSize != 0)
905 {
906 dwChunkCount = ((pFileEntry->dwCmpSize - 1) / pHeader->dwRawChunkSize) + 1;
907 FreeSpacePos += dwChunkCount * MD5_DIGEST_SIZE;
908 }
909 }
910 }
911 }
912
913 // Give the free space position to the caller
914 return FreeSpacePos;
915}

987// Loads a table from MPQ.
988// Can be used for hash table, block table, sector offset table or sector checksum table
989void * LoadMpqTable(
990 TMPQArchive * ha,
991 ULONGLONG ByteOffset,
992 LPBYTE pbTableHash,
993 DWORD dwCompressedSize,
994 DWORD dwTableSize,
995 DWORD dwKey,
996 DWORD * PtrRealTableSize)
997{
998 ULONGLONG FileSize = 0;
999 LPBYTE pbCompressed = NULL;
1000 LPBYTE pbMpqTable;
1001 LPBYTE pbToRead;
1002 DWORD dwBytesToRead = dwCompressedSize;
1003 DWORD dwErrCode = ERROR_SUCCESS;
1004
1005 // Allocate the MPQ table
1006 pbMpqTable = pbToRead = STORM_ALLOC(BYTE, dwTableSize);
1007 if(pbMpqTable != NULL)
1008 {
1009 // Check if the MPQ table is encrypted
1010 if(dwCompressedSize < dwTableSize)
1011 {
1012 // Allocate temporary buffer for holding compressed data
1013 pbCompressed = pbToRead = STORM_ALLOC(BYTE, dwCompressedSize);
1014 if(pbCompressed == NULL)
1015 {
1016 STORM_FREE(pbMpqTable);
1017 return NULL;
1018 }
1019 }
1020
1021 // Get the file offset from which we will read the table
1022 // Note: According to Storm.dll from Warcraft III (version 2002),
1023 // if the hash table position is 0xFFFFFFFF, no SetFilePointer call is done
1024 // and the table is loaded from the current file offset
1025 if(ByteOffset == SFILE_INVALID_POS)
1026 FileStream_GetPos(ha->pStream, &ByteOffset);
1027
1028 // On archives v 1.0, hash table and block table can go beyond EOF.
1029 // Storm.dll reads as much as possible, then fills the missing part with zeros.
1030 // Abused by Spazzler map protector which sets hash table size to 0x00100000
1031 // Abused by NP_Protect in MPQs v4 as well
1032 if(ha->pHeader->wFormatVersion == MPQ_FORMAT_VERSION_1)
1033 {
1034 // Cut the table size
1035 FileStream_GetSize(ha->pStream, &FileSize);
1036 if((ByteOffset + dwBytesToRead) > FileSize)
1037 {
1038 // Fill the extra data with zeros
1039 dwBytesToRead = (DWORD)(FileSize - ByteOffset);
1040 memset(pbMpqTable + dwBytesToRead, 0, (dwTableSize - dwBytesToRead));
1041 }
1042 }
1043
1044 // Give the caller information that the table was cut
1045 if(PtrRealTableSize != NULL)
1046 {
1047 PtrRealTableSize[0] = dwBytesToRead;
1048 }
1049
1050 // If everything succeeded, read the raw table from the MPQ
1051 if(FileStream_Read(ha->pStream, &ByteOffset, pbToRead, dwBytesToRead))
1052 {
1053 // Verify the MD5 of the table, if present
1054 if(!VerifyDataBlockHash(pbToRead, dwBytesToRead, pbTableHash))
1055 {
1056 dwErrCode = ERROR_FILE_CORRUPT;
1057 }
1058 }
1059 else
1060 {
1061 dwErrCode = GetLastError();
1062 }
1063
1064 if(dwErrCode == ERROR_SUCCESS)
1065 {
1066 // First of all, decrypt the table
1067 if(dwKey != 0)
1068 {
1069 BSWAP_ARRAY32_UNSIGNED(pbToRead, dwCompressedSize);
1070 DecryptMpqBlock(pbToRead, dwCompressedSize, dwKey);
1071 BSWAP_ARRAY32_UNSIGNED(pbToRead, dwCompressedSize);
1072 }
1073
1074 // If the table is compressed, decompress it
1075 if(dwCompressedSize < dwTableSize)
1076 {
1077 int cbOutBuffer = (int)dwTableSize;
1078 int cbInBuffer = (int)dwCompressedSize;
1079
1080 if(!SCompDecompress2(pbMpqTable, &cbOutBuffer, pbCompressed, cbInBuffer))
1081 dwErrCode = GetLastError();
1082 }
1083
1084 // Make sure that the table is properly byte-swapped
1085 BSWAP_ARRAY32_UNSIGNED(pbMpqTable, dwTableSize);
1086 }
1087
1088 // If read failed, free the table and return
1089 if(dwErrCode != ERROR_SUCCESS)
1090 {
1091 STORM_FREE(pbMpqTable);
1092 pbMpqTable = NULL;
1093 }
1094
1095 // Free the compression buffer, if any
1096 if(pbCompressed != NULL)
1097 STORM_FREE(pbCompressed);
1098 }
1099
1100 // Return the MPQ table
1101 return pbMpqTable;
1102}

1128// Allocates sector buffer and sector offset table
1129DWORD AllocateSectorBuffer(TMPQFile * hf)

1132
1133 // Caller of AllocateSectorBuffer must ensure these

1153{

1185
1186 // Verify the size of the patch header
1187 // If it's not default size, we have to reload them
1188 if(hf->pPatchInfo->dwLength > dwLength)
1189 {
1190 // Free the patch info
1191 dwLength = hf->pPatchInfo->dwLength;
1192 STORM_FREE(hf->pPatchInfo);
1193 hf->pPatchInfo = NULL;
1194
1195 // If the length is out of all possible ranges, fail the operation
1196 if(dwLength > 0x400)
1197 return ERROR_FILE_CORRUPT;
1198 goto __AllocateAndLoadPatchInfo;
1199 }
1200
1201 // Patch file data size according to the patch header
1202 hf->dwDataSize = hf->pPatchInfo->dwDataSize;
1203 }
1204 else
1205 {
1206 memset(hf->pPatchInfo, 0, dwLength);
1207 }
1208
1209 // Save the final length to the patch header
1210 hf->pPatchInfo->dwLength = dwLength;
1211 hf->pPatchInfo->dwFlags = 0x80000000;
1212 return ERROR_SUCCESS;
1213}

1215// Allocates sector offset table
1216DWORD AllocateSectorOffsets(TMPQFile * hf, bool bLoadFromFile)
1217{
1218 TMPQArchive * ha = hf->ha;
1219 TFileEntry * pFileEntry = hf->pFileEntry;
1220 DWORD dwSectorOffsLen;
1221 bool bSectorOffsetTableCorrupt = false;
1222
1223 // Caller of AllocateSectorOffsets must ensure these
1224 assert(hf->SectorOffsets == NULL);
1225 assert(hf->pFileEntry != NULL);
1226 assert(hf->dwDataSize != 0);
1227 assert(hf->ha != NULL);
1228
1229 // If the file is stored as single unit, just set number of sectors to 1
1230 if(pFileEntry->dwFlags & MPQ_FILE_SINGLE_UNIT)
1231 {
1232 hf->dwSectorCount = 1;
1233 return ERROR_SUCCESS;
1234 }
1235
1236 // Calculate the number of data sectors
1237 // Note that this doesn't work if the file size is zero
1238 hf->dwSectorCount = ((hf->dwDataSize - 1) / hf->dwSectorSize) + 1;
1239
1240 // Calculate the number of file sectors
1241 dwSectorOffsLen = (hf->dwSectorCount + 1) * sizeof(DWORD);
1242
1243 // If MPQ_FILE_SECTOR_CRC flag is set, there will either be extra DWORD
1244 // or an array of MD5's. Either way, we read at least 4 bytes more
1245 // in order to save additional read from the file.
1246 if(pFileEntry->dwFlags & MPQ_FILE_SECTOR_CRC)
1247 dwSectorOffsLen += sizeof(DWORD);
1248
1249 // Only allocate and load the table if the file is compressed
1250 if(pFileEntry->dwFlags & MPQ_FILE_COMPRESS_MASK)
1251 {
1252 __LoadSectorOffsets:
1253
1254 // Allocate the sector offset table
1255 hf->SectorOffsets = STORM_ALLOC(DWORD, (dwSectorOffsLen / sizeof(DWORD)));
1256 if(hf->SectorOffsets == NULL)
1257 return ERROR_NOT_ENOUGH_MEMORY;
1258
1259 // Only read from the file if we are supposed to do so
1260 if(bLoadFromFile)
1261 {
1262 ULONGLONG RawFilePos = hf->RawFilePos;
1263
1264 // Append the length of the patch info, if any
1265 if(hf->pPatchInfo != NULL)
1266 {
1267 if((RawFilePos + hf->pPatchInfo->dwLength) < RawFilePos)
1268 return ERROR_FILE_CORRUPT;
1269 RawFilePos += hf->pPatchInfo->dwLength;
1270 }
1271
1272 // Load the sector offsets from the file
1273 if(!FileStream_Read(ha->pStream, &RawFilePos, hf->SectorOffsets, dwSectorOffsLen))
1274 {
1275 // Free the sector offsets
1276 STORM_FREE(hf->SectorOffsets);
1277 hf->SectorOffsets = NULL;
1278 return GetLastError();
1279 }
1280
1281 // Swap the sector positions
1282 BSWAP_ARRAY32_UNSIGNED(hf->SectorOffsets, dwSectorOffsLen);
1283
1284 // Decrypt loaded sector positions if necessary
1285 if(pFileEntry->dwFlags & MPQ_FILE_ENCRYPTED)
1286 {
1287 // If we don't know the file key, try to find it.
1288 if(hf->dwFileKey == 0)
1289 {
1290 hf->dwFileKey = DetectFileKeyBySectorSize(hf->SectorOffsets, ha->dwSectorSize, dwSectorOffsLen);
1291 if(hf->dwFileKey == 0)
1292 {
1293 STORM_FREE(hf->SectorOffsets);
1294 hf->SectorOffsets = NULL;
1295 return ERROR_UNKNOWN_FILE_KEY;
1296 }
1297 }
1298
1299 // Decrypt sector positions
1300 DecryptMpqBlock(hf->SectorOffsets, dwSectorOffsLen, hf->dwFileKey - 1);
1301 }
1302
1303 //
1304 // Validate the sector offset table
1305 //
1306 // Note: Some MPQ protectors put the actual file data before the sector offset table.
1307 // In this case, the sector offsets are negative (> 0x80000000).
1308 //
1309
1310 for(DWORD i = 0; i < hf->dwSectorCount; i++)
1311 {
1312 DWORD dwSectorOffset1 = hf->SectorOffsets[i+1];
1313 DWORD dwSectorOffset0 = hf->SectorOffsets[i];
1314
1315 // Every following sector offset must be bigger than the previous one
1316 if(dwSectorOffset1 < dwSectorOffset0)
1317 {
1318 bSectorOffsetTableCorrupt = true;
1319 break;
1320 }
1321
1322 // The sector size must not be bigger than compressed file size
1323 // Edit: Yes, but apparently, in original Storm.dll, the compressed
1324 // size is not checked anywhere. However, we need to do this check
1325 // in order to sector offset table malformed by MPQ protectors
1326 if((dwSectorOffset1 - dwSectorOffset0) > ha->dwSectorSize)
1327 {
1328 bSectorOffsetTableCorrupt = true;
1329 break;
1330 }
1331 }
1332
1333 // If data corruption detected, free the sector offset table
1334 if(bSectorOffsetTableCorrupt)
1335 {
1336 STORM_FREE(hf->SectorOffsets);
1337 hf->SectorOffsets = NULL;
1338 return ERROR_FILE_CORRUPT;
1339 }
1340
1341 //
1342 // There may be various extra DWORDs loaded after the sector offset table.
1343 // They are mostly empty on WoW release MPQs, but on MPQs from PTR,
1344 // they contain random non-zero data. Their meaning is unknown.
1345 //
1346 // These extra values are, however, included in the dwCmpSize in the file
1347 // table. We cannot ignore them, because compacting archive would fail
1348 //
1349
1350 // Clear the lower 2 bits in order to make sure that the value is aligned to 4 bytes
1351 if((hf->SectorOffsets[0] & 0xFFFFFFFC) > dwSectorOffsLen)
1352 {
1353 // MPQ protectors put some ridiculous values there. We must limit the extra bytes
1354 if(hf->SectorOffsets[0] > (dwSectorOffsLen + 0x400))
1355 return ERROR_FILE_CORRUPT;
1356
1357 // Free the old sector offset table
1358 dwSectorOffsLen = hf->SectorOffsets[0];
1359 STORM_FREE(hf->SectorOffsets);
1360 goto __LoadSectorOffsets;
1361 }
1362 }
1363 else
1364 {
1365 memset(hf->SectorOffsets, 0, dwSectorOffsLen);
1366 hf->SectorOffsets[0] = dwSectorOffsLen;
1367 }
1368 }
1369
1370 return ERROR_SUCCESS;
1371}

1373DWORD AllocateSectorChecksums(TMPQFile * hf, bool bLoadFromFile)
1374{
1375 TMPQArchive * ha = hf->ha;
1376 TFileEntry * pFileEntry = hf->pFileEntry;
1377 ULONGLONG RawFilePos;
1378 DWORD dwCompressedSize = 0;
1379 DWORD dwExpectedSize;
1380 DWORD dwCrcOffset; // Offset of the CRC table, relative to file offset in the MPQ
1381 DWORD dwCrcSize;
1382
1383 // Caller of AllocateSectorChecksums must ensure these
1384 assert(hf->SectorChksums == NULL);
1385 assert(hf->SectorOffsets != NULL);
1386 assert(hf->pFileEntry != NULL);
1387 assert(hf->ha != NULL);
1388
1389 // Single unit files don't have sector checksums
1390 if(pFileEntry->dwFlags & MPQ_FILE_SINGLE_UNIT)
1391 return ERROR_SUCCESS;
1392
1393 // Caller must ensure that we are only called when we have sector checksums
1394 assert(pFileEntry->dwFlags & MPQ_FILE_SECTOR_CRC);
1395
1396 //
1397 // Older MPQs store an array of CRC32's after
1398 // the raw file data in the MPQ.
1399 //
1400 // In newer MPQs, the (since Cataclysm BETA) the (attributes) file
1401 // contains additional 32-bit values beyond the sector table.
1402 // Their number depends on size of the (attributes), but their
1403 // meaning is unknown. They are usually zeroed in retail game files,
1404 // but contain some sort of checksum in BETA MPQs
1405 //
1406
1407 // Does the size of the file table match with the CRC32-based checksums?
1408 dwExpectedSize = (hf->dwSectorCount + 2) * sizeof(DWORD);
1409 if(hf->SectorOffsets[0] != 0 && hf->SectorOffsets[0] == dwExpectedSize)
1410 {
1411 // If we are not loading from the MPQ file, we just allocate the sector table
1412 // In that case, do not check any sizes
1413 if(bLoadFromFile == false)
1414 {
1415 hf->SectorChksums = STORM_ALLOC(DWORD, hf->dwSectorCount);
1416 if(hf->SectorChksums == NULL)
1417 return ERROR_NOT_ENOUGH_MEMORY;
1418
1419 // Fill the checksum table with zeros
1420 memset(hf->SectorChksums, 0, hf->dwSectorCount * sizeof(DWORD));
1421 return ERROR_SUCCESS;
1422 }
1423 else
1424 {
1425 // Is there valid size of the sector checksums?
1426 if(hf->SectorOffsets[hf->dwSectorCount + 1] >= hf->SectorOffsets[hf->dwSectorCount])
1427 dwCompressedSize = hf->SectorOffsets[hf->dwSectorCount + 1] - hf->SectorOffsets[hf->dwSectorCount];
1428
1429 // Ignore cases when the length is too small or too big.
1430 if(dwCompressedSize < sizeof(DWORD) || dwCompressedSize > hf->dwSectorSize)
1431 return ERROR_SUCCESS;
1432
1433 // Calculate offset of the CRC table
1434 dwCrcSize = hf->dwSectorCount * sizeof(DWORD);
1435 dwCrcOffset = hf->SectorOffsets[hf->dwSectorCount];
1436 RawFilePos = CalculateRawSectorOffset(hf, dwCrcOffset);
1437
1438 // Now read the table from the MPQ
1439 hf->SectorChksums = (DWORD *)LoadMpqTable(ha, RawFilePos, NULL, dwCompressedSize, dwCrcSize, 0, NULL);
1440 if(hf->SectorChksums == NULL)
1441 return ERROR_NOT_ENOUGH_MEMORY;
1442 }
1443 }
1444
1445 // If the size doesn't match, we ignore sector checksums
1446// assert(false);
1447 return ERROR_SUCCESS;
1448}

1525 dwCrcSize = hf->dwSectorCount * sizeof(DWORD);
1526

1533 BSWAP_ARRAY32_UNSIGNED(hf->SectorChksums, dwCrcSize);
1534

1536 SCompCompress(pbCompressed, &nOutSize, hf->SectorChksums, (int)dwCrcSize, MPQ_COMPRESSION_ZLIB, 0, 0);

1540 RawFilePos = hf->RawFilePos + hf->SectorOffsets[hf->dwSectorCount];
1541 if(hf->pPatchInfo != NULL)
1542 RawFilePos += hf->pPatchInfo->dwLength;
1543 if(!FileStream_Write(ha->pStream, &RawFilePos, pbCompressed, dwCompressedSize))
1544 dwErrCode = GetLastError();

1547 // are going to be freed when this is done.
1548// BSWAP_ARRAY32_UNSIGNED(hf->SectorChksums, dwCrcSize);
1549
1550 // Store the sector CRCs
1551 hf->SectorOffsets[hf->dwSectorCount + 1] = hf->SectorOffsets[hf->dwSectorCount] + dwCompressedSize;
1552 pFileEntry->dwCmpSize += dwCompressedSize;
1553 STORM_FREE(pbCompressed);
1554 return dwErrCode;

1557DWORD WriteMemDataMD5(
1558 TFileStream * pStream,
1559 ULONGLONG RawDataOffs,
1560 void * pvRawData,
1561 DWORD dwRawDataSize,
1562 DWORD dwChunkSize,
1563 LPDWORD pcbTotalSize)
1564{
1565 unsigned char * md5_array;
1566 unsigned char * md5;
1567 LPBYTE pbRawData = (LPBYTE)pvRawData;
1568 DWORD dwBytesRemaining = dwRawDataSize;
1569 DWORD dwMd5ArraySize = 0;
1570 DWORD dwErrCode = ERROR_SUCCESS;
1571

1573 md5_array = md5 = AllocateMd5Buffer(dwRawDataSize, dwChunkSize, &dwMd5ArraySize);
1574 if(md5_array == NULL)
1575 return ERROR_NOT_ENOUGH_MEMORY;

1577 // For every file chunk, calculate MD5
1578 while(dwBytesRemaining != 0)
1579 {
1580 // Get the remaining number of bytes to read

1582

1585 md5 += MD5_DIGEST_SIZE;

1595 dwErrCode = GetLastError();

1597 // Give the caller the size of the MD5 array
1598 if(pcbTotalSize != NULL)
1599 *pcbTotalSize = dwRawDataSize + dwMd5ArraySize;
1600
1601 // Free buffers and exit
1602 STORM_FREE(md5_array);
1603 return dwErrCode;
1604}

1755{
