88 printf("DWORD dwBitIndex_Unknown = %08X\n", pBetTable->dwBitIndex_Unknown);
89 printf("DWORD dwBitCount_FilePos = %08X\n", pBetTable->dwBitCount_FilePos);
90 printf("DWORD dwBitCount_FileSize = %08X\n", pBetTable->dwBitCount_FileSize);
91 printf("DWORD dwBitCount_CmpSize = %08X\n", pBetTable->dwBitCount_CmpSize);
92 printf("DWORD dwBitCount_FlagIndex = %08X\n", pBetTable->dwBitCount_FlagIndex);
93 printf("DWORD dwBitCount_Unknown = %08X\n", pBetTable->dwBitCount_Unknown);
94 printf("DWORD dwBitTotal_NameHash2 = %08X\n", pBetTable->dwBitTotal_NameHash2);
95 printf("DWORD dwBitExtra_NameHash2 = %08X\n", pBetTable->dwBitExtra_NameHash2);
96 printf("DWORD dwBitCount_NameHash2 = %08X\n", pBetTable->dwBitCount_NameHash2);
97 printf("DWORD dwEntryCount = %08X\n", pBetTable->dwEntryCount);
98 printf("DWORD dwFlagCount = %08X\n", pBetTable->dwFlagCount);
99 printf("-----------------------------------------------\n\n");
100
101 printf("== HET & Bet Table ======================================================================\n\n");
102 printf("HetIdx HetHash BetIdx BetHash ByteOffset FileSize CmpSize FlgIdx Flags \n");
103 printf("------ ------- ------ ---------------- ---------------- -------- -------- ------ --------\n");
104 for(i = 0; i < pHetTable->dwTotalCount; i++)
105 {
106 ULONGLONG ByteOffset = 0;
107 ULONGLONG BetHash = 0;
108 DWORD dwFileSize = 0;
109 DWORD dwCmpSize = 0;
110 DWORD dwFlagIndex = 0;
111 DWORD dwFlags = 0;
112 DWORD dwBetIndex = 0;
113
114 GetMPQBits(pHetTable->pBetIndexes, i * pHetTable->dwIndexSizeTotal,
115 pHetTable->dwIndexSize,
116 &dwBetIndex, 4);
117
118 if(dwBetIndex < pHetTable->dwTotalCount)
119 {
120 DWORD dwEntryIndex = pBetTable->dwTableEntrySize * dwBetIndex;
121
122 GetMPQBits(pBetTable->pNameHashes, dwBetIndex * pBetTable->dwBitTotal_NameHash2,
123 pBetTable->dwBitCount_NameHash2,
124 &BetHash, 8);
125
126 GetMPQBits(pBetTable->pFileTable, dwEntryIndex + pBetTable->dwBitIndex_FilePos,
127 pBetTable->dwBitCount_FilePos,
128 &ByteOffset, 8);
129
130 GetMPQBits(pBetTable->pFileTable, dwEntryIndex + pBetTable->dwBitIndex_FileSize,
131 pBetTable->dwBitCount_FileSize,
132 &dwFileSize, 4);
133
134 GetMPQBits(pBetTable->pFileTable, dwEntryIndex + pBetTable->dwBitIndex_CmpSize,
135 pBetTable->dwBitCount_CmpSize,
136 &dwCmpSize, 4);
137
138 GetMPQBits(pBetTable->pFileTable, dwEntryIndex + pBetTable->dwBitIndex_FlagIndex,
139 pBetTable->dwBitCount_FlagIndex,
140 &dwFlagIndex, 4);
141
142 dwFlags = pBetTable->pFileFlags[dwFlagIndex];
143 }
144
145 printf(" %04X %02lX %04X %016llX %016llX %08X %08X %04X %08X\n", i,
146 pHetTable->pNameHashes[i],
147 dwBetIndex,
148 BetHash,
149 ByteOffset,
150 dwFileSize,
151 dwCmpSize,
152 dwFlagIndex,
153 dwFlags);
154 }
155 printf("-----------------------------------------------------------------------------------------\n");
156}

158void DumpFileTable(TFileEntry * pFileTable, DWORD dwFileTableSize)
159{
160 DWORD i;
161
162 if(pFileTable == NULL || dwFileTableSize == 0)
163 return;
164
165 printf("== File Table =================================\n");

167 {
168 printf("[%04u] %08X-%08X %08X-%08X %08X-%08X 0x%08X 0x%08X 0x%08X %s\n", i,
169 (DWORD)(pFileTable->FileNameHash >> 0x20),
170 (DWORD)(pFileTable->FileNameHash & 0xFFFFFFFF),
171 (DWORD)(pFileTable->ByteOffset >> 0x20),
172 (DWORD)(pFileTable->ByteOffset & 0xFFFFFFFF),
173 (DWORD)(pFileTable->FileTime >> 0x20),
174 (DWORD)(pFileTable->FileTime & 0xFFFFFFFF),
175 pFileTable->dwFileSize,
176 pFileTable->dwCmpSize,
177 pFileTable->dwFlags,
178 pFileTable->szFileName != NULL ? pFileTable->szFileName : "");
179 }
180 printf("-----------------------------------------------\n\n");
181}
