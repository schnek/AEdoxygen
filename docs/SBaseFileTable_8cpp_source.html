97void TMPQBits::GetBits(
98 unsigned int nBitPosition,
99 unsigned int nBitLength,
100 void * pvBuffer,
101 int nResultByteSize)
102{
103 unsigned char * pbBuffer = (unsigned char *)pvBuffer;
104 unsigned int nBytePosition0 = (nBitPosition / 8);
105 unsigned int nBytePosition1 = nBytePosition0 + 1;
106 unsigned int nByteLength = (nBitLength / 8);
107 unsigned int nBitOffset = (nBitPosition & 0x07);
108 unsigned char BitBuffer;
109
110 // Keep compiler happy for platforms where nResultByteSize is not used
111 nResultByteSize = nResultByteSize;
112
113#ifdef _DEBUG
114 // Check if the target is properly zeroed
115 for(int i = 0; i < nResultByteSize; i++)
116 assert(pbBuffer[i] == 0);
117#endif
118
119#ifndef STORMLIB_LITTLE_ENDIAN
120 // Adjust the buffer pointer for big endian platforms
121 pbBuffer += (nResultByteSize - 1);
122#endif
123
124 // Copy whole bytes, if any
125 while(nByteLength > 0)
126 {
127 // Is the current position in the Elements byte-aligned?
128 if(nBitOffset != 0)
129 {
130 BitBuffer = (unsigned char)((Elements[nBytePosition0] >> nBitOffset) | (Elements[nBytePosition1] << (0x08 - nBitOffset)));
131 }
132 else
133 {
134 BitBuffer = Elements[nBytePosition0];
135 }
136
137#ifdef STORMLIB_LITTLE_ENDIAN
138 *pbBuffer++ = BitBuffer;
139#else
140 *pbBuffer-- = BitBuffer;
141#endif
142
143 // Move byte positions and lengths
144 nBytePosition1++;
145 nBytePosition0++;
146 nByteLength--;
147 }
148
149 // Get the rest of the bits
150 nBitLength = (nBitLength & 0x07);
151 if(nBitLength != 0)
152 {
153 *pbBuffer = (unsigned char)(Elements[nBytePosition0] >> nBitOffset);
154
155 if(nBitLength > (8 - nBitOffset))
156 *pbBuffer = (unsigned char)((Elements[nBytePosition1] << (8 - nBitOffset)) | (Elements[nBytePosition0] >> nBitOffset));
157
158 *pbBuffer &= (0x01 << nBitLength) - 1;
159 }
160}

162void TMPQBits::SetBits(
163 unsigned int nBitPosition,
164 unsigned int nBitLength,
165 void * pvBuffer,
166 int nResultByteSize)
167{
168 unsigned char * pbBuffer = (unsigned char *)pvBuffer;
169 unsigned int nBytePosition = (nBitPosition / 8);
170 unsigned int nBitOffset = (nBitPosition & 0x07);
171 unsigned short BitBuffer = 0;
172 unsigned short AndMask = 0;
173 unsigned short OneByte = 0;
174
175 // Keep compiler happy for platforms where nResultByteSize is not used
176 nResultByteSize = nResultByteSize;
177
178#ifndef STORMLIB_LITTLE_ENDIAN
179 // Adjust the buffer pointer for big endian platforms
180 pbBuffer += (nResultByteSize - 1);
181#endif
182
183 // Copy whole bytes, if any
184 while(nBitLength > 8)
185 {
186 // Reload the bit buffer
187#ifdef STORMLIB_LITTLE_ENDIAN
188 OneByte = *pbBuffer++;
189#else
190 OneByte = *pbBuffer--;
191#endif
192 // Update the BitBuffer and AndMask for the bit array
193 BitBuffer = (BitBuffer >> 0x08) | (OneByte << nBitOffset);
194 AndMask = (AndMask >> 0x08) | (0x00FF << nBitOffset);
195
196 // Update the byte in the array
197 Elements[nBytePosition] = (BYTE)((Elements[nBytePosition] & ~AndMask) | BitBuffer);
198
199 // Move byte positions and lengths
200 nBytePosition++;
201 nBitLength -= 0x08;
202 }
203
204 if(nBitLength != 0)
205 {
206 // Reload the bit buffer
207 OneByte = *pbBuffer;
208
209 // Update the AND mask for the last bit
210 BitBuffer = (BitBuffer >> 0x08) | (OneByte << nBitOffset);
211 AndMask = (AndMask >> 0x08) | (SetBitsMask[nBitLength] << nBitOffset);
212
213 // Update the byte in the array
214 Elements[nBytePosition] = (BYTE)((Elements[nBytePosition] & ~AndMask) | BitBuffer);
215
216 // Update the next byte, if needed
217 if(AndMask & 0xFF00)
218 {
219 nBytePosition++;
220 BitBuffer >>= 0x08;
221 AndMask >>= 0x08;
222
223 Elements[nBytePosition] = (BYTE)((Elements[nBytePosition] & ~AndMask) | BitBuffer);
224 }
225 }
226}

273static ULONGLONG DetermineArchiveSize_V1(
274 TMPQArchive * ha,
275 TMPQHeader * pHeader,
276 ULONGLONG MpqOffset,
277 ULONGLONG FileSize)
278{
279 ULONGLONG ByteOffset;
280 ULONGLONG EndOfMpq = FileSize;
281 DWORD SignatureHeader = 0;
282 DWORD dwArchiveSize32;
283
284 // This could only be called for MPQs version 1.0
285 assert(pHeader->wFormatVersion == MPQ_FORMAT_VERSION_1);
286
287 // Check if we can rely on the archive size in the header
288 if(pHeader->dwBlockTablePos < pHeader->dwArchiveSize)
289 {
290 // The block table cannot be compressed, so the sizes must match
291 if((pHeader->dwArchiveSize - pHeader->dwBlockTablePos) == (pHeader->dwBlockTableSize * sizeof(TMPQBlock)))
292 return pHeader->dwArchiveSize;
293
294 // If the archive size in the header is less than real file size
295 dwArchiveSize32 = (DWORD)(FileSize - MpqOffset);
296 if(pHeader->dwArchiveSize == dwArchiveSize32)
297 return pHeader->dwArchiveSize;
298 }
299
300 // Check if there is a signature header
301 if((EndOfMpq - MpqOffset) > (MPQ_STRONG_SIGNATURE_SIZE + 4))
302 {
303 ByteOffset = EndOfMpq - MPQ_STRONG_SIGNATURE_SIZE - 4;
304 if(FileStream_Read(ha->pStream, &ByteOffset, &SignatureHeader, sizeof(DWORD)))
305 {
306 if(BSWAP_INT32_UNSIGNED(SignatureHeader) == MPQ_STRONG_SIGNATURE_ID)
307 EndOfMpq = EndOfMpq - MPQ_STRONG_SIGNATURE_SIZE - 4;
308 }
309 }
310
311 // Return the returned archive size
312 return (EndOfMpq - MpqOffset);
313}

315static ULONGLONG DetermineArchiveSize_V2(
316 TMPQHeader * pHeader,
317 ULONGLONG MpqOffset,
318 ULONGLONG FileSize)
319{
320 ULONGLONG EndOfMpq = FileSize;
321 DWORD dwArchiveSize32;
322
323 // This could only be called for MPQs version 2.0
324 assert(pHeader->wFormatVersion == MPQ_FORMAT_VERSION_2);
325
326 // Check if we can rely on the archive size in the header

328 {
329 if(pHeader->dwBlockTablePos < pHeader->dwArchiveSize)

332 return pHeader->dwArchiveSize;

336 if(pHeader->dwArchiveSize <= dwArchiveSize32)
337 return pHeader->dwArchiveSize;
338 }
339 }
340
341 // Return the calculated archive size
342 return (EndOfMpq - MpqOffset);

345static ULONGLONG DetermineArchiveSize_V4(
346 TMPQHeader * pHeader,
347 ULONGLONG /* MpqOffset */,
348 ULONGLONG /* FileSize */)
349{
350 ULONGLONG ArchiveSize = 0;
351 ULONGLONG EndOfTable;
352
353 // This could only be called for MPQs version 4
354 assert(pHeader->wFormatVersion == MPQ_FORMAT_VERSION_4);
355
356 // Check position of BET table, if correct
357 if((pHeader->BetTablePos64 >> 0x20) == 0 && (pHeader->BetTableSize64 >> 0x20) == 0)
358 {
359 EndOfTable = pHeader->BetTablePos64 + pHeader->BetTableSize64;
360 if(EndOfTable > ArchiveSize)
361 ArchiveSize = EndOfTable;
362 }
363
364 // Check position of HET table, if correct
365 if((pHeader->HetTablePos64 >> 0x20) == 0 && (pHeader->HetTableSize64 >> 0x20) == 0)
366 {
367 EndOfTable = pHeader->HetTablePos64 + pHeader->HetTableSize64;
368 if(EndOfTable > ArchiveSize)
369 ArchiveSize = EndOfTable;
370 }
371
372 EndOfTable = pHeader->dwHashTablePos + pHeader->dwHashTableSize * sizeof(TMPQHash);
373 if(EndOfTable > ArchiveSize)
374 ArchiveSize = EndOfTable;
375
376 EndOfTable = pHeader->dwBlockTablePos + pHeader->dwBlockTableSize * sizeof(TMPQBlock);
377 if(EndOfTable > ArchiveSize)
378 ArchiveSize = EndOfTable;
379
380 // Return the calculated archive size
381 return ArchiveSize;
382}

398ULONGLONG CalculateRawSectorOffset(
399 TMPQFile * hf,
400 DWORD dwSectorOffset)
401{
402 ULONGLONG RawFilePos;
403
404 // Must be used for files within a MPQ
405 assert(hf->ha != NULL);
406 assert(hf->ha->pHeader != NULL);
407
408 //
409 // Some MPQ protectors place the sector offset table after the actual file data.
410 // Sector offsets in the sector offset table are negative. When added
411 // to MPQ file offset from the block table entry, the result is a correct
412 // position of the file data in the MPQ.
413 //
414 // For MPQs version 1.0, the offset is purely 32-bit
415 //
416
417 RawFilePos = hf->RawFilePos + dwSectorOffset;
418 if(hf->ha->pHeader->wFormatVersion == MPQ_FORMAT_VERSION_1)
419 RawFilePos = (DWORD)hf->ha->MpqPos + (DWORD)hf->pFileEntry->ByteOffset + dwSectorOffset;
420
421 // We also have to add patch header size, if patch header is present
422 if(hf->pPatchInfo != NULL)
423 RawFilePos += hf->pPatchInfo->dwLength;
424
425 // Return the result offset
426 return RawFilePos;
427}

429// This function converts the MPQ header so it always looks like version 4
430DWORD ConvertMpqHeaderToFormat4(
431 TMPQArchive * ha,
432 ULONGLONG ByteOffset,
433 ULONGLONG FileSize,
434 DWORD dwFlags,
435 MTYPE MapType)
436{
437 TMPQHeader * pHeader = (TMPQHeader *)ha->HeaderData;
438 ULONGLONG BlockTablePos64 = 0;
439 ULONGLONG HashTablePos64 = 0;
440 ULONGLONG BlockTableMask = (ULONGLONG)-1;
441 ULONGLONG MaxOffset;
442 USHORT wFormatVersion = BSWAP_INT16_UNSIGNED(pHeader->wFormatVersion);
443 bool bHashBlockOffsetOK = false;
444 bool bHetBetOffsetOK = false;
445 DWORD dwErrCode = ERROR_SUCCESS;
446
447 // If version 1.0 is forced, then the format version is forced to be 1.0
448 // Reason: Storm.dll in Warcraft III ignores format version value
449 if((MapType == MapTypeWarcraft3) || (dwFlags & MPQ_OPEN_FORCE_MPQ_V1))
450 wFormatVersion = MPQ_FORMAT_VERSION_1;
451
452 // Don't accept format 3 for Starcraft II maps
453 if((MapType == MapTypeStarcraft2) && (pHeader->wFormatVersion > MPQ_FORMAT_VERSION_2))
454 wFormatVersion = MPQ_FORMAT_VERSION_4;
455
456 // Format-specific fixes
457 switch(wFormatVersion)
458 {
459 case MPQ_FORMAT_VERSION_1:
460
461 // Check for malformed MPQ header version 1.0
462 BSWAP_TMPQHEADER(pHeader, MPQ_FORMAT_VERSION_1);
463 if(pHeader->wFormatVersion != MPQ_FORMAT_VERSION_1 || pHeader->dwHeaderSize != MPQ_HEADER_SIZE_V1)
464 {
465 pHeader->wFormatVersion = MPQ_FORMAT_VERSION_1;
466 pHeader->dwHeaderSize = MPQ_HEADER_SIZE_V1;
467 ha->dwFlags |= MPQ_FLAG_MALFORMED;
468 }
469
470 //
471 // Note: The value of "dwArchiveSize" member in the MPQ header
472 // is ignored by Storm.dll and can contain garbage value
473 // ("w3xmaster" protector).
474 //
475
476 Label_ArchiveVersion1:
477 if(pHeader->dwBlockTableSize > 1) // Prevent empty MPQs being marked as malformed
478 {
479 if(pHeader->dwHashTablePos <= pHeader->dwHeaderSize || (pHeader->dwHashTablePos & 0x80000000))
480 ha->dwFlags |= MPQ_FLAG_MALFORMED;
481 if(pHeader->dwBlockTablePos <= pHeader->dwHeaderSize || (pHeader->dwBlockTablePos & 0x80000000))
482 ha->dwFlags |= MPQ_FLAG_MALFORMED;
483 }
484
485 // Only low byte of sector size is really used
486 if(pHeader->wSectorSize & 0xFF00)
487 ha->dwFlags |= MPQ_FLAG_MALFORMED;
488 pHeader->wSectorSize = pHeader->wSectorSize & 0xFF;
489
490 // Fill the rest of the header
491 memset((LPBYTE)pHeader + MPQ_HEADER_SIZE_V1, 0, sizeof(TMPQHeader) - MPQ_HEADER_SIZE_V1);
492 pHeader->BlockTableSize64 = pHeader->dwBlockTableSize * sizeof(TMPQBlock);
493 pHeader->HashTableSize64 = pHeader->dwHashTableSize * sizeof(TMPQHash);
494 pHeader->ArchiveSize64 = pHeader->dwArchiveSize;
495
496 // Block table position must be calculated as 32-bit value
497 // Note: BOBA protector puts block table before the MPQ header, so it is negative
498 BlockTablePos64 = (ULONGLONG)((DWORD)ByteOffset + pHeader->dwBlockTablePos);
499 BlockTableMask = 0xFFFFFFF0;
500
501 // Determine the archive size on malformed MPQs
502 if(ha->dwFlags & MPQ_FLAG_MALFORMED)
503 {
504 // Calculate the archive size
505 pHeader->ArchiveSize64 = DetermineArchiveSize_V1(ha, pHeader, ByteOffset, FileSize);
506 pHeader->dwArchiveSize = (DWORD)pHeader->ArchiveSize64;
507 }
508
509 // EWIX_v8_7.w3x: TMPQHeader::dwBlockTableSize = 0x00319601
510 // Size of TFileTable goes to ~200MB, so we artificially cut it
511 if(BlockTablePos64 + (pHeader->dwBlockTableSize * sizeof(TMPQBlock)) > FileSize)
512 {
513 pHeader->dwBlockTableSize = (DWORD)((FileSize - BlockTablePos64) / sizeof(TMPQBlock));
514 pHeader->BlockTableSize64 = pHeader->dwBlockTableSize * sizeof(TMPQBlock);
515 }
516 break;
517
518 case MPQ_FORMAT_VERSION_2:
519
520 // Check for malformed MPQ header version 1.0
521 BSWAP_TMPQHEADER(pHeader, MPQ_FORMAT_VERSION_2);
522 if(pHeader->wFormatVersion != MPQ_FORMAT_VERSION_2 || pHeader->dwHeaderSize != MPQ_HEADER_SIZE_V2)
523 {
524 pHeader->wFormatVersion = MPQ_FORMAT_VERSION_1;
525 pHeader->dwHeaderSize = MPQ_HEADER_SIZE_V1;
526 ha->dwFlags |= MPQ_FLAG_MALFORMED;
527 goto Label_ArchiveVersion1;
528 }
529
530 // Fill the rest of the header with zeros
531 memset((LPBYTE)pHeader + MPQ_HEADER_SIZE_V2, 0, sizeof(TMPQHeader) - MPQ_HEADER_SIZE_V2);
532
533 // Calculate the expected hash table size
534 pHeader->HashTableSize64 = (pHeader->dwHashTableSize * sizeof(TMPQHash));
535 HashTablePos64 = MAKE_OFFSET64(pHeader->wHashTablePosHi, pHeader->dwHashTablePos);
536
537 // Calculate the expected block table size
538 pHeader->BlockTableSize64 = (pHeader->dwBlockTableSize * sizeof(TMPQBlock));
539 BlockTablePos64 = MAKE_OFFSET64(pHeader->wBlockTablePosHi, pHeader->dwBlockTablePos);
540
541 // We require the block table to follow hash table
542 if(BlockTablePos64 >= HashTablePos64)
543 {
544 // HashTableSize64 may be less than TblSize * sizeof(TMPQHash).
545 // That means that the hash table is compressed.
546 pHeader->HashTableSize64 = BlockTablePos64 - HashTablePos64;
547
548 // Calculate the compressed block table size
549 if(pHeader->HiBlockTablePos64 != 0)
550 {
551 // BlockTableSize64 may be less than TblSize * sizeof(TMPQBlock).
552 // That means that the block table is compressed.
553 pHeader->BlockTableSize64 = pHeader->HiBlockTablePos64 - BlockTablePos64;
554 assert(pHeader->BlockTableSize64 <= (pHeader->dwBlockTableSize * sizeof(TMPQBlock)));
555
556 // Determine real archive size
557 pHeader->ArchiveSize64 = DetermineArchiveSize_V2(pHeader, ByteOffset, FileSize);
558
559 // Calculate the size of the hi-block table
560 pHeader->HiBlockTableSize64 = pHeader->ArchiveSize64 - pHeader->HiBlockTablePos64;
561 assert(pHeader->HiBlockTableSize64 == (pHeader->dwBlockTableSize * sizeof(USHORT)));
562 }
563 else
564 {
565 // Determine real archive size
566 pHeader->ArchiveSize64 = DetermineArchiveSize_V2(pHeader, ByteOffset, FileSize);
567
568 // Calculate size of the block table
569 pHeader->BlockTableSize64 = pHeader->ArchiveSize64 - BlockTablePos64;
570 assert(pHeader->BlockTableSize64 <= (pHeader->dwBlockTableSize * sizeof(TMPQBlock)));
571 }
572 }
573 else
574 {
575 pHeader->ArchiveSize64 = pHeader->dwArchiveSize;
576 ha->dwFlags |= MPQ_FLAG_MALFORMED;
577 }
578
579 // Add the MPQ Offset
580 BlockTablePos64 += ByteOffset;
581 break;
582
583 case MPQ_FORMAT_VERSION_3:
584
585 // In MPQ format 3.0, the entire header is optional
586 // and the size of the header can actually be identical
587 // to size of header 2.0
588 BSWAP_TMPQHEADER(pHeader, MPQ_FORMAT_VERSION_3);
589 if(pHeader->dwHeaderSize < MPQ_HEADER_SIZE_V3)
590 {
591 pHeader->ArchiveSize64 = pHeader->dwArchiveSize;
592 pHeader->HetTablePos64 = 0;
593 pHeader->BetTablePos64 = 0;
594 }
595
596 //
597 // We need to calculate the compressed size of each table. We assume the following order:
598 // 1) HET table
599 // 2) BET table
600 // 3) Classic hash table
601 // 4) Classic block table
602 // 5) Hi-block table
603 //
604
605 // Fill the rest of the header with zeros
606 memset((LPBYTE)pHeader + MPQ_HEADER_SIZE_V3, 0, sizeof(TMPQHeader) - MPQ_HEADER_SIZE_V3);
607 BlockTablePos64 = MAKE_OFFSET64(pHeader->wBlockTablePosHi, pHeader->dwBlockTablePos);
608 HashTablePos64 = MAKE_OFFSET64(pHeader->wHashTablePosHi, pHeader->dwHashTablePos);
609 MaxOffset = pHeader->ArchiveSize64;
610
611 // Size of the hi-block table
612 if(pHeader->HiBlockTablePos64)
613 {
614 pHeader->HiBlockTableSize64 = MaxOffset - pHeader->HiBlockTablePos64;
615 MaxOffset = pHeader->HiBlockTablePos64;
616 }
617
618 // Size of the block table
619 if(BlockTablePos64)
620 {
621 pHeader->BlockTableSize64 = MaxOffset - BlockTablePos64;
622 MaxOffset = BlockTablePos64;
623 }
624
625 // Size of the hash table
626 if(HashTablePos64)
627 {
628 pHeader->HashTableSize64 = MaxOffset - HashTablePos64;
629 MaxOffset = HashTablePos64;
630 }
631
632 // Size of the BET table
633 if(pHeader->BetTablePos64)
634 {
635 pHeader->BetTableSize64 = MaxOffset - pHeader->BetTablePos64;
636 MaxOffset = pHeader->BetTablePos64;
637 }
638
639 // Size of the HET table
640 if(pHeader->HetTablePos64)
641 {
642 pHeader->HetTableSize64 = MaxOffset - pHeader->HetTablePos64;
643// MaxOffset = pHeader->HetTablePos64;
644 }
645
646 // Add the MPQ Offset
647 BlockTablePos64 += ByteOffset;
648 break;
649
650 case MPQ_FORMAT_VERSION_4:
651
652 // Verify header MD5. Header MD5 is calculated from the MPQ header since the 'MPQ\x1A'
653 // signature until the position of header MD5 at offset 0xC0
654 BSWAP_TMPQHEADER(pHeader, MPQ_FORMAT_VERSION_4);
655
656 // Apparently, Starcraft II only accepts MPQ headers where the MPQ header hash matches
657 // If MD5 doesn't match, we ignore this offset. We also ignore it if there's no MD5 at all
658 if(!IsValidMD5(pHeader->MD5_MpqHeader))
659 return ERROR_FAKE_MPQ_HEADER;
660 if(!VerifyDataBlockHash(pHeader, MPQ_HEADER_SIZE_V4 - MD5_DIGEST_SIZE, pHeader->MD5_MpqHeader))
661 return ERROR_FAKE_MPQ_HEADER;
662
663 // HiBlockTable must be 0 for archives under 4GB
664 if((pHeader->ArchiveSize64 >> 0x20) == 0 && pHeader->HiBlockTablePos64 != 0)
665 return ERROR_FAKE_MPQ_HEADER;
666
667 // Is the "HET&BET" table tandem OK?
668 bHetBetOffsetOK = VerifyTableTandemPositions(ByteOffset,
669 pHeader->HetTablePos64, pHeader->HetTableSize64,
670 pHeader->BetTablePos64, pHeader->BetTableSize64,
671 FileSize);
672
673 // Is the "Hash&Block" table tandem OK?
674 bHashBlockOffsetOK = VerifyTableTandemPositions(ByteOffset,
675 pHeader->dwHashTablePos, pHeader->HashTableSize64,
676 pHeader->dwBlockTablePos, pHeader->BlockTableSize64,
677 FileSize);

680 if(bHetBetOffsetOK == false && bHashBlockOffsetOK == false)
681 return ERROR_FAKE_MPQ_HEADER;
682
683 // Check for malformed MPQs
684 if(pHeader->wFormatVersion != MPQ_FORMAT_VERSION_4 || (ByteOffset + pHeader->ArchiveSize64) != FileSize || (ByteOffset + pHeader->HiBlockTablePos64) >= FileSize)
685 {
686 pHeader->wFormatVersion = MPQ_FORMAT_VERSION_4;
687 pHeader->dwHeaderSize = MPQ_HEADER_SIZE_V4;
688 ha->dwFlags |= MPQ_FLAG_MALFORMED;
689 }
690
691 // Recalculate archive size
692 if(ha->dwFlags & MPQ_FLAG_MALFORMED)
693 {
694 // Calculate the archive size
695 pHeader->ArchiveSize64 = DetermineArchiveSize_V4(pHeader, ByteOffset, FileSize);
696 pHeader->dwArchiveSize = (DWORD)pHeader->ArchiveSize64;
697 }
698
699 // Calculate the block table position
700 BlockTablePos64 = ByteOffset + MAKE_OFFSET64(pHeader->wBlockTablePosHi, pHeader->dwBlockTablePos);
701 break;
702
703 default:
704
705 // Check if it's a War of the Immortal data file (SQP)
706 // If not, we treat it as malformed MPQ version 1.0
707 if(ConvertSqpHeaderToFormat4(ha, FileSize, dwFlags) != ERROR_SUCCESS)
708 {
709 pHeader->wFormatVersion = MPQ_FORMAT_VERSION_1;
710 pHeader->dwHeaderSize = MPQ_HEADER_SIZE_V1;
711 ha->dwFlags |= MPQ_FLAG_MALFORMED;
712 goto Label_ArchiveVersion1;
713 }
714
715 // Calculate the block table position
716 BlockTablePos64 = ByteOffset + MAKE_OFFSET64(pHeader->wBlockTablePosHi, pHeader->dwBlockTablePos);
717 break;
718 }
719
720 // Handle case when block table is placed before the MPQ header
721 // Used by BOBA protector
722 if(BlockTablePos64 < ByteOffset)
723 ha->dwFlags |= MPQ_FLAG_MALFORMED;
724 return dwErrCode;

1122 ULONGLONG ByteOffset,

1124 DWORD dwSignature,
1125 DWORD dwKey)
1126{

1128 TMPQExtHeader * pExtTable = NULL; // Uncompressed table
1129
1130 // Do nothing if the size is zero
1131 if(ByteOffset != 0 && Size != 0)

1174 STORM_FREE(pExtTable);
1175 pExtTable = NULL;
1176 }
1177 }
1178
1179 // Free the compressed block
1180 STORM_FREE(pCompressed);
1181 }
1182 }
1183 }
1184
1185 // Return the decompressed table to the caller
1186 return pExtTable;

1368TMPQHetTable * CreateHetTable(DWORD dwEntryCount, DWORD dwTotalCount, DWORD dwNameHashBitSize, LPBYTE pbSrcData)
1369{
1370 TMPQHetTable * pHetTable;
1371
1372 pHetTable = STORM_ALLOC(TMPQHetTable, 1);
1373 if(pHetTable != NULL)
1374 {
1375 // Zero the HET table
1376 memset(pHetTable, 0, sizeof(TMPQHetTable));
1377
1378 // Hash sizes less than 0x40 bits are not tested
1379 assert(dwNameHashBitSize == 0x40);
1380
1381 // Calculate masks
1382 pHetTable->AndMask64 = ((dwNameHashBitSize != 0x40) ? ((ULONGLONG)1 << dwNameHashBitSize) : 0) - 1;
1383 pHetTable->OrMask64 = (ULONGLONG)1 << (dwNameHashBitSize - 1);
1384
1385 // If the total count is not entered, use default
1386 if(dwTotalCount == 0)
1387 dwTotalCount = (dwEntryCount * 4) / 3;
1388
1389 // Store the HET table parameters
1390 pHetTable->dwEntryCount = dwEntryCount;
1391 pHetTable->dwTotalCount = dwTotalCount;
1392 pHetTable->dwNameHashBitSize = dwNameHashBitSize;
1393 pHetTable->dwIndexSizeTotal = GetNecessaryBitCount(dwEntryCount);
1394 pHetTable->dwIndexSizeExtra = 0;
1395 pHetTable->dwIndexSize = pHetTable->dwIndexSizeTotal;
1396
1397 // Allocate array of hashes
1398 pHetTable->pNameHashes = STORM_ALLOC(BYTE, dwTotalCount);
1399 if(pHetTable->pNameHashes != NULL)
1400 {
1401 // Make sure the data are initialized
1402 memset(pHetTable->pNameHashes, 0, dwTotalCount);
1403
1404 // Allocate the bit array for file indexes
1405 pHetTable->pBetIndexes = TMPQBits::Create(dwTotalCount * pHetTable->dwIndexSizeTotal, 0xFF);
1406 if(pHetTable->pBetIndexes != NULL)
1407 {
1408 // Initialize the HET table from the source data (if given)
1409 if(pbSrcData != NULL)
1410 {
1411 // Copy the name hashes
1412 memcpy(pHetTable->pNameHashes, pbSrcData, dwTotalCount);
1413
1414 // Copy the file indexes
1415 memcpy(pHetTable->pBetIndexes->Elements, pbSrcData + dwTotalCount, pHetTable->pBetIndexes->NumberOfBytes);
1416 }
1417
1418 // Return the result HET table
1419 return pHetTable;
1420 }
1421
1422 // Free the name hashes
1423 STORM_FREE(pHetTable->pNameHashes);
1424 }
1425
1426 STORM_FREE(pHetTable);
1427 }

1430 return NULL;

1433static DWORD InsertHetEntry(TMPQHetTable * pHetTable, ULONGLONG FileNameHash, DWORD dwFileIndex)
1434{
1435 DWORD StartIndex;
1436 DWORD Index;
1437 BYTE NameHash1;
1438
1439 // Get the start index and the high 8 bits of the name hash
1440 StartIndex = Index = (DWORD)(FileNameHash % pHetTable->dwTotalCount);
1441 NameHash1 = (BYTE)(FileNameHash >> (pHetTable->dwNameHashBitSize - 8));
1442
1443 // Find a place where to put it
1444 for(;;)
1445 {
1446 // Did we find a free HET entry?
1447 if(pHetTable->pNameHashes[Index] == HET_ENTRY_FREE)
1448 {
1449 // Set the entry in the name hash table
1450 pHetTable->pNameHashes[Index] = NameHash1;
1451
1452 // Set the entry in the file index table
1453 pHetTable->pBetIndexes->SetBits(pHetTable->dwIndexSizeTotal * Index,
1454 pHetTable->dwIndexSize,
1455 &dwFileIndex,
1456 4);
1457 return ERROR_SUCCESS;
1458 }
1459
1460 // Move to the next entry in the HET table
1461 // If we came to the start index again, we are done
1462 Index = (Index + 1) % pHetTable->dwTotalCount;
1463 if(Index == StartIndex)
1464 break;
1465 }
1466
1467 // No space in the HET table. Should never happen,
1468 // because the HET table is created according to the number of files
1469 assert(false);
1470 return ERROR_DISK_FULL;

1560static DWORD GetFileIndex_Het(TMPQArchive * ha, const char * szFileName)
1561{
1562 TMPQHetTable * pHetTable = ha->pHetTable;
1563 ULONGLONG FileNameHash;
1564 DWORD StartIndex;
1565 DWORD Index;
1566 BYTE NameHash1; // Upper 8 bits of the masked file name hash
1567
1568 // If there are no entries in the HET table, do nothing
1569 if(pHetTable->dwEntryCount == 0)
1570 return HASH_ENTRY_FREE;
1571
1572 // Do nothing if the MPQ has no HET table
1573 assert(ha->pHetTable != NULL);
1574
1575 // Calculate 64-bit hash of the file name
1576 FileNameHash = (HashStringJenkins(szFileName) & pHetTable->AndMask64) | pHetTable->OrMask64;
1577
1578 // Split the file name hash into two parts:
1579 // NameHash1: The highest 8 bits of the name hash
1580 // NameHash2: File name hash limited to hash size
1581 // Note: Our file table contains full name hash, no need to cut the high 8 bits before comparison
1582 NameHash1 = (BYTE)(FileNameHash >> (pHetTable->dwNameHashBitSize - 8));
1583
1584 // Calculate the starting index to the hash table
1585 StartIndex = Index = (DWORD)(FileNameHash % pHetTable->dwTotalCount);
1586
1587 // Go through HET table until we find a terminator
1588 while(pHetTable->pNameHashes[Index] != HET_ENTRY_FREE)
1589 {
1590 // Did we find a match ?
1591 if(pHetTable->pNameHashes[Index] == NameHash1)
1592 {
1593 DWORD dwFileIndex = 0;
1594
1595 // Get the file index
1596 pHetTable->pBetIndexes->GetBits(pHetTable->dwIndexSizeTotal * Index,
1597 pHetTable->dwIndexSize,
1598 &dwFileIndex,
1599 sizeof(DWORD));
1600
1601 // Verify the FileNameHash against the entry in the table of name hashes
1602 if(dwFileIndex <= ha->dwFileTableSize && ha->pFileTable[dwFileIndex].FileNameHash == FileNameHash)
1603 {
1604 return dwFileIndex;
1605 }
1606 }
1607
1608 // Move to the next entry in the HET table
1609 // If we came to the start index again, we are done
1610 Index = (Index + 1) % pHetTable->dwTotalCount;
1611 if(Index == StartIndex)
1612 break;
1613 }
1614
1615 // File not found
1616 return HASH_ENTRY_FREE;
1617}

2071TFileEntry * AllocateFileEntry(TMPQArchive * ha, const char * szFileName, LCID lcFileLocale, LPDWORD PtrHashIndex)
2072{
2073 TFileEntry * pFileTableEnd = ha->pFileTable + ha->dwFileTableSize;
2074 TFileEntry * pFreeEntry = NULL;
2075 TFileEntry * pFileEntry;
2076 TMPQHash * pHash = NULL;
2077 DWORD dwReservedFiles = ha->dwReservedFiles;
2078 DWORD dwFreeCount = 0;
2079
2080 // Sanity check: File table size must be greater or equal to max file count
2081 assert(ha->dwFileTableSize >= ha->dwMaxFileCount);
2082
2083 // If we are saving MPQ tables, we don't tale number of reserved files into account
2084 dwReservedFiles = (ha->dwFlags & MPQ_FLAG_SAVING_TABLES) ? 0 : ha->dwReservedFiles;
2085
2086 // Now find a free entry in the file table.
2087 // Note that in the case when free entries are in the middle,
2088 // we need to use these
2089 for(pFileEntry = ha->pFileTable; pFileEntry < pFileTableEnd; pFileEntry++)
2090 {
2091 if((pFileEntry->dwFlags & MPQ_FILE_EXISTS) == 0)
2092 {
2093 // Remember the first free entry
2094 if(pFreeEntry == NULL)
2095 pFreeEntry = pFileEntry;
2096 dwFreeCount++;
2097
2098 // If the number of free items is greater than number
2099 // of reserved items, We can add the file
2100 if(dwFreeCount > dwReservedFiles)
2101 break;
2102 }
2103 }
2104
2105 // If the total number of free entries is less than number of reserved files,
2106 // we cannot add the file to the archive
2107 if(pFreeEntry == NULL || dwFreeCount <= dwReservedFiles)
2108 return NULL;
2109
2110 // Initialize the file entry and set its file name
2111 memset(pFreeEntry, 0, sizeof(TFileEntry));
2112 AllocateFileName(ha, pFreeEntry, szFileName);
2113
2114 // If the archive has a hash table, we need to first free entry there
2115 if(ha->pHashTable != NULL)
2116 {
2117 // Make sure that the entry is not there yet
2118 assert(GetHashEntryExact(ha, szFileName, lcFileLocale) == NULL);
2119
2120 // Find a free hash table entry for the name
2121 pHash = AllocateHashEntry(ha, pFreeEntry, lcFileLocale);
2122 if(pHash == NULL)
2123 return NULL;
2124
2125 // Set the file index to the hash table
2126 pHash->dwBlockIndex = (DWORD)(pFreeEntry - ha->pFileTable);
2127 PtrHashIndex[0] = (DWORD)(pHash - ha->pHashTable);
2128 }
2129
2130 // If the archive has a HET table, just do some checks
2131 // Note: Don't bother modifying the HET table. It will be rebuilt from scratch after, anyway
2132 if(ha->pHetTable != NULL)
2133 {
2134 assert(GetFileIndex_Het(ha, szFileName) == HASH_ENTRY_FREE);
2135 }
2136
2137 // Return the free table entry
2138 return pFreeEntry;
