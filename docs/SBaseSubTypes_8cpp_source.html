21typedef struct _TSQPHeader
22{
23 // The ID_MPQ ('MPQ\x1A') signature
24 DWORD dwID;
25
26 // Size of the archive header
27 DWORD dwHeaderSize;
28
29 // 32-bit size of MPQ archive
30 DWORD dwArchiveSize;
31
32 // Offset to the beginning of the hash table, relative to the beginning of the archive.
33 DWORD dwHashTablePos;
34
35 // Offset to the beginning of the block table, relative to the beginning of the archive.
36 DWORD dwBlockTablePos;
37
38 // Number of entries in the hash table. Must be a power of two, and must be less than 2^16 for
39 // the original MoPaQ format, or less than 2^20 for the Burning Crusade format.
40 DWORD dwHashTableSize;
41
42 // Number of entries in the block table
43 DWORD dwBlockTableSize;
44
45 // Must be zero for SQP files
46 USHORT wFormatVersion;
47
48 // Power of two exponent specifying the number of 512-byte disk sectors in each file sector
49 // in the archive. The size of each file sector in the archive is 512 * 2 ^ wSectorSize.
50 USHORT wSectorSize;
51
52} TSQPHeader;

54typedef struct _TSQPHash
55{
56 // Most likely the Locale + Platform
57 DWORD dwAlwaysZero;
58
59 // If the hash table entry is valid, this is the index into the block table of the file.
60 // Otherwise, one of the following two values:
61 // - FFFFFFFFh: Hash table entry is empty, and has always been empty.
62 // Terminates searches for a given file.
63 // - FFFFFFFEh: Hash table entry is empty, but was valid at some point (a deleted file).
64 // Does not terminate searches for a given file.
65 DWORD dwBlockIndex;
66
67 // The hash of the file path, using method A.
68 DWORD dwName1;
69
70 // The hash of the file path, using method B.
71 DWORD dwName2;
72
73} TSQPHash;

94// This function converts SQP file header into MPQ file header
95DWORD ConvertSqpHeaderToFormat4(
96 TMPQArchive * ha,
97 ULONGLONG FileSize,
98 DWORD dwFlags)
99{
100 TSQPHeader * pSqpHeader = (TSQPHeader *)ha->HeaderData;
101 TMPQHeader Header;
102
103 // SQP files from War of the Immortal use MPQ file format with slightly
104 // modified structure. These fields have different position:
105 //
106 // Offset TMPQHeader TSQPHeader
107 // ------ ---------- -----------
108 // 000C wFormatVersion dwHashTablePos (lo)
109 // 000E wSectorSize dwHashTablePos (hi)
110 // 001C dwBlockTableSize (lo) wBlockSize
111 // 001E dwHashTableSize (hi) wFormatVersion
112
113 // Can't open the archive with certain flags
114 if(dwFlags & MPQ_OPEN_FORCE_MPQ_V1)
115 return ERROR_FILE_CORRUPT;
116
117 // The file must not be greater than 4 GB
118 if((FileSize >> 0x20) != 0)
119 return ERROR_FILE_CORRUPT;
120
121 // Translate the SQP header into a MPQ header
122 memset(&Header, 0, sizeof(TMPQHeader));
123 Header.dwID = BSWAP_INT32_UNSIGNED(pSqpHeader->dwID);
124 Header.dwHeaderSize = BSWAP_INT32_UNSIGNED(pSqpHeader->dwHeaderSize);
125 Header.dwArchiveSize = BSWAP_INT32_UNSIGNED(pSqpHeader->dwArchiveSize);
126 Header.dwHashTablePos = BSWAP_INT32_UNSIGNED(pSqpHeader->dwHashTablePos);
127 Header.dwBlockTablePos = BSWAP_INT32_UNSIGNED(pSqpHeader->dwBlockTablePos);
128 Header.dwHashTableSize = BSWAP_INT32_UNSIGNED(pSqpHeader->dwHashTableSize);
129 Header.dwBlockTableSize = BSWAP_INT32_UNSIGNED(pSqpHeader->dwBlockTableSize);
130 Header.wFormatVersion = BSWAP_INT16_UNSIGNED(pSqpHeader->wFormatVersion);
131 Header.wSectorSize = BSWAP_INT16_UNSIGNED(pSqpHeader->wSectorSize);
132
133 // Verify the SQP header
134 if(Header.dwID == g_dwMpqSignature && Header.dwHeaderSize == sizeof(TSQPHeader) && Header.dwArchiveSize == FileSize)
135 {
136 // Check for fixed values of version and sector size
137 if(Header.wFormatVersion == MPQ_FORMAT_VERSION_1 && Header.wSectorSize == 3)
138 {
139 // Initialize the fields of 3.0 header
140 Header.ArchiveSize64 = Header.dwArchiveSize;
141 Header.HashTableSize64 = Header.dwHashTableSize * sizeof(TMPQHash);
142 Header.BlockTableSize64 = Header.dwBlockTableSize * sizeof(TMPQBlock);
143
144 // Copy the converted MPQ header back
145 memcpy(ha->HeaderData, &Header, sizeof(TMPQHeader));
146
147 // Mark this file as SQP file
148 ha->pfnHashString = HashStringSlash;
149 ha->dwFlags |= MPQ_FLAG_READ_ONLY;
150 ha->dwSubType = MPQ_SUBTYPE_SQP;
151 return ERROR_SUCCESS;
152 }
153 }
154
155 return ERROR_FILE_CORRUPT;
156}

159{

162

183TMPQHash * LoadSqpHashTable(TMPQArchive * ha)
184{
185 TMPQHeader * pHeader = ha->pHeader;
186 TSQPHash * pSqpHashTable;
187 TSQPHash * pSqpHashEnd;
188 TSQPHash * pSqpHash;
189 TMPQHash * pMpqHash;
190 DWORD dwErrCode = ERROR_SUCCESS;
191
192 // Load the hash table
193 pSqpHashTable = (TSQPHash *)LoadSqpTable(ha, pHeader->dwHashTablePos, pHeader->dwHashTableSize * sizeof(TSQPHash), MPQ_KEY_HASH_TABLE);
194 if(pSqpHashTable != NULL)
195 {
196 // Parse the entire hash table and convert it to MPQ hash table
197 pSqpHashEnd = pSqpHashTable + pHeader->dwHashTableSize;
198 pMpqHash = (TMPQHash *)pSqpHashTable;
199 for(pSqpHash = pSqpHashTable; pSqpHash < pSqpHashEnd; pSqpHash++, pMpqHash++)
200 {
201 // Ignore free entries
202 if(pSqpHash->dwBlockIndex != HASH_ENTRY_FREE)
203 {
204 // Store the hash entry to a temporary variable
205 TSQPHash TempEntry = *pSqpHash;
206
207 // Check block index against the size of the block table
208 if(pHeader->dwBlockTableSize <= MPQ_BLOCK_INDEX(pSqpHash) && pSqpHash->dwBlockIndex < HASH_ENTRY_DELETED)
209 dwErrCode = ERROR_FILE_CORRUPT;
210
211 // We do not support nonzero locale and platform ID
212 if(pSqpHash->dwAlwaysZero != 0 && pSqpHash->dwAlwaysZero != HASH_ENTRY_FREE)
213 dwErrCode = ERROR_FILE_CORRUPT;
214
215 // Copy the entry to the MPQ hash entry
216 pMpqHash->dwName1 = TempEntry.dwName1;
217 pMpqHash->dwName2 = TempEntry.dwName2;
218 pMpqHash->dwBlockIndex = MPQ_BLOCK_INDEX(&TempEntry);
219 pMpqHash->Locale = 0;
220 pMpqHash->Platform = 0;
221 pMpqHash->Reserved = 0;
222 }
223 }
224
225 // If an error occured, we need to free the hash table
226 if(dwErrCode != ERROR_SUCCESS)
227 {
228 STORM_FREE(pSqpHashTable);
229 pSqpHashTable = NULL;
230 }
231 }
232
233 // Return the converted hash table (or NULL on failure)
234 return (TMPQHash *)pSqpHashTable;
235}

237// Loads the SQP Block table and converts it to a MPQ block table
238TMPQBlock * LoadSqpBlockTable(TMPQArchive * ha)
239{
240 TMPQHeader * pHeader = ha->pHeader;
241 TSQPBlock * pSqpBlockTable;
242 TSQPBlock * pSqpBlockEnd;
243 TSQPBlock * pSqpBlock;
244 TMPQBlock * pMpqBlock;
245 DWORD dwErrCode = ERROR_SUCCESS;
246
247 // Load the hash table
248 pSqpBlockTable = (TSQPBlock *)LoadSqpTable(ha, pHeader->dwBlockTablePos, pHeader->dwBlockTableSize * sizeof(TSQPBlock), MPQ_KEY_BLOCK_TABLE);
249 if(pSqpBlockTable != NULL)
250 {
251 // Parse the entire hash table and convert it to MPQ hash table
252 pSqpBlockEnd = pSqpBlockTable + pHeader->dwBlockTableSize;
253 pMpqBlock = (TMPQBlock *)pSqpBlockTable;
254 for(pSqpBlock = pSqpBlockTable; pSqpBlock < pSqpBlockEnd; pSqpBlock++, pMpqBlock++)
255 {
256 // Store the block entry to a temporary variable
257 TSQPBlock TempEntry = *pSqpBlock;
258
259 // Check for valid flags
260 if(pSqpBlock->dwFlags & ~MPQ_FILE_VALID_FLAGS)
261 dwErrCode = ERROR_FILE_CORRUPT;
262
263 // Convert SQP block table entry to MPQ block table entry
264 pMpqBlock->dwFilePos = TempEntry.dwFilePos;
265 pMpqBlock->dwCSize = TempEntry.dwCSize;
266 pMpqBlock->dwFSize = TempEntry.dwFSize;
267 pMpqBlock->dwFlags = TempEntry.dwFlags;
268 }
269
270 // If an error occured, we need to free the hash table
271 if(dwErrCode != ERROR_SUCCESS)
272 {
273 STORM_FREE(pSqpBlockTable);
274 pSqpBlockTable = NULL;
275 }
276 }
277
278 // Return the converted hash table (or NULL on failure)
279 return (TMPQBlock *)pSqpBlockTable;
280}

389// This function converts MPK file header into MPQ file header
390DWORD ConvertMpkHeaderToFormat4(
391 TMPQArchive * ha,
392 ULONGLONG FileSize,
393 DWORD dwFlags)
394{
395 TMPKHeader * pMpkHeader = (TMPKHeader *)ha->HeaderData;
396 TMPQHeader Header;
397
398 // Can't open the archive with certain flags
399 if(dwFlags & MPQ_OPEN_FORCE_MPQ_V1)
400 return ERROR_FILE_CORRUPT;
401
402 // Translate the MPK header into a MPQ header
403 // Note: Hash table size and block table size are in bytes, not in entries
404 memset(&Header, 0, sizeof(TMPQHeader));
405 Header.dwID = BSWAP_INT32_UNSIGNED(pMpkHeader->dwID);
406 Header.dwArchiveSize = BSWAP_INT32_UNSIGNED(pMpkHeader->dwArchiveSize);
407 Header.dwHeaderSize = BSWAP_INT32_UNSIGNED(pMpkHeader->dwHeaderSize);
408 Header.dwHashTablePos = BSWAP_INT32_UNSIGNED(pMpkHeader->dwHashTablePos);
409 Header.dwHashTableSize = BSWAP_INT32_UNSIGNED(pMpkHeader->dwHashTableSize) / sizeof(TMPKHash);
410 Header.dwBlockTablePos = BSWAP_INT32_UNSIGNED(pMpkHeader->dwBlockTablePos);
411 Header.dwBlockTableSize = BSWAP_INT32_UNSIGNED(pMpkHeader->dwBlockTableSize) / sizeof(TMPKBlock);
412// Header.dwUnknownPos = BSWAP_INT32_UNSIGNED(pMpkHeader->dwUnknownPos);
413// Header.dwUnknownSize = BSWAP_INT32_UNSIGNED(pMpkHeader->dwUnknownSize);
414 assert(Header.dwHeaderSize == sizeof(TMPKHeader));
415
416 // Verify the MPK header
417 if(Header.dwID == ID_MPK && Header.dwHeaderSize == sizeof(TMPKHeader) && Header.dwArchiveSize == (DWORD)FileSize)
418 {
419 // The header ID must be ID_MPQ
420 Header.dwID = g_dwMpqSignature;
421 Header.wFormatVersion = MPQ_FORMAT_VERSION_1;
422 Header.wSectorSize = 3;
423
424 // Initialize the fields of 3.0 header
425 Header.ArchiveSize64 = Header.dwArchiveSize;
426 Header.HashTableSize64 = Header.dwHashTableSize * sizeof(TMPQHash);
427 Header.BlockTableSize64 = Header.dwBlockTableSize * sizeof(TMPQBlock);
428
429 // Copy the converted MPQ header back
430 memcpy(ha->HeaderData, &Header, sizeof(TMPQHeader));
431
432 // Mark this file as MPK file
433 ha->pfnHashString = HashStringLower;
434 ha->dwFlags |= MPQ_FLAG_READ_ONLY;
435 ha->dwSubType = MPQ_SUBTYPE_MPK;
436 return ERROR_SUCCESS;
437 }
438 return ERROR_FILE_CORRUPT;
439}

441// Attempts to search a free hash entry in the hash table being converted.
442// The created hash table must always be of nonzero size,
443// should have no duplicated items and no deleted entries
444TMPQHash * FindFreeHashEntry(TMPQHash * pHashTable, DWORD dwHashTableSize, DWORD dwStartIndex)
445{
446 TMPQHash * pHash;
447 DWORD dwIndex;
448
449 // Set the initial index
450 dwStartIndex = dwIndex = (dwStartIndex & (dwHashTableSize - 1));
451 assert(dwHashTableSize != 0);
452
453 // Search the hash table and return the found entries in the following priority:
454 for(;;)
455 {
456 // We are not expecting to find matching entry in the hash table being built
457 // We are not expecting to find deleted entry either
458 pHash = pHashTable + dwIndex;
459
460 // If we found a free entry, we need to stop searching
461 if(pHash->dwBlockIndex == HASH_ENTRY_FREE)
462 return pHash;
463
464 // Move to the next hash entry.
465 // If we reached the starting entry, it's failure.
466 dwIndex = (dwIndex + 1) & (dwHashTableSize - 1);
467 if(dwIndex == dwStartIndex)
468 break;
469 }
470
471 // We haven't found anything
472 assert(false);
473 return NULL;
474}

511TMPQHash * LoadMpkHashTable(TMPQArchive * ha)
512{
513 TMPQHeader * pHeader = ha->pHeader;
514 TMPQHash * pHashTable = NULL;
515 TMPKHash * pMpkHash;
516 TMPQHash * pHash = NULL;
517 DWORD dwHashTableSize = pHeader->dwHashTableSize;
518
519 // MPKs use different hash table searching.
520 // Instead of using MPQ_HASH_TABLE_INDEX hash as index,
521 // they store the value directly in the hash table.
522 // Also for faster searching, the hash table is sorted ascending by the value
523
524 // Load and decrypt the MPK hash table.
525 pMpkHash = (TMPKHash *)LoadMpkTable(ha, pHeader->dwHashTablePos, pHeader->dwHashTableSize * sizeof(TMPKHash));
526 if(pMpkHash != NULL)
527 {
528 // Calculate the hash table size as if it was real MPQ hash table
529 pHeader->dwHashTableSize = GetNearestPowerOfTwo(pHeader->dwHashTableSize);
530 pHeader->HashTableSize64 = pHeader->dwHashTableSize * sizeof(TMPQHash);
531
532 // Now allocate table that will serve like a true MPQ hash table,
533 // so we translate the MPK hash table to MPQ hash table
534 pHashTable = STORM_ALLOC(TMPQHash, pHeader->dwHashTableSize);
535 if(pHashTable != NULL)
536 {
537 // Set the entire hash table to free
538 memset(pHashTable, 0xFF, (size_t)pHeader->HashTableSize64);
539
540 // Copy the MPK hash table into MPQ hash table
541 for(DWORD i = 0; i < dwHashTableSize; i++)
542 {
543 // Finds the free hash entry in the hash table
544 // We don't expect any errors here, because we are putting files to empty hash table
545 pHash = FindFreeHashEntry(pHashTable, pHeader->dwHashTableSize, pMpkHash[i].dwName1);
546 assert(pHash->dwBlockIndex == HASH_ENTRY_FREE);
547
548 // Copy the MPK hash entry to the hash table
549 pHash->dwBlockIndex = pMpkHash[i].dwBlockIndex;
550 pHash->Locale = 0;
551 pHash->Platform = 0;
552 pHash->Reserved = 0;
553 pHash->dwName1 = pMpkHash[i].dwName2;
554 pHash->dwName2 = pMpkHash[i].dwName3;
555 }
556 }
557
558 // Free the temporary hash table
559 STORM_FREE(pMpkHash);
560 }
561
562 return pHashTable;
563}

565static DWORD ConvertMpkFlagsToMpqFlags(DWORD dwMpkFlags)
566{
567 DWORD dwMpqFlags = MPQ_FILE_EXISTS;
568
569 // Check for flags that are always present
570 assert((dwMpkFlags & MPK_FILE_UNKNOWN_0001) != 0);
571 assert((dwMpkFlags & MPK_FILE_UNKNOWN_0010) != 0);
572 assert((dwMpkFlags & MPK_FILE_UNKNOWN_2000) != 0);
573 assert((dwMpkFlags & MPK_FILE_EXISTS) != 0);
574
575 // Append the compressed flag
576 dwMpqFlags |= (dwMpkFlags & MPK_FILE_COMPRESSED) ? MPQ_FILE_COMPRESS : 0;
577
578 // All files in the MPQ seem to be single unit files
579 dwMpqFlags |= MPQ_FILE_ENCRYPTED | MPQ_FILE_SINGLE_UNIT;
580
581 return dwMpqFlags;
582}

584TMPQBlock * LoadMpkBlockTable(TMPQArchive * ha)
585{
586 TMPQHeader * pHeader = ha->pHeader;
587 TMPKBlock * pMpkBlockTable;
588 TMPKBlock * pMpkBlockEnd;
589 TMPQBlock * pBlockTable = NULL;
590 TMPKBlock * pMpkBlock;
591 TMPQBlock * pMpqBlock;
592
593 // Load and decrypt the MPK block table
594 pMpkBlockTable = pMpkBlock = (TMPKBlock *)LoadMpkTable(ha, pHeader->dwBlockTablePos, pHeader->dwBlockTableSize * sizeof(TMPKBlock));
595 if(pMpkBlockTable != NULL)
596 {
597 // Allocate buffer for MPQ-like block table
598 pBlockTable = pMpqBlock = STORM_ALLOC(TMPQBlock, pHeader->dwBlockTableSize);
599 if(pBlockTable != NULL)
600 {
601 // Convert the MPK block table to MPQ block table
602 pMpkBlockEnd = pMpkBlockTable + pHeader->dwBlockTableSize;
603 while(pMpkBlock < pMpkBlockEnd)
604 {
605 // Translate the MPK block table entry to MPQ block table entry
606 pMpqBlock->dwFilePos = pMpkBlock->dwFilePos;
607 pMpqBlock->dwCSize = pMpkBlock->dwCSize;
608 pMpqBlock->dwFSize = pMpkBlock->dwFSize;
609 pMpqBlock->dwFlags = ConvertMpkFlagsToMpqFlags(pMpkBlock->dwFlags);
610
611 // Move both
612 pMpkBlock++;
613 pMpqBlock++;
614 }
615 }
616
617 // Free the MPK block table
618 STORM_FREE(pMpkBlockTable);
619 }
620
621 return pBlockTable;
622}
