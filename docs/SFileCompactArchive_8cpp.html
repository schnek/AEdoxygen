630 {

633 pTempStream = NULL;
634 else
635 dwErrCode = ERROR_CAN_NOT_COMPLETE;

639 if(dwErrCode == ERROR_SUCCESS && ha->pfnCompactCB != NULL)
640 {
641 ha->CompactBytesProcessed += (ha->pHeader->dwHashTableSize * sizeof(TMPQHash));
642 ha->CompactBytesProcessed += (ha->dwFileTableSize * sizeof(TMPQBlock));
643 ha->pfnCompactCB(ha->pvCompactUserData, CCB_CLOSING_ARCHIVE, ha->CompactBytesProcessed, ha->CompactTotalBytes);
644 }
645

647 if(pTempStream != NULL)
648 FileStream_Close(pTempStream);
649 if(pFileKeys != NULL)
650 STORM_FREE(pFileKeys);
651 if(dwErrCode != ERROR_SUCCESS)
652 SetLastError(dwErrCode);
653 return (dwErrCode == ERROR_SUCCESS);
