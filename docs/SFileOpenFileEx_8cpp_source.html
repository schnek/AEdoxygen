19static DWORD FindHashIndex(TMPQArchive * ha, DWORD dwFileIndex)
20{
21 TMPQHash * pHashTableEnd;
22 TMPQHash * pHash;
23 DWORD dwFirstIndex = HASH_ENTRY_FREE;
24
25 // Should only be called if the archive has hash table
26 assert(ha->pHashTable != NULL);
27
28 // Multiple hash table entries can point to the file table entry.
29 // We need to search all of them
30 pHashTableEnd = ha->pHashTable + ha->pHeader->dwHashTableSize;
31 for(pHash = ha->pHashTable; pHash < pHashTableEnd; pHash++)
32 {
33 if(MPQ_BLOCK_INDEX(pHash) == dwFileIndex)
34 {
35 // Duplicate hash entry found
36 if(dwFirstIndex != HASH_ENTRY_FREE)
37 return HASH_ENTRY_FREE;
38 dwFirstIndex = (DWORD)(pHash - ha->pHashTable);
39 }
40 }
41
42 // Return the hash table entry index
43 return dwFirstIndex;
44}

46static const char * GetPatchFileName(TMPQArchive * ha, const char * szFileName, char * szBuffer)
47{
48 TMPQNamePrefix * pPrefix;
49
50 // Are there patches in the current MPQ?
51 if(ha->dwFlags & MPQ_FLAG_PATCH)
52 {
53 // The patch prefix must be already known here
54 assert(ha->pPatchPrefix != NULL);
55 pPrefix = ha->pPatchPrefix;
56
57 // The patch name for "OldWorld\\XXX\\YYY" is "Base\\XXX\YYY"
58 // We need to remove the "OldWorld\\" prefix
59 if(!_strnicmp(szFileName, "OldWorld\\", 9))
60 szFileName += 9;
61
62 // Create the file name from the known patch entry
63 memcpy(szBuffer, pPrefix->szPatchPrefix, pPrefix->nLength);
64 strcpy(szBuffer + pPrefix->nLength, szFileName);
65 szFileName = szBuffer;
66 }
67
68 return szFileName;
69}

102bool OpenPatchedFile(HANDLE hMpq, const char * szFileName, HANDLE * PtrFile)
103{
104 TMPQArchive * haBase = NULL;
105 TMPQArchive * ha = (TMPQArchive *)hMpq;
106 TFileEntry * pFileEntry;
107 TMPQFile * hfPatch; // Pointer to patch file
108 TMPQFile * hfBase = NULL; // Pointer to base open file
109 TMPQFile * hf = NULL;
110 HANDLE hPatchFile;
111 char szNameBuffer[MAX_PATH];
112
113 // First of all, find the latest archive where the file is in base version
114 // (i.e. where the original, unpatched version of the file exists)
115 while(ha != NULL)
116 {
117 // If the file is there, then we remember the archive
118 pFileEntry = GetFileEntryExact(ha, GetPatchFileName(ha, szFileName, szNameBuffer), 0, NULL);
119 if(pFileEntry != NULL && (pFileEntry->dwFlags & MPQ_FILE_PATCH_FILE) == 0)
120 haBase = ha;
121
122 // Move to the patch archive
123 ha = ha->haPatch;
124 }
125
126 // If we couldn't find the base file in any of the patches, it doesn't exist
127 if((ha = haBase) != NULL)
128 {
129 // Now open the base file
130 if(SFileOpenFileEx((HANDLE)ha, GetPatchFileName(ha, szFileName, szNameBuffer), SFILE_OPEN_BASE_FILE, (HANDLE *)&hfBase))
131 {
132 // The file must be a base file, i.e. without MPQ_FILE_PATCH_FILE
133 assert((hfBase->pFileEntry->dwFlags & MPQ_FILE_PATCH_FILE) == 0);
134 hf = hfBase;
135
136 // Now open all patches and attach them on top of the base file
137 for(ha = ha->haPatch; ha != NULL; ha = ha->haPatch)
138 {
139 // Prepare the file name with a correct prefix
140 if(SFileOpenFileEx((HANDLE)ha, GetPatchFileName(ha, szFileName, szNameBuffer), SFILE_OPEN_BASE_FILE, &hPatchFile))
141 {
142 // Remember the new version
143 hfPatch = (TMPQFile *)hPatchFile;
144
145 // We should not find patch file
146 assert((hfPatch->pFileEntry->dwFlags & MPQ_FILE_PATCH_FILE) != 0);
147
148 // Attach the patch to the base file
149 hf->hfPatch = hfPatch;
150 hf = hfPatch;
151 }
152 }
153 }
154 }
155 else
156 {
157 SetLastError(ERROR_FILE_NOT_FOUND);
158 }
159
160 // Give the updated base MPQ
161 if(PtrFile != NULL)
162 *PtrFile = (HANDLE)hfBase;
163 return (hfBase != NULL);
164}

170//-----------------------------------------------------------------------------
171// SFileEnumLocales enums all locale versions within MPQ.
172// Functions fills all available language identifiers on a file into the buffer
173// pointed by PtrFileLocales. There must be enough entries to copy the localed,
174// otherwise the function returns ERROR_INSUFFICIENT_BUFFER.
175
176DWORD WINAPI SFileEnumLocales(
177 HANDLE hMpq,
178 const char * szFileName,
179 LCID * PtrFileLocales,
180 LPDWORD PtrMaxLocales,
181 DWORD dwSearchScope)
182{
183 TMPQArchive * ha = (TMPQArchive *)hMpq;
184 TMPQHash * pFirstHash;
185 TMPQHash * pHash;
186 DWORD dwFileIndex = 0;
187 DWORD dwMaxLocales;
188 DWORD dwLocales = 0;
189
190 // Test the parameters
191 if(!IsValidMpqHandle(hMpq))
192 return ERROR_INVALID_HANDLE;
193 if(szFileName == NULL || *szFileName == 0)
194 return ERROR_INVALID_PARAMETER;
195 if(ha->pHashTable == NULL)
196 return ERROR_NOT_SUPPORTED;
197 if(PtrMaxLocales == NULL)
198 return ERROR_INVALID_PARAMETER;
199 if(IsPseudoFileName(szFileName, &dwFileIndex))
200 return ERROR_INVALID_PARAMETER;
201
202 // Keep compiler happy
203 dwMaxLocales = PtrMaxLocales[0];
204 dwSearchScope = dwSearchScope;
205
206 // Parse all files with that name
207 pFirstHash = pHash = GetFirstHashEntry(ha, szFileName);
208 while(pHash != NULL)
209 {
210 // Put the locales to the buffer
211 if(PtrFileLocales != NULL && dwLocales < dwMaxLocales)
212 *PtrFileLocales++ = SFILE_MAKE_LCID(pHash->Locale, pHash->Platform);
213 dwLocales++;
214
215 // Get the next locale
216 pHash = GetNextHashEntry(ha, pFirstHash, pHash);
217 }
218
219 // Give the caller the number of locales and return
220 PtrMaxLocales[0] = dwLocales;
221 return (dwLocales <= dwMaxLocales) ? ERROR_SUCCESS : ERROR_INSUFFICIENT_BUFFER;
222}

224//-----------------------------------------------------------------------------
225// SFileOpenFileEx
226//
227// hMpq - Handle of opened MPQ archive
228// szFileName - Name of file to open
229// dwSearchScope - Where to search
230// PtrFile - Pointer to store opened file handle
231
232bool WINAPI SFileOpenFileEx(HANDLE hMpq, const char * szFileName, DWORD dwSearchScope, HANDLE * PtrFile)
233{
234 TMPQArchive * ha = IsValidMpqHandle(hMpq);
235 TFileEntry * pFileEntry = NULL;
236 TMPQFile * hf = NULL;
237 DWORD dwHashIndex = HASH_ENTRY_FREE;
238 DWORD dwFileIndex = 0;
239 DWORD dwErrCode = ERROR_SUCCESS;
240 bool bOpenByIndex = false;
241
242 // Don't accept NULL pointer to file handle
243 if(szFileName == NULL || *szFileName == 0)
244 dwErrCode = ERROR_INVALID_PARAMETER;
245
246 // When opening a file from MPQ, the handle must be valid
247 if(dwSearchScope != SFILE_OPEN_LOCAL_FILE && ha == NULL)
248 dwErrCode = ERROR_INVALID_HANDLE;
249
250 // When not checking for existence, the pointer to file handle must be valid
251 if(dwSearchScope != SFILE_OPEN_CHECK_EXISTS && PtrFile == NULL)
252 dwErrCode = ERROR_INVALID_PARAMETER;
253
254 // Prepare the file opening
255 if(dwErrCode == ERROR_SUCCESS)
256 {
257 switch(dwSearchScope)
258 {
259 case SFILE_OPEN_FROM_MPQ:
260 case SFILE_OPEN_BASE_FILE:
261 case SFILE_OPEN_CHECK_EXISTS:
262
263 // If this MPQ has no patches, open the file from this MPQ directly
264 if(ha->haPatch == NULL || dwSearchScope == SFILE_OPEN_BASE_FILE)
265 {
266 pFileEntry = GetFileEntryLocale(ha, szFileName, g_lcFileLocale, &dwHashIndex);
267 }
268
269 // If this MPQ is a patched archive, open the file as patched
270 else
271 {
272 return OpenPatchedFile(hMpq, szFileName, PtrFile);
273 }
274 break;
275
276 case SFILE_OPEN_ANY_LOCALE:
277
278 // This open option is reserved for opening MPQ internal listfile.
279 // No argument validation. Tries to open file with neutral locale first,
280 // then any other available.
281 pFileEntry = GetFileEntryLocale(ha, szFileName, 0, &dwHashIndex);
282 break;
283
284 case SFILE_OPEN_LOCAL_FILE:
285
286 // Open a local file
287 return OpenLocalFile(szFileName, PtrFile);
288
289 default:
290
291 // Don't accept any other value
292 dwErrCode = ERROR_INVALID_PARAMETER;
293 break;
294 }
295 }
296
297 // Check whether the file really exists in the MPQ
298 if(dwErrCode == ERROR_SUCCESS)
299 {
300 // If we didn't find the file, try to open it using pseudo file name ("File
301 if (pFileEntry == NULL || (pFileEntry->dwFlags & MPQ_FILE_EXISTS) == 0)
302 {
303 // Check the pseudo-file name ("File00000001.ext")
304 if ((bOpenByIndex = IsPseudoFileName(szFileName, &dwFileIndex)) == true)
305 {
306 // Get the file entry for the file
307 if (dwFileIndex < ha->dwFileTableSize)
308 {
309 pFileEntry = ha->pFileTable + dwFileIndex;
310 }
311 }
312
313 // Still not found?
314 if (pFileEntry == NULL)
315 {
316 dwErrCode = ERROR_FILE_NOT_FOUND;
317 }
318 }
319
320 // Perform some checks of invalid files
321 if (pFileEntry != NULL)
322 {
323 // MPQ protectors use insanely amount of fake files, often with very high size.
324 // We won't open any files whose compressed size is bigger than archive size
325 // If the file is not compressed, its size cannot be bigger than archive size
326 if ((pFileEntry->dwFlags & MPQ_FILE_COMPRESS_MASK) == 0 && (pFileEntry->dwFileSize > ha->FileSize))
327 {
328 dwErrCode = ERROR_FILE_CORRUPT;
329 pFileEntry = NULL;
330 }
331
332 // Ignore unknown loading flags (example: MPQ_2016_v1_WME4_4.w3x)
333// if(pFileEntry->dwFlags & ~MPQ_FILE_VALID_FLAGS)
334// {
335// dwErrCode = ERROR_NOT_SUPPORTED;
336// pFileEntry = NULL;
337// }
338 }
339 }
340
341 // Did the caller just wanted to know if the file exists?
342 if(dwErrCode == ERROR_SUCCESS && dwSearchScope != SFILE_OPEN_CHECK_EXISTS)
343 {
344 // Allocate file handle
345 hf = CreateFileHandle(ha, pFileEntry);
346 if(hf != NULL)
347 {
348 // Get the hash index for the file
349 if(ha->pHashTable != NULL && dwHashIndex == HASH_ENTRY_FREE)
350 dwHashIndex = FindHashIndex(ha, dwFileIndex);
351 if(dwHashIndex != HASH_ENTRY_FREE)
352 hf->pHashEntry = ha->pHashTable + dwHashIndex;
353 hf->dwHashIndex = dwHashIndex;
354
355 // If the MPQ has sector CRC enabled, enable if for the file
356 if(ha->dwFlags & MPQ_FLAG_CHECK_SECTOR_CRC)
357 hf->bCheckSectorCRCs = true;
358
359 // If we know the real file name, copy it to the file entry
360 if(bOpenByIndex == false)
361 {
362 // If there is no file name yet, allocate it
363 AllocateFileName(ha, pFileEntry, szFileName);
364
365 // If the file is encrypted, we should detect the file key
366 if(pFileEntry->dwFlags & MPQ_FILE_ENCRYPTED)
367 {
368 hf->dwFileKey = DecryptFileKey(szFileName,
369 pFileEntry->ByteOffset,
370 pFileEntry->dwFileSize,
371 pFileEntry->dwFlags);
372 }
373 }
374 }
375 else
376 {
377 dwErrCode = ERROR_NOT_ENOUGH_MEMORY;
378 }
379 }
380
381 // Give the file entry
382 if(PtrFile != NULL)
383 PtrFile[0] = hf;
384
385 // Return error code
386 if(dwErrCode != ERROR_SUCCESS)
387 SetLastError(dwErrCode);
388 return (dwErrCode == ERROR_SUCCESS);
389}
