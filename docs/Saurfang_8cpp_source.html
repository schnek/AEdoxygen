326void MuradinSaurfangEvent::DoAction(int32_t const action)
327{
328 switch (action)
329 {
330 case EVENT_WIPE:
331 {
332 Reset();
333 break;
334 }
335 case ACTION_START_EVENT:
336 {
337 // Prevent crashes
338 if (getScriptPhase() == PHASE_INTRO_A)
339 return;
340
341 // Guards
342 uint32_t x = 1;
343 GetCreatureListWithEntryInGrid(_guardList, NPC_SE_SKYBREAKER_MARINE, 20.0f);
344 for (auto itr = _guardList.begin(); itr != _guardList.end(); ++x, ++itr)
345 (*itr)->GetScript()->SetCreatureData(0, x);
346 //
347
348 setScriptPhase(PHASE_INTRO_A);
349 sendDBChatMessage(SAY_INTRO_ALLIANCE_1_SE);
350
351 // Start Intro
352 scriptEvents.addEvent(EVENT_INTRO_ALLIANCE_4_SE, 29500, PHASE_INTRO_A);
353
354 // Open Suarfangs Door
355 if (GameObject* Door = mInstance->GetGameObjectByGuid(mInstance->getLocalData(DATA_SAURFANG_DOOR)))
356 Door->setState(GO_STATE_OPEN);
357
358 // Start Intro on Suarfang
359 if (Creature* deathbringer = mInstance->getLocalCreatureData(CN_DEATHBRINGER_SAURFANG))
360 deathbringer->GetScript()->DoAction(PHASE_INTRO_A);
361
362 // Clear NPC FLAGS
363 getCreature()->removeNpcFlags(UNIT_NPC_FLAG_GOSSIP);
364 getCreature()->getAIInterface()->setAllowedToEnterCombat(false);
365 break;
366 }
367 case ACTION_START_OUTRO:
368 {
369 scriptEvents.addEvent(EVENT_OUTRO_ALLIANCE_1_SE, 1000);
370 mInstance->DoAction(ACTION_SPAWN_TRANSPORT);
371 break;
372 }
373 case ACTION_CONTINUE_OUTRO:
374 {
375 scriptEvents.addEvent(EVENT_OUTRO_ALLIANCE_7_SE, 1000);
376 break;
377 }
378 case ACTION_CONTINUE_OUTRO2:
379 {
380 scriptEvents.addEvent(EVENT_OUTRO_ALLIANCE_15_SE, 1000);
381 break;
382 }
383 default:
384 break;
385 }
