14void scriptEventMap::updateEvents(uint32_t _diff, uint32_t _phase)
15{
16 bossPhase = _phase;
17
18 if (!eventMapStore.empty())
19 {
20 for (EventMap::iterator itr = eventMapStore.begin(); itr != eventMapStore.end();)

22 auto& scriptEvent = itr->second;

24 {
