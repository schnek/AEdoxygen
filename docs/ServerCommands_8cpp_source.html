30//.server info
31bool ChatCommandHandler::HandleServerInfoCommand(const char* /*args*/, WorldSession* m_session)
32{
33 uint16_t online_gm = 0;
34 uint16_t online_count = 0;
35 float latency_avg = 0;
36
37 std::lock_guard guard(sObjectMgr.m_playerLock);
38 for (const auto playerPair : sObjectMgr.getPlayerStorage())
39 {
40 Player* player = playerPair.second;
41 if (player->getSession())
42 {
43 online_count++;
44 latency_avg += player->getSession()->GetLatency();
45 if (player->getSession()->hasPermissions())
46 {
47 if (!worldConfig.gm.listOnlyActiveGms)
48 online_gm++;
49 else
50 if (player->isGMFlagSet())
51 online_gm++;
52 }
53 }
54 }
55
56 uint32_t active_sessions = uint32_t(sWorld.getSessionCount());
57
58 greenSystemMessage(m_session, "Info: |r{}AscEmu {}/{}-{}-{} {}(www.ascemu.org)", MSG_COLOR_WHITE, AE_BUILD_HASH, CONFIG, AE_PLATFORM, AE_ARCHITECTURE, MSG_COLOR_LIGHTBLUE);
59 greenSystemMessage(m_session, "Using {}/Library {}", OPENSSL_VERSION_TEXT, SSLeay_version(SSLEAY_VERSION));
60 greenSystemMessage(m_session, "Uptime: |r{}", sWorld.getWorldUptimeString().c_str());
61 greenSystemMessage(m_session, "Active Branch: |r{}", AE_BUILD_BRANCH);
62 greenSystemMessage(m_session, "Active Sessions: |r{}", active_sessions);
63 greenSystemMessage(m_session, "Current GMs: |r{} GMs", online_gm);
64 greenSystemMessage(m_session, "Current Players: |r{} ({} Peak)", online_gm > 0 ? (online_count - online_gm) : online_count, sWorld.getPeakSessionCount());
65 greenSystemMessage(m_session, "Active Thread Count: |r{}", ThreadPool.GetActiveThreadCount());
66 greenSystemMessage(m_session, "Free Thread Count: |r{}", ThreadPool.GetFreeThreadCount());
67 greenSystemMessage(m_session, "Average Latency: |r{}ms", online_count > 0 ? (latency_avg / online_count) : latency_avg);
68 greenSystemMessage(m_session, "CPU Usage: {}%", sWorld.getCPUUsage());
69 greenSystemMessage(m_session, "RAM Usage: {} MB", sWorld.getRAMUsage());
70 greenSystemMessage(m_session, "SQL Query Cache Size (World): |r{} queries delayed", WorldDatabase.GetQueueSize());
71 greenSystemMessage(m_session, "SQL Query Cache Size (Character): |r{} queries delayed", CharacterDatabase.GetQueueSize());
72 greenSystemMessage(m_session, "Socket Count: |r{}", sSocketMgr.GetSocketCount());
73
74 return true;
75}

91//.server save
92bool ChatCommandHandler::HandleServerSaveCommand(const char* args, WorldSession* m_session)
93{
94 Player* player_target = nullptr;
95
96 if (!args)
97 {
98 player_target = GetSelectedPlayer(m_session, false, false);
99 if (player_target == nullptr)
100 {
101 redSystemMessage(m_session, "You need to target or name a player!");
102 redSystemMessage(m_session, "Use: .server save (on a targeted player)");
103 redSystemMessage(m_session, "or: .server save <playername>");
104 return true;
105 }
106 }
107 else
108 {
109 player_target = sObjectMgr.getPlayer(args, false);
110 if (player_target == nullptr)
111 {
112 redSystemMessage(m_session, "A player with name {} is not online / does not exist!", args);
113 return true;
114 }
115 }
116
117 if (player_target->m_nextSave < 180000)
118 {
119 player_target->saveToDB(false);
120 greenSystemMessage(m_session, "Player {} saved to DB", player_target->getName());
121 }
122 else
123 {
124 redSystemMessage(m_session, "You can only save once every 3 minutes.");
125 }
126
127 return true;
128}

151
152 sWorld.sendMessageToOnlineGms(teamAnnounce.str());
153

176//.server shutdown
177bool ChatCommandHandler::HandleServerShutdownCommand(const char* args, WorldSession* m_session)
178{
179 uint32_t shutdowntime;
180 if (!args)
181 shutdowntime = 30;
182 else
183 shutdowntime = std::stoul(args);
184
185 if (shutdowntime < 30)
186 shutdowntime = 30;
187
188 std::stringstream teamAnnounce;
189 teamAnnounce << MSG_COLOR_RED << "[Team]" << MSG_COLOR_GREEN << " |Hplayer:" << m_session->GetPlayer()->getName().c_str();
190 teamAnnounce << "|h[" << m_session->GetPlayer()->getName().c_str() << "]|h:" << MSG_COLOR_YELLOW << " initiated server shutdown timer " << shutdowntime << " sec";
191
192 sWorld.sendMessageToOnlineGms(teamAnnounce.str());
193
194 sGMLog.writefromsession(m_session, "initiated server shutdown timer %u sec", shutdowntime);
195
196 std::stringstream worldAnnounce;
197 worldAnnounce << "Server is shutting down in " << shutdowntime << " seconds.";
198
199 sWorld.sendMessageToAll(worldAnnounce.str());
200 sWorld.sendAreaTriggerMessage(worldAnnounce.str());
201
202 shutdowntime *= 1000;
203 sMaster.m_ShutdownTimer = shutdowntime;
204 sMaster.m_ShutdownEvent = true;
205 sMaster.m_restartEvent = false;
206
207 return true;
208}

225 else
226 {
227 teamAnnounce << MSG_COLOR_RED << "[Team]" << MSG_COLOR_GREEN << " |Hplayer:" << m_session->GetPlayer()->getName().c_str();
228 teamAnnounce << "|h[" << m_session->GetPlayer()->getName().c_str() << "]|h:" << MSG_COLOR_YELLOW << " canceled server shutdown!";
229 }
230
231 sWorld.sendMessageToOnlineGms(teamAnnounce.str());
232 sGMLog.writefromsession(m_session, "canceled server shutdown");
233
234 sWorld.sendGlobalMessage(AscEmu::Packets::SmsgServerMessage(sMaster.m_restartEvent ? SERVER_MSG_RESTART_CANCELLED : SERVER_MSG_SHUTDOWN_CANCELLED).serialise().get());
235
236 sMaster.m_ShutdownTimer = 5000;
237 sMaster.m_ShutdownEvent = false;
238 sMaster.m_restartEvent = false;
239
240 return true;
