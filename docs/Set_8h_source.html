23/**
24 An unordered data structure that has at most one of each element.
25 Provides O(1) time insert, remove, and member test (contains).
26
27 Set uses G3D::Table internally, which means that the template type T
28 must define a hashCode and operator== function. See G3D::Table for
29 a discussion of these functions.
30 */
31// There is not copy constructor or assignment operator defined because
32// the default ones are correct for Set.
33template<class T, class HashFunc = HashTrait<T>, class EqualsFunc = EqualsTrait<T> >
34class Set {
35
36 /**
37 If an object is a member, it is contained in
38 this table.
39 */
40 Table<T, bool, HashFunc, EqualsFunc> memberTable;
41
42public:
43
44 void clearAndSetMemoryManager(const MemoryManager::Ref& m) {
45 memberTable.clearAndSetMemoryManager(m);
46 }
47
48 virtual ~Set() {}
49
50 int size() const {
51 return (int)memberTable.size();
52 }
53
54 bool contains(const T& member) const {
55 return memberTable.containsKey(member);
56 }
57
58 /**
59 Inserts into the table if not already present.
60 Returns true if this is the first time the element was added.
61 */
62 bool insert(const T& member) {
63 bool isNew = false;
64 memberTable.getCreate(member, isNew) = true;
65 return isNew;
66 }
67
68 /**
69 Returns true if the element was present and removed. Returns false
70 if the element was not present.
71 */
72 bool remove(const T& member) {
73 return memberTable.remove(member);
74 }
75
76 /** If @a member is present, sets @a removed to the element
77 being removed and returns true. Otherwise returns false
78 and does not write to @a removed. This is useful when building
79 efficient hashed data structures that wrap Set.
80 */
81 bool getRemove(const T& member, T& removed) {
82 bool ignore;
83 return memberTable.getRemove(member, removed, ignore);
84 }
85
86 /** If a value that is EqualsFunc to @a member is present, returns a pointer to the
87 version stored in the data structure, otherwise returns NULL.
88 */
89 const T* getPointer(const T& member) const {
90 return memberTable.getKeyPointer(member);
91 }
92
93 Array<T> getMembers() const {
