36bool FlametongueWeaponPassive(uint8_t /*effectIndex*/, Aura* pAura, bool apply)
37{
38 Unit* target = pAura->getOwner();
39 if (apply)
40 {
41 // target is always a player
42 Item* item = static_cast<Player*>(target)->getItemInterface()->GetItemByGUID(pAura->itemCasterGUID);
43 target->addProcTriggerSpell(10444, pAura->getSpellInfo()->getId(), pAura->getCasterGuid(), pAura->getSpellInfo()->getProcChance(), PROC_ON_DONE_MELEE_HIT, EXTRA_PROC_NULL, NULL, NULL, pAura, item);
44 }
45 else
46 target->removeProcTriggerSpell(10444, pAura->getCasterGuid(), pAura->itemCasterGUID);
47 return true;
48}

50bool SkyShatterRegalia(uint8_t /*effectIndex*/, Spell* s)
51{
52 // Shaman - Skyshatter Regalia - Two Piece Bonus
53 // it checks for earth, air, water, fire totems and triggers Totemic Mastery spell 38437.
54
55 if (!s->getPlayerCaster())
56 return false;
57
58 if (s->getPlayerCaster()->getSummonInterface()->hasTotemInSlot(SUMMON_SLOT_TOTEM_FIRE) &&
59 s->getPlayerCaster()->getSummonInterface()->hasTotemInSlot(SUMMON_SLOT_TOTEM_EARTH) &&
60 s->getPlayerCaster()->getSummonInterface()->hasTotemInSlot(SUMMON_SLOT_TOTEM_WATER) &&
61 s->getPlayerCaster()->getSummonInterface()->hasTotemInSlot(SUMMON_SLOT_TOTEM_AIR))
62 {
63 auto aur = sSpellMgr.newAura(sSpellMgr.getSpellInfo(38437), 5000, s->getPlayerCaster(), s->getPlayerCaster(), true);
64
65 for (uint8_t j = 0; j < 3; j++)
66 aur->addAuraEffect(static_cast<AuraEffect>(aur->getSpellInfo()->getEffectRadiusIndex(j)), aur->getSpellInfo()->getEffectBasePoints(j) + 1, aur->getSpellInfo()->getEffectMiscValue(j), 1.0f, false, j);
67
68 s->getPlayerCaster()->addAura(std::move(aur));
69 }
70
71 return true;
72}

74bool ManaTide(uint8_t /*effectIndex*/, Spell* s)
75{
76 Unit* unitTarget = s->getUnitTarget();
77
78 if (unitTarget == NULL || unitTarget->isDead() || unitTarget->getClass() == WARRIOR || unitTarget->getClass() == ROGUE)
79 return false;
80
81 uint32_t gain = (uint32_t)(unitTarget->getMaxPower(POWER_TYPE_MANA) * 0.06);
82 unitTarget->energize(unitTarget, 16191, gain, POWER_TYPE_MANA);
83
84 return true;
85}

87bool Reincarnation(uint8_t /*effectIndex*/, Aura* a, bool apply)
88{
89 Unit* u_target = a->getOwner();
90
91 if (!u_target->isPlayer())
92 return true;
93
94 Player* p_target = static_cast<Player*>(u_target);
95
96 if (apply)
97 p_target->m_reincarnation = true;
98 else
99 p_target->m_reincarnation = false;
100
101 return true;
102}
