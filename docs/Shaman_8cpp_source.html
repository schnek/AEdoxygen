96class LightningShieldDummy : public SpellScript
97{
98public:
99 SpellInfo const* getDamageSpellForDummyRank(SpellInfo const* dummyInfo) const
100 {
101 const auto* dmgInfo = sSpellMgr.getSpellInfo(SPELL_LIGHTNING_SHIELD_DMG_R1);
102 if (!dummyInfo->hasSpellRanks() || dmgInfo == nullptr || !dmgInfo->hasSpellRanks())
103 return dmgInfo;
104
105 const auto dummyRank = dummyInfo->getRankInfo()->getRank();
106 return dmgInfo->getRankInfo()->getSpellWithRank(dummyRank);
107 }
108
109 SpellScriptEffectDamage doCalculateEffect(Spell* spell, uint8_t effIndex, int32_t* dmg) override
110 {
111 if (effIndex != EFF_INDEX_0 || spell->getUnitCaster() == nullptr)
112 return SpellScriptEffectDamage::DAMAGE_DEFAULT;
113
114 // Calculate damage done here so correct percent modifiers are applied
115 *dmg = static_cast<int32_t>(std::ceil(spell->getUnitCaster()->applySpellDamageBonus(spell->getSpellInfo(), *dmg, 1.0f, false, spell)));
116 return SpellScriptEffectDamage::DAMAGE_NO_BONUSES;
117 }
118
119 SpellScriptExecuteState beforeAuraEffect(Aura* aur, AuraEffectModifier* aurEff, bool apply) override
120 {
121 if (!apply)
122 return SpellScriptExecuteState::EXECUTE_OK;
123
124 const auto damageSpell = getDamageSpellForDummyRank(aur->getSpellInfo());
125 if (damageSpell == nullptr)
126 return SpellScriptExecuteState::EXECUTE_PREVENT;
127
128 // TODO: classic uses another dummy spell before damage
129
130 // Override effect with real proc effect
131 auto spellProc = aur->getOwner()->addProcTriggerSpell(damageSpell, aur, aur->getCasterGuid());
132 if (spellProc != nullptr)
133 spellProc->setOverrideEffectDamage(EFF_INDEX_0, aurEff->getEffectDamage());
134
135 return SpellScriptExecuteState::EXECUTE_PREVENT;
136 }
137
138 void onAuraRemove(Aura* aur, AuraRemoveMode /*mode*/) override
139 {
140 if (const auto damageSpell = getDamageSpellForDummyRank(aur->getSpellInfo()))
141 aur->getOwner()->removeProcTriggerSpell(damageSpell->getId(), aur->getCasterGuid());
142 }
143};
