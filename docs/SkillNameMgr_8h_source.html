73class SkillNameMgr
74{
75 public:
76 std::unique_ptr<std::unique_ptr<char[]>[]> SkillNames;
77 uint32_t maxskill;
78
79 SkillNameMgr()
80 {
81 /// This will become the size of the skill name lookup table
82 maxskill = sSkillLineStore.getNumRows();
83
84 /// SkillNames = (char **) malloc(maxskill * sizeof(char *));
85 SkillNames = std::make_unique<std::unique_ptr<char[]>[]>(maxskill + 1); //(+1, arrays count from 0.. not 1.)
86 std::fill(SkillNames.get(), SkillNames.get(), nullptr);
87
88 for (uint32_t i = 0; i < sSkillLineStore.getNumRows(); ++i)
89 {
90 auto skill_line = sSkillLineStore.lookupEntry(i);
91 if (skill_line == nullptr)
92 continue;
93
94 uint32_t SkillID = skill_line->id;
95#if VERSION_STRING < Cata
96 char* SkillName = skill_line->Name[sWorld.getDbcLocaleLanguageId()];
97#else
98 char* SkillName = skill_line->Name[0];
99#endif
100
101 SkillNames[SkillID] = std::make_unique<char[]>(strlen(SkillName) + 1);
102 //When the DBCFile gets cleaned up, so does the record data, so make a copy of it..
103 std::strcpy(SkillNames[SkillID].get(), SkillName);
104 }
105 }
106
107 ~SkillNameMgr() = default;
108};
