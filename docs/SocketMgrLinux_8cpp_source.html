17void SocketMgr::AddSocket(Socket* s)
18{
19#ifdef ENABLE_ANTI_DOS
20 uint32_t saddr;
21 int i, count;
22
23 // Check how many connections we already have from that ip
24 saddr = s->GetRemoteAddress().s_addr;
25 for(i = 0, count = 0; i <= max_fd; i++)
26 {
27 if(fds[i])
28 {
29 if(fds[i]->GetRemoteAddress().s_addr == saddr) count++;
30 }
31 }
32
33 // More than 16 connections from the same ip? enough! xD
34 if(count > 16)
35 {
36 s->Disconnect(false);
37 return;
38 }
39#endif
40
41 if(fds[s->GetFd()] != NULL)
42 {
43 //fds[s->GetFd()]->Delete();
44 //fds[s->GetFd()] = NULL;
45 s->Delete();
46 return;
47 }
48
49 if(max_fd < s->GetFd()) max_fd = s->GetFd();
50 fds[s->GetFd()] = s;
51 ++socket_count;
52
53 // Add epoll event based on socket activity.
54 struct epoll_event ev;
55 memset(&ev, 0, sizeof(epoll_event));
56 ev.events = (s->writeBuffer.GetSize()) ? EPOLLOUT : EPOLLIN;
57 ev.events |= EPOLLET; // use edge-triggered instead of level-triggered because we're using nonblocking sockets.
58 ev.data.fd = s->GetFd();
59
60 if(epoll_ctl(epoll_fd, EPOLL_CTL_ADD, ev.data.fd, &ev))
61 sLogger.failure("Could not add event to epoll set on fd {}", s->GetFd());
62}

80void SocketMgr::RemoveSocket(Socket* s)
81{
82 if(fds[s->GetFd()] != s)
83 {
84 sLogger.failure("Could not remove fd {} from the set due to it not existing?", s->GetFd());
85 return;
86 }
87
88 fds[s->GetFd()] = NULL;
89 --socket_count;
90
91 // Remove from epoll list.
92 struct epoll_event ev;
93 memset(&ev, 0, sizeof(epoll_event));
94 ev.data.fd = s->GetFd();
95 ev.events = EPOLLIN | EPOLLOUT | EPOLLERR | EPOLLHUP | EPOLLONESHOT;
96
97 if(epoll_ctl(epoll_fd, EPOLL_CTL_DEL, ev.data.fd, &ev))
98 sLogger.failure("Could not remove fd {} from epoll set, errno {}", s->GetFd(), errno);
99}

120bool SocketWorkerThread::runThread()
121{
122 int fd_count;
123 Socket* ptr;
124 int i;
125 running = true;
126 // events[i].data.fd = ptr->GetFd();
127
128 while(running)
129 {
130 fd_count = epoll_wait(sSocketMgr.epoll_fd, events, THREAD_EVENT_SIZE, 5000);
131 for(i = 0; i < fd_count; ++i)
132 {
133 if(events[i].data.fd >= SOCKET_HOLDER_SIZE)
134 {
135 sLogger.failure("Requested FD that is too high ({})", ptr->GetFd());
136 continue;
137 }
138
139 ptr = sSocketMgr.fds[events[i].data.fd];
140
141 if(ptr == NULL)
142 {
143 if((ptr = ((Socket*)sSocketMgr.listenfds[events[i].data.fd])) != NULL)
144 ((ListenSocketBase*)ptr)->OnAccept();
145 else
146 sLogger.failure("Returned invalid fd (no pointer) of FD {}", ptr->GetFd());
147 continue;
148 }
149
150 if(events[i].events & EPOLLHUP || events[i].events & EPOLLERR)
151 {
152 ptr->Disconnect();
153 continue;
154 }
155 else if(events[i].events & EPOLLIN)
156 {
157 ptr->ReadCallback(0); // Len is unknown at this point.
158
159 /* changing to written state? */
160 if(ptr->writeBuffer.GetSize() && !ptr->HasSendLock() && ptr->IsConnected())
161 ptr->PostEvent(EPOLLOUT);
162 }
163 else if(events[i].events & EPOLLOUT)
164 {
165 ptr->BurstBegin(); // Lock receive mutex
166 ptr->WriteCallback(); // Perform actual send()
167 if(ptr->writeBuffer.GetSize() > 0)
168 {
169 /* we don't have to do anything here. no more oneshots :) */
170 }
171 else
172 {
173 /* change back to a read event */
174 ptr->DecSendLock();
175 ptr->PostEvent(EPOLLIN);
176 }
177 ptr->BurstEnd(); // Unlock
178 }
179 }
180 }
181 return true;
182}
