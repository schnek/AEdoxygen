56 /// constructor > create epoll device handle + initialize event set
57 void initialize()
58 {
59 epoll_fd = epoll_create(SOCKET_HOLDER_SIZE);
60 if(epoll_fd == -1)
61 {
62 sLogger.failure("Could not create epoll fd (/dev/epoll).");
63 exit(-1);
64 }
65
66 // null out the pointer array
67 memset(fds, 0, sizeof(void*) * SOCKET_HOLDER_SIZE);
68 memset(listenfds, 0, sizeof(void*) * SOCKET_HOLDER_SIZE);
69 max_fd = 0;
70 }
71
72 /// destructor > destroy epoll handle
73 void finalize()
74 {
75 // close epoll handle
76 close(epoll_fd);
77 }
