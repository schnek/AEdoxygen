39bool Socket::Connect(const char* Address, uint32_t Port)
40{
41 struct hostent* ci = gethostbyname(Address);
42 if(ci == 0)
43 return false;
44
45 m_client.sin_family = ci->h_addrtype;
46 m_client.sin_port = ntohs((u_short)Port);
47 memcpy(&m_client.sin_addr.s_addr, ci->h_addr_list[0], ci->h_length);
48
49 SocketOps::Blocking(m_fd);
50 if(connect(m_fd, (const sockaddr*)&m_client, sizeof(m_client)) == -1)
51 return false;
52
53 // at this point the connection was established
54#ifdef CONFIG_USE_IOCP
55 m_completionPort = sSocketMgr.GetCompletionPort();
56#endif
57 _OnConnect();
58 return true;
59}
