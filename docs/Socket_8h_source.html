204/** Connect to a server.
205* @param hostname Hostname or IP address to connect to
206* @param port Port to connect to
207* @return templated type if successful, otherwise null
208*/
209template<class T>
210T* ConnectTCPSocket(const char* hostname, u_short port)
211{
212 sockaddr_in conn;
213 hostent* host;
214
215 /* resolve the peer */
216 host = gethostbyname(hostname);
217
218 if(!host)
219 return NULL;
220
221 /* copy into our address struct */
222 memcpy(&conn.sin_addr, host->h_addr_list[0], sizeof(in_addr));
223 conn.sin_family = AF_INET;
224 conn.sin_port = ntohs(port);
225
226 T* s = new T(0);
227 if(!s->Connect(hostname, port))
228 {
229 s->Delete();
230 return 0;
231 }
232 return s;
233}

238class SocketGarbageCollector
239{
240 std::map<Socket*, time_t> deletionQueue;
241 Mutex lock;
242 private:
243 SocketGarbageCollector() = default;
244 ~SocketGarbageCollector() = default;
245
246 public:
247
248 static SocketGarbageCollector& getInstance()
249 {
250 static SocketGarbageCollector mInstance;
251 return mInstance;
252 }
253
254 void finalize()
255 {
256 for (auto i : deletionQueue)
257 delete i.first;
258 }
259
260 SocketGarbageCollector(SocketGarbageCollector&&) = delete;
261 SocketGarbageCollector(SocketGarbageCollector const&) = delete;
262 SocketGarbageCollector& operator=(SocketGarbageCollector&&) = delete;
263 SocketGarbageCollector& operator=(SocketGarbageCollector const&) = delete;
264
265 void Update()
266 {
267 std::map<Socket*, time_t>::iterator i, i2;
268 time_t t = UNIXTIME;
269 lock.acquire();
270 for(i = deletionQueue.begin(); i != deletionQueue.end();)
271 {
272 i2 = i++;
273 if(i2->second <= t)
274 {
275 delete i2->first;
276 deletionQueue.erase(i2);
277 }
278 }
279 lock.release();
280 }
281
282 void QueueSocket(Socket* s)
283 {
284 lock.acquire();
285 deletionQueue.insert(std::map<Socket*, time_t>::value_type(s, UNIXTIME + SOCKET_GC_TIMEOUT));
286 lock.release();
287 }
288};
