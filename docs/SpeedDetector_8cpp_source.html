61void SpeedCheatDetector::AddSample(float x, float y, int stamp, float player_speed)
62{
63 int time_dif = stamp - last_stamp;
64 //don't flood with calculations. Small values might get lost in calculations
65 //also takes care of the delaying of the speed detector
66 if (time_dif <= SPDT_SAMPLINGRATE)
67 return;
68 //we reset the system in case the monitored player changed it's speed. A small loss to avoid false triggers
69 if (player_speed != last_used_speed)
70 {
71 EventSpeedChange();
72 last_used_speed = player_speed;
73 }
74 //seems like we are monitored an interval. Check if we detected any speed hack in it
75 else if (last_stamp != 0)
76 {
77 //get current speed
78 float dif_x = x - last_x;
79 float dif_y = y - last_y;
80 float dist = std::sqrt(dif_x * dif_x + dif_y * dif_y);
81 float cur_speed = dist / (float)time_dif * 1000.0f;
82
83 //check if we really got a cheater here
84 if (cur_speed * SPDT_DETECTION_ERROR > player_speed)
85 {
86 cheat_threat++;
87 float hackspeed_diff = cur_speed - player_speed;
88 if (hackspeed_diff > bigest_hacked_speed_dif)
89 bigest_hacked_speed_dif = hackspeed_diff;
90 }
91 else if (cheat_threat > 0)
92 cheat_threat--;
93
94 }
95 last_stamp = stamp;
96 last_x = x;
97 last_y = y;
98}
