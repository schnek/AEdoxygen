30//////////////////////////////////////////////////////////////////////////////////////////
31///\todo problem : some players advance more then others by using speed cheat
32/// solution : check the amount of advance a player does over time to see if it is higher then should be
33/// situations :
34/// - player plays with mouse triggering a setfacing packet every 16 ms would flood us with incorrect values
35/// - client sends us every 500 ms a movement update
36/// - server makes speed changes -> need to reset detector
37/// - server uses scripts over character -> delay detector
38/// error recovery :
39/// - speed change is detected then system resets state to start measuring again
40/// - speed is measured over interval to avoid small value problems
41/// - cheat detection has to stack up to trigger real cheat alarm
42/// problems :
43/// - it takes SPDT_WINDOW_SIZE*SPDT_SAMPLINGRATE at each speed change until it gives relevant data
44/// - moving in a small area is not considered cheating (it is hard to make that)
45//////////////////////////////////////////////////////////////////////////////////////////
46class SpeedCheatDetector
47{
48public:
49 SpeedCheatDetector();
50
51void addSample(LocationVector v, int timestamp, float speed);
52 /// update the detector with new values
53 void AddSample(float x, float y, int stamp, float player_speed);
54 /// test cheater status
55 inline char IsCheatDetected() { return cheat_threat >= CHEAT_ALARMS_TO_TRIGGER_CHEAT; }
56 /// delay then reset cheat detector
57 void SkipSamplingUntil(int stamp);
58 /// take actions against a cheater
59 void ReportCheater(Player* _player);
60 /// reset internal values on speed change
61 void EventSpeedChange();
62
63private:
64 float last_x, last_y;
65 int last_stamp;
66 signed char cheat_threat; /// don't draw quick conclusions. If player is suspicious over time then kill him
67 float last_used_speed; /// we reset if speed changed since our last measure
68 float bigest_hacked_speed_dif;
69};
