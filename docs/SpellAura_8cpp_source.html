47void Aura::addAuraEffect(AuraEffect auraEffect, int32_t damage, int32_t miscValue, float_t effectPctModifier, bool isStaticDamage, uint8_t effIndex, bool reapplying/* = false*/)
48{
49 if (effIndex >= MAX_SPELL_EFFECTS)
50 return;
51
52 if (auraEffect >= TOTAL_SPELL_AURAS)
53 {
54 sLogger.failure("Aura::addAuraEffect : Unknown aura effect type {} for spell id {}", auraEffect, getSpellId());
55 return;
56 }
57
58 if (m_auraEffects[effIndex].getAuraEffectType() != SPELL_AURA_NONE)
59 {
60 sLogger.failure("Aura::addAuraEffect : Tried to add effect to index {} but effect already exists", effIndex);
61 return;
62 }
63
64 m_auraEffects[effIndex].setAuraEffectType(auraEffect);
65 m_auraEffects[effIndex].setEffectDamage(reapplying ? damage * getStackCount() : damage);
66 m_auraEffects[effIndex].setEffectBaseDamage(damage);
67 m_auraEffects[effIndex].setEffectMiscValue(miscValue);
68 m_auraEffects[effIndex].setEffectPercentModifier(effectPctModifier);
69 m_auraEffects[effIndex].setEffectDamageStatic(isStaticDamage);
70 m_auraEffects[effIndex].setEffectIndex(effIndex);
71 m_auraEffects[effIndex].setAura(this);
72 ++m_auraEffectCount;
73
74 // Add aura effect to unit only if aura has a slot
75 if (m_auraSlot != 0xFFFF)
76 getOwner()->_addAuraEffect(&m_auraEffects[effIndex]);
77
78 if (!reapplying)
79 {
80 // Calculate effect amplitude
81 _calculateEffectAmplitude(effIndex);
82 }
83}

94void Aura::removeAuraEffect(uint8_t effIndex, bool reapplying/* = false*/)
95{
96 if (effIndex >= MAX_SPELL_EFFECTS)
97 return;
98
99 // Remove aura effect from unit
100 getOwner()->_removeAuraEffect(&m_auraEffects[effIndex]);
101
102 if (!reapplying)
103 {
104 // Unapply the modifier
105 m_auraEffects[effIndex].applyEffect(false);
106 }
107
108 m_auraEffects[effIndex].setAuraEffectType(SPELL_AURA_NONE);
109 m_auraEffects[effIndex].setEffectDamage(0.0f);
110 m_auraEffects[effIndex].setEffectBaseDamage(0);
111 m_auraEffects[effIndex].setEffectFixedDamage(0);
112 m_auraEffects[effIndex].setEffectMiscValue(0);
113 m_auraEffects[effIndex].setEffectAmplitude(0);
114 m_auraEffects[effIndex].setEffectDamageFraction(0.0f);
115 m_auraEffects[effIndex].setEffectPercentModifier(1.0f);
116 m_auraEffects[effIndex].setEffectDamageStatic(false);
117 m_auraEffects[effIndex].setEffectIndex(0);
118 m_auraEffects[effIndex].setAura(nullptr);
119 --m_auraEffectCount;
120
121 if (!reapplying)
122 {
123 // Check aura effects on next update
124 m_checkAuraEffects = true;
125 }

173void Aura::removeAura(AuraRemoveMode mode/* = AURA_REMOVE_BY_SERVER*/)
174{
175 sEventMgr.RemoveEvents(this);
176
177 // Check if aura has already been removed
178 if (m_isGarbage)
179 return;
180
181 sScriptMgr.callScriptedAuraOnRemove(this, mode);
182 sHookInterface.OnAuraRemove(this);
183
184 sLogger.debug("Removing aura {} from unit {}", getSpellId(), getOwner()->getGuid());
185
186 m_isGarbage = true;
187
188 // Reset diminishing return timer
189 getOwner()->removeDiminishingReturnTimer(getSpellInfo());
190
191 const auto caster = GetUnitCaster();
192 if (caster != nullptr)
193 {
194 /**********************Cooldown**************************
195 * this is only needed for some spells
196 * for now only spells that have:
197 * (m_spellInfo->Attributes == 0x2050000) && !(m_spellInfo->AttributesEx) ||
198 * m_spellProto->Attributes == 0x2040100
199 * are handled. Its possible there are more spells like this
200 *************************************************************/
201 if (caster->isPlayer() && caster->IsInWorld() && getSpellInfo()->custom_c_is_flags & SPELL_FLAG_IS_REQUIRECOOLDOWNUPDATE)
202 {
203 const auto p = static_cast<Player*>(caster);
204 p->sendSpellCooldownEventPacket(getSpellInfo()->getId());
205 }
206
207 // Far Sight effect
208 ///\ todo: handle this in effects
209 if (caster->isPlayer() && caster->IsInWorld() && static_cast<Player*>(caster)->getFarsightGuid() != 0)
210 {
211 uint8_t j = 0;
212 for (; j < MAX_SPELL_EFFECTS; ++j)
213 if (getSpellInfo()->getEffect(j) == SPELL_EFFECT_ADD_FARSIGHT)
214 break;
215
216 if (j != MAX_SPELL_EFFECTS)
217 static_cast<Player*>(caster)->setFarsightGuid(0);
218 }
219
220 // If this aura can affect one target at a time, remove this target from the caster map
221 if (getSpellInfo()->getAttributesExE() & ATTRIBUTESEXE_SINGLE_TARGET_AURA && getOwner()->getAuraCountForId(getSpellId()) == 1)
222 getOwner()->removeSingleTargetGuidForAura(getSpellInfo()->getId());
223
224 // only remove channel stuff if caster == target, then it's not removed twice, for example, arcane missiles applies a dummy aura to target
225 if (caster == getOwner() && m_spellInfo->getChannelInterruptFlags() != 0)
226 {
227 caster->setChannelObjectGuid(0);
228 caster->setChannelSpellId(0);
229 }
230
231 if (caster->isPlayer() && getSpellInfo()->hasEffect(SPELL_EFFECT_SUMMON))
232 {
233 const auto charm = caster->getWorldMapUnit(caster->getCharmGuid());
234 if (charm != nullptr && charm->getCreatedBySpellId() == getSpellInfo()->getId())
235 caster->unPossess();
236 }
237 }
238
239 // Clear area aura targets
240 if (IsAreaAura() && getCasterGuid() == getOwner()->getGuid())
241 ClearAATargets();
242
243 // Send packet before removing modifiers
244 // Otherwise shapeshift spells can still appear active on stance bar -Appled
245 if (m_visualSlot < AuraSlots::NEGATIVE_VISUAL_SLOT_END)
246 {
247#if VERSION_STRING < WotLK
248 getOwner()->setAura(this, false);
249 getOwner()->setAuraFlags(this, false);
250 getOwner()->setAuraLevel(this);
251 getOwner()->setAuraApplication(this);
252#endif
253
254 getOwner()->m_auraVisualList[m_visualSlot] = 0;
255 getOwner()->sendAuraUpdate(this, true);
256 getOwner()->updateAuraForGroup(m_visualSlot);
257 }
258
259 // Remove aura from unit before removing modifiers
260 auto auraHolder = getOwner()->_removeAura(this);
261
262 // Remove all modifiers
263 removeAllAuraEffects();
264
265 // Remove aurastates
266 if (getSpellInfo()->getMechanicsType() == MECHANIC_ENRAGED && !--m_target->m_ascEnraged)
267 getOwner()->removeAuraStateAndAuras(AURASTATE_FLAG_ENRAGED);
268 else if (getSpellInfo()->getMechanicsType() == MECHANIC_BLEEDING && !--m_target->m_ascBleed)
269 getOwner()->removeAuraStateAndAuras(AURASTATE_FLAG_BLEED);
270 if (getSpellInfo()->custom_BGR_one_buff_on_target & SPELL_TYPE_SEAL && !--m_target->m_ascSeal)
271 getOwner()->removeAuraStateAndAuras(AURASTATE_FLAG_JUDGEMENT);
272
273 getOwner()->addGarbageAura(std::move(auraHolder));
274}

281bool Aura::canPeriodicEffectCrit()
282{
283#if VERSION_STRING < WotLK
284 // Periodic effects never crit in Classic or TBC
285 return false;
286#elif VERSION_STRING == WotLK
287 const auto caster = GetUnitCaster();
288 if (caster == nullptr)
289 return false;
290
291 const auto spellInfo = getSpellInfo();
292
293 if (caster->isPlayer())
294 {
295 // Patch 3.3.3 - The damage-over-time component of Rupture can now produce critical strikes
296 if (spellInfo->getSpellFamilyName() == SPELLFAMILY_ROGUE && spellInfo->getSpellFamilyFlags(0) == 0x100000)
297 return true;
298 }
299
300 for (const auto& aurEff : caster->getAuraEffectList(SPELL_AURA_ALLOW_DOT_TO_CRIT))
301 {
302 if (aurEff->getAura()->getSpellInfo()->isAuraEffectAffectingSpell(SPELL_AURA_ALLOW_DOT_TO_CRIT, spellInfo))
303 return true;
304 }
305
306 return false;
307#else
308 ///\ todo: confirm can ALL periodic effects crit after cata
309 const auto caster = GetUnitCaster();
310 if (caster != nullptr && caster->isPlayer())
311 return true;
312
313 return false;
314#endif
315}

337void Aura::updateModifiers()
338{
339 m_updatingModifiers = true;
340 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
341 {
342 switch (m_auraEffects[i].getAuraEffectType())
343 {
344 case SPELL_AURA_MOD_DECREASE_SPEED:
345 UpdateAuraModDecreaseSpeed(&m_auraEffects[i]);
346 break;
347#if VERSION_STRING >= TBC
348#if VERSION_STRING >= WotLK
349 case SPELL_AURA_MOD_ATTACK_POWER_BY_STAT_PCT:
350#endif
351 case SPELL_AURA_MOD_RANGED_ATTACK_POWER_BY_STAT_PCT:
352 m_auraEffects[i].applyEffect(false);
353 m_auraEffects[i].applyEffect(true);
354 break;
355#endif
356 default:
357 break;
358 }
359
360 sLogger.debug("Aura::updateModifiers : Spell Id {}, Aura Effect {} ({}), Target GUID {}, EffectIndex {}, Duration {}, Damage {}, MiscValue {}",
361 getSpellInfo()->getId(), m_auraEffects[i].getAuraEffectType(), SpellAuraNames[m_auraEffects[i].getAuraEffectType()], getOwner()->getGuid(), i, getTimeLeft(), m_auraEffects[i].getEffectDamage(), m_auraEffects[i].getEffectMiscValue());
362 }
363 m_updatingModifiers = false;
364}

427void Aura::refreshOrModifyStack([[maybe_unused]]bool saveMods/* = false*/, int16_t modifyStackAmount/* = 0*/)
428{
429 int32_t maxStacks = getSpellInfo()->getMaxstack() == 0 ? 1 : static_cast<int32_t>(getSpellInfo()->getMaxstack());
430
431 // Check for aura stack cheat
432 const auto plrHolder = getPlayerOwner();
433 if (plrHolder != nullptr && plrHolder->m_cheats.hasAuraStackCheat)
434 maxStacks = 255;
435
436 const auto curStackCount = getStackCount();
437 int32_t newStackCount = curStackCount + modifyStackAmount;
438
439 if (modifyStackAmount < 0)
440 {
441 // If stack count reaches zero, remove aura
442 if (newStackCount <= 0)
443 {
444 removeAura();
445 return;
446 }
447 }
448 else
449 {
450 if (newStackCount >= maxStacks)
451 newStackCount = maxStacks;
452
453 // Recalculate aura modifiers on reapply or when stack count increases
454#if VERSION_STRING < Cata
455 // Before cata, aura's crit chance, attack power bonus and damage bonuses (NYI) were saved on aura refresh in special situations
456 if (!saveMods)
457 {
458 _calculateAttackPowerBonus();
459 _calculateCritChance();
460 }
461#else
462 _calculateAttackPowerBonus();
463 _calculateCritChance();
464#endif
465 _calculateSpellPowerBonus();
466 _calculateSpellHaste();
467
468 // Recalculate aura duration and effect amplitudes based on new haste
469 setMaxDuration(static_cast<int32_t>(m_originalDuration * m_spellHaste));
470 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
471 {
472 if (m_auraEffects[i].getAuraEffectType() == SPELL_AURA_NONE)
473 continue;
474
475 _calculateEffectAmplitude(i);
476 }
477
478 // Reset charges to max count
479 if (m_originalCharges > 0)
480 setCharges(m_originalCharges, false);
481
482 // Reset duration to max
483 setTimeLeft(getMaxDuration());
484 }
485
486 m_updatingModifiers = true;
487 const auto unsignedStackCount = static_cast<uint8_t>(newStackCount);
488 m_stackCount = unsignedStackCount;
489
490 // Reapply modifiers
491 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
492 {
493 if (m_auraEffects[i].getAuraEffectType() != SPELL_AURA_NONE)
494 {
495 m_auraEffects[i].applyEffect(false, true);
496 m_auraEffects[i].setEffectDamage(m_auraEffects[i].getEffectBaseDamage() * m_stackCount);
497 m_auraEffects[i].applyEffect(true, true);
498 }
499 }
500
501 m_updatingModifiers = false;
502 takeUsedSpellModifiers();
503
504 // Call script hook
505 sScriptMgr.callScriptedAuraOnRefreshOrGainNewStack(this, unsignedStackCount, curStackCount);
506
507#if VERSION_STRING < WotLK
508 getOwner()->setAuraApplication(this);
509#endif
510
511 // Send aura update
512 getOwner()->sendAuraUpdate(this, false);
513 getOwner()->updateAuraForGroup(m_visualSlot);
514}

635void Aura::takeUsedSpellModifiers()
636{
637 if (m_usedModifiers.empty())
638 return;
639
640 for (auto itr = m_usedModifiers.begin(); itr != m_usedModifiers.end();)

648 }
649
650 aurEff->getAura()->removeCharge();

652 }

682void Aura::update(unsigned long diff, bool skipDurationCheck/* = false*/)
683{
684 // Do not update trash
685 if (m_isGarbage)
686 return;
687
688 // Do not update aura while updating modifiers
689 if (m_updatingModifiers)
690 return;
691
692 // Check aura effects if aura has been flagged for check
693 if (m_checkAuraEffects)
694 {
695 if (m_auraEffectCount == 0)
696 {
697 // Aura is active but has no effects -> remove aura
698 removeAura();
699 return;
700 }
701
702 m_checkAuraEffects = false;
703 }
704
705 // Periodic auras
706 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
707 {
708 if (m_periodicTimer[i] == 0)
709 continue;
710 if (m_auraEffects[i].getAuraEffectType() == SPELL_AURA_NONE)
711 continue;
712 // Effects with 0 amplitude are not periodic
713 if (m_auraEffects[i].getEffectAmplitude() <= 0)
714 continue;
715
716 m_periodicTimer[i] -= diff;
717 if (m_periodicTimer[i] <= 0)
718 {
719 periodicTick(&m_auraEffects[i]);
720 m_periodicTimer[i] += m_auraEffects[i].getEffectAmplitude();
721 }
722 }
723
724 // Check if aura is about to expire
725 if (m_duration > 0 && !skipDurationCheck)
726 {
727 m_duration -= diff;
728 if (m_duration <= 0)
729 {
730 m_duration = 0;
731 removeAura(AURA_REMOVE_ON_EXPIRE);
732 }
733 }
