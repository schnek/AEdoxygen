59void SpellCastTargets::read(WorldPacket& data, uint64_t caster)
60{
61 reset();
62
63 data >> m_targetMask;
64
65 if (m_targetMask == TARGET_FLAG_SELF)
66 {
67 m_unitTargetGuid = caster;
68 return;
69 }
70
71 if (m_targetMask & (TARGET_FLAG_OBJECT | TARGET_FLAG_OPEN_LOCK))
72 {
73 WoWGuid guid;
74 data >> guid;
75 m_gameObjectTargetGuid = guid.getRawGuid();
76 }
77
78 if (m_targetMask & (TARGET_FLAG_UNIT | TARGET_FLAG_CORPSE | TARGET_FLAG_CORPSE2 | TARGET_FLAG_UNK17))
79 {
80 WoWGuid guid;
81 data >> guid;
82 m_unitTargetGuid = guid.getRawGuid();
83 }
84
85 if (m_targetMask & (TARGET_FLAG_ITEM | TARGET_FLAG_TRADE_ITEM))
86 {
87 WoWGuid guid;
88 data >> guid;
89 m_itemTargetGuid = guid.getRawGuid();
90 }
91
92 if (m_targetMask & TARGET_FLAG_SOURCE_LOCATION)
93 {
94 WoWGuid sourceGuid;
95
96 data >> sourceGuid;
97 unkuint64_1 = sourceGuid.getRawGuid();
98
99 LocationVector lv;
100 data >> lv.x;
101 data >> lv.y;
102 data >> lv.z;
103
104 setSource(lv);
105 }
106
107 if (m_targetMask & TARGET_FLAG_DEST_LOCATION)
108 {
109 WoWGuid destinationGuid;
110 data >> destinationGuid;
111 unkuint64_2 = destinationGuid.getRawGuid();
112
113 auto transporter = sTransportHandler.getTransporter(destinationGuid.getGuidLow());
114
115 LocationVector lv;
116 data >> lv.x;
117 data >> lv.y;
118 data >> lv.z;
119
120 if (transporter)
121 {
122 transporter->calculatePassengerPosition(lv.x, lv.y, lv.z);
123 }
124
125 setDestination(lv);
126 }
127
128 if (m_targetMask & TARGET_FLAG_STRING)
129 {
130 data >> m_strTarget;
131 }
132}

134void SpellCastTargets::write(WorldPacket& data) const
135{
136 data << m_targetMask;
137
138 if (m_targetMask & (TARGET_FLAG_OBJECT | TARGET_FLAG_OPEN_LOCK))
139 {
140 FastGUIDPack(data, m_gameObjectTargetGuid);
141 }
142
143 if (m_targetMask & (TARGET_FLAG_UNIT | TARGET_FLAG_CORPSE | TARGET_FLAG_CORPSE2 | TARGET_FLAG_UNK17))
144 {
145 FastGUIDPack(data, m_unitTargetGuid);
146 }
147
148 if (m_targetMask & (TARGET_FLAG_ITEM | TARGET_FLAG_TRADE_ITEM))
149 {
150 FastGUIDPack(data, m_itemTargetGuid);
151 }
152
153 if (m_targetMask & TARGET_FLAG_SOURCE_LOCATION)
154 {
155 data << WoWGuid(unkuint64_1);
156 data << m_source;
157 }
158
159 if (m_targetMask & TARGET_FLAG_DEST_LOCATION)
160 {
161 data << WoWGuid(unkuint64_2);
162 data << m_destination;
163 }
164
165 if (m_targetMask & TARGET_FLAG_STRING)
166 {
167 data << m_strTarget.c_str();
168 }
169}
