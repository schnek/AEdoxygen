24struct SpellForcedBasePoints
25{
26 void set(uint8_t effIndex, int32_t value)
27 {
28 if (effIndex >= MAX_SPELL_EFFECTS)
29 return;
30
31 for (auto& values : m_forcedBasePoints)
32 {
33 if (values.first == effIndex)
34 {
35 values.second = value;
36 return;
37 }
38 }
39
40 m_forcedBasePoints.push_back(std::make_pair(effIndex, value));
41 }
42
43 void get(uint8_t effIndex, int32_t* basePoints) const
44 {
45 if (effIndex >= MAX_SPELL_EFFECTS)
46 return;
47
48 for (const auto& values : m_forcedBasePoints)
49 {
50 if (values.first == effIndex)
51 {
52 *basePoints = values.second;
53 break;
54 }
55 }
56 }
57
58private:
59 std::vector<std::pair<uint8_t, int32_t>> m_forcedBasePoints;
60};
