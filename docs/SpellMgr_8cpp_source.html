32bool SpellArea::fitsToRequirements(Player* player, uint32_t newZone, uint32_t newArea) const
33{
34 if (gender != GENDER_NONE)
35 {
36 if (player == nullptr || gender != player->getGender())
37 return false;
38 }
39
40 if (raceMask != 0)
41 {
42 if (player == nullptr || !(raceMask & player->getRaceMask()))
43 return false;
44 }
45
46 if (areaId != 0)
47 {
48 if (newZone != areaId && newArea != areaId)
49 return false;
50 }
51
52 if (questStart != 0)
53 {
54 if (player == nullptr || !player->hasQuestInQuestLog(questStart))
55 return false;
56 }
57
58 if (questEnd != 0)
59 {
60 if (player == nullptr || player->hasQuestFinished(questEnd))
61 return false;
62 }
63
64 if (auraSpell != 0)
65 {
66 const auto auraId = static_cast<uint32_t>(std::abs(auraSpell));
67 if (player == nullptr || (auraSpell > 0 && !player->hasAurasWithId(auraId)) || (auraSpell < 0 && player->hasAurasWithId(auraId)))
68 return false;
69 }
70
71 // Misc Conditions
72 switch (spellId)
73 {
74 case 58600: // Restricted Flight Zone (Dalaran)
75 {
76 if (player == nullptr)
77 return false;
78
79#if VERSION_STRING >= TBC
80 if (!player->hasAuraWithAuraEffect(SPELL_AURA_ENABLE_FLIGHT2) && !player->hasAuraWithAuraEffect(SPELL_AURA_FLY))
81 return false;
82#endif
83 break;
84 }
85 }
86
87 return true;
88}

174std::unique_ptr<Aura> SpellMgr::newAura(SpellInfo const* spellInfo, int32_t duration, Object* caster, Unit* target, bool temporary /*= false*/, Item* i_caster /*= nullptr*/)
175{
176 //\brief... nullptr when downgrading ae from wotlk to tbc (active auras from newer client versions should be removed before entering tbc)
177
178 if (spellInfo == nullptr)
179 {
180 sLogger.failure("You tried to create an Aura without SpellInfo. This is not possible!");
181 return nullptr;
182 }
183
184 // Auras with a script
185 if (spellInfo->auraScriptLink != nullptr)
186 return (*AuraScriptLinker(spellInfo->auraScriptLink))(getMutableSpellInfo(spellInfo->getId()), duration, caster, target, temporary, i_caster);
187 // Auras with absorb effect
188 else if (spellInfo->hasEffectApplyAuraName(SPELL_AURA_SCHOOL_ABSORB))
189 return (*AuraScriptLinker(&AbsorbAura::Create))(getMutableSpellInfo(spellInfo->getId()), duration, caster, target, temporary, i_caster);
190
191 // Standard auras without a script
192 return std::make_unique<Aura>(spellInfo, duration, caster, target, temporary, i_caster);
193}

392bool SpellMgr::checkLocation(SpellInfo const* spellInfo, uint32_t zone_id, uint32_t area_id, Player* player) const
393{
394 if (spellInfo == nullptr)
395 return false;
396
397 // Normal case
398 if (spellInfo->getRequiresAreaId() > 0)
399 {
400 const auto requireAreaId = static_cast<uint32_t>(spellInfo->getRequiresAreaId());
401#if VERSION_STRING == TBC
402 if (requireAreaId != zone_id && requireAreaId != area_id)
403 return false;
404#elif VERSION_STRING >= WotLK
405 auto found = false;
406 auto areaGroup = sAreaGroupStore.lookupEntry(requireAreaId);
407 while (areaGroup != nullptr)
408 {
409 for (uint8_t i = 0; i < 6; ++i)
410 {
411 if (areaGroup->AreaId[i] == zone_id || areaGroup->AreaId[i] == area_id)
412 found = true;
413 }
414
415 if (found || areaGroup->next_group == 0)
416 break;
417
418 // Try next group
419 areaGroup = sAreaGroupStore.lookupEntry(areaGroup->next_group);
420 }
421
422 if (!found)
423 return false;
424#endif
425 }
426
427 // Check for area requirements set in database
428 const auto spellAreaMapBounds = getSpellAreaMapBounds(spellInfo->getId());
429 if (spellAreaMapBounds.first != spellAreaMapBounds.second)
430 {
431 for (SpellAreaMap::const_iterator itr = spellAreaMapBounds.first; itr != spellAreaMapBounds.second; ++itr)
432 {
433 if (itr->second.fitsToRequirements(player, zone_id, area_id))
434 return true;
435 }
436 }
437
438 return false;
439}

1745void SpellMgr::loadSpellRanks()
1746{
1747#if VERSION_STRING < Mop
1748 // Load talent ranks from DBC before spell ranks from database
1749 loadTalentRanks();
1750#endif
1751
1752 // 0 1 2
1753 const auto result = WorldDatabase.Query("SELECT `spell_id`, `first_spell`, `rank` "
1754 "FROM spell_ranks WHERE `min_build` <= %u AND `max_build` >= %u ORDER BY `first_spell`, `rank`", VERSION_STRING, VERSION_STRING);
1755
1756 if (result == nullptr)
1757 {
1758 sLogger.debug("SpellMgr::loadSpellRanks : Your `spell_ranks` table is empty");
1759 return;
1760 }
1761
1762 std::map<uint8_t/*rank*/, uint32_t/*spell id*/> spellRankChain{};
1763 uint32_t totalSpellRankChains = 0;
1764
1765 const auto createSpellRankChain = [&spellRankChain, this, &totalSpellRankChains]() -> void
1766 {
1767 if (spellRankChain.empty())
1768 return;
1769
1770 if (spellRankChain.size() < 2)
1771 {
1772 sLogger.failure("SpellMgr::loadSpellRanks : Spell rank chain for spell id {} must contain at least 2 spells, skipping", spellRankChain.cbegin()->second);
1773 spellRankChain.clear();
1774 return;
1775 }
1776
1777 std::map<uint32_t/*spell id*/, SpellRankInfo> preparedSpellRanks{};
1778 auto areChainSpellsValid = true, areChainRanksValid = true, foundDuplicateChain = false;
1779 uint32_t invalidSpellId = 0;
1780 uint8_t invalidSpellRank = 0, expectedSpellRank = 0;
1781
1782 uint32_t previousSpellId = 0;
1783 uint8_t previousRank = 0;
1784 for (auto itr = spellRankChain.cbegin(); itr != spellRankChain.cend();)
1785 {
1786 const auto spellRank = itr->first;
1787 const auto spellId = itr->second;
1788
1789 if (spellRank == 0)
1790 {
1791 areChainRanksValid = false;
1792 invalidSpellRank = spellRank;
1793 invalidSpellId = spellId;
1794 expectedSpellRank = 1;
1795 break;
1796 }
1797 else if (spellRank > 1 && spellRank != (previousRank + 1))
1798 {
1799 areChainRanksValid = false;
1800 invalidSpellRank = spellRank;
1801 invalidSpellId = spellId;
1802 expectedSpellRank = previousRank + 1;
1803 break;
1804 }
1805
1806 const auto* const spellInfo = getSpellInfo(spellId);
1807 if (spellInfo == nullptr)
1808 {
1809 areChainSpellsValid = false;
1810 invalidSpellId = spellId;
1811 break;
1812 }
1813
1814 if (spellInfo->hasSpellRanks())
1815 {
1816 foundDuplicateChain = true;
1817 invalidSpellId = spellId;
1818 break;
1819 }
1820
1821 SpellRankInfo rankInfo{};
1822 rankInfo.previousSpell = previousSpellId != 0 ? getSpellInfo(previousSpellId) : nullptr;
1823 rankInfo.firstSpell = getSpellInfo(spellRankChain.cbegin()->second);
1824 rankInfo.lastSpell = getSpellInfo(spellRankChain.crbegin()->second);
1825 rankInfo.rank = spellRank;
1826
1827 previousSpellId = spellId;
1828 previousRank = spellRank;
1829 ++itr;
1830
1831 rankInfo.nextSpell = itr != spellRankChain.cend() ? getSpellInfo(itr->second) : nullptr;
1832
1833 // Spell rank data is ready but validate and prepare the entire chain before saving it
1834 preparedSpellRanks.insert({ spellId, rankInfo });
1835 }
1836
1837 if (!areChainSpellsValid)
1838 {
1839 sLogger.failure("SpellMgr::loadSpellRanks : Spell rank chain contains invalid spell id {}, skipping this chain", invalidSpellId);
1840 spellRankChain.clear();
1841 return;
1842 }
1843
1844 if (!areChainRanksValid)
1845 {
1846 sLogger.failure("SpellMgr::loadSpellRanks : Spell rank chain contains invalid spell rank {} for spell id {} (expected {}), skipping this chain",
1847 invalidSpellRank, invalidSpellId, expectedSpellRank);
1848 spellRankChain.clear();
1849 return;
1850 }
1851
1852 if (foundDuplicateChain)
1853 {
1854 sLogger.failure("SpellMgr::loadSpellRanks : Spell id {} already contains a spell rank chain, skipping this chain", invalidSpellId);
1855 spellRankChain.clear();
1856 return;
1857 }
1858
1859 // Finally insert validated spell rank data to SpellInfo class
1860 for (const auto& [spellId, rankInfo] : preparedSpellRanks)
1861 {
1862 auto spellInfo = getMutableSpellInfo(spellId);
1863 spellInfo->setSpellRankData(rankInfo);
1864 }
1865
1866 spellRankChain.clear();
1867 ++totalSpellRankChains;
1868 };
1869
1870 uint32_t lastFetchedSpellId = 0;
1871 do
1872 {
1873 const auto fields = result->Fetch();
1874 const auto spellId = fields[0].asUint32();
1875 const auto firstSpellId = fields[1].asUint32();
1876 const auto rank = fields[2].asUint8();
1877
1878 if (lastFetchedSpellId == 0)
1879 lastFetchedSpellId = firstSpellId;
1880
1881 // First spell id changes => create a rank chain for each rank before continuing
1882 if (lastFetchedSpellId != firstSpellId)
1883 {
1884 createSpellRankChain();
1885 lastFetchedSpellId = firstSpellId;
1886 }
1887
1888 spellRankChain.insert({ rank, spellId });
1889 } while (result->NextRow());
1890
1891 // Remember to create a rank chain for last chain as well
1892 createSpellRankChain();
1893
1894 sLogger.info("SpellMgr : Loaded {} spell rank chains from `spell_ranks` table", totalSpellRankChains);
1895}
