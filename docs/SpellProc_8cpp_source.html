30bool SpellProc::checkExtraProcFlags([[maybe_unused]]Unit* procOwner, DamageInfo damageInfo) const
31{
32 // Quick check
33 if (mExtraProcFlags == EXTRA_PROC_NULL)
34 return true;
35
36 // Main hand hits should not proc off-hand enchantments/poisons
37 if (damageInfo.weaponType == MELEE && mExtraProcFlags & EXTRA_PROC_ON_OFF_HAND_HIT_ONLY)
38 return false;
39 // Off-hand hits should not proc main hand enchantments/poisons
40 if (damageInfo.weaponType == OFFHAND && mExtraProcFlags & EXTRA_PROC_ON_MAIN_HAND_HIT_ONLY)
41 return false;
42
43#if VERSION_STRING < WotLK
44 // In Classic and TBC weapon enchantments never proc while in feral forms
45 if (procOwner->isPlayer() && static_cast<Player*>(procOwner)->isInFeralForm() &&
46 (mExtraProcFlags & EXTRA_PROC_ON_MAIN_HAND_HIT_ONLY ||
47 mExtraProcFlags & EXTRA_PROC_ON_OFF_HAND_HIT_ONLY))
48 return false;
49#endif
50
51 // Check on crit procs
52 if (mExtraProcFlags & EXTRA_PROC_ON_CRIT_ONLY && !damageInfo.isCritical)
53 return false;
54
55 return true;
56}

66bool SpellProc::checkClassMask(SpellInfo const* castingSpell) const
67{
68 if (mProcClassMask[0] == 0 && mProcClassMask[1] == 0 && mProcClassMask[2] == 0)
69 return true;
70
71 if (castingSpell->getSpellFamilyFlags(0) == 0 && castingSpell->getSpellFamilyFlags(1) == 0 && castingSpell->getSpellFamilyFlags(2) == 0)
72 return false;
73
74 if (((castingSpell->getSpellFamilyFlags(0) == 0) || (mProcClassMask[0] & castingSpell->getSpellFamilyFlags(0))) &&
75 ((castingSpell->getSpellFamilyFlags(1) == 0) || (mProcClassMask[1] & castingSpell->getSpellFamilyFlags(1))) &&
76 ((castingSpell->getSpellFamilyFlags(2) == 0) || (mProcClassMask[2] & castingSpell->getSpellFamilyFlags(2))))
77 return true;
78
79 return false;
80}

87uint32_t SpellProc::calcProcChance(Unit* /*victim*/, SpellInfo const* /*castingSpell*/)
88{
89 // Check if proc chance is based on combo points
90 if (mOwner->isPlayer() && mOrigSpell != nullptr && mOrigSpell->getAttributesEx() & ATTRIBUTESEX_REQ_COMBO_POINTS1 && mOrigSpell->getAttributesExD() & ATTRIBUTESEXD_PROCCHANCE_COMBOBASED)
91 return static_cast<uint32_t>(static_cast<Player*>(mOwner)->getComboPoints() * mOrigSpell->getEffectPointsPerComboPoint(0));
92 else
93 return mProcChance;

96void SpellProc::castSpell(Unit* victim, SpellInfo const* castingSpell)
97{
98 if (isCastedOnProcOwner())
99 victim = getProcOwner();
100
101 if (victim == nullptr)
102 return;
103
104 Unit* caster = getProcOwner();
105 if (isCastedByProcCreator())
106 {
107 if (getCasterGuid() == getProcOwner()->getGuid())
108 caster = getProcOwner();
109 else
110 caster = getProcOwner()->getWorldMapUnit(getCasterGuid());
111 }
112
113 if (caster == nullptr)
114 return;
115
116 SpellCastTargets targets(victim->getGuid());
117 Spell* spell = sSpellMgr.newSpell(caster, mSpell, true, nullptr);
118
119 spell->forced_basepoints = mOverrideEffectDamage;
120
121 spell->ProcedOnSpell = castingSpell;
122 if (mOrigSpell != nullptr)
123 spell->pSpellId = mOrigSpell->getId();
124
125 // Final script hook before casting the spell
126 const auto scriptResult = sScriptMgr.callScriptedSpellProcCastSpell(this, caster, victim, spell);
127 if (scriptResult == SpellScriptExecuteState::EXECUTE_PREVENT)
128 {
129 // Script prevents this spell cast => delete spell and do nothing
130 caster->addGarbageSpell(spell);
131 return;
132 }
133
134 spell->prepare(&targets);
135}

231std::unique_ptr<SpellProc> SpellProcMgr::newSpellProc(Unit* owner, SpellInfo const* spellInfo, SpellInfo const* origSpellInfo, uint64_t casterGuid, uint32_t procChance, SpellProcFlags procFlags, SpellExtraProcFlags exProcFlags, uint32_t const* spellFamilyMask, uint32_t const* procClassMask, Aura* createdByAura, Object* obj)
232{
233 if (spellInfo == nullptr)
234 return nullptr;
235
236 spell_proc_factory_function ptr = nullptr;
237
238 // Search for SpellProc script
239 const auto itr = mSpellProc.find(spellInfo->getId());
240 if (itr != mSpellProc.end())
241 ptr = itr->second;
242
243 std::unique_ptr<SpellProc> result = nullptr;
244 if (ptr != nullptr)
245 result = (*ptr)(); // Found, create a new object of this specific class
246 else
247 result = std::make_unique<SpellProc>(); // Not found, create a new object of generic SpellProc
248
249 result->mSpell = spellInfo;
250 result->mOrigSpell = origSpellInfo;
251 result->mOwner = owner;
252 result->mCaster = casterGuid;
253 result->mProcChance = procChance;
254 result->mProcFlags = procFlags;
255 result->mExtraProcFlags = exProcFlags;
256 result->mLastTrigger = 0;
257 result->m_createdByAura = createdByAura;
258 result->mDeleted = false;
259
260 if (spellFamilyMask != nullptr)
261 {
262 result->mGroupRelation[0] = spellFamilyMask[0];
263 result->mGroupRelation[1] = spellFamilyMask[1];
264 result->mGroupRelation[2] = spellFamilyMask[2];
265 }
266 else
267 {
268 result->mGroupRelation[0] = 0;
269 result->mGroupRelation[1] = 0;
270 result->mGroupRelation[2] = 0;
271 }
272
273 if (procClassMask != nullptr)
274 {
275 result->mProcClassMask[0] = procClassMask[0];
276 result->mProcClassMask[1] = procClassMask[1];
277 result->mProcClassMask[2] = procClassMask[2];
278 }
279 else
280 {
281 result->mProcClassMask[0] = 0;
282 result->mProcClassMask[1] = 0;
283 result->mProcClassMask[2] = 0;
284 }
285
286 if (sScriptMgr.getSpellScript(spellInfo->getId()) != nullptr)
287 sScriptMgr.callScriptedSpellProcCreate(result.get(), obj);
288 else
289 result->init(obj);
290
291 return result;
