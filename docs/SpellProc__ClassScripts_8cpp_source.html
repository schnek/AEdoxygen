193class FlametongueWeaponSpellProc : public SpellProc
194{
195public:
196 static std::unique_ptr<SpellProc> Create() { return std::make_unique<FlametongueWeaponSpellProc>(); }
197
198 void init(Object* obj) override
199 {
200 if (obj == nullptr)
201 {
202 deleteProc();
203 //initializing anyway all fields.
204 mItemGUID = 0;
205 damage = 0;
206 return;
207 }
208
209 mItemGUID = obj->getGuid();
210 damage = 0;
211 uint32_t wp_speed;
212 Item* item = static_cast< Item* >(obj);
213 EnchantmentInstance* enchant = item->getEnchantment(TEMP_ENCHANTMENT_SLOT);
214 if (enchant != nullptr)
215 {
216 SpellInfo const* sp = sSpellMgr.getSpellInfo(enchant->Enchantment->spell[0]);
217 if (sp != nullptr)
218 {
219 switch (sp->getId())
220 {
221 //SPELL_HASH_FLAMETONGUE_WEAPON__PASSIVE_
222 case 10400:
223 case 15567:
224 case 15568:
225 case 15569:
226 case 16311:
227 case 16312:
228 case 16313:
229 case 58784:
230 case 58791:
231 case 58792:
232 {
233 wp_speed = item->getItemProperties()->Delay;
234 damage = (sp->calculateEffectValue(0)) * wp_speed / 100000;
235 } break;
236 }
237 }
238 }
239 }
240
241 bool canDeleteProc(uint32_t spellId, uint64_t casterGuid = 0, uint64_t misc = 0) override //in this case misc is the item guid.
242 {
243 if (getSpell()->getId() == spellId && getCasterGuid() == casterGuid && misc == mItemGUID && !isDeleted())
244 return true;
245
246 return false;
247 }
248
249 bool canProc(Unit* /*victim*/, SpellInfo const* /*castingSpell*/) override
250 {
251 if (getProcOwner()->isPlayer())
252 return true;
253 return false;
254 }
255
256 bool doEffect(Unit* /*victim*/, SpellInfo const* /*CastingSpell*/, uint32_t /*flag*/, uint32_t /*dmg*/, uint32_t /*abs*/, uint32_t weaponDamageType) override
257 {
258 Item* item;
259
260 if (weaponDamageType == OFFHAND)
261 item = static_cast< Player* >(getProcOwner())->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_OFFHAND);
262 else
263 item = static_cast< Player* >(getProcOwner())->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_MAINHAND);
264
265 if (item != nullptr && item->getGuid() == mItemGUID)
266 {
267 setOverrideEffectDamage(0, damage);
268 return false;
269 }
270
271 return true;
272 }
273
274private:
275 uint64_t mItemGUID;
276 int damage;
277};

281class PoisonSpellProc : public SpellProc
282{
283public:
284 static std::unique_ptr<SpellProc> Create() { return std::make_unique<PoisonSpellProc>(); }
285
286 PoisonSpellProc()
287 {
288 mItemGUID = 0;
289 mProcPerMinute = 0;
290 }
291
292 void init(Object* obj) override
293 {
294 if (obj == nullptr)
295 {
296 deleteProc();
297 return;
298 }
299
300 mItemGUID = obj->getGuid();
301 if (mProcPerMinute)
302 setProcChance(static_cast<Item*>(obj)->getItemProperties()->Delay * mProcPerMinute / 600);
303 }
304
305 bool canDeleteProc(uint32_t spellId, uint64_t casterGuid = 0, uint64_t misc = 0) override//in this case misc is the item guid.
306 {
307 if (getSpell()->getId() == spellId && getCasterGuid() == casterGuid && misc == mItemGUID && !isDeleted())
308 return true;
309
310 return false;
311 }
312
313 // Allow proc on ability cast (like eviscerate, envenom, fan of knives, rupture)
314 bool canProcOnTriggered(Unit* /*victim*/, SpellInfo const* castingSpell)
315 {
316 if (castingSpell != nullptr && (castingSpell->getSpellFamilyFlags(0) & 0x120000 || castingSpell->getSpellFamilyFlags(1) & 0x240008))
317 return true;
318
319 return false;
320 }
321
322 // Allow proc only if proccing hand is the one where poison was applied
323 bool doEffect(Unit* /*victim*/, SpellInfo const* /*castingSpell*/, uint32_t /*flag*/, uint32_t /*dmg*/, uint32_t /*abs*/, uint32_t weaponDamageType) override
324 {
325 Item* item;
326
327 if (weaponDamageType == OFFHAND)
328 item = static_cast<Player*>(getProcOwner())->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_OFFHAND);
329 else
330 item = static_cast<Player*>(getProcOwner())->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_MAINHAND);
331
332 if (item != nullptr && item->getGuid() == mItemGUID)
333 return false;
334
335 return true;
336 }
337
338protected:
339 uint64_t mItemGUID;
340 uint32_t mProcPerMinute;
341};

400 {

406 int ticks = GetDuration(sSpellDurationStore.lookupEntry(castingSpell->getDurationIndex())) / amplitude;
407
408 setOverrideEffectDamage(0, dmg * ticks * (getOriginalSpell()->calculateEffectValue(0)) / 100);
409
410 return false;

414class EmpoweredRenewSpellProc : public SpellProc
415{
416public:
417 static std::unique_ptr<SpellProc> Create() { return std::make_unique<EmpoweredRenewSpellProc>(); }
418
419 bool doEffect(Unit* /*victim*/, SpellInfo const* castingSpell, uint32_t /*flag*/, uint32_t /*dmg*/, uint32_t /*abs*/, uint32_t /*weapon_damage_type*/) override
420 {
421 // Get heal amt for 1 tick
422 const uint32_t dmg = castingSpell->calculateEffectValue(0);
423
424 // Get total ticks
425 int ticks = GetDuration(sSpellDurationStore.lookupEntry(castingSpell->getDurationIndex())) / castingSpell->getEffectAmplitude(0);
426
427 // Total periodic effect is a single tick amount multiplied by number of ticks
428 setOverrideEffectDamage(0, dmg * ticks * (getOriginalSpell()->calculateEffectValue(0)) / 100);
429
430 return false;
431 }
432
433 void castSpell(Unit* victim, SpellInfo const* CastingSpell) override
434 {
435 SpellCastTargets targets(victim->getGuid());
436
437 Spell* spell = sSpellMgr.newSpell(getProcOwner(), getSpell(), true, nullptr);
438 spell->forced_basepoints->set(0, getOverrideEffectDamage(0));
439 spell->forced_basepoints->set(1, getOverrideEffectDamage(1));
440 spell->forced_basepoints->set(2, getOverrideEffectDamage(2));
441 spell->ProcedOnSpell = CastingSpell;
442
443 spell->prepare(&targets);
444 }

461class PrayerOfMendingProc : public SpellProc
462{
463public:
464 static std::unique_ptr<SpellProc> Create() { return std::make_unique<PrayerOfMendingProc>(); }
465
466 bool doEffect(Unit* /*victim*/, SpellInfo const* /*castingSpell*/, uint32_t /*flag*/, uint32_t /*dmg*/, uint32_t /*abs*/, uint32_t /*weaponDamageType*/) override
467 {
468 Aura* aura = getProcOwner()->getAuraWithId(getSpell()->getId());
469 if (aura == nullptr)
470 return true;
471
472 Unit* caster = dynamic_cast<Unit*>(aura->getCaster());
473 if (caster == nullptr)
474 {
475 getProcOwner()->removeAllAurasById(getSpell()->getId());
476 return true;
477 }
478
479 SpellForcedBasePoints forcedBasePoints;
480 forcedBasePoints.set(0, aura->getEffectDamage(0));
481
482 caster->castSpell(getProcOwner(), 33110, forcedBasePoints, true);
483
484 int32_t count = getProcOwner()->getAuraCountForId(getSpell()->getId());
485
486 if (count <= 1)
487 return true;
488
489 if (getProcOwner()->isPlayer())
490 {
491 Player* playerOwner = dynamic_cast<Player*>(getProcOwner());
492
493 if (const auto group = playerOwner->getGroup())
494 {
495 Player* playerRandom = group->GetRandomPlayerInRangeButSkip(playerOwner, 40.0f, playerOwner);
496 getProcOwner()->removeAllAurasById(getSpell()->getId());
497
498 if (playerRandom)
499 caster->castSpell(playerRandom, getSpell(), forcedBasePoints, count - 1, true);
500 }
501 }
502
503 return true;
504 }
