28class TwinBladesOfAzzinothSpellProc : public SpellProc
29{
30public:
31 static std::unique_ptr<SpellProc> Create() { return std::make_unique<TwinBladesOfAzzinothSpellProc>(); }
32
33 void init(Object* /*obj*/) override
34 {
35 if (!getProcOwner()->isPlayer())
36 return;
37
38 /* The Twin Blades of Azzinoth.
39 * According to comments on wowhead, this proc has ~0.75ppm (procs-per-minute). */
40 Item* mh = static_cast<Player*>(getProcOwner())->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_MAINHAND);
41 Item* of = static_cast<Player*>(getProcOwner())->getItemInterface()->GetInventoryItem(EQUIPMENT_SLOT_OFFHAND);
42 if (mh != nullptr && of != nullptr)
43 {
44 uint32_t mhs = mh->getItemProperties()->Delay;
45 uint32_t ohs = of->getItemProperties()->Delay;
46 setProcChance(mhs * ohs / (800 * (mhs + ohs))); // 0.75 ppm
47 }
48 }
49};
