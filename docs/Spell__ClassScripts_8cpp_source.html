120class CheatDeathAura : public AbsorbAura
121{
122public:
123 CheatDeathAura(SpellInfo* proto, int32_t duration, Object* caster, Unit* target, bool temporary = false, Item* i_caster = nullptr)
124 : AbsorbAura(proto, duration, caster, target, temporary, i_caster)
125 {
126 dSpell = sSpellMgr.getSpellInfo(31231);
127 }
128
129 static std::unique_ptr<Aura> Create(SpellInfo* proto, int32_t duration, Object* caster, Unit* target, bool temporary = false, Item* i_caster = nullptr)
130 {
131 return std::make_unique<CheatDeathAura>(proto, duration, caster, target, temporary, i_caster);
132 }
133
134 uint32_t absorbDamage(SchoolMask /*School*/, uint32_t* dmg, bool checkOnly) override
135 {
136 // Checking for 1 min cooldown
137 if (dSpell == NULL || getPlayerOwner()->hasSpellOnCooldown(dSpell))
138 return 0;
139
140 // Check for proc chance
141 if (Util::getRandomFloat(100.0f) > getSpellInfo()->calculateEffectValue(0))
142 return 0;
143
144 // Check if damage will kill player.
145 uint32_t cur_hlth = getPlayerOwner()->getHealth();
146 if ((*dmg) < cur_hlth)
147 return 0;
148
149 if (checkOnly)
150 return 0;
151
152 uint32_t max_hlth = getPlayerOwner()->getMaxHealth();
153 uint32_t min_hlth = max_hlth / 10;
154
155 /*
156 looks like the following lines are not so good, we check and cast on spell id 31231_
157 and adding the cooldown to it, but it looks like this spell is useless(all it's doing is_
158 casting 45182, so we can do all this stuff on 45182 at first place), BUT_
159 as long as proceeding cheat death is not so height (how many rogue at the same time_
160 gonna get to this point?) so it's better to use it because we wont lose anything!!
161 */
162 getPlayerOwner()->castSpell(getPlayerOwner()->getGuid(), dSpell, true);
163
164 // set dummy effect,
165 // this spell is used to procced the post effect of cheat death later.
166 // Move next line to SPELL::SpellEffectDummy ?!! well it's better in case of dbc changing!!
167 getPlayerOwner()->castSpell(getPlayerOwner()->getGuid(), 45182, true);
168
169 // Better to add custom cooldown procedure then fucking with entry, or not!!
170 getPlayerOwner()->addSpellCooldown(dSpell, nullptr, nullptr, 60000);
171
172 // Calc abs and applying it
173 uint32_t real_dmg = (cur_hlth > min_hlth ? cur_hlth - min_hlth : 0);
174 uint32_t absorbed_dmg = *dmg - real_dmg;
175
176 *dmg = real_dmg;
177 return absorbed_dmg;
178 }
179
180private:
181 SpellInfo const* dSpell;
182};

469
470 int health_pct = caster->getHealthPct();
471 uint32_t cur_health = caster->getHealth();
472 uint32_t max_health = caster->getMaxHealth();
473 uint32_t new_health_pct = (cur_health - *dmg) * 100 / max_health;
474
475 // "Damage that would take you below $s1% health or taken while you are at $s1% health is reduced by $52284s1%."
476 if ((health_pct > 35 && new_health_pct < 35) || health_pct == 35)
477 {
478 uint32_t dmg_absorbed = *dmg * (getSpellInfo()->calculateEffectValue(0)) / 100;
479 *dmg -= dmg_absorbed;
480
481 return dmg_absorbed;
482 }
483 else
484 return 0;
485 }
