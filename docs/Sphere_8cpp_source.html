96void Sphere::merge(const Sphere& other) {
97 if (other.contains(*this)) {
98 *this = other;
99 } else if (! contains(other)) {
100 // The farthest distance is along the axis between the centers, which
101 // must not be colocated since neither contains the other.
102 Vector3 toMe = center - other.center;
103 // Get a point on the axis from each
104 toMe = toMe.direction();
105 const Vector3& A = center + toMe * radius;
106 const Vector3& B = other.center - toMe * other.radius;
107
108 // Now just bound the A->B segment
109 center = (A + B) * 0.5f;
110 radius = (A - B).length();
111 }
112 // (if this contains other, we're done)
113}

135bool Sphere::culledBy(
136 const class Plane* plane,
137 int numPlanes,
138 int& cullingPlane,
139 const uint32 _inMask,
140 uint32& childMask) const {
141
142 if (radius == finf()) {
143 // No plane can cull the infinite box
144 return false;
145 }
146
147 uint32 inMask = _inMask;
148 assert(numPlanes < 31);
149
150 childMask = 0;
151
152 // See if there is one plane for which all of the
153 // vertices are in the negative half space.
154 for (int p = 0; p < numPlanes; ++p) {
155
156 // Only test planes that are not masked
157 if ((inMask & 1) != 0) {
158
159 bool culledLow = ! plane[p].halfSpaceContainsFinite(center + plane[p].normal() * radius);
160 bool culledHigh = ! plane[p].halfSpaceContainsFinite(center - plane[p].normal() * radius);
161
162 if (culledLow) {
163 // Plane p culled the sphere
164 cullingPlane = p;
165
166 // The caller should not recurse into the children,
167 // since the parent is culled. If they do recurse,
168 // make them only test against this one plane, which
169 // will immediately cull the volume.
170 childMask = 1 << p;
171 return true;
172
173 } else if (culledHigh) {
174 // The bounding volume straddled the plane; we have
175 // to keep testing against this plane
176 childMask |= (1 << p);
177 }
178 }
179
180 // Move on to the next bit.
181 inMask = inMask >> 1;
182 }
183
184 // None of the planes could cull this box
185 cullingPlane = -1;
186 return false;
187}

190bool Sphere::culledBy(
191 const class Plane* plane,
192 int numPlanes,
193 int& cullingPlane,
194 const uint32 _inMask) const {
195 // Don't cull if the sphere has infinite radius
196 if(!isFinite(radius)) return false;
197
198 uint32 inMask = _inMask;
199 assert(numPlanes < 31);
200
201 // See if there is one plane for which all of the
202 // vertices are in the negative half space.
203 for (int p = 0; p < numPlanes; ++p) {
204
205 // Only test planes that are not masked
206 if ((inMask & 1) != 0) {
207 bool culled = ! plane[p].halfSpaceContains(center + plane[p].normal() * radius);
208 if (culled) {
209 // Plane p culled the sphere
210 cullingPlane = p;
211 return true;
212 }
213 }
214
215 // Move on to the next bit.
216 inMask = inMask >> 1;
217 }
218
219 // None of the planes could cull this box
220 cullingPlane = -1;
221 return false;
222}
