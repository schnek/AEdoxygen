36uint32_t SplineChainMovementGenerator::sendPathSpline(Unit* owner, float velocity, MovementMgr::PointsArray const& path) const
37{
38 const auto nodeCount = path.size();
39 ASSERT(nodeCount > 1 && "SplineChainMovementGenerator::SendPathSpline: Every path must have source & destination (size > 1)!");
40
41 MovementMgr::MoveSplineInit init(owner);
42 if (nodeCount > 2)
43 init.MovebyPath(path);
44 else
45 init.MoveTo(path[1], false, true);
46
47 if (velocity > 0.f)
48 init.SetVelocity(velocity);
49 init.SetWalk(_walk);
50 return init.Launch();
51}

53void SplineChainMovementGenerator::sendSplineFor(Unit* owner, uint32_t index, uint32_t& duration)
54{
55 ASSERT(index < _chainSize && "SplineChainMovementGenerator::SendSplineFor: referenced index higher than path size!");
56
57 SplineChainLink const& thisLink = _chain[index];
58 uint32_t actualDuration = sendPathSpline(owner, thisLink.Velocity, thisLink.Points);
59 if (actualDuration != thisLink.ExpectedDuration)
60 {
61 duration = uint32_t(double(actualDuration) / double(thisLink.ExpectedDuration) * duration);
62 }
63}

65void SplineChainMovementGenerator::initialize(Unit* owner)
66{
67 removeFlag(MOVEMENTGENERATOR_FLAG_INITIALIZATION_PENDING | MOVEMENTGENERATOR_FLAG_DEACTIVATED);
68 addFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED);
69
70 if (!_chainSize)
71 {
72 return;
73 }
74
75 if (_nextIndex >= _chainSize)
76 {
77 _msToNext = 0;
78 return;
79 }
80
81 if (_nextFirstWP) // this is a resumed movegen that has to start with a partial spline
82 {
83 if (hasFlag(MOVEMENTGENERATOR_FLAG_FINALIZED))
84 return;
85
86 SplineChainLink const& thisLink = _chain[_nextIndex];
87 if (_nextFirstWP >= thisLink.Points.size())
88 {
89 _nextFirstWP = static_cast<uint8_t>(thisLink.Points.size() - 1);
90 }
91
92 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
93 MovementMgr::PointsArray partial(thisLink.Points.begin() + (_nextFirstWP-1), thisLink.Points.end());
94 sendPathSpline(owner, thisLink.Velocity, partial);
95
96 ++_nextIndex;
97 if (_nextIndex >= _chainSize)
98 _msToNext = 0;
99 else if (!_msToNext)
100 _msToNext = 1;
101 _nextFirstWP = 0;
102 }
103 else
104 {
105 _msToNext = std::max(_chain[_nextIndex].TimeToNext, 1u);
106 sendSplineFor(owner, _nextIndex, _msToNext);
107
108 ++_nextIndex;
109 if (_nextIndex >= _chainSize)
110 _msToNext = 0;
111 }
112}

122bool SplineChainMovementGenerator::update(Unit* owner, uint32_t diff)
123{
124 if (!owner || hasFlag(MOVEMENTGENERATOR_FLAG_FINALIZED))
125 return false;
126
127 // _msToNext being zero here means we're on the final spline
128 if (!_msToNext)
129 {
130 if (owner->movespline->Finalized())
131 {
132 addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED);
133 return false;
134 }
135 return true;
136 }
137
138 if (_msToNext <= diff)
139 {
140 // Send next spline
141 _msToNext = std::max(_chain[_nextIndex].TimeToNext, 1u);
142 sendSplineFor(owner, _nextIndex, _msToNext);
143 ++_nextIndex;
144 if (_nextIndex >= _chainSize)
145 {
146 // We have reached the final spline, once it finalizes we should also finalize the movegen (start checking on next update)
147 _msToNext = 0;
148 return true;
149 }
150 }
151 else
152 _msToNext -= diff;
153
154 return true;
155}

183SplineChainResumeInfo SplineChainMovementGenerator::getResumeInfo(Unit const* owner) const
184{
185 if (!_nextIndex)
186 return SplineChainResumeInfo(_id, &_chain, _walk, 0, 0, _msToNext);
187
188 if (owner->movespline->Finalized())
189 {
190 if (_nextIndex < _chainSize)
191 return SplineChainResumeInfo(_id, &_chain, _walk, _nextIndex, 0, 1u);
192 else
193 return SplineChainResumeInfo();
194 }
195
196 return SplineChainResumeInfo(_id, &_chain, _walk, uint8_t(_nextIndex - 1), uint8_t(owner->movespline->_currentSplineIdx()), _msToNext);
197}
