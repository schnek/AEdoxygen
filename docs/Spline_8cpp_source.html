61inline void C_Evaluate(Vector3 const* vertice, float t, Matrix4 const& matr, Vector3 &result)
62{
63 Vector4 tvec(t*t*t, t*t, t, 1.f);
64 Vector4 weights(tvec * matr);
65
66 result = vertice[0] * weights[0] + vertice[1] * weights[1]
67 + vertice[2] * weights[2] + vertice[3] * weights[3];
68}
69
70inline void C_Evaluate_Derivative(Vector3 const* vertice, float t, Matrix4 const& matr, Vector3 &result)
71{
72 Vector4 tvec(3.f*t*t, 2.f*t, 1.f, 0.f);
73 Vector4 weights(tvec * matr);
74
75 result = vertice[0] * weights[0] + vertice[1] * weights[1]
76 + vertice[2] * weights[2] + vertice[3] * weights[3];
77}

123float SplineBase::SegLengthCatmullRom(index_type index) const
124{
125 ASSERT(index >= index_lo && index < index_hi);
126
127 Vector3 curPos, nextPos;
128 const Vector3 * p = &points[index - 1];
129 curPos = nextPos = p[1];
130
131 index_type i = 1;
132 float length = 0;
133 while (i <= STEPS_PER_SEGMENT)
134 {
135 C_Evaluate(p, float(i) / float(STEPS_PER_SEGMENT), s_catmullRomCoeffs, nextPos);
136 length += (nextPos - curPos).length();
137 curPos = nextPos;
138 ++i;
139 }
140 return length;
141}

184void SplineBase::InitLinear(Vector3 const* controls, index_type count, index_type cyclic_point)
185{
186 ASSERT(count >= 2);
187 const int real_size = count + 1;
188
189 points.resize(real_size);
190
191 memcpy(&points[0], controls, sizeof(Vector3) * count);
192
193 // first and last two indexes are space for special 'virtual points'
194 // these points are required for proper C_Evaluate and C_Evaluate_Derivative methtod work
195 if (cyclic)
196 points[count] = controls[cyclic_point];
197 else
198 points[count] = controls[count-1];
199
200 index_lo = 0;
201 index_hi = cyclic ? count : (count - 1);
202}

204void SplineBase::InitCatmullRom(Vector3 const* controls, index_type count, index_type cyclic_point)
205{
206 const int real_size = count + (cyclic ? (1+2) : (1+1));
207
208 points.resize(real_size);
209
210 int lo_index = 1;
211 int high_index = lo_index + count - 1;
212
213 memcpy(&points[lo_index], controls, sizeof(Vector3) * count);
214
215 // first and last two indexes are space for special 'virtual points'
216 // these points are required for proper C_Evaluate and C_Evaluate_Derivative methtod work
217 if (cyclic)
218 {
219 if (cyclic_point == 0)
220 points[0] = controls[count-1];
221 else
222 points[0] = controls[0] - G3D::Vector3{ std::cos(initialOrientation), std::sin(initialOrientation), 0.0f };
223
224 points[high_index+1] = controls[cyclic_point];
225 points[high_index+2] = controls[cyclic_point+1];
226 }
227 else
228 {
229 points[0] = controls[0] - G3D::Vector3{ std::cos(initialOrientation), std::sin(initialOrientation), 0.0f };
230 points[high_index+1] = controls[count-1];
231 }
232
233 index_lo = lo_index;
234 index_hi = high_index + (cyclic ? 1 : 0);
235}
