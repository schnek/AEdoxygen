293 BOOL Init(LPCSTR szSymPath)
294 {
295 if (m_parent == NULL)
296 return FALSE;
297 // Dynamically load the Entry-Points for dbghelp.dll:
298 // First try to load the newest one from
299 TCHAR szTemp[4096];
300 // But before we do this, we first check if the ".local" file exists
301 if (GetModuleFileName(NULL, szTemp, 4096) > 0)
302 {
303 _tcscat_s(szTemp, _T(".local"));
304 if (GetFileAttributes(szTemp) == INVALID_FILE_ATTRIBUTES)
305 {
306 // ".local" file does not exist, so we can try to load the dbghelp.dll from the "Debugging Tools for Windows"
307 // Ok, first try the new path according to the architecture:
308#ifdef _M_IX86
309 if ((m_hDbhHelp == NULL) && (GetEnvironmentVariable(_T("ProgramFiles"), szTemp, 4096) > 0))
310 {
311 _tcscat_s(szTemp, _T("\\Debugging Tools for Windows (x86)\\dbghelp.dll"));
312 // now check if the file exists:
313 if (GetFileAttributes(szTemp) != INVALID_FILE_ATTRIBUTES)
314 {
315 m_hDbhHelp = LoadLibrary(szTemp);
316 }
317 }
318#elif _M_X64
319 if ((m_hDbhHelp == NULL) && (GetEnvironmentVariable(_T("ProgramFiles"), szTemp, 4096) > 0))
320 {
321 _tcscat_s(szTemp, _T("\\Debugging Tools for Windows (x64)\\dbghelp.dll"));
322 // now check if the file exists:
323 if (GetFileAttributes(szTemp) != INVALID_FILE_ATTRIBUTES)
324 {
325 m_hDbhHelp = LoadLibrary(szTemp);
326 }
327 }
328#elif _M_IA64
329 if ((m_hDbhHelp == NULL) && (GetEnvironmentVariable(_T("ProgramFiles"), szTemp, 4096) > 0))
330 {
331 _tcscat_s(szTemp, _T("\\Debugging Tools for Windows (ia64)\\dbghelp.dll"));
332 // now check if the file exists:
333 if (GetFileAttributes(szTemp) != INVALID_FILE_ATTRIBUTES)
334 {
335 m_hDbhHelp = LoadLibrary(szTemp);
336 }
337 }
338#endif
339 // If still not found, try the old directories...
340 if ((m_hDbhHelp == NULL) && (GetEnvironmentVariable(_T("ProgramFiles"), szTemp, 4096) > 0))
341 {
342 _tcscat_s(szTemp, _T("\\Debugging Tools for Windows\\dbghelp.dll"));
343 // now check if the file exists:
344 if (GetFileAttributes(szTemp) != INVALID_FILE_ATTRIBUTES)
345 {
346 m_hDbhHelp = LoadLibrary(szTemp);
347 }
348 }
349#if defined _M_X64 || defined _M_IA64
350 // Still not found? Then try to load the (old) 64-Bit version:
351 if ((m_hDbhHelp == NULL) && (GetEnvironmentVariable(_T("ProgramFiles"), szTemp, 4096) > 0))
352 {
353 _tcscat_s(szTemp, _T("\\Debugging Tools for Windows 64-Bit\\dbghelp.dll"));
354 if (GetFileAttributes(szTemp) != INVALID_FILE_ATTRIBUTES)
355 {
356 m_hDbhHelp = LoadLibrary(szTemp);
357 }
358 }
359#endif
360 }
361 }
362 if (m_hDbhHelp == NULL) // if not already loaded, try to load a default-one
363 m_hDbhHelp = LoadLibrary(_T("dbghelp.dll"));
364 if (m_hDbhHelp == NULL)
365 return FALSE;
366 pSI = (tSI)GetProcAddress(m_hDbhHelp, "SymInitialize");
367 pSC = (tSC)GetProcAddress(m_hDbhHelp, "SymCleanup");
368
369 pSW = (tSW)GetProcAddress(m_hDbhHelp, "StackWalk64");
370 pSGO = (tSGO)GetProcAddress(m_hDbhHelp, "SymGetOptions");
371 pSSO = (tSSO)GetProcAddress(m_hDbhHelp, "SymSetOptions");
372
373 pSFTA = (tSFTA)GetProcAddress(m_hDbhHelp, "SymFunctionTableAccess64");
374 pSGLFA = (tSGLFA)GetProcAddress(m_hDbhHelp, "SymGetLineFromAddr64");
375 pSGMB = (tSGMB)GetProcAddress(m_hDbhHelp, "SymGetModuleBase64");
376 pSGMI = (tSGMI)GetProcAddress(m_hDbhHelp, "SymGetModuleInfo64");
377 pSGSFA = (tSGSFA)GetProcAddress(m_hDbhHelp, "SymGetSymFromAddr64");
378 pUDSN = (tUDSN)GetProcAddress(m_hDbhHelp, "UnDecorateSymbolName");
379 pSLM = (tSLM)GetProcAddress(m_hDbhHelp, "SymLoadModule64");
380 pSGSP = (tSGSP)GetProcAddress(m_hDbhHelp, "SymGetSearchPath");
381
382 if (pSC == NULL || pSFTA == NULL || pSGMB == NULL || pSGMI == NULL || pSGO == NULL ||
383 pSGSFA == NULL || pSI == NULL || pSSO == NULL || pSW == NULL || pUDSN == NULL ||
384 pSLM == NULL)
385 {
386 FreeLibrary(m_hDbhHelp);
387 m_hDbhHelp = NULL;
388 pSC = NULL;
389 return FALSE;
390 }
391
392 // SymInitialize
393 if (this->pSI(m_hProcess, szSymPath, FALSE) == FALSE)
394 this->m_parent->OnDbgHelpErr("SymInitialize", GetLastError(), 0);
395
396 DWORD symOptions = this->pSGO(); // SymGetOptions
397 symOptions |= SYMOPT_LOAD_LINES;
398 symOptions |= SYMOPT_FAIL_CRITICAL_ERRORS;
399 //symOptions |= SYMOPT_NO_PROMPTS;
400 // SymSetOptions
401 symOptions = this->pSSO(symOptions);
402
403 char buf[StackWalker::STACKWALK_MAX_NAMELEN] = {0};
404 if (this->pSGSP != NULL)
405 {
406 if (this->pSGSP(m_hProcess, buf, StackWalker::STACKWALK_MAX_NAMELEN) == FALSE)
407 this->m_parent->OnDbgHelpErr("SymGetSearchPath", GetLastError(), 0);
408 }
409 char szUserName[1024] = {0};
410 DWORD dwSize = 1024;
411 GetUserNameA(szUserName, &dwSize);
412 this->m_parent->OnSymInit(buf, symOptions, szUserName);
413
414 return TRUE;
415 }

738 result = GetLastError();

810 this->m_parent->OnLoadModule(img, mod, baseAddr, size, result, szSymType, pdbName,

978BOOL StackWalker::LoadModules()
979{
980 if (this->m_sw == NULL)
981 {
982 SetLastError(ERROR_DLL_INIT_FAILED);
983 return FALSE;
984 }
985 if (m_modulesLoaded != FALSE)
986 return TRUE;
987
988 // Build the sym-path:
989 char* szSymPath = NULL;
990 if ((this->m_options & SymBuildPath) != 0)
991 {
992 const size_t nSymPathLen = 4096;
993 szSymPath = (char*)malloc(nSymPathLen);
994 if (szSymPath == NULL)
995 {
996 SetLastError(ERROR_NOT_ENOUGH_MEMORY);
997 return FALSE;
998 }
999 szSymPath[0] = 0;
1000 // Now first add the (optional) provided sympath:
1001 if (this->m_szSymPath != NULL)
1002 {
1003 strcat_s(szSymPath, nSymPathLen, this->m_szSymPath);
1004 strcat_s(szSymPath, nSymPathLen, ";");
1005 }
1006
1007 strcat_s(szSymPath, nSymPathLen, ".;");
1008
1009 const size_t nTempLen = 1024;
1010 char szTemp[nTempLen];
1011 // Now add the current directory:
1012 if (GetCurrentDirectoryA(nTempLen, szTemp) > 0)
1013 {
1014 szTemp[nTempLen - 1] = 0;
1015 strcat_s(szSymPath, nSymPathLen, szTemp);
1016 strcat_s(szSymPath, nSymPathLen, ";");
1017 }
1018
1019 // Now add the path for the main-module:
1020 if (GetModuleFileNameA(NULL, szTemp, nTempLen) > 0)
1021 {
1022 szTemp[nTempLen - 1] = 0;
1023 for (char* p = (szTemp + strlen(szTemp) - 1); p >= szTemp; --p)
1024 {
1025 // locate the rightmost path separator
1026 if ((*p == '\\') || (*p == '/') || (*p == ':'))
1027 {
1028 *p = 0;
1029 break;
1030 }
1031 } // for (search for path separator...)
1032 if (strlen(szTemp) > 0)
1033 {
1034 strcat_s(szSymPath, nSymPathLen, szTemp);
1035 strcat_s(szSymPath, nSymPathLen, ";");
1036 }
1037 }
1038 if (GetEnvironmentVariableA("_NT_SYMBOL_PATH", szTemp, nTempLen) > 0)
1039 {
1040 szTemp[nTempLen - 1] = 0;
1041 strcat_s(szSymPath, nSymPathLen, szTemp);
1042 strcat_s(szSymPath, nSymPathLen, ";");
1043 }
1044 if (GetEnvironmentVariableA("_NT_ALTERNATE_SYMBOL_PATH", szTemp, nTempLen) > 0)
1045 {
1046 szTemp[nTempLen - 1] = 0;
1047 strcat_s(szSymPath, nSymPathLen, szTemp);
1048 strcat_s(szSymPath, nSymPathLen, ";");
1049 }
1050 if (GetEnvironmentVariableA("SYSTEMROOT", szTemp, nTempLen) > 0)
1051 {
1052 szTemp[nTempLen - 1] = 0;
1053 strcat_s(szSymPath, nSymPathLen, szTemp);
1054 strcat_s(szSymPath, nSymPathLen, ";");
1055 // also add the "system32"-directory:
1056 strcat_s(szTemp, nTempLen, "\\system32");
1057 strcat_s(szSymPath, nSymPathLen, szTemp);
1058 strcat_s(szSymPath, nSymPathLen, ";");
1059 }
1060
1061 if ((this->m_options & SymUseSymSrv) != 0)
1062 {
1063 if (GetEnvironmentVariableA("SYSTEMDRIVE", szTemp, nTempLen) > 0)
1064 {
1065 szTemp[nTempLen - 1] = 0;
1066 strcat_s(szSymPath, nSymPathLen, "SRV*");
1067 strcat_s(szSymPath, nSymPathLen, szTemp);
1068 strcat_s(szSymPath, nSymPathLen, "\\websymbols");
1069 strcat_s(szSymPath, nSymPathLen, "*https://msdl.microsoft.com/download/symbols;");
1070 }
1071 else
1072 strcat_s(szSymPath, nSymPathLen,
1073 "SRV*c:\\websymbols*https://msdl.microsoft.com/download/symbols;");
1074 }
1075 } // if SymBuildPath
1076
1077 // First Init the whole stuff...
1078 BOOL bRet = this->m_sw->Init(szSymPath);
1079 if (szSymPath != NULL)
1080 free(szSymPath);
1081 szSymPath = NULL;
1082 if (bRet == FALSE)
1083 {
1084 this->OnDbgHelpErr("Error while initializing dbghelp.dll", 0, 0);
1085 SetLastError(ERROR_DLL_INIT_FAILED);
1086 return FALSE;
1087 }
1088
1089 bRet = this->m_sw->LoadModules(this->m_hProcess, this->m_dwProcessId);
1090 if (bRet != FALSE)
1091 m_modulesLoaded = TRUE;
1092 return bRet;
1093}

1096// This has to be done due to a problem with the "hProcess"-parameter in x64...
1097// Because this class is in no case multi-threading-enabled (because of the limitations

1102BOOL StackWalker::ShowCallstack(HANDLE hThread,
1103 const CONTEXT* context,
1104 PReadProcessMemoryRoutine readMemoryFunction,
1105 LPVOID pUserData)
1106{
1107 CONTEXT c;
1108 CallstackEntry csEntry;
1109 IMAGEHLP_SYMBOL64* pSym = NULL;
1110 StackWalkerInternal::IMAGEHLP_MODULE64_V3 Module;
1111 IMAGEHLP_LINE64 Line;
1112 int frameNum;
1113 bool bLastEntryCalled = true;
1114 int curRecursionCount = 0;
1115
1116 if (m_modulesLoaded == FALSE)
1117 this->LoadModules(); // ignore the result...
1118
1119 if (this->m_sw->m_hDbhHelp == NULL)
1120 {
1121 SetLastError(ERROR_DLL_INIT_FAILED);
1122 return FALSE;
1123 }
1124
1125 s_readMemoryFunction = readMemoryFunction;
1126 s_readMemoryFunction_UserData = pUserData;
1127
1128 if (context == NULL)
1129 {
1130 // If no context is provided, capture the context
1131 // See: https://stackwalker.codeplex.com/discussions/446958
1132#if _WIN32_WINNT <= 0x0501
1133 // If we need to support XP, we need to use the "old way", because "GetThreadId" is not available!
1134 if (hThread == GetCurrentThread())
1135#else
1136 if (GetThreadId(hThread) == GetCurrentThreadId())
1137#endif
1138 {
1139 if (m_sw->m_ctx.ContextFlags != 0)
1140 c = m_sw->m_ctx; // context taken at Init
1141 else
1142 GET_CURRENT_CONTEXT_STACKWALKER_CODEPLEX(c, USED_CONTEXT_FLAGS);
1143 }
1144 else
1145 {
1146 SuspendThread(hThread);
1147 memset(&c, 0, sizeof(CONTEXT));
1148 c.ContextFlags = USED_CONTEXT_FLAGS;
1149
1150 // TODO: Detect if you want to get a thread context of a different process, which is running a different processor architecture...
1151 // This does only work if we are x64 and the target process is x64 or x86;
1152 // It cannot work, if this process is x64 and the target process is x64... this is not supported...
1153 // See also: http://www.howzatt.demon.co.uk/articles/DebuggingInWin64.html
1154 if (GetThreadContext(hThread, &c) == FALSE)
1155 {
1156 ResumeThread(hThread);
1157 return FALSE;
1158 }
1159 }
1160 }
1161 else
1162 c = *context;
1163
1164 // init STACKFRAME for first call
1165 STACKFRAME64 s; // in/out stackframe
1166 memset(&s, 0, sizeof(s));
1167 DWORD imageType;
1168#ifdef _M_IX86
1169 // normally, call ImageNtHeader() and use machine info from PE header
1170 imageType = IMAGE_FILE_MACHINE_I386;
1171 s.AddrPC.Offset = c.Eip;
1172 s.AddrPC.Mode = AddrModeFlat;
1173 s.AddrFrame.Offset = c.Ebp;
1174 s.AddrFrame.Mode = AddrModeFlat;
1175 s.AddrStack.Offset = c.Esp;
1176 s.AddrStack.Mode = AddrModeFlat;
1177#elif _M_X64
1178 imageType = IMAGE_FILE_MACHINE_AMD64;
1179 s.AddrPC.Offset = c.Rip;
1180 s.AddrPC.Mode = AddrModeFlat;
1181 s.AddrFrame.Offset = c.Rsp;
1182 s.AddrFrame.Mode = AddrModeFlat;
1183 s.AddrStack.Offset = c.Rsp;
1184 s.AddrStack.Mode = AddrModeFlat;
1185#elif _M_IA64
1186 imageType = IMAGE_FILE_MACHINE_IA64;
1187 s.AddrPC.Offset = c.StIIP;
1188 s.AddrPC.Mode = AddrModeFlat;
1189 s.AddrFrame.Offset = c.IntSp;
1190 s.AddrFrame.Mode = AddrModeFlat;
1191 s.AddrBStore.Offset = c.RsBSP;
1192 s.AddrBStore.Mode = AddrModeFlat;
1193 s.AddrStack.Offset = c.IntSp;
1194 s.AddrStack.Mode = AddrModeFlat;
1195#else
1196#error "Platform not supported!"
1197#endif
1198
1199 pSym = (IMAGEHLP_SYMBOL64*)malloc(sizeof(IMAGEHLP_SYMBOL64) + STACKWALK_MAX_NAMELEN);
1200 if (!pSym)
1201 goto cleanup; // not enough memory...
1202 memset(pSym, 0, sizeof(IMAGEHLP_SYMBOL64) + STACKWALK_MAX_NAMELEN);
1203 pSym->SizeOfStruct = sizeof(IMAGEHLP_SYMBOL64);
1204 pSym->MaxNameLength = STACKWALK_MAX_NAMELEN;
1205
1206 memset(&Line, 0, sizeof(Line));
1207 Line.SizeOfStruct = sizeof(Line);
1208
1209 memset(&Module, 0, sizeof(Module));
1210 Module.SizeOfStruct = sizeof(Module);
1211
1212 for (frameNum = 0;; ++frameNum)
1213 {
1214 // get next stack frame (StackWalk64(), SymFunctionTableAccess64(), SymGetModuleBase64())
1215 // if this returns ERROR_INVALID_ADDRESS (487) or ERROR_NOACCESS (998), you can
1216 // assume that either you are done, or that the stack is so hosed that the next
1217 // deeper frame could not be found.
1218 // CONTEXT need not to be supplied if imageTyp is IMAGE_FILE_MACHINE_I386!
1219 if (!this->m_sw->pSW(imageType, this->m_hProcess, hThread, &s, &c, myReadProcMem,
1220 this->m_sw->pSFTA, this->m_sw->pSGMB, NULL))
1221 {
1222 // INFO: "StackWalk64" does not set "GetLastError"...
1223 this->OnDbgHelpErr("StackWalk64", 0, s.AddrPC.Offset);
1224 break;
1225 }
1226
1227 csEntry.offset = s.AddrPC.Offset;
1228 csEntry.name[0] = 0;
1229 csEntry.undName[0] = 0;
1230 csEntry.undFullName[0] = 0;
1231 csEntry.offsetFromSmybol = 0;
1232 csEntry.offsetFromLine = 0;
1233 csEntry.lineFileName[0] = 0;
1234 csEntry.lineNumber = 0;
1235 csEntry.loadedImageName[0] = 0;
1236 csEntry.moduleName[0] = 0;
1237 if (s.AddrPC.Offset == s.AddrReturn.Offset)
1238 {
1239 if ((this->m_MaxRecursionCount > 0) && (curRecursionCount > m_MaxRecursionCount))
1240 {
1241 this->OnDbgHelpErr("StackWalk64-Endless-Callstack!", 0, s.AddrPC.Offset);
1242 break;
1243 }
1244 curRecursionCount++;
1245 }
