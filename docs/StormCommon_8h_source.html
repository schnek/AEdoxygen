155template <typename XCHAR, typename XINT>
156XCHAR * IntToString(XCHAR * szBuffer, size_t cchMaxChars, XINT nValue, size_t nDigitCount = 0)
157{
158 XCHAR * szBufferEnd = szBuffer + cchMaxChars - 1;
159 XCHAR szNumberRev[0x20];
160 size_t nLength = 0;
161
162 // Always put the first digit
163 szNumberRev[nLength++] = (XCHAR)(nValue % 10) + '0';
164 nValue /= 10;
165
166 // Continue as long as we have non-zero
167 while(nValue != 0)
168 {
169 szNumberRev[nLength++] = (XCHAR)(nValue % 10) + '0';
170 nValue /= 10;
171 }
172
173 // Fill zeros, if needed
174 while(szBuffer < szBufferEnd && nLength < nDigitCount)
175 {
176 *szBuffer++ = '0';
177 nDigitCount--;
178 }
179
180 // Fill the buffer
181 while(szBuffer < szBufferEnd && nLength > 0)
182 {
183 nLength--;
184 *szBuffer++ = szNumberRev[nLength];
185 }
186
187 // Terminate the number with zeros
188 szBuffer[0] = 0;
189 return szBuffer;
190}
