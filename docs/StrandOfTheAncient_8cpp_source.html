432bool StrandOfTheAncient::HookHandleRepop(Player* plr)
433{
434 float dist = 999999.0f;
435 LocationVector dest_pos;
436 uint32_t id = 0;
437
438 // Let's find the closests GY
439 for (uint8_t i = SOTA_GY_EAST; i < NUM_SOTA_GRAVEYARDS; i++)
440 {
441 if (graveyard[i].faction == plr->getTeam())
442 {
443 if (graveyard[i].spiritguide == nullptr)
444 continue;
445
446 float gydist = plr->CalcDistance(graveyard[i].spiritguide);
447 if (gydist > dist)
448 continue;
449
450 dist = gydist;
451 dest_pos = graveyard[i].spiritguide->GetPosition();
452 id = i;
453 }
454 }
455
456 if (id >= NUM_SOTA_GRAVEYARDS)
457 return false;
458
459 // port to it and queue for auto-resurrect
460 plr->safeTeleport(plr->GetMapId(), plr->GetInstanceID(), dest_pos);
461 queuePlayerForResurrect(plr, graveyard[id].spiritguide);
462
463 return true;
464}
