25void SummonHandler::addSummonToHandler(Summon* summon)
26{
27 if (summon == nullptr || summon->getUnitOwner() != m_owner)
28 return;
29
30 if (summon->isPet())
31 {
32 m_summons.push_back(summon);
33
34 // Check if current pet needs to be updated
35 if (m_pet == nullptr || (m_pet->isPermanentSummon() || summon->isPermanentSummon() || m_pet->getEntry() != summon->getEntry()))
36 {
37 m_pet = dynamic_cast<Pet*>(summon);
38 if (m_pet != nullptr)
39 {
40 m_owner->setSummonGuid(m_pet->getGuid());
41 m_pet->sendSpellsToController(m_owner, m_pet->getTimeLeft());
42 }
43 }
44 return;
45 }
46
47 // Do not add scripted summons to handler
48 if (summon->getSummonProperties() == nullptr)
49 return;
50
51 switch (summon->getSummonProperties()->Slot)
52 {
53 // Not handled
54 case SUMMON_SLOT_NONE:
55 break;
56 case SUMMON_SLOT_TOTEM_FIRE:
57 case SUMMON_SLOT_TOTEM_EARTH:
58 case SUMMON_SLOT_TOTEM_WATER:
59 case SUMMON_SLOT_TOTEM_AIR:
60 case SUMMON_SLOT_MINIPET:
61 case SUMMON_SLOT_QUEST:
62 {
63 const uint8_t slot = static_cast<uint8_t>(summon->getSummonProperties()->Slot) - 1;
64
65 // Send summon to totem bar
66 if (summon->getSummonProperties()->Slot != SUMMON_SLOT_MINIPET)
67 {
68 if (auto* const playerUnit = m_owner->isPlayer() ? dynamic_cast<Player*>(m_owner) : nullptr)
69 playerUnit->sendTotemCreatedPacket(slot, summon->getGuid(), summon->getTimeLeft(), summon->getCreatedBySpellId());
70 }
71
72 // Remove old summon if exists
73 if (auto* const oldSummon = m_summonSlots[slot])
74 oldSummon->unSummon();
75
76 m_summonSlots[slot] = summon;
77 } break;
78 default:
79 {
80 sLogger.failure("SummonHandler::addSummonToHandler : Unknown pet slot {}", summon->getSummonProperties()->Slot);
81 } break;
82 }
83
84 m_summons.push_back(summon);
85}

87void SummonHandler::removeSummonFromHandler(Summon* summon)
88{
89 if (summon == nullptr || summon->getUnitOwner() != m_owner)
90 return;
91
92 if (summon->isPet())
93 {
94 std::erase(m_summons, summon);
95
96 if (m_pet != nullptr && m_pet->getGuid() == summon->getGuid())
97 {
98 // Current pet was removed, try find another controllable pet
99 auto foundControllablePet = false;
100 if (!m_summons.empty())
101 {
102 Summon* newCurrentPet = nullptr;
103 for (const auto& existingSummon : m_summons)
104 {
105 if (!existingSummon->isPet() || !existingSummon->isAlive())
106 continue;
107
108 if (existingSummon->isPermanentSummon())
109 {
110 newCurrentPet = existingSummon;
111 break;
112 }
113 else if (newCurrentPet == nullptr)
114 {
115 // Use possibly this pet but try find permanent pet
116 newCurrentPet = existingSummon;
117 }
118 }
119
120 if (newCurrentPet != nullptr)
121 {
122 foundControllablePet = true;
123 m_pet = dynamic_cast<Pet*>(newCurrentPet);
124 if (m_pet != nullptr)
125 {
126 m_owner->setSummonGuid(m_pet->getGuid());
127 m_pet->sendSpellsToController(m_owner, m_pet->getTimeLeft());
128 }
129 }
130 }
131
132 if (!foundControllablePet)
133 {
134 m_pet = nullptr;
135 m_owner->setSummonGuid(0);
136 if (const auto plrOwner = m_owner->isPlayer() ? dynamic_cast<Player*>(m_owner) : nullptr)
137 plrOwner->sendEmptyPetSpellList();
138 }
139 }
140 return;
141 }
142
143 if (summon->getSummonProperties() == nullptr)
144 return;
145
146 switch (summon->getSummonProperties()->Slot)
147 {
148 // Not handled
149 case SUMMON_SLOT_NONE:
150 break;
151 case SUMMON_SLOT_TOTEM_FIRE:
152 case SUMMON_SLOT_TOTEM_EARTH:
153 case SUMMON_SLOT_TOTEM_WATER:
154 case SUMMON_SLOT_TOTEM_AIR:
155 case SUMMON_SLOT_MINIPET:
156 case SUMMON_SLOT_QUEST:
157 {
158 const uint8_t slot = static_cast<uint8_t>(summon->getSummonProperties()->Slot) - 1;
159 m_summonSlots[slot] = nullptr;
160 } break;
161 default:
162 {
163 sLogger.failure("SummonHandler::removeSummonFromHandler : Unknown pet slot {}", summon->getSummonProperties()->Slot);
164 } break;
165 }
166
167 std::erase(m_summons, summon);
168}

249void SummonHandler::notifyOnPetDeath(Summon* pet)
250{
251 if (pet == nullptr || pet->getUnitOwner() != m_owner)
252 return;
253
254 if (m_pet != nullptr && m_pet != pet)
255 return;
256
257 // Some spells summon multiple pets like druid's Force of Nature
258 // If one of those pets die, summon guid should be updated to one that is alive
259 for (const auto& summon : m_summons)
260 {
261 if (!summon->isPet())
262 continue;
263
264 if (summon->getEntry() == pet->getEntry() && summon->isAlive())
265 {
266 m_pet = dynamic_cast<Pet*>(summon);
267 if (m_pet != nullptr)
268 {
269 m_owner->setSummonGuid(m_pet->getGuid());
270 m_pet->sendSpellsToController(m_owner, m_pet->getTimeLeft());
271 }
272 break;
273 }
274 }

297bool SummonHandler::hasTotemInSlot(SummonSlot slot) const
298{
299 if (slot < SUMMON_SLOT_TOTEM_FIRE || slot > SUMMON_SLOT_TOTEM_AIR)
300 return false;
301
302 const auto* totem = getSummonInSlot(slot);
303 return totem != nullptr && totem->isTotem();
304}

306void SummonHandler::killAllTotems() const
307{
308 for (uint8_t i = SUMMON_SLOT_TOTEM_FIRE; i <= SUMMON_SLOT_TOTEM_AIR; ++i)
309 {
310 auto* const totem = getSummonInSlot(static_cast<SummonSlot>(i));
311 if (totem == nullptr || !totem->isTotem())
312 continue;
313
314 totem->dieAndDisappearOnExpire();
315 }
