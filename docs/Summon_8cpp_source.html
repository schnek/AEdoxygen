29{

53void Summon::load(CreatureProperties const* creatureProperties, Unit* unitOwner, LocationVector const& position, uint32_t duration, uint32_t spellId)
54{
55 if (!isPet())
56 Creature::Load(creatureProperties, position.x, position.y, position.z, position.o);
57
58 setTimeLeft(duration);
59 setLifeTime(duration);
60
61 // Despawn Type
62 if (m_despawnType == MANUAL_DESPAWN)
63 m_despawnType = getDefaultDespawnTypeForSummon(this);
64
65 setCreatedBySpellId(spellId);
66 SetSpawnLocation(position);
67
68 m_summonActive = true;
69
70 if (unitOwner == nullptr)
71 return;
72
73 setZoneId(unitOwner->getZoneId());
74
75 if (unitOwner->isPvpFlagSet())
76 setPvpFlag();
77 else
78 removePvpFlag();
79
80 if (unitOwner->isFfaPvpFlagSet())
81 setFfaPvpFlag();
82 else
83 removeFfaPvpFlag();
84
85 if (unitOwner->isSanctuaryFlagSet())
86 setSanctuaryFlag();
87 else
88 removeSanctuaryFlag();
89
90 if (unitOwner->hasUnitFlags(UNIT_FLAG_PVP_ATTACKABLE))
91 addUnitFlags(UNIT_FLAG_PVP_ATTACKABLE);
92
93#if VERSION_STRING == TBC
94 if (unitOwner->hasUnitFlags(UNIT_FLAG_PVP_ATTACKABLE))
95 setPositiveAuraLimit(POS_AURA_LIMIT_PVP_ATTACKABLE);
96 else
97 setPositiveAuraLimit(POS_AURA_LIMIT_CREATURE);
98#endif
99
100 m_unitOwner = unitOwner;
101
102 // Non-scripted summons
103 if (isPet() || m_summonProperties != nullptr)
104 {
105 if (m_summonProperties != nullptr && m_summonProperties->FactionID != 0)
106 setFaction(m_summonProperties->FactionID);
107 else
108 setFaction(unitOwner->getFactionTemplate());
109
110 setCreatedByGuid(unitOwner->getGuid());
111 if (unitOwner->getSummonedByGuid() != 0)
112 setSummonedByGuid(unitOwner->getSummonedByGuid());
113 else
114 setSummonedByGuid(unitOwner->getGuid());
115 }

118void Summon::Update(unsigned long time_passed)
119{
120 if (getDeathState() == DEAD)
121 {
122 unSummon();
123 return;
124 }
125
126 switch (m_despawnType)
127 {
128 case DEAD_DESPAWN:
129 break;
130 case MANUAL_DESPAWN:
131 {
132 // Set correct despawn type if not set
133 m_despawnType = getDefaultDespawnTypeForSummon(this);
134 } break;
135 case TIMED_DESPAWN:
136 {
137 if (getTimeLeft() <= time_passed)
138 {
139 dieAndDisappearOnExpire();
140 return;
141 }
142
143 setTimeLeft(getTimeLeft() - time_passed);
144 } break;
145 case TIMED_DESPAWN_OUT_OF_COMBAT:
146 {
147 if (!isInCombat())
148 {
149 if (getTimeLeft() <= time_passed)
150 {
151 dieAndDisappearOnExpire();
152 return;
153 }
154
155 setTimeLeft(getTimeLeft() - time_passed);
156 }
157 else if (getTimeLeft() != getLifeTime())
158 {
159 setTimeLeft(getLifeTime());
160 }
161 } break;
162 case CORPSE_TIMED_DESPAWN:
163 {
164 if (getDeathState() == CORPSE)
165 {
166 if (getTimeLeft() <= time_passed)
167 {
168 unSummon();
169 return;
170 }
171
172 setTimeLeft(getTimeLeft() - time_passed);
173 }
174 } break;
175 case CORPSE_DESPAWN:
176 {
177 if (getDeathState() == CORPSE)
178 {
179 unSummon();
180 return;
181 }
182 } break;
183 case TIMED_OR_CORPSE_DESPAWN:
184 {
185 if (getDeathState() == CORPSE)
186 {
187 // There is a small delay before summon should despawn
188 m_despawnType = CORPSE_TIMED_DESPAWN;
189 if (_hasExtendedDespawnDelayInDeath())
190 setNewLifeTime(SUMMON_CORPSE_DESPAWN_EXTENDED_TIMER);
191 else
192 setNewLifeTime(SUMMON_CORPSE_DESPAWN_DEFAULT_TIMER);
193 return;
194 }
195
196 if (getTimeLeft() <= time_passed)
197 {
198 dieAndDisappearOnExpire();
199 return;
200 }
201
202 setTimeLeft(getTimeLeft() - time_passed);
203 } break;
204 case TIMED_OR_DEAD_DESPAWN:
205 {
206 if (getDeathState() == DEAD)
207 {
208 unSummon();
209 return;
210 }
211
212 if (getTimeLeft() <= time_passed)
213 {
214 dieAndDisappearOnExpire();
215 return;
216 }
217
218 setTimeLeft(getTimeLeft() - time_passed);
219 } break;
220 default:
221 {
222 unSummon();
223 } break;
224 }
225
226 // Call superclass method last in case if dead summon is removed from world
227 Creature::Update(time_passed);
228}

374void Summon::_onUnsummon()
375{
376 if (m_unitOwner != nullptr)
377 {
378 // If this summon is summoned by a totem, unsummon the totem also
379 if (auto* const totemOwner = m_unitOwner->isTotem() ? dynamic_cast<TotemSummon*>(m_unitOwner) : nullptr)
380 {
381 // Prevent unsummon loop if owner is already dead
382 if (totemOwner->isSummonActive() && totemOwner->isAlive())
383 totemOwner->unSummon();
384 }
385
386 // Script Call
387 if (m_unitOwner->ToCreature() && m_unitOwner->IsInWorld() && m_unitOwner->ToCreature()->GetScript())
388 m_unitOwner->ToCreature()->GetScript()->OnSummonDespawn(this);
389
390 if (getCreatedBySpellId() != 0)
391 m_unitOwner->removeAllAurasById(getCreatedBySpellId());
392
393 // Remove summon from owner
394 if (m_summonProperties != nullptr || isPet())
395 m_unitOwner->getSummonInterface()->removeSummonFromHandler(this);
396 }
397
398 // Prevent unsummon loop
399 m_summonActive = false;
400}

506{
507 Summon::load(creatureProperties, unitOwner, position, duration, spellId);
508
509 uint32_t displayId = 0;
510 if (unitOwner != nullptr)
511 {
512 if (const auto displayIds = sMySQLStore.getTotemDisplayId(unitOwner->getRace(), creature_properties->Male_DisplayID))
513 displayId = displayIds->race_specific_id;
514 }
515
516 if (displayId == 0)
517 displayId = creature_properties->Male_DisplayID;
518
519 setLevel(unitOwner != nullptr ? unitOwner->getLevel() : 1);
520 setRace(0);
521 setClass(1); // Creature class warrior
522 setGender(GENDER_NONE);
523 setPowerType(POWER_TYPE_MANA);
524 setBaseAttackTime(MELEE, 2000);
525 setBaseAttackTime(OFFHAND, 2000);
526 setBoundingRadius(1.0f);
527 setCombatReach(1.0f);
528 setDisplayId(displayId);
529 setNativeDisplayId(displayId);
530 setModCastSpeed(1.0f);
531 setDynamicFlags(0);
532
533 // Totems should always be unsummoned on death or when expired
534 m_despawnType = duration == 0 ? CORPSE_DESPAWN : TIMED_OR_CORPSE_DESPAWN;
535
536 if (unitOwner != nullptr)
537 {
538 for (uint8_t school = 0; school < TOTAL_SPELL_SCHOOLS; school++)
539 {
540 ModDamageDone[school] = unitOwner->GetDamageDoneMod(school);
541 m_healDoneMod[school] = unitOwner->m_healDoneMod[school];
542 }
543
544 m_aiInterface->Init(this, unitOwner);
545
546 setAItoUse(false);
547 }
548}

571void TotemSummon::setupSpells()
572{
573 if (getUnitOwner() == nullptr)
574 return;
575
576 const auto creatorSpell = sSpellMgr.getSpellInfo(getCreatedBySpellId());
577 const auto totemSpell = sSpellMgr.getSpellInfo(creature_properties->AISpells[0]);
578 if (totemSpell == nullptr)
579 {
580 sLogger.debug("Totem {} does not have any spells to cast", creature_properties->Id);
581 return;
582 }
583
584 // Set up AI, depending on our spells.
585 bool isCastingTotem = true;
586
587 if (totemSpell->hasEffect(SPELL_EFFECT_SUMMON) ||
588 totemSpell->hasEffect(SPELL_EFFECT_APPLY_GROUP_AREA_AURA) ||
589 totemSpell->hasEffect(SPELL_EFFECT_APPLY_RAID_AREA_AURA) ||
590 totemSpell->hasEffect(SPELL_EFFECT_PERSISTENT_AREA_AURA) ||
591 totemSpell->hasEffect(SPELL_EFFECT_APPLY_AURA) && totemSpell->appliesAreaAura(SPELL_AURA_PERIODIC_TRIGGER_SPELL))
592 isCastingTotem = false;
593
594 if (!isCastingTotem)
595 {
596 // We're an area aura. Simply cast the spell.
597 m_aiInterface->totemspell = creatorSpell;
598
599 auto spell = sSpellMgr.newSpell(this, totemSpell, true, nullptr);
600 SpellCastTargets targets;
601
602 if (!totemSpell->hasEffect(SPELL_AURA_PERIODIC_TRIGGER_SPELL))
603 {
604 targets.setDestination(GetPosition());
605 targets.setTargetMask(TARGET_FLAG_DEST_LOCATION);
606 }
607 spell->prepare(&targets);
608
609 }
610 else
611 {
612 // We're a casting totem. Switch AI on, and tell it to cast this spell.
613 setAItoUse(true);
614 m_aiInterface->totemspell = totemSpell;
615 m_aiInterface->m_totemspelltimer = 0;
616 m_aiInterface->m_totemspelltime = 3 * TimeVarsMs::Second;
617 }
