478#ifdef _MSC_VER
479# ifdef _M_IX86
480 // 32-bit
481 inline uint64 System::getCycleCount() {
482 uint32 timehi, timelo;
483
484 // Use the assembly instruction rdtsc, which gets the current
485 // cycle count (since the process started) and puts it in edx:eax.
486 __asm
487 {
488 rdtsc;
489 mov timehi, edx;
490 mov timelo, eax;
491 }
492
493 return ((uint64)timehi << 32) + (uint64)timelo;
494 }
495# else
496 // 64-bit
497 inline uint64 System::getCycleCount() {
498 LARGE_INTEGER now;
499 QueryPerformanceCounter(&now);
500 return now.QuadPart;
501 }
502
503# endif
504
505#elif defined(G3D_LINUX)
506
507 inline uint64 System::getCycleCount() {
508# if defined(__aarch64__)
509# if (__ARM_ARCH >= 6) // V6 is the earliest arch that has a standard cyclecount
510 uint32_t pmccntr;
511 uint32_t pmuseren;
512 uint32_t pmcntenset;
513 // Read the user mode perf monitor counter access permissions.
514 __asm__ __volatile__("mrc p15, 0, %w0, c9, c14, 0" : "=r"(pmuseren));
515 if (pmuseren & 1) { // Allows reading perfmon counters for user mode code.
516 __asm__ __volatile__("mrc p15, 0, %w0, c9, c12, 1" : "=r"(pmcntenset));
517 if (pmcntenset & 0x80000000ul) { // Is it counting?
518 __asm__ __volatile__("mrc p15, 0, %w0, c9, c13, 0" : "=r"(pmccntr));
519 // The counter is set up to count every 64th cycle
520 return static_cast<uint64>(pmccntr) * 64; // Should optimize to << 6
521 }
522 }
523# endif
524
525 struct timeval tv;
526 gettimeofday(&tv, nullptr);
527 return static_cast<uint64>(tv.tv_sec) * 1000000 + tv.tv_usec;
528# else
529 uint32 timehi, timelo;
530
531 __asm__ __volatile__ (
532 "rdtsc "
533 : "=a" (timelo),
534 "=d" (timehi)
535 : );
536
537 return ((uint64)timehi << 32) + (uint64)timelo;
538# endif
539 }

560inline void System::endCycleCount(uint64& cycleCount) {
561#ifndef G3D_OSX
562 cycleCount = getCycleCount() - cycleCount;
563#else
564 AbsoluteTime end = UpTime();
565 Nanoseconds diffNS =
566 AbsoluteDeltaToNanoseconds(end, UInt64ToUnsignedWide(cycleCount));
567 cycleCount =
568 (uint64) ((double) (instance().m_OSXCPUSpeed) *
569 (double) UnsignedWideToUInt64(diffNS) * instance().m_secondsPerNS);
570#endif
571}
