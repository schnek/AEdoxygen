28static uint8_t getFieldForTaxiNode(uint32_t nodeId)
29{
30#if VERSION_STRING < Cata
31 return static_cast<uint8_t>((nodeId - 1) / 32);
32#else
33 return static_cast<uint8_t>((nodeId - 1) / 8);
34#endif
35}

37static uint8_t getSubmaskForTaxiNode(uint32_t nodeId)
38{
39#if VERSION_STRING < Cata
40 return 1 << ((nodeId - 1) % 32);
41#else
42 return 1 << ((nodeId - 1) % 8);
43#endif
44}

190void TaxiPath::loadTaxiMask(std::string const& data)
191{
192 const auto tokens = AscEmu::Util::Strings::split(data, " ");
193 auto iter = tokens.cbegin();
194 uint8_t index = 0;
195 for (; index < DBC_TAXI_MASK_SIZE && iter != tokens.cend(); ++iter, ++index)
196 {
197 if (const uint32_t mask = std::stoul((*iter).c_str()))
198 {
199 // load and set bits only for existing taxi nodes
200 m_taximask[index] = sTaxiNodesMask[index] & mask;
201 }
202 else
203 {
204 m_taximask[index] = 0;
205 }
206 }
207}

216bool TaxiPath::setTaximaskNode(uint32_t nodeidx)
217{
218 const auto field = getFieldForTaxiNode(nodeidx);
219 const auto submask = getSubmaskForTaxiNode(nodeidx);
220 if ((m_taximask[field] & submask) != submask)
221 {
222 m_taximask[field] |= submask;
223 return true;
224 }
225 return false;

236bool TaxiPath::loadTaxiDestinationsFromString(std::string const& values, uint32_t team)
237{
238 clearTaxiDestinations();
239
240 const auto tokens = AscEmu::Util::Strings::split(values, " ");
241 for (const auto& itr : tokens)
242 {
243 if (const uint32_t node = std::stoul((itr).c_str()))
244 addTaxiDestination(node);
245 else
246 return false;
247 }
248
249 if (m_TaxiDestinations.empty())
250 return true;
251
252 // Check integrity
253 if (m_TaxiDestinations.size() < 2)
254 return false;
255
256 for (size_t i = 1; i < m_TaxiDestinations.size(); ++i)
257 {
258 uint32_t cost;
259 uint32_t path;
260 sTaxiMgr.getTaxiPath(m_TaxiDestinations[i - 1], m_TaxiDestinations[i], path, cost);
261 if (!path)
262 return false;
263 }
264
265 // can't load taxi path without mount set (quest taxi path?)
266 if (!sTaxiMgr.getTaxiMountDisplayId(getTaxiSource(), team, true))
267 return false;
268
269 return true;
270}

313uint32_t TaxiMgr::getNearestTaxiNode(LocationVector const& pos, uint32_t mapid, uint32_t team) const
314{
315 bool found = false;
316 float dist = 10000;
317 uint32_t id = 0;
318
319 for (uint32_t i = 1; i < sTaxiNodesStore.getNumRows(); ++i)
320 {
321 const auto node = sTaxiNodesStore.lookupEntry(i);
322 if (node == nullptr || node->mapid != mapid || (!node->mountCreatureID[team == TEAM_ALLIANCE ? 1 : 0] && node->mountCreatureID[0] != 32981)) // dk flight
323 continue;
324
325 const auto field = getFieldForTaxiNode(i);
326 const auto submask = getSubmaskForTaxiNode(i);
327
328 // skip not taxi network nodes
329 if ((sTaxiNodesMask[field] & submask) == 0)
330 continue;
331
332 float dist2 = (node->x - pos.x) * (node->x - pos.x) + (node->y - pos.y) * (node->y - pos.y) + (node->z - pos.z) * (node->z - pos.z);
333 if (found)
334 {
335 if (dist2 < dist)
336 {
337 dist = dist2;
338 id = i;
339 }
340 }
341 else
342 {
343 found = true;
344 dist = dist2;
345 id = i;
346 }
347 }
348
349 return id;
350}

352void TaxiMgr::getTaxiPath(uint32_t source, uint32_t destination, uint32_t& path, uint32_t& cost) const
353{
354 const auto src_i = sTaxiPathSetBySource.find(source);
355 if (src_i == sTaxiPathSetBySource.cend())
356 {
357 path = 0;
358 cost = 0;
359 return;
360 }
361
362 const auto& pathSet = src_i->second;
363
364 const auto dest_i = pathSet.find(destination);
365 if (dest_i == pathSet.cend())
366 {
367 path = 0;
368 cost = 0;
369 return;
370 }
371
372 cost = dest_i->second.price;
373 path = dest_i->second.ID;
374}

376uint32_t TaxiMgr::getTaxiMountDisplayId(uint32_t id, uint32_t team, bool allowed_alt_team /* = false */) const
377{
378 uint32_t mount_id = 0;
379
380 // select mount creature id
381 if (const auto node = sTaxiNodesStore.lookupEntry(id))
382 {
383 uint32_t mount_entry = 0;
384 if (team == TEAM_ALLIANCE)
385 mount_entry = node->mountCreatureID[1];
386 else
387 mount_entry = node->mountCreatureID[0];
388
389 // Fix for Alliance not being able to use Acherus taxi
390 // only one mount type for both sides
391 if (mount_entry == 0 && allowed_alt_team)
392 {
393 // Simply reverse the selection. At least one team in theory should have a valid mount ID to choose.
394 mount_entry = team == TEAM_ALLIANCE ? node->mountCreatureID[0] : node->mountCreatureID[1];
395 }
396
397 if (const auto mount_info = sMySQLStore.getCreatureProperties(mount_entry))
398 {
399 mount_id = mount_info->getRandomModelId();
400 if (mount_id == 0)
401 {
402 sLogger.failure("TaxiMgr:::No displayid found for the taxi mount with the entry {}! Can't load it!", mount_entry);
403 return 0;
404 }
405 }
406 }
407
408 return mount_id;
409}

411void TaxiMgr::loadTaxiNodeLevelData()
412{
413 auto oldMSTime = Util::TimeNow();
414
415 // 0 1
416 auto result = WorldDatabase.Query("SELECT TaxiNodeId, `Level` FROM taxi_level_data ORDER BY TaxiNodeId ASC");
417
418 if (!result)
419 {
420 sLogger.info("TaxiMgr:: Loaded 0 taxi node level definitions. DB table `taxi_level_data` is empty.");
421 return;
422 }
423
424 uint32_t count = 0;
425 do
426 {
427 Field* fields = result->Fetch();
428
429 uint32_t taxiNodeId = fields[0].asUint16();
430 uint8_t level = fields[1].asUint8();
431
432 const auto node = sTaxiNodesStore.lookupEntry(taxiNodeId);
433 if (node == nullptr)
434 {
435 sLogger.failure("TaxiMgr:: Table `taxi_level_data` has data for nonexistent taxi node (ID: {}), skipped", taxiNodeId);
436 continue;
437 };
438
439 _taxiNodeLevelDataStore.emplace(taxiNodeId, level);
440
441 ++count;
442 } while (result->NextRow());
443
444 sLogger.info("TaxiMgr:: Loaded {} taxi node level definitions in {} ms", count, Util::GetTimeDifferenceToNow(oldMSTime));
445}

456void TaxiMgr::initialize()
457{
458#if VERSION_STRING > WotLK
459 loadTaxiNodeLevelData();
460#endif
461
462 // Initialize global taxinodes mask
463 // include existed nodes that have at least single not spell base (scripted) path
464 std::set<uint32_t> spellPaths;
465
466#if VERSION_STRING > WotLK
467 for (uint32_t i = 0; i < sSpellEffectStore.getNumRows(); ++i)
468 {
469 if (const auto sInfo = sSpellEffectStore.lookupEntry(i))
470 {
471 if (sInfo->Effect == SPELL_EFFECT_START_TAXI)
472 spellPaths.insert(sInfo->EffectMiscValue);
473 }
474 }
475#else
476 for (uint32_t i = 0; i < sSpellStore.getNumRows(); ++i)
477 {
478 if (const auto sInfo = sSpellStore.lookupEntry(i))
479 {
480 for (uint8_t j = 0; j < MAX_SPELL_EFFECTS; ++j)
481 {
482 if (sInfo->Effect[j] == SPELL_EFFECT_START_TAXI)
483 spellPaths.insert(sInfo->EffectMiscValue[j]);
484 }
485 }
486 }
487#endif
488
489 sTaxiNodesMask.fill(0);
490 sOldContinentsNodesMask.fill(0);
491 sHordeTaxiNodesMask.fill(0);
492 sAllianceTaxiNodesMask.fill(0);
493 sDeathKnightTaxiNodesMask.fill(0);
494 for (uint32_t i = 1; i < sTaxiNodesStore.getNumRows(); ++i)
495 {
496 const auto node = sTaxiNodesStore.lookupEntry(i);
497 if (node == nullptr)
498 continue;
499
500 const auto src_i = std::as_const(sTaxiPathSetBySource).find(i);
501 if (src_i != sTaxiPathSetBySource.cend() && !src_i->second.empty())
502 {
503 bool ok = false;
504 for (auto dest_i = src_i->second.begin(); dest_i != src_i->second.end(); ++dest_i)
505 {
506 // not spell path
507 if (spellPaths.find(dest_i->second.ID) == spellPaths.end())
508 {
509 ok = true;
510 break;
511 }
512 }
513
514 if (!ok)
515 continue;
516 }
517
518 // valid taxi network node
519 const auto field = getFieldForTaxiNode(i);
520 const auto submask = getSubmaskForTaxiNode(i);
521
522 sTaxiNodesMask[field] |= submask;
523 if (node->mountCreatureID[0] && node->mountCreatureID[0] != 32981)
524 sHordeTaxiNodesMask[field] |= submask;
525 if (node->mountCreatureID[1] && node->mountCreatureID[1] != 32981)
526 sAllianceTaxiNodesMask[field] |= submask;
527 if (node->mountCreatureID[0] == 32981 || node->mountCreatureID[1] == 32981)
528 sDeathKnightTaxiNodesMask[field] |= submask;
529
530 // old continent node (+ nodes virtually at old continents, check explicitly to avoid loading map files for zone info)
531 if (node->mapid < 2 || i == 82 || i == 83 || i == 93 || i == 94)
532 sOldContinentsNodesMask[field] |= submask;
533
534 // fix DK node at Ebon Hold and Shadow Vault flight master
535 if (i == 315 || i == 333)
536 ((WDB::Structures::TaxiNodesEntry*)node)->mountCreatureID[1] = 32981;
537 }
