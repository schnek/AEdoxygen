32void TileMap::load(char* filename)
33{
34 sLogger.debug("Loading {}", filename);
35 FILE* f = fopen(filename, "rb");
36
37 if (f == NULL)
38 {
39 sLogger.failure("{} does not exist", filename);
40 return;
41 }
42
43 TileMapHeader header;
44
45 if (fread(&header, sizeof(header), 1, f) != 1)
46 {
47 fclose(f);
48 return;
49 }
50
51#if VERSION_STRING < Mop
52 if (header.buildMagic != BUILD_VERSION) // wow version
53 {
54 sLogger.failure("{}: from incorrect client (you: {} us: {})", filename, header.buildMagic, BUILD_VERSION);
55 fclose(f);
56 return;
57 }
58#endif
59
60 if (header.areaMapOffset != 0)
61 loadAreaData(f, header);
62
63 if (header.heightMapOffset != 0)
64 loadHeightData(f, header);
65
66 if (header.liquidMapOffset != 0)
67 loadLiquidData(f, header);
68
69 if (header.holesSize != 0)
70 loadHolesData(f, header);
71
72 fclose(f);
73}

75void TileMap::loadAreaData(FILE* f, TileMapHeader& header)
76{
77 TileMapAreaHeader areaHeader;
78
79 if (fseek(f, header.areaMapOffset, SEEK_SET) != 0)
80 return;
81
82 if (fread(&areaHeader, sizeof(areaHeader), 1, f) != 1)
83 return;
84
85 m_tileArea = areaHeader.gridArea;
86 if (!(areaHeader.flags & MAP_AREA_NO_AREA))
87 {
88 m_areaMap = std::make_unique<uint16_t[]>(16 * 16);
89 if (fread(m_areaMap.get(), sizeof(uint16_t), 16 * 16, f) != 16 * 16)
90 return;
91 }

94void TileMap::loadHeightData(FILE* f, TileMapHeader& header)
95{
96 TileMapHeightHeader mapHeader;
97
98 if (fseek(f, header.heightMapOffset, SEEK_SET) != 0)
99 return;
100
101 if (fread(&mapHeader, sizeof(mapHeader), 1, f) != 1)
102 return;
103
104 m_tileHeight = mapHeader.gridHeight;
105 m_heightMapFlags = mapHeader.flags;
106
107 if (!(m_heightMapFlags & MAP_HEIGHT_NO_HEIGHT))
108 {
109 if (m_heightMapFlags & MAP_HEIGHT_AS_INT16)
110 {
111 m_tileHeightMultiplier = (mapHeader.gridMaxHeight - mapHeader.gridHeight) / 65535;
112 m_gridGetHeight = &TileMap::getHeightFromUint16;
113
114 m_heightMap9S = std::make_unique<uint16_t[]>(129 * 129);
115 m_heightMap8S = std::make_unique<uint16_t[]>(128 * 128);
116 if (fread(m_heightMap9S.get(), sizeof(uint16_t), 129 * 129, f) != 129 * 129 ||
117 fread(m_heightMap8S.get(), sizeof(uint16_t), 128 * 128, f) != 128 * 128)
118 return;
119 }
120 else if (m_heightMapFlags & MAP_HEIGHT_AS_INT8)
121 {
122 m_tileHeightMultiplier = (mapHeader.gridMaxHeight - mapHeader.gridHeight) / 255;
123 m_gridGetHeight = &TileMap::getHeightFromUint8;
124
125 m_heightMap9B = std::make_unique<uint8_t[]>(129 * 129);
126 m_heightMap8B = std::make_unique<uint8_t[]>(128 * 128);
127 if (fread(m_heightMap9B.get(), sizeof(uint8_t), 129 * 129, f) != 129 * 129 ||
128 fread(m_heightMap8B.get(), sizeof(uint8_t), 128 * 128, f) != 128 * 128)
129 return;
130 }
131 else
132 {
133 m_heightMap9F = std::make_unique<float[]>(129 * 129);
134 m_heightMap8F = std::make_unique<float[]>(128 * 128);
135 if (fread(m_heightMap9F.get(), sizeof(float), 129 * 129, f) != 129 * 129 ||
136 fread(m_heightMap8F.get(), sizeof(float), 128 * 128, f) != 128 * 128)
137 return;
138
139 m_gridGetHeight = &TileMap::getHeightFromFloat;
140 }
141 }
142 else
143 {
144 m_gridGetHeight = &TileMap::getHeightFromFlat;
145 }
146}

148void TileMap::loadLiquidData(FILE* f, TileMapHeader& header)
149{
150 TileMapLiquidHeader liquidHeader;
151
152 if (fseek(f, header.liquidMapOffset, SEEK_SET) != 0)
153 return;
154
155 if (fread(&liquidHeader, sizeof(liquidHeader), 1, f) != 1)
156 return;
157
158 m_liquidGlobalEntry = liquidHeader.liquidType;
159 m_liquidGlobalFlags = liquidHeader.liquidFlags;
160 m_liquidOffX = liquidHeader.offsetX;
161 m_liquidOffY = liquidHeader.offsetY;
162 m_liquidWidth = liquidHeader.width;
163 m_liquidHeight = liquidHeader.height;
164 m_liquidLevel = liquidHeader.liquidLevel;
165
166 if (!(liquidHeader.flags & MAP_LIQUID_NO_TYPE))
167 {
168 m_liquidEntry = std::make_unique<uint16_t[]>(16 * 16);
169 if (fread(m_liquidEntry.get(), sizeof(uint16_t), 16 * 16, f) != 16 * 16)
170 return;
171
172 m_liquidFlags = std::make_unique<uint8_t[]>(16 * 16);
173 if (fread(m_liquidFlags.get(), sizeof(uint8_t), 16 * 16, f) != 16 * 16)
174 return;
175 }
176
177 if (!(liquidHeader.flags & MAP_LIQUID_NO_HEIGHT))
178 {
179 m_liquidMap = std::make_unique<float[]>(m_liquidWidth * m_liquidHeight);
180 if (fread(m_liquidMap.get(), sizeof(float), m_liquidWidth * m_liquidHeight, f) != 16 * 16)
181 return;
182 }
183}

195bool TileMap::isHole(int row, int col) const
196{
197 if (!m_holes)
198 return false;
199
200 int cellRow = row / 8; // 8 squares per cell
201 int cellCol = col / 8;
202 int holeRow = row % 8 / 2;
203 int holeCol = (col - (cellCol * 8)) / 2;
204
205 uint16_t hole = m_holes[cellRow * 16 + cellCol];
206
207 return (hole & holetab_h[holeCol] & holetab_v[holeRow]) != 0;
208}

215float TileMap::getHeightFromFloat(float x, float y) const
216{
217 if (!m_heightMap8F || !m_heightMap9F)
218 return m_tileHeight;
219
220 x = Map::Terrain::MapResoloution * (Map::Terrain::MapCenter - x / Map::Terrain::TileSize);
221 y = Map::Terrain::MapResoloution * (Map::Terrain::MapCenter - y / Map::Terrain::TileSize);
222
223 int x_int = static_cast<int>(x);
224 int y_int = static_cast<int>(y);
225 x -= x_int;
226 y -= y_int;
227 x_int &= (Map::Terrain::MapResoloution - 1);
228 y_int &= (Map::Terrain::MapResoloution - 1);
229
230 if (isHole(x_int, y_int))
231 return INVALID_HEIGHT;
232
233 // Height stored as: h5 - its v8 grid, h1-h4 - its v9 grid
234 // +--------------> X
235 // | h1-------h2 Coordinates is:
236 // | | \ 1 / | h1 0, 0
237 // | | \ / | h2 0, 1
238 // | | 2 h5 3 | h3 1, 0
239 // | | / \ | h4 1, 1
240 // | | / 4 \ | h5 1/2, 1/2
241 // | h3-------h4
242 // V Y
243 // For find height need
244 // 1 - detect triangle
245 // 2 - solve linear equation from triangle points
246 // Calculate coefficients for solve h = a*x + b*y + c
247
248 float a, b, c;
249 // Select triangle:
250 if (x + y < 1)
251 {
252 if (x > y)
253 {
254 // 1 triangle (h1, h2, h5 points)
255 float h1 = m_heightMap9F[(x_int) * 129 + y_int];
256 float h2 = m_heightMap9F[(x_int + 1) * 129 + y_int];
257 float h5 = 2 * m_heightMap8F[x_int * 128 + y_int];
258 a = h2 - h1;
259 b = h5 - h1 - h2;
260 c = h1;
261 }
262 else
263 {
264 // 2 triangle (h1, h3, h5 points)
265 float h1 = m_heightMap9F[x_int * 129 + y_int];
266 float h3 = m_heightMap9F[x_int * 129 + y_int + 1];
267 float h5 = 2 * m_heightMap8F[x_int * 128 + y_int];
268 a = h5 - h1 - h3;
269 b = h3 - h1;
270 c = h1;
271 }
272 }
273 else
274 {
275 if (x > y)
276 {
277 // 3 triangle (h2, h4, h5 points)
278 float h2 = m_heightMap9F[(x_int + 1) * 129 + y_int];
279 float h4 = m_heightMap9F[(x_int + 1) * 129 + y_int + 1];
280 float h5 = 2 * m_heightMap8F[x_int * 128 + y_int];
281 a = h2 + h4 - h5;
282 b = h4 - h2;
283 c = h5 - h4;
284 }
285 else
286 {
287 // 4 triangle (h3, h4, h5 points)
288 float h3 = m_heightMap9F[(x_int) * 129 + y_int + 1];
289 float h4 = m_heightMap9F[(x_int + 1) * 129 + y_int + 1];
290 float h5 = 2 * m_heightMap8F[x_int * 128 + y_int];
291 a = h4 - h3;
292 b = h3 + h4 - h5;
293 c = h5 - h4;
294 }
295 }
296 // Calculate height
297 return a * x + b * y + c;
298}

300float TileMap::getHeightFromUint8(float x, float y) const
301{
302 if (!m_heightMap8B || !m_heightMap9B)
303 return m_tileHeight;
304
305 x = Map::Terrain::MapResoloution * (Map::Terrain::MapCenter - x / Map::Terrain::TileSize);
306 y = Map::Terrain::MapResoloution * (Map::Terrain::MapCenter - y / Map::Terrain::TileSize);
307
308 int x_int = static_cast<int>(x);
309 int y_int = static_cast<int>(y);
310 x -= x_int;
311 y -= y_int;
312 x_int &= (Map::Terrain::MapResoloution - 1);
313 y_int &= (Map::Terrain::MapResoloution - 1);
314
315 if (isHole(x_int, y_int))
316 return INVALID_HEIGHT;
317
318 int32_t a, b, c;
319 uint8_t* V9_h1_ptr = &m_heightMap9B[x_int * 128 + x_int + y_int];
320 if (x + y < 1)
321 {
322 if (x > y)
323 {
324 // 1 triangle (h1, h2, h5 points)
325 int32_t h1 = V9_h1_ptr[0];
326 int32_t h2 = V9_h1_ptr[129];
327 int32_t h5 = 2 * m_heightMap8B[x_int * 128 + y_int];
328 a = h2 - h1;
329 b = h5 - h1 - h2;
330 c = h1;
331 }
332 else
333 {
334 // 2 triangle (h1, h3, h5 points)
335 int32_t h1 = V9_h1_ptr[0];
336 int32_t h3 = V9_h1_ptr[1];
337 int32_t h5 = 2 * m_heightMap8B[x_int * 128 + y_int];
338 a = h5 - h1 - h3;
339 b = h3 - h1;
340 c = h1;
341 }
342 }
343 else
344 {
345 if (x > y)
346 {
347 // 3 triangle (h2, h4, h5 points)
348 int32_t h2 = V9_h1_ptr[129];
349 int32_t h4 = V9_h1_ptr[130];
350 int32_t h5 = 2 * m_heightMap8B[x_int * 128 + y_int];
351 a = h2 + h4 - h5;
352 b = h4 - h2;
353 c = h5 - h4;
354 }
355 else
356 {
357 // 4 triangle (h3, h4, h5 points)
358 int32_t h3 = V9_h1_ptr[1];
359 int32_t h4 = V9_h1_ptr[130];
360 int32_t h5 = 2 * m_heightMap8B[x_int * 128 + y_int];
361 a = h4 - h3;
362 b = h3 + h4 - h5;
363 c = h5 - h4;
364 }
365 }
366 // Calculate height
367 return static_cast<float>((a * x) + (b * y) + c) * m_tileHeightMultiplier + m_tileHeight;
368}

453{

471// Get water state on map
472ZLiquidStatus TileMap::getLiquidStatus(LocationVector pos, uint8_t ReqLiquidType, LiquidData* data, float collisionHeight)
473{
474 // Check water type (if no water return)
475 if (!m_liquidGlobalFlags && !m_liquidFlags)
476 return LIQUID_MAP_NO_WATER;
477
478 // Get cell
479 float cx = Map::Terrain::MapResoloution * (Map::Terrain::MapCenter - pos.x / Map::Terrain::TileSize);
480 float cy = Map::Terrain::MapResoloution * (Map::Terrain::MapCenter - pos.y / Map::Terrain::TileSize);
481
482 int x_int = static_cast<int>(cx) & (Map::Terrain::MapResoloution - 1);
483 int y_int = static_cast<int>(cy) & (Map::Terrain::MapResoloution - 1);
484
485 // Check water type in cell
486 int idx = (x_int >> 3) * 16 + (y_int >> 3);
487 uint8_t type = m_liquidFlags ? m_liquidFlags[idx] : m_liquidGlobalFlags;
488 uint32_t entry = m_liquidEntry ? m_liquidEntry[idx] : m_liquidGlobalEntry;
489 if (WDB::Structures::LiquidTypeEntry const* liquidEntry = sLiquidTypeStore.lookupEntry(entry))
490 {
491 type &= MAP_LIQUID_TYPE_DARK_WATER;
492 uint32_t liqTypeIdx = liquidEntry->Type;
493 if (entry < 21)
494 {
495 if (WDB::Structures::AreaTableEntry const* area = sAreaStore.lookupEntry(getArea(pos.x, pos.y)))
496 {
497#if VERSION_STRING > Classic
498 uint32_t overrideLiquid = area->liquid_type_override[liquidEntry->Type];
499 if (!overrideLiquid && area->zone)
500 {
501 area = MapManagement::AreaManagement::AreaStorage::GetAreaById(area->zone);
502 if (area)
503 overrideLiquid = area->liquid_type_override[liquidEntry->Type];
504 }
505#else
506 uint32_t overrideLiquid = 0;
507#endif
508
509 if (WDB::Structures::LiquidTypeEntry const* liq = sLiquidTypeStore.lookupEntry(overrideLiquid))
510 {
511 entry = overrideLiquid;
512 liqTypeIdx = liq->Type;
513 }
514 }
515 }
516
517 type |= 1 << liqTypeIdx;
518 }
519
520 if (type == 0)
521 return LIQUID_MAP_NO_WATER;
522
523 // Check req liquid type mask
524 if (ReqLiquidType && !(ReqLiquidType & type))
525 return LIQUID_MAP_NO_WATER;
526
527 // Check water level:
528 // Check water height map
529 int lx_int = x_int - m_liquidOffY;
530 int ly_int = y_int - m_liquidOffX;
531 if (lx_int < 0 || lx_int >= m_liquidHeight)
532 return LIQUID_MAP_NO_WATER;
533 if (ly_int < 0 || ly_int >= m_liquidWidth)
534 return LIQUID_MAP_NO_WATER;
535
536 // Get water level
537 float liquid_level = m_liquidMap ? m_liquidMap[lx_int * m_liquidWidth + ly_int] : m_liquidLevel;
538 // Get ground level (sub 0.2 for fix some errors)
539 float ground_level = getHeight(pos.x, pos.y);
540
541 // Check water level and ground level
542 if (liquid_level < ground_level || pos.z < ground_level)
543 return LIQUID_MAP_NO_WATER;
544
545 // All ok in water -> store data
546 if (data)
547 {
548 data->entry = entry;
549 data->type_flags = type;
550 data->level = liquid_level;
551 data->depth_level = ground_level;
552 }
553
554 // For speed check as int values
555 float delta = liquid_level - pos.z;
556
557 if (delta > collisionHeight) // Under water
558 return LIQUID_MAP_UNDER_WATER;
559 if (delta > 0.0f) // In water
560 return LIQUID_MAP_IN_WATER;
561 if (delta > -0.1f) // Walk on water
562 return LIQUID_MAP_WATER_WALK;
563 // Above water
564 return LIQUID_MAP_ABOVE_WATER;
565}

586TerrainHolder::TerrainHolder(uint32_t mapid)
587{
588 TileCountX = TileCountY = 0;
589 TileStartX = TileEndX = 0;
590 TileStartY = TileEndY = 0;
591
592 for (uint8_t i = 0; i < Map::Terrain::TilesCount; ++i)
593 {
594 for (uint8_t j = 0; j < Map::Terrain::TilesCount; ++j)
595 {
596 m_tiles[i][j] = nullptr;
597 if (TileOffsets[i][j])
598 {
599 if (!TileStartX || TileStartX > i)
600 TileStartX = i;
601 if (!TileStartY || TileStartY > j)
602 TileStartY = j;
603 if (i > TileEndX)
604 TileEndX = i;
605 if (j > TileEndY)
606 TileEndY = j;
607 }
608 }
609 }
610
611 m_mapid = mapid;
612 TileCountX = (TileEndX - TileStartX) + 1;
613 TileCountY = (TileEndY - TileStartY) + 1;
614}
