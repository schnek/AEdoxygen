39void ThreatReference::updateOffline()
40{
41 bool const shouldBeOff = shouldBeOffline();
42 if (shouldBeOff == isOffline())
43 return;
44
45 if (shouldBeOff)
46 {
47 _online = ONLINE_STATE_OFFLINE;
48 _mgr.heapNotifyChanged();
49 _mgr.sendRemoveToClients(_victim);
50 }
51 else
52 {
53 _online = shouldBeSuppressed() ? ONLINE_STATE_SUPPRESSED : ONLINE_STATE_ONLINE;
54 _mgr.heapNotifyChanged();
55 }

58bool ThreatReference::shouldBeOffline() const
59{
60 if (!_owner->canSee(_victim))
61 return true;
62
63 if (!_owner->getAIInterface()->canOwnerAttackUnit(_victim))
64 return true;
65
66 if (!flagsAllowFighting(_owner, _victim) || !flagsAllowFighting(_victim, _owner))
67 return true;
68
69 return false;
70}

72bool ThreatReference::shouldBeSuppressed() const
73{
74 if (isTaunting()) // a taunting victim can never be suppressed
75 return false;
76
77 if (_owner->isCreature())
78 if (_victim->m_schoolImmunityList[_owner->BaseAttackType] != 0)
79 return true;
80
81 // check if we have any aura that suppresses us
82 return _victim->hasAuraWithAuraEffect(SPELL_AURA_MOD_STUN) || _victim->hasAuraWithAuraEffect(SPELL_AURA_MOD_CONFUSE);
83}

85void ThreatReference::updateTauntState(TauntState state)
86{
87#if VERSION_STRING >= TBC
88 // Check for SPELL_AURA_MOD_DETAUNT
89 if (state < TAUNT_STATE_TAUNT && _victim->hasAuraWithAuraEffect(SPELL_AURA_MOD_DETAUNT))
90 state = TAUNT_STATE_DETAUNT;
91#endif
92
93 if (state == _taunted)
94 return;
95
96 std::swap(state, _taunted);
97
98 _mgr.heapNotifyChanged();
99
100 _mgr._needClientUpdate = true;
101}

146/*static*/ bool ThreatManager::canHaveThreatList(Unit const* pUnit)
147{
148 // only creatures can have threat list
149 if (!pUnit->isCreature())
150 return false;
151
152 // pets, totems cannot have threat list
153 if (pUnit->isPet() || pUnit->isTotem())
154 return false;
155
156 // summons cannot have a threat list if they were summoned by a player
157 if (pUnit->isSummon())
158 {
159 // TODO: Summoner is set in Summon::Load and we call ThreatManager initialize from Summon constructor
160 // so summoner is always nullptr here. Maybe in the future it's set in constructor ;) -Appled
161 if (Unit* summoner = pUnit->getWorldMap()->getUnit(WoWGuid::getGuidLowPartFromUInt64(pUnit->getSummonedByGuid())))
162 if (summoner->isPlayer())
163 return false;
164 }
165
166 return true;
167}

220void ThreatManager::updateVictim()
221{
222 ThreatReference const* const newVictim = reselectVictim();

233ThreatReference const* ThreatManager::reselectVictim()
234{
235 if (_sortedThreatList.empty())
236 return nullptr;
237
238 for (auto & pair : _myThreatListEntries)
239 pair.second->updateOffline(); // AI notifies are processed in ::UpdateVictim caller
240
241 // fixated target is always preferred
242 if (_fixateRef && _fixateRef->isAvailable())
243 return _fixateRef;
244
245 ThreatReference const* oldVictimRef = _currentVictimRef;
246 if (oldVictimRef && oldVictimRef->isOffline())
247 oldVictimRef = nullptr;
248 // in 99% of cases - we won't need to actually look at anything beyond the first element
249 const auto& highest = _sortedThreatList.front();
250
251 // if the highest reference is offline, the entire list is offline, and we indicate this
252 if (!highest->isAvailable())
253 return nullptr;
254 // if we have no old victim, or old victim is still highest, then highest is our target and we're done
255 if (!oldVictimRef || highest.get() == oldVictimRef)
256 return highest.get();
257 // if highest threat doesn't break 110% of old victim, nothing below it is going to do so either; new victim = old victim and done
258 if (!ThreatManager::compareReferencesLT(oldVictimRef, highest.get(), 1.1f))
259 return oldVictimRef;
260 // if highest threat breaks 130%, it's our new target regardless of range (and we're done)
261 if (ThreatManager::compareReferencesLT(oldVictimRef, highest.get(), 1.3f))
262 return highest.get();
263 // if it doesn't break 130%, we need to check if it's melee - if yes, it breaks 110% (we checked earlier) and is our new target
264 if (_owner->canReachWithAttack(highest->_victim))
265 return highest.get();
266 // If we get here, highest threat is ranged, but below 130% of current - there might be a melee that breaks 110% below us somewhere, so now we need to actually look at the next highest element
267 // luckily, this is a heap, so getting the next highest element is O(log n), and we're just gonna do that repeatedly until we've seen enough targets (or find a target)
268 auto it = _sortedThreatList.begin(), end = _sortedThreatList.end();
269 while (it != end)
270 {
271 const auto& next = *it;
272 // if we've found current victim, we're done (nothing above is higher, and nothing below can be higher)
273 if (next.get() == oldVictimRef)
274 return next.get();
275 // if next isn't above 110% threat, then nothing below it can be either - we're done, old victim stays
276 if (!ThreatManager::compareReferencesLT(oldVictimRef, next.get(), 1.1f))
277 return oldVictimRef;
278 // if next is melee, he's above 110% and our new victim
279 if (_owner->canReachWithAttack(next->_victim))
280 return next.get();
281 // otherwise the next highest target may still be a melee above 110% and we need to look further
282 ++it;
283 }
284 // we should have found the old victim at some point in the loop above, so execution should never get to this point
285 return nullptr;
286}

288// returns true if a is LOWER on the threat list than b
289/*static*/ bool ThreatManager::compareReferencesLT(ThreatReference const* a, ThreatReference const* b, float aWeight)
290{
291 if (a->_online != b->_online) // online state precedence (ONLINE > SUPPRESSED > OFFLINE)
292 return a->_online < b->_online;
293 if (a->_taunted != b->_taunted) // taunt state precedence (TAUNT > NONE > DETAUNT)
294 return a->_taunted < b->_taunted;
295 return (a->getThreat()*aWeight < b->getThreat());
296}

320void ThreatManager::updateRedirectInfo()
321{
322 _redirectInfo.clear();
323 uint32_t totalPct = 0;
324 for (auto const& pair : _redirectRegistry) // (spellid, victim -> pct)
325 {
326 for (auto const& victimPair : pair.second) // (victim,pct)
327 {
328 uint32_t thisPct = std::min<uint32_t>(100 - totalPct, victimPair.second);
329 if (thisPct > 0)
330 {
331 _redirectInfo.push_back({ victimPair.first, thisPct });
332 totalPct += thisPct;
333 ASSERT(totalPct <= 100);
334 if (totalPct == 100)
335 return;
336 }
337 }
338 }

341void ThreatManager::forwardThreatForAssistingMe(Unit* assistant, float baseAmount, SpellInfo const* spell, bool ignoreModifiers)
342{
343 if (spell && spell->getAttributesEx() == ATTRIBUTESEX_NO_INITIAL_AGGRO) // shortcut, none of the calls would do anything
344 return;
345
346 if (_threatenedByMe.empty())
347 return;
348
349 std::vector<Creature*> canBeThreatened, cannotBeThreatened;
350 for (auto const& pair : _threatenedByMe)
351 {
352 Creature* owner = pair.second->getOwner();
353 if (!owner->hasUnitFlags(UNIT_FLAG_PLAYER_CONTROLLED_CREATURE))
354 canBeThreatened.push_back(owner);
355 else
356 cannotBeThreatened.push_back(owner);
357 }
358
359 if (!canBeThreatened.empty()) // targets under CC cannot gain assist threat - split evenly among the rest
360 {
361 float const perTarget = baseAmount / canBeThreatened.size();
362 for (Creature* threatened : canBeThreatened)
363 threatened->getThreatManager().addThreat(assistant, perTarget, spell, ignoreModifiers);
364 }
365
366 for (Creature* threatened : cannotBeThreatened)
367 threatened->getThreatManager().addThreat(assistant, 0.0f, spell, true);
368}

379/*static*/ float ThreatManager::calculateModifiedThreat(Unit* owner, float threat, Unit* victim, SpellInfo const* spell, Spell* castingSpell/* = nullptr*/)
380{
381 // modifiers by spell
382 if (spell)
383 {
384 if (spell->custom_ThreatForSpell)
385 if (spell->custom_ThreatForSpellCoef != 1.0f)
386 threat *= spell->custom_ThreatForSpellCoef;
387 }
388
389 if (castingSpell != nullptr)
390 owner->applySpellModifiers(SPELLMOD_THREAT_REDUCED, &threat, castingSpell->getSpellInfo(), castingSpell);
391
392 // modifiers by effect school
393 ThreatManager const& victimMgr = victim->getThreatManager();
394 SchoolMask const mask = spell ? SchoolMask(spell->getSchoolMask()) : SCHOOL_MASK_NORMAL;
395 switch (mask)
396 {
397 case SCHOOL_MASK_NORMAL:
398 threat *= victimMgr._singleSchoolModifiers[SCHOOL_NORMAL];
399 break;
400 case SCHOOL_MASK_HOLY:
401 threat *= victimMgr._singleSchoolModifiers[SCHOOL_HOLY];
402 break;
403 case SCHOOL_MASK_FIRE:
404 threat *= victimMgr._singleSchoolModifiers[SCHOOL_FIRE];
405 break;
406 case SCHOOL_MASK_NATURE:
407 threat *= victimMgr._singleSchoolModifiers[SCHOOL_NATURE];
408 break;
409 case SCHOOL_MASK_FROST:
410 threat *= victimMgr._singleSchoolModifiers[SCHOOL_FROST];
411 break;
412 case SCHOOL_MASK_SHADOW:
413 threat *= victimMgr._singleSchoolModifiers[SCHOOL_SHADOW];
414 break;
415 case SCHOOL_MASK_ARCANE:
416 threat *= victimMgr._singleSchoolModifiers[SCHOOL_ARCANE];
417 break;
418 default:
419 {
420 auto it = victimMgr._multiSchoolModifiers.find(mask);
421 if (it != victimMgr._multiSchoolModifiers.end())
422 {
423 threat *= it->second;
424 break;
425 }
426 float mod = static_cast<float>(victim->getGeneratedThreatModifyer(mask));
427 victimMgr._multiSchoolModifiers[mask] = mod;
428 threat *= mod;
429 break;
430 }
431 }
432 return threat;
433}

472void ThreatManager::addThreat(Unit* target, float amount, SpellInfo const* spell, bool ignoreModifiers, bool ignoreRedirects, Spell* castingSpell/* = nullptr*/)
473{
474 // Can we even have a Threat List if not return
475 if (!canHaveThreatList())
476 return;
477
478 if (!_owner->getAIInterface()->isAllowedToEnterCombat())
479 return;
480
481 if (!getOwner()->isAlive())
482 return;
483
484 if (!target)
485 return;
486
487 if (spell)
488 {
489 if (spell->getAttributesEx() == ATTRIBUTESEX_NO_INITIAL_AGGRO)
490 return;
491
492 if (!_owner->isInCombat() && spell->getAttributesExC() & ATTRIBUTESEXC_UNK19)
493 return;
494 }
495
496#ifdef FT_VEHICLES
497 // while riding a vehicle, all threat goes to the vehicle, not the pilot
498 if (Unit* vehicle = target->getVehicleBase())
499 {
500 addThreat(vehicle, amount, spell, ignoreModifiers, ignoreRedirects);
501 if (target->hasUnitStateFlag(UNIT_STATE_ACCESSORY)) // accessories are fully treated as components of the parent and cannot have threat
502 return;
503 amount = 0.0f;
504 }
505#endif
506
507 // apply threat modifiers to the amount
508 if (!ignoreModifiers)
509 amount = calculateModifiedThreat(getOwner(), amount, target, spell, castingSpell);
510
511 // if we're increasing threat, send some/all of it to redirection targets instead if applicable
512 if (!ignoreRedirects && amount > 0.0f)
513 {
514 auto const& redirInfo = target->getThreatManager()._redirectInfo;
515 if (!redirInfo.empty())
516 {
517 float const origAmount = amount;
518 // intentional iteration by index - there's a nested AddThreat call further down that might cause AI calls which might modify redirect info through spells
519 for (size_t i = 0; i < redirInfo.size(); ++i)
520 {
521 auto const pair = redirInfo[i]; // (victim,pct)
522 Unit* redirTarget = nullptr;
523 auto it = _myThreatListEntries.find(pair.first); // try to look it up in our threat list first (faster)
524 if (it != _myThreatListEntries.end())
525 redirTarget = it->second->_victim;
526 else
527 redirTarget = target->getWorldMapUnit(pair.first);
528
529 if (redirTarget)
530 {
531 float amountRedirected = (origAmount * static_cast<float>(pair.second) / 100.0f);
532 addThreat(redirTarget, amountRedirected, spell, true, true);
533 amount -= amountRedirected;
534 }
535 }
536 }
537 }
538
539 // Damage Dealt for Scripts
540 getOwner()->getAIInterface()->handleEvent(EVENT_DAMAGETAKEN, target, static_cast<uint32_t>(amount));
541
542 // ok, now we actually apply threat
543 // check if we already have an entry - if we do, just increase threat for that entry and we're done
544 auto it = _myThreatListEntries.find(target->getGuid());
545 if (it != _myThreatListEntries.end())
546 {
547 const auto& ref = it->second;
548 // SUPPRESSED threat states don't go back to ONLINE until threat is caused by them (retail behavior)
549 if (ref->getOnlineState() == ThreatReference::ONLINE_STATE_SUPPRESSED)
550 if (!ref->shouldBeSuppressed())
551 {
552 ref->_online = ThreatReference::ONLINE_STATE_ONLINE;
553 heapNotifyChanged();
554 }
555
556 ref->updateOffline();
557 if (ref->isOnline())
558 ref->addThreat(amount);
559 return;
560 }
561
562 // ok, we're now in combat - create the threat list reference and push it to the respective managers
563 // std::make_shared does not work with private ctors -Appled
564 auto ref = std::shared_ptr<ThreatReference>(new ThreatReference(this, target));
565 putThreatListRef(target->getGuid(), ref);
566 target->getThreatManager().putThreatenedByMeRef(_owner->getGuid(), ref);
567
568 ref->updateOffline();
569 if (ref->isOnline()) // we only add the threat if the ref is currently available
570 ref->addThreat(amount);
571
572 if (!_currentVictimRef)
573 updateVictim();
574}

600void ThreatManager::matchUnitThreatToHighestThreat(Unit* target)
601{
602 if (_sortedThreatList.empty())
603 return;
604
605 auto it = _sortedThreatList.begin(), end = _sortedThreatList.end();
606 ThreatReference const* highest = (*it).get();
607 if (!highest->isAvailable())
608 return;
609
610 if (highest->isTaunting() && ((++it) != end)) // might need to skip this - max threat could be the preceding element (there is only one taunt element)
611 {
612 const auto& a = *it;
613 if (a->isAvailable() && a->getThreat() > highest->getThreat())
614 highest = a.get();
615 }
616
617 addThreat(target, highest->getThreat() - getThreat(target, true), nullptr, true, true);
618}

620void ThreatManager::tauntUpdate()
621{
622 _tauntEffects.clear();
623
624 // check if we have any aura that taunts us
625 for (const auto& aurEff : _owner->getAuraEffectList(SPELL_AURA_MOD_TAUNT))
626 {
627 _tauntEffects.push_back(aurEff->getAura()->getCasterGuid());
628 }
629
630 uint32_t state = ThreatReference::TAUNT_STATE_TAUNT;
631 std::unordered_map<uint64_t, ThreatReference::TauntState> tauntStates;
632 // Only the last taunt effect applied by something still on our threat list is considered
633 for (auto it = _tauntEffects.begin(), end = _tauntEffects.end(); it != end; ++it)
634 tauntStates[(*it)] = ThreatReference::TauntState(state++);
635
636 for (auto const& pair : _myThreatListEntries)
637 {
638 auto it = tauntStates.find(pair.first);
639 if (it != tauntStates.end())
640 pair.second->updateTauntState(it->second);
641 else
642 pair.second->updateTauntState();
643 }
644
645 // taunt aura update also re-evaluates all suppressed states (retail behavior)
646 evaluateSuppressed(true);
647}

709void ThreatManager::sendClearAllThreatToClients() const
710{
711 ByteBuffer packedGuidOwner;
