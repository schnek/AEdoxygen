99{
100 char* Comment = strchr(args, ' ');
101 if (Comment != nullptr)
102 {
103 if (isdigit(*(Comment + 1))) // this is the duration of the ban
104 {
105 *Comment = 0; // nullptr-terminate the first string (character/account/ip)

112bool ChatCommandHandler::HandleTicketCloseCommand(const char* args, WorldSession* m_session)
113{
114 if (!args)
115 {
116 redSystemMessage(m_session, "You need to specify a ticket ID and a comment! Use .ticket close <ID> <comment text>");
117 return false;
118 }
119
120 char* ticket = (char*)args;
121 char* comment;
122 ParseTicketArgs(ticket, &comment);
123
124 if (comment == nullptr)
125 {
126 redSystemMessage(m_session, "You need to add a comment for closing an ticket!");
127 return false;
128 }
129
130 uint32_t ticketID = std::stoul(args);
131
132 Player* player = m_session->GetPlayer();
133
134 auto result = CharacterDatabase.Query("SELECT * FROM gm_tickets WHERE ticketid = %u AND deleted = 0", ticketID);
135 if (!result)
136 {
137 redSystemMessage(m_session, "Ticket {} is already closed!", ticketID);
138 return false;
139 }
140 Field* fields = result->Fetch();
141 uint32_t playerGuid = fields[1].asUint32();
142
143 GM_Ticket* gm_ticket = sTicketMgr.getGMTicketByPlayer(playerGuid);
144 if (gm_ticket == nullptr)
145 {
146 redSystemMessage(m_session, "Ticket not found.");
147 return true;
148 }
149
150 sTicketMgr.closeTicket(gm_ticket->guid);
151
152 Player* ticketOwner = sObjectMgr.getPlayer(playerGuid);
153
154 if (ticketOwner != nullptr)
155 {
156 ticketOwner->getSession()->SystemMessage("Your Ticket was closed by %s Comment: %s", player->getName().c_str(), comment);
157
158 // Notify player about removing ticket
159 ticketOwner->getSession()->SendPacket(SmsgGmTicketDeleteTicket(9).serialise().get());
160 // Response - Send GM Survey
161 ticketOwner->getSession()->SendPacket(SmsgGmTicketStatusUpdate(3).serialise().get());
162 }
163 else
164 {
165 std::string ticketIdString = std::to_string(ticketID);
166 std::string subject = "Your Ticket: ";
167 subject += ticketIdString;
168 sMailSystem.SendAutomatedMessage(MAIL_TYPE_NORMAL, player->getGuid(), playerGuid, subject, comment, 0, 0, 0, MAIL_STATIONERY_GM, MAIL_CHECK_MASK_NONE);
169 }
170
171 CharacterDatabase.Execute("UPDATE gm_tickets SET deleted = 1, comment = 'GM: %s %s', assignedto = %u WHERE ticketid = %u", player->getName().c_str(), comment, player->getGuid(), ticketID);
172 greenSystemMessage(m_session, "Ticket {} is now closed and assigned to you.", ticketID);
173 sGMLog.writefromsession(m_session, "closed ticket %u ", ticketID);
174 return true;
175}

177bool ChatCommandHandler::HandleTicketDeleteCommand(const char* args, WorldSession* m_session)
178{
179 if (!args)
180 {
181 redSystemMessage(m_session, "You need to specify a ticket ID");
182 return false;
183 }
184
185 uint32_t ticketID = std::stoul(args);
186
187 auto result = CharacterDatabase.Query("SELECT * FROM gm_tickets WHERE ticketid = %u AND deleted = 1", ticketID);
188 if (!result)
189 {
190 redSystemMessage(m_session, "Ticket {} is not available in gm_tickets table or not closed!", ticketID);
191 return false;
192 }
193
194 CharacterDatabase.Execute("DELETE FROM gm_tickets WHERE ticketid = %u", ticketID);
195 greenSystemMessage(m_session, "Ticket {} is deleted", ticketID);
196 sGMLog.writefromsession(m_session, "deleted ticket %u ", ticketID);
197
198 return true;
