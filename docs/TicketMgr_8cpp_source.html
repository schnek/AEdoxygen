87void TicketMgr::loadGMTickets()
88{

90 if (result == nullptr)

92 sLogger.info("TicketMgr : 0 active GM Tickets loaded.");

95

105void TicketMgr::saveGMTicket(GM_Ticket* ticket, QueryBuffer* buf)
106{
107 std::stringstream ss;
108
109 ss << "DELETE FROM gm_tickets WHERE ticketid = ";
110 ss << ticket->guid;
111 ss << ";";
112
113 if (buf == nullptr)
114 CharacterDatabase.ExecuteNA(ss.str().c_str());
115 else
116 buf->AddQueryStr(ss.str());
117
118 ss.rdbuf()->str("");
119
120 ss << "INSERT INTO gm_tickets (ticketid, playerguid, name, level, map, posX, posY, posZ, message, timestamp, deleted, assignedto, comment) VALUES(";
121 ss << ticket->guid << ", ";
122 ss << ticket->playerGuid << ", '";
123 ss << CharacterDatabase.EscapeString(ticket->name) << "', ";
124 ss << ticket->level << ", ";
125 ss << ticket->map << ", ";
126 ss << ticket->posX << ", ";
127 ss << ticket->posY << ", ";
128 ss << ticket->posZ << ", '";
129 ss << CharacterDatabase.EscapeString(ticket->message) << "', ";
130 ss << ticket->timestamp << ", ";
131
132 if (ticket->deleted)
133 ss << uint32_t(1);
134 else
135 ss << uint32_t(0);
136 ss << ",";
137
138 ss << ticket->assignedToPlayer << ", '";
139 ss << CharacterDatabase.EscapeString(ticket->comment) << "');";
140
141 if (buf == nullptr)
142 CharacterDatabase.ExecuteNA(ss.str().c_str());
143 else
144 buf->AddQueryStr(ss.str());
145}
