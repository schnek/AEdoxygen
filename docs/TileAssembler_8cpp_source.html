74 bool TileAssembler::convertWorld2()
75 {
76 bool success = readMapSpawns();
77 if (!success)
78 {
79 return false;
80 }
81
82 // export Map data
83 for (MapData::iterator map_iter = mapData.begin(); map_iter != mapData.end() && success; ++map_iter)
84 {
85 // build global map tree
86 std::vector<ModelSpawn*> mapSpawns;
87 UniqueEntryMap::iterator entry;
88 printf("Calculating model bounds for map %u...\n", map_iter->first);
89 for (entry = map_iter->second->UniqueEntries.begin(); entry != map_iter->second->UniqueEntries.end(); ++entry)
90 {
91 // M2 models don't have a bound set in WDT/ADT placement data, i still think they're not used for LoS at all on retail
92 if (entry->second.flags & MOD_M2)
93 {
94 if (!calculateTransformedBound(entry->second))
95 {
96 break;
97 }
98 }
99 else if (entry->second.flags & MOD_WORLDSPAWN) // WMO maps and terrain maps use different origin, so we need to adapt :/
100 {
101 /// @todo remove extractor hack and uncomment below line:
102 //entry->second.iPos += Vector3(533.33333f*32, 533.33333f*32, 0.f);
103 entry->second.iBound = entry->second.iBound + Vector3(533.33333f*32, 533.33333f*32, 0.f);
104 }
105 mapSpawns.push_back(&(entry->second));
106 spawnedModelFiles.insert(entry->second.name);
107 }
108
109 printf("Creating map tree for map %u...\n", map_iter->first);
110 BIH pTree;
111
112 try
113 {
114 pTree.build(mapSpawns, BoundsTrait<ModelSpawn*>::getBounds);
115 }
116 catch (std::exception& e)
117 {
118 printf("Exception ""%s"" when calling pTree.build", e.what());
119 return false;
120 }
121
122 // ===> possibly move this code to StaticMapTree class
123 std::map<uint32_t, uint32_t> modelNodeIdx;
124 for (uint32_t i = 0; i < mapSpawns.size(); ++i)
125 {
126 modelNodeIdx.insert(pair<uint32_t, uint32_t>(mapSpawns[i]->ID, i));
127 }
128
129 // write map tree file
130 std::stringstream mapfilename;
131 mapfilename << iDestDir << '/' << std::setfill('0') << std::setw(4) << map_iter->first << ".vmtree";
132 FILE* mapfile = fopen(mapfilename.str().c_str(), "wb");
133 if (!mapfile)
134 {
135 success = false;
136 printf("Cannot open %s\n", mapfilename.str().c_str());
137 break;
138 }
139
140 //general info
141 if (success && fwrite(VMAP_MAGIC, 1, 8, mapfile) != 8)
142 {
143 success = false;
144 }
145
146 uint32_t globalTileID = StaticMapTree::packTileID(65, 65);
147 pair<TileMap::iterator, TileMap::iterator> globalRange = map_iter->second->TileEntries.equal_range(globalTileID);
148 char isTiled = globalRange.first == globalRange.second; // only maps without terrain (tiles) have global WMO
149 if (success && fwrite(&isTiled, sizeof(char), 1, mapfile) != 1)
150 {
151 success = false;
152 }
153
154 // Nodes
155 if (success && fwrite("NODE", 4, 1, mapfile) != 1)
156 {
157 success = false;
158 }
159
160 if (success)
161 {
162 success = pTree.writeToFile(mapfile);
163 }
164
165 // global map spawns (WDT), if any (most instances)
166 if (success && fwrite("GOBJ", 4, 1, mapfile) != 1)
167 {
168 success = false;
169 }
170
171 for (TileMap::iterator glob=globalRange.first; glob != globalRange.second && success; ++glob)
172 {
173 success = ModelSpawn::writeToFile(mapfile, map_iter->second->UniqueEntries[glob->second]);
174 }
175
176 fclose(mapfile);
177
178 // <====
179
180 // write map tile files, similar to ADT files, only with extra BSP tree node info
181 TileMap &tileEntries = map_iter->second->TileEntries;
182 TileMap::iterator tile;
183 for (tile = tileEntries.begin(); tile != tileEntries.end(); ++tile)
184 {
185 const ModelSpawn &spawn = map_iter->second->UniqueEntries[tile->second];
186 if (spawn.flags & MOD_WORLDSPAWN) // WDT spawn, saved as tile 65/65 currently...
187 {
188 continue;
189 }
190
191 size_t nSpawns = tileEntries.count(tile->first);
192 std::stringstream tilefilename;
193 tilefilename.fill('0');
194 tilefilename << iDestDir << '/' << std::setw(4) << map_iter->first << '_';
195 uint32_t x, y;
196 StaticMapTree::unpackTileID(tile->first, x, y);
197 tilefilename << std::setw(2) << x << '_' << std::setw(2) << y << ".vmtile";
198 if (FILE* tilefile = fopen(tilefilename.str().c_str(), "wb"))
199 {
200 // file header
201 if (success && fwrite(VMAP_MAGIC, 1, 8, tilefile) != 8)
202 {
203 success = false;
204 }
205
206 // write number of tile spawns
207 if (success && fwrite(&nSpawns, sizeof(uint32_t), 1, tilefile) != 1)
208 {
209 success = false;
210 }
211
212 // write tile spawns
213 for (uint32_t s = 0; s < nSpawns; ++s)
214 {
215 if (s)
216 {
217 ++tile;
218 }
219
220 const ModelSpawn &spawn2 = map_iter->second->UniqueEntries[tile->second];
221 success = success && ModelSpawn::writeToFile(tilefile, spawn2);
222 // MapTree nodes to update when loading tile:
223 std::map<uint32_t, uint32_t>::iterator nIdx = modelNodeIdx.find(spawn2.ID);
224
225 if (success && fwrite(&nIdx->second, sizeof(uint32_t), 1, tilefile) != 1)
226 {
227 success = false;
228 }
229 }
230 fclose(tilefile);
231 }
232 }
233 // break; //test, extract only first map; TODO: remvoe this line
234 }

236 // add an object models, listed in temp_gameobject_models file
237 exportGameobjectModels();
238 // export objects
239 std::cout << "\nConverting Model Files" << "\n";
240 for (std::set<std::string>::iterator mfile = spawnedModelFiles.begin(); mfile != spawnedModelFiles.end(); ++mfile)

242 std::cout << "Converting " << *mfile << "\n";
243 if (!convertRawFile(*mfile))

245 std::cout << "error converting " << *mfile << "\n";
246 success = false;
247 break;
248 }
249 }
250
251 //cleanup:
252 mapData.clear();
253
254 return success;

257 bool TileAssembler::readMapSpawns()
258 {
259 std::string fname = iSrcDir + "/dir_bin";
260 FILE* dirf = fopen(fname.c_str(), "rb");
261 if (!dirf)
262 {
263 printf("Could not read dir_bin file!\n");
264 return false;
265 }
266 printf("Read coordinate mapping...\n");
267 uint32_t mapID, tileX, tileY = 0;
268 size_t check;
269 G3D::Vector3 v1, v2;
270 ModelSpawn spawn;
271 while (!feof(dirf))
272 {
273 check = 0;
274 // read mapID, tileX, tileY, Flags, adtID, ID, Pos, Rot, Scale, Bound_lo, Bound_hi, name
275 check += fread(&mapID, sizeof(uint32_t), 1, dirf);
276 if (check == 0) // EoF...
277 {
278 break;
279 }
280
281 check += fread(&tileX, sizeof(uint32_t), 1, dirf);
282 if (check == 0)
283 {
284 printf("Read tileX ends with an invalid check value of 0..\n");
285 }
286
287 check += fread(&tileY, sizeof(uint32_t), 1, dirf);
288 if (check == 0)
289 {
290 printf("Read tileY ends with an invalid check value of 0..\n");
291 }
292
293 if (!ModelSpawn::readFromFile(dirf, spawn))
294 {
295 break;
296 }
297
298 MapSpawns *current;
299 MapData::iterator map_iter = mapData.find(mapID);
300 if (map_iter == mapData.end())
301 {
302 printf("spawning Map %u\n", mapID);
303 mapData[mapID] = std::make_unique<MapSpawns>();
304 current = mapData[mapID].get();
305 }
306 else
307 {
308 current = (*map_iter).second.get();
309 }
310
311 current->UniqueEntries.emplace(pair<uint32_t, ModelSpawn>(spawn.ID, spawn));
312 current->TileEntries.emplace(pair<uint32_t, uint32_t>(StaticMapTree::packTileID(tileX, tileY), spawn.ID));
313 }
314 bool success = (ferror(dirf) == 0);
315 fclose(dirf);
316 return success;
317 }

319 bool TileAssembler::calculateTransformedBound(ModelSpawn &spawn)
320 {
321 std::string modelFilename(iSrcDir);
322 modelFilename.push_back('/');
323 modelFilename.append(spawn.name);
324
325 ModelPosition modelPosition;
326 modelPosition.iDir = spawn.iRot;
327 modelPosition.iScale = spawn.iScale;
328 modelPosition.init();
329
330 WorldModel_Raw raw_model;
331 if (!raw_model.Read(modelFilename.c_str()))
332 {
333 return false;
334 }
335
336 size_t groups = raw_model.groupsArray.size();
337 if (groups != 1)
338 {
339 printf("Warning: '%s' does not seem to be a M2 model!\n", modelFilename.c_str());
340 }
341
342 AABox modelBound;
343 bool boundEmpty = true;
344
345 for (uint32_t g = 0; g < groups; ++g) // should be only one for M2 files...
346 {
347 std::vector<Vector3>& vertices = raw_model.groupsArray[g].vertexArray;
348
349 if (vertices.empty())
350 {
351 std::cout << "error: model '" << spawn.name << "' has no geometry!" << "\n";
352 continue;
353 }
354
355 size_t nvectors = vertices.size();
356 for (uint32_t i = 0; i < nvectors; ++i)
357 {
358 Vector3 v = modelPosition.transform(vertices[i]);
359
360 if (boundEmpty)
361 {
362 modelBound = AABox(v, v), boundEmpty = false;
363 }
364 else
365 {
366 modelBound.merge(v);
367 }
368 }
369 }
370 spawn.iBound = modelBound + spawn.iPos;
371 spawn.flags |= MOD_HAS_BOUND;
372 return true;
373 }

384 bool TileAssembler::convertRawFile(const std::string& pModelFilename)
385 {
386 bool success = true;
387 std::string filename = iSrcDir;
388 if (filename.length() > 0)
389 {
390 filename.push_back('/');
391 }
392
393 filename.append(pModelFilename);
394
395 WorldModel_Raw raw_model;
396 if (!raw_model.Read(filename.c_str()))
397 {
398 return false;
399 }
400
401 // write WorldModel
402 WorldModel model;
403 model.setRootWmoID(raw_model.RootWMOID);
404 if (!raw_model.groupsArray.empty())
405 {
406 std::vector<GroupModel> groupsArray;
407
408 size_t groups = raw_model.groupsArray.size();
409 for (uint32_t g = 0; g < groups; ++g)
410 {
411 GroupModel_Raw& raw_group = raw_model.groupsArray[g];
412 groupsArray.emplace_back(GroupModel(raw_group.mogpflags, raw_group.GroupWMOID, raw_group.bounds ));
413 groupsArray.back().setMeshData(raw_group.vertexArray, raw_group.triangles);
414 groupsArray.back().setLiquidData(std::move(raw_group.liquid));
415 }
416
417 model.setGroupModels(groupsArray);
418 }
419
420 success = model.writeFile(iDestDir + "/" + pModelFilename + ".vmo");
421 //std::cout << "readRawFile2: '" << pModelFilename << "' tris: " << nElements << " nodes: " << nNodes << "\n";
422 return success;

425 void TileAssembler::exportGameobjectModels()
426 {
427 FILE* model_list = fopen((iSrcDir + "/" + "temp_gameobject_models").c_str(), "rb");
428 if (!model_list)
429 {
430 return;
431 }
432
433 FILE* model_list_copy = fopen((iDestDir + "/" + GAMEOBJECT_MODELS).c_str(), "wb");
434 if (!model_list_copy)
435 {
436 fclose(model_list);
437 return;
438 }
439
440 uint32_t name_length, displayId;
441 uint8_t isWmo;
442 char buff[500];
443 while (!feof(model_list))
444 {
445 if (fread(&displayId, sizeof(uint32_t), 1, model_list) != 1)
446 if (feof(model_list)) // EOF flag is only set after failed reading attempt
447 break;
448
449 if (fread(&isWmo, sizeof(uint8_t), 1, model_list) != 1
450 || fread(&name_length, sizeof(uint32_t), 1, model_list) != 1
451 || name_length >= sizeof(buff)
452 || fread(&buff, sizeof(char), name_length, model_list) != name_length)
453 {
454 std::cout << "\nFile 'temp_gameobject_models' seems to be corrupted" << "\n";
455 break;
456 }
457
458 std::string model_name(buff, name_length);
459
460 WorldModel_Raw raw_model;
461 if (!raw_model.Read((iSrcDir + "/" + model_name).c_str()))
462 {
463 continue;
464 }
465
466 spawnedModelFiles.insert(model_name);
467 AABox bounds;
468 bool boundEmpty = true;
469 for (uint32_t g = 0; g < raw_model.groupsArray.size(); ++g)
470 {
471 std::vector<Vector3>& vertices = raw_model.groupsArray[g].vertexArray;
472
473 size_t nvectors = vertices.size();
474 for (uint32_t i = 0; i < nvectors; ++i)
475 {
476 Vector3& v = vertices[i];
477 if (boundEmpty)
478 {
479 bounds = AABox(v, v), boundEmpty = false;
480 }
481 else
482 {
483 bounds.merge(v);
484 }
485 }
486 }
487
488 if (bounds.isEmpty())
489 {
490 std::cout << "\nModel " << std::string(buff, name_length) << " has empty bounding box" << "\n";
491 continue;
492 }
493
494 if (!bounds.isFinite())
495 {
496 std::cout << "\nModel " << std::string(buff, name_length) << " has invalid bounding box" << "\n";
497 continue;
498 }
499
500 fwrite(&displayId, sizeof(uint32_t), 1, model_list_copy);
501 fwrite(&isWmo, sizeof(uint8_t), 1, model_list_copy);
502 fwrite(&name_length, sizeof(uint32_t), 1, model_list_copy);
503 fwrite(&buff, sizeof(char), name_length, model_list_copy);
504 fwrite(&bounds.low(), sizeof(Vector3), 1, model_list_copy);
505 fwrite(&bounds.high(), sizeof(Vector3), 1, model_list_copy);
506 }
507
508 fclose(model_list);
509 fclose(model_list_copy);
510 }

519 bool GroupModel_Raw::Read(FILE* rf)
520 {
521 char blockId[5];
522 blockId[4] = 0;
523 int blocksize;
524 int readOperation = 0;
525
526 READ_OR_RETURN(&mogpflags, sizeof(uint32_t));
527 READ_OR_RETURN(&GroupWMOID, sizeof(uint32_t));
528
529
530 Vector3 vec1, vec2;
531 READ_OR_RETURN(&vec1, sizeof(Vector3));
532
533 READ_OR_RETURN(&vec2, sizeof(Vector3));
534 bounds.set(vec1, vec2);
535
536 READ_OR_RETURN(&liquidflags, sizeof(uint32_t));
537
538 // will this ever be used? what is it good for anyway??
539 uint32_t branches;
540 READ_OR_RETURN(&blockId, 4);
541 CMP_OR_RETURN(blockId, "GRP ");
542 READ_OR_RETURN(&blocksize, sizeof(int));
543 READ_OR_RETURN(&branches, sizeof(uint32_t));
544 for (uint32_t b=0; b<branches; ++b)
545 {
546 uint32_t indexes;
547 // indexes for each branch (not used jet)
548 READ_OR_RETURN(&indexes, sizeof(uint32_t));
549 }
550
551 // ---- indexes
552 READ_OR_RETURN(&blockId, 4);
553 CMP_OR_RETURN(blockId, "INDX");
554 READ_OR_RETURN(&blocksize, sizeof(int));
555 uint32_t nindexes;
556 READ_OR_RETURN(&nindexes, sizeof(uint32_t));
557 if (nindexes >0)
558 {
559 auto indexarray = std::make_unique<uint16_t[]>(nindexes);
560 READ_OR_RETURN_WITH_DELETE(indexarray.get(), nindexes * sizeof(uint16_t));
561 triangles.reserve(nindexes / 3);
562 for (uint32_t i = 0; i < nindexes; i += 3)
563 {
564 triangles.push_back(MeshTriangle(indexarray[i], indexarray[i + 1], indexarray[i + 2]));
565 }
566 }
567
568 // ---- vectors
569 READ_OR_RETURN(&blockId, 4);
570 CMP_OR_RETURN(blockId, "VERT");
571 READ_OR_RETURN(&blocksize, sizeof(int));
572 uint32_t nvectors;
573 READ_OR_RETURN(&nvectors, sizeof(uint32_t));
574
575 if (nvectors >0)
576 {
577 auto vectorarray = std::make_unique<float[]>(nvectors * 3);
578 READ_OR_RETURN_WITH_DELETE(vectorarray.get(), nvectors * sizeof(float) * 3);
579 for (uint32_t i = 0; i < nvectors; ++i)
580 {
581 vertexArray.push_back(Vector3(vectorarray.get() + 3 * i));
582 }
583 }
584 // ----- liquid
585 liquid = 0;
586 if (liquidflags& 1)
587 {
588 WMOLiquidHeader hlq;
589
590 // Function fread does not terminate string blockId
591 READ_OR_RETURN(&blockId, 4);
592
593 CMP_OR_RETURN(blockId, "LIQU");
594
595 READ_OR_RETURN(&blocksize, sizeof(int));
596 READ_OR_RETURN(&hlq, sizeof(WMOLiquidHeader));
597 liquid = std::make_unique<WmoLiquid>(hlq.xtiles, hlq.ytiles, Vector3(hlq.pos_x, hlq.pos_y, hlq.pos_z), hlq.type);
598 uint32_t size = hlq.xverts*hlq.yverts;
599 READ_OR_RETURN(liquid->GetHeightStorage(), size*sizeof(float));
600 size = hlq.xtiles*hlq.ytiles;
601 READ_OR_RETURN(liquid->GetFlagsStorage(), size);
602 }
603
604 return true;
605 }

611 {
612 FILE* rf = fopen(path, "rb");
613 if (!rf)
614 {
615 printf("ERROR: Can't open raw model file: %s\n", path);
616 return false;
617 }
618
619 char ident[9];
620 ident[8] = '\0';
621 int readOperation = 0;
622
623 // Function fread does not terminate string ident
624 READ_OR_RETURN(&ident, 8);
625
626 CMP_OR_RETURN(ident, RAW_VMAP_MAGIC);
627
628 // we have to read one int. This is needed during the export and we have to skip it here
629 uint32_t tempNVectors;
630 READ_OR_RETURN(&tempNVectors, sizeof(tempNVectors));
631
632 uint32_t groups;
633 READ_OR_RETURN(&groups, sizeof(uint32_t));
634 READ_OR_RETURN(&RootWMOID, sizeof(uint32_t));
635
636 groupsArray.resize(groups);
637 bool succeed = true;
638 for (uint32_t g = 0; g < groups && succeed; ++g)
639 {
640 succeed = groupsArray[g].Read(rf);
641 }
642
643 if (succeed) /// rf will be freed inside Read if the function had any errors.
644 {
645 fclose(rf);
646 }
647
648 return succeed;
649 }
