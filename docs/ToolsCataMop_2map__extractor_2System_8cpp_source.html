263void HandleArgs(int argc, char* arg[])
264{
265 for (int c = 1; c < argc; ++c)
266 {
267 // i - input path
268 // o - output path
269 // e - extract only MAP(1)/DBC(2)/Camera(4) - standard: all(7)
270 // f - use float to int conversion
271 // h - limit minimum height
272 // b - target client build
273 if (arg[c][0] != '-')
274 Usage(arg[0]);
275
276 switch (arg[c][1])
277 {
278 case 'i':
279 if (c + 1 < argc && strlen(arg[c + 1]) < MAX_PATH_LENGTH) // all ok
280 {
281 strncpy(input_path, arg[c++ + 1], MAX_PATH_LENGTH);
282 input_path[MAX_PATH_LENGTH - 1] = '\0';
283 }
284 else
285 Usage(arg[0]);
286 break;
287 case 'o':
288 if (c + 1 < argc && strlen(arg[c + 1]) < MAX_PATH_LENGTH) // all ok
289 {
290 strncpy(output_path, arg[c++ + 1], MAX_PATH_LENGTH);
291 output_path[MAX_PATH_LENGTH - 1] = '\0';
292 }
293 else
294 Usage(arg[0]);
295 break;
296 case 'f':
297 if (c + 1 < argc) // all ok
298 CONF_allow_float_to_int = atoi(arg[c++ + 1]) != 0;
299 else
300 Usage(arg[0]);
301 break;
302 case 'e':
303 if (c + 1 < argc) // all ok
304 {
305 CONF_extract = atoi(arg[c++ + 1]);
306 if (!(CONF_extract > 0 && CONF_extract < 8))
307 Usage(arg[0]);
308 }
309 else
310 Usage(arg[0]);
311 break;
312 case 'b':
313 if (c + 1 < argc) // all ok
314 CONF_TargetBuild = atoi(arg[c++ + 1]);
315 else
316 Usage(arg[0]);
317 break;
318 default:
319 break;
320 }
321 }
322}

324uint32_t ReadBuild(int locale)
325{
326 // include build info file also
327 std::string filename = std::string("component.wow-") + Locales[locale] + ".txt";
328 //printf("Read %s file... ", filename.c_str());
329
330 HANDLE dbcFile;
331 if (!SFileOpenFileEx(LocaleMpq, filename.c_str(), SFILE_OPEN_FROM_MPQ, &dbcFile))
332 {
333 printf("Fatal error: Not found %s file!\n", filename.c_str());
334 exit(1);
335 }
336
337 char buff[512];
338 DWORD readBytes = 0;
339 SFileReadFile(dbcFile, buff, 512, &readBytes, nullptr);
340 if (!readBytes)
341 {
342 printf("Fatal error: Not found %s file!\n", filename.c_str());
343 exit(1);
344 }
345
346 std::string text = buff;
347 SFileCloseFile(dbcFile);
348
349 size_t pos = text.find("version=\"");
350 size_t pos1 = pos + strlen("version=\"");
351 size_t pos2 = text.find("\"", pos1);
352 if (pos == text.npos || pos2 == text.npos || pos1 >= pos2)
353 {
354 printf("Fatal error: Invalid %s file format!\n", filename.c_str());
355 exit(1);
356 }
357
358 std::string build_str = text.substr(pos1, pos2 - pos1);
359
360 int build = atoi(build_str.c_str());
361 if (build <= 0)
362 {
363 printf("Fatal error: Invalid %s file format!\n", filename.c_str());
364 exit(1);
365 }
366
367 return build;
368}

605bool ConvertADT(std::string const& inputPath, std::string const& outputPath, int /*cell_y*/, int /*cell_x*/, uint32_t build, bool ignoreDeepWater)
606{
607 ChunkedFile adt;
608
609 if (!adt.loadFile(WorldMpq, inputPath))
610 return false;
611
612 // Prepare map header
613 map_fileheader map;
614 map.mapMagic = *reinterpret_cast<uint32_t const*>(MAP_MAGIC);
615 map.versionMagic = *(uint32_t const*)MAP_VERSION_MAGIC;
616 map.buildMagic = build;
617
618 // Get area flags data
619 memset(area_ids, 0, sizeof(area_ids));
620 memset(V9, 0, sizeof(V9));
621 memset(V8, 0, sizeof(V8));
622
623 memset(liquid_show, 0, sizeof(liquid_show));
624 memset(liquid_flags, 0, sizeof(liquid_flags));
625 memset(liquid_entry, 0, sizeof(liquid_entry));
626
627 memset(holes, 0, sizeof(holes));
628
629 bool hasHoles = false;
630 bool hasFlightBox = false;
631
632 for (std::multimap<std::string, FileChunk*>::const_iterator itr = adt.chunks.lower_bound("MCNK"); itr != adt.chunks.upper_bound("MCNK"); ++itr)
633 {
634 adt_MCNK* mcnk = itr->second->As<adt_MCNK>();
635
636 if (mcnk->iy > ADT_CELLS_PER_GRID - 1 || mcnk->ix > ADT_CELLS_PER_GRID - 1)
637 {
638 printf("\n ADT_CELLS_PER_GRID %u max %u\n", mcnk->iy, ADT_CELLS_PER_GRID);
639 continue;
640 }
641
642 // Area data
643 area_ids[mcnk->iy][mcnk->ix] = mcnk->areaid;
644
645 // Height
646 // Height values for triangles stored in order:
647 // 1 2 3 4 5 6 7 8 9
648 // 10 11 12 13 14 15 16 17
649 // 18 19 20 21 22 23 24 25 26
650 // 27 28 29 30 31 32 33 34
651 // . . . . . . . .
652 // For better get height values merge it to V9 and V8 map
653 // V9 height map:
654 // 1 2 3 4 5 6 7 8 9
655 // 18 19 20 21 22 23 24 25 26
656 // . . . . . . . .
657 // V8 height map:
658 // 10 11 12 13 14 15 16 17
659 // 27 28 29 30 31 32 33 34
660 // . . . . . . . .
661
662 // Set map height as grid height
663 for (int y = 0; y <= ADT_CELL_SIZE; y++)
664 {
665 int cy = mcnk->iy * ADT_CELL_SIZE + y;
666 for (int x = 0; x <= ADT_CELL_SIZE; x++)
667 {
668 int cx = mcnk->ix * ADT_CELL_SIZE + x;
669 V9[cy][cx] = mcnk->ypos;
670 }
671 }
672
673 for (int y = 0; y < ADT_CELL_SIZE; y++)
674 {
675 int cy = mcnk->iy * ADT_CELL_SIZE + y;
676 for (int x = 0; x < ADT_CELL_SIZE; x++)
677 {
678 int cx = mcnk->ix * ADT_CELL_SIZE + x;
679 V8[cy][cx] = mcnk->ypos;
680 }
681 }
682
683 // Get custom height
684 if (FileChunk* chunk = itr->second->GetSubChunk("MCVT"))
685 {
686 adt_MCVT* mcvt = chunk->As<adt_MCVT>();
687 // get V9 height map
688 for (int y = 0; y <= ADT_CELL_SIZE; y++)
689 {
690 int cy = mcnk->iy * ADT_CELL_SIZE + y;
691 for (int x = 0; x <= ADT_CELL_SIZE; x++)
692 {
693 int cx = mcnk->ix * ADT_CELL_SIZE + x;
694 V9[cy][cx] += mcvt->height_map[y * (ADT_CELL_SIZE * 2 + 1) + x];
695 }
696 }
697 // get V8 height map
698 for (int y = 0; y < ADT_CELL_SIZE; y++)
699 {
700 int cy = mcnk->iy * ADT_CELL_SIZE + y;
701 for (int x = 0; x < ADT_CELL_SIZE; x++)
702 {
703 int cx = mcnk->ix * ADT_CELL_SIZE + x;
704 V8[cy][cx] += mcvt->height_map[y * (ADT_CELL_SIZE * 2 + 1) + ADT_CELL_SIZE + 1 + x];
705 }
706 }
707 }
708
709 // Liquid data
710 if (mcnk->sizeMCLQ > 8)
711 {
712 if (FileChunk* chunk = itr->second->GetSubChunk("MCLQ"))
713 {
714 adt_MCLQ* liquid = chunk->As<adt_MCLQ>();
715 int count = 0;
716 for (int y = 0; y < ADT_CELL_SIZE; ++y)
717 {
718 int cy = mcnk->iy * ADT_CELL_SIZE + y;
719 for (int x = 0; x < ADT_CELL_SIZE; ++x)
720 {
721 int cx = mcnk->ix * ADT_CELL_SIZE + x;
722 if (liquid->flags[y][x] != 0x0F)
723 {
724 liquid_show[cy][cx] = true;
725 if (!ignoreDeepWater && liquid->flags[y][x] & (1 << 7))
726 liquid_flags[mcnk->iy][mcnk->ix] |= MAP_LIQUID_TYPE_DARK_WATER;
727 ++count;
728 }
729 }
730 }
731
732 uint32_t c_flag = mcnk->flags;
733 if (c_flag & (1 << 2))
734 {
735 liquid_entry[mcnk->iy][mcnk->ix] = 1;
736 liquid_flags[mcnk->iy][mcnk->ix] |= MAP_LIQUID_TYPE_WATER; // water
737 }
738 if (c_flag & (1 << 3))
739 {
740 liquid_entry[mcnk->iy][mcnk->ix] = 2;
741 liquid_flags[mcnk->iy][mcnk->ix] |= MAP_LIQUID_TYPE_OCEAN; // ocean
742 }
743 if (c_flag & (1 << 4))
744 {
745 liquid_entry[mcnk->iy][mcnk->ix] = 3;
746 liquid_flags[mcnk->iy][mcnk->ix] |= MAP_LIQUID_TYPE_MAGMA; // magma/slime
747 }
748
749 if (!count && liquid_flags[mcnk->iy][mcnk->ix])
750 fprintf(stderr, "Wrong liquid detect in MCLQ chunk");
751
752 for (int y = 0; y <= ADT_CELL_SIZE; ++y)
753 {
754 int cy = mcnk->iy * ADT_CELL_SIZE + y;
755 for (int x = 0; x <= ADT_CELL_SIZE; ++x)
756 {
757 int cx = mcnk->ix * ADT_CELL_SIZE + x;
758 liquid_height[cy][cx] = liquid->liquid[y][x].height;
759 }
760 }
761 }
762 }
763
764 // Hole data
765 holes[mcnk->iy][mcnk->ix] = mcnk->holes;
766 if (!hasHoles && mcnk->holes != 0)
767 hasHoles = true;
768 }
769
770 // Get liquid map for grid (in WOTLK used MH2O chunk)
771 if (FileChunk* chunk = adt.GetChunk("MH2O"))
772 {
773 adt_MH2O* h2o = chunk->As<adt_MH2O>();
774 for (int32_t i = 0; i < ADT_CELLS_PER_GRID; i++)
775 {
776 for (int32_t j = 0; j < ADT_CELLS_PER_GRID; j++)
777 {
778 adt_liquid_instance const* h = h2o->GetLiquidInstance(i, j);
779 if (!h)
780 continue;
781
782 adt_liquid_attributes attrs = h2o->GetLiquidAttributes(i, j);
783
784 int32_t count = 0;
785 uint64_t existsMask = h2o->GetLiquidExistsBitmap(h);
786 for (int32_t y = 0; y < h->GetHeight(); y++)
787 {
788 int32_t cy = i * ADT_CELL_SIZE + y + h->GetOffsetY();
789 for (int32_t x = 0; x < h->GetWidth(); x++)
790 {
791 int32_t cx = j * ADT_CELL_SIZE + x + h->GetOffsetX();
792 if (existsMask & 1)
793 {
794 liquid_show[cy][cx] = true;
795 ++count;
796 }
797 existsMask >>= 1;
798 }
799 }
800
801 liquid_entry[i][j] = h2o->GetLiquidType(h);
802 switch (LiquidTypes.at(liquid_entry[i][j]).SoundBank)
803 {
804 case LIQUID_TYPE_WATER: liquid_flags[i][j] |= MAP_LIQUID_TYPE_WATER; break;
805 case LIQUID_TYPE_OCEAN: liquid_flags[i][j] |= MAP_LIQUID_TYPE_OCEAN; if (!ignoreDeepWater && attrs.Deep) liquid_flags[i][j] |= MAP_LIQUID_TYPE_DARK_WATER; break;
806 case LIQUID_TYPE_MAGMA: liquid_flags[i][j] |= MAP_LIQUID_TYPE_MAGMA; break;
807 case LIQUID_TYPE_SLIME: liquid_flags[i][j] |= MAP_LIQUID_TYPE_SLIME; break;
808 default:
809 printf("\nCan't find Liquid type %u for map %s\nchunk %d,%d\n", h->LiquidType, inputPath.c_str(), i, j);
810 break;
811 }
812
813 if (!count && liquid_flags[i][j])
814 printf("Wrong liquid detect in MH2O chunk");
815
816 int32_t pos = 0;
817 for (int32_t y = 0; y <= h->GetHeight(); y++)
818 {
819 int32_t cy = i * ADT_CELL_SIZE + y + h->GetOffsetY();
820 for (int32_t x = 0; x <= h->GetWidth(); x++)
821 {
822 int32_t cx = j * ADT_CELL_SIZE + x + h->GetOffsetX();
823 liquid_height[cy][cx] = h2o->GetLiquidHeight(h, pos);
824 pos++;
825 }
826 }
827 }
828 }
829 }
830
831 if (FileChunk* chunk = adt.GetChunk("MFBO"))
832 {
833 adt_MFBO* mfbo = chunk->As<adt_MFBO>();
834 memcpy(flight_box_max, &mfbo->max, sizeof(flight_box_max));
835 memcpy(flight_box_min, &mfbo->min, sizeof(flight_box_min));
836 hasFlightBox = true;
837 }
838
839 //============================================
840 // Try pack area data
841 //============================================
842 bool fullAreaData = false;
843 uint32_t areaId = area_ids[0][0];
844 for (int y = 0; y < ADT_CELLS_PER_GRID; ++y)
845 {
846 for (int x = 0; x < ADT_CELLS_PER_GRID; ++x)
847 {
848 if (area_ids[y][x] != areaId)
849 {
850 fullAreaData = true;
851 break;
852 }
853 }
854 }
855
856 map.areaMapOffset = sizeof(map);
857 map.areaMapSize = sizeof(map_areaHeader);
858
859 map_areaHeader areaHeader;
860 areaHeader.fourcc = *reinterpret_cast<uint32_t const*>(MAP_AREA_MAGIC);
861 areaHeader.flags = 0;
862 if (fullAreaData)
863 {
864 areaHeader.gridArea = 0;
865 map.areaMapSize += sizeof(area_ids);
866 }
867 else
868 {
869 areaHeader.flags |= MAP_AREA_NO_AREA;
870 areaHeader.gridArea = static_cast<uint16_t>(areaId);
871 }
872
873
874 //============================================
875 // Try pack height data
876 //============================================
877 float maxHeight = -20000;
878 float minHeight = 20000;
879 for (int y = 0; y < ADT_GRID_SIZE; y++)
880 {
881 for (int x = 0; x < ADT_GRID_SIZE; x++)
882 {
883 float h = V8[y][x];
884 if (maxHeight < h) maxHeight = h;
885 if (minHeight > h) minHeight = h;
886 }
887 }
888 for (int y = 0; y <= ADT_GRID_SIZE; y++)
889 {
890 for (int x = 0; x <= ADT_GRID_SIZE; x++)
891 {
892 float h = V9[y][x];
893 if (maxHeight < h) maxHeight = h;
894 if (minHeight > h) minHeight = h;
895 }
896 }
897
898 // Check for allow limit minimum height (not store height in deep ochean - allow save some memory)
899 if (CONF_allow_height_limit && minHeight < CONF_use_minHeight)
900 {
901 for (int y = 0; y < ADT_GRID_SIZE; y++)
902 for (int x = 0; x < ADT_GRID_SIZE; x++)
903 if (V8[y][x] < CONF_use_minHeight)
904 V8[y][x] = CONF_use_minHeight;
905 for (int y = 0; y <= ADT_GRID_SIZE; y++)
906 for (int x = 0; x <= ADT_GRID_SIZE; x++)
907 if (V9[y][x] < CONF_use_minHeight)
908 V9[y][x] = CONF_use_minHeight;
909 if (minHeight < CONF_use_minHeight)
910 minHeight = CONF_use_minHeight;
911 if (maxHeight < CONF_use_minHeight)
912 maxHeight = CONF_use_minHeight;
913 }
914
915 map.heightMapOffset = map.areaMapOffset + map.areaMapSize;
916 map.heightMapSize = sizeof(map_heightHeader);
917
918 map_heightHeader heightHeader;
919 heightHeader.fourcc = *reinterpret_cast<uint32_t const*>(MAP_HEIGHT_MAGIC);
920 heightHeader.flags = 0;
921 heightHeader.gridHeight = minHeight;
922 heightHeader.gridMaxHeight = maxHeight;
923
924 if (maxHeight == minHeight)
925 heightHeader.flags |= MAP_HEIGHT_NO_HEIGHT;
926
927 // Not need store if flat surface
928 if (CONF_allow_float_to_int && (maxHeight - minHeight) < CONF_flat_height_delta_limit)
929 heightHeader.flags |= MAP_HEIGHT_NO_HEIGHT;
930
931 if (hasFlightBox)
932 {
933 heightHeader.flags |= MAP_HEIGHT_HAS_FLIGHT_BOUNDS;
934 map.heightMapSize += sizeof(flight_box_max) + sizeof(flight_box_min);
935 }
936
937 // Try store as packed in uint16_t or uint8_t values
938 if (!(heightHeader.flags & MAP_HEIGHT_NO_HEIGHT))
939 {
940 float step = 0;
941 // Try Store as uint values
942 if (CONF_allow_float_to_int)
943 {
944 float diff = maxHeight - minHeight;
945 if (diff < CONF_float_to_int8_limit) // As uint8_t (max accuracy = CONF_float_to_int8_limit/256)
946 {
947 heightHeader.flags |= MAP_HEIGHT_AS_INT8;
948 step = selectUInt8StepStore(diff);
949 }
950 else if (diff < CONF_float_to_int16_limit) // As uint16_t (max accuracy = CONF_float_to_int16_limit/65536)
951 {
952 heightHeader.flags |= MAP_HEIGHT_AS_INT16;
953 step = selectUInt16StepStore(diff);
954 }
955 }
956
957 // Pack it to int values if need
958 if (heightHeader.flags & MAP_HEIGHT_AS_INT8)
959 {
960 for (int y = 0; y < ADT_GRID_SIZE; y++)
961 for (int x = 0; x < ADT_GRID_SIZE; x++)
962 uint8_V8[y][x] = uint8_t((V8[y][x] - minHeight) * step + 0.5f);
963 for (int y = 0; y <= ADT_GRID_SIZE; y++)
964 for (int x = 0; x <= ADT_GRID_SIZE; x++)
965 uint8_V9[y][x] = uint8_t((V9[y][x] - minHeight) * step + 0.5f);
966 map.heightMapSize += sizeof(uint8_V9) + sizeof(uint8_V8);
967 }
968 else if (heightHeader.flags & MAP_HEIGHT_AS_INT16)
969 {
970 for (int y = 0; y < ADT_GRID_SIZE; y++)
971 for (int x = 0; x < ADT_GRID_SIZE; x++)
972 uint16_V8[y][x] = uint16_t((V8[y][x] - minHeight) * step + 0.5f);
973 for (int y = 0; y <= ADT_GRID_SIZE; y++)
974 for (int x = 0; x <= ADT_GRID_SIZE; x++)
975 uint16_V9[y][x] = uint16_t((V9[y][x] - minHeight) * step + 0.5f);
976 map.heightMapSize += sizeof(uint16_V9) + sizeof(uint16_V8);
977 }
978 else
979 map.heightMapSize += sizeof(V9) + sizeof(V8);
980 }
981
982 //============================================
983 // Pack liquid data
984 //============================================
985 uint16_t firstLiquidType = liquid_entry[0][0];
986 uint8_t firstLiquidFlag = liquid_flags[0][0];
987 bool fullType = false;
988 for (int y = 0; y < ADT_CELLS_PER_GRID; y++)
989 {
990 for (int x = 0; x < ADT_CELLS_PER_GRID; x++)
991 {
992 if (liquid_entry[y][x] != firstLiquidType || liquid_flags[y][x] != firstLiquidFlag)
993 {
994 fullType = true;
995 y = ADT_CELLS_PER_GRID;
996 break;
997 }
998 }
999 }
1000
1001 map_liquidHeader liquidHeader;
1002
1003 // no water data (if all grid have 0 liquid type)
1004 if (firstLiquidFlag == 0 && !fullType)
1005 {
1006 // No liquid data
1007 map.liquidMapOffset = 0;
1008 map.liquidMapSize = 0;
1009 }
1010 else
1011 {
1012 int minX = 255, minY = 255;
1013 int maxX = 0, maxY = 0;
1014 maxHeight = -20000;
1015 minHeight = 20000;
1016 for (int y = 0; y < ADT_GRID_SIZE; y++)
1017 {
1018 for (int x = 0; x < ADT_GRID_SIZE; x++)
1019 {
1020 if (liquid_show[y][x])
1021 {
1022 if (minX > x) minX = x;
1023 if (maxX < x) maxX = x;
1024 if (minY > y) minY = y;
1025 if (maxY < y) maxY = y;
1026 float h = liquid_height[y][x];
1027 if (maxHeight < h) maxHeight = h;
1028 if (minHeight > h) minHeight = h;
1029 }
1030 else
1031 {
1032 liquid_height[y][x] = CONF_use_minHeight;
1033 if (minHeight > CONF_use_minHeight) minHeight = CONF_use_minHeight;
1034 }
1035 }
1036 }
1037 map.liquidMapOffset = map.heightMapOffset + map.heightMapSize;
1038 map.liquidMapSize = sizeof(map_liquidHeader);
1039 liquidHeader.fourcc = *reinterpret_cast<uint32_t const*>(MAP_LIQUID_MAGIC);
1040 liquidHeader.flags = 0;
1041 liquidHeader.liquidType = 0;
1042 liquidHeader.offsetX = minX;
1043 liquidHeader.offsetY = minY;
1044 liquidHeader.width = maxX - minX + 1 + 1;
1045 liquidHeader.height = maxY - minY + 1 + 1;
1046 liquidHeader.liquidLevel = minHeight;
1047
1048 if (maxHeight == minHeight)
1049 liquidHeader.flags |= MAP_LIQUID_NO_HEIGHT;
1050
1051 // Not need store if flat surface
1052 if (CONF_allow_float_to_int && (maxHeight - minHeight) < CONF_flat_liquid_delta_limit)
1053 liquidHeader.flags |= MAP_LIQUID_NO_HEIGHT;
1054
1055 if (!fullType)
1056 liquidHeader.flags |= MAP_LIQUID_NO_TYPE;
1057
1058 if (liquidHeader.flags & MAP_LIQUID_NO_TYPE)
1059 {
1060 liquidHeader.liquidFlags = firstLiquidFlag;
1061 liquidHeader.liquidType = firstLiquidType;
1062 }
1063 else
1064 map.liquidMapSize += sizeof(liquid_entry) + sizeof(liquid_flags);
1065
1066 if (!(liquidHeader.flags & MAP_LIQUID_NO_HEIGHT))
1067 map.liquidMapSize += sizeof(float) * liquidHeader.width * liquidHeader.height;
1068 }
1069
1070 if (hasHoles)
1071 {
1072 if (map.liquidMapOffset)
1073 map.holesOffset = map.liquidMapOffset + map.liquidMapSize;
1074 else
1075 map.holesOffset = map.heightMapOffset + map.heightMapSize;
1076
1077 map.holesSize = sizeof(holes);
1078 }
1079 else
1080 {
1081 map.holesOffset = 0;
1082 map.holesSize = 0;
1083 }
1084
1085 // Ok all data prepared - store it
1086 std::ofstream outFile(outputPath, std::ofstream::out | std::ofstream::binary);
1087 if (!outFile)
1088 {
1089 printf("Can't create the output file '%s'\n", outputPath.c_str());
1090 return false;
1091 }
1092
1093 outFile.write(reinterpret_cast<const char*>(&map), sizeof(map));
1094 // Store area data
1095 outFile.write(reinterpret_cast<const char*>(&areaHeader), sizeof(areaHeader));
1096 if (!(areaHeader.flags & MAP_AREA_NO_AREA))
1097 outFile.write(reinterpret_cast<const char*>(area_ids), sizeof(area_ids));
1098
1099 // Store height data
1100 outFile.write(reinterpret_cast<const char*>(&heightHeader), sizeof(heightHeader));
1101 if (!(heightHeader.flags & MAP_HEIGHT_NO_HEIGHT))
1102 {
1103 if (heightHeader.flags & MAP_HEIGHT_AS_INT16)
1104 {
1105 outFile.write(reinterpret_cast<const char*>(uint16_V9), sizeof(uint16_V9));
1106 outFile.write(reinterpret_cast<const char*>(uint16_V8), sizeof(uint16_V8));
1107 }
1108 else if (heightHeader.flags & MAP_HEIGHT_AS_INT8)
1109 {
1110 outFile.write(reinterpret_cast<const char*>(uint8_V9), sizeof(uint8_V9));
1111 outFile.write(reinterpret_cast<const char*>(uint8_V8), sizeof(uint8_V8));
1112 }
1113 else
1114 {
1115 outFile.write(reinterpret_cast<const char*>(V9), sizeof(V9));
1116 outFile.write(reinterpret_cast<const char*>(V8), sizeof(V8));
1117 }
1118 }
1119
1120 if (heightHeader.flags & MAP_HEIGHT_HAS_FLIGHT_BOUNDS)
1121 {
1122 outFile.write(reinterpret_cast<char*>(flight_box_max), sizeof(flight_box_max));
1123 outFile.write(reinterpret_cast<char*>(flight_box_min), sizeof(flight_box_min));
1124 }
1125
1126 // Store liquid data if need
1127 if (map.liquidMapOffset)
1128 {
1129 outFile.write(reinterpret_cast<const char*>(&liquidHeader), sizeof(liquidHeader));
1130 if (!(liquidHeader.flags & MAP_LIQUID_NO_TYPE))
1131 {
1132 outFile.write(reinterpret_cast<const char*>(liquid_entry), sizeof(liquid_entry));
1133 outFile.write(reinterpret_cast<const char*>(liquid_flags), sizeof(liquid_flags));
1134 }
1135
1136 if (!(liquidHeader.flags & MAP_LIQUID_NO_HEIGHT))
1137 {
1138 for (int y = 0; y < liquidHeader.height; y++)
1139 outFile.write(reinterpret_cast<const char*>(&liquid_height[y + liquidHeader.offsetY][liquidHeader.offsetX]), sizeof(float) * liquidHeader.width);
1140 }
1141 }
1142
1143 // store hole data
1144 if (hasHoles)
1145 outFile.write(reinterpret_cast<const char*>(holes), map.holesSize);
1146
1147 outFile.close();
1148
1149 return true;
1150}
