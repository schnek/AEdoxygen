200 if (liquid[x][y].OffsetAttributes)
201 return *((adt_liquid_attributes*)((uint8_t*)this + 8 + liquid[x][y].OffsetAttributes));
202 return { 0xFFFFFFFFFFFFFFFF, 0xFFFFFFFFFFFFFFFF };
203 }

207 uint16_t GetLiquidType(adt_liquid_instance const* h) const
208 {
209 if (GetLiquidVertexFormat(h) == LiquidVertexFormatType::Depth)
210 return 2;
211
212 return h->LiquidType;
213 }
214
215 float GetLiquidHeight(adt_liquid_instance const* h, int32_t pos) const
216 {
217 if (!h->OffsetVertexData)
218 return 0.0f;
219 if (GetLiquidVertexFormat(h) == LiquidVertexFormatType::Depth)
220 return 0.0f;
221
222 switch (GetLiquidVertexFormat(h))
223 {
224 case LiquidVertexFormatType::HeightDepth:
225 case LiquidVertexFormatType::HeightTextureCoord:
226 case LiquidVertexFormatType::HeightDepthTextureCoord:
227 return ((float const*)((uint8_t*)this + 8 + h->OffsetVertexData))[pos];
228 case LiquidVertexFormatType::Depth:
229 return 0.0f;
230 case LiquidVertexFormatType::Unk4:
231 case LiquidVertexFormatType::Unk5:
232 return ((float const*)((uint8_t*)this + 8 + h->OffsetVertexData + 4))[pos * 2];
233 default:
234 break;
235 }
236
237 return 0.0f;
238 }
239
240 int8_t GetLiquidDepth(adt_liquid_instance const* h, int32_t pos) const
241 {
242 if (!h->OffsetVertexData)
243 return -1;
244
245 switch (GetLiquidVertexFormat(h))
246 {
247 case LiquidVertexFormatType::HeightDepth:
248 return ((int8_t const*)((int8_t const*)this + 8 + h->OffsetVertexData + (h->GetWidth() + 1) * (h->GetHeight() + 1) * 4))[pos];
249 case LiquidVertexFormatType::HeightTextureCoord:
250 return 0;
251 case LiquidVertexFormatType::Depth:
252 return ((int8_t const*)((uint8_t*)this + 8 + h->OffsetVertexData))[pos];
253 case LiquidVertexFormatType::HeightDepthTextureCoord:
254 return ((int8_t const*)((int8_t const*)this + 8 + h->OffsetVertexData + (h->GetWidth() + 1) * (h->GetHeight() + 1) * 8))[pos];
255 case LiquidVertexFormatType::Unk4:
256 return ((int8_t const*)((uint8_t*)this + 8 + h->OffsetVertexData))[pos * 8];
257 case LiquidVertexFormatType::Unk5:
258 return 0;
259 default:
260 break;
261 }
262 return 0;
263 }
264
265 uint16_t const* GetLiquidTextureCoordMap(adt_liquid_instance const* h, int32_t pos) const
266 {
267 if (!h->OffsetVertexData)
268 return nullptr;
269
270 switch (GetLiquidVertexFormat(h))
271 {
272 case LiquidVertexFormatType::HeightDepth:
273 case LiquidVertexFormatType::Depth:
274 case LiquidVertexFormatType::Unk4:
275 return nullptr;
276 case LiquidVertexFormatType::HeightTextureCoord:
277 case LiquidVertexFormatType::HeightDepthTextureCoord:
278 return (uint16_t const*)((uint8_t const*)this + 8 + h->OffsetVertexData + 4 * ((h->GetWidth() + 1) * (h->GetHeight() + 1) + pos));
279 case LiquidVertexFormatType::Unk5:
280 return (uint16_t const*)((uint8_t const*)this + 8 + h->OffsetVertexData + 8 * ((h->GetWidth() + 1) * (h->GetHeight() + 1) + pos));
281 default:
282 break;
283 }
284 return nullptr;
285 }

287 uint64_t GetLiquidExistsBitmap(adt_liquid_instance const* h) const

289 if (h->OffsetExistsBitmap)
290 return *((uint64_t*)((uint8_t*)this + 8 + h->OffsetExistsBitmap));
291 else
292 return 0xFFFFFFFFFFFFFFFFuLL;
293 }
