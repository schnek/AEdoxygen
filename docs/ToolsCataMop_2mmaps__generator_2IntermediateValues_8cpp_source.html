42
43 std::string name("meshes/%04u%02i%02i.");

45#define DEBUG_WRITE(fileExtension,data) \
46 do { \
47 sprintf(fileName, (name + fileExtension).c_str(), mapID, tileY, tileX); \
48 FILE* file = fopen(fileName, "wb"); \
49 if (!file) \
50 { \
51 char message[1024]; \

58 printf("%sWriting debug output... \r", tileString); \

75 void IntermediateValues::debugWrite(FILE* file, const rcHeightfield* mesh)
76 {
77 if (!file || !mesh)
78 return;
79
80 fwrite(&(mesh->cs), sizeof(float), 1, file);
81 fwrite(&(mesh->ch), sizeof(float), 1, file);
82 fwrite(&(mesh->width), sizeof(int), 1, file);
83 fwrite(&(mesh->height), sizeof(int), 1, file);
84 fwrite(mesh->bmin, sizeof(float), 3, file);
85 fwrite(mesh->bmax, sizeof(float), 3, file);
86
87 for (int y = 0; y < mesh->height; ++y)
88 for (int x = 0; x < mesh->width; ++x)
89 {
90 rcSpan* span = mesh->spans[x+y*mesh->width];
91
92 // first, count the number of spans
93 int spanCount = 0;
94 while (span)
95 {
96 spanCount++;
97 span = span->next;
98 }
99
100 // write the span count
101 fwrite(&spanCount, sizeof(int), 1, file);
102
103 // write the spans
104 span = mesh->spans[x+y*mesh->width];
105 while (span)
106 {
107 fwrite(span, sizeof(rcSpan), 1, file);
108 span = span->next;
109 }
110 }
111 }

127
128 fwrite(chf->bmin, sizeof(chf->bmin), 1, file);
129 fwrite(chf->bmax, sizeof(chf->bmax), 1, file);
130
131 fwrite(&(chf->cs), sizeof(chf->cs), 1, file);
132 fwrite(&(chf->ch), sizeof(chf->ch), 1, file);
133
134 int tmp = 0;
135 if (chf->cells) tmp |= 1;
136 if (chf->spans) tmp |= 2;
137 if (chf->dist) tmp |= 4;
138 if (chf->areas) tmp |= 8;
139
140 fwrite(&tmp, sizeof(tmp), 1, file);
141
142 if (chf->cells)
143 fwrite(chf->cells, sizeof(rcCompactCell), chf->width*chf->height, file);
144 if (chf->spans)
145 fwrite(chf->spans, sizeof(rcCompactSpan), chf->spanCount, file);
146 if (chf->dist)
147 fwrite(chf->dist, sizeof(unsigned short), chf->spanCount, file);
148 if (chf->areas)
149 fwrite(chf->areas, sizeof(unsigned char), chf->spanCount, file);

205 void IntermediateValues::generateObjFile(uint32_t mapID, uint32_t tileX, uint32_t tileY, MeshData &meshData)
206 {
207 char objFileName[255];
208 sprintf(objFileName, "meshes/map%04u%02u%02u.obj", mapID, tileY, tileX);
209
210 FILE* objFile = fopen(objFileName, "wb");
211 if (!objFile)
212 {
213 char message[1024];
214 sprintf(message, "Failed to open %s for writing!\n", objFileName);
215 perror(message);
216 return;
217 }
218
219 G3D::Array<float> allVerts;
220 G3D::Array<int> allTris;
221
222 allTris.append(meshData.liquidTris);
223 allVerts.append(meshData.liquidVerts);
224 TerrainBuilder::copyIndices(meshData.solidTris, allTris, allVerts.size() / 3);
225 allVerts.append(meshData.solidVerts);
226
227 float* verts = allVerts.getCArray();
228 int vertCount = allVerts.size() / 3;
229 int* tris = allTris.getCArray();
230 int triCount = allTris.size() / 3;
231
232 for (int i = 0; i < allVerts.size() / 3; i++)
233 fprintf(objFile, "v %f %f %f\n", verts[i*3], verts[i*3 + 1], verts[i*3 + 2]);
234
235 for (int i = 0; i < allTris.size() / 3; i++)
236 fprintf(objFile, "f %i %i %i\n", tris[i*3] + 1, tris[i*3 + 1] + 1, tris[i*3 + 2] + 1);
237
238 fclose(objFile);
239
240
241 char tileString[25];
242 sprintf(tileString, "[%02u,%02u]: ", tileY, tileX);
243 printf("%sWriting debug output... \r", tileString);
244
245 sprintf(objFileName, "meshes/%04u.map", mapID);
246
247 objFile = fopen(objFileName, "wb");
248 if (!objFile)
249 {
250 char message[1024];
251 sprintf(message, "Failed to open %s for writing!\n", objFileName);
252 perror(message);
253 return;
254 }
255
256 char b = '\0';
257 fwrite(&b, sizeof(char), 1, objFile);
258 fclose(objFile);
259
260 sprintf(objFileName, "meshes/%04u%02u%02u.mesh", mapID, tileY, tileX);
261 objFile = fopen(objFileName, "wb");
262 if (!objFile)
263 {
264 char message[1024];
265 sprintf(message, "Failed to open %s for writing!\n", objFileName);
266 perror(message);
267 return;
268 }
269
270 fwrite(&vertCount, sizeof(int), 1, objFile);
271 fwrite(verts, sizeof(float), vertCount*3, objFile);
272 fflush(objFile);
273
274 fwrite(&triCount, sizeof(int), 1, objFile);
275 fwrite(tris, sizeof(int), triCount*3, objFile);
276 fflush(objFile);
277
278 fclose(objFile);
279 }
