48void FixNameCase(char* name, size_t len)
49{
50 char* ptr = name + len - 1;
51
52 //extension in lowercase
53 for (; *ptr != '.'; --ptr)
54 *ptr |= 0x20;
55
56 for (; ptr >= name; --ptr)
57 {
58 if (ptr > name && *ptr >= 'A' && *ptr <= 'Z' && isalpha(*(ptr - 1)))
59 *ptr |= 0x20;
60 else if ((ptr == name || !isalpha(*(ptr - 1))) && *ptr >= 'a' && *ptr <= 'z')
61 *ptr &= ~0x20;
62 }
63}

65void FixNameSpaces(char* name, size_t len)
66{
67 for (size_t i = 0; i<len - 3; i++)
68 {
69 if (name[i] == ' ')
70 name[i] = '_';
71 }
72}
