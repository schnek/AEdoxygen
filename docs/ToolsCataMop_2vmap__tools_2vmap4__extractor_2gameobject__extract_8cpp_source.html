28bool ExtractSingleModel(std::string& fname)
29{
30 if (fname.substr(fname.length() - 4, 4) == ".mdx")
31 {
32 fname.erase(fname.length() - 2, 2);
33 fname.append("2");
34 }
35
36 std::string originalName = fname;
37
38 char* name = GetPlainName((char*)fname.c_str());
39 FixNameCase(name, strlen(name));
40 FixNameSpaces(name, strlen(name));
41
42 std::string output(szWorkDirWmo);
43 output += "/";
44 output += name;
45
46 if (FileExists(output.c_str()))
47 return true;
48
49 Model mdl(originalName);
50 if (!mdl.open())
51 return false;
52
53 return mdl.ConvertToVMAPModel(output.c_str());
54}

58void ExtractGameobjectModels()
59{
60 printf("Extracting GameObject models...");
61 DBCFile dbc(LocaleMpq, "DBFilesClient\\GameObjectDisplayInfo.dbc");
62 if (!dbc.open())
63 {
64 printf("Fatal error: Invalid GameObjectDisplayInfo.dbc file format!\n");
65 exit(1);
66 }
67
68 std::string basepath = szWorkDirWmo;
69 basepath += "/";
70 std::string path;
71
72 std::string modelListPath = basepath + "temp_gameobject_models";
73 FILE* model_list = fopen(modelListPath.c_str(), "wb");
74 if (!model_list)
75 {
76 printf("Fatal error: Could not open file %s\n", modelListPath.c_str());
77 return;
78 }
79
80 for (DBCFile::Iterator it = dbc.begin(); it != dbc.end(); ++it)
81 {
82 path = it->getString(1);
83
84 if (path.length() < 4)
85 continue;
86
87 FixNameCase((char*)path.c_str(), path.size());
88 char * name = GetPlainName((char*)path.c_str());
89 FixNameSpaces(name, strlen(name));
90
91 char * ch_ext = GetExtension(name);
92 if (!ch_ext)
93 continue;
94
95 strToLower(ch_ext);
96
97 bool result = false;
98 uint8_t isWmo = 0;
99 if (!strcmp(ch_ext, ".wmo"))
100 {
101 isWmo = 1;
102 result = ExtractSingleWmo(path);
103 }
104 else if (!strcmp(ch_ext, ".mdl")) // TODO: extract .mdl files, if needed
105 continue;
106 else //if (!strcmp(ch_ext, ".mdx") || !strcmp(ch_ext, ".m2"))
107 result = ExtractSingleModel(path);
108
109 if (result)
110 {
111 uint32_t displayId = it->getUInt(0);
112 uint32_t path_length = static_cast<uint32_t>(strlen(name));
113 fwrite(&displayId, sizeof(uint32_t), 1, model_list);
114 fwrite(&isWmo, sizeof(uint8_t), 1, model_list);
115 fwrite(&path_length, sizeof(uint32_t), 1, model_list);
116 fwrite(name, sizeof(char), path_length, model_list);
117 }
118 }
119
120 fclose(model_list);
121
122 printf("Done!\n");
123}
