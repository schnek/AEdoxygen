147bool LoadLocaleMPQFile(int locale)
148{
149 TCHAR buff[512];
150 memset(buff, 0, sizeof(buff));
151 _stprintf(buff, _T("%s%s/locale-%s.MPQ"), input_path, LocalesT[locale], LocalesT[locale]);
152 if (!SFileOpenArchive(buff, 0, MPQ_OPEN_READ_ONLY, &LocaleMpq))
153 {
154 if (GetLastError() != ERROR_PATH_NOT_FOUND)
155 {
156 _tprintf(_T("Loading %s locale MPQs\n"), LocalesT[locale]);
157 _tprintf(_T("Cannot open archive %s\n"), buff);
158 }
159 return false;
160 }
161
162 _tprintf(_T("Loading %s locale MPQs\n"), LocalesT[locale]);
163 char const* prefix = NULL;
164 for (int i = 0; Builds[i] && Builds[i] <= CONF_TargetBuild; ++i)
165 {
166 // Do not attempt to read older MPQ patch archives past this build, they were merged with base
167 // and trying to read them together with new base will not end well
168 if (CONF_TargetBuild >= NEW_BASE_SET_BUILD && Builds[i] < NEW_BASE_SET_BUILD)
169 continue;
170
171 memset(buff, 0, sizeof(buff));
172 if (Builds[i] > LAST_DBC_IN_DATA_BUILD)
173 {
174 prefix = "";
175 _stprintf(buff, _T("%s%s/wow-update-%s-%u.MPQ"), input_path, LocalesT[locale], LocalesT[locale], Builds[i]);
176 }
177 else
178 {
179 prefix = Locales[locale];
180 _stprintf(buff, _T("%swow-update-%u.MPQ"), input_path, Builds[i]);
181 }
182
183 if (!SFileOpenPatchArchive(LocaleMpq, buff, prefix, 0))
184 {
185 if (GetLastError() != ERROR_FILE_NOT_FOUND)
186 _tprintf(_T("Cannot open patch archive %s\n"), buff);
187 continue;
188 }
189 }
190
191 printf("\n");
192 return true;
193}

376bool ExtractSingleWmo(std::string& fname)
377{
378 // Copy files from archive
379
380 char szLocalFile[1024];
381 const char * plain_name = GetPlainName(fname.c_str());
382 sprintf(szLocalFile, "%s/%s", szWorkDirWmo, plain_name);
383 FixNameCase(szLocalFile,strlen(szLocalFile));
384
385 if (FileExists(szLocalFile))
386 return true;
387
388 int p = 0;
389 // Select root wmo files
390 char const* rchr = strrchr(plain_name, '_');
391 if (rchr != NULL)
392 {
393 char cpy[4];
394 memcpy(cpy, rchr, 4);
395 for (int i = 0; i < 4; ++i)
396 {
397 int m = cpy[i];
398 if (isdigit(m))
399 p++;
400 }
401 }
402
403 if (p == 3)
404 return true;
405
406 bool file_ok = true;
407 std::cout << "Extracting " << fname << std::endl;
408 WMORoot froot(fname);
409 if(!froot.open())
410 {
411 printf("Couldn't open RootWmo!!!\n");
412 return true;
413 }
414 FILE *output = fopen(szLocalFile,"wb");
415 if(!output)
416 {
417 printf("couldn't open %s for writing!\n", szLocalFile);
418 return false;
419 }
420 froot.ConvertToVMAPRootWmo(output);
421 int Wmo_nVertices = 0;
422 //printf("root has %d groups\n", froot->nGroups);
423 if (froot.nGroups !=0)
424 {
425 for (uint32_t i = 0; i < froot.nGroups; ++i)
426 {
427 char temp[1024];
428 strcpy(temp, fname.c_str());
429 temp[fname.length()-4] = 0;
430 char groupFileName[1024];
431 sprintf(groupFileName, "%s_%03u.wmo", temp, i);
432 //printf("Trying to open groupfile %s\n",groupFileName);
433
434 std::string s = groupFileName;
435 WMOGroup fgroup(s);
436 if(!fgroup.open())
437 {
438 printf("Could not open all Group file for: %s\n", plain_name);
439 file_ok = false;
440 break;
441 }
442
443 Wmo_nVertices += fgroup.ConvertToVMAPGroupWmo(output, &froot, preciseVectorData);
444 }
445 }
446
447 fseek(output, 8, SEEK_SET); // store the correct no of vertices
448 fwrite(&Wmo_nVertices,sizeof(int),1,output);
449 fclose(output);
450
451 // Delete the extracted file in the case of an error
452 if (!file_ok)
453 remove(szLocalFile);
454 return true;
455}

498bool processArgv(int argc, char ** argv, const char *versionString)
499{
500 bool result = true;
501 bool hasInputPathParam = false;
502 preciseVectorData = false;
503
504 for(int i = 1; i < argc; ++i)
505 {
506 if(strcmp("-s",argv[i]) == 0)
507 {
508 preciseVectorData = false;
509 }
510 else if(strcmp("-d",argv[i]) == 0)
511 {
512 if((i+1)<argc)
513 {
514 hasInputPathParam = true;
515 strcpy(input_path, argv[i+1]);
516 if (input_path[strlen(input_path) - 1] != '\\' && input_path[strlen(input_path) - 1] != '/')
517 strcat(input_path, "/");
518 ++i;
519 }
520 else
521 {
522 result = false;
523 }
524 }
525 else if(strcmp("-?",argv[1]) == 0)
526 {
527 result = false;
528 }
529 else if(strcmp("-l",argv[i]) == 0)
530 {
531 preciseVectorData = true;
532 }
533 else if(strcmp("-b",argv[i]) == 0)
534 {
535 if (i + 1 < argc) // all ok
536 CONF_TargetBuild = atoi(argv[i++ + 1]);
537 }
538 else
539 {
540 result = false;
541 break;
542 }
543 }
544
545 if(!result)
546 {
547 printf("Extract %s.\n",versionString);
548 printf("%s [-?][-s][-l][-d <path>]\n", argv[0]);
549 printf(" -s : (default) small size (data size optimization), ~500MB less vmap data.\n");
550 printf(" -l : large size, ~500MB more vmap data. (might contain more details)\n");
551 printf(" -d <path>: Path to the vector data source folder.\n");
552 printf(" -b : target build (default %u)\n", CONF_TargetBuild);
553 printf(" -? : This message.\n");
554 }
555
556 if(!hasInputPathParam)
557 getGamePath();
558
559 return result;
560}
