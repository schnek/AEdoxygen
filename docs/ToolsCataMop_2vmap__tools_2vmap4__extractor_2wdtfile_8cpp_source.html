41bool WDTFile::init(char* /*map_id*/, unsigned int mapID)
42{
43 if (WDT.isEof())
44 {
45 //printf("Can't find WDT file.\n");
46 return false;
47 }
48
49 char fourcc[5];
50 uint32_t size;
51
52 std::string dirname = std::string(szWorkDirWmo) + "/dir_bin";
53 FILE *dirfile;
54 dirfile = fopen(dirname.c_str(), "ab");
55 if (!dirfile)
56 {
57 printf("Can't open dirfile!'%s'\n", dirname.c_str());
58 return false;
59 }
60
61 while (!WDT.isEof())
62 {
63 WDT.read(fourcc, 4);
64 WDT.read(&size, 4);
65
66 flipcc(fourcc);
67 fourcc[4] = 0;
68
69 size_t nextpos = WDT.getPos() + size;
70
71 if (!strcmp(fourcc, "MAIN"))
72 {
73 }
74 if (!strcmp(fourcc, "MWMO"))
75 {
76 // global map objects
77 if (size)
78 {
79 char *buf = new char[size];
80 WDT.read(buf, size);
81 char *p = buf;
82 int q = 0;
83 gWmoInstansName = new string[size];
84 while (p < buf + size)
85 {
86 char* s = wdtGetPlainName(p);
87 FixNameCase(s, strlen(s));
88 p = p + strlen(p) + 1;
89 gWmoInstansName[q++] = s;
90 }
91 delete[] buf;
92 }
93 }
94 else if (!strcmp(fourcc, "MODF"))
95 {
96 // global wmo instance data
97 if (size)
98 {
99 gnWMO = (int)size / 64;
100
101 for (int i = 0; i < gnWMO; ++i)
102 {
103 int id;
104 WDT.read(&id, 4);
105 WMOInstance inst(WDT, gWmoInstansName[id].c_str(), mapID, 65, 65, dirfile);
106 }
107
108 delete[] gWmoInstansName;
109 }
110 }
111 WDT.seek((int)nextpos);
112 }
113
114 WDT.close();
115 fclose(dirfile);
116 return true;
117}
