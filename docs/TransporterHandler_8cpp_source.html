31void TransportHandler::loadTransportTemplates()
32{
33 sLogger.debugFlag(AscEmu::Logging::LF_MAP, "TransportHandler : Start Loading TransportTemplates...");
34
35 uint32_t createCount = 0;
36
37 for (auto& it : sMySQLStore._transportEntryStore)
38 {
39 uint32_t entry = it.first;
40
41 GameObjectProperties const* gameobject_info = sMySQLStore.getGameObjectProperties(entry);
42 if (gameobject_info == nullptr)
43 {
44 sLogger.failure("Transport {} has no associated GameObjectProperties from `gameobject_properities` , skipped.", entry);
45 continue;
46 }
47
48 // paths are generated per template, saves us from generating it again in case of instanced transports
49 TransportTemplate& transport = _transportTemplates[entry];
50 transport.entry = entry;
51 generatePath(gameobject_info, &transport);
52
53 // transports in instance are only on one map
54 if (transport.inInstance)
55 _instanceTransports[*transport.mapsUsed.begin()].insert(entry);
56
57 ++createCount;
58 }
59
60 sLogger.debugFlag(AscEmu::Logging::LF_MAP, "Transporter Handler : Loaded {} transport templates", createCount);
61}

84Transporter* TransportHandler::createTransport(uint32_t entry, WorldMap* map /*= nullptr*/)
85{
86 TransportTemplate const* tInfo = getTransportTemplate(entry);
87 if (!tInfo)
88 {
89 sLogger.failure("Transport {} will not be loaded, `transport_template` missing", entry);
90 return nullptr;
91 }
92
93 // create transport...
94 Transporter* trans = new Transporter((uint64_t)HIGHGUID_TYPE_TRANSPORTER << 32 | entry);
95
96 // ...at first waypoint
97#if VERSION_STRING == Classic
98 if (tInfo->keyFrames.size() == 0)
99 return nullptr;
100
101#endif
102 PathNode startNode = tInfo->keyFrames.begin()->Node;
103 uint32_t mapId = startNode.mapid;
104 float x = startNode.x;
105 float y = startNode.y;
106 float z = startNode.z;
107 float o = tInfo->keyFrames.begin()->InitialOrientation;
108
109 if (!trans->Create(entry, mapId, x, y, z, o, 255))
110 {
111 delete trans;
112 return nullptr;
113 }
114
115 // Storage
116 _Transports.insert(trans);
117 if (map)
118 _TransportersByInstanceIdMap[map->getInstanceId()].insert(trans);
119 else
120 _TransportersByInstanceIdMap[trans->GetInstanceID()].insert(trans);
121
122 addTransport(trans);
123
124 if (const auto mapEntry = sMapStore.lookupEntry(mapId))
125 {
126 if (mapEntry->isInstanceableMap() != tInfo->inInstance)
127 {
128 sLogger.failure("Transport {} attempted creation in instance map (id: {}) but it is not an instanced transport!", entry, mapId);
129 delete trans;
130 return nullptr;
131 }
132 }
133
134 // Instance Case
135 if (map)
136 {
137 trans->AddToWorld(map);
138 trans->SetInstanceID(map->getInstanceId());
139 }
140 else
141 trans->AddToWorld();
142
143 trans->getWorldMap()->addToMapMgr(trans);
144
145 // Load Creatures
146 trans->LoadStaticPassengers();
147
148 return trans;
149}

173bool FillTransporterPathVector(uint32_t PathID, TransportPath & Path)
174{
175 // Store dbc values into current Path array
176 Path.resize(sTaxiPathNodeStore.getNumRows());
177
178 uint32_t i = 0;
179 for (uint32_t j = 0; j < sTaxiPathNodeStore.getNumRows(); ++j)
180 {
181 auto pathnode = sTaxiPathNodeStore.lookupEntry(j);
182 if (pathnode == nullptr)
183 continue;
184
185 if (pathnode->pathId == PathID)
186 {
187 Path[i].mapid = pathnode->mapid;
188 Path[i].x = pathnode->x;
189 Path[i].y = pathnode->y;
190 Path[i].z = pathnode->z;
191 Path[i].flags = pathnode->flags;
192 Path[i].delay = pathnode->waittime;
193#if VERSION_STRING > Classic
194 Path[i].ArrivalEventID = pathnode->arivalEventID;
195 Path[i].DepartureEventID = pathnode->departureEventID;
196#endif
197 ++i;
198 }
199 }
200
201 Path.resize(i);
202 return (i > 0 ? true : false);
203}

222void TransportHandler::generatePath(GameObjectProperties const* goInfo, TransportTemplate* transport)
223{
224 uint32_t pathId = goInfo->mo_transport.taxi_path_id;
225 TransportPath path;
226 FillTransporterPathVector(pathId, path);
227
228 if (path.size() == 0)
229 return;
230
231 std::vector<KeyFrame>& keyFrames = transport->keyFrames;
232 MovementMgr::PointsArray splinePath, allPoints;
233 bool mapChange = false;
234
235 for (uint16_t i = 0; i < path.size(); ++i)
236 allPoints.emplace_back(G3D::Vector3(path[i].x, path[i].y, path[i].z));
237
238 // Add extra points to allow derivative calculations for all path nodes
239 allPoints.insert(allPoints.begin(), allPoints.front().lerp(allPoints[1], -0.2f));
240 allPoints.push_back(allPoints.back().lerp(allPoints[allPoints.size() - 2], -0.2f));
241 allPoints.push_back(allPoints.back().lerp(allPoints[allPoints.size() - 2], -1.0f));
242
243 SplineRawInitializer initer(allPoints);
244 TransportSpline orientationSpline;
245 orientationSpline.init_spline_custom(initer);
246 orientationSpline.initLengths();
247
248 for (uint16_t i = 0; i < path.size(); ++i)
249 {
250 if (!mapChange)
251 {
252 PathNode node_i = path[i];
253 if (i != path.size() - 1 && (node_i.flags & 1 || node_i.mapid != path[i + 1].mapid))
254 {
255 keyFrames.back().Teleport = true;
256 mapChange = true;
257 }
258 else
259 {
260 KeyFrame k(node_i);
261 G3D::Vector3 h;
262 orientationSpline.evaluate_derivative(i + 1, 0.0f, h);
263 k.InitialOrientation = normalizeOrientation(std::atan2(h.y, h.x) + float(M_PI));
264
265 keyFrames.push_back(k);
266 splinePath.emplace_back(G3D::Vector3(node_i.x, node_i.y, node_i.z));
267 transport->mapsUsed.insert(k.Node.mapid);
268 }
269 }
270 else
271 {
272 mapChange = false;
273 }
274 }
275
276 // to calculate proper splines we need at least 4 spline points
277 if (splinePath.size() >= 2)
278 {
279 // Remove special catmull-rom spline points
280 if (!keyFrames.front().isStopFrame() && !keyFrames.front().Node.ArrivalEventID && !keyFrames.front().Node.DepartureEventID)
281 {
282 splinePath.erase(splinePath.begin());
283 keyFrames.erase(keyFrames.begin());
284 }
285 if (!keyFrames.back().isStopFrame() && !keyFrames.back().Node.ArrivalEventID && !keyFrames.back().Node.DepartureEventID)
286 {
287 splinePath.pop_back();
288 keyFrames.pop_back();
289 }
290 }
291
292 if (keyFrames.empty())
293 {
294 sLogger.failure("TransportHandler::generatePath no keyFrames available for Transport {}", goInfo->entry);
295 return;
296 }
297
298 if (transport->mapsUsed.size() > 1)
299 {
300 for (std::set<uint32_t>::const_iterator itr = transport->mapsUsed.begin(); itr != transport->mapsUsed.end(); ++itr)
301 {
302 if (const auto map = sMapStore.lookupEntry(*itr))
303 {
304 if (map->isInstanceableMap())
305 {
306 sLogger.failure("TransportHandler::generatePath not allowed to create a path to a instance map!");
307 return;
308 }
309 }
310 }
311
312 transport->inInstance = false;
313 }
314 else
315 {
316 if (const auto map = sMapStore.lookupEntry(*transport->mapsUsed.begin()))
317 transport->inInstance = map->isInstanceableMap();
318 }
319
320 // last to first is always "teleport", even for closed paths
321 keyFrames.back().Teleport = true;
322
323 const float speed = float(goInfo->mo_transport.move_speed);
324 const float accel = float(goInfo->mo_transport.accel_rate);
325 const float accel_dist = 0.5f * speed * speed / accel;
326
327 transport->accelTime = speed / accel;
328 transport->accelDist = accel_dist;
329
330 int32_t firstStop = -1;
331 int32_t lastStop = -1;
332
333 // first cell is arrived at by teleportation :S
334 keyFrames[0].DistFromPrev = 0;
335 keyFrames[0].Index = 1;
336 if (keyFrames[0].isStopFrame())
337 {
338 lastStop = 0;
339 firstStop = 0;
340 }
341
342 // find the rest of the distances between key points
343 // Every path segment has its own spline
344 uint16_t start = 0;
345 for (uint16_t i = 1; i < keyFrames.size(); ++i)
346 {
347 if (keyFrames[i - 1].Teleport || i + 1 == keyFrames.size())
348 {
349 auto extra = !keyFrames[i - 1].Teleport ? 1 : 0;
350 std::shared_ptr<TransportSpline> spline = std::make_shared<TransportSpline>();
351 spline->init_spline(&splinePath[start], i - start + extra, MovementMgr::SplineBase::ModeCatmullrom);
352 spline->initLengths();
353 for (auto j = start; j < i + extra; ++j)
354 {
355 keyFrames[j].Index = j - start + 1;
356 keyFrames[j].DistFromPrev = float(spline->length(j - start, j + 1 - start));
357 if (j > 0)
358 keyFrames[j - 1].NextDistFromPrev = keyFrames[j].DistFromPrev;
359 keyFrames[j].Spline = spline;
360 }
361
362 if (keyFrames[i - 1].Teleport)
363 {
364 keyFrames[i].Index = i - start + 1;
365 keyFrames[i].DistFromPrev = 0.0f;
366 keyFrames[i - 1].NextDistFromPrev = 0.0f;
367 keyFrames[i].Spline = spline;
368 }
369
370 start = i;
371 }
372
373 if (keyFrames[i].isStopFrame())
374 {
375 // remember first stop frame
376 if (firstStop == -1)
377 firstStop = i;
378 lastStop = i;
379 }
380 }
381
382 keyFrames.back().NextDistFromPrev = keyFrames.front().DistFromPrev;
383
384 if (firstStop == -1 || lastStop == -1)
385 firstStop = lastStop = 0;
386
387 // at stopping keyframes, we define distSinceStop == 0,
388 // and distUntilStop is to the next stopping keyframe.
389 // this is required to properly handle cases of two stopping frames in a row (yes they do exist)
390 float tmpDist = 0.0f;
391 for (size_t i = 0; i < keyFrames.size(); ++i)
392 {
393 auto j = (i + lastStop) % keyFrames.size();
394 if (keyFrames[j].isStopFrame() || j == lastStop)
395 tmpDist = 0.0f;
396 else
397 tmpDist += keyFrames[j].DistFromPrev;
398 keyFrames[j].DistSinceStop = tmpDist;
399 }
400
401 tmpDist = 0.0f;
402 for (int32_t i = static_cast<int32_t>(keyFrames.size()) - 1; i >= 0; i--)
403 {
404 int32_t j = (i + firstStop) % keyFrames.size();
405 tmpDist += keyFrames[(j + 1) % keyFrames.size()].DistFromPrev;
406 keyFrames[j].DistUntilStop = tmpDist;
407 if (keyFrames[j].isStopFrame() || j == firstStop)
408 tmpDist = 0.0f;
409 }
410
411 for (size_t i = 0; i < keyFrames.size(); ++i)
412 {
413 float total_dist = keyFrames[i].DistSinceStop + keyFrames[i].DistUntilStop;
414 if (total_dist < 2 * accel_dist) // won't reach full speed
415 {
416 if (keyFrames[i].DistSinceStop < keyFrames[i].DistUntilStop) // is still accelerating
417 {
418 // calculate accel+brake time for this short segment
419 float segment_time = 2.0f * std::sqrt((keyFrames[i].DistUntilStop + keyFrames[i].DistSinceStop) / accel);
420 // substract acceleration time
421 keyFrames[i].TimeTo = segment_time - std::sqrt(2 * keyFrames[i].DistSinceStop / accel);
422 }
423 else // slowing down
424 {
425 keyFrames[i].TimeTo = std::sqrt(2 * keyFrames[i].DistUntilStop / accel);
426 }
427 }
428 else if (keyFrames[i].DistSinceStop < accel_dist) // still accelerating (but will reach full speed)
429 {
430 // calculate accel + cruise + brake time for this long segment
431 float segment_time = (keyFrames[i].DistUntilStop + keyFrames[i].DistSinceStop) / speed + (speed / accel);
432 // substract acceleration time
433 keyFrames[i].TimeTo = segment_time - std::sqrt(2 * keyFrames[i].DistSinceStop / accel);
434 }
435 else if (keyFrames[i].DistUntilStop < accel_dist) // already slowing down (but reached full speed)
436 {
437 keyFrames[i].TimeTo = std::sqrt(2 * keyFrames[i].DistUntilStop / accel);
438 }
439 else // at full speed
440 {
441 keyFrames[i].TimeTo = (keyFrames[i].DistUntilStop / speed) + (0.5f * speed / accel);
442 }
443 }
444
445 // calculate tFrom times from tTo times
446 float segmentTime = 0.0f;
447 for (size_t i = 0; i < keyFrames.size(); ++i)
448 {
449 auto j = (i + lastStop) % keyFrames.size();
450 if (keyFrames[j].isStopFrame() || j == lastStop)
451 segmentTime = keyFrames[j].TimeTo;
452 keyFrames[j].TimeFrom = segmentTime - keyFrames[j].TimeTo;
453 }
454
455 // calculate path times
456 keyFrames[0].ArriveTime = 0;
457 float curPathTime = 0.0f;
458 if (keyFrames[0].isStopFrame())
459 {
460 curPathTime = float(keyFrames[0].Node.delay);
461 keyFrames[0].DepartureTime = uint32_t(curPathTime * static_cast<float>(IN_MILLISECONDS));
462 }
463
464 for (size_t i = 1; i < keyFrames.size(); ++i)
465 {
466 curPathTime += keyFrames[i - 1].TimeTo;
467 if (keyFrames[i].isStopFrame())
468 {
469 keyFrames[i].ArriveTime = uint32_t(curPathTime * static_cast<float>(IN_MILLISECONDS));
470 keyFrames[i - 1].NextArriveTime = keyFrames[i].ArriveTime;
471 curPathTime += float(keyFrames[i].Node.delay);
472 keyFrames[i].DepartureTime = uint32_t(curPathTime * static_cast<float>(IN_MILLISECONDS));
473 }
474 else
475 {
476 curPathTime -= keyFrames[i].TimeTo;
477 keyFrames[i].ArriveTime = uint32_t(curPathTime * static_cast<float>(IN_MILLISECONDS));
478 keyFrames[i - 1].NextArriveTime = keyFrames[i].ArriveTime;
479 keyFrames[i].DepartureTime = keyFrames[i].ArriveTime;
480 }
481 }
482
483 keyFrames.back().NextArriveTime = keyFrames.back().DepartureTime;
484
485 transport->pathTime = keyFrames.back().DepartureTime;
486 sLogger.debugFlag(AscEmu::Logging::LF_MAP, "TransportHandler: total time {} at transport {} \n", transport->pathTime, transport->entry);
487}

489float TransportHandler::normalizeOrientation(float o)
490{
491 if (o < 0)
492 {
493 float mod = o * -1;
494 mod = std::fmod(mod, 2.0f * static_cast<float>(M_PI));
495 mod = -mod + 2.0f * static_cast<float>(M_PI);
496 return mod;
497 }
498 return std::fmod(o, 2.0f * static_cast<float>(M_PI));
499}
