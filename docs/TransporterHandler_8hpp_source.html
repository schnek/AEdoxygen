58public:
59 inline void setLength(const unsigned int sz)
60 {
61 i_nodes.resize(sz);
62 }
63
64 inline size_t size(void) const { return i_nodes.size(); }
65 inline void resize(unsigned int sz) { i_nodes.resize(sz); }
66 inline void clear(void) { i_nodes.clear(); }
67 inline PathNode* getNodes(void) { return &i_nodes[0]; }
68 float getTotalLength(void)
69 {
70 float len = 0, xd, yd, zd;
71 for (unsigned int idx = 1; idx < i_nodes.size(); ++idx)
72 {
73 xd = i_nodes[idx].x - i_nodes[idx - 1].x;
74 yd = i_nodes[idx].y - i_nodes[idx - 1].y;
75 zd = i_nodes[idx].z - i_nodes[idx - 1].z;
76 len += std::sqrt(xd * xd + yd * yd + zd * zd);
77 }
78 return len;
79 }
80
81 PathNode & operator[](const unsigned int idx) { return i_nodes[idx]; }
82 const PathNode & operator()(const unsigned int idx) const { return i_nodes[idx]; }
83
84protected:

86};
