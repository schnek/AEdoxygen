62void TwinsAI::OnCombatStart(Unit*)
63{
64 getInstanceScript()->sendUnitEncounter(EncounterFrameEngage, getCreature());
65 setZoneWideCombat();
66
67 if (CreatureAIScript* pSister = getLinkedCreatureAIScript())
68 {
69 SpellInfo const* spellInfo = sSpellMgr.getSpellInfo(MyEmphatySpellId);
70 auto pAura = sSpellMgr.newAura(spellInfo, (int32_t)GetDuration(sSpellDurationStore.lookupEntry(spellInfo->getDurationIndex())), getCreature(), pSister->getCreature());
71 getCreature()->addAura(std::move(pAura));
72 setZoneWideCombat(pSister->getCreature());
73 }
74
75 getInstanceScript()->setBossState(DATA_TWIN_VALKIRIES, EncounterStates::InProgress);
76
77 castSpellOnSelf(SurgeSpellId);
78
79 SpellDesc mBerserkInfo(twins::SPELL_BERSERK, FilterArgs(TargetFilter_Self), false);
80 mBerserkInfo.setUseSpellCD(true);
81 mBerserkInfo.addDBMessage(twins::SAY_BERSERK);
82 addAISpell(mBerserkInfo, DoLoopScheduler(isHeroic() ? 6min : 8min, 100.0f));
83
84 SpellDesc mTwinSpikeInfo(SpikeSpellId, FilterArgs(TargetFilter_Current), false);
85 addAISpell(mTwinSpikeInfo, DoLoopScheduler(20s, 100.0f));
86
87 if (isHeroic())
88 {
89 SpellDesc mTouchInfo(TouchSpellId, FilterArgs(TargetFilter_InRangeOnly, 0.0f, 200.0f, OtherEssenceSpellId), false);
90 addAISpell(mTouchInfo, DoLoopScheduler(10s, 20s, 33.0f));
91 }

237void FjolaAI::AIUpdate(unsigned long time_passed)
238{
239 if (!getCreature()->getAIInterface()->getCurrentTarget() && !getScriptPhase() != twins::PHASE_EVENT)
240 return;
241
242 scriptEvents.updateEvents(time_passed, getScriptPhase());
243
244 if (_isCasting())
245 return;
246
247 if (scriptEvents.getFinishedEvent() == twins::EVENT_SPECIAL_ABILITY)
248 {
249 if (CurrentStage == twins::MAX_STAGES)
250 generateStageSequence();
251
252 switch (Stage[CurrentStage])
253 {
254 case twins::STAGE_DARK_VORTEX:
255 if (CreatureAIScript* sister = getLinkedCreatureAIScript())
256 sister->DoAction(twins::ACTION_VORTEX);
257 break;
258 case twins::STAGE_DARK_PACT:
259 if (CreatureAIScript* sister = getLinkedCreatureAIScript())
260 sister->DoAction(twins::ACTION_PACT);
261 break;
262 case twins::STAGE_LIGHT_VORTEX:
263 DoAction(twins::ACTION_VORTEX);
264 break;
265 case twins::STAGE_LIGHT_PACT:
266 DoAction(twins::ACTION_PACT);
267 break;
268 default:
269 break;
270 }
271 ++CurrentStage;
272 scriptEvents.addEvent(twins::EVENT_SPECIAL_ABILITY, 45000);
273 }
274}

281void FjolaAI::generateStageSequence()
282{
283 CurrentStage = 0;
284
285 // Initialize and clean up.
286 for (int i = 0; i < twins::MAX_STAGES; ++i)
287 Stage[i] = i;
288
289 // Allocate an unique random stage to each position in the array.
290 for (int i = 0; i < twins::MAX_STAGES - 1; ++i)
291 {
292 int random = i + Util::getRandomUInt(0, twins::MAX_STAGES - i);
293 std::swap(Stage[i], Stage[random]);
294 }
295}

524 {

546//////////////////////////////////////////////////////////////////////////////////////////
547/// Spell Dark Essence and Light Essence
548SpellScriptExecuteState EssenceScript::beforeAuraEffect(Aura* aur, AuraEffectModifier* aurEff, bool apply)
549{
550 if (aurEff->getEffectIndex() == EFF_INDEX_0)
551 {
552 if (Unit* owner = aur->getOwner())
553 {
554 if (aur->getSpellInfo())
555 {
556 if (uint32_t poweringUp = twins::SPELL_POWERING_UP)
557 {
558 if (Util::getRandomUInt(0, 99) < 5)
559 aur->GetUnitTarget()->castSpell(aur->GetUnitTarget(), twins::SPELL_SURGE_OF_SPEED, true);
560
561 // Twin Vortex part
562 int32_t stacksCount = aur->getSpellInfo()->calculateEffectValue(EFF_INDEX_0) / 1000 - 1;
563
564 if (stacksCount)
565 {
566 if (aur->getSpellInfo()->getId() == twins::SPELL_DARK_VORTEX_DAMAGE || aur->getSpellInfo()->getId() == twins::SPELL_LIGHT_VORTEX_DAMAGE)
567 {
568 if (Aura* aura = owner->getAuraWithId(poweringUp))
569 {
570 aura->refreshOrModifyStack(false, stacksCount);
571 owner->castSpell(owner, poweringUp, true);
572 }
573 else
574 {
575 owner->castSpell(owner, poweringUp, true);
576 if (Aura* newAura = owner->getAuraWithId(poweringUp))
577 newAura->refreshOrModifyStack(false, stacksCount);
578 }
579 }
580 }
581
582 // Picking floating balls
583 if (aur->getSpellInfo()->getId() == twins::SPELL_UNLEASHED_DARK || aur->getSpellInfo()->getId() == twins::SPELL_UNLEASHED_LIGHT)
584 {
585 // need to do the things in this order, else players might have 100 charges of Powering Up without anything happening
586 if (Aura* aura = owner->getAuraWithId(poweringUp))
587 {
588 // 2 lines together add the correct amount of buff stacks
589 aura->refreshOrModifyStack(false, stacksCount);
590 owner->castSpell(owner, poweringUp, true);
591 }
592 else
593 {
594 owner->castSpell(owner, poweringUp, true);
595 if (Aura* newAura = owner->getAuraWithId(poweringUp))
596 newAura->refreshOrModifyStack(false, stacksCount);
597 }
598 }
599 }
600 }
601 }
602 }
603
604 return SpellScriptExecuteState();
605}
