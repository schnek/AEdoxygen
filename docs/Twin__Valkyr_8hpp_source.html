178 template<class T> inline
179 const T& RAID_MODE(Creature* pCreature, const T& normal10, const T& normal25, const T& heroic10, const T& heroic25) const
180 {
181 if (pCreature->getWorldMap()->getInstance())
182 {
183 switch (pCreature->getWorldMap()->getDifficulty())
184 {
185 case InstanceDifficulty::RAID_10MAN_NORMAL:
186 return normal10;
187 case InstanceDifficulty::RAID_25MAN_NORMAL:
188 return normal25;
189 case InstanceDifficulty::RAID_10MAN_HEROIC:
190 return heroic10;
191 case InstanceDifficulty::RAID_25MAN_HEROIC:
192 return heroic25;
193 default:
194 break;
195 }
196 }
197
198 return normal10;
199 }
