113void UpdateManager::internalProcessPendingUpdates()
114{
115 if (!readyForUpdate())
116 return;
117
118 ByteBuffer buffer(calculateBufferSize());
119
120 if (m_creationBuffer.size() > 0 || m_outOfRangeIdCount > 0)
121 {
122#if VERSION_STRING >= Cata
123 buffer << uint16_t(m_owner->GetMapId());
124#endif
125
126 if (m_outOfRangeIds.size() > 0)
127 buffer << uint32_t(m_creationCount + 1);
128 else
129 buffer << uint32_t(m_creationCount);
130
131
132#if VERSION_STRING <= TBC
133 buffer << uint8_t(1);
134#endif
135
136 if (m_outOfRangeIdCount > 0)
137 {
138 buffer << uint8_t(UPDATETYPE_OUT_OF_RANGE_OBJECTS);
139 buffer << uint32_t(m_outOfRangeIdCount);
140 buffer.append(m_outOfRangeIds);
141 m_outOfRangeIds.clear();
142 m_outOfRangeIdCount = 0;
143 }
144
145 if (m_creationBuffer.size() > 0)
146 {
147 buffer.append(m_creationBuffer);
148 m_creationBuffer.clear();
149 m_creationCount = 0;
150 }
151
152 auto sent_packet = false;
153#if VERSION_STRING < Cata
154 if (buffer.wpos() > m_compressionThreshold)
155 sent_packet = m_owner->compressAndSendUpdateBuffer(uint32_t(buffer.wpos()), buffer.contents());
156#endif
157
158 if (!sent_packet)
159 m_owner->getSession()->OutPacket(SMSG_UPDATE_OBJECT, uint16_t(buffer.wpos()), buffer.contents());
160}
161
162 if (m_updateBuffer.size() > 0)
163 {
164 buffer.clear();
165
166#if VERSION_STRING >= Cata
167 buffer << uint16_t(m_owner->GetMapId());
168#endif
169
170 if (m_outOfRangeIds.size() > 0)
171 buffer << uint32_t(m_updateCount + 1);
172 else
173 buffer << uint32_t(m_updateCount);
174
175#if VERSION_STRING <= TBC
176 buffer << uint8_t(1);
177#endif
178
179 buffer.append(m_updateBuffer);
180 m_updateBuffer.clear();
181 m_updateCount = 0;
182
183 auto sent_packet = false;
184#if VERSION_STRING < Cata
185 if (buffer.wpos() > m_compressionThreshold)
186 sent_packet = m_owner->compressAndSendUpdateBuffer(uint32_t(buffer.wpos()), buffer.contents());
187#endif
188
189 if (!sent_packet)
190 m_owner->getSession()->OutPacket(SMSG_UPDATE_OBJECT, uint16_t(buffer.wpos()), buffer.contents());
191 }
192
193 m_processPending = false;
194 internalSendDelayedPackets();
195}
