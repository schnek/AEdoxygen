93 std::size_t max_consecutive(std::string_view name, const bool case_insensitive, const std::locale& locale)
94 {
95 std::size_t current_run = 0;
96 std::size_t longest_run = 0;
97 char last = 0;
98
99 for(auto c : name)
100 {
101 if(case_insensitive)
102 {
103 c = std::tolower(c, locale);
104 }
105
106 if(c == last)
107 {
108 ++current_run;
109 } else
110 {
111 current_run = 1;
112 }
113
114 if(current_run > longest_run)
115 {
116 longest_run = current_run;
117 }
118
119 last = c;
120 }
121
122 return longest_run;
123 }

214 std::string GetDateStringFromSeconds(uint32_t seconds)

216 uint32_t in_seconds = seconds;

219 uint32_t in_days = 0;

223 in_years = (in_seconds / 60 / 60 / 24 / 30 / 12);
224 in_months = (in_seconds / 60 / 60 / 24 / 30) % 12;
225 in_days = (in_seconds / 60 / 60 / 24) % 30;
226 in_hours = (in_seconds / 60 / 60) % 24;
227 in_minutes = (in_seconds / 60) % 60;
228 in_seconds = in_seconds % 60;
229
230 std::stringstream date_time_stream;

232 date_time_stream << in_years << " years, ";

234 if (in_months)

244 date_time_stream << in_minutes << " minutes, ";
245
246 if (in_seconds)
247 date_time_stream << in_seconds << " seconds, ";

249 return date_time_stream.str();

252 uint32_t GetTimePeriodFromString(const char* str)
253 {
254 uint32_t multiplier;
255
256 std::string time_str = str;

258 std::istringstream read_time_var(time_str);
259 uint32_t time_period = 0;

261 read_time_var >> time_period;
262 read_time_var >> time_var;
263
264 AscEmu::Util::Strings::toLowerCase(time_var);
265
266 if (time_var.compare("y") == 0)
267 multiplier = TimeVars::Year;
268 else if (time_var.compare("m") == 0)
269 multiplier = TimeVars::Month;
270 else if (time_var.compare("d") == 0)
271 multiplier = TimeVars::Day;
272 else if (time_var.compare("h") == 0)
273 multiplier = TimeVars::Hour;
274 else
275 multiplier = TimeVars::Minute;
276
277 time_period = (multiplier * time_period);
278
279 return time_period;

282 uint32_t getGameTime()
283 {
284 const auto now = std::chrono::system_clock::now();
285 auto inTimeT = std::chrono::system_clock::to_time_t(now);
286
287 const uint32_t currentYear = localtime(&inTimeT)->tm_year - 100;
288 const uint32_t currentMonth = localtime(&inTimeT)->tm_mon;
289 const uint32_t currentDayInMonth = localtime(&inTimeT)->tm_mday - 1;
290
291 const uint32_t currentDayInWeek = localtime(&inTimeT)->tm_wday == 0 ? 6 : localtime(&inTimeT)->tm_wday - 1;
292
293 const uint32_t currentHours = localtime(&inTimeT)->tm_hour;
294 const uint32_t currentMinutes = localtime(&inTimeT)->tm_min;
295
296
297 uint32_t gameTimeValue = currentMinutes << TimeShiftmask::Minute & TimeBitmask::Minute;
298 gameTimeValue |= currentHours << TimeShiftmask::Hour & TimeBitmask::Hour;
299 gameTimeValue |= currentDayInWeek << TimeShiftmask::Weekday & TimeBitmask::Weekday;
300 gameTimeValue |= currentDayInMonth << TimeShiftmask::Day & TimeBitmask::Day;
301 gameTimeValue |= currentMonth << TimeShiftmask::Month & TimeBitmask::Month;
302 gameTimeValue |= currentYear << TimeShiftmask::Year & TimeBitmask::Year;
303
304 return gameTimeValue;
305 }

423 unsigned int makeIP(std::string_view _str)
424 {
425 std::array<unsigned int, 4> bytes;
426 char dot; // To capture the dots between the numbers
427
428 std::stringstream ss{ std::string{_str} }; // Convert string_view to string for stringstream
429
430 for (auto& byte : bytes)
431 {
432 if (!(ss >> byte) || byte > 255)
433 {
434 return 0; // Return 0 if parsing fails or if any byte is out of range
435 }
436 if (&byte != &bytes.back())
437 {
438 if (!(ss >> dot) || dot != '.')
439 {
440 return 0; // Return 0 if dots are incorrectly placed
441 }
442 }
443 }
444
445 return (bytes[0]) | (bytes[1] << 8) | (bytes[2] << 16) | (bytes[3] << 24);
446 }

448 bool parseCIDRBan(uint32_t _ip, uint32_t _mask, uint32_t _maskBits)
449 {
450 if (_maskBits > 32)
451 return false; // Sanity check
452
453 // Define the leftover bits comparison table
454 constexpr std::array<uint8_t, 9> leftover_bits_compare = { 0x00, 0x80, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC, 0xFE, 0xFF };
455
456 int full_bytes = _maskBits / 8;
457 int leftover_bits = _maskBits % 8;
458
459 // Convert IP and Mask to arrays of bytes
460 auto ip_bytes = reinterpret_cast<uint8_t*>(&_ip);
461 auto mask_bytes = reinterpret_cast<uint8_t*>(&_mask);
462
463 // Compare full bytes
464 if (full_bytes > 0)
465 {
466 if (std::memcmp(ip_bytes, mask_bytes, full_bytes) != 0)
467 return false;
468 }
469
470 // Compare leftover bits
471 if (leftover_bits > 0)
472 {
473 uint8_t mask = leftover_bits_compare[leftover_bits];
474 if ((ip_bytes[full_bytes] & mask) != (mask_bytes[full_bytes] & mask))
475 return false;
476 }
477
478 return true; // All testable bits match
479 }
