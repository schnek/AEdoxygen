124 template<class Factory>
125 struct LazyInstanceCreator
126 {
127 /*!
128 * \brief Converts a struct to a specified type using a lazy evaluation approach.
129 *
130 * This struct utilizes an internal lambda factory to generate an instance of the provided type
131 * when a conversion is necessary. It is particularly useful when working with objects and
132 * emplace/try_emplace.
133 *
134 * Important notes:
135 *
136 * - The internal lambda will only be invoked if a conversion is required, ensuring that the
137 * evaluation is lazy and avoids unnecessary overhead. If i.e. emplace operation fails, no
138 * conversion will take place, meaning that resources such as unique_ptr will not be allocated,
139 * preventing even temporary unnecessary memory allocations.
140 *
141 * - Works with both emplace and try_emplace
142 */
143 constexpr LazyInstanceCreator(Factory&& factory) : m_factory(std::move(factory))
144 {
145 }
146
147 constexpr operator auto() const noexcept(std::is_nothrow_invocable_v<const Factory&>)
148 {
149 return m_factory();
150 }
151
152 Factory m_factory;
153 };
