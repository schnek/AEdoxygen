64 {
65 Vector3 pos;

108 // load one tile (internal use only)
109 bool VMapManager2::_loadMap(uint32_t mapId, const std::string& basePath, uint32_t tileX, uint32_t tileY)
110 {
111 InstanceTreeMap::iterator instanceTree = iInstanceMapTrees.find(mapId);
112 if (instanceTree == iInstanceMapTrees.end())
113 {
114 if (thread_safe_environment)
115 instanceTree = iInstanceMapTrees.insert(InstanceTreeMap::value_type(mapId, nullptr)).first;
116 else
117 {
118 sLogger.failure("Invalid mapId {} tile [{}, {}] passed to VMapManager2 after startup in thread unsafe environment", mapId, tileX, tileY);
119 ASSERT(false);
120 }
121 }
122
123 if (!instanceTree->second)
124 {
125 std::string mapFileName = getMapFileName(mapId);
126 auto newTree = std::make_unique<StaticMapTree>(mapId, basePath);
127 if (!newTree->InitMap(mapFileName, this))
128 {
129 return false;
130 }
131 instanceTree->second = std::move(newTree);
132 }
133
134 return instanceTree->second->LoadMapTile(tileX, tileY, this);
135 }

163 bool VMapManager2::isInLineOfSight(unsigned int mapId, float x1, float y1, float z1, float x2, float y2, float z2)
164 {
165 if (!isLineOfSightCalcEnabled() || IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_LOS))
166 return true;
167
168 InstanceTreeMap::const_iterator instanceTree = GetMapTree(mapId);
169 if (instanceTree != iInstanceMapTrees.end())
170 {
171 Vector3 pos1 = convertPositionToInternalRep(x1, y1, z1);
172 Vector3 pos2 = convertPositionToInternalRep(x2, y2, z2);
173 if (pos1 != pos2)
174 {
175 return instanceTree->second->isInLineOfSight(pos1, pos2);
176 }
177 }
178
179 return true;
180 }

182 /**
183 get the hit position and return true if we hit something
184 otherwise the result pos will be the dest pos
185 */
186 bool VMapManager2::getObjectHitPos(unsigned int mapId, float x1, float y1, float z1, float x2, float y2, float z2, float& rx, float &ry, float& rz, float modifyDist)
187 {
188 if (isLineOfSightCalcEnabled() && !IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_LOS))
189 {
190 InstanceTreeMap::const_iterator instanceTree = GetMapTree(mapId);
191 if (instanceTree != iInstanceMapTrees.end())
192 {
193 Vector3 pos1 = convertPositionToInternalRep(x1, y1, z1);
194 Vector3 pos2 = convertPositionToInternalRep(x2, y2, z2);
195 Vector3 resultPos;
196 bool result = instanceTree->second->getObjectHitPos(pos1, pos2, resultPos, modifyDist);
197 resultPos = convertPositionToInternalRep(resultPos.x, resultPos.y, resultPos.z);
198 rx = resultPos.x;
199 ry = resultPos.y;
200 rz = resultPos.z;
201 return result;
202 }
203 }
204
205 rx = x2;
206 ry = y2;
207 rz = z2;
208
209 return false;
210 }

216 float VMapManager2::getHeight(unsigned int mapId, float x, float y, float z, float maxSearchDist)
217 {
218 if (isHeightCalcEnabled() && !IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_HEIGHT))
219 {
220 InstanceTreeMap::const_iterator instanceTree = GetMapTree(mapId);
221 if (instanceTree != iInstanceMapTrees.end())
222 {
223 Vector3 pos = convertPositionToInternalRep(x, y, z);
224 float height = instanceTree->second->getHeight(pos, maxSearchDist);
225 if (!(height < G3D::finf()))
226 return height = VMAP_INVALID_HEIGHT_VALUE; // No height
227
228 return height;
229 }
230 }
231
232 return VMAP_INVALID_HEIGHT_VALUE;

236 {

253 bool VMapManager2::getLiquidLevel(uint32_t mapId, float x, float y, float z, uint8_t reqLiquidType, float& level, float& floor, uint32_t& type, uint32_t& mogpFlags) const
254 {
255 if (!IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_LIQUIDSTATUS))
256 {
257 InstanceTreeMap::const_iterator instanceTree = GetMapTree(mapId);
258 if (instanceTree != iInstanceMapTrees.end())
259 {
260 LocationInfo info;
261 Vector3 pos = convertPositionToInternalRep(x, y, z);
262 if (instanceTree->second->GetLocationInfo(pos, info))
263 {
264 floor = info.ground_Z;
265 ASSERT(floor < std::numeric_limits<float>::max());
266 ASSERT(info.hitModel);
267 if (info.hitModel)
268 {
269 type = info.hitModel->GetLiquidType(); // entry from LiquidType.dbc
270 mogpFlags = info.hitModel->GetMogpFlags();
271 }
272 if (reqLiquidType && !(GetLiquidFlagsPtr(type) & reqLiquidType))
273 return false;
274 ASSERT(info.hitInstance);
275 if (info.hitInstance && info.hitInstance->GetLiquidLevel(pos, info, level))
276 return true;
277 }
278 }
279 }
280
281 return false;
282 }

284 void VMapManager2::getAreaAndLiquidData(unsigned int mapId, float x, float y, float z, uint8_t reqLiquidType, AreaAndLiquidData& data) const
285 {
286 if (IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_LIQUIDSTATUS))
287 {
288 data.floorZ = z;
289 int32_t adtId, rootId, groupId;
290 uint32_t flags;
291 if (getAreaInfo(mapId, x, y, data.floorZ, flags, adtId, rootId, groupId))
292 data.areaInfo.emplace(adtId, rootId, groupId, flags);
293 return;
294 }
295 InstanceTreeMap::const_iterator instanceTree = GetMapTree(mapId);
296 if (instanceTree != iInstanceMapTrees.end())
297 {
298 LocationInfo info;
299 Vector3 pos = convertPositionToInternalRep(x, y, z);
300 if (instanceTree->second->GetLocationInfo(pos, info))
301 {
302 ASSERT(info.hitModel);
303 ASSERT(info.hitInstance);
304 data.floorZ = info.ground_Z;
305 uint32_t liquidType = info.hitModel->GetLiquidType();
306 float liquidLevel;
307 if (!reqLiquidType || (GetLiquidFlagsPtr(liquidType) & reqLiquidType))
308 if (info.hitInstance->GetLiquidLevel(pos, info, liquidLevel))
309 data.liquidInfo.emplace(liquidType, liquidLevel);
310
311 if (!IsVMAPDisabledForPtr(mapId, VMAP_DISABLE_AREAFLAG))
312 data.areaInfo.emplace(info.hitInstance->adtId, info.rootId, info.hitModel->GetWmoID(), info.hitModel->GetMogpFlags());
313 }
314 }

317 WorldModel* VMapManager2::acquireModelInstance(const std::string& basepath, const std::string& filename)
318 {
319 //! Critical section, thread safe access to iLoadedModelFiles
320 std::lock_guard<std::mutex> lock(LoadedModelFilesLock);
321
322 ModelFileMap::iterator model = iLoadedModelFiles.find(filename);
323 if (model == iLoadedModelFiles.end())
324 {
325 auto worldmodel = std::make_unique<WorldModel>();
326 if (!worldmodel->readFile(basepath + filename + ".vmo"))
327 {
328 sLogger.failure("could not load '{}{}.vmo'", basepath, filename);
329 return nullptr;
330 }
331 sLogger.debug("VMapManager2 loading file '{}{}'", basepath, filename);
332 model = iLoadedModelFiles.insert(std::pair<std::string, ManagedModel>(filename, ManagedModel())).first;
333 model->second.setModel(std::move(worldmodel));
334 }
335 model->second.incRefCount();
336 return model->second.getModel();
337 }

339 void VMapManager2::releaseModelInstance(const std::string &filename)
340 {
341 //! Critical section, thread safe access to iLoadedModelFiles
342 std::lock_guard<std::mutex> lock(LoadedModelFilesLock);
343
344 ModelFileMap::iterator model = iLoadedModelFiles.find(filename);
345 if (model == iLoadedModelFiles.end())
346 {
347 sLogger.failure("trying to unload non-loaded file '{}'", filename);
348 return;
349 }
350 if (model->second.decRefCount() == 0)
351 {
352 sLogger.debug("VMapManager2 unloading file '{}'", filename);
353 iLoadedModelFiles.erase(model);
354 }
355 }
