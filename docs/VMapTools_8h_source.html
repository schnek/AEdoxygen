55 static bool collisionLocationForMovingPointFixedAABox(
56 const G3D::Vector3& origin,
57 const G3D::Vector3& dir,
58 const G3D::AABox& box,
59 G3D::Vector3& location,
60 bool& Inside)
61 {
62
63 // Integer representation of a floating-point value.
64#define IR(x) (reinterpret_cast<G3D::uint32_t const&>(x))
65
66 Inside = true;
67 const G3D::Vector3& MinB = box.low();
68 const G3D::Vector3& MaxB = box.high();
69 G3D::Vector3 MaxT(-1.0f, -1.0f, -1.0f);
70
71 // Find candidate planes.
72 for (int i = 0; i < 3; ++i)
73 {
74 if (origin[i] < MinB[i])
75 {
76 location[i] = MinB[i];
77 Inside = false;
78
79 // Calculate T distances to candidate planes
80 if (IR(dir[i]))
81 {
82 MaxT[i] = (MinB[i] - origin[i]) / dir[i];
83 }
84 }
85 else if (origin[i] > MaxB[i])
86 {
87 location[i] = MaxB[i];
88 Inside = false;
89
90 // Calculate T distances to candidate planes
91 if (IR(dir[i]))
92 {
93 MaxT[i] = (MaxB[i] - origin[i]) / dir[i];
94 }
95 }
96 }
97
98 if (Inside)
99 {
100 // definite hit
101 location = origin;
102 return true;
103 }
104
105 // Get largest of the maxT's for final choice of intersection
106 int WhichPlane = 0;
107 if (MaxT[1] > MaxT[WhichPlane])
108 {
109 WhichPlane = 1;
110 }
111
112 if (MaxT[2] > MaxT[WhichPlane])
113 {
114 WhichPlane = 2;
115 }
116
117 // Check final candidate actually inside box
118 if (IR(MaxT[WhichPlane]) & 0x80000000)
119 {
120 // Miss the box
121 return false;
122 }
123
124 for (int i = 0; i < 3; ++i)
125 {
126 if (i != WhichPlane)
127 {
128 location[i] = origin[i] + MaxT[WhichPlane] * dir[i];
129 if ((location[i] < MinB[i]) ||
130 (location[i] > MaxB[i]))
131 {
132 // On this plane we're outside the box extents, so
133 // we miss the box
134 return false;
135 }
136 }
137 }
138 /*
139 // Choose the normal to be the plane normal facing into the ray
140 normal = G3D::Vector3::zero();
141 normal[WhichPlane] = (dir[WhichPlane] > 0) ? -1.0 : 1.0;
142 */
143 return true;
144
145#undef IR
146 }
