434inline Vector2 Vector2::directionOrZero() const {
435 float mag = length();
436 if (mag < 0.0000001f) {
437 return Vector2::zero();
438 } else if (mag < 1.00001f && mag > 0.99999f) {
439 return *this;
440 } else {
441 return *this * (1.0f / mag);
442 }
443}
