27Vector3int32::Vector3int32(const Any& any) {
28 *this = Vector3int32();
29 any.verifyNameBeginsWith("Vector3int32", "Point3int32");
30
31 switch (any.type()) {
32 case Any::TABLE:
33
34 for (Any::AnyTable::Iterator it = any.table().begin(); it.isValid(); ++it) {
35 const std::string& key = toLower(it->key);
36
37 if (key == "x") {
38 x = it->value;
39 } else if (key == "y") {
40 y = it->value;
41 } else if (key == "z") {
42 z = it->value;
43 } else {
44 any.verify(false, "Illegal key: " + it->key);
45 }
46 }
47 break;
48
49 case Any::ARRAY:
50
51 (void)any.name();

53 x = y = z = any[0];
