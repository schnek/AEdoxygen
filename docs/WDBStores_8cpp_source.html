352 {
353 std::map<uint32_t, uint32_t> InspectTalentTabPos;
354 std::map<uint32_t, uint32_t> InspectTalentTabSize;
355 std::map<uint32_t, uint32_t> InspectTalentTabBit;
356
357 uint32_t talent_max_rank;
358 uint32_t talent_pos;
359 uint32_t talent_class;
360
361 for (uint32_t i = 0; i < sTalentStore.getNumRows(); ++i)
362 {
363 auto talent_info = sTalentStore.lookupEntry(i);
364 if (talent_info == nullptr)
365 continue;
366
367 // Don't add invalid talents or Hunter Pet talents (trees 409, 410 and 411) to the inspect table
368 if (talent_info->TalentTree == 409 || talent_info->TalentTree == 410 || talent_info->TalentTree == 411)
369 continue;
370
371 auto talent_tab = sTalentTabStore.lookupEntry(talent_info->TalentTree);
372 if (talent_tab == nullptr)
373 continue;
374
375 talent_max_rank = 0;
376 for (uint32_t j = 5; j > 0; --j)
377 {
378 if (talent_info->RankID[j - 1])
379 {
380 talent_max_rank = j;
381 break;
382 }
383 }
384
385 InspectTalentTabBit[(talent_info->Row << 24) + (talent_info->Col << 16) + talent_info->TalentID] = talent_max_rank;
386 InspectTalentTabSize[talent_info->TalentTree] += talent_max_rank;
387 }
388
389 for (uint32_t i = 0; i < sTalentTabStore.getNumRows(); ++i)
390 {
391 auto talent_tab = sTalentTabStore.lookupEntry(i);
392 if (talent_tab == nullptr)
393 continue;
394
395 if (talent_tab->ClassMask == 0)
396 continue;
397
398 talent_pos = 0;
399
400 for (talent_class = 0; talent_class < 12; ++talent_class)
401 {
402 if (talent_tab->ClassMask & (1 << talent_class))
403 break;
404 }
405
406 if (talent_class > 0 && talent_class < 12)
407 InspectTalentTabPages[talent_class][talent_tab->TabPage] = talent_tab->TalentTabID;
408
409 for (std::map<uint32_t, uint32_t>::iterator itr = InspectTalentTabBit.begin(); itr != InspectTalentTabBit.end(); ++itr)
410 {
411 uint32_t talent_id = itr->first & 0xFFFF;
