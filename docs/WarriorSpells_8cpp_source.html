29bool Execute(uint8_t effectIndex, Spell* pSpell)
30{
31 if (pSpell->getPlayerCaster() == NULL || pSpell->getUnitTarget() == NULL)
32 {
33 return true;
34 }
35
36 Player* Caster = pSpell->getPlayerCaster();
37 Unit* Target = pSpell->getUnitTarget();
38
39 uint32_t rage = Caster->getPower(POWER_TYPE_RAGE);
40
41 if (Caster->hasAurasWithId(58367)) // Glyph of Execution: Your Execute ability deals damage as if you had 10 additional rage.
42 {
43 rage += 10;
44 }
45
46 uint32_t toadd = 0;
47 int32_t dmg = 0;
48 uint32_t multiple[] = { 0, 3, 6, 9, 12, 15, 18, 21, 30, 38, };
49
50 const uint8_t rank = pSpell->getSpellInfo()->hasSpellRanks() ? pSpell->getSpellInfo()->getRankInfo()->getRank() : 1;
51 if (rage >= 30)
52 {
53 toadd = (multiple[rank] * 30);
54 }
55 else
56 {
57 toadd = (multiple[rank] * rage);
58 }
59
60 dmg = pSpell->calculateEffect(effectIndex);
61 dmg += Caster->getAttackPower() / 5;
62 dmg += toadd;
63
64 Caster->strike(Target, MELEE, pSpell->getSpellInfo(), 0, 0, dmg, false, false);
65
66 return true;
67}

119bool Charge(uint8_t effectIndex, Spell* s)
120{
121 if (!s->getUnitCaster())
122 {
123 return false;
124 }
125
126 uint32_t rage_to_gen = s->getSpellInfo()->getEffectBasePoints(effectIndex) + 1;
127 if (s->getPlayerCaster())
128 {
129 for (auto itr = s->getPlayerCaster()->getSpellSet().begin(); itr != s->getPlayerCaster()->getSpellSet().end(); ++itr)
130 {
131 if (*itr == 12697)
132 {
133 rage_to_gen += 100;
134 }
135
136 if (*itr == 12285)
137 {
138 rage_to_gen += 50;
139 }
140 }
141 }
142
143 // Add the rage to the caster
144 s->getUnitCaster()->modPower(POWER_TYPE_RAGE, rage_to_gen);
145
146 return true;
147}

167bool BerserkerRage(uint8_t /*effectIndex*/, Aura* a, bool apply)
168{
169 Unit* u = a->getOwner();
170 Player* p_target = NULL;
171
172 if (u->isPlayer())
173 {
174 p_target = static_cast<Player*>(u);
175 }
176
177 if (p_target == NULL)
178 {
179 return true;
180 }
181
182 if (apply)
183 {
184 p_target->m_rageFromDamageTaken += 100;
185 }
186 else
187 {
188 p_target->m_rageFromDamageTaken -= 100;
189 }
190
191 SpellMechanic mechanics[4] = { MECHANIC_NONE };
192 for (uint8_t i = 0; i < 3; i++)
193 {
194 if (apply)
195 {
196 p_target->m_mechanicsDispels[a->getSpellInfo()->getEffectMiscValue(i)]++;
197 mechanics[i] = static_cast<SpellMechanic>(a->getSpellInfo()->getEffectMiscValue(i));
198 }
199 else
200 {
201 p_target->m_mechanicsDispels[a->getSpellInfo()->getEffectMiscValue(i)]--;
202 }
203 }
204
205 if (apply)
206 p_target->removeAllAurasBySpellMechanic(mechanics);
207
208 return true;
209}
