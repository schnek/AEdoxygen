20#if VERSION_STRING < Mop
21class Flurry : public SpellScript
22{
23public:
24 void onCreateSpellProc(SpellProc* proc, Object* /*obj*/) override
25 {
26 proc->setExtraProcFlags(EXTRA_PROC_ON_CRIT_ONLY);
27 }
28
29 void onAuraApply(Aura* aur) override
30 {
31 // Make the proc aura proc its mother spell
32 // This is only needed to consume proc charges from this aura on melee attacks
33 // DBC data also have PROC_ON_DONE_MELEE_HIT flag set so this is not probably far-fetched
34 motherAuraId = aur->pSpellId;
35 spellProc = aur->getOwner()->addProcTriggerSpell(sSpellMgr.getSpellInfo(motherAuraId), aur, aur->getCasterGuid());
36 if (spellProc != nullptr)
37 spellProc->skipOnNextHandleProc(true);
38 }
39
40 void onAuraRefreshOrGainNewStack(Aura* /*aur*/, uint32_t /*newStacks*/, uint32_t /*oldStacks*/) override
41 {
42 // If this proc is not skipped in next ::handleProc event,
43 // the same attack, that created this aura, will consume one proc charge
44 if (spellProc != nullptr)
45 spellProc->skipOnNextHandleProc(true);
46 }
47
48 void onAuraRemove(Aura* aur, AuraRemoveMode /*mode*/) override
49 {
50 aur->getOwner()->removeProcTriggerSpell(motherAuraId, aur->getCasterGuid());
51 motherAuraId = 0;
52 spellProc = nullptr;
53 }
54
55private:
56 uint32_t motherAuraId = 0;
57 SpellProc* spellProc = nullptr;
58};
