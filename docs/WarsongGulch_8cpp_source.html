91{
92 // gates are always spawned, so mapmgr will clean them up
93 for (uint8_t i = 0; i < 6; ++i)
94 {

111 if ((*itr) != nullptr)
112 {
113 if (!(*itr)->IsInWorld())

116 }
117

134void WarsongGulch::HookOnAreaTrigger(Player* plr, uint32_t id)
135{
136 int32_t buffslot = -1;
137 switch (id)
138 {
139 case AREATRIGGER_A_SPEED: // Speed
140 buffslot = 0;
141 break;
142 case AREATRIGGER_H_SPEED: // Speed (Horde)
143 buffslot = 1;
144 break;
145 case AREATRIGGER_A_RESTORATION: // Restoration
146 buffslot = 2;
147 break;
148 case AREATRIGGER_H_RESTORATION: // Restoration (Horde)
149 buffslot = 3;
150 break;
151 case AREATRIGGER_A_BERSERKING: // Berserking
152 buffslot = 4;
153 break;
154 case AREATRIGGER_H_BERSERKING: // Berserking (Horde)
155 buffslot = 5;
156 break;
157 case AREATRIGGER_WSG_ENCOUNTER_01:
158 case AREATRIGGER_WSG_ENCOUNTER_02:
159 case AREATRIGGER_WSG_ENCOUNTER_03:
160 case AREATRIGGER_WSG_ENCOUNTER_04:
161 case AREATRIGGER_WSG_A_SPAWN:
162 case AREATRIGGER_WSG_H_SPAWN:
163 break;
164 default:
165 DLLLogDetail("Encountered unhandled areatrigger id %u", id);
166 return;
167 break;
168 }
169
170 if (buffslot >= 0)
171 {
172 if (m_buffs[buffslot] != nullptr && m_buffs[buffslot]->IsInWorld())
173 {
174 // apply the buff
175 SpellInfo const* sp = sSpellMgr.getSpellInfo(m_buffs[buffslot]->GetGameObjectProperties()->raw.parameter_3);
176 Spell* s = sSpellMgr.newSpell(plr, sp, true, 0);
177 SpellCastTargets targets(plr->getGuid());
178 s->prepare(&targets);
179
180 // despawn the gameobject (not delete!)
181 m_buffs[buffslot]->despawn(0, BUFF_RESPAWN_TIME);
182 }
183 return;
184 }
185
186 if (((id == AREATRIGGER_WSG_A_SPAWN && plr->isTeamAlliance()) || (id == AREATRIGGER_WSG_H_SPAWN && plr->isTeamHorde())) && (plr->hasBgFlag() && m_flagHolders[plr->getTeam()] == plr->getGuidLow()))
187 {
188 if (m_flagHolders[plr->isTeamHorde() ? TEAM_ALLIANCE : TEAM_HORDE] != 0 || m_dropFlags[plr->isTeamHorde() ? TEAM_ALLIANCE : TEAM_HORDE]->IsInWorld())
189 {
190 // can't cap while flag dropped
191 return;
192 }
193 float distance = plr->isTeamAlliance() ? plr->CalcDistance(1540.29f, 1481.34f, 352.64f) : plr->CalcDistance(915.367f, 1433.78f, 346.089f);
194 if (distance > 50.0f)
195 {
196 //50 yards from the spawn, gtfo hacker.
197 sCheatLog.writefromsession(plr->getSession(), "Tried to capture the flag in WSG while being more then 50 yards away. (%f yards)", plr->CalcDistance(915.367f, 1433.78f, 346.089f));
198 plr->getSession()->Disconnect();
199 return;
200 }
201
202 // remove the bool from the player so the flag doesn't drop
203 m_flagHolders[plr->getTeam()] = 0;
204 plr->setHasBgFlag(false);
205
206 // remove flag aura from player
207 plr->removeAllAurasById(23333 + (plr->getTeam() * 2));
208
209 // capture flag points
210 plr->m_bgScore.MiscData[BattlegroundDef::WSG_FLAGS_CAPTURED]++;
211
212 playSoundToAll(plr->isTeamHorde() ? BattlegroundDef::HORDE_SCORES : BattlegroundDef::ALLIANCE_SCORES);
213
214 if (plr->isTeamHorde())
215 sendChatMessage(CHAT_MSG_BG_EVENT_HORDE, plr->getGuid(), "%s captured the Alliance flag!", plr->getName().c_str());
216 else
217 sendChatMessage(CHAT_MSG_BG_EVENT_ALLIANCE, plr->getGuid(), "%s captured the Horde flag!", plr->getName().c_str());
218
219 setWorldState(plr->isTeamHorde() ? WORLDSTATE_WSG_ALLIANCE_FLAG_DISPLAY : WORLDSTATE_WSG_HORDE_FLAG_DISPLAY, 1);
220
221 // Remove the Other Flag
222 if (m_homeFlags[plr->isTeamHorde() ? TEAM_ALLIANCE : TEAM_HORDE]->IsInWorld())
223 m_homeFlags[plr->isTeamHorde() ? TEAM_ALLIANCE : TEAM_HORDE]->RemoveFromWorld(false);
224
225 // Add the Event to respawn the Flags
226 sEventMgr.AddEvent(this, &WarsongGulch::EventReturnFlags, EVENT_BATTLEGROUND_WSG_AUTO_RETURN_FLAG, 20000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
227
228 // give each player on that team bonus honor and reputation
229 uint32_t honorToAdd = 2 * m_honorPerKill;
230 uint32_t repToAdd = m_isWeekend ? 45 : 35;
231 uint32_t fact = plr->isTeamHorde() ? 889 : 890; //Warsong Outriders : Sliverwing Sentinels
232 for (std::set<Player*>::iterator itr = m_players[plr->getTeam()].begin(); itr != m_players[plr->getTeam()].end(); ++itr)
233 {
234 (*itr)->m_bgScore.BonusHonor += honorToAdd;
235 HonorHandler::AddHonorPointsToPlayer((*itr), honorToAdd);
236 plr->modFactionStanding(fact, repToAdd);
237 }
238
239 m_scores[plr->getTeam()]++;
240 if (m_scores[plr->getTeam()] == 3)
241 {
242 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_CLOSE);
243 sEventMgr.AddEvent(static_cast<Battleground*>(this), &Battleground::close, EVENT_BATTLEGROUND_CLOSE, 120000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
244
245 this->endBattleground(plr->getTeam());
246 }
247
248 // increment the score world state
249 setWorldState(plr->isTeamHorde() ? WORLDSTATE_WSG_HORDE_SCORE : WORLDSTATE_WSG_ALLIANCE_SCORE, m_scores[plr->getTeam()]);
250
251 updatePvPData();
252 }
253}

295void WarsongGulch::HookFlagDrop(Player* plr, GameObject* obj)
296{
297 // picking up a dropped flag
298 if (m_dropFlags[plr->getTeam()] != obj)
299 {
300 // are we returning it?
301 if ((obj->getEntry() == SILVERWING_FLAG && plr->isTeamAlliance()) || (obj->getEntry() == WARSONG_FLAG && plr->isTeamHorde()))
302 {
303 uint32_t x = plr->getTeam() ? TEAM_ALLIANCE : TEAM_HORDE;
304 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_WSG_AUTO_RETURN_FLAG + (plr->isTeamHorde() ? TEAM_ALLIANCE : TEAM_HORDE));
305
306 if (m_dropFlags[x]->IsInWorld())
307 m_dropFlags[x]->RemoveFromWorld(false);
308
309 if (m_homeFlags[x]->IsInWorld() == false)
310 m_homeFlags[x]->PushToWorld(m_mapMgr);
311
312 plr->m_bgScore.MiscData[BattlegroundDef::WSG_FLAGS_RETURNED]++;
313 updatePvPData();
314
315 playSoundToAll(BattlegroundDef::FLAG_RETURNED);
316
317 if (plr->isTeamHorde())
318 sendChatMessage(CHAT_MSG_BG_EVENT_HORDE, plr->getGuid(), "The Horde flag was returned to its base by %s!", plr->getName().c_str());
319 else
320 sendChatMessage(CHAT_MSG_BG_EVENT_ALLIANCE, plr->getGuid(), "The Alliance flag was returned to its base by %s!", plr->getName().c_str());
321
322 setWorldState(plr->isTeamHorde() ? WORLDSTATE_WSG_ALLIANCE_FLAG_DISPLAY : WORLDSTATE_WSG_HORDE_FLAG_DISPLAY, 1);
323 }
324 return;
325 }
326
327 std::map<uint32_t, Standing>::iterator itr = plr->m_forcedReactions.find(1059);
328 if (itr != plr->m_forcedReactions.end())
329 {
330 return;
331 }
332
333 if (plr->isTeamAlliance())
334 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_WSG_AUTO_RETURN_FLAG);
335 else
336 sEventMgr.RemoveEvents(this, EVENT_BATTLEGROUND_WSG_AUTO_RETURN_FLAG + 1);
337
338 if (m_dropFlags[plr->getTeam()]->IsInWorld())
339 m_dropFlags[plr->getTeam()]->RemoveFromWorld(false);
340
341 m_flagHolders[plr->getTeam()] = plr->getGuidLow();
342 plr->setHasBgFlag(true);
343
344 /* This is *really* strange. Even though the A9 create sent to the client is exactly the same as the first one, if
345 * you spawn and despawn it, then spawn it again it will not show. So we'll assign it a new guid, hopefully that
346 * will work.
347 * - Burlex
348 */
349 m_dropFlags[plr->getTeam()]->SetNewGuid(m_mapMgr->generateGameobjectGuid());
350
351 SpellInfo const* pSp = sSpellMgr.getSpellInfo(23333 + (plr->getTeam() * 2));
352 Spell* sp = sSpellMgr.newSpell(plr, pSp, true, 0);
353 SpellCastTargets targets(plr->getGuid());
354 sp->prepare(&targets);
355 setWorldState(plr->isTeamHorde() ? WORLDSTATE_WSG_ALLIANCE_FLAG_DISPLAY : WORLDSTATE_WSG_HORDE_FLAG_DISPLAY, 2);
356 playSoundToAll(plr->isTeamHorde() ? BattlegroundDef::HORDE_CAPTURE : BattlegroundDef::ALLIANCE_CAPTURE);
357
358 if (plr->isTeamHorde())
359 sendChatMessage(CHAT_MSG_BG_EVENT_HORDE, plr->getGuid(), "The Alliance's flag has been taken by %s !", plr->getName().c_str());
360 else
361 sendChatMessage(CHAT_MSG_BG_EVENT_ALLIANCE, plr->getGuid(), "The Horde's flag has been taken by %s !", plr->getName().c_str());
362}
