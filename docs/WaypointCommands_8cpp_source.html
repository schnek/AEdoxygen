17bool ChatCommandHandler::HandleWayPointAddCommand(const char* args, WorldSession* m_session)
18{
19 AIInterface* ai = nullptr;
20 Creature* creature_target = nullptr;
21 Player* player = m_session->GetPlayer();
22
23 if (player->m_aiInterfaceWaypoint != nullptr)
24 {
25 systemMessage(m_session, "Using Previous Unit.");
26 ai = player->m_aiInterfaceWaypoint;
27 if (!ai)
28 {
29 systemMessage(m_session, "Invalid Creature, please select another one.");
30 return true;
31 }
32
33 creature_target = static_cast<Creature*>(ai->getUnit());
34 if (creature_target == nullptr || creature_target->isPet())
35 {
36 systemMessage(m_session, "Invalid Creature, please select another one.");
37 return true;
38 }
39 }
40 else
41 {
42 creature_target = GetSelectedCreature(m_session, true);
43 if (creature_target == nullptr)
44 return true;
45
46 ai = creature_target->getAIInterface();
47 }
48
49 uint32_t pathId = creature_target->getWaypointPath();
50
51 char* pWaitTime = strtok((char*)args, " ");
52 uint32_t WaitTime = (pWaitTime) ? atoi(pWaitTime) : 10000;
53
54 char* pFlags = strtok(nullptr, " ");
55 uint32_t Flags = (pFlags) ? atoi(pFlags) : 0;
56
57 bool showing = ai->isShowWayPointsActive();
58
59 if (!pathId)
60 {
61 pathId = sWaypointMgr->generateWaypointPathId();
62 creature_target->loadPath(pathId);
63 sLogger.debug("New Waypoint Path {} Startet for Creature {}.", pathId, creature_target->getSpawnId());
64
65 // Start Movement
66 creature_target->setDefaultMovementType(WAYPOINT_MOTION_TYPE);
67 creature_target->getMovementManager()->movePath(pathId, true);
68
69 WorldDatabase.Execute("UPDATE creature_spawns SET movetype = %u, waypoint_group = %u WHERE id = %u AND min_build <= %u AND max_build >= %u", WAYPOINT_MOTION_TYPE, pathId, creature_target->spawnid, VERSION_STRING, VERSION_STRING);
70 }
71
72 WaypointNode waypoint;
73 waypoint.id = uint32_t(ai->getWayPointsCount() + 1);
74 waypoint.x = player->GetPositionX();
75 waypoint.y = player->GetPositionY();
76 waypoint.z = player->GetPositionZ();
77 waypoint.orientation = player->GetOrientation();
78 waypoint.moveType = Flags;
79
80 if (waypoint.moveType >= WAYPOINT_MOVE_TYPE_MAX)
81 {
82 sLogger.failure("Waypoint {} has invalid move_type, setting default", waypoint.id);
83 waypoint.moveType = WAYPOINT_MOVE_TYPE_WALK;
84 }
85
86 waypoint.delay = WaitTime;
87 waypoint.eventId = 0;
88 waypoint.eventChance = 0;
89
90 if (showing)
91 ai->hideWayPoints(player);
92
93 // Save Our New Waypoint
94 sWaypointMgr->addWayPoint(pathId, waypoint, true);
95 systemMessage(m_session, "Waypoint {} added to Creature {}.", waypoint.id, creature_target->GetCreatureProperties()->Name);
96
97 if (showing)
98 ai->activateShowWayPoints(player, false);
99
100 return true;
101}

104bool ChatCommandHandler::HandleWayPointDeleteCommand(const char* /*args*/, WorldSession* m_session)
105{
106 uint32_t wpid = GetSelectedWayPointId(m_session);
107 Player* player = m_session->GetPlayer();
108 AIInterface* ai = player->m_aiInterfaceWaypoint;
109 if (ai == nullptr || !ai->getUnit())
110 {
111 systemMessage(m_session, "Invalid Creature, please select another one.");
112 return true;
113 }
114
115 if (wpid)
116 {
117 bool show = ai->isShowWayPointsActive();
118 if (show == true)
119 ai->hideWayPoints(player);
120
121 sWaypointMgr->deleteWayPointById(ai->getUnit()->ToCreature()->getWaypointPath(), wpid);
122
123 systemMessage(m_session, "Waypoint {} deleted.", wpid);
124 }
125 else
126 {
127 systemMessage(m_session, "Invalid Waypoint.");
128 }
129 return true;
130}

176//.waypoint show
177bool ChatCommandHandler::HandleWayPointShowCommand(const char* args, WorldSession* m_session)
178{
179 auto creature_target = GetSelectedCreature(m_session, true);
180 if (creature_target == nullptr)
181 return true;
182
183 char* pBackwards = strtok((char*)args, " ");
184 bool Backwards = (pBackwards) ? ((atoi(pBackwards) > 0) ? true : false) : false;
185
186 AIInterface* ai = creature_target->getAIInterface();
187 Player* player = m_session->GetPlayer();
188 if (player->m_aiInterfaceWaypoint != ai)
189 {
190 if (ai->isShowWayPointsActive() == true)
191 {
192 redSystemMessage(m_session, "Some one else is also Viewing this Creatures WayPoints.");
193 redSystemMessage(m_session, "Viewing WayPoints at the same time as some one else can cause undesireble results.");
194 return true;
195 }
196
197 if (player->m_aiInterfaceWaypoint != nullptr)
198 player->m_aiInterfaceWaypoint->hideWayPoints(player);
199
200 player->m_aiInterfaceWaypoint = ai;
201 ai->activateShowWayPoints(player, Backwards);
202 }
203 else
204 {
205 if (ai->isShowWayPointsActive() == true)
206 systemMessage(m_session, "Waypoints Already Showing.");
207 else
208 ai->activateShowWayPoints(m_session->GetPlayer(), Backwards);
209 }
210
211 systemMessage(m_session, "Showing waypoints for creature {}", creature_target->GetSQL_id());
212 return true;
213}
