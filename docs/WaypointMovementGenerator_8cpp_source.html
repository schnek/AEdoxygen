44{
45 if (timer)
46 {

49 return;
50

122bool WaypointMovementGenerator<Creature>::doUpdate(Creature* owner, uint32_t diff)
123{
124 if (!owner || !owner->isAlive())
125 return true;
126
127 if (hasFlag(MOVEMENTGENERATOR_FLAG_FINALIZED | MOVEMENTGENERATOR_FLAG_PAUSED) || !_path || _path->nodes.empty())
128 return true;
129
130 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE | UNIT_STATE_LOST_CONTROL) || owner->isCastingSpell())
131 {
132 addFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
133 owner->stopMoving();
134 return true;
135 }
136
137 if (hasFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED))
138 {
139 if (hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED) && (_nextMoveTime->isTimePassed() || !hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED)))
140 {
141 startMove(owner, true);
142 return true;
143 }
144
145 removeFlag(MOVEMENTGENERATOR_FLAG_INTERRUPTED);
146 }
147
148 // if it's moving
149 if (!owner->movespline->Finalized())
150 {
151 // set home position at place (every MotionMaster::UpdateMotion)
152#if VERSION_STRING <= WotLK
153 if (!owner->hasUnitMovementFlag(MOVEFLAG_TRANSPORT) || owner->getTransGuid())
154 owner->SetSpawnLocation(owner->GetPosition());
155#else
156 if (owner->getTransGuid())
157 owner->SetSpawnLocation(owner->GetPosition());
158#endif
159
160 // relaunch movement if its speed has changed
161 if (hasFlag(MOVEMENTGENERATOR_FLAG_SPEED_UPDATE_PENDING))
162 startMove(owner, true);
163 }
164 else if (!_nextMoveTime->isTimePassed()) // it's not moving, is there a timer?
165 {
166 if (updateTimer(diff))
167 {
168 if (!hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED)) // initial movement call
169 {
170 startMove(owner);
171 return true;
172 }
173 else if (!hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED)) // timer set before node was reached, resume now
174 {
175 startMove(owner, true);
176 return true;
177 }
178 }
179 else
180 return true; // keep waiting
181 }
182 else // not moving, no timer
183 {
184 if (hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED) && !hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED))
185 {
186 onArrived(owner); // hooks and wait timer reset (if necessary)
187 addFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED); // signals to future startMove that it reached a node
188 }
189
190 if (_nextMoveTime->isTimePassed()) // OnArrived might have set a timer
191 startMove(owner); // check path status, get next point and move if necessary & can
192 }
193
194 return true;
195}

221void WaypointMovementGenerator<Creature>::onArrived(Creature* owner)
222{
223 if (!_path || _path->nodes.empty())
224 return;
225
226 ASSERT(_currentNode < _path->nodes.size() && "WaypointMovementGenerator::OnArrived: tried to reference a node id which is not included in path");
227 WaypointNode const &waypoint = _path->nodes.at(_currentNode);
228 if (waypoint.delay)
229 {
230 owner->removeUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
231 _nextMoveTime->resetInterval(waypoint.delay);
232 }
233
234 if (waypoint.eventId && Util::getRandomUInt(0, 99) < waypoint.eventChance)
235 {
236 sLogger.debug("Creature movement start script {} at point {} for {}.", waypoint.eventId, _currentNode, owner->getGuid());
237 owner->removeUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
238 // add waypoint scripts
239 }
240
241 // inform AI
242 if (AIInterface* AI = owner->getAIInterface())
243 {
244 AI->movementInform(WAYPOINT_MOTION_TYPE, _currentNode);
245 AI->waypointReached(waypoint.id, _path->id);
246 }
247
248 owner->updateCurrentWaypointInfo(waypoint.id, _path->id);
249}

251void WaypointMovementGenerator<Creature>::startMove(Creature* owner, bool relaunch/* = false*/)
252{
253 // sanity checks
254 if (!owner || !owner->isAlive() || hasFlag(MOVEMENTGENERATOR_FLAG_FINALIZED) || !_path || _path->nodes.empty() || (relaunch && (hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED) || !hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED))))
255 return;
256
257 if (owner->hasUnitStateFlag(UNIT_STATE_NOT_MOVE) || owner->isCastingSpell() || (owner->isFormationLeader() && !owner->isFormationLeaderMoveAllowed())) // if cannot move OR cannot move because of formation
258 {
259 _nextMoveTime->resetInterval(1000); // delay 1s
260 return;
261 }
262
263#if VERSION_STRING <= WotLK
264 bool const transportPath = owner->hasUnitMovementFlag(MOVEFLAG_TRANSPORT) && owner->getTransGuid();
265#else
266 bool const transportPath = owner->GetTransport() != nullptr;
267#endif
268
269 if (hasFlag(MOVEMENTGENERATOR_FLAG_INFORM_ENABLED) && hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED))
270 {
271 if (computeNextNode())
272 {
273 ASSERT(_currentNode < _path->nodes.size() && "WaypointMovementGenerator::startMove: tried to reference a node id which is not included in path");
274
275 // inform AI
276 if (AIInterface* AI = owner->getAIInterface())
277 AI->waypointStarted(_path->nodes[_currentNode].id, _path->id);
278 }
279 else
280 {
281 WaypointNode const &waypoint = _path->nodes[_currentNode];
282 float x = waypoint.x;
283 float y = waypoint.y;
284 float z = waypoint.z;
285 float o = owner->GetOrientation();
286
287 if (!transportPath)
288 owner->SetSpawnLocation(x, y, z, o);
289 else
290 {
291 if (Transporter* trans = owner->GetTransport())
292 {
293 o -= trans->GetOrientation();
294 owner->SetTransportHomePosition(x, y, z, o);
295 trans->calculatePassengerPosition(x, y, z, &o);
296 owner->SetSpawnLocation(x, y, z, o);
297 }
298 // else if (vehicle) - this should never happen, vehicle offsets are const
299 }
300 addFlag(MOVEMENTGENERATOR_FLAG_FINALIZED);
301 owner->updateCurrentWaypointInfo(0, 0);
302
303 // inform AI
304 if (AIInterface* AI = owner->getAIInterface())
305 AI->waypointPathEnded(waypoint.id, _path->id);
306 return;
307 }
308 }
309 else if (!hasFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED))
310 {
311 addFlag(MOVEMENTGENERATOR_FLAG_INITIALIZED);
312
313 // inform AI
314 if (AIInterface* AI = owner->getAIInterface())
315 AI->waypointStarted(_path->nodes[_currentNode].id, _path->id);
316 }
317
318 ASSERT(_currentNode < _path->nodes.size() && "WaypointMovementGenerator::startMove: tried to reference a node id which is not included in path");
319 WaypointNode const &waypoint = _path->nodes[_currentNode];
320
321 removeFlag(MOVEMENTGENERATOR_FLAG_TRANSITORY | MOVEMENTGENERATOR_FLAG_INFORM_ENABLED | MOVEMENTGENERATOR_FLAG_TIMED_PAUSED);
322
323 owner->addUnitStateFlag(UNIT_STATE_ROAMING_MOVE);
324
325 MovementMgr::MoveSplineInit init(owner);
326
327 //! If creature is on transport, we assume waypoints set in DB are already transport offsets
328 if (transportPath)
329 init.DisableTransportPathTransformations();
330
331 //! Do not use formationDest here, MoveTo requires transport offsets due to DisableTransportPathTransformations() call
332 //! but formationDest contains global coordinates
333 init.MoveTo(waypoint.x, waypoint.y, waypoint.z);
334
335 //! Accepts angles such as 0.00001 and -0.00001, 0 must be ignored, default value in waypoint table
336 if (waypoint.orientation && waypoint.delay)
337 init.SetFacing(waypoint.orientation);
338
339 switch (waypoint.moveType)
340 {
341#if VERSION_STRING >= WotLK
342 case WAYPOINT_MOVE_TYPE_LAND:
343 init.SetAnimation(ANIMATION_FLAG_GROUND);
344 break;
345 case WAYPOINT_MOVE_TYPE_TAKEOFF:
346 init.SetAnimation(ANIMATION_FLAG_HOVER);
347 break;
348#endif
349 case WAYPOINT_MOVE_TYPE_RUN:
350 init.SetWalk(false);
351 break;
352 case WAYPOINT_MOVE_TYPE_WALK:
353 init.SetWalk(true);
354 break;
355 default:
356 break;
357 }
358
359 init.Launch();
360
361 // inform formation
362 owner->signalFormationMovement();
363}

365bool WaypointMovementGenerator<Creature>::computeNextNode()
366{
367 if ((_currentNode == _path->nodes.size() - 1) && !_repeating)
368 return false;
369
370 _currentNode = (_currentNode + 1) % _path->nodes.size();
371 return true;
