42uint32_t WeatherMgr::getSound(uint32_t effect, float density)
43{
44 if (density <= 0.30f)
45 return WEATHER_NOSOUND;
46
47 uint32_t sound;
48
49 switch (effect)
50 {
51 case WEATHER_TYPE_RAIN:
52 case WEATHER_TYPE_HEAVY_RAIN:
53 {
54 if (density < 0.40f)
55 sound = WEATHER_RAINLIGHT;
56 else if (density < 0.70f)
57 sound = WEATHER_RAINMEDIUM;
58 else
59 sound = WEATHER_RAINHEAVY;
60 } break;
61 case WEATHER_TYPE_SNOW:
62 {
63 if (density < 0.40f)
64 sound = WEATHER_SNOWLIGHT;
65 else if (density < 0.70f)
66 sound = WEATHER_SNOWMEDIUM;
67 else
68 sound = WEATHER_SNOWHEAVY;
69 } break;
70 case WEATHER_TYPE_SANDSTORM:
71 {
72 if (density < 0.40f)
73 sound = WEATHER_SANDSTORMLIGHT;
74 else if (density < 0.70f)
75 sound = WEATHER_SANDSTORMMEDIUM;
76 else
77 sound = WEATHER_SANDSTORMHEAVY;
78 } break;
79 default:
80 sound = WEATHER_NOSOUND;
81 break;
82 }
83 return sound;
84}

97void WeatherMgr::loadFromDB()
98{
99 sLogger.info("Loading Weather...");
100 auto result = WorldDatabase.Query("SELECT zoneId,high_chance,high_type,med_chance,med_type,low_chance,low_type FROM weather");
101 if (!result)
102 return;
103
104 do
105 {
106 Field* fields = result->Fetch();
107 auto weatherInfo = std::make_unique<WeatherInfo>();
108 weatherInfo->m_zoneId = fields[0].asUint32();
109 weatherInfo->m_effectValues[0] = fields[1].asUint32(); // high_chance
110 weatherInfo->m_effectValues[1] = fields[2].asUint32(); // high_type
111 weatherInfo->m_effectValues[2] = fields[3].asUint32(); // med_chance
112 weatherInfo->m_effectValues[3] = fields[4].asUint32(); // med_type
113 weatherInfo->m_effectValues[4] = fields[5].asUint32(); // low_chance
114 weatherInfo->m_effectValues[5] = fields[6].asUint32(); // low_type
115 const auto [itr, _] = m_zoneWeathers.try_emplace(fields[0].asUint32(), std::move(weatherInfo));
116
117 itr->second->_generateWeather();
118 }
119 while (result->NextRow());
120 sLogger.info("WeatherMgr : Loaded weather information for {} zones.", result->GetRowCount());
121}

167void WeatherInfo::_generateWeather()
168{
169 m_currentTime = 0;
170 m_currentEffect = 0;
171 m_currentDensity = 0.30f;
172
173 const float randomFloat = Util::getRandomFloat(1);
174 m_maxDensity = randomFloat + 1;
175 m_totalTime = (Util::getRandomUInt(11) + 5) * 1000 * 120;
176
177 const uint32_t randomUInt = Util::getRandomUInt(100);
178
179 if (randomUInt <= m_effectValues[4]) // low chance
180 m_currentEffect = m_effectValues[5]; // low type
181 else if (randomUInt <= m_effectValues[2]) // med chance
182 m_currentEffect = m_effectValues[3]; // med type
183 else if (randomUInt <= m_effectValues[0]) // high chance
184 m_currentEffect = m_effectValues[1]; // high type
185
186 sendUpdate();
187
188 sEventMgr.AddEvent(this, &WeatherInfo::buildUp, EVENT_WEATHER_UPDATE, static_cast<uint32_t>(m_totalTime / ceil(m_maxDensity / m_densityUpdate) * 2), 0, 0);
189 sLogger.debugFlag(AscEmu::Logging::LF_MAP, "Forecast for zone:{} new type:{} new interval:{} ms", m_zoneId, m_currentEffect, static_cast<uint32_t>(m_totalTime / ceil(m_maxDensity / m_densityUpdate) * 2));
190}

192void WeatherInfo::buildUp()
193{
194 if (m_currentDensity >= 0.50f)
195 {
196 sEventMgr.RemoveEvents(this, EVENT_WEATHER_UPDATE);
197 sEventMgr.AddEvent(this, &WeatherInfo::update, EVENT_WEATHER_UPDATE, static_cast<uint32_t>(m_totalTime / ceil(m_maxDensity / m_densityUpdate) * 4), 0, 0);
198 }
199 else
200 {
201 m_currentDensity += m_densityUpdate;
202 sendUpdate();
203 }
204}

206void WeatherInfo::update()
207{
208 if (m_currentEffect == 0 || Util::getRandomUInt(100) < 66)
209 {
210 m_currentDensity -= m_densityUpdate;
211 if (m_currentDensity < 0.30f)
212 {
213 m_currentDensity = 0.0f;
214 m_currentEffect = 0;
215 sEventMgr.RemoveEvents(this, EVENT_WEATHER_UPDATE);
216 _generateWeather();
217 return;
218 }
219 }
220 else
221 {
222 m_currentDensity += m_densityUpdate;
223 if (m_currentDensity >= m_maxDensity)
224 {
225 m_currentDensity = m_maxDensity;
226 return;
227 }
228 }
229 sendUpdate();
230}
