33/* Parse a command line buffer into arguments */
34static int ParseCommandLine(char *cmdline, char **argv) {
35 char *bufp;
36 int argc;
37
38 argc = 0;
39 for (bufp = cmdline; *bufp;) {
40 /* Skip leading whitespace */
41 while (isspace(*bufp)) {
42 ++bufp;
43 }
44 /* Skip over argument */
45 if (*bufp == '"') {
46 ++bufp;
47 if (*bufp) {
48 if (argv) {
49 argv[argc] = bufp;
50 }
51 ++argc;
52 }
53 /* Skip over word */
54 while (*bufp && (*bufp != '"')) {
55 ++bufp;
56 }
57 } else {
58 if (*bufp) {
59 if (argv) {
60 argv[argc] = bufp;
61 }
62 ++argc;
63 }
64 /* Skip over word */
65 while (*bufp && !isspace(*bufp)) {
66 ++bufp;
67 }
68 }
69 if (*bufp) {
70 if (argv) {
71 *bufp = '\0';
72 }
73 ++bufp;
74 }
75 }
76 if (argv) {
77 argv[argc] = NULL;
78 }
79 return (argc);
80}
