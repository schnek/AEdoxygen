308 void AppendField(uint8_t field)
309 {
310 ASSERT(!m_compiled)
311 ASSERT(m_fieldcount < BitCount8(guidmask))
312
313 m_guidfields[m_fieldcount++] = field;
314
315 if (m_fieldcount == BitCount8(guidmask))
316 _CompileByNew();
317 }

329 void _CompileByOld()
330 {
331 ASSERT(!m_compiled)
332
333 m_fieldcount = 0;
334
335 for (uint8_t x = 0; x < 8; x++)
336 {
337
338 uint8_t p = reinterpret_cast<uint8_t*>(&m_rawGuid)[x];
339
340 if (p)
341 {
342 m_guidfields[m_fieldcount++] = p;
343 guidmask |= 1 << x;
344 }
345 }
346
347 m_compiled = true;
348 }
349
350 void _CompileByNew()
351 {
352 ASSERT(!m_compiled || m_fieldcount == BitCount8(guidmask))
353
354 int j = 0;
355
356 if (guidmask & 0x01) //1
357 {
358 m_rawGuid |= static_cast<uint64_t>(m_guidfields[j]);
359 j++;
360 }
361 if (guidmask & 0x02) //2
362 {
363 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 8);
364 j++;
365 }
366 if (guidmask & 0x04) //4
367 {
368 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 16);
369 j++;
370 }
371 if (guidmask & 0x08) //8
372 {
373 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 24);
374 j++;
375 }
376 if (guidmask & 0x10) //16
377 {
378 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 32);
379 j++;
380 }
381 if (guidmask & 0x20) //32
382 {
383 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 40);
384 j++;
385 }
386 if (guidmask & 0x40) //64
387 {
388 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 48);
389 j++;
390 }
391 if (guidmask & 0x80) //128
392 {
393 m_rawGuid |= (static_cast<uint64_t>(m_guidfields[j]) << 56);
394 j++;
395 }
396
397 m_compiled = true;
398 }
