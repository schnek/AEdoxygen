218void WorldMap::Do()
219{
220 m_threadRunning = true;
221
222 if (!m_terminateThread)
223 {
224 const auto exec_start = Util::getMSTime();
225
226 // Time In Milliseconds ( exact Difftime Since last update Cycle )
227 const uint32_t diffTime = exec_start - m_lastUpdateTime;
228
229 //first push to world new objects
230 {
231 std::scoped_lock<std::mutex> lock(m_objectinsertlock);
232 if (!m_objectinsertpool.empty())
233 {
234 for (const auto& o : m_objectinsertpool)
235 o->PushToWorld(this);
236
237 m_objectinsertpool.clear();
238 }
239 }
240
241 // Update Our Map
242 update(diffTime);
243
244 m_lastUpdateTime = Util::getMSTime();
245 return;
246 }
247
248 m_threadRunning = false;
249
250 // Map is added to MapMgr remove pool which deletes map when thread has finished all its work
251 m_thread->requestKill();
252}

254void WorldMap::update(uint32_t t_diff)
255{
256 const auto msTime = Util::getMSTime();
257
258 // Update any events.
259 // we make update of events before objects so in case there are 0 timediff events they do not get deleted after update but on next server update loop
260 eventHolder.Update(t_diff);
261
262 // Update Dynamic Map
263 _dynamicTree.update(t_diff);
264
265 // Update Transporters
266 auto diffTime = msTime - m_lastTransportUpdateTimer;
267 if (diffTime >= 100)
268 {
269 {
270 std::scoped_lock<std::mutex> guard(m_transportsLock);
271 for (auto itr = m_TransportStorage.cbegin(); itr != m_TransportStorage.cend();)
272 {
273 Transporter* trans = *itr;
274 ++itr;
275
276 if (!trans || !trans->IsInWorld())
277 continue;
278
279 trans->Update(diffTime);
280 }
281 }
282
283 {
284 std::scoped_lock<std::mutex> guard(m_delayedTransportLock);
285 for (auto itr = m_TransportDelayedRemoveStorage.cbegin(); itr != m_TransportDelayedRemoveStorage.cend();)
286 {
287 auto* trans = itr->first;
288 if (!trans || !trans->IsInWorld())
289 {
290 itr = m_TransportDelayedRemoveStorage.erase(itr);
291 continue;
292 }
293
294 if (itr->second)
295 {
296 removeFromMapMgr(trans);
297 trans->RemoveFromWorld(true);
298 }
299 else
300 {
301 trans->delayedTeleportTransport();
302 }
303
304 itr = m_TransportDelayedRemoveStorage.erase(itr);
305 }
306 }
307
308 m_lastTransportUpdateTimer = msTime;
309 }
310
311 // Update Creatures
312 {
313 auto diffTime = msTime - m_lastCreatureUpdateTimer;
314 creature_iterator = activeCreatures.begin();
315 for (; creature_iterator != activeCreatures.end();)
316 {
317 Creature* ptr = *creature_iterator;
318 ++creature_iterator;
319 ptr->Update(diffTime);
320 }
321
322 m_lastCreatureUpdateTimer = msTime;
323 }
324
325 // Update Pets
326 {
327 const auto diffTime = msTime - m_lastPetUpdateTimer;
328 for (auto itr = m_PetStorage.cbegin(); itr != m_PetStorage.cend();)
329 {
330 Pet* ptr = itr->second;
331 ++itr;
332 ptr->Update(diffTime);
333 }
334
335 m_lastPetUpdateTimer = msTime;
336 }
337
338 // Update Players
339 {
340 auto diffTime = msTime - m_lastPlayerUpdateTimer;
341 for (auto itr = m_PlayerStorage.cbegin(); itr != m_PlayerStorage.cend();)
342 {
343 Player* ptr = itr->second;
344 ++itr;
345 ptr->Update(diffTime);
346 }
347
348 m_lastPlayerUpdateTimer = msTime;
349 }
350
351 // Dynamic objects are updated every 100ms
352 diffTime = msTime - m_lastDynamicUpdateTimer;
353 if (diffTime >= 100)
354 {
355 for (auto itr = m_DynamicObjectStorage.cbegin(); itr != m_DynamicObjectStorage.cend();)
356 {
357 DynamicObject* o = itr->second;
358 ++itr;
359 o->updateTargets();
360 }
361
362 m_lastDynamicUpdateTimer = msTime;
363 }
364
365 // Update gameobjects only every 200ms
366 diffTime = msTime - m_lastGameObjectUpdateTimer;
367 if (diffTime >= 200)
368 {
369 gameObject_iterator = activeGameObjects.begin();
370 for (; gameObject_iterator != activeGameObjects.end();)
371 {
372 GameObject* gameobject = *gameObject_iterator;
373 ++gameObject_iterator;
374 if (gameobject != nullptr)
375 gameobject->Update(diffTime);
376 }
377
378 m_lastGameObjectUpdateTimer = msTime;
379 }
380
381 // Update Sessions every 100ms
382 diffTime = msTime - m_lastSessionUpdateTimer;
383 if (diffTime >= 100)
384 {
385 for (auto itr = Sessions.cbegin(); itr != Sessions.cend();)
386 {
387 WorldSession* session = (*itr);
388 auto it2 = itr;
389 ++itr;
390
391 if (session->GetInstance() != getInstanceId())
392 {
393 Sessions.erase(it2);
394 continue;
395 }
396
397 // Don't update players not on our map.
398 // If we abort in the handler, it means we will "lose" packets, or not process this.
399 // .. and that could be disastrous to our client :P
400 if (session->GetPlayer() && (session->GetPlayer()->getWorldMap() != this && session->GetPlayer()->getWorldMap() != nullptr))
401 continue;
402
403 uint8_t result;
404
405 if ((result = session->Update(getInstanceId())) != 0)
406 {
407 if (result == 1)
408 {
409 // complete deletion
410 sWorld.deleteSession(session);
411 }
412 Sessions.erase(it2);
413 }
414 }
415
416 m_lastSessionUpdateTimer = msTime;
417 }
418
419 /// Update Respawns every 1000ms
420 diffTime = msTime - m_lastRespawnUpdateTimer;
421 if (diffTime >= 1000)
422 {
423 processRespawns();
424
425 // Time In Seconds
426 const auto now = std::chrono::system_clock::to_time_t(std::chrono::system_clock::now());
427
428 while (!_corpseDespawnTimes.empty())
429 {
430 CorpseInfo next = _corpseDespawnTimes.top();
431 if (now < next.time)
432 break;
433
434 _corpseDespawnTimes.pop();
435 if (auto* const pCorpse = sObjectMgr.getCorpseByGuid(static_cast<uint32_t>(next.guid)))
436 {
437 if (pCorpse->getWorldMap() != this)
438 break;
439
440 pCorpse->despawn();
441 }
442 break;
443 }
444
445 m_lastRespawnUpdateTimer = msTime;
446 }
447
448 // Finally, A9 Building/Distribution
449 updateObjects();
450}

452void WorldMap::processRespawns()
453{
454 const auto now = Util::getTimeNow();
455
456 while (!_respawnTimes.empty())
457 {
458 RespawnInfo* next = _respawnTimes.top().get();
459 if (now < next->time) // done for this tick
460 break;
461
462 if (checkRespawn(next)) // see if we're allowed to respawn
463 {
464 // ok, respawn
465 getRespawnMapForType(next->type).erase(next->spawnId);
466 doRespawn(next->type, next->obj, next->spawnId, next->cellX, next->cellY);
467 _respawnTimes.pop();
468 }
469 else if (!next->time) // just remove respawn entry without rescheduling
470 {
471 getRespawnMapForType(next->type).erase(next->spawnId);
472 _respawnTimes.pop();
473 }
474 else
475 {
476 saveRespawnDB(*next);
477 }
478 }

570void WorldMap::PushObject(Object* obj)
571{
572 if (obj != nullptr)
573 {
574 //\todo That object types are not map objects. TODO: add AI groups here?
575 if (obj->isItem() || obj->isContainer())
576 {
577 // mark object as updatable and exit
578 return;
579 }
580
581 //Zyres: this was an old ASSERT MapMgr for map x is not allowed to push objects for mapId z
582 if (obj->GetMapId() != getBaseMap()->getMapId())
583 {
584 sLogger.failure("WorldMap::PushObject manager for mapId {} tried to push object for mapId {}, return!", getBaseMap()->getMapId(), obj->GetMapId());
585 return;
586 }
587
588 if (obj->GetPositionY() > Map::Terrain::_maxY || obj->GetPositionY() < Map::Terrain::_minY)
589 {
590 sLogger.failure("WorldMap::PushObject not allowed to push object to y: {} (max {}/min {}), return!", obj->GetPositionY(), Map::Terrain::_maxY, Map::Terrain::_minY);
591 return;
592 }
593
594 if (_cells == nullptr)
595 {
596 sLogger.failure("WorldMap::PushObject not allowed to push object to invalid cell (nullptr), return!");
597 return;
598 }
599
600 if (obj->isCorpse())
601 {
602 m_corpses.insert(static_cast<Corpse*>(obj));
603 }
604
605 obj->clearInRangeSets();
606
607 // Check valid cell x/y values
608 if (!(obj->GetPositionX() < Map::Terrain::_maxX && obj->GetPositionX() > Map::Terrain::_minX) || !(obj->GetPositionY() < Map::Terrain::_maxY && obj->GetPositionY() > Map::Terrain::_minY))
609 {
610 outOfMapBoundariesTeleport(obj);
611 }
612
613 // Get cell coordinates
614 uint32_t x = getPosX(obj->GetPositionX());
615 uint32_t y = getPosY(obj->GetPositionY());
616
617 if (x >= Map::Cell::_sizeX || y >= Map::Cell::_sizeY)
618 {
619 outOfMapBoundariesTeleport(obj);
620
621 x = getPosX(obj->GetPositionX());
622 y = getPosY(obj->GetPositionY());
623 }
624
625 MapCell* objCell = getCell(x, y);
626 if (objCell == nullptr)
627 {
628 objCell = create(x, y);
629 if (objCell != nullptr)
630 {
631 objCell->init(x, y, this);
632 }
633 else
634 {
635 sLogger.fatal("MapCell for x f% and y f% seems to be invalid!", x, y);
636 return;
637 }
638 }
639
640 // Build update-block for player
641 std::unique_ptr<ByteBuffer> buf;
642 uint32_t count;
643 Player* plObj = nullptr;
644
645 if (obj->isPlayer())
646 {
647 plObj = static_cast<Player*>(obj);
648
649 sLogger.debug("Creating player {} for himself.", std::to_string(obj->getGuid()));
650 ByteBuffer pbuf(10000);
651 count = plObj->buildCreateUpdateBlockForPlayer(&pbuf, plObj);
652 plObj->getUpdateMgr().pushCreationData(&pbuf, count);
653 }
654
655 // Build in-range data
656 uint8_t cellNumber = worldConfig.server.mapCellNumber;
657
658 uint32_t endX = (x <= Map::Cell::_sizeX) ? x + cellNumber : (Map::Cell::_sizeX - cellNumber);
659 uint32_t endY = (y <= Map::Cell::_sizeY) ? y + cellNumber : (Map::Cell::_sizeY - cellNumber);
660 uint32_t startX = x > 0 ? x - cellNumber : 0;
661 uint32_t startY = y > 0 ? y - cellNumber : 0;
662
663 for (uint32_t posX = startX; posX <= endX; posX++)
664 {
665 for (uint32_t posY = startY; posY <= endY; posY++)
666 {
667 MapCell* cell = getCell(posX, posY);
668 if (cell)
669 {
670 updateInRangeSet(obj, plObj, cell, buf);
671 }
672 }
673 }
674
675 // Forced Cells
676 for (auto& cell : m_forcedcells)
677 updateInRangeSet(obj, plObj, cell, buf);
678
679 //Add to the cell's object list
680 objCell->addObject(obj);
681
682 // Add Object
683 if (getScript())
684 getScript()->addObject(obj);
685
686 obj->SetMapCell(objCell);
687 //Add to the mapmanager's object list
688 if (plObj != nullptr)
689 {
690 m_PlayerStorage[plObj->getGuidLow()] = plObj;
691 updateCellActivity(x, y, 2U + cellNumber);
692 }
693 else
694 {
695 switch (obj->GetTypeFromGUID())
696 {
697 case HIGHGUID_TYPE_PET:
698 {
699 m_PetStorage[obj->GetUIdFromGUID()] = static_cast<Pet*>(obj);
700 } break;
701 case HIGHGUID_TYPE_UNIT:
702 case HIGHGUID_TYPE_VEHICLE:
703 {
704 if (obj->GetUIdFromGUID() <= m_CreatureHighGuid)
705 {
706 m_CreatureStorage[obj->GetUIdFromGUID()] = static_cast<Creature*>(obj);
707 if (static_cast<Creature*>(obj)->m_spawn != nullptr)
708 {
709 _sqlids_creatures.insert(std::make_pair(static_cast<Creature*>(obj)->m_spawn->id, static_cast<Creature*>(obj)));
710 }
711 }
712 } break;
713 case HIGHGUID_TYPE_GAMEOBJECT:
714 {
715 m_GameObjectStorage[obj->GetUIdFromGUID()] = static_cast<GameObject*>(obj);
716 if (static_cast<GameObject*>(obj)->m_spawn != nullptr)
717 {
718 _sqlids_gameobjects.insert(std::make_pair(static_cast<GameObject*>(obj)->m_spawn->id, static_cast<GameObject*>(obj)));
719 }
720 } break;
721
722 case HIGHGUID_TYPE_DYNAMICOBJECT:
723 {
724 m_DynamicObjectStorage[obj->getGuidLow()] = static_cast<DynamicObject*>(obj);
725 } break;
726 }
727 }
728
729 // Handle activation of that object.
730 if (objCell->isActive() && obj->CanActivate())
731 obj->Activate(this);
732
733 // Add the session to our set if it is a player.
734 if (plObj)
735 {
736 Sessions.insert(plObj->getSession());
737
738 // Change the instance ID, this will cause it to be removed from the world thread (return value 1)
739 plObj->getSession()->SetInstance(getInstanceId());
740
741 // Add the map wide objects
742 if (_mapWideStaticObjects.size())
743 {
744 uint32_t globalcount = 0;
745 if (buf == nullptr)
746 buf = std::make_unique<ByteBuffer>(300);
747
748 for (auto _mapWideStaticObject : _mapWideStaticObjects)
749 {
750 count = _mapWideStaticObject->buildCreateUpdateBlockForPlayer(buf.get(), plObj);
751 globalcount += count;
752 }
753 /*VLack: It seems if we use the same buffer then it is a BAD idea to try and push created data one by one, add them at once!
754 If you try to add them one by one, then as the buffer already contains data, they'll end up repeating some object.
755 Like 6 object updates for Deeprun Tram, but the built package will contain these entries: 2AFD0, 2AFD0, 2AFD1, 2AFD0, 2AFD1, 2AFD2*/
756 if (globalcount > 0)
757 plObj->getUpdateMgr().pushCreationData(buf.get(), globalcount);
758 }
759 }
760 }
761 else
762 {
763 sLogger.failure("WorldMap::PushObject tried to push invalid object (nullptr)!");
764 }
765}

767void WorldMap::PushStaticObject(Object* obj)
768{
769 _mapWideStaticObjects.insert(obj);
770
771 obj->SetInstanceID(getInstanceId());
772 obj->SetMapId(getBaseMap()->getMapId());
773
774 switch (obj->GetTypeFromGUID())
775 {
776 case HIGHGUID_TYPE_UNIT:
777 case HIGHGUID_TYPE_VEHICLE:
778 m_CreatureStorage[obj->GetUIdFromGUID()] = static_cast<Creature*>(obj);
779 break;
780
781 case HIGHGUID_TYPE_GAMEOBJECT:
782 m_GameObjectStorage[obj->GetUIdFromGUID()] = static_cast<GameObject*>(obj);
783 break;
784
785 default:
786 sLogger.debug("WorldMap::PushStaticObject called for invalid type {}.", obj->GetTypeFromGUID());
787 break;
788 }
789
790 if (getScript())

794void WorldMap::RemoveObject(Object* obj, bool free_guid)
795{
796 // Assertions
797 if (obj == nullptr)
798 {
799 sLogger.failure("WorldMap::RemoveObject tried to remove invalid object (nullptr)");
800 return;
801 }
802
803 if (obj->GetMapId() != getBaseMap()->getMapId())
804 {
805 sLogger.failure("WorldMap::RemoveObject tried to remove object with map {} but WorldMap is for map {}!", obj->GetMapId(), getBaseMap()->getMapId());
806 return;
807 }
808
809 if (_cells == nullptr)
810 {
811 sLogger.failure("WorldMap::RemoveObject tried to remove invalid cells (nullptr)");
812 return;
813 }
814
815 // Call Script Object Got Removed
816 if (getScript())
817 getScript()->removeObject(obj);
818
819 if (obj->IsActive())
820 obj->deactivate(this);
821
822 //there is a very small chance that on double player ports on same update player is added to multiple insertpools but not removed
823 //one clear example was the double port proc when exploiting double resurrect
824 {
825 std::scoped_lock<std::mutex> lock(m_objectinsertlock);
826 m_objectinsertpool.erase(obj);
827 }
828
829 {
830 std::scoped_lock<std::mutex> lock(m_updateMutex);
831 _updates.erase(obj);
832 obj->ClearUpdateMask();
833 }
834
835 // Remove object from all needed places
836 switch (obj->GetTypeFromGUID())
837 {
838 case HIGHGUID_TYPE_UNIT:
839 case HIGHGUID_TYPE_VEHICLE:
840 {
841 if (obj->GetUIdFromGUID() <= m_CreatureHighGuid)
842 {
843 m_CreatureStorage[obj->GetUIdFromGUID()] = nullptr;
844
845 if (static_cast<Creature*>(obj)->m_spawn != nullptr)
846 _sqlids_creatures.erase(static_cast<Creature*>(obj)->m_spawn->id);
847
848 if (free_guid)
849 _reusable_guids_creature.push_back(obj->GetUIdFromGUID());
850 }
851 } break;
852 case HIGHGUID_TYPE_PET:
853 {
854 m_PetStorage.erase(obj->GetUIdFromGUID());
855 } break;
856 case HIGHGUID_TYPE_DYNAMICOBJECT:
857 {
858 m_DynamicObjectStorage.erase(obj->getGuidLow());
859 } break;
860 case HIGHGUID_TYPE_GAMEOBJECT:
861 {
862 if (obj->GetUIdFromGUID() <= m_GOHighGuid)
863 {
864 m_GameObjectStorage[obj->GetUIdFromGUID()] = nullptr;
865 if (static_cast<GameObject*>(obj)->m_spawn != nullptr)
866 _sqlids_gameobjects.erase(static_cast<GameObject*>(obj)->m_spawn->id);
867
868 if (free_guid)
869 _reusable_guids_gameobject.push_back(obj->GetUIdFromGUID());
870 }
871 } break;
872 case HIGHGUID_TYPE_TRANSPORTER:
873 break;
874 default:
875 {
876 if (!obj->isPlayer())
877 sLogger.debug("WorldMap::RemoveObject called for invalid type {} (not player).", obj->GetTypeFromGUID());
878 break;
879 }
880 }
881
882 //\todo That object types are not map objects. TODO: add AI groups here?
883 if (obj->isItem() || obj->isContainer())
884 {
885 return;
886 }
887
888 if (obj->isCorpse())
889 {
890 m_corpses.erase(static_cast<Corpse*>(obj));
891 }
892
893 MapCell* cell = getCell(obj->GetMapCellX(), obj->GetMapCellY());
894 if (cell == nullptr)
895 {
896 // set the map cell correctly
897 if (obj->GetPositionX() < Map::Terrain::_maxX || obj->GetPositionX() > Map::Terrain::_minY || obj->GetPositionY() < Map::Terrain::_maxY || obj->GetPositionY() > Map::Terrain::_minY)
898 {
899 cell = this->getCellByCoords(obj->GetPositionX(), obj->GetPositionY());
900 obj->SetMapCell(cell);
901 }
902 }
903
904 if (cell != nullptr)
905 {
906 cell->removeObject(obj); // Remove object from cell
907 obj->SetMapCell(nullptr); // Unset object's cell
908 }
909
910 Player* plObj = nullptr;
911 if (obj->isPlayer())
912 {
913 plObj = static_cast<Player*>(obj);
914
915 _processQueue.erase(plObj); // Clear any updates pending
916 plObj->getUpdateMgr().clearPendingUpdates();
917
918 removePlayerFromMap(plObj);
919 }
920
921 obj->removeSelfFromInrangeSets();
922 obj->clearInRangeSets(); // Clear object's in-range set
923
924 uint8_t cellNumber = worldConfig.server.mapCellNumber;
925
926 // If it's a player - update his nearby cells
927 if (obj->isPlayer())
928 {// get x/y
929 if (obj->GetPositionX() < Map::Terrain::_maxX || obj->GetPositionX() > Map::Terrain::_minY || obj->GetPositionY() < Map::Terrain::_maxY || obj->GetPositionY() > Map::Terrain::_minY)
930 {
931 uint32_t x = getPosX(obj->GetPositionX());
932 uint32_t y = getPosY(obj->GetPositionY());
933 updateCellActivity(x, y, 2U + cellNumber);
934 }
935 m_PlayerStorage.erase(obj->getGuidLow());
936 }
937 else if (obj->isCreatureOrPlayer() && static_cast<Unit*>(obj)->m_playerControler != nullptr)
938 {
939 if (obj->GetPositionX() < Map::Terrain::_maxX || obj->GetPositionX() > Map::Terrain::_minY || obj->GetPositionY() < Map::Terrain::_maxY || obj->GetPositionY() > Map::Terrain::_minY)
940 {
941 uint32_t x = getPosX(obj->GetPositionX());
942 uint32_t y = getPosY(obj->GetPositionY());
943 updateCellActivity(x, y, 2U + cellNumber);
944 }
945 }
946
947 // Remove the session from our set if it is a player.
948 if (obj->isPlayer() || obj->isCreatureOrPlayer() && static_cast<Unit*>(obj)->m_playerControler != nullptr)
949 {
950 for (auto _mapWideStaticObject : _mapWideStaticObjects)
951 {
952 if (_mapWideStaticObject != nullptr && plObj)
953 plObj->getUpdateMgr().pushOutOfRangeGuid(_mapWideStaticObject->GetNewGUID());
954 }
955
956 // Setting an instance ID here will trigger the session to be removed by WorldMap::run(). :)
957 if (plObj && plObj->getSession())
958 {
959 plObj->getSession()->SetInstance(0);
960
961 // Add it to the global session set. Don't "re-add" to session if it is being deleted.
962 if (!plObj->getSession()->bDeleted)
963 sWorld.addGlobalSession(plObj->getSession());
964 }
965 }

1014 {
1015 getTerrain()->loadTile(TileX, TileY);

1018 }
1019
1020 for (uint32_t xc = (CellX % Map::Cell::CellsPerTile) * 16 / Map::Cell::CellsPerTile; xc < (CellX % Map::Cell::CellsPerTile) * 16 / Map::Cell::CellsPerTile + 16 / Map::Cell::CellsPerTile; xc++)

1028 Result = true;

1468 uint32_t cellY = getPosY(obj->GetPositionY());

1520 {
1521 updateCellActivity(pOldCell->_x, pOldCell->_y, cellNumber);
1522 }
1523 }
1524 }
1525 }

1539 startY = cellY > 5 ? cellY - 6 : 0;

1549 updateInRangeSet(obj, plObj, cell, buf);
