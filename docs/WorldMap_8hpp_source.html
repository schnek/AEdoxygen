78struct CompareRespawnInfo
79{
80 bool operator()(std::unique_ptr<RespawnInfo> const& a, std::unique_ptr<RespawnInfo> const& b)
81 {
82 if (a == b)
83 return false;
84 if (a->time != b->time)
85 return (a->time > b->time);
86 if (a->spawnId != b->spawnId)
87 return a->spawnId < b->spawnId;
88 ASSERT(a->type != b->type);
89 return a->type < b->type;
90 }
91};

101 if (a.type != b.type)
102 return false;

104 return true;
105}

107class respawnQueue : public std::priority_queue<std::unique_ptr<RespawnInfo>, std::vector<std::unique_ptr<RespawnInfo>>, CompareRespawnInfo>
108{
109public:
110 bool remove(RespawnInfo const* value)
111 {
112 auto it = std::find_if(this->c.begin(), this->c.end(), [value](std::unique_ptr<RespawnInfo> const& respawn) { return respawn.get() == value; });
113 if (it != this->c.end())
114 {
115 this->c.erase(it);
116 std::make_heap(this->c.begin(), this->c.end(), this->comp);
117 return true;
118 }
119
120 return false;
121 }
122
123 void clear()
124 {
125 this->c.clear();
126 }
127};
