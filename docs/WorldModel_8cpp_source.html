34 bool IntersectTriangle(const MeshTriangle &tri, std::vector<Vector3>::const_iterator points, const G3D::Ray &ray, float &distance)
35 {
36 static const float EPS = 1e-5f;
37
38 // See RTR2 ch. 13.7 for the algorithm.
39
40 const Vector3 e1 = points[tri.idx1] - points[tri.idx0];
41 const Vector3 e2 = points[tri.idx2] - points[tri.idx0];
42 const Vector3 p(ray.direction().cross(e2));
43 const float a = e1.dot(p);
44
45 if (std::fabs(a) < EPS) {
46 // Determinant is ill-conditioned; abort early
47 return false;
48 }
49
50 const float f = 1.0f / a;
51 const Vector3 s(ray.origin() - points[tri.idx0]);
52 const float u = f * s.dot(p);
53
54 if ((u < 0.0f) || (u > 1.0f)) {
55 // We hit the plane of the m_geometry, but outside the m_geometry
56 return false;
57 }
58
59 const Vector3 q(s.cross(e1));
60 const float v = f * ray.direction().dot(q);
61
62 if ((v < 0.0f) || ((u + v) > 1.0f)) {
63 // We hit the plane of the triangle, but outside the triangle
64 return false;
65 }
66
67 const float t = f * e2.dot(q);
68
69 if ((t > 0.0f) && (t < distance))
70 {
71 // This is a new hit, closer than the previous one
72 distance = t;
73
74 /* baryCoord[0] = 1.0 - u - v;
75 baryCoord[1] = u;
76 baryCoord[2] = v; */
77
78 return true;
79 }
80 // This hit is after the previous hit, so ignore it
81 return false;
82 }

84 class TriBoundFunc
85 {
86 public:
87 TriBoundFunc(std::vector<Vector3> &vert): vertices(vert.begin()) { }
88 void operator()(const MeshTriangle &tri, G3D::AABox &out) const
89 {
90 G3D::Vector3 lo = vertices[tri.idx0];
91 G3D::Vector3 hi = lo;
92
93 lo = (lo.min(vertices[tri.idx1])).min(vertices[tri.idx2]);
94 hi = (hi.max(vertices[tri.idx1])).max(vertices[tri.idx2]);
95
96 out = G3D::AABox(lo, hi);
97 }
98 protected:
99 const std::vector<Vector3>::const_iterator vertices;
100 };

143 bool WmoLiquid::GetLiquidHeight(const Vector3 &pos, float &liqHeight) const
144 {
145 float tx_f = (pos.x - iCorner.x)/LIQUID_TILE_SIZE;
146 uint32_t tx = uint32_t(tx_f);
147 if (tx_f < 0.0f || tx >= iTilesX)
148 return false;
149 float ty_f = (pos.y - iCorner.y)/LIQUID_TILE_SIZE;
150 uint32_t ty = uint32_t(ty_f);
151 if (ty_f < 0.0f || ty >= iTilesY)
152 return false;
153
154 // check if tile shall be used for liquid level
155 // checking for 0x08 *might* be enough, but disabled tiles always are 0x?F:
156 if ((iFlags[tx + ty*iTilesX] & 0x0F) == 0x0F)
157 return false;
158
159 // (dx, dy) coordinates inside tile, in [0, 1]^2
160 float dx = tx_f - (float)tx;
161 float dy = ty_f - (float)ty;
162
163 /* Tesselate tile to two triangles (not sure if client does it exactly like this)
164
165 ^ dy
166 |
167 1 x---------x (1, 1)
168 | (b) / |
169 | / |
170 | / |
171 | / (a) |
172 x---------x---> dx
173 0 1
174 */
175
176 const uint32_t rowOffset = iTilesX + 1;
177 if (dx > dy) // case (a)
178 {
179 float sx = iHeight[tx+1 + ty * rowOffset] - iHeight[tx + ty * rowOffset];
180 float sy = iHeight[tx+1 + (ty+1) * rowOffset] - iHeight[tx+1 + ty * rowOffset];
181 liqHeight = iHeight[tx + ty * rowOffset] + dx * sx + dy * sy;
182 }
183 else // case (b)
184 {
185 float sx = iHeight[tx+1 + (ty+1) * rowOffset] - iHeight[tx + (ty+1) * rowOffset];
186 float sy = iHeight[tx + (ty+1) * rowOffset] - iHeight[tx + ty * rowOffset];
187 liqHeight = iHeight[tx + ty * rowOffset] + dx * sx + dy * sy;
188 }
189 return true;
190 }

270 bool GroupModel::writeToFile(FILE* wf)
271 {
272 bool result = true;
273
274 if (result && fwrite(&iBound, sizeof(G3D::AABox), 1, wf) != 1) result = false;
275 if (result && fwrite(&iMogpFlags, sizeof(uint32_t), 1, wf) != 1) result = false;
276 if (result && fwrite(&iGroupWMOID, sizeof(uint32_t), 1, wf) != 1) result = false;
277
278 // write vertices
279 if (result && fwrite("VERT", 1, 4, wf) != 4)
280 result = false;
281 size_t count = vertices.size();
282 size_t chunkSize = sizeof(uint32_t)+ sizeof(Vector3)*count;
283 if (result && fwrite(&chunkSize, sizeof(uint32_t), 1, wf) != 1)
284 result = false;
285 if (result && fwrite(&count, sizeof(uint32_t), 1, wf) != 1)
286 result = false;
287 if (!count) // models without (collision) geometry end here, unsure if they are useful
288 return result;
289 if (result && fwrite(&vertices[0], sizeof(Vector3), count, wf) != count)
290 result = false;
291
292 // write triangle mesh
293 if (result && fwrite("TRIM", 1, 4, wf) != 4)
294 result = false;
295 count = triangles.size();
296 chunkSize = sizeof(uint32_t)+ sizeof(MeshTriangle)*count;
297 if (result && fwrite(&chunkSize, sizeof(uint32_t), 1, wf) != 1)
298 result = false;
299 if (result && fwrite(&count, sizeof(uint32_t), 1, wf) != 1)
300 result = false;
301 if (result && fwrite(&triangles[0], sizeof(MeshTriangle), count, wf) != count)
302 result = false;
303
304 // write mesh BIH
305 if (result && fwrite("MBIH", 1, 4, wf) != 4)
306 result = false;
307 if (result) result = meshTree.writeToFile(wf);
308
309 // write liquid data
310 if (result && fwrite("LIQU", 1, 4, wf) != 4)
311 result = false;
312 if (!iLiquid)
313 {
314 chunkSize = 0;
315 if (result && fwrite(&chunkSize, sizeof(uint32_t), 1, wf) != 1)
316 result = false;
317 return result;
318 }
319 chunkSize = iLiquid->GetFileSize();
320 if (result && fwrite(&chunkSize, sizeof(uint32_t), 1, wf) != 1)
321 result = false;
322 if (result) result = iLiquid->writeToFile(wf);
323
324 return result;

327 bool GroupModel::readFromFile(FILE* rf)
328 {
329 char chunk[8];
330 bool result = true;
331 uint32_t chunkSize = 0;
332 uint32_t count = 0;
333 triangles.clear();
334 vertices.clear();
335 iLiquid = nullptr;
336
337 if (result && fread(&iBound, sizeof(G3D::AABox), 1, rf) != 1) result = false;
338 if (result && fread(&iMogpFlags, sizeof(uint32_t), 1, rf) != 1) result = false;
339 if (result && fread(&iGroupWMOID, sizeof(uint32_t), 1, rf) != 1) result = false;
340
341 // read vertices
342 if (result && !readChunk(rf, chunk, "VERT", 4)) result = false;
343 if (result && fread(&chunkSize, sizeof(uint32_t), 1, rf) != 1) result = false;
344 if (result && fread(&count, sizeof(uint32_t), 1, rf) != 1) result = false;
345 if (!count) // models without (collision) geometry end here, unsure if they are useful
346 return result;
347 if (result) vertices.resize(count);
348 if (result && fread(&vertices[0], sizeof(Vector3), count, rf) != count) result = false;
349
350 // read triangle mesh
351 if (result && !readChunk(rf, chunk, "TRIM", 4)) result = false;
352 if (result && fread(&chunkSize, sizeof(uint32_t), 1, rf) != 1) result = false;
353 if (result && fread(&count, sizeof(uint32_t), 1, rf) != 1) result = false;
354 if (result) triangles.resize(count);
355 if (result && fread(&triangles[0], sizeof(MeshTriangle), count, rf) != count) result = false;
356
357 // read mesh BIH
358 if (result && !readChunk(rf, chunk, "MBIH", 4)) result = false;
359 if (result) result = meshTree.readFromFile(rf);
360
361 // write liquid data
362 if (result && !readChunk(rf, chunk, "LIQU", 4)) result = false;
363 if (result && fread(&chunkSize, sizeof(uint32_t), 1, rf) != 1) result = false;
364 if (result && chunkSize > 0)
365 result = WmoLiquid::readFromFile(rf, iLiquid);
366 return result;
367 }
