127static inline void FastGUIDPack(ByteBuffer & buf, const uint64_t & oldguid)
128{
129 // hehe speed freaks
130 uint8_t guidmask = 0;
131 uint8_t guidfields[9] = {0, 0, 0, 0, 0, 0, 0, 0};

133 int j = 1;
134 uint8_t* test = (uint8_t*)&oldguid;
135
136 if (*test) //7*8
137 {
138 guidfields[j] = *test;
139 guidmask |= 1;
140 j++;
141 }
