40
41 uint8_t encryptHash[SHA_DIGEST_LENGTH];
42 uint8_t decryptHash[SHA_DIGEST_LENGTH];
43
44 uint8_t pass[1024] = { 0 };
45 uint32_t mdLength;
46
47 HMAC(EVP_sha1(), send, seedLenght, key, 40, decryptHash, &mdLength);

54 m_servertWotlkEncrypt.setup(encryptHash, SHA_DIGEST_LENGTH);

153void WowCrypt::generateTbcKey(uint8_t* key, uint8_t* sessionkey)
154{
155 uint8_t seedKey[seedLenght] = { 0x38, 0xA7, 0x83, 0x15, 0xF8, 0x92, 0x25, 0x30, 0x71, 0x98, 0x67, 0xB1, 0x8C, 0x4, 0xE2, 0xAA };
156
157 uint8_t firstBuffer[64];
158 uint8_t secondBuffer[64];
159
160 memset(firstBuffer, 0x36, 64);
161 memset(secondBuffer, 0x5C, 64);
162
163 for (uint8_t i = 0; i < seedLenght; ++i)
164 {
165 firstBuffer[i] = (uint8_t)(seedKey[i] ^ firstBuffer[i]);
166 secondBuffer[i] = (uint8_t)(seedKey[i] ^ secondBuffer[i]);
167 }
168
169 Sha1Hash sha1;
170 sha1.updateData(firstBuffer, 64);
171 sha1.updateData(sessionkey, 40);
172 sha1.finalize();
173
174 uint8_t* tempDigest = sha1.getDigest();
175 Sha1Hash sha2;
176 sha2.updateData(secondBuffer, 64);
177 sha2.updateData(tempDigest, SHA_DIGEST_LENGTH);
178 sha2.finalize();
179
180 memcpy(key, sha2.getDigest(), SHA_DIGEST_LENGTH);
181}
