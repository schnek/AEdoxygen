48void XevozzAI::AIUpdate(unsigned long time_passed)
49{
50 scriptEvents.updateEvents(time_passed, getScriptPhase());
51
52 while (uint32_t eventId = scriptEvents.getFinishedEvent())
53 {
54 switch (eventId)
55 {
56 case 1:
57 {
58 sendRandomDBChatMessage({ Xevozz::SAY_REPEAT_SUMMON1 , Xevozz::SAY_REPEAT_SUMMON2 }, nullptr);
59
60 uint8_t spell = SelectRandomContainerElement(summonSpells);
61
62 getCreature()->castSpell(getCreature(), Xevozz::EtherealSphereSummonSpells[spell]);
63 summonSpells.remove(spell);
64
65 scriptEvents.addEvent(1, Util::getRandomUInt(45000, 50000));
66
67 if (isHeroic())
68 scriptEvents.addEvent(2, 2500);
69 } break;
70 case 2:
71 {
72 uint8_t spell = SelectRandomContainerElement(summonSpells);
73 getCreature()->castSpell(getCreature(), Xevozz::EtherealSphereSummonSpells[spell]);
74 } break;
75 default:
76 break;
77 }
78 }
79}
