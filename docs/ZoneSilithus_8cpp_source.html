77void AreatriggerHook(PlayerPointer pPlayer, uint32_t triggerID)
78{
79 if(!pPlayer)
80 return;
81
82 if(triggerID == ALLIANCE_RETURN ||
83 triggerID == HORDE_RETURN)
84 {
85 if(pPlayer->HasAura(SILITHYST_SPELL)) // if we doesn't have it, do nothing
86 {
87 pPlayer->RemoveAura(SILITHYST_SPELL);
88
89 /* Rewards */
90 pPlayer->CastSpell(pPlayer, TRACES_OF_SILITHYST, true);
91 pPlayer->m_honorPoints += REWARD_HONOR;
92 pPlayer->ModStanding(REWARD_REPUTATION_FACTION, REWARD_REPUTATION_VALUE);
93
94 uint32_t quest = pPlayer->GetTeam() == ALLIANCE ? ALLIANCE_SILITHYST_QUEST : HORDE_SILITHYST_QUEST;
95 QuestLogEntry* qle = pPlayer->GetQuestLogForEntry(quest);
96 if(qle)
97 qle->SendQuestComplete();
98
99 if(locked)
100 return;
101 silithyst_gathered[ pPlayer->GetTeam() ]++;
102 uint32_t worldstate = pPlayer->GetTeam() == ALLIANCE ? WORLDSTATE_SILITHUS_GATHERED_ALLIANCE : WORLDSTATE_SILITHUS_GATHERED_HORDE;
103 pPlayer->GetMapMgr()->GetStateManager().UpdateWorldState(worldstate , silithyst_gathered[ pPlayer->GetTeam() ]);
104 if(silithyst_gathered[ pPlayer->GetTeam() ] >= SILITHYST_MAX)
105 {
106 locked = true; // we win, so opposite team must wait for server restart
107 winners = pPlayer->GetTeam();
108 pPlayer->GetMapMgr()->CastSpellOnPlayers(winners, CENARION_FAVOR);
109 pPlayer->GetMapMgr()->RemovePositiveAuraFromPlayers(winners == ALLIANCE ? HORDE : ALLIANCE, CENARION_FAVOR);
110 }
111 }
112 }
113}

131void DropFlag(PlayerPointer pPlayer, uint32_t spellID)
132{
133 if(!pPlayer || spellID != SILITHYST_SPELL)
134 return;
135 uint32_t triggerID = pPlayer->GetTeam() == ALLIANCE ? ALLIANCE_RETURN : HORDE_RETURN;
136 // we have to use AreaTrigger.dbc here
137 AreaTrigger* pAreaTrigger = AreaTriggerStorage.LookupEntry(triggerID);
138 if(pAreaTrigger)
139 {
140 if(pPlayer->CalcDistance(pAreaTrigger->x, pAreaTrigger->y, pAreaTrigger->z) > 10.0f)
141 {
142 GameObjectPointer pGo = pPlayer->GetMapMgr()->GetInterface()->SpawnGameObject(SILITHYST_MOUND, pPlayer->GetPositionX(), pPlayer->GetPositionY(), pPlayer->GetPositionZ(), 0, true, 0, 0);
143 if(pGo == NULL)
144 {
145 pGo->Destructor();
146 pGo = NULL;
147 }
148 };
149 }
