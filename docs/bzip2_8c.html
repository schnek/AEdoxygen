734 "%s: %d specified on command line, %d not processed yet.\n\n",

1008/*---------------------------------------------*/
1009/* Copy modification date, access date, permissions and owner from the
1010 source to destination file. We have to copy this meta-info off
1011 into fileMetaInfo before starting to compress / decompress it,
1012 because doing it afterwards means we get the wrong access time.
1013
1014 To complicate matters, in compress() and decompress() below, the
1015 sequence of tests preceding the call to saveInputFileMetaInfo()
1016 involves calling fileExists(), which in turn establishes its result
1017 by attempting to fopen() the file, and if successful, immediately
1018 fclose()ing it again. So we have to assume that the fopen() call
1019 does not cause the access time field to be updated.
1020
1021 Reading of the man page for stat() (man 2 stat) on RedHat 7.2 seems
1022 to imply that merely doing open() will not affect the access time.
1023 Therefore we merely need to hope that the C library only does
1024 open() as a result of fopen(), and not any kind of read()-ahead
1025 cleverness.
1026
1027 It sounds pretty fragile to me. Whether this carries across
1028 robustly to arbitrary Unix-like platforms (or even works robustly
1029 on this one, RedHat 7.2) is unknown to me. Nevertheless ...
1030*/
1031#if BZ_UNIX
1032static
1033struct MY_STAT fileMetaInfo;
1034#endif
1035
1036static
1037void saveInputFileMetaInfo ( Char *srcName )
1038{
1039# if BZ_UNIX
1040 IntNative retVal;
1041 /* Note use of stat here, not lstat. */
1042 retVal = MY_STAT( srcName, &fileMetaInfo );
1043 ERROR_IF_NOT_ZERO ( retVal );
1044# endif
1045}

1056 uTimBuf.modtime = fileMetaInfo.st_mtime;

1058 retVal = utime ( dstName, &uTimBuf );
1059 ERROR_IF_NOT_ZERO ( retVal );
1060# endif

1130/*---------------------------------------------*/
1131static
1132void compress ( Char *name )
1133{
1134 FILE *inStr;
1135 FILE *outStr;
1136 Int32 n, i;
1137 struct MY_STAT statBuf;
1138
1139 deleteOutputOnInterrupt = False;
1140
1141 if (name == NULL && srcMode != SM_I2O)
1142 panic ( "compress: bad modes\n" );
1143
1144 switch (srcMode) {
1145 case SM_I2O:
1146 copyFileName ( inName, (Char*)"(stdin)" );
1147 copyFileName ( outName, (Char*)"(stdout)" );
1148 break;
1149 case SM_F2F:
1150 copyFileName ( inName, name );
1151 copyFileName ( outName, name );
1152 strcat ( outName, ".bz2" );
1153 break;
1154 case SM_F2O:
1155 copyFileName ( inName, name );
1156 copyFileName ( outName, (Char*)"(stdout)" );
1157 break;
1158 }
1159
1160 if ( srcMode != SM_I2O && containsDubiousChars ( inName ) ) {
1161 if (noisy)
1162 fprintf ( stderr, "%s: There are no files matching `%s'.\n",
1163 progName, inName );
1164 setExit(1);
1165 return;
1166 }
1167 if ( srcMode != SM_I2O && !fileExists ( inName ) ) {
1168 fprintf ( stderr, "%s: Can't open input file %s: %s.\n",
1169 progName, inName, strerror(errno) );
1170 setExit(1);
1171 return;
1172 }
1173 for (i = 0; i < BZ_N_SUFFIX_PAIRS; i++) {
1174 if (hasSuffix(inName, zSuffix[i])) {
1175 if (noisy)
1176 fprintf ( stderr,
1177 "%s: Input file %s already has %s suffix.\n",
1178 progName, inName, zSuffix[i] );
1179 setExit(1);
1180 return;
1181 }
1182 }
1183 if ( srcMode == SM_F2F || srcMode == SM_F2O ) {
1184 MY_STAT(inName, &statBuf);
1185 if ( MY_S_ISDIR(statBuf.st_mode) ) {
1186 fprintf( stderr,
1187 "%s: Input file %s is a directory.\n",
1188 progName,inName);
1189 setExit(1);
1190 return;
1191 }
1192 }
1193 if ( srcMode == SM_F2F && !forceOverwrite && notAStandardFile ( inName )) {
1194 if (noisy)
1195 fprintf ( stderr, "%s: Input file %s is not a normal file.\n",
1196 progName, inName );
1197 setExit(1);
1198 return;
1199 }
1200 if ( srcMode == SM_F2F && fileExists ( outName ) ) {
1201 if (forceOverwrite) {
1202 remove(outName);
1203 } else {
1204 fprintf ( stderr, "%s: Output file %s already exists.\n",
1205 progName, outName );
1206 setExit(1);
1207 return;
1208 }
1209 }
1210 if ( srcMode == SM_F2F && !forceOverwrite &&
1211 (n=countHardLinks ( inName )) > 0) {
1212 fprintf ( stderr, "%s: Input file %s has %d other link%s.\n",
1213 progName, inName, n, n > 1 ? "s" : "" );
1214 setExit(1);
1215 return;
1216 }
1217
1218 if ( srcMode == SM_F2F ) {
1219 /* Save the file's meta-info before we open it. Doing it later
1220 means we mess up the access times. */
1221 saveInputFileMetaInfo ( inName );
1222 }
1223
1224 switch ( srcMode ) {
1225
1226 case SM_I2O:
1227 inStr = stdin;
1228 outStr = stdout;
1229 if ( isatty ( fileno ( stdout ) ) ) {
1230 fprintf ( stderr,
1231 "%s: I won't write compressed data to a terminal.\n",
1232 progName );
1233 fprintf ( stderr, "%s: For help, type: `%s --help'.\n",
1234 progName, progName );
1235 setExit(1);
1236 return;
1237 };
1238 break;
1239
1240 case SM_F2O:
1241 inStr = fopen ( inName, "rb" );
1242 outStr = stdout;
1243 if ( isatty ( fileno ( stdout ) ) ) {
1244 fprintf ( stderr,
1245 "%s: I won't write compressed data to a terminal.\n",
1246 progName );
1247 fprintf ( stderr, "%s: For help, type: `%s --help'.\n",
1248 progName, progName );
1249 if ( inStr != NULL ) fclose ( inStr );
1250 setExit(1);
1251 return;
1252 };
1253 if ( inStr == NULL ) {
1254 fprintf ( stderr, "%s: Can't open input file %s: %s.\n",
1255 progName, inName, strerror(errno) );
1256 setExit(1);
1257 return;
1258 };
1259 break;
1260
1261 case SM_F2F:
1262 inStr = fopen ( inName, "rb" );
1263 outStr = fopen_output_safely ( outName, "wb" );
1264 if ( outStr == NULL) {
1265 fprintf ( stderr, "%s: Can't create output file %s: %s.\n",
1266 progName, outName, strerror(errno) );
1267 if ( inStr != NULL ) fclose ( inStr );
1268 setExit(1);
1269 return;
1270 }
1271 if ( inStr == NULL ) {
1272 fprintf ( stderr, "%s: Can't open input file %s: %s.\n",
1273 progName, inName, strerror(errno) );
1274 if ( outStr != NULL ) fclose ( outStr );
1275 setExit(1);
1276 return;
1277 };
1278 break;
1279
1280 default:
1281 panic ( "compress: bad srcMode" );
1282 break;
1283 }
1284
1285 if (verbosity >= 1) {
1286 fprintf ( stderr, " %s: ", inName );
1287 pad ( inName );
1288 fflush ( stderr );
1289 }
1290
1291 /*--- Now the input and output handles are sane. Do the Biz. ---*/
1292 outputHandleJustInCase = outStr;
1293 deleteOutputOnInterrupt = True;
1294 compressStream ( inStr, outStr );
1295 outputHandleJustInCase = NULL;
1296
1297 /*--- If there was an I/O error, we won't get here. ---*/
1298 if ( srcMode == SM_F2F ) {
1299 applySavedTimeInfoToOutputFile ( outName );
1300 deleteOutputOnInterrupt = False;
1301 if ( !keepInputFiles ) {
1302 IntNative retVal = remove ( inName );
1303 ERROR_IF_NOT_ZERO ( retVal );
1304 }
1305 }
1306
1307 deleteOutputOnInterrupt = False;
1308}

1312static
1313void uncompress ( Char *name )
1314{
1315 FILE *inStr;
1316 FILE *outStr;
1317 Int32 n, i;
1318 Bool magicNumberOK;
1319 Bool cantGuess;
1320 struct MY_STAT statBuf;
1321
1322 deleteOutputOnInterrupt = False;

1324 if (name == NULL && srcMode != SM_I2O)
1325 panic ( "uncompress: bad modes\n" );

1327 cantGuess = False;
1328 switch (srcMode) {
1329 case SM_I2O:
1330 copyFileName ( inName, (Char*)"(stdin)" );
1331 copyFileName ( outName, (Char*)"(stdout)" );
1332 break;

1334 copyFileName ( inName, name );
1335 copyFileName ( outName, name );
1336 for (i = 0; i < BZ_N_SUFFIX_PAIRS; i++)
1337 if (mapSuffix(outName,zSuffix[i],unzSuffix[i]))
1338 goto zzz;
1339 cantGuess = True;
1340 strcat ( outName, ".out" );
1341 break;
1342 case SM_F2O:
1343 copyFileName ( inName, name );
1344 copyFileName ( outName, (Char*)"(stdout)" );
1345 break;
1346 }
1347
1348 zzz:
1349 if ( srcMode != SM_I2O && containsDubiousChars ( inName ) ) {
1350 if (noisy)
1351 fprintf ( stderr, "%s: There are no files matching `%s'.\n",
1352 progName, inName );
1353 setExit(1);
1354 return;
1355 }
1356 if ( srcMode != SM_I2O && !fileExists ( inName ) ) {
1357 fprintf ( stderr, "%s: Can't open input file %s: %s.\n",
1358 progName, inName, strerror(errno) );
1359 setExit(1);
1360 return;
1361 }
1362 if ( srcMode == SM_F2F || srcMode == SM_F2O ) {
1363 MY_STAT(inName, &statBuf);
1364 if ( MY_S_ISDIR(statBuf.st_mode) ) {
1365 fprintf( stderr,
1366 "%s: Input file %s is a directory.\n",
1367 progName,inName);
1368 setExit(1);
1369 return;
1370 }
1371 }
1372 if ( srcMode == SM_F2F && !forceOverwrite && notAStandardFile ( inName )) {
1373 if (noisy)
1374 fprintf ( stderr, "%s: Input file %s is not a normal file.\n",
1375 progName, inName );
1376 setExit(1);
1377 return;
1378 }
1379 if ( /* srcMode == SM_F2F implied && */ cantGuess ) {
1380 if (noisy)
1381 fprintf ( stderr,
1382 "%s: Can't guess original name for %s -- using %s\n",
1383 progName, inName, outName );
1384 /* just a warning, no return */
1385 }
1386 if ( srcMode == SM_F2F && fileExists ( outName ) ) {
1387 if (forceOverwrite) {
1388 remove(outName);
1389 } else {
1390 fprintf ( stderr, "%s: Output file %s already exists.\n",
1391 progName, outName );
1392 setExit(1);
1393 return;
1394 }
1395 }
1396 if ( srcMode == SM_F2F && !forceOverwrite &&
1397 (n=countHardLinks ( inName ) ) > 0) {
1398 fprintf ( stderr, "%s: Input file %s has %d other link%s.\n",
1399 progName, inName, n, n > 1 ? "s" : "" );
1400 setExit(1);
1401 return;
1402 }
1403
1404 if ( srcMode == SM_F2F ) {
1405 /* Save the file's meta-info before we open it. Doing it later
1406 means we mess up the access times. */
1407 saveInputFileMetaInfo ( inName );
1408 }
1409
1410 switch ( srcMode ) {
1411
1412 case SM_I2O:
1413 inStr = stdin;
1414 outStr = stdout;
1415 if ( isatty ( fileno ( stdin ) ) ) {
1416 fprintf ( stderr,
1417 "%s: I won't read compressed data from a terminal.\n",
1418 progName );
1419 fprintf ( stderr, "%s: For help, type: `%s --help'.\n",
1420 progName, progName );
1421 setExit(1);
1422 return;
1423 };
1424 break;
1425
1426 case SM_F2O:
1427 inStr = fopen ( inName, "rb" );
1428 outStr = stdout;
1429 if ( inStr == NULL ) {
1430 fprintf ( stderr, "%s: Can't open input file %s:%s.\n",
1431 progName, inName, strerror(errno) );
1432 if ( inStr != NULL ) fclose ( inStr );
1433 setExit(1);
1434 return;
1435 };
1436 break;
1437
1438 case SM_F2F:
1439 inStr = fopen ( inName, "rb" );
1440 outStr = fopen_output_safely ( outName, "wb" );
1441 if ( outStr == NULL) {
1442 fprintf ( stderr, "%s: Can't create output file %s: %s.\n",
1443 progName, outName, strerror(errno) );
1444 if ( inStr != NULL ) fclose ( inStr );
1445 setExit(1);
1446 return;
