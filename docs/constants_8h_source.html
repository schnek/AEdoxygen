45/** Values for UniversalSurface::GPUGeom::refractionHint. */
46class RefractionQuality {
47public:
48 enum Value {
49 /** No refraction; a translucent object will appear as if it had the same index of refraction
50 as the surrounding medium and objects will be undistorted in the background. */
51 NONE = 0,
52
53 /** Use a static environment map (cube or paraboloid) for computing transmissivity.*/
54 STATIC_ENV = 25,
55
56 /** Use a dynamically rendered 2D environment map; distort the background. This looks good for many scenes
57 but avoids the cost of rendering a cube map for DYNAMIC_ENV. */
58 DYNAMIC_FLAT = 50,
59
60 /** Use a dynamically rendered 2D environment map that is re-captured per transparent object. This works well
61 for transparent objects that are separated by a significant camera space z distance but overlap in screen space.*/
62 DYNAMIC_FLAT_MULTILAYER = 55,
63
64 /** Render a dynamic environment map */
65 DYNAMIC_ENV = 75,
66
67 /** Use the best method available, ideally true ray tracing. */
68 BEST = 100
69 };
