18/**
19 Find a registered PRNG by name
20 @param name The name of the PRNG to look for
21 @return >= 0 if found, -1 if not present
22*/
23int find_prng(const char *name)
24{
25 int x;
26 LTC_ARGCHK(name != NULL);
27 LTC_MUTEX_LOCK(&ltc_prng_mutex);
28 for (x = 0; x < TAB_SIZE; x++) {
29 if ((prng_descriptor[x].name != NULL) && XSTRCMP(prng_descriptor[x].name, name) == 0) {
30 LTC_MUTEX_UNLOCK(&ltc_prng_mutex);
31 return x;
32 }
33 }
34 LTC_MUTEX_UNLOCK(&ltc_prng_mutex);
35 return -1;
36}
