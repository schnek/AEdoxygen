48static void postToClipboard(const char *text) {
49 if (OpenClipboard(NULL)) {
50 HGLOBAL hMem = GlobalAlloc(GHND | GMEM_DDESHARE, strlen(text) + 1);
51 if (hMem) {
52 char *pMem = (char*)GlobalLock(hMem);
53 strcpy(pMem, text);
54 GlobalUnlock(hMem);
55
56 EmptyClipboard();
57 SetClipboardData(CF_TEXT, hMem);
58 }
59
60 CloseClipboard();
61 GlobalFree(hMem);
62 }
63}
64#endif

69static void createErrorMessage(
70 const char* expression,
71 const std::string& message,
72 const char* filename,
73 int lineNumber,
74 std::string& outTitle,
75 std::string& outMessage) {
76
77 std::string le = "";
78 const char* newline = "\n";
79
80 #ifdef G3D_WINDOWS
81 newline = "\r\n";
82
83 // The last error value. (Which is preserved across the call).
84 DWORD lastErr = GetLastError();
85
86 // The decoded message from FormatMessage
87 LPTSTR formatMsg = NULL;
88
89 if (NULL == formatMsg) {
90 FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER |
91 FORMAT_MESSAGE_IGNORE_INSERTS |
92 FORMAT_MESSAGE_FROM_SYSTEM,
93 NULL,
94 lastErr,
95 0,
96 (LPTSTR)&formatMsg,
97 0,
98 NULL);
99 }
100
101 // Make sure the message got translated into something.
102 LPTSTR realLastErr;
103 if (NULL != formatMsg) {
104 realLastErr = formatMsg;
105 } else {
106 const char* noLastError = "Last error code does not exist.";
107 realLastErr = const_cast<LPSTR>(noLastError);
108 }
109
110 if (lastErr != 0) {
111 le = G3D::format("Last Error (0x%08X): %s\r\n\r\n", lastErr, (LPCSTR)realLastErr);
112 }
113
114 // Get rid of the allocated memory from FormatMessage.
115 if (NULL != formatMsg) {
116 LocalFree((LPVOID)formatMsg);
117 }
118
119 char modulePath[MAX_PATH];
120 GetModuleFileNameA(NULL, modulePath, MAX_PATH);
121
122 const char* moduleName = strrchr(modulePath, '\\');
123 outTitle = outTitle + string(" - ") + string(moduleName ? (moduleName + 1) : modulePath);
124
125 #else
126 (void)outTitle;
127 #endif
128
129 // Build the message.
130 outMessage =
131 G3D::format("%s%s%sExpression: %s%s%s:%d%s%s%s",
132 message.c_str(), newline, newline, expression, newline,
133 filename, lineNumber, newline, newline, le.c_str());
134}

137bool _handleDebugAssert_(
138 const char* expression,
139 const std::string& message,
140 const char* filename,
141 int lineNumber,
142 bool useGuiPrompt) {
143
144 std::string dialogTitle = "Assertion Failure";
145 std::string dialogText = "";
146 createErrorMessage(expression, message, filename, lineNumber, dialogTitle, dialogText);
147
148 #ifdef G3D_WINDOWS
149 DWORD lastErr = GetLastError();
150 postToClipboard(dialogText.c_str());
151 debugPrintf("\n%s\n", dialogText.c_str());
152 #endif
153
154 const int cBreak = 0;
155 const int cIgnore = 1;
156 const int cAbort = 2;
157
158 static const char* choices[] = {"Debug", "Ignore", "Exit"};
159
160 // Log the error
161 Log::common()->print(std::string("\n**************************\n\n") + dialogTitle + "\n" + dialogText);
162
163 int result = G3D::prompt(dialogTitle.c_str(), dialogText.c_str(), (const char**)choices, 3, useGuiPrompt);
164
165# ifdef G3D_WINDOWS
166 // Put the incoming last error back.
167 SetLastError(lastErr);
168# endif
169
170 switch (result) {
171 // -1 shouldn't actually occur because it means
172 // that we're in release mode.
173 case -1:
174 case cBreak:
175 return true;
176 break;
177
178 case cIgnore:
179 return false;
180 break;
181
182 case cAbort:
183 exit(-1);
184 break;
185 }
186
187 // Should never get here
188 return false;
189}

212
213 static const char* choices[] = {"Ok"};
214
215 const std::string& m =
216 std::string("An internal error has occured in this program and it will now close. "

218 Log::getCommonLogFilename() + "\".\n\n" + dialogText;
219
220 int result = G3D::prompt("Error", m.c_str(), (const char**)choices, 1, useGuiPrompt);
221 (void)result;
222
223 return true;
224}

234void _releaseInputGrab_() {
235 #ifdef G3D_WINDOWS
236
237 GetCursorPos(&oldCursorPos);
238
239 // Stop hiding the cursor if the application hid it.
240 oldShowCursorCount = ShowCursor(true) - 1;
241
242 if (oldShowCursorCount < -1) {
243 for (int c = oldShowCursorCount; c < -1; ++c) {
244 ShowCursor(true);
245 }
246 }
247
248 // Set the default cursor in case the application
249 // set the cursor to NULL.
250 oldCursor = GetCursor();
251 SetCursor(LoadCursor(NULL, IDC_ARROW));
252
253 // Allow the cursor full access to the screen
254 GetClipCursor(&oldCursorRect);
255 ClipCursor(NULL);
256
257 #elif defined(G3D_LINUX)
258#if 0 /* G3DFIX: Disabled to avoid requirement for X11 libraries */
259 if (x11Display != NULL) {
260 XUngrabPointer(x11Display, CurrentTime);
261 XUngrabKeyboard(x11Display, CurrentTime);
262 if (x11Window != 0) {
263 //XUndefineCursor(x11Display, x11Window);
264 // TODO: Note that we leak this cursor; it should be
265 // freed in the restore code.
266 Cursor c = XCreateFontCursor(x11Display, 68);
267 XDefineCursor(x11Display, x11Window, c);
268 }
269 XSync(x11Display, false);
270 XAllowEvents(x11Display, AsyncPointer, CurrentTime);
271 XFlush(x11Display);
272 }
273#endif
274 #elif defined(G3D_OSX)
275 // TODO: OS X
276 #endif
277}

334std::string __cdecl debugPrint(const std::string& s) {
335# ifdef G3D_WINDOWS
336 const int MAX_STRING_LEN = 1024;
337
338 // Windows can't handle really long strings sent to
339 // the console, so we break the string.
340 if (s.size() < MAX_STRING_LEN) {
341 OutputDebugStringA(s.c_str());
342 } else {
343 for (unsigned int i = 0; i < s.size(); i += MAX_STRING_LEN) {
344 std::string sub = s.substr(i, MAX_STRING_LEN);
345 OutputDebugStringA(sub.c_str());
346 }
347 }
348# else
349 fprintf(stderr, "%s", s.c_str());
350 fflush(stderr);
351# endif
352
353 return s;
354}
