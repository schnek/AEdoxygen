31bool DBCFile::open()
32{
33 MPQFile f(filename.c_str());
34 char header[4];
35 unsigned int na,nb,es,ss;
36
37 if(f.read(header,4)!=4) // Number of records
38 return false;
39
40 if(header[0]!='W' || header[1]!='D' || header[2]!='B' || header[3]!='C')
41 return false;
42
43 if(f.read(&na,4)!=4) // Number of records
44 return false;
45 if(f.read(&nb,4)!=4) // Number of fields
46 return false;
47 if(f.read(&es,4)!=4) // Size of a record
48 return false;
49 if(f.read(&ss,4)!=4) // String size
50 return false;
51
52 recordSize = es;
53 recordCount = na;
54 fieldCount = nb;
55 stringSize = ss;
56 if(fieldCount*4 != recordSize)
57 return false;
58
59 data = std::make_unique<unsigned char[]>(recordSize * recordCount + stringSize);
60 stringTable = data.get() + recordSize * recordCount;
61
62 size_t data_size = recordSize*recordCount+stringSize;
63 if(f.read(data.get(), data_size) != data_size)
64 return false;
65 f.close();
66 return true;
67}
