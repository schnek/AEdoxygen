130void System::init() {
131 // NOTE: Cannot use most G3D data structures or utility functions
132 // in here because they are not initialized.
133
134 if (m_initialized) {
135 return;
136 } else {
137 m_initialized = true;
138 }
139
140 getG3DVersion(m_version);
141
142 m_machineEndian = checkEndian();
143
144 m_hasCPUID = checkForCPUID();
145 // Process the CPUID information
146 if (m_hasCPUID) {
147 // We read the standard CPUID level 0x00000000 which should
148 // be available on every x86 processor. This fills out
149 // a string with the processor vendor tag.
150 unsigned int eaxreg = 0, ebxreg = 0, ecxreg = 0, edxreg = 0;
151
152 cpuid(CPUID_VENDOR_ID, eaxreg, ebxreg, ecxreg, edxreg);
153
154 {
155 char c[100];
156 // Then we connect the single register values to the vendor string
157 *((unsigned int*) c) = ebxreg;
158 *((unsigned int*) (c + 4)) = edxreg;
159 *((unsigned int*) (c + 8)) = ecxreg;
160 c[12] = '\0';
161 m_cpuVendor = c;
162 }
163
164 switch (ebxreg) {
165 case 0x756E6547: // GenuineIntel
166 m_cpuArch = "Intel Processor";
167 break;
168
169 case 0x68747541: // AuthenticAMD
170 m_cpuArch = "AMD Processor";
171 break;
172
173 default:
174 m_cpuArch = "Unknown Processor Vendor";
175 break;
176 }
177
178
179 unsigned int highestFunction = eaxreg;
180 if (highestFunction >= CPUID_NUM_CORES) {
181 cpuid(CPUID_NUM_CORES, eaxreg, ebxreg, ecxreg, edxreg);
182 // Number of cores is in (eax>>26) + 1
183 m_numCores = (eaxreg >> 26) + 1;
184 }
185
186 cpuid(CPUID_GET_HIGHEST_FUNCTION, m_highestCPUIDFunction, ebxreg, ecxreg, edxreg);
187 }
188
189
190 // Get the operating system name (also happens to read some other information)
191# ifdef G3D_WINDOWS
192 HRESULT r = OleInitialize(NULL);
193 // Note that this overrides some of the values computed above
194 bool success = RegistryUtil::readInt32
195 ("HKEY_LOCAL_MACHINE\\HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\0",
196 "~MHz", m_cpuSpeed);
197 HRESULT s = OleInitialize(NULL);
198 SYSTEM_INFO systemInfo;
199 GetSystemInfo(&systemInfo);
200 const char* arch = NULL;
201 switch (systemInfo.wProcessorArchitecture) {
202 case PROCESSOR_ARCHITECTURE_INTEL:
203 arch = "x86 Intel";
204 break;
205
206 case PROCESSOR_ARCHITECTURE_AMD64:
207 arch = "x64 Intel/AMD";
208 break;
209
210 case PROCESSOR_ARCHITECTURE_ARM:
211 arch = "ARM";
212 break;
213
214 case PROCESSOR_ARCHITECTURE_ARM64:
215 arch = "ARM64";
216 break;
217
218 default:
219 arch = "Unknown";
220 break;
221 }
222
223 m_numCores = systemInfo.dwNumberOfProcessors;
224 uint64_t maxAddr = reinterpret_cast<uint64_t>(systemInfo.lpMaximumApplicationAddress);
225 {
226 char c[1024];
227 sprintf(c, "%d - %s cores",
228 systemInfo.dwNumberOfProcessors,
229 arch);
230 m_cpuArch = c;
231 }
232
233 OSVERSIONINFO osVersionInfo;
234 osVersionInfo.dwOSVersionInfoSize = sizeof(OSVERSIONINFO);
235 success = GetVersionEx(&osVersionInfo) != 0;
236
237 if (success) {
238 char c[1000];
239 sprintf(c, "Windows %d.%d build %d Platform %d %s",
240 osVersionInfo.dwMajorVersion,
241 osVersionInfo.dwMinorVersion,
242 osVersionInfo.dwBuildNumber,
243 osVersionInfo.dwPlatformId,
244 osVersionInfo.szCSDVersion);
245 m_operatingSystem = c;
246 } else {
247 m_operatingSystem = "Windows";
248 }
249
250# elif defined(G3D_LINUX) || defined(G3D_FREEBSD)
251
252 {
253 // Find the operating system using the 'uname' command
254 FILE* f = popen("uname -a", "r");
255
256 int len = 100;
257 char* r = (char*)::malloc(len * sizeof(char));
258 (void)fgets(r, len, f);
259 // Remove trailing newline
260 if (r[strlen(r) - 1] == '\n') {
261 r[strlen(r) - 1] = '\0';
262 }
263 fclose(f);
264 f = NULL;
265
266 m_operatingSystem = r;
267 ::free(r);
268 }
269
270# elif defined(G3D_OSX)
271
272 // Operating System:
273 SInt32 macVersion;
274 Gestalt(gestaltSystemVersion, &macVersion);
275
276 int major = (macVersion >> 8) & 0xFF;
277 int minor = (macVersion >> 4) & 0xF;
278 int revision = macVersion & 0xF;
279
280 {
281 char c[1000];
282 sprintf(c, "OS X %x.%x.%x", major, minor, revision);
283 m_operatingSystem = c;
284 }
285
286 // Clock Cycle Timing Information:
287 Gestalt('pclk', &m_OSXCPUSpeed);
288 m_cpuSpeed = iRound((double)m_OSXCPUSpeed / (1024 * 1024));
289 m_secondsPerNS = 1.0 / 1.0e9;
290
291 // System Architecture:
292 const NXArchInfo* pInfo = NXGetLocalArchInfo();
293
294 if (pInfo) {
295 m_cpuArch = pInfo->description;
296
297 switch (pInfo->cputype) {
298 case CPU_TYPE_POWERPC:
299 switch(pInfo->cpusubtype){
300 case CPU_SUBTYPE_POWERPC_750:
301 case CPU_SUBTYPE_POWERPC_7400:
302 case CPU_SUBTYPE_POWERPC_7450:
303 m_cpuVendor = "Motorola";
304 break;
305 case CPU_SUBTYPE_POWERPC_970:
306 m_cpuVendor = "IBM";
307 break;
308 }
309 break;
310
311 case CPU_TYPE_I386:
312 m_cpuVendor = "Intel";
313 break;
314 }
315 }
316# endif
317
318 initTime();
319
320 getStandardProcessorExtensions();
321
322 m_appDataDir = FilePath::parent(System::currentProgramFilename());
323}

368#define MARK_LOG()
369//#define MARK_LOG() logPrintf("%s(%d)\n", __FILE__, __LINE__)
370std::string System::findDataFile
371(const std::string& _full,
372 bool errorIfNotFound,
373 bool caseSensitive) {
374MARK_LOG();
375
376 const std::string full = FilePath::expandEnvironmentVariables(_full);
377
378 // First check if the file exists as requested. This will go
379 // through the FileSystemCache, so most calls do not touch disk.
380 if (FileSystem::exists(full, true, caseSensitive)) {
381 return full;
382 }
383
384MARK_LOG();
385 // Now check where we previously found this file.
386 std::string* last = lastFound.getPointer(full);
387 if (last != NULL) {
388 if (FileSystem::exists(*last, true, caseSensitive)) {
389 // Even if cwd has changed the file is still present.
390 // We won't notice if it has been deleted, however.
391 return *last;
392 } else {
393 // Remove this from the cache it is invalid
394 lastFound.remove(full);
395 }
396 }
397
398MARK_LOG();
399
400 const std::string initialAppDataDir(instance().m_appDataDir);
401 const char* g3dPath = getenv("G3D9DATA");
402
403 if (directoryArray.size() == 0) {
404 // Initialize the directory array
405 RealTime t0 = System::time();
406
407 Array<std::string> baseDirArray;
408
409 baseDirArray.append(FileSystem::currentDirectory());
410MARK_LOG();
411 if (! initialAppDataDir.empty()) {
412MARK_LOG();
413 baseDirArray.append(initialAppDataDir);
414 baseDirArray.append(pathConcat(initialAppDataDir, "data"));
415 baseDirArray.append(pathConcat(initialAppDataDir, "data.zip"));
416 } else {
417MARK_LOG();
418 baseDirArray.append("data");
419 baseDirArray.append("data.zip");
420 }
421MARK_LOG();
422
423# ifdef G3D_WINDOWS
424 if (g3dPath == NULL) {
425 // If running the demos under visual studio from the G3D.sln file,
426 // this will locate the data directory.
427 const char* paths[] = {"../data-files/", "../../data-files/", "../../../data-files/", NULL};
428 for (int i = 0; paths[i]; ++i) {
429 if (FileSystem::exists(pathConcat(paths[i], "G3D-DATA-README.TXT"), true, caseSensitive)) {
430 g3dPath = paths[i];
431 break;
432 }
433 }
434 }
435# endif
436
437 if (g3dPath && (initialAppDataDir != g3dPath)) {
438 baseDirArray.append(g3dPath);
439 }
440
441 static const std::string subdirs[] =
442 {"font", "gui", "shader", "model", "cubemap", "icon", "material", "image", "md2", "md3", "ifs", "3ds", "sky", "music", "sound", "scene", ""};
443 for (int j = 0; j < baseDirArray.size(); ++j) {
444 std::string d = baseDirArray[j];
445//logPrintf("%s", d.c_str());
446 if ((d == "") || FileSystem::exists(d)) {
447//logPrintf(" exists\n");
448 directoryArray.append(d);
449 for (int i = 0; ! subdirs[i].empty(); ++i) {
450 const std::string& p = pathConcat(d, subdirs[i]);
451 if (FileSystem::exists(p, true, caseSensitive)) {
452 directoryArray.append(p);
453 }
454 }
455 } else {
456//logPrintf(" does not exist\n");
457 }
458 }
459
460 logLazyPrintf("Initializing System::findDataFile took %fs\n", System::time() - t0);
461
462 }
463MARK_LOG();
464
465 for (int i = 0; i < directoryArray.size(); ++i) {
466 const std::string& p = pathConcat(directoryArray[i], full);
467 if (FileSystem::exists(p, true, caseSensitive)) {
468 lastFound.set(full, p);
469 return p;
470 }
471 }
472MARK_LOG();
473
474 if (errorIfNotFound) {
475 // Generate an error message. Delay this operation until we know that we need it;
476 // otherwise all of the string concatenation would run on each successful find.
477 std::string locations;
478 for (int i = 0; i < directoryArray.size(); ++i) {
479 locations += "\'" + pathConcat(directoryArray[i], full) + "'\n";
480 }
481MARK_LOG();
482
483 std::string msg = "Could not find '" + full + "'.\n\n";
484 msg += " cwd = '" + FileSystem::currentDirectory() + "'\n";
485 if (g3dPath) {
486 msg += " G3D9DATA = '" + std::string(g3dPath) + "'";
487 if (! FileSystem::exists(g3dPath, true, caseSensitive)) {
488 msg += " (illegal path!)";
489 }
490 msg += "\n";
491 } else {
492 msg += " G3D9DATA = (environment variable is not defined)\n";
493 }
494MARK_LOG();
495 msg += " GApp::Settings.dataDir = '" + initialAppDataDir + "'";
496 if (! FileSystem::exists(initialAppDataDir, true, caseSensitive)) {
497 msg += " (illegal path!)";
498 }
499 msg += "\n";
500
501 msg += "\nFilenames tested:\n" + locations;
502
503MARK_LOG();
504logPrintf("%s\n", msg.c_str());
505 throw FileNotFound(full, msg);
506 alwaysAssertM(false, msg);
507 }
508MARK_LOG();
509
510 // Not found
511 return "";
512}
513#undef MARK_LOG

561void System::getStandardProcessorExtensions() {
562#if ! defined(G3D_OSX) || defined(G3D_OSX_INTEL)
563 if (! m_hasCPUID) {
564 return;
565 }
566
567 uint32 eaxreg = 0, ebxreg = 0, ecxreg = 0, features = 0;
568
569 cpuid(CPUID_PROCESSOR_FEATURES, eaxreg, ebxreg, ecxreg, features);
570
571# define checkBit(var, bit) ((var & (1 << bit)) ? true : false)
572
573 m_hasRDTSC = checkBit(features, 4);
574 m_hasMMX = checkBit(features, 23);
575 m_hasSSE = checkBit(features, 25);
576 m_hasSSE2 = checkBit(features, 26);
577 // Bit 28 is HTT; not checked by G3D
578
579 m_hasSSE3 = checkBit(ecxreg, 0);
580
581 if (m_highestCPUIDFunction >= CPUID_EXTENDED_FEATURES) {
582 cpuid(CPUID_EXTENDED_FEATURES, eaxreg, ebxreg, ecxreg, features);
583 m_hasAMDMMX = checkBit(features, 22); // Only on AMD
584 m_has3DNOW = checkBit(features, 31); // Only on AMD
585 m_has3DNOW2 = checkBit(features, 30); // Only on AMD
586 } else {
587 m_hasAMDMMX = false;
588 m_has3DNOW = false;
589 m_has3DNOW2 = false;
590 }
591
592# undef checkBit
593#endif

597// 32-bit
598/** Michael Herf's fast memcpy. Assumes 16-byte alignment */
599void memcpyMMX(void* dst, const void* src, int nbytes) {
600 int remainingBytes = nbytes;
601
602 alwaysAssertM((int)dst % 16 == 0, format("Must be on 16-byte boundary. dst = 0x%x", dst));
603 alwaysAssertM((int)src % 16 == 0, format("Must be on 16-byte boundary. src = 0x%x", src));
604
605 if (nbytes > 64) {
606 _asm {
607 mov esi, src
608 mov edi, dst
609 mov ecx, nbytes
610 shr ecx, 6 // 64 bytes per iteration
611
612 loop1:
613 movq mm1, 0[ESI] // Read in source data
614 movq mm2, 8[ESI]
615 movq mm3, 16[ESI]
616 movq mm4, 24[ESI]
617 movq mm5, 32[ESI]
618 movq mm6, 40[ESI]
619 movq mm7, 48[ESI]
620 movq mm0, 56[ESI]
621
622 movntq 0[EDI], mm1 // Non-temporal stores
623 movntq 8[EDI], mm2
624 movntq 16[EDI], mm3
625 movntq 24[EDI], mm4
626 movntq 32[EDI], mm5
627 movntq 40[EDI], mm6
628 movntq 48[EDI], mm7
629 movntq 56[EDI], mm0
630
631 add esi, 64
632 add edi, 64
633 dec ecx
634 jnz loop1
635
636 emms
637 }
638 remainingBytes -= ((nbytes >> 6) << 6);
639 }
640
641 if (remainingBytes > 0) {
642 // Memcpy the rest
643 memcpy((uint8*)dst + (nbytes - remainingBytes),
644 (const uint8*)src + (nbytes - remainingBytes), remainingBytes);
645 }
646}
647#endif

663/** Michael Herf's fastest memset. n32 must be filled with the same
664 character repeated. */
665#if defined(G3D_WINDOWS) && defined(_M_IX86) && !defined(__MINGW32__) /* G3DFIX: Don't check if on 64-bit Windows platforms or using MinGW */
666// On x86 processors, use MMX
667void memfill(void *dst, int n32, unsigned long i) {
668
669 int originalSize = i;
670 int bytesRemaining = i;
671
672 if (i > 16) {
673
674 bytesRemaining = i % 16;
675 i -= bytesRemaining;
676 __asm {
677 movq mm0, n32
678 punpckldq mm0, mm0
679 mov edi, dst
680
681 loopwrite:
682
683 movntq 0[edi], mm0
684 movntq 8[edi], mm0
685
686 add edi, 16
687 sub i, 16
688 jg loopwrite
689
690 emms
691 }
692 }
693
694 if (bytesRemaining > 0) {
695 ::memset((uint8*)dst + (originalSize - bytesRemaining), n32, bytesRemaining);
696 }
697}

717/** Removes the 'd' that icompile / Morgan's VC convention appends. */
718static std::string computeAppName(const std::string& start) {
719 if (start.size() < 2) {
720 return start;
721 }
722
723 if (start[start.size() - 1] == 'd') {
724 // Maybe remove the 'd'; see if ../ or ../../ has the same name
725 char tmp[1024];
726 (void)getcwd(tmp, sizeof(tmp));
727 std::string drive, base, ext;
728 Array<std::string> path;
729 parseFilename(tmp, drive, path, base, ext);
730
731 std::string shortName = start.substr(0, start.size() - 1);
732
733 if ((path.size() > 1) && (toLower(path.last()) == toLower(shortName))) {
734 return shortName;
735 }
736
737 if ((path.size() > 2) && (toLower(path[path.size() - 2]) == toLower(shortName))) {
738 return shortName;
739 }
740 }
741
742 return start;
743}

771 // filename will contain a newline. Overwrite it:
772 filename[s - 1] = '\0';
773 pclose(fd);
774 }
775# else
776 {
777 int ret = readlink("/proc/self/exe", filename, sizeof(filename));
778
779 // In case of an error, leave the handling up to the caller
780 if (ret == -1) {
781 return "";
782 }
783
784 debugAssert((int)sizeof(filename) > ret);
785
786 // Ensure proper NULL termination
787 filename[ret] = 0;
788 }
789 #endif
790
791 return filename;
792}

795void System::sleep(RealTime t) {
796
797 // Overhead of calling this function, measured from a previous run.
798 static const RealTime OVERHEAD = 0.00006f;
799
800 RealTime now = time();
801 RealTime wakeupTime = now + t - OVERHEAD;
802
803 RealTime remainingTime = wakeupTime - now;
804 RealTime sleepTime = 0;
805
806 // On Windows, a "time slice" is measured in quanta of 3-5 ms (http://support.microsoft.com/kb/259025)
807 // Sleep(0) yields the remainder of the time slice, which could be a long time.
808 // A 1 ms minimum time experimentally kept the "Empty GApp" at nearly no CPU load at 100 fps,
809 // yet nailed the frame timing perfectly.
810 static RealTime minRealSleepTime = 3 * units::milliseconds();
811
812 while (remainingTime > 0) {
813
814 if (remainingTime > minRealSleepTime * 2.5) {
815 // Safe to use Sleep with a time... sleep for half the remaining time
816 sleepTime = max(remainingTime * 0.5, 0.0005);
817 } else if (remainingTime > minRealSleepTime) {
818 // Safe to use Sleep with a zero time;
819 // causes the program to yield only
820 // the current time slice, and then return.
821 sleepTime = 0;
822 } else {
823 // Not safe to use Sleep; busy wait
824 sleepTime = -1;
825 }
826
827 if (sleepTime >= 0) {
828 #ifdef G3D_WINDOWS
829 // Translate to milliseconds
830 Sleep((int)(sleepTime * 1e3));
831 #else
832 // Translate to microseconds
833 usleep((int)(sleepTime * 1e6));
834 #endif
835 }
836
837 now = time();
838 remainingTime = wakeupTime - now;
839 }
840}

964#define REALPTR_TO_USERPTR(x) ((uint8*)(x) + sizeof(size_t))
965#define USERPTR_TO_REALPTR(x) ((uint8*)(x) - sizeof(size_t))
966#define USERSIZE_TO_REALSIZE(x) ((x) + sizeof(size_t))
967#define REALSIZE_FROM_USERPTR(u) (*(size_t*)USERPTR_TO_REALPTR(ptr) + sizeof(size_t))
968#define USERSIZE_FROM_USERPTR(u) (*(size_t*)USERPTR_TO_REALPTR(ptr))

970class BufferPool {
971public:
972
973 /** Only store buffers up to these sizes (in bytes) in each pool->
974 Different pools have different management strategies.
975
976 A large block is preallocated for tiny buffers; they are used with
977 tremendous frequency. Other buffers are allocated as demanded.
978 Tiny buffers are 128 bytes long because that seems to align well with
979 cache sizes on many machines.
980 */
981#ifdef G3D_64BIT
982 // 64-bit machines have larger pointers...and probably have more memory as well
983 enum {tinyBufferSize = 256, smallBufferSize = 2048, medBufferSize = 8192};
984#else
985 enum {tinyBufferSize = 128, smallBufferSize = 1024, medBufferSize = 4096};
986#endif
987
988 /**
989 Most buffers we're allowed to store.
990 250000 * { 128 | 256} = {32 | 64} MB (preallocated)
991 40000 * {1024 | 2048} = {40 | 80} MB (allocated on demand)
992 5000 * {4096 | 8192} = {20 | 40} MB (allocated on demand)
993 */
994 enum {maxTinyBuffers = 250000, maxSmallBuffers = 40000, maxMedBuffers = 5000};
995
996private:
997
998 /** Pointer given to the program. Unless in the tiny heap, the user size of the block is stored right in front of the pointer as a uint32.*/
999 typedef void* UserPtr;
1000
1001 /** Actual block allocated on the heap */
1002 typedef void* RealPtr;
1003
1004 class MemBlock {
1005 public:
1006 UserPtr ptr;
1007 size_t bytes;
1008
1009 inline MemBlock() : ptr(NULL), bytes(0) {}
1010 inline MemBlock(UserPtr p, size_t b) : ptr(p), bytes(b) {}
1011 };
1012
1013 MemBlock smallPool[maxSmallBuffers];
1014 int smallPoolSize;
1015
1016 MemBlock medPool[maxMedBuffers];
1017 int medPoolSize;
1018
1019 /** The tiny pool is a single block of storage into which all tiny
1020 objects are allocated. This provides better locality for
1021 small objects and avoids the search time, since all tiny
1022 blocks are exactly the same size. */
1023 void* tinyPool[maxTinyBuffers];
1024 int tinyPoolSize;
1025
1026 /** Pointer to the data in the tiny pool */
1027 void* tinyHeap;
1028
1029 Spinlock m_lock;
1030
1031 void lock() {
1032 m_lock.lock();
1033 }
1034
1035 void unlock() {
1036 m_lock.unlock();
1037 }
1038
1039 /**
1040 Malloc out of the tiny heap. Returns NULL if allocation failed.
1041 */
1042 inline UserPtr tinyMalloc(size_t bytes) {
1043 // Note that we ignore the actual byte size
1044 // and create a constant size block.
1045 (void)bytes;
1046 assert(tinyBufferSize >= bytes);
1047
1048 UserPtr ptr = NULL;
1049
1050 if (tinyPoolSize > 0) {
1051 --tinyPoolSize;
1052
1053 // Return the old last pointer from the freelist
1054 ptr = tinyPool[tinyPoolSize];
1055
1056# ifdef G3D_DEBUG
1057 if (tinyPoolSize > 0) {
1058 assert(tinyPool[tinyPoolSize - 1] != ptr);
1059 // "System::malloc heap corruption detected: "
1060 // "the last two pointers on the freelist are identical (during tinyMalloc).");
1061 }
1062# endif
1063
1064 // NULL out the entry to help detect corruption
1065 tinyPool[tinyPoolSize] = NULL;
1066 }
1067
1068 return ptr;
1069 }
1070
1071 /** Returns true if this is a pointer into the tiny heap. */
1072 bool inTinyHeap(UserPtr ptr) {
1073 return
1074 (ptr >= tinyHeap) &&
1075 (ptr < (uint8*)tinyHeap + maxTinyBuffers * tinyBufferSize);
1076 }
1077
1078 void tinyFree(UserPtr ptr) {
1079 assert(ptr);
1080 assert(tinyPoolSize < maxTinyBuffers);
1081 // "Tried to free a tiny pool buffer when the tiny pool freelist is full.");
1082
1083# ifdef G3D_DEBUG
1084 if (tinyPoolSize > 0) {
1085 UserPtr prevOnHeap = tinyPool[tinyPoolSize - 1];
1086 assert(prevOnHeap != ptr);
1087// "System::malloc heap corruption detected: "
1088// "the last two pointers on the freelist are identical (during tinyFree).");
1089 }
1090# endif
1091
1092 assert(tinyPool[tinyPoolSize] == NULL);
1093
1094 // Put the pointer back into the free list
1095 tinyPool[tinyPoolSize] = ptr;
1096 ++tinyPoolSize;
1097
1098 }

1100 void flushPool(MemBlock* pool, int& poolSize) {
1101 for (int i = 0; i < poolSize; ++i) {
1102 bytesAllocated -= USERSIZE_TO_REALSIZE(pool[i].bytes);
1103 ::free(USERPTR_TO_REALPTR(pool[i].ptr));
1104 pool[i].ptr = NULL;
1105 pool[i].bytes = 0;
1106 }
1107 poolSize = 0;
1108 }

1111 /** Allocate out of a specific pool. Return NULL if no suitable
1112 memory was found. */
1113 UserPtr malloc(MemBlock* pool, int& poolSize, size_t bytes) {
1114
1115 // OPT: find the smallest block that satisfies the request.
1116
1117 // See if there's something we can use in the buffer pool.
1118 // Search backwards since usually we'll re-use the last one.
1119 for (int i = (int)poolSize - 1; i >= 0; --i) {
1120 if (pool[i].bytes >= bytes) {
1121 // We found a suitable entry in the pool.
1122
1123 // No need to offset the pointer; it is already offset
1124 UserPtr ptr = pool[i].ptr;
1125
1126 // Remove this element from the pool, replacing it with
1127 // the one from the end (same as Array::fastRemove)
1128 --poolSize;
1129 pool[i] = pool[poolSize];
1130
1131 return ptr;
1132 }
1133 }
1134
1135 return NULL;
1136 }

1138public:
1139
1140 /** Count of memory allocations that have occurred. */
1141 int totalMallocs;
1142 int mallocsFromTinyPool;
1143 int mallocsFromSmallPool;
1144 int mallocsFromMedPool;
1145
1146 /** Amount of memory currently allocated (according to the application).
1147 This does not count the memory still remaining in the buffer pool,
1148 but does count extra memory required for rounding off to the size
1149 of a buffer.
1150 Primarily useful for detecting leaks.*/
1151 // TODO: make me an atomic int!
1152 volatile size_t bytesAllocated;

1154 BufferPool() {
1155 totalMallocs = 0;
1156
1157 mallocsFromTinyPool = 0;
1158 mallocsFromSmallPool = 0;
1159 mallocsFromMedPool = 0;
1160
1161 bytesAllocated = 0;
1162
1163 tinyPoolSize = 0;
1164 tinyHeap = NULL;
1165
1166 smallPoolSize = 0;
1167
1168 medPoolSize = 0;
1169

1171 // Initialize the tiny heap as a bunch of pointers into one
1172 // pre-allocated buffer.
1173 tinyHeap = ::malloc(maxTinyBuffers * tinyBufferSize);
1174 for (int i = 0; i < maxTinyBuffers; ++i) {
1175 tinyPool[i] = (uint8*)tinyHeap + (tinyBufferSize * i);
1176 }
1177 tinyPoolSize = maxTinyBuffers;
1178
1179#if 0 ///---------------------------------- old mutex
1180# ifdef G3D_WINDOWS

1182# else

1184# endif
1185#endif ///---------------------------------- old mutex

1196# else

1198# endif
1199#endif //--------------------------------old mutex

1203 UserPtr realloc(UserPtr ptr, size_t bytes) {
1204 if (ptr == NULL) {
1205 return malloc(bytes);
1206 }
1207
1208 if (inTinyHeap(ptr)) {
1209 if (bytes <= tinyBufferSize) {
1210 // The old pointer actually had enough space.
1211 return ptr;
1212 } else {
1213 // Free the old pointer and malloc
1214
1215 UserPtr newPtr = malloc(bytes);
1216 System::memcpy(newPtr, ptr, tinyBufferSize);
1217 tinyFree(ptr);
1218 return newPtr;
1219
1220 }
1221 } else {
1222 // In one of our heaps.
1223
1224 // See how big the block really was
1225 size_t userSize = USERSIZE_FROM_USERPTR(ptr);
1226 if (bytes <= userSize) {
1227 // The old block was big enough.
1228 return ptr;
1229 }
1230
1231 // Need to reallocate and move
1232 UserPtr newPtr = malloc(bytes);
1233 System::memcpy(newPtr, ptr, userSize);
1234 free(ptr);
1235 return newPtr;
1236 }
1237 }

1239
1240 UserPtr malloc(size_t bytes) {
1241 lock();
1242 ++totalMallocs;
1243
1244 if (bytes <= tinyBufferSize) {
1245
1246 UserPtr ptr = tinyMalloc(bytes);
1247
1248 if (ptr) {
1249 ++mallocsFromTinyPool;
1250 unlock();
1251 return ptr;
1252 }
1253
1254 }
1255
1256 // Failure to allocate a tiny buffer is allowed to flow
1257 // through to a small buffer
1258 if (bytes <= smallBufferSize) {
1259
1260 UserPtr ptr = malloc(smallPool, smallPoolSize, bytes);
1261
1262 if (ptr) {
1263 ++mallocsFromSmallPool;
1264 unlock();
1265 return ptr;
1266 }
1267
1268 } else if (bytes <= medBufferSize) {
1269 // Note that a small allocation failure does *not* fall
1270 // through into a medium allocation because that would
1271 // waste the medium buffer's resources.
1272
1273 UserPtr ptr = malloc(medPool, medPoolSize, bytes);
1274
1275 if (ptr) {
1276 ++mallocsFromMedPool;
1277 unlock();
1278 debugAssertM(ptr != NULL, "BufferPool::malloc returned NULL");
1279 return ptr;
1280 }
1281 }
1282
1283 bytesAllocated += USERSIZE_TO_REALSIZE(bytes);
1284 unlock();
1285
1286 // Heap allocate
1287
1288 // Allocate 4 extra bytes for our size header (unfortunate,
1289 // since malloc already added its own header).
1290 RealPtr ptr = ::malloc(USERSIZE_TO_REALSIZE(bytes));
1291
1292 if (ptr == NULL) {
1293# ifdef G3D_WINDOWS
1294 // Check for memory corruption
1295 alwaysAssertM(_CrtCheckMemory() == TRUE, "Heap corruption detected.");
1296# endif
1297
1298 // Flush memory pools to try and recover space
1299 flushPool(smallPool, smallPoolSize);
1300 flushPool(medPool, medPoolSize);
1301 ptr = ::malloc(USERSIZE_TO_REALSIZE(bytes));
1302 }
1303
1304 if (ptr == NULL) {
1305 if ((System::outOfMemoryCallback() != NULL) &&
1306 (System::outOfMemoryCallback()(USERSIZE_TO_REALSIZE(bytes), true) == true)) {
1307 // Re-attempt the malloc
1308 ptr = ::malloc(USERSIZE_TO_REALSIZE(bytes));
1309 }
1310 }
1311
1312 if (ptr == NULL) {
1313 if (System::outOfMemoryCallback() != NULL) {
1314 // Notify the application
1315 System::outOfMemoryCallback()(USERSIZE_TO_REALSIZE(bytes), false);
1316 }
1317# ifdef G3D_DEBUG
1318 debugPrintf("::malloc(%d) returned NULL\n", (int)USERSIZE_TO_REALSIZE(bytes));
1319# endif
1320 debugAssertM(ptr != NULL,
1321 "::malloc returned NULL. Either the "
1322 "operating system is out of memory or the "
1323 "heap is corrupt.");
1324 return NULL;
1325 }
1326
1327 ((size_t*)ptr)[0] = bytes;
1328
1329 return REALPTR_TO_USERPTR(ptr);
1330 }

1332
1333 void free(UserPtr ptr) {
1334 if (ptr == NULL) {
1335 // Free does nothing on null pointers
1336 return;
1337 }
1338
1339 assert(isValidPointer(ptr));
1340
1341 if (inTinyHeap(ptr)) {
1342 lock();
1343 tinyFree(ptr);
1344 unlock();
1345 return;
1346 }
1347
1348 size_t bytes = USERSIZE_FROM_USERPTR(ptr);
1349
1350 lock();
1351 if (bytes <= smallBufferSize) {
1352 if (smallPoolSize < maxSmallBuffers) {
1353 smallPool[smallPoolSize] = MemBlock(ptr, bytes);
1354 ++smallPoolSize;
1355 unlock();
1356 return;
1357 }
1358 } else if (bytes <= medBufferSize) {
1359 if (medPoolSize < maxMedBuffers) {
1360 medPool[medPoolSize] = MemBlock(ptr, bytes);
1361 ++medPoolSize;
1362 unlock();
1363 return;
1364 }
1365 }
1366 bytesAllocated -= USERSIZE_TO_REALSIZE(bytes);
1367 unlock();
1368
1369 // Free; the buffer pools are full or this is too big to store.
1370 ::free(USERPTR_TO_REALPTR(ptr));
1371 }

1488void* System::alignedMalloc(size_t bytes, size_t alignment) {
1489
1490 alwaysAssertM(isPow2((uint32)alignment), "alignment must be a power of 2");
1491
1492 // We must align to at least a word boundary.
1493 alignment = max(alignment, sizeof(void *));
1494
1495 // Pad the allocation size with the alignment size and the size of
1496 // the redirect pointer. This is the worst-case size we'll need.
1497 // Since the alignment size is at least teh word size, we don't
1498 // need to allocate space for the redirect pointer. We repeat the max here
1499 // for clarity.
1500 size_t totalBytes = bytes + max(alignment, sizeof(void*));
1501
1502 size_t truePtr = (size_t)System::malloc(totalBytes);
1503
1504 if (truePtr == 0) {
1505 // malloc returned NULL
1506 return NULL;
1507 }
1508
1509 debugAssert(isValidHeapPointer((void*)truePtr));
1510 #ifdef G3D_WINDOWS
1511 // The blocks we return will not be valid Win32 debug heap
1512 // pointers because they are offset
1513 // debugAssert(_CrtIsValidPointer((void*)truePtr, totalBytes, TRUE) );
1514 #endif
1515
1516
1517 // We want alignedPtr % alignment == 0, which we'll compute with a
1518 // binary AND because 2^n - 1 has the form 1111... in binary.
1519 const size_t bitMask = (alignment - 1);
1520
1521 // The return pointer will be the next aligned location that is at
1522 // least sizeof(void*) after the true pointer. We need the padding
1523 // to have a place to write the redirect pointer.
1524 size_t alignedPtr = truePtr + sizeof(void*);
1525
1526 const size_t remainder = alignedPtr & bitMask;
1527
1528 // Add what we need to make it to the next alignment boundary, but
1529 // if the remainder was zero, let it wrap to zero and don't add
1530 // anything.
1531 alignedPtr += ((alignment - remainder) & bitMask);
1532
1533 debugAssert((alignedPtr & bitMask) == 0);
1534 debugAssert((alignedPtr - truePtr + bytes) <= totalBytes);
1535
1536 // Immediately before the aligned location, write the true array location
1537 // so that we can free it correctly.
1538 size_t* redirectPtr = (size_t *)(alignedPtr - sizeof(void *));
1539 redirectPtr[0] = truePtr;
1540
1541 debugAssert(isValidHeapPointer((void*)truePtr));
1542
1543 #if defined(G3D_WINDOWS) && defined(G3D_DEBUG)
1544 if (bytes < 0xFFFFFFFF) {
1545 debugAssert( _CrtIsValidPointer((void*)alignedPtr, (int)bytes, TRUE) );
1546 }
1547 #endif
1548 return (void *)alignedPtr;
1549}

1552void System::alignedFree(void* _ptr) {
1553 if (_ptr == NULL) {
1554 return;
1555 }
1556
