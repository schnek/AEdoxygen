19/**
20 Locate the indices of the break between of the two
21 sections of the circular queue. These are used to
22 construct two for loops that iterate over the whole
23 sequence without using the modulo operator.
24
25 [0 ... secondEnd) [head .... firstEnd)
26
27 \sa ThreadsafeQueue
28 */
29#define FIND_ENDS \
30 int firstEnd = head + num;\
31 int secondEnd = 0;\
32 if (firstEnd > numAllocated) {\
33 secondEnd = firstEnd - numAllocated;\
34 firstEnd = numAllocated;\
35 }

38/**
39 Dynamic queue that uses a circular buffer for performance.
40
41 Faster than std::deque for objects with constructors.
42 */

44class Queue {
45private:

75 /** If a clear was needed, assumes it already occured */
76 void _copy(const Queue& other) {
77 debugAssert(data == NULL);
78 data = (T*)System::malloc(sizeof(T) * other.numAllocated);
79 debugAssert(data);
80 head = other.head;
81 num = other.num;
82 numAllocated = other.numAllocated;
83
84 FIND_ENDS;
85
86 for (int i = head; i < firstEnd; ++i) {
87 new (data + i)T(other.data[i]);
88 }
89
90 for (int i = 0; i < secondEnd; ++i) {
91 new (data + i)T(other.data[i]);
92 }
93 }

96 /**
97 Computes a data array index from a queue position. The queue position
98 may be negative.
99 */
100 inline int index(int i) const {
101 return (head + i + numAllocated) % numAllocated;
102 }

104 /**
105 Allocates newSize elements and repacks the array.
106 */
107 void repackAndRealloc(int newSize) {
108 // TODO: shrink queue
109 T* old = data;
110 data = (T*)System::malloc(newSize * sizeof(T));
111 debugAssert(data != NULL);
112
113 FIND_ENDS;
114
115 int j = 0;
116 for (int i = head; i < firstEnd; ++i, ++j) {
117 new (data + j)T(old[i]);
118 (old + i)->~T();
119 }
120
121 for (int i = 0; i < secondEnd; ++i, ++j) {
122 new (data + j)T(old[i]);
123 (old + i)->~T();
124 }
125
126 head = 0;
127 System::free(old);
128 numAllocated = newSize;
129 }

131 /**
132 Ensure that there is at least one element between
133 the tail and head, wrapping around in the circular
134 buffer.
135 */
136 inline void reserveSpace() {
137 if (num == numAllocated) {
138 repackAndRealloc(numAllocated * 3 + 20);
139 }
140 }

170 /**
171 Insert a new element into the front of the queue
172 (a traditional queue only uses pushBack).
173 */
174 inline void pushFront(const T& e) {
175 reserveSpace();
176
177 // Get the index of head-1
178 int i = index(-1);
179
180 // Call the constructor on the newly exposed element.
181 new (data + i)T(e);
182
183 // Reassign the head to point to this index
184 head = i;
185 ++num;
186 }

191 inline void pushBack(const T& e) {
192 reserveSpace();
193
194 // Get the index of 1+tail
195 int i = index(num);
196
197 // Initialize that element
198 new (data + i)T(e);

210 /**
211 Remove the last element from the queue. The queue will never
212 shrink in size. (A typical queue only uses popFront).
213 */
214 inline T popBack() {
215 int tail = index(num - 1);
216 T result(data[tail]);
217
218 // Call the destructor
219 (data + tail)->~T();
220 --num;
221
222 return result;
223 }
224
225 /**
226 Remove the next element from the head of the queue. The queue will never
227 shrink in size. */
228 inline T popFront() {
229 T result(data[head]);
230 // Call the destructor
231 (data + head)->~T();
232 head = (head + 1) % numAllocated;
233 --num;
234 return result;
235 }

245 /**
246 Removes all elements (invoking their destructors).
247
248 @param freeStorage If false, the underlying array is not deallocated
249 (allowing fast push in the future), however, the size of the Queue
250 is reported as zero.
251
252 */
253 void clear(bool freeStorage = true) {
254
255 FIND_ENDS;
256
257 // Invoke the destructors on the elements
258 int i;
259 for (i = head; i < firstEnd; ++i) {
260 (data + i)->~T();
261 }
262
263 for (i = 0; i < secondEnd; ++i) {
264 (data + i)->~T();
265 }
266
267 num = 0;
268 head = 0;
269 if (freeStorage) {
270 numAllocated = 0;
271 System::free(data);
272 data = NULL;
273 }
274 }
