43bool FileLoader::loadFile(std::string const& fileName, bool log)
44{
45 free();
46 MPQFile mf(fileName.c_str());
47 if(mf.isEof())
48 {
49 if (log)
50 printf("No such file %s\n", fileName.c_str());
51 return false;
52 }
53
54 data_size = mf.getSize();
55
56 data = std::make_unique<uint8_t[]>(data_size);
57 mf.read(data.get(), data_size);
58 mf.close();
59 if (prepareLoadedData())
60 return true;
61
62 printf("Error loading %s", fileName.c_str());
63 mf.close();
64 free();
65 return false;
66}

68bool FileLoader::prepareLoadedData()
69{
70 // Check version
71 version = (file_MVER *) data.get();
72 if (version->fcc != MverMagic.fcc)
73 return false;
74 if (version->ver != FILE_FORMAT_VERSION)
75 return false;
76 return true;
77}
