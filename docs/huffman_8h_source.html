42/* input stream for huffman decompression. */
43struct huffman_input_stream_s {
44 uint8_t *in_buf; /* 00 - input data. */
45 uint32_t bit_buf; /* 04 - input bit buffer. */
46 uint32_t bits; /* 08 - number of bits remaining in byte. */
47};
48
49/* huffman tree item. */
50struct huffman_tree_item_s {
51 struct huffman_tree_item_s *next; /* 00 - pointer to next huffman tree item. */
52 struct huffman_tree_item_s *prev; /* 04 - pointer to prev huffman tree item (< 0 if none). */
53 uint32_t dcmp_byte; /* 08 - index of this item in item pointer array, decompressed byte value. */
54 uint32_t byte_value; /* 0C - some byte value. */
55 struct huffman_tree_item_s *parent; /* 10 - pointer to parent huffman tree item (NULL if none). */
56 struct huffman_tree_item_s *child; /* 14 - pointer to child huffman tree item. */
57};
58
59/* structure used for quick decompression. */
60struct huffman_decompress_s {
61 uint32_t offs00; /* 00 - 1 if resolved. */
62 uint32_t bits; /* 04 - bit count. */
63 union {
64 uint32_t dcmp_byte; /* 08 - byte value for decompress (if bitCount <= 7). */
65 struct huffman_tree_item_s *p_item; /* 08 - huffman tree item (if number of bits is greater than 7). */
66 };
67};
68
69/* structure for huffman tree. */
70struct huffman_tree_s {
71 uint32_t cmp0; /* 0000 - 1 if compression type 0. */
72 uint32_t offs0004; /* 0004 - some flag. */
73 struct huffman_tree_item_s items0008[0x203]; /* 0008 - huffman tree items. */
74 struct huffman_tree_item_s *item3050; /* 3050 - always NULL? */
75 struct huffman_tree_item_s *item3054; /* 3054 - pointer to huffman tree item. */
76 struct huffman_tree_item_s *item3058; /* 3058 - pointer to huffman tree item (< 0 if invalid). */
77 struct huffman_tree_item_s *item305C; /* 305C - usually NULL. */
78 struct huffman_tree_item_s *first; /* 3060 - pointer to top (first) huffman tree item. */
79 struct huffman_tree_item_s *last; /* 3064 - pointer to bottom (last) huffman tree item (< 0 if invalid). */
80 uint32_t items; /* 3068 - number of used huffman tree items. */
81 struct huffman_tree_item_s *items306C[0x102]; /* 306C - huffman tree item pointer array. */
82 struct huffman_decompress_s qd3474[0x80]; /* 3474 - array for quick decompression. */
83 uint8_t table_1502A630[]; /* some table to make struct size flexible. */
84};
