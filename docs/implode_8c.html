538 }

558 OutputBits(pWork, pWork->dsize_bits, pWork->dsize_mask & pWork->distance);
559 }

562 input_data += rep_length;

567 // just output the 9-bit literal for the one character
568 OutputBits(pWork, pWork->nChBits[*input_data], pWork->nChCodes[*input_data]);
569 input_data++;

571 }

574 {
575 input_data -= 0x1000;
576 memmove(pWork->work_buff, pWork->work_buff + 0x1000, pWork->dsize_bytes + 0x204);
577 }
578 }

585 pWork->out_bytes++;
