194/*
195** Barrier that moves collector forward, that is, marks the white object
196** 'v' being pointed by the black object 'o'. In the generational
197** mode, 'v' must also become old, if 'o' is old; however, it cannot
198** be changed directly to OLD, because it may still point to non-old
199** objects. So, it is marked as OLD0. In the next cycle it will become
200** OLD1, and in the next it will finally become OLD (regular old). By
201** then, any object it points to will also be old. If called in the
202** incremental sweep phase, it clears the black object to white (sweep
203** it) to avoid other barrier calls for this same object. (That cannot
204** be done is generational mode, as its sweep does not distinguish
205** whites from deads.)
206*/
207void luaC_barrier_ (lua_State *L, GCObject *o, GCObject *v) {
208 global_State *g = G(L);
209 lua_assert(isblack(o) && iswhite(v) && !isdead(g, v) && !isdead(g, o));
210 if (keepinvariant(g)) { /* must keep invariant? */
211 reallymarkobject(g, v); /* restore invariant */
212 if (isold(o)) {
213 lua_assert(!isold(v)); /* white object could not be old */
214 setage(v, G_OLD0); /* restore generational invariant */
215 }
216 }
217 else { /* sweep phase */
218 lua_assert(issweepphase(g));
219 if (g->gckind == KGC_INC) /* incremental mode? */
220 makewhite(g, o); /* mark 'o' as white to avoid other barriers */
221 }
222}

278/*
279** Mark an object. Userdata with no user values, strings, and closed
280** upvalues are visited and turned black here. Open upvalues are
281** already indirectly linked through their respective threads in the
282** 'twups' list, so they don't go to the gray list; nevertheless, they
283** are kept gray to avoid barriers, as their values will be revisited
284** by the thread or by 'remarkupvals'. Other objects are added to the
285** gray list to be visited (and turned black) later. Both userdata and
286** upvalues can call this function recursively, but this recursion goes
287** for at most two levels: An upvalue cannot refer to another upvalue
288** (only closures can), and a userdata's metatable must be a table.
289*/
290static void reallymarkobject (global_State *g, GCObject *o) {
291 switch (o->tt) {
292 case LUA_VSHRSTR:
293 case LUA_VLNGSTR: {
294 set2black(o); /* nothing to visit */
295 break;
296 }
297 case LUA_VUPVAL: {
298 UpVal *uv = gco2upv(o);
299 if (upisopen(uv))
300 set2gray(uv); /* open upvalues are kept gray */
301 else
302 set2black(uv); /* closed upvalues are visited here */
303 markvalue(g, uv->v); /* mark its content */
304 break;
305 }
306 case LUA_VUSERDATA: {
307 Udata *u = gco2u(o);
308 if (u->nuvalue == 0) { /* no user values? */
309 markobjectN(g, u->metatable); /* mark its metatable */
310 set2black(u); /* nothing else to mark */
311 break;
312 }
313 /* else... */
314 } /* FALLTHROUGH */
315 case LUA_VLCL: case LUA_VCCL: case LUA_VTABLE:
316 case LUA_VTHREAD: case LUA_VPROTO: {
317 linkobjgclist(o, g->gray); /* to be visited later */
318 break;
319 }
320 default: lua_assert(0); break;
321 }
322}

349/*
350** For each non-marked thread, simulates a barrier between each open
351** upvalue and its value. (If the thread is collected, the value will be
352** assigned to the upvalue, but then it can be too late for the barrier
353** to act. The "barrier" does not need to check colors: A non-marked
354** thread must be young; upvalues cannot be older than their threads; so
355** any visited upvalue must be young too.) Also removes the thread from
356** the list, as it was already visited. Removes also threads with no
357** upvalues, as they have nothing to be checked. (If the thread gets an
358** upvalue later, it will be linked in the list again.)
359*/
360static int remarkupvals (global_State *g) {
361 lua_State *thread;
362 lua_State **p = &g->twups;
363 int work = 0; /* estimate of how much work was done here */
364 while ((thread = *p) != NULL) {
365 work++;
366 if (!iswhite(thread) && thread->openupval != NULL)
367 p = &thread->twups; /* keep marked thread with upvalues in the list */
368 else { /* thread is not marked or without upvalues */
369 UpVal *uv;
370 lua_assert(!isold(thread) || thread->openupval == NULL);
371 *p = thread->twups; /* remove thread from the list */
372 thread->twups = thread; /* mark that it is out of list */
373 for (uv = thread->openupval; uv != NULL; uv = uv->u.open.next) {
374 lua_assert(getage(uv) <= getage(thread));
375 work++;
376 if (!iswhite(uv)) { /* upvalue already visited? */
377 lua_assert(upisopen(uv) && isgray(uv));
378 markvalue(g, uv->v); /* mark its value */
379 }
380 }
381 }
382 }
383 return work;
384}

434** Traverse a table with weak values and link it to proper list. During
435** propagate phase, keep it in 'grayagain' list, to be revisited in the
436** atomic phase. In the atomic phase, if table has any white value,
437** put it in 'weak' list, to be cleared.

439static void traverseweakvalue (global_State *g, Table *h) {
440 Node *n, *limit = gnodelast(h);
441 /* if there is array part, assume it may have white values (it is not
442 worth traversing it now just to check) */
443 int hasclears = (h->alimit > 0);
444 for (n = gnode(h, 0); n < limit; n++) { /* traverse hash part */
445 if (isempty(gval(n))) /* entry is empty? */
446 clearkey(n); /* clear its key */
447 else {
448 lua_assert(!keyisnil(n));
449 markkey(g, n);
450 if (!hasclears && iscleared(g, gcvalueN(gval(n)))) /* a white value? */
451 hasclears = 1; /* table will have to be cleared */
452 }
453 }
454 if (g->gcstate == GCSatomic && hasclears)
455 linkgclist(h, g->weak); /* has to be cleared later */
456 else
457 linkgclist(h, g->grayagain); /* must retraverse it in atomic phase */

461/*
462** Traverse an ephemeron table and link it to proper list. Returns true
463** iff any object was marked during this traversal (which implies that
464** convergence has to continue). During propagation phase, keep table
465** in 'grayagain' list, to be visited again in the atomic phase. In
466** the atomic phase, if table has any white->white entry, it has to
467** be revisited during ephemeron convergence (as that key may turn
468** black). Otherwise, if it has any white key, table has to be cleared
469** (in the atomic phase). In generational mode, some tables
470** must be kept in some gray list for post-processing; this is done
471** by 'genlink'.
472*/
473static int traverseephemeron (global_State *g, Table *h, int inv) {
474 int marked = 0; /* true if an object is marked in this traversal */
475 int hasclears = 0; /* true if table has white keys */
476 int hasww = 0; /* true if table has entry "white-key -> white-value" */
477 unsigned int i;
478 unsigned int asize = luaH_realasize(h);
479 unsigned int nsize = sizenode(h);
480 /* traverse array part */
481 for (i = 0; i < asize; i++) {
482 if (valiswhite(&h->array[i])) {
483 marked = 1;
484 reallymarkobject(g, gcvalue(&h->array[i]));
485 }
486 }
487 /* traverse hash part; if 'inv', traverse descending
488 (see 'convergeephemerons') */
489 for (i = 0; i < nsize; i++) {
490 Node *n = inv ? gnode(h, nsize - 1 - i) : gnode(h, i);
491 if (isempty(gval(n))) /* entry is empty? */
492 clearkey(n); /* clear its key */
493 else if (iscleared(g, gckeyN(n))) { /* key is not marked (yet)? */
494 hasclears = 1; /* table must be cleared */
495 if (valiswhite(gval(n))) /* value not marked yet? */
496 hasww = 1; /* white-white entry */
497 }
498 else if (valiswhite(gval(n))) { /* value not marked yet? */
499 marked = 1;
500 reallymarkobject(g, gcvalue(gval(n))); /* mark it now */
501 }
502 }
503 /* link table into proper list */
504 if (g->gcstate == GCSpropagate)
505 linkgclist(h, g->grayagain); /* must retraverse it in atomic phase */
506 else if (hasww) /* table has white->white entries? */
507 linkgclist(h, g->ephemeron); /* have to propagate again */
508 else if (hasclears) /* table has white keys? */
509 linkgclist(h, g->allweak); /* may have to clean white keys */
510 else
511 genlink(g, obj2gco(h)); /* check whether collector still needs to see it */
512 return marked;
513}

535static lu_mem traversetable (global_State *g, Table *h) {
536 const char *weakkey, *weakvalue;
537 const TValue *mode = gfasttm(g, h->metatable, TM_MODE);
538 markobjectN(g, h->metatable);
539 if (mode && ttisstring(mode) && /* is there a weak mode? */
540 (cast_void(weakkey = strchr(svalue(mode), 'k')),
541 cast_void(weakvalue = strchr(svalue(mode), 'v')),
542 (weakkey || weakvalue))) { /* is really weak? */
543 if (!weakkey) /* strong keys? */
544 traverseweakvalue(g, h);
545 else if (!weakvalue) /* strong values? */
546 traverseephemeron(g, h, 0);
547 else /* all weak */
548 linkgclist(h, g->allweak); /* nothing to traverse now */
549 }
550 else /* not weak */
551 traversestrongtable(g, h);
552 return 1 + h->alimit + 2 * allocsizenode(h);
553}

608/*
609** Traverse a thread, marking the elements in the stack up to its top
610** and cleaning the rest of the stack in the final traversal. That
611** ensures that the entire stack have valid (non-dead) objects.
612** Threads have no barriers. In gen. mode, old threads must be visited
613** at every cycle, because they might point to young objects. In inc.
614** mode, the thread can still be modified before the end of the cycle,
615** and therefore it must be visited again in the atomic phase. To ensure
616** these visits, threads must return to a gray list if they are not new
617** (which can only happen in generational mode) or if the traverse is in
618** the propagate phase (which can only happen in incremental mode).
619*/
620static int traversethread (global_State *g, lua_State *th) {
621 UpVal *uv;
622 StkId o = th->stack;
623 if (isold(th) || g->gcstate == GCSpropagate)
624 linkgclist(th, g->grayagain); /* insert into 'grayagain' list */
625 if (o == NULL)
626 return 1; /* stack not completely built yet */
627 lua_assert(g->gcstate == GCSatomic ||
628 th->openupval == NULL || isintwups(th));
629 for (; o < th->top; o++) /* mark live elements in the stack */
630 markvalue(g, s2v(o));
631 for (uv = th->openupval; uv != NULL; uv = uv->u.open.next)
632 markobject(g, uv); /* open upvalues cannot be collected */
633 if (g->gcstate == GCSatomic) { /* final traversal? */
634 for (; o < th->stack_last + EXTRA_STACK; o++)
635 setnilvalue(s2v(o)); /* clear dead stack slice */
636 /* 'remarkupvals' may have removed thread from 'twups' list */
637 if (!isintwups(th) && th->openupval != NULL) {
638 th->twups = g->twups; /* link it back to the list */
639 g->twups = th;
640 }
641 }
642 else if (!g->gcemergency)
643 luaD_shrinkstack(th); /* do not change stack in emergency cycle */
644 return 1 + stacksize(th);
645}

652 GCObject *o = g->gray;
653 nw2black(o);

675/*
676** Traverse all ephemeron tables propagating marks from keys to values.
677** Repeat until it converges, that is, nothing new is marked. 'dir'
678** inverts the direction of the traversals, trying to speed up
679** convergence on chains in the same table.
680**
681*/
682static void convergeephemerons (global_State *g) {
683 int changed;
684 int dir = 0;
685 do {
686 GCObject *w;
687 GCObject *next = g->ephemeron; /* get ephemeron list */
688 g->ephemeron = NULL; /* tables may return to this list when traversed */
689 changed = 0;
690 while ((w = next) != NULL) { /* for each ephemeron table */
691 Table *h = gco2t(w);
692 next = h->gclist; /* list is rebuilt during loop */
693 nw2black(h); /* out of the list (for now) */
694 if (traverseephemeron(g, h, dir)) { /* marked some value? */
695 propagateall(g); /* propagate changes */
696 changed = 1; /* will have to revisit all ephemeron tables */
697 }
698 }
699 dir = !dir; /* invert direction next time */
700 } while (changed); /* repeat until no more changes */
701}

808/*
809** sweep at most 'countin' elements from a list of GCObjects erasing dead
810** objects, where a dead object is one marked with the old (non current)
811** white; change all non-dead objects back to white, preparing for next
812** collection cycle. Return where to continue the traversal or NULL if
813** list is finished. ('*countout' gets the number of elements traversed.)
814*/
815static GCObject **sweeplist (lua_State *L, GCObject **p, int countin,
816 int *countout) {
817 global_State *g = G(L);
818 int ow = otherwhite(g);
819 int i;
820 int white = luaC_white(g); /* current white */
821 for (i = 0; *p != NULL && i < countin; i++) {
822 GCObject *curr = *p;
823 int marked = curr->marked;
824 if (isdeadm(ow, marked)) { /* is 'curr' dead? */
825 *p = curr->next; /* remove 'curr' from list */
826 freeobj(L, curr); /* erase 'curr' */
827 }
828 else { /* change mark to 'white' */
829 curr->marked = cast_byte((marked & ~maskgcbits) | white);
830 p = &curr->next; /* go to next element */
831 }
832 }
833 if (countout)
834 *countout = i; /* number of elements traversed */
835 return (*p == NULL) ? NULL : p;
836}

898static void GCTM (lua_State *L) {
899 global_State *g = G(L);
900 const TValue *tm;
901 TValue v;
902 lua_assert(!g->gcemergency);
903 setgcovalue(L, &v, udata2finalize(g));
904 tm = luaT_gettmbyobj(L, &v, TM_GC);
905 if (!notm(tm)) { /* is there a finalizer? */
906 int status;
907 lu_byte oldah = L->allowhook;
908 int oldgcstp = g->gcstp;
909 g->gcstp |= GCSTPGC; /* avoid GC steps */
910 L->allowhook = 0; /* stop debug hooks during GC metamethod */
911 setobj2s(L, L->top++, tm); /* push finalizer... */
912 setobj2s(L, L->top++, &v); /* ... and its argument */
913 L->ci->callstatus |= CIST_FIN; /* will run a finalizer */
914 status = luaD_pcall(L, dothecall, NULL, savestack(L, L->top - 2), 0);
915 L->ci->callstatus &= ~CIST_FIN; /* not running a finalizer anymore */
916 L->allowhook = oldah; /* restore hooks */
917 g->gcstp = oldgcstp; /* restore state */
918 if (l_unlikely(status != LUA_OK)) { /* error while running __gc? */
919 luaE_warnerror(L, "__gc");
920 L->top--; /* pops error object */
921 }
922 }

960** finalization from list 'finobj' to list 'tobefnz' (to be finalized).

1046/*
1047** Sweep a list of objects to enter generational mode. Deletes dead
1048** objects and turns the non dead to old. All non-dead threads---which
1049** are now old---must be in a gray list. Everything else is not in a
1050** gray list. Open upvalues are also kept gray.
1051*/
1052static void sweep2old (lua_State *L, GCObject **p) {
1053 GCObject *curr;
1054 global_State *g = G(L);
1055 while ((curr = *p) != NULL) {
1056 if (iswhite(curr)) { /* is 'curr' dead? */
1057 lua_assert(isdead(g, curr));
1058 *p = curr->next; /* remove 'curr' from list */
1059 freeobj(L, curr); /* erase 'curr' */
1060 }
1061 else { /* all surviving objects become old */
1062 setage(curr, G_OLD);
1063 if (curr->tt == LUA_VTHREAD) { /* threads must be watched */
1064 lua_State *th = gco2th(curr);
1065 linkgclist(th, g->grayagain); /* insert into 'grayagain' list */
1066 }
1067 else if (curr->tt == LUA_VUPVAL && upisopen(gco2upv(curr)))
1068 set2gray(curr); /* open upvalues are always gray */
1069 else /* everything else is black */
1070 nw2black(curr);
1071 p = &curr->next; /* go to next element */
1072 }
1073 }
1074}

1077/*
1078** Sweep for generational mode. Delete dead objects. (Because the
1079** collection is not incremental, there are no "new white" objects
1080** during the sweep. So, any white object must be dead.) For
1081** non-dead objects, advance their ages and clear the color of
1082** new objects. (Old objects keep their colors.)
1083** The ages of G_TOUCHED1 and G_TOUCHED2 objects cannot be advanced
1084** here, because these old-generation objects are usually not swept
1085** here. They will all be advanced in 'correctgraylist'. That function
1086** will also remove objects turned white here from any gray list.
1087*/
1088static GCObject **sweepgen (lua_State *L, global_State *g, GCObject **p,
1089 GCObject *limit, GCObject **pfirstold1) {
1090 static const lu_byte nextage[] = {
1091 G_SURVIVAL, /* from G_NEW */
1092 G_OLD1, /* from G_SURVIVAL */
1093 G_OLD1, /* from G_OLD0 */
1094 G_OLD, /* from G_OLD1 */
1095 G_OLD, /* from G_OLD (do not change) */
1096 G_TOUCHED1, /* from G_TOUCHED1 (do not change) */
1097 G_TOUCHED2 /* from G_TOUCHED2 (do not change) */
1098 };
1099 int white = luaC_white(g);
1100 GCObject *curr;
1101 while ((curr = *p) != limit) {
1102 if (iswhite(curr)) { /* is 'curr' dead? */
1103 lua_assert(!isold(curr) && isdead(g, curr));
1104 *p = curr->next; /* remove 'curr' from list */
1105 freeobj(L, curr); /* erase 'curr' */
1106 }
1107 else { /* correct mark and age */
1108 if (getage(curr) == G_NEW) { /* new objects go back to white */
1109 int marked = curr->marked & ~maskgcbits; /* erase GC bits */
1110 curr->marked = cast_byte(marked | G_SURVIVAL | white);
1111 }
1112 else { /* all other objects will be old, and so keep their color */
1113 setage(curr, nextage[getage(curr)]);
1114 if (getage(curr) == G_OLD1 && *pfirstold1 == NULL)
1115 *pfirstold1 = curr; /* first OLD1 object in the list */
1116 }
1117 p = &curr->next; /* go to next element */
1118 }
1119 }
1120 return p;
1121}

1136/*
1137** Correct a list of gray objects. Return pointer to where rest of the
1138** list should be linked.
1139** Because this correction is done after sweeping, young objects might
1140** be turned white and still be in the list. They are only removed.
1141** 'TOUCHED1' objects are advanced to 'TOUCHED2' and remain on the list;
1142** Non-white threads also remain on the list; 'TOUCHED2' objects become
1143** regular old; they and anything else are removed from the list.
1144*/
1145static GCObject **correctgraylist (GCObject **p) {
1146 GCObject *curr;
1147 while ((curr = *p) != NULL) {
1148 GCObject **next = getgclist(curr);
1149 if (iswhite(curr))

1151 else if (getage(curr) == G_TOUCHED1) { /* touched in this cycle? */

1353/*
1354** Does a major collection after last collection was a "bad collection".
1355**
1356** When the program is building a big structure, it allocates lots of
1357** memory but generates very little garbage. In those scenarios,
1358** the generational mode just wastes time doing small collections, and
1359** major collections are frequently what we call a "bad collection", a
1360** collection that frees too few objects. To avoid the cost of switching
1361** between generational mode and the incremental mode needed for full
1362** (major) collections, the collector tries to stay in incremental mode
1363** after a bad collection, and to switch back to generational mode only
1364** after a "good" collection (one that traverses less than 9/8 objects
1365** of the previous one).
1366** The collector must choose whether to stay in incremental mode or to
1367** switch back to generational mode before sweeping. At this point, it
1368** does not know the real memory in use, so it cannot use memory to
1369** decide whether to return to generational mode. Instead, it uses the
1370** number of objects traversed (returned by 'atomic') as a proxy. The
1371** field 'g->lastatomic' keeps this count from the last collection.
1372** ('g->lastatomic != 0' also means that the last collection was bad.)
1373*/
1374static void stepgenfull (lua_State *L, global_State *g) {
1375 lu_mem newatomic; /* count of traversed objects */
1376 lu_mem lastatomic = g->lastatomic; /* count from last collection */
1377 if (g->gckind == KGC_GEN) /* still in generational mode? */
1378 enterinc(g); /* enter incremental mode */
1379 luaC_runtilstate(L, bitmask(GCSpropagate)); /* start new cycle */
1380 newatomic = atomic(L); /* mark everybody */
1381 if (newatomic < lastatomic + (lastatomic >> 3)) { /* good collection? */
1382 atomic2gen(L, g); /* return to generational mode */
1383 setminordebt(g);
1384 }
1385 else { /* another bad collection; stay in incremental mode */
1386 g->GCestimate = gettotalbytes(g); /* first estimate */;
1387 entersweep(L);
1388 luaC_runtilstate(L, bitmask(GCSpause)); /* finish collection */
1389 setpause(g);
1390 g->lastatomic = newatomic;
1391 }
1392}

1395/*
1396** Does a generational "step".
1397** Usually, this means doing a minor collection and setting the debt to
1398** make another collection when memory grows 'genminormul'% larger.
1399**
1400** However, there are exceptions. If memory grows 'genmajormul'%
1401** larger than it was at the end of the last major collection (kept
1402** in 'g->GCestimate'), the function does a major collection. At the
1403** end, it checks whether the major collection was able to free a
1404** decent amount of memory (at least half the growth in memory since
1405** previous major collection). If so, the collector keeps its state,
1406** and the next collection will probably be minor again. Otherwise,
1407** we have what we call a "bad collection". In that case, set the field
1408** 'g->lastatomic' to signal that fact, so that the next collection will
1409** go to 'stepgenfull'.
1410**
1411** 'GCdebt <= 0' means an explicit call to GC step with "size" zero;
1412** in that case, do a minor collection.
1413*/
1414static void genstep (lua_State *L, global_State *g) {
1415 if (g->lastatomic != 0) /* last collection was a bad one? */
1416 stepgenfull(L, g); /* do a full step */
1417 else {
1418 lu_mem majorbase = g->GCestimate; /* memory after last major collection */
1419 lu_mem majorinc = (majorbase / 100) * getgcparam(g->genmajormul);
1420 if (g->GCdebt > 0 && gettotalbytes(g) > majorbase + majorinc) {
1421 lu_mem numobjs = fullgen(L, g); /* do a major collection */
1422 if (gettotalbytes(g) < majorbase + (majorinc / 2)) {
1423 /* collected at least half of memory growth since last major
1424 collection; keep doing minor collections */
1425 setminordebt(g);
1426 }
1427 else { /* bad collection */
1428 g->lastatomic = numobjs; /* signal that last collection was bad */
1429 setpause(g); /* do a long wait for next (major) collection */
1430 }
1431 }
1432 else { /* regular case; do a minor collection */
1433 youngcollection(L, g);
1434 setminordebt(g);
1435 g->GCestimate = majorbase; /* preserve base value */
1436 }
1437 }
1438 lua_assert(isdecGCmodegen(g));
1439}

1577static lu_mem singlestep (lua_State *L) {
1578 global_State *g = G(L);
1579 lu_mem work;
1580 lua_assert(!g->gcstopem); /* collector is not reentrant */

1582 switch (g->gcstate) {
1583 case GCSpause: {
1584 restartcollection(g);
1585 g->gcstate = GCSpropagate;
1586 work = 1;
1587 break;
1588 }
1589 case GCSpropagate: {
1590 if (g->gray == NULL) { /* no more gray objects? */
1591 g->gcstate = GCSenteratomic; /* finish propagate phase */
1592 work = 0;
1593 }
1594 else
1595 work = propagatemark(g); /* traverse one gray object */
1596 break;
1597 }
1598 case GCSenteratomic: {
1599 work = atomic(L); /* work is what was traversed by 'atomic' */
1600 entersweep(L);
1601 g->GCestimate = gettotalbytes(g); /* first estimate */;
1602 break;
1603 }
1604 case GCSswpallgc: { /* sweep "regular" objects */
1605 work = sweepstep(L, g, GCSswpfinobj, &g->finobj);
1606 break;
1607 }
1608 case GCSswpfinobj: { /* sweep objects with finalizers */
1609 work = sweepstep(L, g, GCSswptobefnz, &g->tobefnz);
1610 break;
1611 }
1612 case GCSswptobefnz: { /* sweep objects to be finalized */
1613 work = sweepstep(L, g, GCSswpend, NULL);
1614 break;
1615 }
1616 case GCSswpend: { /* finish sweeps */
1617 checkSizes(L, g);
1618 g->gcstate = GCScallfin;
1619 work = 0;
1620 break;
1621 }
1622 case GCScallfin: { /* call remaining finalizers */
1623 if (g->tobefnz && !g->gcemergency) {
1624 g->gcstopem = 0; /* ok collections during finalizers */
1625 work = runafewfinalizers(L, GCFINMAX) * GCFINALIZECOST;
1626 }
1627 else { /* emergency mode or no more finalizers */
1628 g->gcstate = GCSpause; /* finish collection */

1631 break;
1632 }
1633 default: lua_assert(0); return 0;

1652/*
1653** Performs a basic incremental step. The debt and step size are
1654** converted from bytes to "units of work"; then the function loops
1655** running single steps until adding that many units of work or
1656** finishing a cycle (pause state). Finally, it sets the debt that
1657** controls when next step will be performed.
1658*/
1659static void incstep (lua_State *L, global_State *g) {
1660 int stepmul = (getgcparam(g->gcstepmul) | 1); /* avoid division by 0 */
1661 l_mem debt = (g->GCdebt / WORK2MEM) * stepmul;
1662 l_mem stepsize = (g->gcstepsize <= log2maxs(l_mem))
1663 ? ((cast(l_mem, 1) << g->gcstepsize) / WORK2MEM) * stepmul
1664 : MAX_LMEM; /* overflow; keep maximum value */
1665 do { /* repeat until pause or enough "credit" (negative debt) */
1666 lu_mem work = singlestep(L); /* perform one single step */
1667 debt -= work;
1668 } while (debt > -stepsize && g->gcstate != GCSpause);
1669 if (g->gcstate == GCSpause)
1670 setpause(g); /* pause until next cycle */
1671 else {
1672 debt = (debt / stepmul) * WORK2MEM; /* convert 'work units' to bytes */
1673 luaE_setdebt(g, debt);
1674 }
1675}
