135/* skips given number of bits. */
136static int32_t skip_bit(pkzip_cmp_s *mpq_pkzip, uint32_t bits) {
137
138 /* check if number of bits required is less than number of bits in the buffer. */
139 if (bits <= mpq_pkzip->extra_bits) {
140 mpq_pkzip->extra_bits -= bits;
141 mpq_pkzip->bit_buf >>= bits;
142 return 0;
143 }
144
145 /* load input buffer if necessary. */
146 mpq_pkzip->bit_buf >>= mpq_pkzip->extra_bits;
147 if (mpq_pkzip->in_pos == mpq_pkzip->in_bytes) {
148 mpq_pkzip->in_pos = sizeof(mpq_pkzip->in_buf);
149 if ((mpq_pkzip->in_bytes = mpq_pkzip->read_buf((char *)mpq_pkzip->in_buf, &mpq_pkzip->in_pos, mpq_pkzip->param)) == 0) {
150 return 1;
151 }
152 mpq_pkzip->in_pos = 0;
153 }
154
155 /* update bit buffer. */
156 mpq_pkzip->bit_buf |= (mpq_pkzip->in_buf[mpq_pkzip->in_pos++] << 8);
157 mpq_pkzip->bit_buf >>= (bits - mpq_pkzip->extra_bits);
158 mpq_pkzip->extra_bits = (mpq_pkzip->extra_bits - bits) + 8;
159
160 /* if no error was found, return zero. */
161 return 0;
162}

164/* this function generate the decode tables used for decryption. */
165static void generate_tables_decode(int32_t count, uint8_t *bits, const uint8_t *code, uint8_t *buf2) {
166
167 /* some common variables. */
168 int32_t i;
169
170 /* EBX - count */
171 for (i = count-1; i >= 0; i--) {
172
173 /* some common variables. */
174 uint32_t idx1 = code[i];
175 uint32_t idx2 = 1 << bits[i];
176
177 /* loop until table is ready. */
178 do {
179 buf2[idx1] = (uint8_t)i;
180 idx1 += idx2;
181 } while (idx1 < 0x100);
182 }
183}

185/* this function generate the tables for ascii decompression. */
186static void generate_tables_ascii(pkzip_cmp_s *mpq_pkzip) {
187
188 /* some common variables. */
189 const uint16_t *code_asc = &pkzip_code_asc[0xFF];
190 uint32_t acc;
191 uint32_t add;
192 uint16_t count;
193
194 /* loop through ascii table. */
195 for (count = 0x00FF; code_asc >= pkzip_code_asc; code_asc--, count--) {
196 uint8_t *bits_asc = mpq_pkzip->bits_asc + count;
197 uint8_t bits_tmp = *bits_asc;
198
199 /* check if byte is finished. */
200 if (bits_tmp <= 8) {
201 add = (1 << bits_tmp);
202 acc = *code_asc;
203 do {
204 mpq_pkzip->offs_2c34[acc] = (uint8_t)count;
205 acc += add;
206 } while (acc < 0x100);
207 } else {
208 if ((acc = (*code_asc & 0xFF)) != 0) {
209 mpq_pkzip->offs_2c34[acc] = 0xFF;
210 if (*code_asc & 0x3F) {
211
212 /* decrease bit by four. */
213 bits_tmp -= 4;
214 *bits_asc = bits_tmp;
215 add = (1 << bits_tmp);
216 acc = *code_asc >> 4;
217 do {
218 mpq_pkzip->offs_2d34[acc] = (uint8_t)count;
219 acc += add;
220 } while (acc < 0x100);
221 } else {
222
223 /* decrease bit by six. */
224 bits_tmp -= 6;
225 *bits_asc = bits_tmp;
226 add = (1 << bits_tmp);
227 acc = *code_asc >> 6;
228 do {
229 mpq_pkzip->offs_2e34[acc] = (uint8_t)count;
230 acc += add;
231 } while (acc < 0x80);
232 }
233 } else {
234
235 /* decrease bit by eight. (one byte) */
236 bits_tmp -= 8;
237 *bits_asc = bits_tmp;
238 add = (1 << bits_tmp);
239 acc = *code_asc >> 8;
240 do {
241 mpq_pkzip->offs_2eb4[acc] = (uint8_t)count;
242 acc += add;
243 } while (acc < 0x100);
244 }
245 }
246 }
247}

249/*
250 * decompress the imploded data using coded literals.
251 *
252 * returns: 0x000 - 0x0FF : one byte from compressed file.
253 * 0x100 - 0x305 : copy previous block. (0x100 = 1 byte)
254 * 0x306 : out of buffer?
255 */
256static uint32_t decode_literal(pkzip_cmp_s *mpq_pkzip) {
257
258 /* number of bits to skip. */
259 uint32_t bits;
260
261 /* position in buffers. */
262 uint32_t value;
263
264 /* check if bit the current buffer is set, if not return the next byte. */
265 if (mpq_pkzip->bit_buf & 1) {
266
267 /* skip current bit in the buffer. */
268 if (skip_bit(mpq_pkzip, 1)) {
269 return 0x306;
270 }
271
272 /* the next bits are position in buffers. */
273 value = mpq_pkzip->pos2[(mpq_pkzip->bit_buf & 0xFF)];
274
275 /* get number of bits to skip. */
276 if (skip_bit(mpq_pkzip, mpq_pkzip->slen_bits[value])) {
277 return 0x306;
278 }
279
280 /* check bits. */
281 if ((bits = mpq_pkzip->clen_bits[value]) != 0) {
282
283 /* some common variables. */
284 uint32_t val2 = mpq_pkzip->bit_buf & ((1 << bits) - 1);
285
286 /* check if we should skip one bit. */
287 if (skip_bit(mpq_pkzip, bits)) {
288
289 /* check position if we should skip the bit. */
290 if ((value + val2) != 0x10E) {
291 return 0x306;
292 }
293 }
294
295 /* fill values. */
296 value = mpq_pkzip->len_base[value] + val2;
297 }
298
299 /* return number of bytes to repeat. */
300 return value + 0x100;
301 }
302
303 /* skip one bit. */
304 if (skip_bit(mpq_pkzip, 1)) {
305 return 0x306;
306 }
307
308 /* check the binary compression type, read 8 bits and return them as one byte. */
309 if (mpq_pkzip->cmp_type == LIBMPQ_PKZIP_CMP_BINARY) {
310
311 /* fill values. */
312 value = mpq_pkzip->bit_buf & 0xFF;
313
314 /* check if we should skip one bit. */
315 if (skip_bit(mpq_pkzip, 8)) {
316 return 0x306;
317 }
318
319 /* return value from bit buffer. */
320 return value;
321 }
322
323 /* check if ascii compression is used. */
324 if (mpq_pkzip->bit_buf & 0xFF) {
325
326 /* fill values. */
327 value = mpq_pkzip->offs_2c34[mpq_pkzip->bit_buf & 0xFF];
328
329 /* check value. */
330 if (value == 0xFF) {
331 if (mpq_pkzip->bit_buf & 0x3F) {
332
333 /* check if four bits are in bit buffer for skipping. */
334 if (skip_bit(mpq_pkzip, 4)) {
335 return 0x306;
336 }
337
338 /* fill values. */
339 value = mpq_pkzip->offs_2d34[mpq_pkzip->bit_buf & 0xFF];
340 } else {
341
342 /* check if six bits are in bit buffer for skipping. */
343 if (skip_bit(mpq_pkzip, 6)) {
344 return 0x306;
345 }
346
347 /* fill values. */
348 value = mpq_pkzip->offs_2e34[mpq_pkzip->bit_buf & 0x7F];
349 }
350 }
351 } else {
352
353 /* check if eight bits are in bit buffer for skipping. */
354 if (skip_bit(mpq_pkzip, 8)) {
355 return 0x306;
356 }
357
358 /* fill values. */
359 value = mpq_pkzip->offs_2eb4[mpq_pkzip->bit_buf & 0xFF];
360 }
361
362 /* return out of buffer error (0x306) or position in buffer. */
363 return skip_bit(mpq_pkzip, mpq_pkzip->bits_asc[value]) ? 0x306 : value;
364}

366/* this function retrieves the number of bytes to move back. */
367static uint32_t decode_distance(pkzip_cmp_s *mpq_pkzip, uint32_t length) {
368
369 /* some common variables. */
370 uint32_t pos = mpq_pkzip->pos1[(mpq_pkzip->bit_buf & 0xFF)];
371
372 /* number of bits to skip. */
373 uint32_t skip = mpq_pkzip->dist_bits[pos];
374
375 /* skip the appropriate number of bits. */
376 if (skip_bit(mpq_pkzip, skip) == 1) {
377 return 0;
378 }
379
380 /* check if length is two. */
381 if (length == 2) {
382 pos = (pos << 2) | (mpq_pkzip->bit_buf & 0x03);
383
384 /* skip the bits. */
385 if (skip_bit(mpq_pkzip, 2) == 1) {
386 return 0;
387 }
388 } else {
389 pos = (pos << mpq_pkzip->dsize_bits) | (mpq_pkzip->bit_buf & mpq_pkzip->dsize_mask);
390
391 /* skip the bits */
392 if (skip_bit(mpq_pkzip, mpq_pkzip->dsize_bits) == 1) {
393 return 0;
394 }
395 }
396
397 /* return the bytes to move back. */
398 return pos + 1;
399}

401/*
402 * function loads data from the input buffer used by mpq_pkzip
403 * "implode" and "explode" function as user defined callback and
404 * returns number of bytes loaded.
405 *
406 * char *buf - pointer to a buffer where to store loaded data.
407 * uint32_t *size - maximum number of bytes to read.
408 * void *param - custom pointer, parameter of implode/explode.
409 */
410static uint32_t data_read_input(char *buf, uint32_t *size, void *param) {
411
412 /* some common variables. */
413 pkzip_data_s *info = (pkzip_data_s *)param;
414 uint32_t max_avail = (info->in_bytes - info->in_pos);
415 uint32_t to_read = *size;
416
417 /* check the case when not enough data available. */
418 if (to_read > max_avail) {
419 to_read = max_avail;
420 }
421
422 /* load data and increment offsets. */
423 memcpy(buf, info->in_buf + info->in_pos, to_read);
424 info->in_pos += to_read;
425
426 /* return bytes read. */
427 return to_read;
428}

471 while ((result = one_byte = decode_literal(mpq_pkzip)) < 0x305) {
472
473 /* check if one byte is greater than 0x100, which means 'repeat n - 0xFE bytes'. */
474 if (one_byte >= 0x100) {
475
476 /* ECX */
477 uint8_t *source;
478
479 /* EDX */
480 uint8_t *target;
481
482 /* some common variables. */
483 uint32_t copy_length = one_byte - 0xFE;
484 uint32_t move_back;
485
486 /* get length of data to copy. */
487 if ((move_back = decode_distance(mpq_pkzip, copy_length)) == 0) {
488 result = 0x306;
489 break;
490 }
491
492 /* target and source pointer. */
493 target = &mpq_pkzip->out_buf[mpq_pkzip->out_pos];
494 source = target - move_back;
495 mpq_pkzip->out_pos += copy_length;
496
497 /* copy until nothing left. */
498 while (copy_length-- > 0) {
499 *target++ = *source++;
500 }
501 } else {
502
503 /* byte is 0x100 great, so add one byte. */
504 mpq_pkzip->out_buf[mpq_pkzip->out_pos++] = (uint8_t)one_byte;
505 }
506
507 /* check if number of extracted bytes has reached 1/2 of output buffer, so flush output buffer. */
508 if (mpq_pkzip->out_pos >= 0x2000) {
509
510 /* copy decompressed data into user buffer. */
511 copy_bytes = 0x1000;
512 mpq_pkzip->write_buf((char *)&mpq_pkzip->out_buf[0x1000], &copy_bytes, mpq_pkzip->param);
513
514 /* check if there are some data left, keep them alive. */
515 memcpy(mpq_pkzip->out_buf, &mpq_pkzip->out_buf[0x1000], mpq_pkzip->out_pos - 0x1000);
516 mpq_pkzip->out_pos -= 0x1000;
517 }
518 }
519
520 /* copy the rest. */
521 copy_bytes = mpq_pkzip->out_pos - 0x1000;
522 mpq_pkzip->write_buf((char *)&mpq_pkzip->out_buf[0x1000], &copy_bytes, mpq_pkzip->param);
523
524 /* return copied bytes. */
525 return result;
526}

597 return LIBMPQ_PKZIP_CMP_NO_ERROR;
598 }

601 return LIBMPQ_PKZIP_CMP_ABORT;
