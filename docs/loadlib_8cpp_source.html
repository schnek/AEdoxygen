38bool ChunkedFile::loadFile(HANDLE mpq, std::string const& fileName, bool log)
39{
40 free();
41 HANDLE file;
42 if (!SFileOpenFileEx(mpq, fileName.c_str(), SFILE_OPEN_FROM_MPQ, &file))
43 {
44 if (log)
45 printf("No such file %s\n", fileName.c_str());
46 return false;
47 }
48
49 data_size = SFileGetFileSize(file, nullptr);
50 data = new uint8_t[data_size];
51 SFileReadFile(file, data, data_size, nullptr/*bytesRead*/, nullptr);
52
53 parseChunks();
54 if (prepareLoadedData())
55 {
56 SFileCloseFile(file);
57 return true;
58 }
59
60 printf("Error loading %s\n", fileName.c_str());
61 SFileCloseFile(file);
62 free();
63
64 return false;
65}

67bool ChunkedFile::prepareLoadedData()
68{
69 FileChunk* chunk = GetChunk("MVER");
70 if (!chunk)
71 return false;
72
73 // Check version
74 file_MVER* version = chunk->As<file_MVER>();
75 if (version->fcc != MverMagic.fcc)
76 return false;
77 if (version->ver != FILE_FORMAT_VERSION)
78 return false;
79 return true;
80}

115void ChunkedFile::parseChunks()
116{
117 uint8_t* ptr = GetData();
118 // Make sure there's enough data to read u_map_fcc struct and the uint32_t size after it
119 while (ptr <= GetData() + GetDataSize() - 8)
120 {
121 u_map_fcc header = *(u_map_fcc*)ptr;
122 if (IsInterestingChunk(header))
123 {
124 uint32_t size = *(uint32_t*)(ptr + 4);
125 if (size <= data_size)
126 {
127 std::swap(header.fcc_txt[0], header.fcc_txt[3]);
128 std::swap(header.fcc_txt[1], header.fcc_txt[2]);
129
130 FileChunk* chunk = new FileChunk(ptr, size);
131 chunk->parseSubChunks();
132 chunks.insert({ std::string(header.fcc_txt, 4), chunk });
133 }
134
135 // move to next chunk
136 ptr += size + 8;
137 }
138 else
139 ++ptr;
140 }
141}

160void FileChunk::parseSubChunks()
161{
162 uint8_t* ptr = data + 8; // skip self
163 while (ptr < data + size)
164 {
165 u_map_fcc header = *(u_map_fcc*)ptr;
166 if (IsInterestingChunk(header))
167 {
168 uint32_t subsize = *(uint32_t*)(ptr + 4);
169 if (subsize < size)
170 {
171 std::swap(header.fcc_txt[0], header.fcc_txt[3]);
172 std::swap(header.fcc_txt[1], header.fcc_txt[2]);
173
174 FileChunk* chunk = new FileChunk(ptr, subsize);
175 chunk->parseSubChunks();
176 subchunks.insert({ std::string(header.fcc_txt, 4), chunk });
177 }
178
179 // move to next chunk
180 ptr += subsize + 8;
181 }
182 else
183 ++ptr;
184 }
185}
