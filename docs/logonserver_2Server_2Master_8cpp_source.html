113 // Spawn auth listener

124

129 sLogger.info("Success! Ready for connections");
130 while (mrunning)

132 if (!(++loop_counter % 20)) // 20 seconds
133 CheckForDeadSockets();
134
135 if (!(loop_counter % 300)) // 5mins
136 ThreadPool.IntegrityCheck();
137
138 if (!(loop_counter % 5))
139 {
140 sRealmManager.timeoutSockets();
141 sSocketGarbageCollector.Update();
142 CheckForDeadSockets(); // Flood Protection
143 UNIXTIME = time(nullptr);
144 g_localTime = *localtime(&UNIXTIME);

152

157 sLogger.failure("Error creating sockets. Shutting down...");
158 }
159

161 logonServerSocket->Close();
162 sSocketMgr.CloseAll();
163#ifdef WIN32
164 sSocketMgr.ShutdownThreads();
165#endif
166 sLogonConsole.Kill();
167 sAccountMgr.finalize();
168 sRealmManager.finalize();
169
170 // kill db

172 sLogonSQL->EndThreads();
173 sLogonSQL->Shutdown();

175
176 ThreadPool.Shutdown();
177

179 if (remove("logonserver.pid") != 0)
180 sLogger.failure("Error deleting file logonserver.pid");

182 sLogger.debug("File logonserver.pid successfully deleted");
183
184 sSocketMgr.finalize();
185 sSocketGarbageCollector.finalize();
186 //delete periodicReloadAccounts;
187 sLogger.info("Shutdown complete.");
188 sLogger.finalize();

196void MasterLogon::CheckForDeadSockets()
197{
198 std::lock_guard guard(m_authSocketLock);
199 time_t t = time(nullptr);
200 for (auto itr = m_authSockets.begin(); itr != m_authSockets.end();)
201 {
202 auto it2 = itr;
203 auto s = (*it2);
204 ++itr;
205
206 time_t diff = t - s->GetLastRecv();
207 if (diff > 300) // More than 5mins
208 {
209 m_authSockets.erase(it2);
210 s->removedFromSet = true;
211 s->Disconnect();
212 }
213 }
214}

275bool MasterLogon::StartDb()
276{
277 std::string dbHostname = logonConfig.logonDb.host;
278 std::string dbUsername = logonConfig.logonDb.user;
279 std::string dbPassword = logonConfig.logonDb.password;
280 std::string dbDatabase = logonConfig.logonDb.db;
281
282 const int dbPort = logonConfig.logonDb.port;
283
284 // Configure Main Database
285 const bool existsUsername = !dbUsername.empty();
286 const bool existsPassword = !dbPassword.empty();
287 const bool existsHostname = !dbHostname.empty();
288 const bool existsName = !dbDatabase.empty();
289
290 const bool result = existsUsername && existsPassword && existsHostname && existsName;
291
292 if (!result)
293 {
294 //Build informative error message
295 //Built as one string and then printed rather than calling Log.outString(...) for every line,
296 // as experiments has seen other thread write to the console inbetween calls to Log.outString(...)
297 // resulting in unreadable error messages.
298 //If the <LogonDatabase> tag is malformed, all parameters will fail, and a different error message is given
299
300 std::string errorMessage = "sql: Certain <LogonDatabase> parameters not found in " CONFDIR "\\logon.conf \r\n";
301 if (!(existsHostname || existsUsername || existsPassword || existsName))
302 {
303 errorMessage += " Double check that you have remembered the entire <LogonDatabase> tag.\r\n";
304 errorMessage += " All parameters missing. It is possible you forgot the first '<' character.\r\n";
305 }
306 else
307 {
308 errorMessage += " Missing paramer(s):\r\n";
309
310 if (!existsHostname)
311 errorMessage += " Hostname\r\n";
312
313 if (!existsUsername)
314 errorMessage += " Username\r\n";
315
316 if (!existsPassword)
317 errorMessage += " Password\r\n";
318
319 if (!existsName)
320 errorMessage += " Name\r\n";
321 }
322
323 //sLogger.fatal(errorMessage); FIX fmt
324 return false;
325 }
326
327 sLogonSQL = Database::CreateDatabaseInterface();
328
329 // Initialize it
330 if (!sLogonSQL->Initialize(dbHostname.c_str(), (unsigned int)dbPort, dbUsername.c_str(),
331 dbPassword.c_str(), dbDatabase.c_str(), logonConfig.logonDb.connections,
332 16384, logonConfig.logonDb.isLegacyAuth))
333 {
334 sLogger.fatal("sql: Logon database initialization failed. Exiting.");
335 return false;
336 }
337
338 return true;
339}

341bool MasterLogon::CheckDBVersion()
342{
343 auto cqr = sLogonSQL->QueryNA("SELECT LastUpdate FROM logon_db_version ORDER BY id DESC LIMIT 1;");
344 if (cqr == nullptr)
345 {
346 sLogger.failure("Database : logon database is missing the table `logon_db_version` OR the table doesn't contain any rows. Can't validate database version. Exiting.");
347 sLogger.failure("Database : You may need to update your database");
348 return false;
349 }
350
351 Field* f = cqr->Fetch();
352 const char *LogonDBVersion = f->asCString();
353
354 sLogger.info("Database : Last logon database update: {}", LogonDBVersion);
355 int result = strcmp(LogonDBVersion, REQUIRED_LOGON_DB_VERSION);
356 if (result != 0)
357 {
358 sLogger.failure("Database : Last logon database update doesn't match the required one which is {}.", REQUIRED_LOGON_DB_VERSION);
359 if (result < 0)
360 {
361 sLogger.failure("Database : You need to apply the logon update queries that are newer than {}. Exiting.", LogonDBVersion);
362 sLogger.failure("Database : You can find the logon update queries in the sql/logon/updates sub-directory of your AscEmu source directory.");
363 }
364 else
365 sLogger.failure("Database : Your logon database is too new for this AscEmu version, you need to update your server. Exiting.");
366
367 return false;
368 }
369
370 sLogger.info("Database : Database successfully validated.");
371
372 return true;
373}

394bool MasterLogon::SetLogonConfiguration()
395{
396 logonConfig.loadConfigValues();
397
398 std::vector<std::string> allowedIPs = AscEmu::Util::Strings::split(logonConfig.logonServer.allowedIps, " ");
399 std::vector<std::string> allowedModIPs = AscEmu::Util::Strings::split(logonConfig.logonServer.allowedModIps, " ");
400
401 std::lock_guard lock(m_allowedIpLock);
402 m_allowedIps.clear();
403 m_allowedModIps.clear();
404
405 std::vector<std::string>::iterator itr;
406
407 for (const auto allowedIP : allowedIPs)
408 {
409 std::string::size_type i = allowedIP.find('/');
410 if (i == std::string::npos)
411 {
412 sLogger.failure("Ips: {} could not be parsed. Ignoring", allowedIP);
413 continue;
414 }
415
416 std::string stmp = allowedIP.substr(0, i);
417 std::string smask = allowedIP.substr(i + 1);
418
419 const unsigned int ipraw = Util::makeIP(stmp.c_str());
420 const unsigned char ipmask = static_cast<char>(atoi(smask.c_str()));
421 if (ipraw == 0 || ipmask == 0)
422 {
423 sLogger.failure("Ips: {} could not be parsed. Ignoring", allowedIP);
424 continue;
425 }
426
427 AllowedIP tmp;
428 tmp.Bytes = ipmask;
429 tmp.IP = ipraw;
430 m_allowedIps.push_back(tmp);
431 }
432
433 for (const auto allowedModIP : allowedModIPs)
434 {
435 std::string::size_type i = allowedModIP.find('/');
436 if (i == std::string::npos)
437 {
438 sLogger.failure("ModIps: {} could not be parsed. Ignoring", allowedModIP);
439 continue;
440 }
441
442 std::string stmp = allowedModIP.substr(0, i);
443 std::string smask = allowedModIP.substr(i + 1);
444
445 unsigned int ipraw = Util::makeIP(stmp.c_str());
446 unsigned char ipmask = static_cast<char>(atoi(smask.c_str()));
447 if (ipraw == 0 || ipmask == 0)
448 {
449 sLogger.failure("ModIps: {} could not be parsed. Ignoring", allowedModIP);
450 continue;
451 }
452
453 AllowedIP tmp;
454 tmp.Bytes = ipmask;
455 tmp.IP = ipraw;
456 m_allowedModIps.push_back(tmp);
457 }
458
459 sRealmManager.checkServers();
460
461 return true;
462}

490void MasterLogon::_OnSignal(int s)
491{
492 switch (s)
493 {
494#ifndef WIN32
495 case SIGHUP:
496 {
497 sLogger.info("Received SIGHUP signal, reloading accounts.");
498 sAccountMgr.reloadAccounts(true);
499 }

501#endif

503 case SIGTERM:
504 case SIGABRT:
505#ifdef _WIN32

509 break;
510 }
511
512 signal(s, _OnSignal);
