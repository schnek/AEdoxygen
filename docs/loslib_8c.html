199/*
200** About the overflow check: an overflow cannot occur when time
201** is represented by a lua_Integer, because either lua_Integer is
202** large enough to represent all int fields or it is not large enough
203** to represent a time that cause a field to overflow. However, if
204** times are represented as doubles and lua_Integer is int, then the
205** time 0x1.e1853b0d184f6p+55 would cause an overflow when adding 1900
206** to compute the year.

208static void setfield (lua_State *L, const char *key, int value, int delta) {
209 #if (defined(LUA_NUMTIME) && LUA_MAXINTEGER <= INT_MAX)
210 if (l_unlikely(value > LUA_MAXINTEGER - delta))
211 luaL_error(L, "field '%s' is out-of-bound", key);
212 #endif
213 lua_pushinteger(L, (lua_Integer)value + delta);
214 lua_setfield(L, -2, key);
215}

250static int getfield (lua_State *L, const char *key, int d, int delta) {
251 int isnum;
252 int t = lua_getfield(L, -1, key); /* get field and its type */
253 lua_Integer res = lua_tointegerx(L, -1, &isnum);
254 if (!isnum) { /* field is not an integer? */
255 if (l_unlikely(t != LUA_TNIL)) /* some other value? */
256 return luaL_error(L, "field '%s' is not an integer", key);
257 else if (l_unlikely(d < 0)) /* absent field; no default? */
258 return luaL_error(L, "field '%s' missing in date table", key);
259 res = d;
260 }
261 else {
262 /* unsigned avoids overflow when lua_Integer has 32 bits */
263 if (!(res >= 0 ? (lua_Unsigned)res <= (lua_Unsigned)INT_MAX + delta
264 : (lua_Integer)INT_MIN + delta <= res))
265 return luaL_error(L, "field '%s' is out-of-bound", key);
266 res -= delta;
267 }
268 lua_pop(L, 1);
269 return (int)res;
270}

322 else {
