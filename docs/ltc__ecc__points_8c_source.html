30ecc_point *ltc_ecc_new_point(void)
31{
32 ecc_point *p;
33 p = XCALLOC(1, sizeof(*p));
34 if (p == NULL) {
35 return NULL;
36 }
37 if (mp_init_multi(&p->x, &p->y, &p->z, NULL) != CRYPT_OK) {
38 XFREE(p);
39 return NULL;
40 }
41 return p;
42}
