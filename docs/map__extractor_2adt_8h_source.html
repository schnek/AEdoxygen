197class adt_MH2O
198{
199public:
200 union{
201 uint32_t fcc;
202 char fcc_txt[4];
203 };
204 uint32_t size;
205
206 struct adt_LIQUID{
207 uint32_t offsData1;
208 uint32_t used;
209 uint32_t offsData2;
210 } liquid[ADT_CELLS_PER_GRID][ADT_CELLS_PER_GRID];
211
212 bool prepareLoadedData();
213
214 adt_liquid_header *getLiquidData(int x, int y)
215 {
216 if (liquid[x][y].used && liquid[x][y].offsData1)
217 return (adt_liquid_header *)((uint8_t*)this + 8 + liquid[x][y].offsData1);
218 return 0;
219 }
220
221 float *getLiquidHeightMap(adt_liquid_header *h)
222 {
223 if (h->formatFlags & ADT_LIQUID_HEADER_NO_HIGHT)
224 return 0;
225 if (h->offsData2b)
226 return (float *)((uint8_t*)this + 8 + h->offsData2b);
227 return 0;
228 }
229
230 uint8_t *getLiquidLightMap(adt_liquid_header *h)
231 {
232 if (h->formatFlags&ADT_LIQUID_HEADER_FULL_LIGHT)
233 return 0;
234 if (h->offsData2b)
235 {
236 if (h->formatFlags & ADT_LIQUID_HEADER_NO_HIGHT)
237 return (uint8_t *)((uint8_t*)this + 8 + h->offsData2b);
238 return (uint8_t *)((uint8_t*)this + 8 + h->offsData2b + (h->width+1)*(h->height+1)*4);
239 }
240 return 0;
241 }
242
243 uint32_t *getLiquidFullLightMap(adt_liquid_header *h)
244 {
245 if (!(h->formatFlags&ADT_LIQUID_HEADER_FULL_LIGHT))
246 return 0;
247 if (h->offsData2b)
248 {
249 if (h->formatFlags & ADT_LIQUID_HEADER_NO_HIGHT)
250 return (uint32_t *)((uint8_t*)this + 8 + h->offsData2b);
251 return (uint32_t *)((uint8_t*)this + 8 + h->offsData2b + (h->width+1)*(h->height+1)*4);
252 }
253 return 0;
254 }
255
256 uint64_t getLiquidShowMap(adt_liquid_header *h)
257 {
258 if (h->offsData2a)
259 return *((uint64_t *)((uint8_t*)this + 8 + h->offsData2a));
260 else
261 return 0xFFFFFFFFFFFFFFFFuLL;
262 }
