38template <typename T>
39class ProducerConsumerQueue
40{
41private:
42 std::mutex _queueLock;
43 std::queue<T> _queue;
44 std::condition_variable _condition;
45 std::atomic<bool> _shutdown;
46
47public:
48 ProducerConsumerQueue() : _shutdown(false) { }
49
50 void Push(const T& value)
51 {
52 std::lock_guard<std::mutex> lock(_queueLock);
53 _queue.push(std::move(value));
54
55 _condition.notify_one();
56 }
57
58 bool Empty()
59 {
60 std::lock_guard<std::mutex> lock(_queueLock);
61
62 return _queue.empty();
63 }
64
65 bool Pop(T& value)
66 {
67 std::lock_guard<std::mutex> lock(_queueLock);
68
69 if (_queue.empty() || _shutdown)
70 return false;
71
72 value = _queue.front();
73
74 _queue.pop();
75
76 return true;
77 }
78
79 void WaitAndPop(T& value)
80 {
81 std::unique_lock<std::mutex> lock(_queueLock);
82
83 // we could be using .wait(lock, predicate) overload here but it is broken
84 // https://connect.microsoft.com/VisualStudio/feedback/details/1098841
85 while (_queue.empty() && !_shutdown)
86 _condition.wait(lock);
87
88 if (_queue.empty() || _shutdown)
89 return;
90
91 value = _queue.front();
92
93 _queue.pop();
94 }
95
96 void Cancel()
97 {
98 std::unique_lock<std::mutex> lock(_queueLock);
99
100 while (!_queue.empty())
101 {
102 T& value = _queue.front();
103
104 DeleteQueuedObject(value);
105
106 _queue.pop();
107 }
108
109 _shutdown = true;
110
111 _condition.notify_all();
112 }
113
114private:
115 template<typename E = T>
116 typename std::enable_if<std::is_pointer<E>::value>::type DeleteQueuedObject(E& obj) { delete obj; }
117
118 template<typename E = T>
119 typename std::enable_if<!std::is_pointer<E>::value>::type DeleteQueuedObject(E const& /*packet*/) { }
120};
