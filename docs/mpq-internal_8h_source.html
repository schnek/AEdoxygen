66/* mpq archive header. */
67typedef struct {
68 uint32_t mpq_magic; /* the 0x1A51504D ('MPQ\x1A') signature. */
69 uint32_t header_size; /* mpq archive header size. */
70 uint32_t archive_size; /* size of mpq archive. */
71 uint16_t version; /* 0000 for starcraft and broodwar. */
72 uint16_t block_size; /* size of file block is (512 * 2 ^ block size). */
73 uint32_t hash_table_offset; /* file position of mpq_hash. */
74 uint32_t block_table_offset; /* file position of mpq_block, each entry has 16 bytes. */
75 uint32_t hash_table_count; /* number of entries in hash table. */
76 uint32_t block_table_count; /* number of entries in the block table. */
77} PACK_STRUCT mpq_header_s;
78
79/* mpq extended archive header, used since world of warcraft - the burning crusade. */
80typedef struct {
81 uint64_t extended_offset; /* offset to the beginning of the extended block table, relative to the beginning of the archive. */
82 uint16_t hash_table_offset_high; /* upper 16 bits of the hash table offset for large archives. */
83 uint16_t block_table_offset_high;/* upper 16 bits of the block table offset for large archives.*/
84} PACK_STRUCT mpq_header_ex_s;
85
86/* hash entry, all files in the archive are searched by their hashes. */
87typedef struct {
88 uint32_t hash_a; /* the first two uint32_ts are the encrypted file. */
89 uint32_t hash_b; /* the first two uint32_ts are the encrypted file. */
90 uint16_t locale; /* locale information. */
91 uint16_t platform; /* platform information and zero is default. */
92 uint32_t block_table_index; /* index to file description block. */
93} PACK_STRUCT mpq_hash_s;
94
95/* file description block contains informations about the file. */
96typedef struct {
97 uint32_t offset; /* block file starting position in the archive. */
98 uint32_t packed_size; /* packed file size. */
99 uint32_t unpacked_size; /* unpacked file size. */
100 uint32_t flags; /* flags. */
101} PACK_STRUCT mpq_block_s;
102
103/* extended file description block contains information about the offset beyond 2^32 (4GB). */
104typedef struct {
105 uint16_t offset_high; /* upper 16 bit of the file offset in archive. */
106} PACK_STRUCT mpq_block_ex_s;
107
108/* file structure used since diablo 1.00 (0x38 bytes). */
109typedef struct {
110 uint32_t seed; /* seed used for file decrypt. */
111 uint32_t *packed_offset; /* position of each file block (only for packed files). */
112 uint32_t open_count; /* number of times it has been opened - used for freeing */
113} PACK_STRUCT mpq_file_s;
114
115/* map structure for valid blocks and hashes (first seen in warcraft 3 archives). */
116typedef struct {
117 uint32_t block_table_indices; /* real mapping for file number to block entry. */
118 uint32_t block_table_diff; /* block table difference between valid blocks and invalid blocks before. */
119} PACK_STRUCT mpq_map_s;
120#include "pack_end.h"

122/* archive structure used since diablo 1.00 by blizzard. */
123struct mpq_archive {
124
125 /* generic file information. */
126 FILE *fp; /* file handle. */
127
128 /* generic size information. */
129 uint32_t block_size; /* size of the mpq block. */
130 off_t archive_offset; /* absolute start position of archive. */
131
132 /* archive related buffers and tables. */
133 mpq_header_s mpq_header; /* mpq file header. */
134 mpq_header_ex_s mpq_header_ex; /* mpq extended file header. */
135 mpq_hash_s *mpq_hash; /* hash table. */
136 mpq_block_s *mpq_block; /* block table. */
137 mpq_block_ex_s *mpq_block_ex; /* extended block table. */
138 mpq_file_s **mpq_file; /* pointer to the file pointers which are opened. */
139
140 /* non archive structure related members. */
141 mpq_map_s *mpq_map; /* map table between valid blocks and hashes. */
142 uint32_t files; /* number of files in archive, which could be extracted. */
143};
