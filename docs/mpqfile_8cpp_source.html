25MPQFile::MPQFile(HANDLE mpq, const char* filename, bool warnNoExist /*= true*/) :
26 eof(false),
27 buffer(0),
28 pointer(0),
29 size(0)
30{
31 HANDLE file;
32 if (!SFileOpenFileEx(mpq, filename, SFILE_OPEN_FROM_MPQ, &file))
33 {
34 if (warnNoExist || GetLastError() != ERROR_FILE_NOT_FOUND)
35 fprintf(stderr, "Can't open %s, err=%u!\n", filename, GetLastError());
36 eof = true;
37 return;
38 }
39
40 DWORD hi = 0;
41 size = SFileGetFileSize(file, &hi);
42
43 if (hi)
44 {
45 fprintf(stderr, "Can't open %s, size[hi] = %u!\n", filename, uint32_t(hi));
46 SFileCloseFile(file);
47 eof = true;
48 return;
49 }
50
51 if (size <= 1)
52 {
53 fprintf(stderr, "Can't open %s, size = %u!\n", filename, uint32_t(size));
54 SFileCloseFile(file);
55 eof = true;
56 return;
57 }
58
59 DWORD read = 0;
60 buffer = new char[size];
61 if (!SFileReadFile(file, buffer, size, &read, nullptr) || size != read)
62 {
63 fprintf(stderr, "Can't read %s, size=%u read=%u!\n", filename, uint32_t(size), uint32_t(read));
64 SFileCloseFile(file);
65 eof = true;
66 return;
67 }
68
69 SFileCloseFile(file);
70}

72size_t MPQFile::read(void* dest, size_t bytes)
73{
74 if (eof) return 0;
75
76 size_t rpos = pointer + bytes;
77 if (rpos > size)
78 {
79 bytes = size - pointer;
80 eof = true;
81 }
82
83 memcpy(dest, &(buffer[pointer]), bytes);
84
85 pointer = rpos;
86
87 return bytes;
88}
