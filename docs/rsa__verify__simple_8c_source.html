44
45 /* default to invalid */
46 *stat = 0;
47
48 /* get modulus len in bits */
49 modulus_bitlen = mp_count_bits( (key->N));
50
51 /* outlen must be at least the size of the modulus */
52 modulus_bytelen = mp_unsigned_bin_size( (key->N));
53 if (modulus_bytelen != siglen) {
54 return CRYPT_INVALID_PACKET;
55 }
56
57 /* allocate temp buffer for decoded sig */
58 tmpbuf = XMALLOC(siglen);
59 if (tmpbuf == NULL) {
60 return CRYPT_MEM;
61 }
62
63 /* RSA decode it */
64 x = siglen;
65 if ((err = ltc_mp.rsa_me(sig, siglen, tmpbuf, &x, PK_PUBLIC, key)) != CRYPT_OK) {
66 XFREE(tmpbuf);
67 return err;
68 }
69
70 /* make sure the output is the right size */
71 if (x != siglen) {
72 XFREE(tmpbuf);
73 return CRYPT_INVALID_PACKET;
74 }
75
76 /* compare the decrypted signature with the given hash */
77 if(x == hashlen && XMEMCMP(tmpbuf, hash, hashlen) == 0)
78 *stat = 1;
79
80#ifdef LTC_CLEAN_STACK
81 zeromem(tmpbuf, siglen);
82#endif
83 XFREE(tmpbuf);
84 return CRYPT_OK;
85}
