56// This is a common script to handle Death Knight Presences
57class Presences : public SpellScript
58{
59public:
60 virtual bool isBloodPresence() const { return false; }
61 virtual bool isFrostPresence() const { return false; }
62 virtual bool isUnholyPresence() const { return false; }
63
64 void onAuraApply(Aura* aur) override
65 {
66 // Handle improved presences here
67 Aura const* improvedBloodPresence = nullptr;
68 Aura const* improvedFrostPresence = nullptr;
69 Aura const* improvedUnholyPresence = nullptr;
70
71 // Get all auras in single loop
72 for (uint16_t i = AuraSlots::PASSIVE_SLOT_START; i < AuraSlots::PASSIVE_SLOT_END; ++i)
73 {
74 const auto* const unitAura = aur->getOwner()->getAuraWithAuraSlot(i);
75 if (unitAura == nullptr)
76 continue;
77
78 if (unitAura->getSpellId() == SPELL_IMPROVED_BLOOD_PRESENCE_R2)
79 improvedBloodPresence = unitAura;
80 if (improvedBloodPresence == nullptr && unitAura->getSpellId() == SPELL_IMPROVED_BLOOD_PRESENCE_R1)
81 improvedBloodPresence = unitAura;
82
83 if (unitAura->getSpellId() == SPELL_IMPROVED_FROST_PRESENCE_R2)
84 improvedFrostPresence = unitAura;
85 if (improvedFrostPresence == nullptr && unitAura->getSpellId() == SPELL_IMPROVED_FROST_PRESENCE_R1)
86 improvedFrostPresence = unitAura;
87
88 if (unitAura->getSpellId() == SPELL_IMPROVED_UNHOLY_PRESENCE_R2)
89 improvedUnholyPresence = unitAura;
90 if (improvedUnholyPresence == nullptr && unitAura->getSpellId() == SPELL_IMPROVED_UNHOLY_PRESENCE_R1)
91 improvedUnholyPresence = unitAura;
92 }
93
94 // Cast Improved Blood Presence in Frost and Unholy presences
95 if (!isBloodPresence() && improvedBloodPresence != nullptr)
96 {
97 SpellForcedBasePoints forcedBasePoints;
98 forcedBasePoints.set(EFF_INDEX_0, improvedBloodPresence->getAuraEffect(EFF_INDEX_0)->getEffectDamage());
99 aur->getOwner()->castSpell(aur->getOwner(), sSpellMgr.getSpellInfo(SPELL_IMPROVED_BLOOD_PRESENCE_DUMMY), forcedBasePoints, true);
100 }
101
102 // Cast Improved Frost Presence in Blood and Unholy presences
103 if (!isFrostPresence() && improvedFrostPresence != nullptr)
104 {
105 SpellForcedBasePoints forcedBasePoints;
106 forcedBasePoints.set(EFF_INDEX_0, improvedFrostPresence->getAuraEffect(EFF_INDEX_0)->getEffectDamage());
107 aur->getOwner()->castSpell(aur->getOwner(), sSpellMgr.getSpellInfo(SPELL_IMPROVED_FROST_PRESENCE_DUMMY), forcedBasePoints, true);
108 }
109
110 if (improvedUnholyPresence != nullptr)
111 {
112#if VERSION_STRING == WotLK
113 // Cast Unholy Presence speed aura in Blood and Frost presences
114 if (!isUnholyPresence())
115 {
116 SpellForcedBasePoints forcedBasePoints;
117 forcedBasePoints.set(EFF_INDEX_0, improvedUnholyPresence->getAuraEffect(EFF_INDEX_0)->getEffectDamage());
118 aur->getOwner()->castSpell(aur->getOwner(), sSpellMgr.getSpellInfo(SPELL_UNHOLY_PRESENCE_MOVE_SPEED), forcedBasePoints, true);
119 }
120 // Cast Improved Unholy Presence in Unholy Presence
121 else
122 {
123 SpellForcedBasePoints forcedBasePoints;
124 const auto basePoints = improvedUnholyPresence->getSpellInfo()->calculateEffectValue(EFF_INDEX_1);
125 for (uint8_t i = 0; i < MAX_SPELL_EFFECTS; ++i)
126 forcedBasePoints.set(i, basePoints);
127
128 aur->getOwner()->castSpell(aur->getOwner(), sSpellMgr.getSpellInfo(SPELL_IMPROVED_UNHOLY_PRESENCE_DUMMY), forcedBasePoints, true);
129 }
130#else
131 // Cast Improved Unholy Presence in Blood and Frost presences
132 if (!isUnholyPresence())
133 {
134 SpellForcedBasePoints forcedBasePoints;
135 forcedBasePoints.set(EFF_INDEX_0, improvedUnholyPresence->getAuraEffect(EFF_INDEX_0)->getEffectDamage());
136 aur->getOwner()->castSpell(aur->getOwner(), sSpellMgr.getSpellInfo(SPELL_IMPROVED_UNHOLY_PRESENCE_DUMMY), forcedBasePoints, true);
137 }
138#endif
139 }
140 }

154 }

175#if VERSION_STRING == WotLK
176class BloodPresenceHeal : public SpellScript
177{
178public:
179 bool canProc(SpellProc* /*spellProc*/, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, DamageInfo damageInfo) override
180 {
181 return damageInfo.realDamage > 0;
182 }
183
184 SpellScriptExecuteState onDoProcEffect(SpellProc* spellProc, Unit* /*victim*/, SpellInfo const* /*castingSpell*/, DamageInfo damageInfo) override
185 {
186 heal = damageInfo.realDamage * spellProc->getOverrideEffectDamage(EFF_INDEX_0) / 100;
187 if (heal == 0)
188 return SpellScriptExecuteState::EXECUTE_PREVENT;
189
190 return SpellScriptExecuteState::EXECUTE_OK;
191 }
192
193 SpellScriptExecuteState onCastProcSpell(SpellProc* /*spellProc*/, Unit* /*caster*/, Unit* /*victim*/, Spell* spell) override
194 {
195 spell->forced_basepoints->set(EFF_INDEX_0, heal);
196 heal = 0;
197 return SpellScriptExecuteState::EXECUTE_OK;
198 }
199
200private:
201 uint32_t heal = 0;
202};
