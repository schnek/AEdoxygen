26void CompressSparse(void * pvOutBuffer, int * pcbOutBuffer, void * pvInBuffer, int cbInBuffer)
27{
28 unsigned char * pbOutBufferEnd = (unsigned char *)pvOutBuffer + *pcbOutBuffer;
29 unsigned char * pbInBufferEnd = (unsigned char *)pvInBuffer + cbInBuffer;
30 unsigned char * pbLastNonZero = (unsigned char *)pvInBuffer;
31 unsigned char * pbOutBuffer0 = (unsigned char *)pvOutBuffer;
32 unsigned char * pbInBuffPtr = (unsigned char *)pvInBuffer;
33 unsigned char * pbOutBuffer = (unsigned char *)pvOutBuffer;
34 unsigned char * pbInBuffer = (unsigned char *)pvInBuffer;
35 size_t NumberOfNonZeros;
36 size_t NumberOfZeros;
37
38 // There must be at least 4 bytes of free space in the output buffer now
39 if((pbInBuffer + 4) >= pbInBufferEnd)
40 return;
41
42 // Put the original data length (in little endian)
43 *pbOutBuffer++ = (unsigned char)(cbInBuffer >> 0x18);
44 *pbOutBuffer++ = (unsigned char)(cbInBuffer >> 0x10);
45 *pbOutBuffer++ = (unsigned char)(cbInBuffer >> 0x08);
46 *pbOutBuffer++ = (unsigned char)(cbInBuffer >> 0x00);
47
48 // If there is at least 3 bytes in the input buffer, do this loop
49 while(pbInBuffer < (pbInBufferEnd - 3))
50 {
51 // Reset the zero count and frontal pointer
52 pbLastNonZero = pbInBuffer;
53 pbInBuffPtr = pbInBuffer;
54 NumberOfZeros = 0;
55
56 if(pbInBuffPtr < pbInBufferEnd)
57 {
58 do
59 {
60 // Count number of zeros
61 if(*pbInBuffPtr == 0)
62 {
63 NumberOfZeros++;
64 }
65 else
66 {
67 // Were there at least 3 zeros before? If yes, we need to flush the data
68 if(NumberOfZeros >= 3)
69 break;
70 pbLastNonZero = pbInBuffPtr + 1;
71 NumberOfZeros = 0;
72 }
73 }
74 while(++pbInBuffPtr < pbInBufferEnd);
75 }
76
77 // Get number of nonzeros that we found so far and flush them
78 NumberOfNonZeros = pbLastNonZero - pbInBuffer;
79 if(NumberOfNonZeros != 0)
80 {
81 // Process blocks that are longer than 0x81 nonzero bytes
82 while(NumberOfNonZeros > 0x81)
83 {
84 // Verify if we still have enough space in output buffer
85 if((pbOutBuffer + 0x81) >= pbOutBufferEnd)
86 return;
87
88 // Put marker that means "0x80 of nonzeros"
89 *pbOutBuffer++ = 0xFF;
90 memcpy(pbOutBuffer, pbInBuffer, 0x80);
91
92 // Adjust counter of nonzeros and both pointers
93 NumberOfNonZeros -= 0x80;
94 pbOutBuffer += 0x80;
95 pbInBuffer += 0x80;
96 }
97
98 // BUGBUG: The following code will be triggered if the NumberOfNonZeros
99 // was 0x81 before. It will copy just one byte. This seems like a bug to me,
100 // but since I want StormLib to be exact like Blizzard code is, I will keep
101 // it that way here
102 if(NumberOfNonZeros > 0x80)
103 {
104 // Verify if we still have enough space in output buffer
105 if((pbOutBuffer + 2) >= pbOutBufferEnd)
106 return;
107
108 // Put marker that means "1 nonzero byte"
109 *pbOutBuffer++ = 0x80;
110 memcpy(pbOutBuffer, pbInBuffer, 1);
111
112 // Adjust counter of nonzeros and both pointers
113 NumberOfNonZeros--;
114 pbOutBuffer++;
115 pbInBuffer++;
116 }
117
118 // If there is 1 nonzero or more, put the block
119 if(NumberOfNonZeros >= 0x01)
120 {
121 // Verify if we still have enough space in output buffer
122 if((pbOutBuffer + NumberOfNonZeros + 1) >= pbOutBufferEnd)
123 return;
124
125 // Put marker that means "Several nonzero bytes"
126 *pbOutBuffer++ = (unsigned char)(0x80 | (NumberOfNonZeros - 1));
127 memcpy(pbOutBuffer, pbInBuffer, NumberOfNonZeros);
128
129 // Adjust pointers
130 pbOutBuffer += NumberOfNonZeros;
131 pbInBuffer += NumberOfNonZeros;
132 }
133 }
134
135 // Now flush all zero bytes
136 while(NumberOfZeros > 0x85)
137 {
138 // Do we have at least 2 bytes in the output buffer ?
139 if((pbOutBuffer + 1) >= pbOutBufferEnd)
140 return;
141
142 // Put marker that means "0x82 zeros"
143 *pbOutBuffer++ = 0x7F;
144
145 // Adjust zero counter and input pointer
146 NumberOfZeros -= 0x82;
147 pbInBuffer += 0x82;
148 }
149
150 // If we got more than 0x82 zeros, flush 3 of them now
151 if(NumberOfZeros > 0x82)
152 {
153 // Do we have at least 2 bytes in the output buffer ?
154 if((pbOutBuffer + 1) >= pbOutBufferEnd)
155 return;
156
157 // Put marker that means "0x03 zeros"
158 *pbOutBuffer++ = 0;
159
160 // Adjust zero counter and input pointer
161 NumberOfZeros -= 0x03;
162 pbInBuffer += 0x03;
163 }
164
165 // Is there at least three zeros ?
166 if(NumberOfZeros >= 3)
167 {
168 // Do we have at least 2 bytes in the output buffer ?
169 if((pbOutBuffer + 1) >= pbOutBufferEnd)
170 return;
171
172 // Put marker that means "Several zeros"
173 *pbOutBuffer++ = (unsigned char)(NumberOfZeros - 3);
174
175 // Adjust pointer
176 pbInBuffer += NumberOfZeros;
177 }
178 }
179
180 // Flush last three bytes
181 if(pbInBuffer < pbInBufferEnd)
182 {
183 pbInBuffPtr = pbInBuffer;
184
185 for(;;)
186 {
187 if(*pbInBuffPtr++ != 0)
188 {
189 // Get number of bytes remaining
190 NumberOfNonZeros = (pbInBufferEnd - pbInBuffer);
191
192 // Not enough space in the output buffer ==> exit
193 if((pbOutBuffer + NumberOfNonZeros + 1) >= pbOutBufferEnd)
194 return;
195
196 // Terminate with a marker that means "0x80 of nonzeros"
197 *pbOutBuffer++ = 0xFF;
198 memcpy(pbOutBuffer, pbInBuffer, NumberOfNonZeros);
199
200 // Adjust pointer
201 pbOutBuffer += NumberOfNonZeros;
202 break;
203 }
204 else
205 {
206 // Is there are more chars in the input buffer
207 if(pbInBuffPtr < pbInBufferEnd)
208 continue;
209
210 // If the compression will not compress it by even 1 byte, do nothing
211 if((pbOutBuffer + 1) >= pbOutBufferEnd)
212 return;
213
214 // Terminate with a chunk that means "0x82 of zeros"
215 *pbOutBuffer++ = 0x7F;
216 break;
217 }
218 }
219 }
220
221 // Out the length of the output buffer
222 *pcbOutBuffer = (int)(pbOutBuffer - pbOutBuffer0);
223}

225int DecompressSparse(void * pvOutBuffer, int * pcbOutBuffer, void * pvInBuffer, int cbInBuffer)
226{
227 unsigned char * pbInBufferEnd = (unsigned char *)pvInBuffer + cbInBuffer;
228 unsigned char * pbOutBuffer = (unsigned char *)pvOutBuffer;
229 unsigned char * pbInBuffer = (unsigned char *)pvInBuffer;
230 unsigned int cbChunkSize;
231 unsigned int cbOutBuffer = 0;
232 unsigned int OneByte;
233
234 // Don't decompress anything that is shorter than 5 bytes
235 if(cbInBuffer < 5)
236 return 0;
237
238 // Get the 32-bits from the input stream
239 OneByte = *pbInBuffer++;
240 cbOutBuffer |= (OneByte << 0x18);
241 OneByte = *pbInBuffer++;
242 cbOutBuffer |= (OneByte << 0x10);
243 OneByte = *pbInBuffer++;
244 cbOutBuffer |= (OneByte << 0x08);
245 OneByte = *pbInBuffer++;
246 cbOutBuffer |= (OneByte << 0x00);
247
248 // Verify the size of the stream against the output buffer size
249 if(cbOutBuffer > (unsigned int)*pcbOutBuffer)
250 return 0;
251
252 // Put the output size to the buffer
253 *pcbOutBuffer = cbOutBuffer;
254
255 // Process the input buffer
256 while(pbInBuffer < pbInBufferEnd)
257 {
258 // Get (next) byte from the stream
259 OneByte = *pbInBuffer++;
260
261 // If highest bit, it means that that normal data follow
262 if(OneByte & 0x80)
263 {
264 cbChunkSize = (OneByte & 0x7F) + 1;
265 cbChunkSize = (cbChunkSize < cbOutBuffer) ? cbChunkSize : cbOutBuffer;
266 memcpy(pbOutBuffer, pbInBuffer, cbChunkSize);
267 pbInBuffer += cbChunkSize;
268 }
269 else
270 {
271 cbChunkSize = (OneByte & 0x7F) + 3;
272 cbChunkSize = (cbChunkSize < cbOutBuffer) ? cbChunkSize : cbOutBuffer;
273 memset(pbOutBuffer, 0, cbChunkSize);
274 }
275
276 // Increment output buffer pointer
277 pbOutBuffer += cbChunkSize;
278 cbOutBuffer -= cbChunkSize;
279 }
280
281 return 1;
282}
