20/**
21 Interpolates a property according to a piecewise linear spline. This provides
22 C0 continuity but the derivatives are not smooth.
23 <P>
24 Example:
25 <CODE>
26 const double times[] = {MIDNIGHT, SUNRISE - HOUR, SUNRISE, SUNRISE + sunRiseAndSetTime / 4, SUNRISE + sunRiseAndSetTime, SUNSET - sunRiseAndSetTime, SUNSET - sunRiseAndSetTime / 2, SUNSET, SUNSET + HOUR/2, DAY};
27 const Color3 color[] = {Color3(0, .0, .1), Color3(0, .0, .1), Color3::black(), Color3::black(), Color3::white() * .25, Color3::white() * .25, Color3(.5, .2, .2), Color3(.05, .05, .1), Color3(0, .0, .1), Color3(0, .0, .1)};
28 ambient = linearSpline(time, times, color, 10);
29 </CODE>
30
31 See also G3D::Spline
32
33 @param x The spline is a function of x; this is the sample to choose.
34 @param controlX controlX[i], controlY[i] is a control points. It is assumed
35 that controlX are strictly increasing. XType must support
36 the "<" operator and a subtraction operator that returns
37 a number.
38 @param controlY YType must support multiplication and addition.
39 @param numControl The number of control points.
40 */
41template<class XType, class YType>
42YType linearSpline(double x, const XType* controlX, const YType* controlY, int numControl) {
43 debugAssert(numControl >= 1);
44
45 // Off the beginning
46 if ((numControl == 1) || (x < controlX[0])) {
47 return controlY[0];
48 }
49
50 for (int i = 1; i < numControl; ++i) {
51 if (x < controlX[i]) {
52 const double alpha = (double)(controlX[i] - x) / (controlX[i] - controlX[i - 1]);
53 return controlY[i] * (1 - alpha) + controlY[i - 1] * alpha;
54 }
55 }
56
57 // Off the end
58 return controlY[numControl - 1];
59}

63template<class YType> YType cyclicCatmullRomSpline(
64 double t,
65 const YType* controlY,
66 int numPoints) {
67
68 debugAssert(numPoints >= 3);
69
70 t = wrap(t, numPoints);
71
72 // Find the indices of adjacent control points
73 int i = iFloor(t);
74
75 // Compute the distance from the control point
76 t = t - i;
77
78 // Shift back one point for correct indexing
79 i += numPoints - 1;
80
81 // Pick up four control points
82 const YType& P0 = controlY[(i + 0) % numPoints];
83 const YType& P1 = controlY[(i + 1) % numPoints];
84 const YType& P2 = controlY[(i + 2) % numPoints];
85 const YType& P3 = controlY[(i + 3) % numPoints];
86
87 return 0.5 * ((2 * P1) +
88 (-P0 + P2) * t +
89 (2*P0 - 5*P1 + 4*P2 - P3) * t*t +
90 (-P0 + 3*P1- 3*P2 + P3) * t*t*t);
91}

93/**
94 A cubic spline with regularly spaced
95 control points. The spline interpolates
96 the control points. The spline
97 will wrap from the last point back to the first.
98
99 The t parameter is on the range [0, controlY.size()],
100 where integers correspond to control points exactly.
101
102 See also G3D::Spline
103
104 @cite http://www.mvps.org/directx/articles/catmull/
105*/
106template<class YType> YType cyclicCatmullRomSpline(
107 double t,
108 const Array<YType>& controlY) {
109
110 int numPoints = controlY.size();
111 return cyclicCatmullRomSpline(t, controlY.getCArray(), numPoints);
112}
