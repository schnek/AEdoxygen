67std::string getWowExeName()
68{
69 const std::string path("./");
70 const std::string extension(".exe");
71 for (const auto& files : fs::recursive_directory_iterator(path))
72 {
73 if (files.path().extension() == extension)
74 {
75 auto name = files.path().stem().string().substr(0, 6);
76 std::transform(name.begin(), name.end(), name.begin(), [](unsigned char chars) { return std::tolower(chars); });
77 if (name + extension == "wow.exe")
78 return files.path().stem().string() + extension;
79 }
80 }
81
82 return "";
83}

86uint32_t getBuildNumber()
87{
88 // buffers used for working on the file's bytes
89 unsigned char byteSearchBuffer[1]; // used for reading in a single character, ready to be
90 // tested for the required text we are searching for: 1, 5, 6, or 8
91 unsigned char jumpBytesBuffer[128]; // used for skipping past the bytes from the file's start
92 // to the base # area, before we start searching for the base #, for faster processing
93
94 unsigned char preWOTLKbuildNumber[3]; // will hold the last 3 digits of the build number
95 unsigned char postTBCbuildNumber[4]; // will hold the last 4 digits of the build number
96
97 // These do not include the first digit
98 // as the first digit is used to locate the possible location of the build number
99 // then the following bytes are grabbed, 3 for pre WOTLK, 4 for WOTLK and later.
100 // Those grabbed bytes are then compared with the below variables in order to idenity the exe's build
101 unsigned char vanillaBuild1[3] = { 0x38, 0x37, 0x35 }; // (5)875
102 unsigned char vanillaBuild2[3] = { 0x30, 0x30, 0x35 }; // (6)005
103 unsigned char vanillaBuild3[3] = { 0x31, 0x34, 0x31 }; // (6)141
104 unsigned char tbcBuild[3] = { 0x36, 0x30, 0x36 }; // (8)606
105 unsigned char wotlkBuild[4] = { 0x32, 0x33, 0x34, 0x30 }; // (1)2340
106 unsigned char cataBuild[4] = { 0x35, 0x35, 0x39, 0x35 }; // (1)5595
107 unsigned char mopBuild[4] = { 0x38, 0x34, 0x31, 0x34 }; // (1)8414
108
109 const auto wowExeFile = fopen(getWowExeName().c_str(), "rb");
110
111 /// jump over as much of the file as possible, before we start searching for the base #
112 for (auto i = 0; i < 3300; ++i)
113 fread(jumpBytesBuffer, sizeof(jumpBytesBuffer), 1, wowExeFile);
114
115 // Search for the build #
116 while (fread(byteSearchBuffer, 1, 1, wowExeFile))
117 {
118 // we are looking for 1, 5, 6, or 8
119 // these values are the first digit of the build versions we are interested in
120 // Vanilla and TBC
121 if (byteSearchBuffer[0] == 0x35 || byteSearchBuffer[0] == 0x36 || byteSearchBuffer[0] == 0x38)
122 {
123 // grab the next 4 bytes
124 fread(preWOTLKbuildNumber, sizeof(preWOTLKbuildNumber), 1, wowExeFile);
125
126 if (!memcmp(preWOTLKbuildNumber, vanillaBuild1, sizeof(preWOTLKbuildNumber))) // build is Vanilla?
127 return 5875;
128
129 if (!memcmp(preWOTLKbuildNumber, vanillaBuild2, sizeof(preWOTLKbuildNumber))) // build is Vanilla?
130 return 6005;
131
132 if (!memcmp(preWOTLKbuildNumber, vanillaBuild3, sizeof(preWOTLKbuildNumber))) // build is Vanilla?
133 return 6141;
134
135 if (!memcmp(preWOTLKbuildNumber, tbcBuild, sizeof(preWOTLKbuildNumber))) // build is TBC?
136 return 8606;
137 }
138
139 // WOTLK, CATA, MoP
140 if (byteSearchBuffer[0] == 0x31)
141 {
142 // grab the next 4 bytes
143 fread(postTBCbuildNumber, sizeof(postTBCbuildNumber), 1, wowExeFile);
144
145 if (!memcmp(postTBCbuildNumber, wotlkBuild, sizeof(postTBCbuildNumber))) /// build is WOTLK?
146 return 12340;
147
148 if (!memcmp(postTBCbuildNumber, cataBuild, sizeof(postTBCbuildNumber))) /// build is CATA?
149 return 15595;
150
151 if (!memcmp(postTBCbuildNumber, mopBuild, sizeof(postTBCbuildNumber))) /// build is MoP?
152 return 18414;
153 }
154 }
155
156 return 0;
157}

289void HandleArgs(int argc, char* arg[])
290{
291 for (int c = 1; c < argc; ++c)
292 {
293 // i - input path
294 // o - output path
295 // e - extract only MAP(1)/DBC(2) - standard both(3)
296 // f - use float to int conversion
297 // h - limit minimum height
298 if (arg[c][0] != '-')
299 Usage(arg[0]);
300
301 switch (arg[c][1])
302 {
303 case 'i':
304 if (c + 1 < argc && strlen(arg[c + 1]) < MAX_PATH_LENGTH) // all ok
305 {
306 strncpy(input_path, arg[c++ + 1], MAX_PATH_LENGTH);
307 input_path[MAX_PATH_LENGTH - 1] = '\0';
308 }
309 else
310 Usage(arg[0]);
311 break;
312 case 'o':
313 if (c + 1 < argc && strlen(arg[c + 1]) < MAX_PATH_LENGTH) // all ok
314 {
315 strncpy(output_path, arg[c++ + 1], MAX_PATH_LENGTH);
316 output_path[MAX_PATH_LENGTH - 1] = '\0';

319 Usage(arg[0]);

322 if (c + 1 < argc) // all ok
323 CONF_allow_float_to_int = atoi(arg[(c++) + 1]) != 0;

334 else
335 Usage(arg[0]);
336 break;
337 }
338 }

341uint32_t ReadBuild(int locale)
342{
343 // include build info file also if available
344
345 std::string filename = std::string("component.wow-") + langs[locale] + ".txt";
346 printf("Read %s file... ", filename.c_str());
347
348 MPQFile m(filename.c_str());
349 if (m.isEof())
350 {
351 printf("Fatal error: Not found %s file!\n", filename.c_str());
352 exit(1);
353 }
354
355 std::string text = std::string(m.getPointer(), m.getSize());
356 m.close();
357
358 size_t pos = text.find("version=\"");
359 size_t pos1 = pos + strlen("version=\"");
360 size_t pos2 = text.find("\"", pos1);
361 if (pos == text.npos || pos2 == text.npos || pos1 >= pos2)
362 {
363 printf("Fatal error: Invalid %s file format!\n", filename.c_str());
364 exit(1);
365 }
366
367 std::string build_str = text.substr(pos1, pos2 - pos1);
368
369 int build = atoi(build_str.c_str());
370 if (build <= 0)
371 {
372 printf("Fatal error: Invalid %s file format!\n", filename.c_str());
373 exit(1);
374 }
375
376 std::cout << "Detected file build: " << build << std::endl;
377
378 return build;
379}

554bool ConvertADT(char* filename, char* filename2, int /*cell_y*/, int /*cell_x*/, uint32_t build)
555{
556 ADT_file adt;
557
558 if (!adt.loadFile(filename))
559 return false;
560
561 adt_MCIN* cells = adt.a_grid->getMCIN();
562 if (!cells)
563 {
564 printf("Can't find cells in '%s'\n", filename);
565 return false;
566 }
567
568 memset(liquid_show, 0, sizeof(liquid_show));
569 memset(liquid_flags, 0, sizeof(liquid_flags));
570 memset(liquid_entry, 0, sizeof(liquid_entry));
571
572 // Prepare map header
573 map_fileheader map;
574 map.mapMagic = *reinterpret_cast<uint32_t const*>(MAP_MAGIC);
575 map.versionMagic = *reinterpret_cast<uint32_t const*>(MAP_VERSION_MAGIC);
576 map.buildMagic = build;
577
578 // Get area flags data
579 for (int i = 0; i < ADT_CELLS_PER_GRID; i++)
580 {
581 for (int j = 0; j < ADT_CELLS_PER_GRID; j++)
582 {
583 adt_MCNK* cell = cells->getMCNK(i, j);
584 uint32_t areaid = cell->areaid;
585 if (areaid && areaid <= maxAreaId)
586 {
587 if (areas[areaid] != 0xffff)
588 {
589 area_flags[i][j] = areas[areaid];
590 continue;
591 }
592 printf("File: %s\nCan't find area flag for areaid %u [%d, %d].\n", filename, areaid, cell->ix, cell->iy);
593 }
594 area_flags[i][j] = 0xffff;
595 }
596 }
597 //============================================
598 // Try pack area data
599 //============================================
600 bool fullAreaData = false;
601 uint32_t areaflag = area_flags[0][0];
602 for (int y = 0; y < ADT_CELLS_PER_GRID; y++)
603 {
604 for (int x = 0; x < ADT_CELLS_PER_GRID; x++)
605 {
606 if (area_flags[y][x] != areaflag)
607 {
608 fullAreaData = true;
609 break;
610 }
611 }
612 }
613
614 map.areaMapOffset = sizeof(map);
615 map.areaMapSize = sizeof(map_areaHeader);
616
617 map_areaHeader areaHeader;
618 areaHeader.fourcc = *reinterpret_cast<uint32_t const*>(MAP_AREA_MAGIC);
619 areaHeader.flags = 0;
620 if (fullAreaData)
621 {
622 areaHeader.gridArea = 0;
623 map.areaMapSize += sizeof(area_flags);
624 }
625 else
626 {
627 areaHeader.flags |= MAP_AREA_NO_AREA;
628 areaHeader.gridArea = static_cast<uint16_t>(areaflag);
629 }
630
631 //
632 // Get Height map from grid
633 //
634 for (int i = 0; i < ADT_CELLS_PER_GRID; i++)
635 {
636 for (int j = 0; j < ADT_CELLS_PER_GRID; j++)
637 {
638 adt_MCNK* cell = cells->getMCNK(i, j);
639 if (!cell)
640 continue;
641 // Height values for triangles stored in order:
642 // 1 2 3 4 5 6 7 8 9
643 // 10 11 12 13 14 15 16 17
644 // 18 19 20 21 22 23 24 25 26
645 // 27 28 29 30 31 32 33 34
646 // . . . . . . . .
647 // For better get height values merge it to V9 and V8 map
648 // V9 height map:
649 // 1 2 3 4 5 6 7 8 9
650 // 18 19 20 21 22 23 24 25 26
651 // . . . . . . . .
652 // V8 height map:
653 // 10 11 12 13 14 15 16 17
654 // 27 28 29 30 31 32 33 34
655 // . . . . . . . .
656
657 // Set map height as grid height
658 for (int y = 0; y <= ADT_CELL_SIZE; y++)
659 {
660 int cy = i * ADT_CELL_SIZE + y;
661 for (int x = 0; x <= ADT_CELL_SIZE; x++)
662 {
663 int cx = j * ADT_CELL_SIZE + x;
664 V9[cy][cx] = cell->ypos;
665 }
666 }
667 for (int y = 0; y < ADT_CELL_SIZE; y++)
668 {
669 int cy = i * ADT_CELL_SIZE + y;
670 for (int x = 0; x < ADT_CELL_SIZE; x++)
671 {
672 int cx = j * ADT_CELL_SIZE + x;
673 V8[cy][cx] = cell->ypos;
674 }
675 }
676 // Get custom height
677 adt_MCVT* v = cell->getMCVT();
678 if (!v)
679 continue;
680 // get V9 height map
681 for (int y = 0; y <= ADT_CELL_SIZE; y++)
682 {
683 int cy = i * ADT_CELL_SIZE + y;
684 for (int x = 0; x <= ADT_CELL_SIZE; x++)
685 {
686 int cx = j * ADT_CELL_SIZE + x;
687 V9[cy][cx] += v->height_map[y * (ADT_CELL_SIZE * 2 + 1) + x];
688 }
689 }
690 // get V8 height map
