163 void init_cyclic_spline(const Vector3 * controls, index_type count, EvaluationMode m, index_type cyclic_point, float orientation = 0) { SplineBase::init_cyclic_spline(controls, count, m, cyclic_point, orientation);}
164
165 // Initializes lengths with SplineBase::SegLength method.
166 void initLengths();
167
168 // Initializes lengths in some custom way
169 // Note that value returned by cacher must be greater or equal to previous value.
170 template<class T> inline void initLengths(T& cacher)
171 {
172 index_type i = index_lo;
173 lengths.resize(index_hi+1);
174 length_type prev_length = 0, new_length = 0;
175 while (i < index_hi)
176 {
177 new_length = cacher(*this, i);
178 // length overflowed, assign to max positive value
179 if (new_length < 0)
180 new_length = std::numeric_limits<length_type>::max();
181 lengths[++i] = new_length;
182
183 ASSERT(prev_length <= new_length);
184 prev_length = new_length;
185 }
186 }
187
188 // Returns length of the whole spline.
189 length_type length() const
190 {
191 if (lengths.empty())
192 return 0;
193 return lengths[index_hi];
194 }
195 // Returns length between given nodes.
196 length_type length(index_type first, index_type last) const { return lengths[last]-lengths[first];}
197 length_type length(index_type Idx) const { return lengths[Idx];}
198
199 void set_length(index_type i, length_type length) { lengths[i] = length;}
200 void clear();
201};
