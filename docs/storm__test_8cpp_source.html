56static void CalculateMD5(LPBYTE md5_digest, LPBYTE pbData, DWORD cbData)
57{
58 HCRYPTPROV hCryptProv = NULL;
59 HCRYPTHASH hCryptHash = NULL;
60
61 if(CryptAcquireContext(&hCryptProv, NULL, NULL, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT))
62 {
63 if(CryptCreateHash(hCryptProv, CALG_MD5, NULL, 0, &hCryptHash))
64 {
65 DWORD dwHashLen = 0x10;
66
67 CryptHashData(hCryptHash, pbData, cbData, 0);
68 CryptGetHashParam(hCryptHash, HP_HASHVAL, md5_digest, &dwHashLen, 0);
69 CryptDestroyHash(hCryptHash);
70 }
71
72 CryptReleaseContext(hCryptProv, 0);
73 }
74}

76template <typename XCHAR>
77DWORD BinaryToString(XCHAR * szBuffer, size_t cchBuffer, LPCVOID pvBinary, size_t cbBinary)
78{
79 LPCBYTE pbBinary = (LPCBYTE)pvBinary;
80
81 // The size of the string must be enough to hold the binary + EOS
82 if(cchBuffer < ((cbBinary * 2) + 1))
83 return ERROR_INSUFFICIENT_BUFFER;
84
85 // Convert the string to the array of MD5
86 // Copy the blob data as text
87 for(size_t i = 0; i < cbBinary; i++)
88 {
89 *szBuffer++ = IntToHexChar[pbBinary[0] >> 0x04];
90 *szBuffer++ = IntToHexChar[pbBinary[0] & 0x0F];
91 pbBinary++;
92 }
93
94 // Terminate the string
95 *szBuffer = 0;
96 return ERROR_SUCCESS;
97}
