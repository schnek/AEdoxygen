30bool checkDirectories(bool debugOutput)
31{
32 std::vector<std::string> dirFiles;
33
34 if (getDirContents(dirFiles, "maps") == LISTFILE_DIRECTORY_NOT_FOUND || dirFiles.empty())
35 {
36 printf("'maps' directory is empty or does not exist\n");
37 return false;
38 }
39
40 dirFiles.clear();
41 if (getDirContents(dirFiles, "vmaps", "*.vmtree") == LISTFILE_DIRECTORY_NOT_FOUND || dirFiles.empty())
42 {
43 printf("'vmaps' directory is empty or does not exist\n");
44 return false;
45 }
46
47 dirFiles.clear();
48 if (getDirContents(dirFiles, "mmaps") == LISTFILE_DIRECTORY_NOT_FOUND)
49 {
50 return fs::create_directory("mmaps");
51 }
52
53 dirFiles.clear();
54 if (debugOutput)
55 {
56 if (getDirContents(dirFiles, "meshes") == LISTFILE_DIRECTORY_NOT_FOUND)
57 {
58 printf("'meshes' directory does not exist (no place to put debugOutput files)\n");
59 return false;
60 }
61 }
62
63 return true;
64}

247int main(int argc, char** argv)
248{
249 unsigned int threads = std::thread::hardware_concurrency();
250 int mapnum = -1;
251 float maxAngle = 70.0f;
252 int tileX = -1, tileY = -1;
253 bool skipLiquid = false,
254 skipContinents = false,
255 skipJunkMaps = true,
256 skipBattlegrounds = false,
257 debugOutput = false,
258 silent = false,
259 bigBaseUnit = false;
260 char* offMeshInputPath = NULL;
261 char* file = NULL;
262
263 bool validParam = handleArgs(argc, argv, mapnum,
264 tileX, tileY, maxAngle,
265 skipLiquid, skipContinents, skipJunkMaps, skipBattlegrounds,
266 debugOutput, silent, bigBaseUnit, offMeshInputPath, file, threads);
267
268 if (!validParam)
269 return silent ? -1 : finish("You have specified invalid parameters", -1);
270
271 if (mapnum == -1 && debugOutput)
272 {
273 if (silent)
274 return -2;
275
276 printf("You have specifed debug output, but didn't specify a map to generate.\n");
277 printf("This will generate debug output for ALL maps.\n");
278 printf("Are you sure you want to continue? (y/n) ");
279 if (getchar() != 'y')
280 return 0;
281 }
282
283 if (!checkDirectories(debugOutput))
284 return silent ? -3 : finish("Press ENTER to close...", -3);
285
286 MapBuilder builder(maxAngle, skipLiquid, skipContinents, skipJunkMaps,
287 skipBattlegrounds, debugOutput, bigBaseUnit, offMeshInputPath);
288
289 auto startTime = Util::TimeNow();
290 if (file)
291 builder.buildMeshFromFile(file);
292 else if (tileX > -1 && tileY > -1 && mapnum >= 0)
293 builder.buildSingleTile(mapnum, tileX, tileY);
294 else if (mapnum >= 0)
295 builder.buildMap(uint32_t(mapnum));
296 else
297 builder.buildAllMaps(threads);
298
299 if (!silent)
300 printf("Finished. MMAPS were built in %I64d ms!\n", Util::GetTimeDifferenceToNow(startTime));
301 return 0;
302}
