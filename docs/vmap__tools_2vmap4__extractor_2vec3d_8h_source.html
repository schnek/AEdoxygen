242inline void rotate(float x0, float y0, float *x, float *y, float angle)
243{
244 float xa = *x - x0, ya = *y - y0;
245 *x = xa*cosf(angle) - ya*sinf(angle) + x0;
246 *y = xa*sinf(angle) + ya*cosf(angle) + y0;
247}
