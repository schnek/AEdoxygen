245int WMOGroup::ConvertToVMAPGroupWmo(FILE *output, WMORoot *rootWMO, bool preciseVectorData)
246{
247 fwrite(&mogpFlags,sizeof(uint32_t),1,output);
248 fwrite(&groupWMOID,sizeof(uint32_t),1,output);
249 // group bound
250 fwrite(bbcorn1, sizeof(float), 3, output);
251 fwrite(bbcorn2, sizeof(float), 3, output);
252 fwrite(&liquflags,sizeof(uint32_t),1,output);
253 int nColTriangles = 0;
254 if (preciseVectorData)
255 {
256 char GRP[] = "GRP ";
257 fwrite(GRP,1,4,output);
258
259 int k = 0;
260 int moba_batch = moba_size/12;
261 MobaEx = new int[moba_batch*4];
262 for(int i=8; i<moba_size; i+=12)
263 {
264 MobaEx[k++] = MOBA[i];
265 }
266 int moba_size_grp = moba_batch*4+4;
267 fwrite(&moba_size_grp,4,1,output);
268 fwrite(&moba_batch,4,1,output);
269 fwrite(MobaEx,4,k,output);
270 delete [] MobaEx;
271
272 uint32_t nIdexes = nTriangles * 3;
273
274 if(fwrite("INDX",4, 1, output) != 1)
275 {
276 printf("Error while writing file nbraches ID");
277 exit(0);
278 }
279 int wsize = sizeof(uint32_t) + sizeof(unsigned short) * nIdexes;
280 if(fwrite(&wsize, sizeof(int), 1, output) != 1)
281 {
282 printf("Error while writing file wsize");
283 // no need to exit?
284 }
285 if(fwrite(&nIdexes, sizeof(uint32_t), 1, output) != 1)
286 {
287 printf("Error while writing file nIndexes");
288 exit(0);
289 }
290 if(nIdexes >0)
291 {
292 if(fwrite(MOVI, sizeof(unsigned short), nIdexes, output) != nIdexes)
293 {
294 printf("Error while writing file indexarray");
295 exit(0);
296 }
297 }
298
299 if(fwrite("VERT",4, 1, output) != 1)
300 {
301 printf("Error while writing file nbraches ID");
302 exit(0);
303 }
304 wsize = sizeof(int) + sizeof(float) * 3 * nVertices;
305 if(fwrite(&wsize, sizeof(int), 1, output) != 1)
306 {
307 printf("Error while writing file wsize");
308 // no need to exit?
309 }
310 if(fwrite(&nVertices, sizeof(int), 1, output) != 1)
311 {
312 printf("Error while writing file nVertices");
313 exit(0);
314 }
315 if(nVertices >0)
316 {
317 if(fwrite(MOVT, sizeof(float)*3, nVertices, output) != nVertices)
318 {
319 printf("Error while writing file vectors");
320 exit(0);
321 }
322 }
323
324 nColTriangles = nTriangles;
325 }
326 else
327 {
328 char GRP[] = "GRP ";
329 fwrite(GRP,1,4,output);
330 int k = 0;
331 int moba_batch = moba_size/12;
332 MobaEx = new int[moba_batch*4];
333 for(int i=8; i<moba_size; i+=12)
334 {
335 MobaEx[k++] = MOBA[i];
336 }
337
338 int moba_size_grp = moba_batch*4+4;
339 fwrite(&moba_size_grp,4,1,output);
340 fwrite(&moba_batch,4,1,output);
341 fwrite(MobaEx,4,k,output);
342 delete [] MobaEx;
343
344 //-------INDX------------------------------------
345 //-------MOPY--------
346 MoviEx = new uint16_t[nTriangles*3]; // "worst case" size...
347 int *IndexRenum = new int[nVertices];
348 memset(IndexRenum, 0xFF, nVertices*sizeof(int));
349 for (int i=0; i<nTriangles; ++i)
350 {
351 // Skip no collision triangles
352 if (MOPY[2*i]&WMO_MATERIAL_NO_COLLISION ||
353 !(MOPY[2*i]&(WMO_MATERIAL_HINT|WMO_MATERIAL_COLLIDE_HIT)) )
354 continue;
355 // Use this triangle
356 for (int j=0; j<3; ++j)
357 {
358 IndexRenum[MOVI[3*i + j]] = 1;
359 MoviEx[3*nColTriangles + j] = MOVI[3*i + j];
360 }
361 ++nColTriangles;
362 }
363
364 // assign new vertex index numbers
365 int nColVertices = 0;
366 for (uint32_t i=0; i<nVertices; ++i)
367 {
368 if (IndexRenum[i] == 1)
369 {
370 IndexRenum[i] = nColVertices;
371 ++nColVertices;
372 }
373 }
374
375 // translate triangle indices to new numbers
376 for (int i=0; i<3*nColTriangles; ++i)
377 {
378 assert(MoviEx[i] < nVertices);
379 MoviEx[i] = IndexRenum[MoviEx[i]];
380 }
381
382 // write triangle indices
383 int INDX[] = {0x58444E49, nColTriangles*6+4, nColTriangles*3};
384 fwrite(INDX,4,3,output);
385 fwrite(MoviEx,2,nColTriangles*3,output);
386
387 // write vertices
388 int VERT[] = {0x54524556, nColVertices*3*static_cast<int>(sizeof(float))+4, nColVertices};// "VERT"
389 int check = 3*nColVertices;
390 fwrite(VERT,4,3,output);
391 for (uint32_t i=0; i<nVertices; ++i)
392 if(IndexRenum[i] >= 0)
393 check -= static_cast<int>(fwrite(MOVT+3*i, sizeof(float), 3, output));
394
395 assert(check==0);
396
397 delete [] MoviEx;
398 delete [] IndexRenum;
399 }
400
401 //------LIQU------------------------
402 if (LiquEx_size != 0)
403 {
404 int LIQU_h[] = {0x5551494C, static_cast<int>(sizeof(WMOLiquidHeader) + LiquEx_size) + hlq->xtiles*hlq->ytiles};// "LIQU"
405 fwrite(LIQU_h, 4, 2, output);
406
407 // according to WoW.Dev Wiki:
408 uint32_t liquidEntry;
409 if (rootWMO->liquidType & 4)
410 liquidEntry = liquidType;
411 else if (liquidType == 15)
412 liquidEntry = 0;
413 else
414 liquidEntry = liquidType + 1;
415
416 if (!liquidEntry)
417 {
418 int v1; // edx@1
419 int v2; // eax@1
420
421 v1 = hlq->xtiles * hlq->ytiles;
422 v2 = 0;
423 if (v1 > 0)
424 {
425 while ((LiquBytes[v2] & 0xF) == 15)
426 {
427 ++v2;
428 if (v2 >= v1)
429 break;
430 }
431
432 if (v2 < v1 && (LiquBytes[v2] & 0xF) != 15)
433 liquidEntry = (LiquBytes[v2] & 0xF) + 1;
434 }
435 }
436
437 if (liquidEntry && liquidEntry < 21)
438 {
439 switch ((liquidEntry - 1) & 3)
440 {
441 case 0:
442 liquidEntry = ((mogpFlags & 0x80000) != 0) + 13;
443 break;
444 case 1:
445 liquidEntry = 14;
446 break;
447 case 2:
448 liquidEntry = 19;
449 break;
450 case 3:
451 liquidEntry = 20;
452 break;
453 }
454 }
455
456 hlq->type = liquidEntry;
457
458 /* std::ofstream llog("Buildings/liquid.log", ios_base::out | ios_base::app);
459 llog << filename;
460 llog << ":\nliquidEntry: " << liquidEntry << " type: " << hlq->type << " (root:" << rootWMO->liquidType << " group:" << liquidType << ")\n";
461 llog.close(); */
462
463 fwrite(hlq, sizeof(WMOLiquidHeader), 1, output);
464 // only need height values, the other values are unknown anyway
465 for (uint32_t i = 0; i<LiquEx_size/sizeof(WMOLiquidVert); ++i)
466 fwrite(&LiquEx[i].height, sizeof(float), 1, output);
467 // todo: compress to bit field
468 fwrite(LiquBytes, 1, hlq->xtiles*hlq->ytiles, output);
469 }
470
471 return nColTriangles;
472}
