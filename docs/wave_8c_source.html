58/* this function decompress a wave file, mono or stereo, 1500F230 offset. */
59int32_t libmpq__do_decompress_wave(uint8_t *out_buf, int32_t out_length, uint8_t *in_buf, int32_t in_length, int32_t channels) {
60
61 /* some common variables. */
62 byte_and_int16_t out;
63 byte_and_int16_t in;
64 uint32_t index;
65 int32_t nr_array1[2];
66 int32_t nr_array2[2];
67 int32_t count = 0;
68
69 /* end on input buffer. */
70 uint8_t *in_end = in_buf + in_length;
71
72 /* assign default values. */
73 out.pb = out_buf;
74 in.pb = in_buf;
75 nr_array1[0] = 0x2C;
76 nr_array1[1] = 0x2C;
77
78 /* increase. */
79 in.pw++;
80
81 /* 15007AD7 */
82 for (count = 0; count < channels; count++) {
83
84 /* some common variables. */
85 int32_t temp;
86
87 /* save pointer. */
88 temp = *(int16_t *)in.pw++;
89 nr_array2[count] = temp;
90
91 /* check if should break. */
92 if (out_length < 2) {
93 return out.pb - out_buf;
94 }
95
96 /* return values. */
97 *out.pw++ = (uint16_t)temp;
98 out_length -= 2;
99 }
100
101 /* decrease channels. */
102 index = channels - 1;
103
104 /* loop through input buffer until end reached. */
105 while (in.pb < in_end) {
106
107 /* save the byte. */
108 uint8_t one_byte = *in.pb++;
109
110 /* check how many channels and set index. */
111 if (channels == 2) {
112 index = (index == 0) ? 1 : 0;
113 }
114
115 /* 15007B25 - get one byte from input buffer. */
116 if (one_byte & 0x80) {
117
118 /* 15007B32 */
119 switch (one_byte & 0x7F) {
120 case 0:
121
122 /* 15007B8E */
123 if (nr_array1[index] != 0) {
124 nr_array1[index]--;
125 }
126
127 /* check if should break. */
128 if (out_length < 2) {
129 break;
130 }
131
132 /* return values. */
133 *out.pw++ = (uint16_t)nr_array2[index];
134 out_length -= 2;
135
136 /* continue loop. */
137 continue;
138 case 1:
139 /* 15007B72 and EBX. */
140 nr_array1[index] += 8;
141
142 /* check index. */
143 if (nr_array1[index] > 0x58) {
144 nr_array1[index] = 0x58;
145 }
146
147 /* check how many channels and set index. */
148 if (channels == 2) {
149 index = (index == 0) ? 1 : 0;
150 }
151
152 /* continue loop. */
153 continue;
154 case 2:
155
156 /* nothing todo, so continue. */
157 continue;
158 default:
159
160 /* decrease index. */
161 nr_array1[index] -= 8;
162
163 /* check index. */
164 if (nr_array1[index] < 0) {
165 nr_array1[index] = 0;
166 }
167
168 /* check if two channels left. */
169 if (channels != 2) {
170 continue;
171 }
172 index = (index == 0) ? 1 : 0;
173
174 /* continue loop. */
175 continue;
176 }
177 } else {
178
179 /* EDI */
180 uint32_t temp1 = wave_table_1503f1a0[nr_array1[index]];
181
182 /* ESI */
183 uint32_t temp2 = temp1 >> in_buf[1];
184
185 /* ECX */
186 int32_t temp3 = nr_array2[index];
187
188 /* EBX = one byte. */
189 if (one_byte & 0x01) {
190 temp2 += (temp1 >> 0);
191 }
192 if (one_byte & 0x02) {
193 temp2 += (temp1 >> 1);
194 }
195 if (one_byte & 0x04) {
196 temp2 += (temp1 >> 2);
197 }
198 if (one_byte & 0x08) {
199 temp2 += (temp1 >> 3);
200 }
201 if (one_byte & 0x10) {
202 temp2 += (temp1 >> 4);
203 }
204 if (one_byte & 0x20) {
205 temp2 += (temp1 >> 5);
206 }
207 if (one_byte & 0x40) {
208 temp3 -= temp2;
209 if (temp3 <= (int32_t)0xFFFF8000) {
210 temp3 = (int32_t)0xFFFF8000;
211 }
212 } else {
213 temp3 += temp2;
214 if (temp3 >= 0x7FFF) {
215 temp3 = 0x7FFF;
216 }
217 }
218
219 /* restore index. */
220 nr_array2[index] = temp3;
221
222 /* check if should break. */
223 if (out_length < 2) {
224 break;
225 }
226
227 /* assign values. */
228 temp2 = nr_array1[index];
229 one_byte &= 0x1F;
230 *out.pw++ = (uint16_t)temp3;
231 out_length -= 2;
232 temp2 += wave_table_1503f120[one_byte];
233 nr_array1[index] = temp2;
234
235 /* check index. */
236 if (nr_array1[index] < 0) {
237 nr_array1[index] = 0;
238 } else {
239
240 /* check index. */
241 if (nr_array1[index] > 0x58) {
242 nr_array1[index] = 0x58;
243 }
244 }
245 }
246 }
247
248 /* return copied bytes. */
249 return (out.pb - out_buf);
250}
