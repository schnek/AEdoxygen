20 _polyLength(0), _type(PATHFIND_BLANK), _useStraightPath(false),
21 _forceDestination(false), _pointPathLimit(MAX_POINT_PATH_LENGTH), _useRaycast(false),
22 _endPosition(G3D::Vector3::zero()), _source(owner), _navMesh(nullptr),

33 }
34
35 createFilter();

38bool PathGenerator::calculatePath(float destX, float destY, float destZ, bool forceDest)
39{
40 float x, y, z;
41 _source->getPosition(x, y, z);
42
43 G3D::Vector3 dest(destX, destY, destZ);
44 setEndPosition(dest);
45
46 G3D::Vector3 start(x, y, z);
47 setStartPosition(start);
48
49 _forceDestination = forceDest;
50
51 // make sure navMesh works - we can run on map w/o mmap
52 // check if the start and end point have a .mmtile loaded (can we pass via not loaded tile on the way?)
53 Unit* _sourceUnit = _source->ToUnit();
54 if (!_navMesh || !_navMeshQuery || (_sourceUnit && _sourceUnit->hasUnitStateFlag(UNIT_STATE_IGNORE_PATHFINDING)) ||
55 !haveTile(start) || !haveTile(dest))
56 {
57 buildShortcut();
58 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
59 return true;
60 }
61
62 updateFilter();
63
64 buildPolyPath(start, dest);
65 return true;
66}

68dtPolyRef PathGenerator::getPathPolyByPosition(dtPolyRef const* polyPath, uint32_t polyPathSize, float const* point, float* distance) const
69{
70 if (!polyPath || !polyPathSize)
71 return INVALID_POLYREF;
72
73 dtPolyRef nearestPoly = INVALID_POLYREF;
74 float minDist = FLT_MAX;
75
76 for (uint32_t i = 0; i < polyPathSize; ++i)
77 {
78 float closestPoint[VERTEX_SIZE];
79 if (dtStatusFailed(_navMeshQuery->closestPointOnPoly(polyPath[i], point, closestPoint, nullptr)))
80 continue;
81
82 float d = dtVdistSqr(point, closestPoint);
83 if (d < minDist)
84 {
85 minDist = d;
86 nearestPoly = polyPath[i];
87 }
88
89 if (minDist < 1.0f) // shortcut out - close enough for us
90 break;
91 }
92
93 if (distance)
94 *distance = dtMathSqrtf(minDist);
95
96 return (minDist < 3.0f) ? nearestPoly : INVALID_POLYREF;
97}

99dtPolyRef PathGenerator::getPolyByLocation(float const* point, float* distance) const
100{
101 // first we check the current path
102 // if the current path doesn't contain the current poly,
103 // we need to use the expensive navMesh.findNearestPoly
104 dtPolyRef polyRef = getPathPolyByPosition(_pathPolyRefs, _polyLength, point, distance);
105 if (polyRef != INVALID_POLYREF)
106 return polyRef;
107
108 // we don't have it in our old path
109 // try to get it by findNearestPoly()
110 // first try with low search box
111 float extents[VERTEX_SIZE] = {3.0f, 5.0f, 3.0f}; // bounds of poly search area
112 float closestPoint[VERTEX_SIZE] = {0.0f, 0.0f, 0.0f};
113 if (dtStatusSucceed(_navMeshQuery->findNearestPoly(point, extents, &_filter, &polyRef, closestPoint)) && polyRef != INVALID_POLYREF)
114 {
115 *distance = dtVdist(closestPoint, point);
116 return polyRef;
117 }
118
119 // still nothing ..
120 // try with bigger search box
121 // Note that the extent should not overlap more than 128 polygons in the navmesh (see dtNavMeshQuery::findNearestPoly)
122 extents[1] = 50.0f;
123
124 if (dtStatusSucceed(_navMeshQuery->findNearestPoly(point, extents, &_filter, &polyRef, closestPoint)) && polyRef != INVALID_POLYREF)
125 {
126 *distance = dtVdist(closestPoint, point);
127 return polyRef;
128 }
129
130 *distance = FLT_MAX;
131 return INVALID_POLYREF;
132}

134void PathGenerator::buildPolyPath(G3D::Vector3 const& startPos, G3D::Vector3 const& endPos)
135{
136 //////////////////////////////////////////////////////////////////////////////////////////
137 // getting start/end poly logic
138 float distToStartPoly, distToEndPoly;
139 float startPoint[VERTEX_SIZE] = {startPos.y, startPos.z, startPos.x};
140 float endPoint[VERTEX_SIZE] = {endPos.y, endPos.z, endPos.x};
141
142 dtPolyRef startPoly = getPolyByLocation(startPoint, &distToStartPoly);
143 dtPolyRef endPoly = getPolyByLocation(endPoint, &distToEndPoly);
144
145 _type = PathType(PATHFIND_NORMAL);
146
147 // we have a hole in our mesh
148 // make shortcut path and mark it as NOPATH ( with flying and swimming exception )
149 // its up to caller how he will use this info
150
151 bool waterPath = false;
152 bool path = false;
153
154 if (_source->isCreature())
155 {
156 waterPath = _source->ToCreature()->canSwim();
157 path = _source->ToCreature()->canFly();
158 }
159 else
160 {
161 waterPath = _source->ToPlayer()->canSwim();
162 }
163
164 if (startPoly == INVALID_POLYREF || endPoly == INVALID_POLYREF || waterPath && _source->ToUnit()->isInWater() || path && _source->ToUnit()->IsFlying())
165 {
166 buildShortcut();
167
168 if (waterPath)
169 {
170 // Check both start and end points, if they're both in water, then we can *safely* let the creature move
171 for (uint32_t i = 0; i < _pathPoints.size(); ++i)
172 {
173 ZLiquidStatus liquidStatus = _source->getWorldMap()->getLiquidStatus(_source->GetPhase(), LocationVector(_pathPoints[i].x, _pathPoints[i].y, _pathPoints[i].z), MAP_ALL_LIQUIDS, nullptr, _source->getCollisionHeight());
174
175 // One of the points is not in the water, cancel movement.
176 if (liquidStatus == LIQUID_MAP_NO_WATER)
177 {
178 waterPath = false;
179 break;
180 }
181 }
182 }
183
184 if (path || waterPath)
185 {
186 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
187 return;
188 }
189
190 // raycast doesn't need endPoly to be valid
191 if (!_useRaycast)
192 {
193 _type = PATHFIND_NOPATH;
194 return;
195 }
196 }
197
198 // we may need a better number here
199 bool startFarFromPoly = distToStartPoly > 7.0f;
200 bool endFarFromPoly = distToEndPoly > 7.0f;
201 if (startFarFromPoly || endFarFromPoly)
202 {
203 bool buildShotrcut = false;
204
205 G3D::Vector3 const& p = (distToStartPoly > 7.0f) ? startPos : endPos;
206 if (_source->getWorldMap()->isUnderWater(_source->GetPhase(), LocationVector(p.x, p.y, p.z)))
207 {
208 if (Unit* _sourceUnit = _source->ToUnit())
209 if (_sourceUnit->canSwim())
210 buildShotrcut = true;
211 }
212 else
213 {
214 if (Unit* _sourceUnit = _source->ToUnit())
215 {
216 if (_sourceUnit->canFly())
217 buildShotrcut = true;
218 // Allow to build a shortcut if the unit is falling and it's trying to move downwards towards a target (i.e. charging)
219 else if (_sourceUnit->IsFalling() && endPos.z < startPos.z)
220 buildShotrcut = true;
221 }
222 }
223
224 if (buildShotrcut)
225 {
226 buildShortcut();
227 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
228
229 addFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
230
231 return;
232 }
233 else
234 {
235 float closestPoint[VERTEX_SIZE];
236 // we may want to use closestPointOnPolyBoundary instead
237 if (dtStatusSucceed(_navMeshQuery->closestPointOnPoly(endPoly, endPoint, closestPoint, nullptr)))
238 {
239 dtVcopy(endPoint, closestPoint);
240 setActualEndPosition(G3D::Vector3(endPoint[2], endPoint[0], endPoint[1]));
241 }
242
243 _type = PathType(PATHFIND_INCOMPLETE);
244
245 addFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
246 }
247 }
248
249 // *** poly path generating logic ***
250
251 // start and end are on same polygon
252 // handle this case as if they were 2 different polygons, building a line path split in some few points
253 if (startPoly == endPoly && !_useRaycast)
254 {
255 _pathPolyRefs[0] = startPoly;
256 _polyLength = 1;
257
258 if (startFarFromPoly || endFarFromPoly)
259 {
260 _type = PathType(PATHFIND_INCOMPLETE);
261
262 addFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
263 }
264 else
265 _type = PATHFIND_NORMAL;
266
267 buildPointPath(startPoint, endPoint);
268 return;
269 }
270
271 // look for startPoly/endPoly in current path
272 /// @todo we can merge it with getPathPolyByPosition() loop
273 bool startPolyFound = false;
274 bool endPolyFound = false;
275 uint32_t pathStartIndex = 0;
276 uint32_t pathEndIndex = 0;
277
278 if (_polyLength)
279 {
280 for (; pathStartIndex < _polyLength; ++pathStartIndex)
281 {
282 // here to catch few bugs
283 if (_pathPolyRefs[pathStartIndex] == INVALID_POLYREF)
284 {
285 break;
286 }
287
288 if (_pathPolyRefs[pathStartIndex] == startPoly)
289 {
290 startPolyFound = true;
291 break;
292 }
293 }
294
295 for (pathEndIndex = _polyLength-1; pathEndIndex > pathStartIndex; --pathEndIndex)
296 if (_pathPolyRefs[pathEndIndex] == endPoly)
297 {
298 endPolyFound = true;
299 break;
300 }
301 }
302
303 if (startPolyFound && endPolyFound)
304 {
305 // we moved along the path and the target did not move out of our old poly-path
306 // our path is a simple subpath case, we have all the data we need
307 // just "cut" it out
308
309 _polyLength = pathEndIndex - pathStartIndex + 1;
310 memmove(_pathPolyRefs, _pathPolyRefs + pathStartIndex, _polyLength * sizeof(dtPolyRef));
311 }
312 else if (startPolyFound && !endPolyFound)
313 {
314 // we are moving on the old path but target moved out
315 // so we have atleast part of poly-path ready
316
317 _polyLength -= pathStartIndex;
318
319 // try to adjust the suffix of the path instead of recalculating entire length
320 // at given interval the target cannot get too far from its last location
321 // thus we have less poly to cover
322 // sub-path of optimal path is optimal
323
324 // take ~80% of the original length
325 /// @todo play with the values here
326 uint32_t prefixPolyLength = uint32_t(_polyLength * 0.8f + 0.5f);
327 memmove(_pathPolyRefs, _pathPolyRefs+pathStartIndex, prefixPolyLength * sizeof(dtPolyRef));
328
329 dtPolyRef suffixStartPoly = _pathPolyRefs[prefixPolyLength-1];
330
331 // we need any point on our suffix start poly to generate poly-path, so we need last poly in prefix data
332 float suffixEndPoint[VERTEX_SIZE];
333 if (dtStatusFailed(_navMeshQuery->closestPointOnPoly(suffixStartPoly, endPoint, suffixEndPoint, nullptr)))
334 {
335 // we can hit offmesh connection as last poly - closestPointOnPoly() don't like that
336 // try to recover by using prev polyref
337 --prefixPolyLength;
338 suffixStartPoly = _pathPolyRefs[prefixPolyLength-1];
339 if (dtStatusFailed(_navMeshQuery->closestPointOnPoly(suffixStartPoly, endPoint, suffixEndPoint, nullptr)))
340 {
341 // suffixStartPoly is still invalid, error state
342 buildShortcut();
343 _type = PATHFIND_NOPATH;
344 return;
345 }
346 }
347
348 // generate suffix
349 uint32_t suffixPolyLength = 0;
350
351 dtStatus dtResult;
352 if (_useRaycast)
353 {
354 buildShortcut();
355 _type = PATHFIND_NOPATH;
356 return;
357 }
358 else
359 {
360 dtResult = _navMeshQuery->findPath(
361 suffixStartPoly, // start polygon
362 endPoly, // end polygon
363 suffixEndPoint, // start position
364 endPoint, // end position
365 &_filter, // polygon search filter
366 _pathPolyRefs + prefixPolyLength - 1, // [out] path
367 (int*)&suffixPolyLength,
368 MAX_POINT_PATH_LENGTH - prefixPolyLength); // max number of polygons in output path
369 }
370
371 if (!suffixPolyLength || dtStatusFailed(dtResult))
372 {
373 // this is probably an error state, but we'll leave it
374 // and hopefully recover on the next Update
375 // we still need to copy our preffix
376 }
377
378 // new path = prefix + suffix - overlap
379 _polyLength = prefixPolyLength + suffixPolyLength - 1;
380 }
381 else
382 {
383 // either we have no path at all -> first run
384 // or something went really wrong -> we aren't moving along the path to the target
385 // just generate new path
386
387 // free and invalidate old path data
388 clear();
389
390 dtStatus dtResult;
391 if (_useRaycast)
392 {
393 float hit = 0;
394 float hitNormal[3];
395 memset(hitNormal, 0, sizeof(hitNormal));
396
397 dtResult = _navMeshQuery->raycast(
398 startPoly,
399 startPoint,
400 endPoint,
401 &_filter,
402 &hit,
403 hitNormal,
404 _pathPolyRefs,
405 (int*)&_polyLength,
406 MAX_POINT_PATH_LENGTH);
407
408 if (!_polyLength || dtStatusFailed(dtResult))
409 {
410 buildShortcut();
411 _type = PATHFIND_NOPATH;
412 addFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
413 return;
414 }
415
416 // raycast() sets hit to FLT_MAX if there is a ray between start and end
417 if (hit != FLT_MAX)
418 {
419 float hitPos[3];
420
421 // Walk back a bit from the hit point to make sure it's in the mesh (sometimes the point is actually outside of the polygons due to float precision issues)
422 hit *= 0.99f;
423 dtVlerp(hitPos, startPoint, endPoint, hit);
424
425 // if it fails again, clamp to poly boundary
426 if (dtStatusFailed(_navMeshQuery->getPolyHeight(_pathPolyRefs[_polyLength - 1], hitPos, &hitPos[1])))
427 _navMeshQuery->closestPointOnPolyBoundary(_pathPolyRefs[_polyLength - 1], hitPos, hitPos);
428
429 _pathPoints.resize(2);
430 _pathPoints[0] = getStartPosition();
431 _pathPoints[1] = G3D::Vector3(hitPos[2], hitPos[0], hitPos[1]);
432
433 normalizePath();
434 _type = PATHFIND_INCOMPLETE;
435 addFarFromPolyFlags(startFarFromPoly, false);
436 return;
437 }
438 else
439 {
440 // clamp to poly boundary if we fail to get the height
441 if (dtStatusFailed(_navMeshQuery->getPolyHeight(_pathPolyRefs[_polyLength - 1], endPoint, &endPoint[1])))
442 _navMeshQuery->closestPointOnPolyBoundary(_pathPolyRefs[_polyLength - 1], endPoint, endPoint);
443
444 _pathPoints.resize(2);
445 _pathPoints[0] = getStartPosition();
446 _pathPoints[1] = G3D::Vector3(endPoint[2], endPoint[0], endPoint[1]);
447
448 normalizePath();
449 if (startFarFromPoly || endFarFromPoly)
450 {
451 _type = PathType(PATHFIND_INCOMPLETE);
452
453 addFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
454 }
455 else
456 _type = PATHFIND_NORMAL;
457 return;
458 }
459 }
460 else
461 {
462 dtResult = _navMeshQuery->findPath(
463 startPoly, // start polygon
464 endPoly, // end polygon
465 startPoint, // start position
466 endPoint, // end position
467 &_filter, // polygon search filter
468 _pathPolyRefs, // [out] path
469 (int*)&_polyLength,
470 MAX_POINT_PATH_LENGTH); // max number of polygons in output path
471 }
472
473 if (!_polyLength || dtStatusFailed(dtResult))
474 {
475 // only happens if we passed bad data to findPath(), or navmesh is messed up
476 buildShortcut();
477 _type = PATHFIND_NOPATH;
478 return;
479 }
480 }
481
482 // by now we know what type of path we can get
483 if (_pathPolyRefs[_polyLength - 1] == endPoly && !(_type & PATHFIND_INCOMPLETE))
484 _type = PATHFIND_NORMAL;
485 else
486 _type = PATHFIND_INCOMPLETE;
487
488 addFarFromPolyFlags(startFarFromPoly, endFarFromPoly);
489
490 // generate the point-path out of our up-to-date poly-path
491 buildPointPath(startPoint, endPoint);
492}

494void PathGenerator::buildPointPath(const float *startPoint, const float *endPoint)
495{
496 float pathPoints[MAX_POINT_PATH_LENGTH*VERTEX_SIZE];
497 uint32_t pointCount = 0;
498 dtStatus dtResult = DT_FAILURE;
499 if (_useRaycast)
500 {
501 // _straightLine uses raycast and it currently doesn't support building a point path, only a 2-point path with start and hitpoint/end is returned
502 buildShortcut();
503 _type = PATHFIND_NOPATH;
504 return;
505 }
506 else if (_useStraightPath)
507 {
508 dtResult = _navMeshQuery->findStraightPath(
509 startPoint, // start position
510 endPoint, // end position
511 _pathPolyRefs, // current path
512 _polyLength, // lenth of current path
513 pathPoints, // [out] path corner points
514 nullptr, // [out] flags
515 nullptr, // [out] shortened path
516 (int*)&pointCount,
517 _pointPathLimit); // maximum number of points/polygons to use
518 }
519 else
520 {
521 dtResult = findSmoothPath(
522 startPoint, // start position
523 endPoint, // end position
524 _pathPolyRefs, // current path
525 _polyLength, // length of current path
526 pathPoints, // [out] path corner points
527 (int*)&pointCount,
528 _pointPathLimit); // maximum number of points
529 }
530
531 // Special case with start and end positions very close to each other
532 if (_polyLength == 1 && pointCount == 1)
533 {
534 // First point is start position, append end position
535 dtVcopy(&pathPoints[1 * VERTEX_SIZE], endPoint);
536 pointCount++;
537 }
538 else if (pointCount < 2 || dtStatusFailed(dtResult))
539 {
540 // only happens if pass bad data to findStraightPath or navmesh is broken
541 // single point paths can be generated here
542 /// @todo check the exact cases
543 buildShortcut();
544 _type = PathType(_type | PATHFIND_NOPATH);
545 return;
546 }
547 else if (pointCount >= _pointPathLimit)
548 {
549 buildShortcut();
550 _type = PathType(_type | PATHFIND_SHORT);
551 return;
552 }
553
554 _pathPoints.resize(pointCount);
555 for (uint32_t i = 0; i < pointCount; ++i)
556 _pathPoints[i] = G3D::Vector3(pathPoints[i*VERTEX_SIZE+2], pathPoints[i*VERTEX_SIZE], pathPoints[i*VERTEX_SIZE+1]);
557
558 normalizePath();
559
560 // first point is always our current location - we need the next one
561 setActualEndPosition(_pathPoints[pointCount-1]);
562
563 // force the given destination, if needed
564 if (_forceDestination &&
565 (!(_type & PATHFIND_NORMAL) || !inRange(getEndPosition(), getActualEndPosition(), 1.0f, 1.0f)))
566 {
567 // we may want to keep partial subpath
568 if (dist3DSqr(getActualEndPosition(), getEndPosition()) < 0.3f * dist3DSqr(getStartPosition(), getEndPosition()))
569 {
570 setActualEndPosition(getEndPosition());
571 _pathPoints[_pathPoints.size()-1] = getEndPosition();
572 }
573 else
574 {
575 setActualEndPosition(getEndPosition());
576 buildShortcut();
577 }
578
579 _type = PathType(PATHFIND_NORMAL | PATHFIND_NOT_USING_PATH);
580 }
581}

583void PathGenerator::normalizePath()
584{
585 for (uint32_t i = 0; i < _pathPoints.size(); ++i)

589void PathGenerator::buildShortcut()
590{
591 clear();
592
593 // make two point path, our curr pos is the start, and dest is the end
594 _pathPoints.resize(2);
595
596 // set start and a default next position
597 _pathPoints[0] = getStartPosition();
598 _pathPoints[1] = getActualEndPosition();
599
600 normalizePath();
601
602 _type = PATHFIND_SHORTCUT;
603}

605void PathGenerator::createFilter()
606{
607 uint16_t includeFlags = 0;
608 uint16_t excludeFlags = 0;
609
610 if (_source->isCreature())
611 {
612 Creature* creature = (Creature*)_source;
613 if (creature->canWalk())
614 includeFlags |= NAV_GROUND; // walk
615
616 // creatures don't take environmental damage
617 if (creature->canSwim())
618 includeFlags |= (NAV_WATER | NAV_MAGMA_SLIME); // swim
619 }
620 else // assume Player
621 {
622 // perfect support not possible, just stay 'safe'
623 includeFlags |= (NAV_GROUND | NAV_WATER | NAV_MAGMA_SLIME);
624 }
625
626 _filter.setIncludeFlags(includeFlags);
627 _filter.setExcludeFlags(excludeFlags);
628
629 updateFilter();
630}
631
632void PathGenerator::updateFilter()
633{
634 // allow creatures to cheat and use different movement types if they are moved
635 // forcefully into terrain they can't normally move in
636 if (Unit const* _sourceUnit = _source->ToUnit())
637 {
638 if (_sourceUnit->isInWater() || _sourceUnit->isUnderWater())
639 {
640 uint16_t includedFlags = _filter.getIncludeFlags();
641 includedFlags |= getNavTerrain(_source->GetPositionX(),
642 _source->GetPositionY(),
643 _source->GetPositionZ());
644
645 _filter.setIncludeFlags(includedFlags);
646 }
647
648 if (Creature* _sourceCreature = _source->ToCreature())
649 if (_sourceCreature->isInCombat() || _sourceCreature->isInEvadeMode())
650 _filter.setIncludeFlags(_filter.getIncludeFlags() | NAV_GROUND_STEEP);
651 }
652}

654NavTerrainFlag PathGenerator::getNavTerrain(float x, float y, float z)
655{
656 LiquidData data;
657 ZLiquidStatus liquidStatus = _source->getWorldMap()->getLiquidStatus(_source->GetPhase(), LocationVector(x, y, z), MAP_ALL_LIQUIDS, &data, _source->getCollisionHeight());
658 if (liquidStatus == LIQUID_MAP_NO_WATER)
659 return NAV_GROUND;
660
661 switch (data.type_flags)
662 {
663 case MAP_LIQUID_TYPE_WATER:
664 case MAP_LIQUID_TYPE_OCEAN:
665 return NAV_WATER;
666 case MAP_LIQUID_TYPE_MAGMA:
667 case MAP_LIQUID_TYPE_SLIME:
668 return NAV_MAGMA_SLIME;
669 default:
670 return NAV_GROUND;
671 }
672}

674bool PathGenerator::haveTile(const G3D::Vector3& p) const
675{
676 int tx = -1, ty = -1;
677 float point[VERTEX_SIZE] = {p.y, p.z, p.x};
678
679 _navMesh->calcTileLoc(point, &tx, &ty);
680
681 /// Workaround
682 /// For some reason, often the tx and ty variables wont get a valid value
683 /// Use this check to prevent getting negative tile coords and crashing on getTileAt
684 if (tx < 0 || ty < 0)
685 return false;
686
687 return (_navMesh->getTileAt(tx, ty, 0) != nullptr);
688}

690uint32_t PathGenerator::fixupCorridor(dtPolyRef* path, uint32_t npath, uint32_t maxPath, dtPolyRef const* visited, uint32_t nvisited)
691{
692 int32_t furthestPath = -1;
693 int32_t furthestVisited = -1;
694
695 // Find furthest common polygon.
696 for (int32_t i = npath-1; i >= 0; --i)
697 {
698 bool found = false;
699 for (int32_t j = nvisited-1; j >= 0; --j)
700 {
701 if (path[i] == visited[j])
702 {
703 furthestPath = i;
704 furthestVisited = j;
705 found = true;
706 }
707 }
708 if (found)
709 break;
710 }
711
712 // If no intersection found just return current path.
713 if (furthestPath == -1 || furthestVisited == -1)
714 return npath;
715
716 // Concatenate paths.
717
718 // Adjust beginning of the buffer to include the visited.
719 uint32_t req = nvisited - furthestVisited;
720 uint32_t orig = uint32_t(furthestPath + 1) < npath ? furthestPath + 1 : npath;
721 uint32_t size = npath > orig ? npath - orig : 0;
722 if (req + size > maxPath)
723 size = maxPath-req;
724
725 if (size)
726 memmove(path + req, path + orig, size * sizeof(dtPolyRef));
727
728 // Store visited
729 for (uint32_t i = 0; i < req; ++i)
730 path[i] = visited[(nvisited - 1) - i];
731
732 return req+size;

735bool PathGenerator::getSteerTarget(float const* startPos, float const* endPos,
736 float minTargetDist, dtPolyRef const* path, uint32_t pathSize,
737 float* steerPos, unsigned char& steerPosFlag, dtPolyRef& steerPosRef)
738{
739 // Find steer target.
740 static const uint32_t MAX_STEER_POINTS = 3;
741 float steerPath[MAX_STEER_POINTS*VERTEX_SIZE];
742 unsigned char steerPathFlags[MAX_STEER_POINTS];
743 dtPolyRef steerPathPolys[MAX_STEER_POINTS];
744 uint32_t nsteerPath = 0;
745 dtStatus dtResult = _navMeshQuery->findStraightPath(startPos, endPos, path, pathSize,
746 steerPath, steerPathFlags, steerPathPolys, (int*)&nsteerPath, MAX_STEER_POINTS);
747 if (!nsteerPath || dtStatusFailed(dtResult))
748 return false;
749
750 // Find vertex far enough to steer to.
751 uint32_t ns = 0;
752 while (ns < nsteerPath)
753 {
754 // Stop at Off-Mesh link or when point is further than slop away.
755 if ((steerPathFlags[ns] & DT_STRAIGHTPATH_OFFMESH_CONNECTION) ||
756 !inRangeYZX(&steerPath[ns*VERTEX_SIZE], startPos, minTargetDist, 1000.0f))
757 break;
758 ns++;
759 }
760 // Failed to find good point to steer to.
761 if (ns >= nsteerPath)
762 return false;
763
764 dtVcopy(steerPos, &steerPath[ns*VERTEX_SIZE]);
765 steerPos[1] = startPos[1]; // keep Z value
766 steerPosFlag = steerPathFlags[ns];
767 steerPosRef = steerPathPolys[ns];
768
769 return true;
770}

772dtStatus PathGenerator::findSmoothPath(float const* startPos, float const* endPos,
773 dtPolyRef const* polyPath, uint32_t polyPathSize,
774 float* smoothPath, int* smoothPathSize, uint32_t maxSmoothPathSize)
775{
776 *smoothPathSize = 0;
777 uint32_t nsmoothPath = 0;
778
779 dtPolyRef polys[MAX_POINT_PATH_LENGTH];
780 memcpy(polys, polyPath, sizeof(dtPolyRef)*polyPathSize);
781 uint32_t npolys = polyPathSize;
782
783 float iterPos[VERTEX_SIZE], targetPos[VERTEX_SIZE];
784
785 if (polyPathSize > 1)
786 {
787 // Pick the closest points on poly border
788 if (dtStatusFailed(_navMeshQuery->closestPointOnPolyBoundary(polys[0], startPos, iterPos)))
789 return DT_FAILURE;
790
791 if (dtStatusFailed(_navMeshQuery->closestPointOnPolyBoundary(polys[npolys - 1], endPos, targetPos)))
792 return DT_FAILURE;
793 }
794 else
795 {
796 // Case where the path is on the same poly
797 dtVcopy(iterPos, startPos);
798 dtVcopy(targetPos, endPos);
799 }
800
801 dtVcopy(&smoothPath[nsmoothPath*VERTEX_SIZE], iterPos);
802 nsmoothPath++;
803
804 // Move towards target a small advancement at a time until target reached or
805 // when ran out of memory to store the path.
806 while (npolys && nsmoothPath < maxSmoothPathSize)
807 {
808 // Find location to steer towards.
809 float steerPos[VERTEX_SIZE];
810 unsigned char steerPosFlag;
811 dtPolyRef steerPosRef = INVALID_POLYREF;
812
813 if (!getSteerTarget(iterPos, targetPos, SMOOTH_PATH_SLOP, polys, npolys, steerPos, steerPosFlag, steerPosRef))
814 break;
815
816 bool endOfPath = (steerPosFlag & DT_STRAIGHTPATH_END) != 0;
817 bool offMeshConnection = (steerPosFlag & DT_STRAIGHTPATH_OFFMESH_CONNECTION) != 0;
818
819 // Find movement delta.
820 float delta[VERTEX_SIZE];
821 dtVsub(delta, steerPos, iterPos);
822 float len = dtMathSqrtf(dtVdot(delta, delta));
823 // If the steer target is end of path or off-mesh link, do not move past the location.
824 if ((endOfPath || offMeshConnection) && len < SMOOTH_PATH_STEP_SIZE)
825 len = 1.0f;
826 else
827 len = SMOOTH_PATH_STEP_SIZE / len;
828
829 float moveTgt[VERTEX_SIZE];
830 dtVmad(moveTgt, iterPos, delta, len);
831
832 // Move
833 float result[VERTEX_SIZE];
834 const static uint32_t MAX_VISIT_POLY = 16;
835 dtPolyRef visited[MAX_VISIT_POLY];
836
837 uint32_t nvisited = 0;
838 if (dtStatusFailed(_navMeshQuery->moveAlongSurface(polys[0], iterPos, moveTgt, &_filter, result, visited, (int*)&nvisited, MAX_VISIT_POLY)))
839 return DT_FAILURE;
840 npolys = fixupCorridor(polys, npolys, MAX_POINT_PATH_LENGTH, visited, nvisited);
841
842 result[1] += 0.5f;
843 dtVcopy(iterPos, result);
844
845 // Handle end of path and off-mesh links when close enough.
846 if (endOfPath && inRangeYZX(iterPos, steerPos, SMOOTH_PATH_SLOP, 1.0f))
847 {
848 // Reached end of path.
849 dtVcopy(iterPos, targetPos);
850 if (nsmoothPath < maxSmoothPathSize)
851 {
852 dtVcopy(&smoothPath[nsmoothPath*VERTEX_SIZE], iterPos);
853 nsmoothPath++;
854 }
855 break;
856 }
857 else if (offMeshConnection && inRangeYZX(iterPos, steerPos, SMOOTH_PATH_SLOP, 1.0f))
858 {
859 // Advance the path up to and over the off-mesh connection.
860 dtPolyRef prevRef = INVALID_POLYREF;
861 dtPolyRef polyRef = polys[0];
862 uint32_t npos = 0;
863 while (npos < npolys && polyRef != steerPosRef)
864 {
865 prevRef = polyRef;
866 polyRef = polys[npos];
867 npos++;
868 }
869
870 for (uint32_t i = npos; i < npolys; ++i)
871 polys[i-npos] = polys[i];
872
873 npolys -= npos;
874
875 // Handle the connection.
876 float connectionStartPos[VERTEX_SIZE], connectionEndPos[VERTEX_SIZE];
877 if (dtStatusSucceed(_navMesh->getOffMeshConnectionPolyEndPoints(prevRef, polyRef, connectionStartPos, connectionEndPos)))
878 {
879 if (nsmoothPath < maxSmoothPathSize)
880 {
881 dtVcopy(&smoothPath[nsmoothPath*VERTEX_SIZE], connectionStartPos);
882 nsmoothPath++;
883 }
884 // Move position at the other side of the off-mesh link.
885 dtVcopy(iterPos, connectionEndPos);
886 if (dtStatusFailed(_navMeshQuery->getPolyHeight(polys[0], iterPos, &iterPos[1])))
887 return DT_FAILURE;
888 iterPos[1] += 0.5f;
889 }
890 }
891
892 // Store results.
893 if (nsmoothPath < maxSmoothPathSize)
894 {
895 dtVcopy(&smoothPath[nsmoothPath*VERTEX_SIZE], iterPos);
896 nsmoothPath++;
897 }
898 }
899
900 *smoothPathSize = nsmoothPath;
901
902 // this is most likely a loop
903 return nsmoothPath < MAX_POINT_PATH_LENGTH ? DT_SUCCESS : DT_FAILURE;
904}

906bool PathGenerator::inRangeYZX(float const* v1, float const* v2, float r, float h) const
907{
908 const float dx = v2[0] - v1[0];
909 const float dy = v2[1] - v1[1]; // elevation
910 const float dz = v2[2] - v1[2];
911 return (dx * dx + dz * dz) < r * r && fabsf(dy) < h;
912}

925void PathGenerator::shortenPathUntilDist(G3D::Vector3 const& target, float dist)
926{
927 if (getPathType() == PATHFIND_BLANK || _pathPoints.size() < 2)
928 {
929 return;
930 }
931
932 float const distSq = dist * dist;
933
934 // the first point of the path must be outside the specified range
935 // (this should have really been checked by the caller...)
936 if ((_pathPoints[0] - target).squaredLength() < distSq)
937 return;
938
939 // check if we even need to do anything
940 if ((*_pathPoints.rbegin() - target).squaredLength() >= distSq)
941 return;
942
943 size_t i = _pathPoints.size()-1;
944 float x, y, z, collisionHeight = _source->getCollisionHeight();
945 // find the first i s.t.:
946 // - _pathPoints[i] is still too close
947 // - _pathPoints[i-1] is too far away
948 // => the end point is somewhere on the line between the two
949 while (1)
950 {
951 // we know that pathPoints[i] is too close already (from the previous iteration)
952 if ((_pathPoints[i-1] - target).squaredLength() >= distSq)
953 break; // bingo!
954
955 // check if the shortened path is still in LoS with the target
956 _source->getHitSpherePointFor({ _pathPoints[i - 1].x, _pathPoints[i - 1].y, _pathPoints[i - 1].z + collisionHeight }, x, y, z);
957 if (!_source->getWorldMap()->isInLineOfSight(LocationVector(x, y, z), LocationVector(_pathPoints[i - 1].x, _pathPoints[i - 1].y, _pathPoints[i - 1].z + collisionHeight), _source->GetPhase(), LINEOFSIGHT_ALL_CHECKS))
958 {
959 // whenver we find a point that is not in LoS anymore, simply use last valid path
960 _pathPoints.resize(i + 1);
961 return;
962 }
963
964 if (!--i)
965 {
966 // no point found that fulfills the condition
967 _pathPoints[0] = _pathPoints[1];
968 _pathPoints.resize(2);
969 return;
970 }
971 }
972
973 // ok, _pathPoints[i] is too close, _pathPoints[i-1] is not, so our target point is somewhere between the two...
974 // ... settle for a guesstimate since i'm not confident in doing trig on every chase motion tick...
975 // (@todo review this)
976 _pathPoints[i] += (_pathPoints[i - 1] - _pathPoints[i]).direction() * (dist - (_pathPoints[i] - target).length());
977 _pathPoints.resize(i+1);
978}
