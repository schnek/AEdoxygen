58uint32_t DeathKnight::HasRunes(uint8_t type, uint32_t count)
59{
60 uint32_t found = 0;
61 for (uint8_t i = 0; i < MAX_RUNES && count != found; ++i)
62 if (m_runes[i].type == type && !m_runes[i].is_used)
63 found++;
64
65 return (count - found);
66}

68uint32_t DeathKnight::TakeRunes(uint8_t type, uint32_t count)
69{
70 uint8_t found = 0;
71 for (uint8_t i = 0; i < MAX_RUNES && count != found; ++i)
72 {
73 if (m_runes[i].type == type && !m_runes[i].is_used)
74 {
75 m_runes[i].is_used = true;
76 m_last_used_rune_slot = i;
77 sEventMgr.AddEvent(this, &DeathKnight::ResetRune, i, EVENT_PLAYER_RUNE_REGEN + i, 10000, 1, EVENT_FLAG_DO_NOT_EXECUTE_IN_WORLD_CONTEXT);
78 found++;
79 }
80 }
81 return (count - found);
82}

90uint8_t DeathKnight::GetRuneFlags()
91{
92 uint8_t result = 0;
93 for (uint8_t k = 0; k < MAX_RUNES; k++)
94 if (!m_runes[k].is_used)
95 result |= (1 << k);
96
97 return result;
98}
